12impl Formatter<'_> {
13    /// Writes a single space, skipping any whitespace and comments.
14    /// That is, suppose the next token is a big whitespace, possibly with multiple lines.
15    /// Those are skipped but only one space is written. In this way if we have
16    /// "mod     foo" it's transformed to "mod foo".
17    /// If there are comments in between `mod` and `foo` they are written, though!
18    /// No comment is ever lost.
19    ///
20    /// A space is not appended to the buffer is it already ends with a space.
21    pub(crate) fn write_space(&mut self) {
22        self.skip_comments_and_whitespace();
23        self.write_space_without_skipping_whitespace_and_comments();
24    }
25
26    /// Writes a single space, but doesn't skip whitespace and comments before doing that.
27    ///
28    /// A space is not appended to the buffer is it already ends with a space.
29    pub(crate) fn write_space_without_skipping_whitespace_and_comments(&mut self) {
30        if !self.buffer.ends_with_newline() && !self.buffer.ends_with_space() {
31            self.write(" ");
32        }
33    }

35    pub(crate) fn skip_whitespace(&mut self) {
36        while let Token::Whitespace(..) = &self.token {
37            self.bump();
38        }
39    }
40
41    /// Only skips whitespace if it doesn't have newlines in it.
42    /// Note that this doesn't write whitespace or comments at all.
43    pub(crate) fn skip_whitespace_if_it_is_not_a_newline(&mut self) {
44        while let Token::Whitespace(whitespace) = &self.token {
45            if whitespace.contains('\n') {
46                break;
47            }
48            self.bump();
49        }
50    }

52    /// Skips comments and whitespace, writing newlines if there are any.
53    /// If there are multiple consecutive newlines, only one is written.
54    pub(crate) fn skip_comments_and_whitespace(&mut self) {
55        self.skip_comments_and_whitespace_impl(
56            false, // write multiple lines
57            false, // at beginning
58        );
59    }

61    /// Similar to skip_comments_and_whitespace, but will write two lines if
62    /// multiple newlines are found (but at most two lines at a time).
63    pub(crate) fn skip_comments_and_whitespace_writing_multiple_lines_if_found(&mut self) {
64        self.skip_comments_and_whitespace_impl(
65            true,  // write multiple lines
66            false, // at beginning
67        );
68    }

70    pub(crate) fn skip_comments_and_whitespace_impl(
71        &mut self,
72        write_multiple_lines: bool,
73        at_beginning: bool,
74    ) {
75        // Number of newlines we just skipped.
76        let mut number_of_newlines = 0;
77
78        // Did we just passed some whitespace?
79        let mut passed_whitespace = false;
80
81        // Was the last token we processed a block comment?
82        let mut last_was_block_comment = false;
83
84        let mut ignore_next = self.ignore_next;
85
86        loop {
87            match &self.token {
88                Token::Whitespace(whitespace) => {
89                    number_of_newlines = whitespace.chars().filter(|char| *char == '\n').count();
90                    passed_whitespace = whitespace.ends_with(' ');
91
92                    if last_was_block_comment && number_of_newlines > 0 {
93                        if number_of_newlines > 1 {
94                            self.write_multiple_lines_without_skipping_whitespace_and_comments();
95                        } else {
96                            self.write_line_without_skipping_whitespace_and_comments();
97                        }
98
99                        self.bump();
100
101                        // Only indent for what's coming next if it's a comment
102                        // (otherwise a closing brace must come and we wouldn't want to indent that)
103                        if matches!(
104                            &self.token,
105                            Token::LineComment(_, None) | Token::BlockComment(_, None),
106                        ) {
107                            self.write_indentation();
108                        }
109
110                        number_of_newlines = 0;
111                        passed_whitespace = false;
112                    } else {
113                        self.bump();
114                    }
115
116                    last_was_block_comment = false;
117                }
118                Token::LineComment(comment, None) => {
119                    let comment = comment.clone();
120
121                    if comment.trim() == "noir-fmt:ignore" {
122                        ignore_next = true;
123                        self.ignore_next = true;
124                    }
125
126                    // Here we check if we need to write one line, two lines or none after the
127                    // end of the line comment.
128                    if number_of_newlines > 1 && write_multiple_lines {
129                        self.write_multiple_lines_without_skipping_whitespace_and_comments();
130                        self.write_indentation();
131                    } else if number_of_newlines > 0 {
132                        self.write_line_without_skipping_whitespace_and_comments();
133                        self.write_indentation();
134                    } else if !(at_beginning && self.buffer.is_empty()) {
135                        // We write a space before a line comment so if you have code like this:
136                        // "1// comment" it's transformed to "1 // comment".
137                        // What if there was already a space? It's all good, `write_space`
138                        // will never write two consecutive spaces.
139                        self.write_space_without_skipping_whitespace_and_comments();
140                    }
141
142                    // If this comment is trailing-inline (no newline came before it and we
143                    // aren't at the start of the file), keep it isolated. Subsequent
144                    // standalone `//` comments on the following lines belong to the next
145                    // "block" of comments, not to this one — they would arrive in a
146                    // separate `skip_comments_and_whitespace_impl` call in normal flow,
147                    // but here they'd get swallowed by the group loop. The buffer check
148                    // alone isn't enough because chunk construction runs us against a
149                    // fresh sub-buffer, so we rely on `number_of_newlines` instead.
150                    let first_is_trailing_inline = number_of_newlines == 0 && !at_beginning;
151
152                    let mut group = vec![comment];
153                    self.bump();
154                    let mut group_count = 1;
155
156                    // Extend the group with consecutive `//` line comments at the same
157                    // indentation, so that reflow can treat them as one paragraph. We
158                    // never group across blank lines, ignore directives, doc-style
159                    // changes (`///` and `//!` arrive as separate token variants), or
160                    // when the leading comment is attached to code on its own line.
161                    if self.config.wrap_comments && !self.ignore_next && !first_is_trailing_inline {
162                        loop {
163                            let Token::Whitespace(ws) = &self.token else { break };
164                            let newlines = ws.chars().filter(|c| *c == '\n').count();
165                            if newlines != 1 {
166                                break;
167                            }
168                            let extends = match self.peek_next_token() {
169                                Token::LineComment(next_body, None) => {
170                                    next_body.trim() != "noir-fmt:ignore"
171                                }
172                                _ => false,
173                            };
174                            if !extends {
175                                break;
176                            }
177                            self.bump();
178                            let Token::LineComment(next_body, None) = self.bump() else {
179                                unreachable!("peek confirmed a plain line comment")
180                            };
181                            group.push(next_body);
182                            group_count += 1;
183                        }
184                    }
185
186                    self.write_line_comment_group(&group, "//");
187                    self.write_line_without_skipping_whitespace_and_comments();
188                    number_of_newlines = 1;
189                    passed_whitespace = false;
190                    last_was_block_comment = false;
191                    self.written_comments_count += group_count;
192                }
193                Token::BlockComment(comment, None) => {
194                    let comment = comment.clone();
195
196                    if comment.trim() == "noir-fmt:ignore" {
197                        ignore_next = true;
198                        self.ignore_next = true;
199                    }
200
201                    // Here we check if we need to write one line, two lines or none after the
202                    // end of the block comment.
203                    if number_of_newlines > 1 && write_multiple_lines {
204                        self.write_multiple_lines_without_skipping_whitespace_and_comments();
205                        self.write_indentation();
206                    } else if number_of_newlines > 0 {
207                        self.write_line_without_skipping_whitespace_and_comments();
208                        self.write_indentation();
209                    } else if passed_whitespace {
210                        // We write a space before a line comment so if you have code like this:
211                        // "1/* comment */" it's transformed to "1 /* comment */".
212                        // What if there was already a space? It's all good, `write_space`
213                        // will never write two consecutive spaces.
214                        self.write_space_without_skipping_whitespace_and_comments();
215                    }
216                    self.write_block_comment(&comment, "/*");
217                    self.bump();
218                    passed_whitespace = false;
219                    last_was_block_comment = true;
220                    self.written_comments_count += 1;
221                }
222                _ => break,
223            }
224        }
225
226        // Case when we passed some whitespace with newlines but no comments followed it.
227        if number_of_newlines > 1 && write_multiple_lines {
228            self.write_multiple_lines_without_skipping_whitespace_and_comments();
229        }
230
231        self.ignore_next = ignore_next;
232    }

234    /// Reflow a sequence of consecutive line-comment bodies as a single paragraph-aware
235    /// group. The bodies arrive with the per-line `//` prefix already stripped (i.e. they
236    /// are the lexer's comment body strings). The caller is responsible for having written
237    /// any leading indentation for the first comment.
238    ///
239    /// While inside a chunk's sub-buffer we don't yet know the rendered position of the
240    /// emitted text, so we don't reflow here — we just emit each body verbatim. The
241    /// chunks layer (`write_chunk_lines`) does the actual reflow once we're back in the
242    /// main buffer and the budget is known.
243    pub(crate) fn write_line_comment_group(&mut self, bodies: &[String], prefix: &str) {
244        if self.ignore_next || !self.config.wrap_comments || self.in_chunk {
245            for (index, body) in bodies.iter().enumerate() {
246                if index > 0 {
247                    self.start_new_line();
248                }
249                self.write(prefix);
250                self.write(body.trim_end());
251            }
252            return;
253        }
254
255        let reflow_enabled = reflow_enabled_for_prefix(self.config, prefix);
256
257        // With reflow disabled, take the pre-reflow per-line wrap path: each body wraps
258        // independently and internal whitespace (e.g. multiple spaces inside the
259        // comment) is preserved. The paragraph-aware engine normalizes runs of
260        // whitespace via `split_whitespace`, which we don't want here.
261        if !reflow_enabled {
262            for (index, body) in bodies.iter().enumerate() {
263                if index > 0 {
264                    self.start_new_line();
265                }
266                self.write_comment_with_prefix(body.trim_end(), prefix);
267            }
268            return;
269        }
270
271        let stripped: Vec<String> = bodies
272            .iter()
273            .map(|body| {
274                let trimmed = body.trim_end();
275                trimmed.strip_prefix(' ').unwrap_or(trimmed).to_string()
276            })
277            .collect();
278        let lines: Vec<&str> = stripped.iter().map(String::as_str).collect();
279
280        let prefix_cols = prefix.chars().count() + 1;
281        let indent_cols = (self.indentation.max(0) as usize) * self.config.tab_spaces;
282        let comment_width = self.config.comment_width;
283        let first_budget =
284            comment_width.saturating_sub(self.current_line_width()).saturating_sub(prefix_cols);
285        let cont_budget = comment_width.saturating_sub(indent_cols).saturating_sub(prefix_cols);
286
287        let config = self.config;
288        let snippet_width =
289            self.config.max_width.saturating_sub(indent_cols).saturating_sub(prefix_cols);
290        let outputs = comment_reflow::reflow_comment_with_code_formatter(
291            &lines,
292            first_budget,
293            cont_budget,
294            reflow_enabled,
295            |source, lang| format_noir_snippet(source, lang, config, snippet_width),
296        );
297
298        for (index, content) in outputs.iter().enumerate() {
299            if index > 0 {
300                self.start_new_line();
301            }
302            self.write(prefix);
303            if !content.is_empty() {
304                self.write(" ");
305                self.write(content);
306            }
307        }
308    }

310    /// Wraps a single comment body word-by-word with the given prefix. Kept for the
311    /// in-chunk path which performs line-by-line wrapping over already-grouped chunk
312    /// strings; the standalone-comment path uses `write_line_comment_group` instead.
313    pub(crate) fn write_comment_with_prefix(&mut self, comment: &str, prefix: &str) {
314        self.write(prefix);
315        for word in comment.split_inclusive([' ', '\n', '\t']) {
316            if self.current_line_width() + word.trim().chars().count() > self.config.comment_width {
317                self.start_new_line();
318                if !prefix.is_empty() {
319                    self.write(prefix);
320                    self.write(" ");
321                }
322            }
323
324            self.write(word);
325        }
326    }

328    /// Pre-reflow body emit for `write_block_comment`. Each source line wraps
329    /// independently with the same word-by-word logic as `write_comment_with_prefix`,
330    /// preserving internal whitespace. The opening `/*`-style prefix has already been
331    /// written by the caller; we write the body and the closing `*/`.
332    fn write_block_comment_non_reflow(&mut self, comment: &str, all_stars: bool) {
333        if comment.trim_start_matches([' ', '\t']).starts_with('\n') {
334            self.start_new_line_no_indentation();
335        }
336
337        for (index, line) in comment.lines().enumerate() {
338            if index > 0 {
339                self.start_new_line_no_indentation();
340            }
341
342            for word in line.split_inclusive([' ', '\n', '\t']) {
343                if self.current_line_width() + word.trim().chars().count()
344                    > self.config.comment_width
345                {
346                    self.start_new_line();
347                    if all_stars {
348                        self.write(" * ");
349                    }
350                }
351
352                self.write(word);
353            }
354        }
355
356        if comment.ends_with('\n') {
357            self.start_new_line();
358        }
359
360        if self.current_line_width() + 2 > self.config.comment_width {
361            self.start_new_line();
362        }
363
364        self.write("*/");
365    }

367    pub(crate) fn write_block_comment(&mut self, comment: &str, prefix: &str) {
368        self.write(prefix);
369
370        if self.ignore_next || !self.config.wrap_comments || self.in_chunk {
371            self.write(comment);
372            self.write("*/");
373            return;
374        }
375
376        let all_stars = block_comment_has_all_leading_stars(comment);
377        let indent_cols = (self.indentation.max(0) as usize) * self.config.tab_spaces;
378        let comment_width = self.config.comment_width;
379
380        let reflow_enabled = reflow_enabled_for_prefix(self.config, prefix);
381
382        // With reflow disabled, take the pre-reflow per-line wrap path: each source
383        // line wraps independently and internal whitespace within each line is
384        // preserved (the paragraph-aware engine normalizes via `split_whitespace`).
385        if !reflow_enabled {
386            self.write_block_comment_non_reflow(comment, all_stars);
387            return;
388        }
389
390        // A block comment has two emit shapes. We pick based on whether the source
391        // body opens with a newline; that decision is idempotent because the wrapped
392        // form preserves the opening shape.
393        //
394        // Single-line shape (no leading newline):
395        //     /* first wrap
396        //      * second wrap */
397        //
398        // Multi-line shape (leading newline):
399        //     /*
400        //     content
401        //     */
402        let body_opens_with_newline = comment.trim_start_matches([' ', '\t']).starts_with('\n');
403
404        let source_lines: Vec<&str> = comment.lines().collect();
405        let mut content_lines: Vec<String> = source_lines
406            .iter()
407            .map(|line| {
408                let after_ws = line.trim_start();
409                if all_stars {
410                    if let Some(rest) = after_ws.strip_prefix("* ") {
411                        rest.to_string()
412                    } else if let Some(rest) = after_ws.strip_prefix('*') {
413                        rest.strip_prefix(' ').unwrap_or(rest).to_string()
414                    } else {
415                        after_ws.to_string()
416                    }
417                } else {
418                    after_ws.to_string()
419                }
420            })
421            .collect();
422
423        while content_lines.first().is_some_and(|s| s.trim().is_empty()) {
424            content_lines.remove(0);
425        }
426        while content_lines.last().is_some_and(|s| s.trim().is_empty()) {
427            content_lines.pop();
428        }
429
430        if content_lines.is_empty() {
431            self.write(comment);
432            self.write("*/");
433            return;
434        }
435
436        let line_refs: Vec<&str> = content_lines.iter().map(String::as_str).collect();
437
438        if !body_opens_with_newline {
439            let has_leading_space = comment.starts_with(' ') || comment.starts_with('\t');
440            let has_trailing_space = comment.ends_with(' ') || comment.ends_with('\t');
441
442            let first_used = self.current_line_width() + if has_leading_space { 1 } else { 0 };
443            let cont_used = indent_cols + 3;
444            let first_budget = comment_width.saturating_sub(first_used);
445            let cont_budget = comment_width.saturating_sub(cont_used);
446
447            let config = self.config;
448            let snippet_width = self.config.max_width.saturating_sub(cont_used);
449            let outputs = comment_reflow::reflow_comment_with_code_formatter(
450                &line_refs,
451                first_budget,
452                cont_budget,
453                reflow_enabled,
454                |source, lang| format_noir_snippet(source, lang, config, snippet_width),
455            );
456
457            if has_leading_space {
458                self.write(" ");
459            }
460            if let Some((first, rest)) = outputs.split_first() {
461                self.write(first);
462                for line in rest {
463                    self.start_new_line();
464                    if all_stars {
465                        self.write(" * ");
466                    }
467                    self.write(line);
468                }
469            }
470            if has_trailing_space {
471                self.write(" ");
472            }
473            self.write("*/");
474            return;
475        }
476
477        let prefix_used = if all_stars { 3 } else { 0 };
478        let content_budget = comment_width.saturating_sub(indent_cols + prefix_used);
479        let config = self.config;
480        let snippet_width = self.config.max_width.saturating_sub(indent_cols + prefix_used);
481        let outputs = comment_reflow::reflow_comment_with_code_formatter(
482            &line_refs,
483            content_budget,
484            content_budget,
485            reflow_enabled,
486            |source, lang| format_noir_snippet(source, lang, config, snippet_width),
487        );
488
489        self.start_new_line_no_indentation();
490        for content in &outputs {
491            if content.is_empty() && !all_stars {
492                self.write(NEWLINE);
493                continue;
494            }
495            self.write_indentation();
496            if all_stars {
497                if content.is_empty() {
498                    self.write(" *");
499                } else {
500                    self.write(" * ");
501                    self.write(content);
502                }
503            } else {
504                self.write(content);
505            }
506            self.start_new_line_no_indentation();
507        }
508        self.write_indentation();
509        if all_stars {
510            self.write(" ");
511        }
512        self.write("*/");
513    }

540        if !self.buffer.ends_with_newline() && !self.buffer.ends_with_space() {
541            self.write(NEWLINE);
542            true
543        } else {
544            false
545        }

548    // Modifies the current buffer so that it will always have two newlines at the end.
549    pub(crate) fn write_multiple_lines_without_skipping_whitespace_and_comments(&mut self) {
550        if self.buffer.ends_with_double_newline() {
551            // Nothing
552        } else if self.buffer.ends_with_newline() {
553            self.write(NEWLINE);
554        } else {
555            self.write(NEWLINE);
556            self.write(NEWLINE);
557        }
558    }

581/// Returns whether paragraph reflow is enabled for the comment kind identified by its
582/// per-line prefix. Plain `//` and `/*` opt in via `reflow_non_doc_comments`; doc-style
583/// `///`, `//!`, `/**`, and `/*!` opt in via `reflow_doc_comments`.
584fn reflow_enabled_for_prefix(config: &Config, prefix: &str) -> bool {
585    match prefix {
586        "//" | "/*" => config.reflow_non_doc_comments,
587        _ => config.reflow_doc_comments,
588    }

591/// Recursively formats a Noir code snippet pulled from a doc-comment fenced block.
592/// Returns `None` when the snippet should be left alone — either the fence has a
593/// non-Noir language tag, the body is empty, or the snippet fails to parse cleanly.
594///
595/// `available_width` is how many columns the rendered fence body actually gets — the
596/// parent's `max_width` minus the comment-line prefix (e.g. `/// `) and any outer
597/// indentation. The inner formatter's `max_width` and related single-line thresholds
598/// are tightened to this so output that lands inside the prefix still respects the
599/// configured maximum.
600fn format_noir_snippet(
601    source: &str,
602    lang: Option<&str>,
603    config: &Config,
604    available_width: usize,
605) -> Option<String> {
606    if !config.format_code_blocks {
607        return None;
608    }
609    let lang_ok = match lang {
610        None => true,
611        Some(l) => l.eq_ignore_ascii_case("noir"),
612    };
613    if !lang_ok || source.trim().is_empty() {
614        return None;
615    }
616    let (parsed, errors) = noirc_frontend::parse_program_with_dummy_file(source);
617    if errors.into_iter().any(|e| !e.is_warning()) {
618        return None;
619    }
620
621    // Floor at 20 so deeply-indented or very tight comment positions still produce
622    // legible output rather than degenerate one-token-per-line wrapping.
623    let inner_width = available_width.max(20);
624    let mut snippet_config = config.clone();
625    snippet_config.max_width = inner_width;
626    snippet_config.fn_call_width = config.fn_call_width.min(inner_width);
627    snippet_config.array_width = config.array_width.min(inner_width);
628    snippet_config.single_line_if_else_max_width =
629        config.single_line_if_else_max_width.min(inner_width);
