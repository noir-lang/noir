7impl Formatter<'_> {
8    pub(super) fn format_module_declaration(&mut self, module_declaration: ModuleDeclaration) {
9        self.format_secondary_attributes(module_declaration.outer_attributes);
10        self.write_indentation();
11        self.format_item_visibility(module_declaration.visibility);
12        self.write_keyword(Keyword::Mod);
13        self.write_space();
14        self.write_identifier(module_declaration.ident);
15        self.write_semicolon();
16    }
17
18    pub(super) fn format_submodule(&mut self, submodule: ParsedSubModule) {
19        self.format_secondary_attributes(submodule.outer_attributes);
20        self.write_indentation();
21        self.format_item_visibility(submodule.visibility);
22        if submodule.is_contract {
23            self.write_keyword(Keyword::Contract);
24        } else {
25            self.write_keyword(Keyword::Mod);
26        }
27        self.write_space();
28        self.write_identifier(submodule.name);
29        self.write_space();
30        self.write_left_brace();
31        if parsed_module_is_empty(&submodule.contents) {
32            self.format_empty_block_contents();
33        } else {
34            self.increase_indentation();
35            self.write_line();
36            self.format_parsed_module(submodule.contents, self.ignore_next);
37            self.decrease_indentation();
38            self.write_indentation();
39        }
40        self.write_right_brace();
41    }
42}
43
44fn parsed_module_is_empty(parsed_module: &ParsedModule) -> bool {
45    parsed_module.inner_doc_comments.is_empty() && parsed_module.items.is_empty()
46}
