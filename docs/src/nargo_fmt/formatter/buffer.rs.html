1/// A buffer to write to.
2/// It keeps track of the current line width and provides a few useful methods
3/// to deal with the buffer contents.
4#[derive(Default, Debug)]
5pub(crate) struct Buffer {
6    buffer: String,
7
8    /// How many characters we've written so far in the current line
9    /// (useful to avoid exceeding the configurable maximum)
10    current_line_width: usize,
11}
12
13impl Buffer {
14    pub(crate) fn is_empty(&self) -> bool {
15        self.buffer.is_empty()
16    }
17
18    pub(crate) fn ends_with_newline(&self) -> bool {
19        self.buffer.ends_with('\n')
20    }
21
22    pub(crate) fn ends_with_double_newline(&self) -> bool {
23        self.buffer.ends_with("\n\n")
24    }
25
26    pub(crate) fn ends_with_space(&self) -> bool {
27        self.buffer.ends_with(' ')
28    }
29
30    pub(crate) fn write(&mut self, str: &str) {
31        self.buffer.push_str(str);
32
33        if str.ends_with('\n') {
34            self.current_line_width = 0;
35        } else {
36            self.current_line_width += str.chars().count();
37        }
38    }
39
40    /// Trim spaces from the end of the buffer.
41    pub(crate) fn trim_spaces(&mut self) {
42        while self.buffer.ends_with(' ') {
43            self.buffer.truncate(self.buffer.len() - 1);
44            self.current_line_width -= 1;
45        }
46    }
47
48    /// Trim commas from the end of the buffer. Returns true if a comma was trimmed.
49    pub(super) fn trim_comma(&mut self) -> bool {
50        if self.buffer.ends_with(',') {
51            self.buffer.truncate(self.buffer.len() - 1);
52            self.current_line_width -= 1;
53            true
54        } else {
55            false
56        }
57    }
58
59    /// Trim multiple newlines from the end of the buffer, keeping at most one.
60    pub(crate) fn trim_multiple_newlines(&mut self) {
61        while self.buffer.ends_with("\n\n") {
62            self.buffer.truncate(self.buffer.len() - 1);
63        }
64    }
65
66    pub(crate) fn contents(self) -> String {
67        self.buffer
68    }
69
70    pub(crate) fn current_line_width(&self) -> usize {
71        self.current_line_width
72    }
73}
