1//! This module has all the logic to format a series of chunks (a piece of text) in a way
2//! that we (almost always) never exceed the configurable maximum line width.
3//!
4//! It's heavily inspired by this excellent blog post:
5//!
6//! <https://yorickpeterse.com/articles/how-to-write-a-code-formatter/>
7//!
8//! However, some changes were introduces to handle comments and other particularities of Noir.
9use std::{fmt::Display, ops::Deref};
10
11use noirc_frontend::token::Token;

13use super::Formatter;

15/// A text chunk. It precomputes the text width and whether it has newlines.
16#[derive(Debug)]
17pub(crate) struct TextChunk {
18    pub(crate) string: String,
19    pub(crate) width: usize,
20    pub(crate) has_newlines: bool,
21}
22
23impl TextChunk {
24    pub(crate) fn new(string: String) -> Self {
25        TextChunk {
26            width: string.chars().count(),
27            has_newlines: string.chars().any(|char| char == '\n'),
28            string,
29        }
30    }
31}
32
33/// A chunk can either be text or a directive that instructs the formatter to do something
34/// (for example: increase or decrease the current indentation).
35#[derive(Debug)]
36pub(crate) enum Chunk {
37    /// A text chunk. It might contain leading comments.
38    Text(TextChunk, bool /* verbatim */),
39    /// A trailing comma that's only written if we decide to format chunks in multiple lines
40    /// (for example for a call we'll add a trailing comma to the last argument).
41    TrailingComma,
42    /// A trailing comment (happens at the end of a line, and always after something else have been written).
43    TrailingComment(TextChunk, bool /* at_block_end */),
44    /// A leading comment. Happens at the beginning of a line.
45    LeadingComment(TextChunk),
46    /// A group of chunks.
47    Group(ChunkGroup),
48    /// Write a line (or two) if we decide to format chunks in multiple lines, otherwise do nothing.
49    Line { two: bool },
50    /// Writes a space if we can write a group in one line, otherwise writes a line.
51    /// However, a space might be written if `one_chunk_per_line` of a Chunks object is set to false.
52    SpaceOrLine,
53    /// Command to increase the current indentation.
54    IncreaseIndentation,
55    /// Command to decrease the current indentation.
56    DecreaseIndentation,
57    /// Push the current indentation to the indentation stack.
58    PushIndentation,
59    /// Set the current indentation by popping it from the indentation stack.
60    PopIndentation,
61}
62
63impl Chunk {
64    pub(crate) fn width(&self) -> usize {
65        match self {
66            Chunk::Text(chunk, _)
67            | Chunk::TrailingComment(chunk, _)
68            | Chunk::LeadingComment(chunk) => chunk.width,
69            Chunk::Group(group) => group.width(),
70            Chunk::SpaceOrLine => 1,
71            Chunk::Line { .. }
72            | Chunk::IncreaseIndentation
73            | Chunk::DecreaseIndentation
74            | Chunk::TrailingComma
75            | Chunk::PushIndentation
76            | Chunk::PopIndentation => 0,
77        }
78    }
79
80    /// Computes the width of this chunk considering it's inside an ExpressionList.
81    /// The only thing that changes here compared to `width` is that a LambdaAsLastExpressionInList's
82    /// width is considered to be only the first line, so we can avoid splitting the entire call
83    /// arguments into separate lines.
84    pub(crate) fn width_inside_an_expression_list(&self) -> usize {
85        if let Chunk::Group(group) = &self
86            && let GroupKind::LambdaAsLastExpressionInList { first_line_width, .. } = &group.kind
87        {
88            return *first_line_width;
89        }
90
91        self.width()
92    }
93
94    pub(crate) fn has_newlines(&self) -> bool {
95        match self {
96            Chunk::Text(chunk, _)
97            | Chunk::TrailingComment(chunk, _)
98            | Chunk::LeadingComment(chunk) => chunk.has_newlines,
99            Chunk::Group(group) => group.has_newlines(),
100            Chunk::TrailingComma
101            | Chunk::Line { .. }
102            | Chunk::SpaceOrLine
103            | Chunk::IncreaseIndentation
104            | Chunk::DecreaseIndentation
105            | Chunk::PushIndentation
106            | Chunk::PopIndentation => false,
107        }
108    }
109
110    /// Returns the current chunk as a Group, if it is one. Otherwise returns None.
111    pub(crate) fn group(self) -> Option<ChunkGroup> {
112        if let Chunk::Group(group) = self { Some(group) } else { None }
113    }
114}
115
116impl Display for Chunk {
117    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
118        match self {
119            Chunk::Text(text_chunk, _)
120            | Chunk::TrailingComment(text_chunk, _)
121            | Chunk::LeadingComment(text_chunk) => {
122                write!(f, "{}", text_chunk.string)
123            }
124            Chunk::TrailingComma => write!(f, ","),
125            Chunk::Group(chunk_group) => {
126                write!(f, "`(`")?;
127                chunk_group.fmt(f)?;
128                write!(f, "`)`")
129            }
130            Chunk::SpaceOrLine => write!(f, "`space_or_line`"),
131            Chunk::Line { .. } => write!(f, "`line`"),
132            Chunk::IncreaseIndentation => write!(f, "`+`"),
133            Chunk::DecreaseIndentation => write!(f, "`-`"),
134            Chunk::PushIndentation => write!(f, "`push`"),
135            Chunk::PopIndentation => write!(f, "`pop`"),
136        }
137    }
138}

140#[derive(Debug)]
141pub(crate) struct ChunkGroup {
142    pub(crate) chunks: Vec<Chunk>,
143
144    /// If `true`, when formatting in multiple lines, and after a SpaceOrLine,
145    /// a line will be written.
146    /// If `false`, when formatting in multiple lines, and after a SpaceOrLine,
147    /// a space will be inserted and the next chunk will go in the same line if
148    /// it fits that line.
149    ///
150    /// This is used to, for example, control how arrays are formatted. If each
151    /// element is short, we'll format the array like this:
152    ///
153    /// [
154    ///   1, 2, 3,
155    ///   4, 5
156    /// ]
157    ///
158    /// but if one of the elements is long, each one will go in a separate line:
159    ///
160    /// ```text
161    /// [
162    ///     1,
163    ///     1234567890123,
164    ///     3
165    /// ]
166    /// ```
167    pub(crate) one_chunk_per_line: bool,
168
169    /// If true, regardless of this group's chunks, this group will be formatted in
170    /// multiple lines.
171    /// This is set to true when, for example, we format a block that has at least
172    /// two statements: we always want to show that in multiple lines.
173    pub(crate) force_multiple_lines: bool,
174
175    /// Groups can be tagged. For example we tag all consequences and alternative blocks
176    /// of an `if` expression. If we determine one of them needs to be formatted in multiple
177    /// lines, we find all other chunks with the same tag and mark them too to be formatted
178    /// in multiple lines.
179    pub(crate) tag: Option<GroupTag>,
180
181    /// The kind of this group. Some group kinds are formatted in a special way
182    /// (mainly lambda arguments that are the last expression in a list).
183    pub(crate) kind: GroupKind,
184
185    /// This name is a bit long and explicit, but it's to make things clearer:
186    /// if we determine that this group needs to be formatted in multiple lines,
187    /// children groups with the same tag will also be formatted in multiple lines.
188    ///
189    /// This is used for example in infix expressions like `a + b + c + d`, where if we
190    /// determine that `a + b` needs to be formatted in multiple lines, we want the entire
191    /// tree (of those infix expressions) to be formatted in multiple lines.
192    pub(crate) force_multiline_on_children_with_same_tag_if_multiline: bool,
193}

195impl ChunkGroup {
196    pub(crate) fn new() -> Self {
197        Self {
198            chunks: Vec::new(),
199            one_chunk_per_line: true,
200            force_multiple_lines: false,
201            tag: None,
202            kind: GroupKind::Regular,
203            force_multiline_on_children_with_same_tag_if_multiline: false,
204        }
205    }
206
207    /// Appends a text to this group.
208    /// If the last chunk in this group is a text or verbatim,
209    /// no new chunk is inserted and instead the last text chunk is extended.
210    pub(crate) fn text(&mut self, chunk: TextChunk) {
211        self.text_impl(chunk, false /* verbatim */);
212    }
213
214    /// Appends a verbatim text chunk to this group.
215    pub(crate) fn verbatim(&mut self, chunk: TextChunk) {
216        self.text_impl(chunk, true /* verbatim */);
217    }
218
219    pub(crate) fn text_impl(&mut self, chunk: TextChunk, verbatim: bool) {
220        if chunk.width == 0 {
221            return;
222        }
223
224        if let Some(Chunk::Text(text_chunk, _)) = self.chunks.last_mut() {
225            text_chunk.string.push_str(&chunk.string);
226            text_chunk.width += chunk.width;
227            text_chunk.has_newlines |= chunk.has_newlines;
228        } else {
229            self.push(Chunk::Text(chunk, verbatim));
230        }
231    }
232
233    /// Appends a single space to this group by reading it from the given formatter.
234    pub(crate) fn space(&mut self, formatter: &mut ChunkFormatter<'_, '_>) {
235        self.text(formatter.chunk(|formatter| {
236            formatter.write_space();
237        }));
238    }
239
240    /// Appends a semicolon to this group by reading it from the given formatter.
241    /// This will actually end up attaching the semicolon to the last text in this
242    /// group so that we don't end up with stray semicolons.
243    pub(crate) fn semicolon(&mut self, formatter: &mut ChunkFormatter<'_, '_>) {
244        self.text_attached_to_last_group(formatter.chunk(|formatter| {
245            formatter.write_semicolon();
246        }));
247    }
248
249    /// Appends a TextChunk to this chunks chunks. However, if the last chunk is a group,
250    /// it's appended to that group's last text.
251    pub(crate) fn text_attached_to_last_group(&mut self, chunk: TextChunk) {
252        if chunk.width == 0 {
253            return;
254        }
255
256        if let Some(Chunk::Group(group)) = self.chunks.last_mut() {
257            group.text(chunk);
258        } else {
259            self.text(chunk);
260        }
261    }
262
263    /// Appends a trailing comment (it's formatted slightly differently than a regular text chunk).
264    pub(crate) fn trailing_comment(&mut self, chunk: TextChunk) {
265        if chunk.width > 0 {
266            self.push(Chunk::TrailingComment(chunk, false));
267        }
268    }
269
270    /// Similar to `trailing_comment` but happens in a block end so no newline+indent will be
271    /// produced afterwards.
272    pub(crate) fn trailing_comment_at_block_end(&mut self, chunk: TextChunk) {
273        if chunk.width > 0 {
274            self.push(Chunk::TrailingComment(chunk, true));
275        }
276    }
277
278    /// Appends a leading comment (it's formatted slightly differently than a regular text chunk).
279    pub(crate) fn leading_comment(&mut self, chunk: TextChunk) {
280        if chunk.width > 0 {
281            self.push(Chunk::LeadingComment(chunk));
282        }
283    }
284
285    /// Appends a trailing comma (will only show up if the group is formatted in multiple lines).
286    pub(crate) fn trailing_comma(&mut self) {
287        self.push(Chunk::TrailingComma);
288    }
289
290    /// Appends another group as a nested group.
291    pub(crate) fn group(&mut self, group: ChunkGroup) {
292        self.push(Chunk::Group(group));
293    }
294
295    /// Append one line to this chunk.
296    pub(crate) fn line(&mut self) {
297        self.lines(false);
298    }
299
300    /// Append one or two lines to this chunk.
301    pub(crate) fn lines(&mut self, two: bool) {
302        self.push(Chunk::Line { two });
303    }
304
305    /// Appends a SpaceOrLine chunk, which means that it's a space when this group is
306    /// formatted in a single line, or a line when it's formatted in multiple lines.
307    pub(crate) fn space_or_line(&mut self) {
308        self.push(Chunk::SpaceOrLine);
309    }
310
311    pub(crate) fn increase_indentation(&mut self) {
312        self.push(Chunk::IncreaseIndentation);
313    }
314
315    pub(crate) fn decrease_indentation(&mut self) {
316        self.push(Chunk::DecreaseIndentation);
317    }
318
319    pub(crate) fn push_indentation(&mut self) {
320        self.push(Chunk::PushIndentation);
321    }
322
323    pub(crate) fn pop_indentation(&mut self) {
324        self.push(Chunk::PopIndentation);
325    }
326
327    pub(crate) fn push(&mut self, chunk: Chunk) {
328        self.chunks.push(chunk);
329    }
330
331    pub(crate) fn width(&self) -> usize {
332        self.chunks.iter().map(|chunk| chunk.width()).sum()
333    }
334
335    pub(crate) fn expression_list_width(&self) -> usize {
336        self.chunks.iter().map(|chunk| chunk.width_inside_an_expression_list()).sum()
337    }
338
339    pub(crate) fn has_newlines(&self) -> bool {
340        self.force_multiple_lines || self.chunks.iter().any(|chunk| chunk.has_newlines())
341    }
342
343    /// Determines if this group has a LambdaAsLastExpressionInList chunk.
344    /// Note that if this group is a MethodCall, this is checked for the ExpressionList group
345    /// inside it.
346    pub(crate) fn has_lambda_as_last_expression_in_list(&self) -> bool {
347        self.chunks.iter().any(|chunk| {
348            if let Chunk::Group(group) = chunk {
349                if self.kind.is_method_call() && group.kind.is_expression_list() {
350                    group.has_lambda_as_last_expression_in_list()
351                } else {
352                    matches!(group.kind, GroupKind::LambdaAsLastExpressionInList { .. })
353                }
354            } else {
355                false
356            }
357        })
358    }
359
360    /// Finds the `LambdaAsLastExpressionInList` associated to this group and sets its indentation
361    /// to the given value.
362    pub(crate) fn set_lambda_as_last_expression_in_list_indentation(
363        &mut self,
364        indentation_to_set: i32,
365    ) {
366        for chunk in self.chunks.iter_mut() {
367            if let Chunk::Group(group) = chunk {
368                if self.kind.is_method_call() && group.kind.is_expression_list() {
369                    group.set_lambda_as_last_expression_in_list_indentation(indentation_to_set);
370                } else if let GroupKind::LambdaAsLastExpressionInList { indentation, .. } =
371                    &mut group.kind
372                    && indentation.is_none()
373                {
374                    *indentation = Some(indentation_to_set);
375                }
376            }
377        }
378    }

380    /// Before writing a Chunks object in multiple lines, create a new one where `TrailingComma`
381    /// is turned into `Text`. Because Chunks will glue two consecutive `Text`s together, if we
382    /// have two chunks `Text("123"), TrailingComma`, we'll consider the entire string "123,"
383    /// when deciding whether we can still write in the current line or not.
384    pub(crate) fn prepare_for_multiple_lines(self) -> ChunkGroup {
385        let mut group = ChunkGroup { chunks: Vec::new(), ..self };
386
387        for chunk in self.chunks {
388            match chunk {
389                Chunk::Text(chunk, verbatim) => group.text_impl(chunk, verbatim),
390                Chunk::TrailingComma => {
391                    // If there's a trailing comma after a group, append the text to that group
392                    // so that it glues with the last text present there (if any)
393                    group.add_trailing_comma_to_last_text();
394                }
395                Chunk::TrailingComment(chunk, at_block_end) => {
396                    if at_block_end {
397                        group.trailing_comment_at_block_end(chunk);
398                    } else {
399                        group.trailing_comment(chunk);
400                    }
401                }
402                Chunk::LeadingComment(chunk) => group.leading_comment(chunk),
403                Chunk::Group(inner_group) => group.group(inner_group),
404                Chunk::Line { two } => group.lines(two),
405                Chunk::SpaceOrLine => group.space_or_line(),
406                Chunk::IncreaseIndentation => group.increase_indentation(),
407                Chunk::DecreaseIndentation => group.decrease_indentation(),
408                Chunk::PushIndentation => group.push_indentation(),
409                Chunk::PopIndentation => group.pop_indentation(),
410            }
411        }
412        group
413    }
414
415    fn add_trailing_comma_to_last_text(&mut self) {
416        if let Some(Chunk::Group(group)) = self.chunks.last_mut() {
417            group.add_trailing_comma_to_last_text();
418        } else {
419            self.text(TextChunk::new(",".to_string()));
420        }
421    }
422
423    /// Returns the width of text until we hit a Line or LineOrSpace, together
424    /// with whether we hit a Line or LineOrSpace.
425    fn width_until_line(&self) -> (usize, bool) {
426        let mut width = 0;
427        for chunk in &self.chunks {
428            match chunk {
429                Chunk::Text(text_chunk, _)
430                | Chunk::TrailingComment(text_chunk, _)
431                | Chunk::LeadingComment(text_chunk) => {
432                    width += text_chunk.width;
433                }
434                Chunk::Group(chunk_group) => {
435                    let (group_width, hit_line) = chunk_group.width_until_line();
436                    width += group_width;
437                    if hit_line {
438                        return (width, true);
439                    }
440                }
441                Chunk::Line { .. } | Chunk::SpaceOrLine => {
442                    return (width, true);
443                }
444                Chunk::IncreaseIndentation
445                | Chunk::DecreaseIndentation
446                | Chunk::PushIndentation
447                | Chunk::PopIndentation
448                | Chunk::TrailingComma => (),
449            }
450        }
451
452        (width, false)
453    }
454
455    fn first_group(&self) -> Option<&ChunkGroup> {
456        self.chunks
457            .iter()
458            .filter_map(|chunk| if let Chunk::Group(group) = chunk { Some(group) } else { None })
459            .next()
460    }
461
462    fn has_expression_list_or_method_call_group(&self) -> bool {
463        for chunk in &self.chunks {
464            if let Chunk::Group(group) = chunk
465                && (group.kind.is_expression_list() || group.kind.is_method_call())
466            {
467                return true;
468            }
469        }
470
471        false
472    }
473
474    /// Assuming this is a MethodCall group, if the ExpressionList nested in it
475    /// has a LambdaAsLastExpressionInList, returns its `first_line_width`.
476    fn method_call_lambda_first_line_width(&self) -> Option<usize> {
477        for chunk in &self.chunks {
478            let Chunk::Group(group) = chunk else {
479                continue;
480            };
481
482            let GroupKind::ExpressionList { expressions_count: 1, .. } = group.kind else {
483                continue;
484            };
485
486            for chunk in &group.chunks {
487                let Chunk::Group(group) = chunk else {
488                    continue;
489                };
490
491                let GroupKind::LambdaAsLastExpressionInList { first_line_width, .. } = group.kind
492                else {
493                    continue;
494                };
495
496                return Some(first_line_width);
497            }
498        }
499
500        None
501    }
502}

506        for chunk in &self.chunks {
507            chunk.fmt(f)?;
508        }
509        Ok(())
510    }

513#[derive(Debug, Copy, Clone, PartialEq, Eq)]
514pub(crate) struct GroupTag(usize);
515
516#[derive(Debug, Copy, Clone, PartialEq, Eq)]
517pub(crate) enum GroupKind {
518    /// Most chunks are regular chunks and are not of interest.
519    Regular,
520    /// This is a chunk that has a list of expression in it, for example:
521    /// a call, a method call, an array literal, a tuple literal, etc.
522    /// `prefix_width` is the width of whatever is before the actual expression list.
523    /// For example, for an array this is 1 (for "["), for a vector it's 2 ("@["), etc.
524    ExpressionList { prefix_width: usize, expressions_count: usize },
525    /// This is a chunk for a lambda argument that is the last expression of an ExpressionList.
526    /// `first_line_width` is the width of the first line of the lambda argument: the parameters
527    /// list and the left bracket.
528    LambdaAsLastExpressionInList { first_line_width: usize, indentation: Option<i32> },
529    /// The body of a lambda.
530    /// We track this as a group kind so that when we have to write it, if it doesn't
531    /// fit in the current line and it's not a block, instead of splitting that expression
532    /// somewhere that's probably undesired, we'll "turn it" into a block
533    /// (write the "{" and "}" delimiters) and write the lambda body in the next line.
534    LambdaBody {
535        block_statement_count: Option<usize>,
536        has_comments: bool,
537        lambda_has_return_type: bool,
538    },
539    /// A method call.
540    /// We track all this information to see, if we end up needing to format this call
541    /// in multiple lines, if we can write everything up to the left parentheses (inclusive)
542    /// in one line, and just the call arguments in multiple lines.
543    MethodCall {
544        /// This is the width of the group until the left parenthesis (inclusive).
545        width_until_left_paren_inclusive: usize,
546        /// Are there newlines before the left parentheses in this group?
547        has_newlines_before_left_paren: bool,
548        /// Is this method call the left-hand side of a call chain? If so, this is true,
549        /// otherwise this is false an it means it's the outermost call.
550        lhs: bool,
551    },
552    /// The value of an assignment or let statement. We know this is the last group in a chunk so
553    /// if it doesn't fit in the current line but it fits in the next line, we can
554    /// write a newline, indent, and put it there (instead of writing the value in
555    /// multiple lines).
556    AssignValue,
557}
558
559impl GroupKind {
560    fn is_method_call(&self) -> bool {
561        matches!(self, GroupKind::MethodCall { .. })
562    }
563
564    fn is_expression_list(&self) -> bool {
565        matches!(self, GroupKind::ExpressionList { .. })
566    }
567}

569/// Interface for creating TextChunks.
570pub(crate) struct ChunkFormatter<'a, 'b>(&'b mut Formatter<'a>);
571
572impl<'a, 'b> ChunkFormatter<'a, 'b> {
573    pub(crate) fn new(formatter: &'b mut Formatter<'a>) -> Self {
574        Self(formatter)
575    }
576
577    /// Stops writing to the current buffer for the duration of the `f` call, which takes
578    /// a formatter to write to. Then, returns a `TextChunk` with the written text.
579    ///
580    /// This allows a caller to format pieces of code and then pre-process them before
581    /// writing to the main buffer.
582    pub(crate) fn chunk(&mut self, f: impl FnOnce(&mut Formatter)) -> TextChunk {
583        let previous_buffer = std::mem::take(&mut self.0.buffer);
584        let previous_indentation = self.0.indentation;
585        let previous_in_chunk = self.0.in_chunk;
586        self.0.indentation = 0;
587        self.0.in_chunk = true;
588
589        f(self.0);
590
591        self.0.indentation = previous_indentation;
592        self.0.in_chunk = previous_in_chunk;
593
594        let buffer = std::mem::replace(&mut self.0.buffer, previous_buffer);
595        TextChunk::new(buffer.contents())
596    }
597
598    /// Stops writing to the current buffer, skips comments and whitespaces (formatting them)
599    /// and returns the formatted result as a `TextChunk`.
600    pub(crate) fn skip_comments_and_whitespace_chunk(&mut self) -> TextChunk {
601        self.chunk(|formatter| {
602            formatter.skip_comments_and_whitespace();
603        })
604    }
605
606    pub(super) fn new_group_tag(&mut self) -> GroupTag {
607        self.0.new_group_tag()
608    }
609
610    pub(super) fn bump(&mut self) -> Token {
611        self.0.bump()
612    }
613}

619

621        self.0
622    }

625impl<'a> Formatter<'a> {
626    /// Returns an object that has a `chunk` method to get a TextChunk.
627    /// This method exists so that we can't mix the two operation modes:
628    /// using the formatter directly while writing to the buffer, or creating text chunks.
629    pub(super) fn chunk_formatter(&mut self) -> ChunkFormatter<'a, '_> {
630        ChunkFormatter::new(self)
631    }
632
633    /// Main interface to format a chunk group.
634    /// Here it's determined if the chunk will group in a single line or multiple lines.
635    pub(super) fn format_chunk_group(&mut self, group: ChunkGroup) {
636        let previous_indentation = self.indentation;
637        self.format_chunk_group_impl(group);
638        self.indentation = previous_indentation;
639    }
640
641    pub(super) fn format_chunk_group_impl(&mut self, group: ChunkGroup) {
642        if let GroupKind::LambdaAsLastExpressionInList { indentation: Some(indentation), .. } =
643            group.kind
644        {
645            let previous_indentation = self.indentation;
646            self.indentation = indentation;
647            self.format_chunks_group_impl_without_lambda_handling(group);
648            self.indentation = previous_indentation;
649        } else {
650            self.format_chunks_group_impl_without_lambda_handling(group);
651        }
652    }
653
654    pub(super) fn format_chunks_group_impl_without_lambda_handling(
655        &mut self,
656        mut group: ChunkGroup,
657    ) {
658        let chunks_width = group.width();
659        let total_width = self.current_line_width() + chunks_width;
660
661        if total_width > self.max_width {
662            // If this is a method call that doesn't fit in the current line, we check if
663            // everything that follows up to the left parentheses fits in the current line.
664            // If so, we write that and we'll end up formatting the arguments in multiple
665            // lines, instead of splitting this entire call chain in multiple lines.
666            //
667            // For example, a call like this:
668            //
669            // foo.bar.baz.qux(1)
670            //
671            // if it exceeds the maximum width, will end up being formatted like this:
672            //
673            // foo.bar.baz.qux(
674            //     1,
675            // )
676            //
677            // instead of like this (many more lines):
678            //
679            // foo
680            //     .bar
681            //     .baz
682            //     .qux(1)
683            //
684            // This is something that rustfmt seems to do too.
685            if let GroupKind::MethodCall {
686                width_until_left_paren_inclusive,
687                has_newlines_before_left_paren: false,
688                lhs: false,
689            } = group.kind
690            {
691                let mut total_width = self.current_line_width() + width_until_left_paren_inclusive;
692
693                if total_width <= self.max_width {
694                    // Check if this method call has a single lambda argument, like this:
695                    //
696                    // foo.bar.baz(|x| { x + 1 })
697                    //
698                    // In this case we can't just consider the width up to `(`, we need to include `|x| {`
699                    // because we'll want to format it in the same line as the `(` (we could put the lambda
700                    // on a separate line but it would be less readable).
701                    if let Some(first_line_width) = group.method_call_lambda_first_line_width() {
702                        total_width += first_line_width;
703                    }
704                }
705
706                if total_width <= self.max_width {
707                    // Check if this method call has another call or method call nested in it.
708                    // If not, it means this is the last nested call and after it we'll need to start
709                    // writing at least one closing parentheses. So the argument list will actually
710                    // have one less character available for writing, and that's why we (temporarily) decrease
711                    // max width.
712                    let expression_list_group = group.first_group().unwrap();
713                    let has_expression_list_or_call_group =
714                        expression_list_group.has_expression_list_or_method_call_group();
715                    if !has_expression_list_or_call_group {
716                        self.max_width -= 1;
717                    }
718
719                    // When a method call's group is formed, we indent after the first dot. But with that
720                    // indentation, and the arguments indentation, we'll end up with too much indentation,
721                    // so here we decrease it to compensate that.
722                    self.decrease_indentation();
723                    self.format_chunk_group_in_one_line(group);
724                    self.increase_indentation();
725
726                    if !has_expression_list_or_call_group {
727                        self.max_width += 1;
728                    }
729                    return;
730                }
731            }
732
733            // If this is an expression list with a single expression, see if we can fit whatever
734            // comes next until a line in the current line. For example, if we have this:
735            //
736            // foo(bar(baz(1)))
737            //
738            // then `foo(...)` is an ExpressionList. We check if `foo(` fits in the current line.
739            // If yes, we write it in the current line and continue. Then we'll find `bar(...)`,
740            // which is also an ExpressionList, and if `bar(` fits the current line, we'll write it,
741            // etc. But we only do this if we have nested calls (nested expression lists, etc.)
742            //
743            // This is to avoid formatting the above like this:
744            //
745            // foo(
746            //     bar(
747            //         baz(
748            //             1,
749            //         ),
750            //     ),
751            // )
752            //
753            // (rustfmt seems to do the same thing)
754            if let GroupKind::ExpressionList { prefix_width, expressions_count: 1 } = group.kind
755                && let Some(inner_group) = group.first_group()
756                && (inner_group.kind.is_expression_list() || inner_group.kind.is_method_call())
757            {
758                let total_width =
759                    self.current_line_width() + prefix_width + inner_group.width_until_line().0;
760                if total_width <= self.max_width {
761                    self.decrease_indentation();
762                    self.format_chunk_group_in_one_line(group);
763                    self.increase_indentation();
764                    return;
765                }
766            }
767        }
768
769        if group.force_multiple_lines {
770            self.format_chunk_group_in_multiple_lines(group);
771            return;
772        }
773
774        // When formatting an expression list we have to check if the last argument is a lambda,
775        // because we format that in a special way:
776        // 1. to compute the group width we'll consider only the `|...| {` part of the lambda
777        // 2. If it fits in a line, we'll format this expression list in a single line
778        // 3. However, an expression list is instructed to increase indentation after, say,
779        //    `(` or `[` (depending on the expression list) and then the `{` part of a lambda
780        //    will also increase the indentation, resulting in too much indentation.
781        // 4. For that reason we adjust the lambda to be formatted with the indentation
782        //    we have right that (that is, that of the call that holds the lambda).
783        //    We do that by setting the `indentation` field of the LambdaAsLastExpressionInList.
784        //
785        // Note that this logic is a bit complex because for method calls, the arguments list
786        // is in a group so all arguments can potentially be formatted in a single line, and
787        // that group has the `ExpressionList` kind. The method call itself has the `MethodCall`
788        // kind. So when determining the first line width of a method call with a lambda as
789        // the last argument we have to find the nested ExpressionList and do some nested calls.
790        if (group.kind.is_expression_list() || group.kind.is_method_call())
791            && group.has_lambda_as_last_expression_in_list()
792        {
793            let chunks_width = group.expression_list_width();
794            let total_width = self.current_line_width() + chunks_width;
795            if total_width <= self.max_width {
796                group.set_lambda_as_last_expression_in_list_indentation(self.indentation);
797                self.format_chunk_group_in_one_line(group);
798                return;
799            }
800        }
801
802        if group.has_newlines() {
803            self.format_chunk_group_in_multiple_lines(group);
804            return;
805        }
806
807        // Check if the group fits in the remainder of the current line.
808        if total_width > self.max_width {
809            // If this chunk is the value of an assignment (either a normal assignment or a let statement)
810            // and it doesn't fit the current line, we check if it fits the next line with an increased
811            // indentation.
812            //
813            // That way this:
814            //
815            // let x = foo(1, 2);
816            //                 ^
817            //                 assume the max width is here
818            //
819            // is formatted like this:
820            //
821            // let x =
822            //     foo(1, 2);
823            //
824            // instead of:
825            //
826            // let x = foo(
827            //     1,
828            //     2,
829            // )
830            if group.kind == GroupKind::AssignValue {
831                let total_width_next_line =
832                    (self.indentation as usize + 1) * self.config.tab_spaces + chunks_width;
833                if total_width_next_line <= self.max_width {
834                    // We might have trailing spaces
835                    // (for example a space after the `=` of a let statement or an assignment)
836                    self.trim_spaces();
837                    self.write_line_without_skipping_whitespace_and_comments();
838                    self.increase_indentation();
839                    self.write_indentation();
840                    self.format_chunk_group_in_one_line(group);
841                    self.decrease_indentation();
842                    return;
843                }
844            }
845
846            // If a lambda body doesn't fit in the current line and it's not a block,
847            // we can turn it into a block and write it in the next line, so its contents fit.
848            if let GroupKind::LambdaBody { block_statement_count: None, .. } = group.kind {
849                // Try to format it again in the next line, but we don't want to recurse
850                // infinitely so we change the group kind.
851                group.kind = GroupKind::Regular;
852                self.write("{");
853                self.trim_spaces();
854                self.increase_indentation();
855                self.write_line_without_skipping_whitespace_and_comments();
856                self.write_indentation();
857                self.format_chunk_group_impl(group);
858
859                // If this lambda was in an expression list and it was formatted in multiple
860                // lines, it might be that the trailing comma happened after the lambda body:
861                //
862                // foo(
863                //     1,
864                //     |lambda| body,
865                // )
866                //
867                // Because we attach commas to the last text to avoid splitting it, the body
868                // in this case is "body,", so if we end up writing it as a block it will
869                // look like this:
870                //
871                // foo(
872                //     1,
873                //     |lambda| {
874                //         body,
875                //     }
876                // )
877                //
878                // So, if after writing the body we find a comma (there will be at most one)
879                // we remove it, but place it after the right brace, so it looks like this:
880                //
881                // foo(
882                //     1,
883                //     |lambda| {
884                //         body
885                //     },
886                // )
887                let comma_trimmed = self.trim_comma();
888                self.decrease_indentation();
889                self.write_line_without_skipping_whitespace_and_comments();
890                self.write_indentation();
891                self.write("}");
892                if comma_trimmed {
893                    self.write(",");
894                }
895                return;
896            }
897
898            self.format_chunk_group_in_multiple_lines(group);
899            return;
900        }
901
902        // At this point we determined we are going to write this group in a single line.
903        // If the current group is a lambda body that is a block with a single statement, like this:
904        //
905        // |x| { 1 + 2 }
906        //
907        // given that we determined the block fits the current line, if we remove the surrounding
908        // `{ .. }` it will still fit the current line, and reduce some noise from the code
909        // (this is what rustfmt seems to do too).
910        if let GroupKind::LambdaBody {
911            block_statement_count: Some(1),
912            has_comments: false,
913            lambda_has_return_type: false,
914        } = group.kind
915        {
916            self.format_lambda_body_removing_braces(group);
917            return;
918        }
919
920        self.format_chunk_group_in_one_line(group);
921    }

923    pub(super) fn format_chunk_group_in_one_line(&mut self, group: ChunkGroup) {
924        for chunk in group.chunks {
925            match chunk {
926                Chunk::Text(text_chunk, _) => {
927                    self.write(&text_chunk.string);
928                }
929                Chunk::TrailingComment(text_chunk, _) => {
930                    self.write(&text_chunk.string);
931                }
932                Chunk::LeadingComment(text_chunk) => {
933                    self.write(&text_chunk.string);
934                    self.write_space_without_skipping_whitespace_and_comments();
935                }
936                Chunk::Group(chunks) => self.format_chunk_group_impl(chunks),
937                Chunk::SpaceOrLine => self.write_space_without_skipping_whitespace_and_comments(),
938                Chunk::IncreaseIndentation => self.increase_indentation(),
939                Chunk::DecreaseIndentation => self.decrease_indentation(),
940                Chunk::PushIndentation => self.push_indentation(),
941                Chunk::PopIndentation => self.pop_indentation(),
942                Chunk::TrailingComma | Chunk::Line { .. } => (),
943            }
944        }
945    }

947    pub(super) fn format_chunk_group_in_multiple_lines(&mut self, group: ChunkGroup) {
948        let chunks = group.prepare_for_multiple_lines();
949
950        let mut last_was_space_or_line = false;
951
952        // Indentation increases when a chunk will exceed the max width
953        let mut increased_indentation = 0;
954
955        for chunk in chunks.chunks {
956            if last_was_space_or_line {
957                if chunks.one_chunk_per_line {
958                    self.write_line_without_skipping_whitespace_and_comments();
959                    self.write_indentation();
960                } else {
961                    // "+ 1" because we still need to add a space before the next chunk
962                    if self.current_line_width() + chunk.width() + 1 > self.max_width {
963                        self.write_line_without_skipping_whitespace_and_comments();
964                        self.write_indentation();
965                    } else {
966                        self.write_space_without_skipping_whitespace_and_comments();
967                    }
968                }
969            }
970
971            last_was_space_or_line = false;
972
973            match chunk {
974                Chunk::Text(text_chunk, verbatim) => {
975                    if verbatim {
976                        self.write(&text_chunk.string);
977                    } else if text_chunk.has_newlines {
978                        self.write_chunk_lines(&text_chunk.string, false);
979                    } else {
980                        // If we didn't exceed the max width, but this chunk will, insert a newline,
981                        // increase indentation and indent (the indentation will be undone
982                        // after `format_chunks` finishes).
983                        // This is the logic to automatically wrap a line when a ChunkGroup doesn't
984                        // have Line or SpaceOrLine in it.
985                        if self.current_line_width() <= self.max_width
986                            && self.current_line_width() + text_chunk.width > self.max_width
987                            && !self.buffer.ends_with_space()
988                        {
989                            self.write_line_without_skipping_whitespace_and_comments();
990                            self.increase_indentation();
991                            self.write_indentation();
992                            increased_indentation += 1;
993                        }
994                        self.write(&text_chunk.string);
995                    }
996                }
997                Chunk::TrailingComment(text_chunk, at_block_end) => {
998                    let is_comment = true;
999                    self.write_chunk_lines(&text_chunk.string, is_comment);
1000                    if !at_block_end {
1001                        self.write_line_without_skipping_whitespace_and_comments();
1002                        self.write_indentation();
1003                    }
1004                }
1005                Chunk::LeadingComment(text_chunk) => {
1006                    let ends_with_multiple_newlines = text_chunk.string.ends_with("\n\n");
1007                    let ends_with_newline =
1008                        ends_with_multiple_newlines || text_chunk.string.ends_with('\n');
1009                    let is_comment = true;
1010                    self.write_chunk_lines(text_chunk.string.trim(), is_comment);
1011
1012                    // Respect whether the leading comment had a newline before what comes next or not
1013                    if ends_with_multiple_newlines {
1014                        // Remove any indentation that could exist (we'll add it later)
1015                        self.buffer.trim_spaces();
1016                        self.write_multiple_lines_without_skipping_whitespace_and_comments();
1017                        self.write_indentation();
1018                    } else if ends_with_newline {
1019                        self.write_line_without_skipping_whitespace_and_comments();
1020                        self.write_indentation();
1021                    } else {
1022                        self.write_space_without_skipping_whitespace_and_comments();
1023                    }
1024                }
1025                Chunk::Group(mut group) => {
1026                    if chunks.force_multiline_on_children_with_same_tag_if_multiline
1027                        && chunks.tag == group.tag
1028                    {
1029                        group.force_multiple_lines = true;
1030                        group.force_multiline_on_children_with_same_tag_if_multiline = true;
1031                    }
1032
1033                    self.format_chunk_group_impl(group);
1034                }
1035                Chunk::Line { two } => {
1036                    if two {
1037                        self.write_multiple_lines_without_skipping_whitespace_and_comments();
1038                    } else {
1039                        self.write_line_without_skipping_whitespace_and_comments();
1040                    }
1041                    self.write_indentation();
1042                }
1043                Chunk::SpaceOrLine => {
1044                    last_was_space_or_line = true;
1045                }
1046                Chunk::IncreaseIndentation => {
1047                    self.increase_indentation();
1048                }
1049                Chunk::DecreaseIndentation => {
1050                    self.decrease_indentation();
1051                }
1052                Chunk::PushIndentation => {
1053                    self.push_indentation();
1054                }
1055                Chunk::PopIndentation => {
1056                    self.pop_indentation();
1057
1058                    // Any increased indentation that we were planning to undo must not be undone
1059                    // if we change the current indentation to something completely different.
1060                    increased_indentation = 0;
1061                }
1062                Chunk::TrailingComma => {
1063                    unreachable!(
1064                        "TrailingComma should have been removed by `prepare_for_multiple_lines`"
1065                    )
1066                }
1067            }
1068        }
1069
1070        // Reset indentation back to what it was before we started formatting this group
1071        for _ in 0..increased_indentation {
1072            self.decrease_indentation();
1073        }
1074    }

1076    fn format_lambda_body_removing_braces(&mut self, group: ChunkGroup) {
1077        // Write to an intermediate string so we can remove the braces if needed.
1078        let text_chunk = self.chunk_formatter().chunk(|formatter| {
1079            formatter.format_chunk_group_in_one_line(group);
1080        });
1081        let string = text_chunk.string;
1082
1083        // Don't remove the braces if the lambda's body is a Semi expression.
1084        if string.ends_with("; }") || string.ends_with("; },") {
1085            self.write(&string);
1086            return;
1087        }
1088
1089        let string = string.strip_prefix("{ ").unwrap();
1090
1091        // The lambda might have a trailing comma if it's inside an arguments list
1092        if let Some(string) = string.strip_suffix(" },") {
1093            self.write(string);
1094            self.write(",");
1095        } else {
1096            let string = string.strip_suffix(" }").unwrap();
1097            self.write(string);
1098        }
1099    }

1101    /// Appends the string to the current buffer line by line, with some pre-checks.
1102    fn write_chunk_lines(&mut self, string: &str, is_comment: bool) {
1103        // Here we also wrap comments if this chunk is a comment.
1104        // The logic involves checking whether each line is part of a line or block comment
1105        // and wrapping those accordingly.
1106        let mut inside_block_comment = false;
1107
1108        let lines: Vec<_> = string.lines().collect();
1109
1110        let mut index = 0;
1111        while index < lines.len() {
1112            let mut line = lines[index];
1113
1114            let starts_with_space = line.starts_with(' ');
1115            let is_line_comment =
1116                is_comment && !inside_block_comment && line.trim_start().starts_with("//");
1117            let is_block_comment =
1118                is_comment && !inside_block_comment && line.trim_start().starts_with("/*");
1119
1120            // Don't indent the first line (it should already be indented).
1121            // Also don't indent if the current line already has a space as the last char
1122            // (it means it's already indented)
1123            if index > 0 && !self.buffer.ends_with_space() {
1124                self.write_line_without_skipping_whitespace_and_comments();
1125                // Only indent if the line doesn't start with a space. When that happens
1126                // it's likely a block comment part that we don't want to modify.
1127                if !line.starts_with(' ') {
1128                    self.write_indentation();
1129                }
1130            }
1131
1132            // If we already have a space in the buffer and the line starts with a space,
1133            // don't repeat that space.
1134            if self.buffer.ends_with_space() && line.starts_with(' ') {
1135                line = line.trim_start();
1136            }
1137
1138            if is_line_comment && self.config.wrap_comments {
1139                if starts_with_space && !self.buffer.ends_with_space() {
1140                    self.write(" ");
1141                }
1142                self.write_comment_with_prefix(
1143                    line.trim_start().strip_prefix("//").unwrap_or(line),
1144                    "//",
1145                );
1146            } else if (is_block_comment || inside_block_comment) && self.config.wrap_comments {
1147                // Only append a space if it's the start of a block comment (no leading spaces in following lines)
1148                if starts_with_space && is_block_comment && !self.buffer.ends_with_space() {
1149                    self.write(" ");
1150                }
1151                self.write_comment_with_prefix(line, "");
1152                if inside_block_comment {
1153                    self.start_new_line();
1154                }
1155                inside_block_comment = true;
1156            } else if index < lines.len() - 1 {
1157                // Trim the end of all lines except the last one, to avoid trailing spaces
1158                self.write(line.trim_end());
1159            } else {
1160                self.write(line);
1161            }
1162
1163            index += 1;
1164
1165            // If a newline comes next too, write multiple lines to preserve original formatting
1166            while index < lines.len() && lines[index].is_empty() {
1167                self.write_multiple_lines_without_skipping_whitespace_and_comments();
1168                index += 1;
1169            }
1170
1171            if inside_block_comment && line.trim_end().ends_with("*/") {
1172                inside_block_comment = false;
1173            }
1174        }
1175    }
