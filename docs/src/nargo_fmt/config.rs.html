7macro_rules! config {

9        pub struct Config {
10            $(

18                $(
19                    if let Some(value) = toml.$field_name {

21                    }
22                )+
23            }

60impl Config {
61    /// Reads a Config starting at the given path and going through the path parents
62    /// until a `noirfmt.toml` file is found in one of them or the root is reached.
63    pub fn read(mut path: &Path) -> Result<Self, ConfigError> {
64        loop {
65            let config_path = path.join("noirfmt.toml");
66            if config_path.exists() {
67                match std::fs::read_to_string(&config_path) {
68                    Ok(input) => return Self::of(&input, &config_path),
69                    Err(cause) => return Err(ConfigError::ReadFailed(config_path, cause)),
70                };
71            }
72
73            let Some(parent_path) = path.parent() else {
74                return Ok(Config::default());
75            };
76
77            path = parent_path;
78        }
79    }
80
81    pub fn of(s: &str, path: &Path) -> Result<Self, ConfigError> {
82        let mut config = Self::default();
83        let toml =
84            toml::from_str(s).map_err(|err| ConfigError::MalformedFile(path.to_path_buf(), err))?;
85        config.fill_from_toml(toml);
86        Ok(config)
87    }
88}
