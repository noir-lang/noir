4//! The Noir formatter.
5//!
6//! It works by using two techniques:
7//!
8//! 1. Surfing a parsed module by using a lexer.
9//! 2. Trying to not exceed the maximum line width by formatting to intermediate chunk (see chunks.rs)
10//!
11//! What is lexer surfing?
12//!
13//! Suppose we need to format this code:
14//!
15//! fn foo ( ) { let x : Field = ( 2 , 3 ) ; }
16//!
17//! We first parse the above code so we end up with a ParsedModule. Next we traverse this module
18//! contents and process each item, statement, expression, type, etc., we find.
19//!
20//! For example, the first thing we'll find is a function. We know it has no visibility and no doc
21//! comments so we can expect an `fn` keyword to be there. We write it. Next will come the identifier.
22//! For that we "skip" any spaces and comments between `fn` and `foo`, writing only one space instead
23//! of possibly multiple spaces, then write "foo". If there were comments between `fn` and `foo`,
24//! we write them (the formatter will never lose comments).
25//!
26//! Next we know there are no generics, so we can expect a `(`, etc.
27//!
28//! In this way we go token by token, inserting newlines when needed, removing extra spaces,
29//! indenting things as we go deep inside structures, etc.
30//!
31//! But that's not all. The formatter will try to not exceed the configurable maximum width.
32//! It will do that but, for simplicity, only for function parameters list, statements and expressions
33//! (we assume an `impl Foo ...` line won't exceed the maximum length, and if it does it's not a big deal,
34//! or we can always improve things later). For this, read the comments in chunks.rs.

45pub fn format(source: &str, parsed_module: ParsedModule, config: &Config) -> String {
46    let mut formatter = Formatter::new(source, config);
47    formatter.format_program(parsed_module);
48    formatter.buffer.contents()
49}

62#[cfg(test)]
63pub(crate) fn assert_format_with_config(src: &str, expected: &str, config: Config) {
64    use noirc_frontend::parser;
65
66    let (parsed_module, errors) = parser::parse_program_with_dummy_file(src);
67    let errors: Vec<_> = errors.into_iter().filter(|error| !error.is_warning()).collect();
68    if !errors.is_empty() {
69        panic!("Expected no errors, got: {errors:?}");
70    }
71    let result = format(src, parsed_module, &config);
72    if result != expected {
73        println!("Expected:\n~~~\n{expected}\n~~~\nGot:\n~~~\n{result}\n~~~");
74    }
75
76    similar_asserts::assert_eq!(result, expected);
77
78    let src = &result;
79    let (parsed_module, errors) = parser::parse_program_with_dummy_file(src);
80    let errors: Vec<_> = errors.into_iter().filter(|error| !error.is_warning()).collect();
81    if !errors.is_empty() {
82        panic!("Expected no errors in idempotent check, got: {errors:?}");
83    }
84    let result = format(src, parsed_module, &config);
85    if result != expected {
86        println!("Expected (idempotent):\n~~~\n{expected}\n~~~\nGot:\n~~~\n{result}\n~~~");
87    }
88    similar_asserts::assert_eq!(result, expected, "idempotent check failed");
89}
