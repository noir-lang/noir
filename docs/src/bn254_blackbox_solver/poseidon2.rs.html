14struct Poseidon2<'a> {
15    config: &'a Poseidon2Config,
16}
17
18impl Poseidon2<'_> {
19    fn new() -> Self {
20        Poseidon2 { config: &POSEIDON2_CONFIG }
21    }
22
23    fn single_box(x: FieldElement) -> FieldElement {
24        let s = x * x;
25        s * s * x
26    }
27
28    fn s_box(input: &mut [FieldElement]) {
29        for i in input {
30            *i = Self::single_box(*i);
31        }
32    }
33
34    fn add_round_constants(&self, state: &mut [FieldElement], round: usize) {
35        for (state_element, constant_element) in
36            state.iter_mut().zip(self.config.round_constant[round])
37        {
38            *state_element += constant_element;
39        }
40    }
41
42    /// Algorithm is taken directly from the Poseidon2 implementation in Barretenberg crypto module.
43    fn matrix_multiplication_4x4(input: &mut [FieldElement]) {
44        assert!(input.len() == 4);
45        let t0 = input[0] + input[1]; // A + B
46        let t1 = input[2] + input[3]; // C + D
47        let mut t2 = input[1] + input[1]; // 2B
48        t2 += t1; // 2B + C + D
49        let mut t3 = input[3] + input[3]; // 2D
50        t3 += t0; // 2D + A + B
51        let mut t4 = t1 + t1;
52        t4 += t4;
53        t4 += t3; // A + B + 4C + 6D
54        let mut t5 = t0 + t0;
55        t5 += t5;
56        t5 += t2; // 4A + 6B + C + D
57        let t6 = t3 + t5; // 5A + 7B + C + 3D
58        let t7 = t2 + t4; // A + 3B + 5C + 7D
59        input[0] = t6;
60        input[1] = t5;
61        input[2] = t7;
62        input[3] = t4;
63    }
64
65    fn internal_m_multiplication(&self, input: &mut [FieldElement]) {
66        let mut sum = FieldElement::zero();
67        for i in input.iter() {
68            sum += *i;
69        }
70        for (index, i) in input.iter_mut().enumerate() {
71            *i = *i * self.config.internal_matrix_diagonal[index];
72            *i += sum;
73        }
74    }
75
76    pub(crate) fn permutation(
77        &self,
78        inputs: &[FieldElement],
79    ) -> Result<Vec<FieldElement>, BlackBoxResolutionError> {
80        if inputs.len() != self.config.t as usize {
81            return Err(BlackBoxResolutionError::Failed(
82                acir::BlackBoxFunc::Poseidon2Permutation,
83                format!("Expected {} values but encountered {}", self.config.t, inputs.len()),
84            ));
85        }
86        // Read witness assignments
87        let mut state = [FieldElement::zero(); 4];
88        for (index, input) in inputs.iter().enumerate() {
89            state[index] = *input;
90        }
91        // Apply 1st linear layer
92        Self::matrix_multiplication_4x4(&mut state);
93
94        // First set of external rounds
95        let rf_first = self.config.rounds_f / 2;
96        for r in 0..rf_first {
97            self.add_round_constants(&mut state, r as usize);
98            Self::s_box(&mut state);
99            Self::matrix_multiplication_4x4(&mut state);
100        }
101        // Internal rounds
102        let p_end = rf_first + self.config.rounds_p;
103        for r in rf_first..p_end {
104            state[0] += self.config.round_constant[r as usize][0];
105            state[0] = Self::single_box(state[0]);
106            self.internal_m_multiplication(&mut state);
107        }
108
109        // Remaining external rounds
110        let num_rounds = self.config.rounds_f + self.config.rounds_p;
111        for i in p_end..num_rounds {
112            self.add_round_constants(&mut state, i as usize);
113            Self::s_box(&mut state);
114            Self::matrix_multiplication_4x4(&mut state);
115        }
116        Ok(state.into())
117    }
118}

164    fn run_both_poseidon2_permutations(
165        inputs: Vec<FieldElement>,
166    ) -> (Vec<ark_bn254::Fr>, Vec<ark_bn254::Fr>) {
167        let poseidon2_t = POSEIDON2_CONFIG.t as usize;
168        let poseidon2_d = 5;
169        let rounds_f = POSEIDON2_CONFIG.rounds_f as usize;
170        let rounds_p = POSEIDON2_CONFIG.rounds_p as usize;
171        let mat_internal_diag_m_1: Vec<ark_bn254_v04::Fr> =
172            POSEIDON2_CONFIG.internal_matrix_diagonal.into_iter().map(into_old_ark_field).collect();
173        let mat_internal = vec![];
174        let round_constants: Vec<Vec<ark_bn254_v04::Fr>> = POSEIDON2_CONFIG
175            .round_constant
176            .into_iter()
177            .map(|fields| fields.into_iter().map(into_old_ark_field).collect())
178            .collect();
179
180        let external_poseidon2 = zkhash::poseidon2::poseidon2::Poseidon2::new(&Arc::new(
181            zkhash::poseidon2::poseidon2_params::Poseidon2Params::new(
182                poseidon2_t,
183                poseidon2_d,
184                rounds_f,
185                rounds_p,
186                &mat_internal_diag_m_1,
187                &mat_internal,
188                &round_constants,
189            ),
190        ));
191
192        let result =
193            poseidon2_permutation(&inputs).unwrap().into_iter().map(|x| x.into_repr()).collect();
194
195        let expected_result = external_poseidon2.permutation(
196            &inputs.into_iter().map(into_old_ark_field).collect::<Vec<ark_bn254_v04::Fr>>(),
197        );
198        (result, expected_result.into_iter().map(into_new_ark_field).collect())
199    }
