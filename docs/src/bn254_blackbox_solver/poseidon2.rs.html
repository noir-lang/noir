19struct Poseidon2<'a> {
20    config: &'a Poseidon2Config,
21}
22
23impl Poseidon2<'_> {
24    fn new() -> Self {
25        Poseidon2 { config: &POSEIDON2_CONFIG }
26    }
27
28    fn single_box(x: FieldElement) -> FieldElement {
29        let s = x * x;
30        s * s * x
31    }
32
33    fn s_box(input: &mut [FieldElement]) {
34        for i in input {
35            *i = Self::single_box(*i);
36        }
37    }
38
39    fn add_round_constants(&self, state: &mut [FieldElement], round: usize) {
40        for (state_element, constant_element) in
41            state.iter_mut().zip_eq(self.config.round_constant[round])
42        {
43            *state_element += constant_element;
44        }
45    }
46
47    /// Algorithm is taken directly from the Poseidon2 implementation in Barretenberg crypto module.
48    fn matrix_multiplication_4x4(input: &mut [FieldElement]) {
49        assert!(input.len() == 4);
50        let t0 = input[0] + input[1]; // A + B
51        let t1 = input[2] + input[3]; // C + D
52        let mut t2 = input[1] + input[1]; // 2B
53        t2 += t1; // 2B + C + D
54        let mut t3 = input[3] + input[3]; // 2D
55        t3 += t0; // 2D + A + B
56        let mut t4 = t1 + t1;
57        t4 += t4;
58        t4 += t3; // A + B + 4C + 6D
59        let mut t5 = t0 + t0;
60        t5 += t5;
61        t5 += t2; // 4A + 6B + C + D
62        let t6 = t3 + t5; // 5A + 7B + C + 3D
63        let t7 = t2 + t4; // A + 3B + 5C + 7D
64        input[0] = t6;
65        input[1] = t5;
66        input[2] = t7;
67        input[3] = t4;
68    }
69
70    fn internal_m_multiplication(&self, input: &mut [FieldElement]) {
71        let mut sum = FieldElement::zero();
72        for i in input.iter() {
73            sum += *i;
74        }
75        for (index, i) in input.iter_mut().enumerate() {
76            *i = *i * self.config.internal_matrix_diagonal[index];
77            *i += sum;
78        }
79    }
80
81    pub(crate) fn permutation(
82        &self,
83        inputs: &[FieldElement],
84    ) -> Result<Vec<FieldElement>, BlackBoxResolutionError> {
85        if inputs.len() != self.config.t as usize {
86            return Err(BlackBoxResolutionError::Failed(
87                acir::BlackBoxFunc::Poseidon2Permutation,
88                format!("Expected {} values but encountered {}", self.config.t, inputs.len()),
89            ));
90        }
91        // Read witness assignments
92        let mut state = [FieldElement::zero(); 4];
93        for (index, input) in inputs.iter().enumerate() {
94            state[index] = *input;
95        }
96        // Apply 1st linear layer
97        Self::matrix_multiplication_4x4(&mut state);
98
99        // First set of external rounds
100        let rf_first = self.config.rounds_f / 2;
101        for r in 0..rf_first {
102            self.add_round_constants(&mut state, r as usize);
103            Self::s_box(&mut state);
104            Self::matrix_multiplication_4x4(&mut state);
105        }
106        // Internal rounds
107        let p_end = rf_first + self.config.rounds_p;
108        for r in rf_first..p_end {
109            state[0] += self.config.round_constant[r as usize][0];
110            state[0] = Self::single_box(state[0]);
111            self.internal_m_multiplication(&mut state);
112        }
113
114        // Remaining external rounds
115        let num_rounds = self.config.rounds_f + self.config.rounds_p;
116        for i in p_end..num_rounds {
117            self.add_round_constants(&mut state, i as usize);
118            Self::s_box(&mut state);
119            Self::matrix_multiplication_4x4(&mut state);
120        }
121        Ok(state.into())
122    }
123}
