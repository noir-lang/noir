11/// Converts a field element to u128, returning an error if it doesn't fit.
12fn field_to_u128_limb(
13    limb: &FieldElement,
14    func: BlackBoxFunc,
15) -> Result<u128, BlackBoxResolutionError> {
16    limb.try_into_u128().ok_or_else(|| {
17        BlackBoxResolutionError::Failed(
18            func,
19            format!("Limb {} is not less than 2^128", limb.to_hex()),
20        )
21    })
22}

24/// Performs multi scalar multiplication of points with scalars.
25pub fn multi_scalar_mul(
26    points: &[FieldElement],
27    scalars_lo: &[FieldElement],
28    scalars_hi: &[FieldElement],
29) -> Result<(FieldElement, FieldElement, FieldElement), BlackBoxResolutionError> {
30    if points.len() != 3 * scalars_lo.len() || scalars_lo.len() != scalars_hi.len() {
31        return Err(BlackBoxResolutionError::Failed(
32            BlackBoxFunc::MultiScalarMul,
33            "Points and scalars must have the same length".to_string(),
34        ));
35    }
36
37    // Collect all bases (affine points) and scalars for batch MSM
38    let mut bases = Vec::new();
39    let mut big_ints = Vec::new();
40
41    for i in (0..points.len()).step_by(3) {
42        if points[i + 2] > FieldElement::one() {
43            return Err(BlackBoxResolutionError::Failed(
44                BlackBoxFunc::MultiScalarMul,
45                "EmbeddedCurvePoint is malformed (non-boolean `is_infinite` flag)".to_string(),
46            ));
47        }
48        let point = create_point(points[i], points[i + 1], points[i + 2])
49            .map_err(|e| BlackBoxResolutionError::Failed(BlackBoxFunc::MultiScalarMul, e))?;
50
51        let scalar_low: u128 =
52            field_to_u128_limb(&scalars_lo[i / 3], BlackBoxFunc::MultiScalarMul)?;
53
54        let scalar_high: u128 =
55            field_to_u128_limb(&scalars_hi[i / 3], BlackBoxFunc::MultiScalarMul)?;
56
57        // Convert to BigInt<4>, using u64 limbs.
58        let limbs_array = [
59            scalar_low as u64,
60            (scalar_low >> 64) as u64,
61            scalar_high as u64,
62            (scalar_high >> 64) as u64,
63        ];
64        let scalar_bigint = BigInt::new(limbs_array);
65
66        // Check if this is smaller than the grumpkin modulus
67        if scalar_bigint >= ark_grumpkin::FrConfig::MODULUS {
68            // Format as hex string (big-endian, most significant limb first)
69            let hex_str = format!(
70                "{:016x}{:016x}{:016x}{:016x}",
71                limbs_array[3], limbs_array[2], limbs_array[1], limbs_array[0]
72            );
73            return Err(BlackBoxResolutionError::Failed(
74                BlackBoxFunc::MultiScalarMul,
75                format!("{hex_str} is not a valid grumpkin scalar"),
76            ));
77        }
78
79        bases.push(point);
80        big_ints.push(scalar_bigint);
81    }
82
83    // Perform batch multi-scalar multiplication
84    let output_point = ark_grumpkin::Projective::msm_bigint(&bases, &big_ints);
85    let output_point = ark_grumpkin::Affine::from(output_point);
86
87    if let Some((out_x, out_y)) = output_point.xy() {
88        Ok((
89            FieldElement::from_repr(out_x),
90            FieldElement::from_repr(out_y),
91            FieldElement::from(u128::from(output_point.is_zero())),
92        ))
93    } else {
94        Ok((FieldElement::from(0_u128), FieldElement::from(0_u128), FieldElement::from(1_u128)))
95    }
96}

98pub fn embedded_curve_add(
99    input1: [FieldElement; 3],
100    input2: [FieldElement; 3],
101) -> Result<(FieldElement, FieldElement, FieldElement), BlackBoxResolutionError> {
102    if input1[2] > FieldElement::one() || input2[2] > FieldElement::one() {
103        return Err(BlackBoxResolutionError::Failed(
104            BlackBoxFunc::EmbeddedCurveAdd,
105            "EmbeddedCurvePoint is malformed (non-boolean `is_infinite` flag)".to_string(),
106        ));
107    }
108
109    let point1 = create_point(input1[0], input1[1], input1[2])
110        .map_err(|e| BlackBoxResolutionError::Failed(BlackBoxFunc::EmbeddedCurveAdd, e))?;
111    let point2 = create_point(input2[0], input2[1], input2[2])
112        .map_err(|e| BlackBoxResolutionError::Failed(BlackBoxFunc::EmbeddedCurveAdd, e))?;
113
114    for point in [point1, point2] {
115        if point == ark_grumpkin::Affine::zero() {
116            return Err(BlackBoxResolutionError::Failed(
117                BlackBoxFunc::EmbeddedCurveAdd,
118                format!("Infinite input: embedded_curve_add({point1}, {point2})"),
119            ));
120        }
121    }
122
123    let res = ark_grumpkin::Affine::from(point1 + point2);
124    if let Some((res_x, res_y)) = res.xy() {
125        Ok((FieldElement::from_repr(res_x), FieldElement::from_repr(res_y), FieldElement::zero()))
126    } else {
127        assert!(res.is_zero());
128        Ok((FieldElement::from(0_u128), FieldElement::from(0_u128), FieldElement::from(1_u128)))
129    }
130}

132fn create_point(
133    x: FieldElement,
134    y: FieldElement,
135    is_infinite: FieldElement,
136) -> Result<ark_grumpkin::Affine, String> {
137    if is_infinite.is_one() {
138        return Ok(ark_grumpkin::Affine::zero());
139    } else if !is_infinite.is_zero() {
140        return Err("`is_infinite` flag is non-boolean".to_string());
141    }
142
143    let point = ark_grumpkin::Affine::new_unchecked(x.into_repr(), y.into_repr());
144    if !point.is_on_curve() {
145        return Err(format!("Point ({}, {}) is not on curve", x.to_hex(), y.to_hex()));
146    };
147    if !point.is_in_correct_subgroup_assuming_on_curve() {
148        return Err(format!("Point ({}, {}) is not in correct subgroup", x.to_hex(), y.to_hex()));
149    };
150    Ok(point)
151}

342    fn msm_against_add_and_mul(
343        points: &[FieldElement],
344        scalars_lo: &[FieldElement],
345        scalars_hi: &[FieldElement],
346    ) {
347        // Manual MSM via add and mul
348        let mut output_point = ark_grumpkin::Affine::zero();
349        for i in (0..points.len()).step_by(3) {
350            let point = create_point(points[i], points[i + 1], points[i + 2]).unwrap();
351
352            let scalar_low: u128 =
353                field_to_u128_limb(&scalars_lo[i / 3], BlackBoxFunc::MultiScalarMul).unwrap();
354
355            let scalar_high: u128 =
356                field_to_u128_limb(&scalars_hi[i / 3], BlackBoxFunc::MultiScalarMul).unwrap();
357
358            let mut bytes = scalar_high.to_be_bytes().to_vec();
359            bytes.extend_from_slice(&scalar_low.to_be_bytes());
360
361            let grumpkin_integer = BigUint::from_bytes_be(&bytes);
362
363            // Check if this is smaller than the grumpkin modulus
364            assert!(
365                grumpkin_integer < ark_grumpkin::FrConfig::MODULUS.into(),
366                "invalid grumpkin scalar",
367            );
368
369            let iteration_output_point =
370                ark_grumpkin::Affine::from(point.mul_bigint(grumpkin_integer.to_u64_digits()));
371
372            output_point = ark_grumpkin::Affine::from(output_point + iteration_output_point);
373        }
374
375        // Batch MSM
376        let output_point2 = multi_scalar_mul(points, scalars_lo, scalars_hi).unwrap();
377
378        // Checks both implementations have the same result
379        if let Some((out_x, out_y)) = output_point.xy() {
380            assert_eq!(FieldElement::from_repr(out_x), output_point2.0);
381            assert_eq!(FieldElement::from_repr(out_y), output_point2.1);
382        } else {
383            // Point at infinity
384            assert_eq!(output_point2.2, FieldElement::from(1u128));
385        }
386    }
