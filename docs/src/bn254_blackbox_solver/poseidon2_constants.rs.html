1/// Poseidon2 constants for BN254 field
2///
3/// This module contains the precomputed constants used in the Poseidon2 hash function
4/// for the BN254 curve.
5use acir::AcirField;
6use lazy_static::lazy_static;
7
8use crate::FieldElement;
9
10/// Configuration parameters for Poseidon2 permutation
11pub(crate) struct Poseidon2Config {
12    /// State width (or state size, number of field elements in the permutation state).
13    pub(crate) t: u32,
14
15    /// Number of full rounds (also called R_F).
16    /// In full rounds, the S-box is applied to all state elements.
17    pub(crate) rounds_f: u32,
18
19    /// Number of partial rounds (also called R_P).
20    /// In partial rounds, the S-box is applied only to the first state element.
21    pub(crate) rounds_p: u32,
22
23    /// Internal matrix diagonal values used in the linear layer of partial rounds.
24    pub(crate) internal_matrix_diagonal: [FieldElement; 4],
25
26    /// Round constants added to the state in each round.
27    pub(crate) round_constant: [[FieldElement; 4]; 64],
28}
29
30/// Helper function to convert hex strings to field elements
31pub(crate) fn field_from_hex(hex: &str) -> FieldElement {
32    FieldElement::from_be_bytes_reduce(&hex::decode(hex).expect("Should be passed only valid hex"))
33}
