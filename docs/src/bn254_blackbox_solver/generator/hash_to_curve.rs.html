12/// Hash a seed buffer into a point
13///
14/// # ALGORITHM DESCRIPTION
15///
16///  1. Initialize unsigned integer `attempt_count = 0`
17///  2. Copy seed into a buffer whose size is 2 bytes greater than `seed` (initialized to `0`)
18///  3. Interpret `attempt_count` as a byte and write into buffer at `[buffer.size() - 2]`
19///  4. Compute Blake3 hash of buffer
20///  5. Set the end byte of the buffer to `1`
21///  6. Compute Blake3 hash of buffer
22///  7. Interpret the two hash outputs as the high / low 256 bits of a 512-bit integer (big-endian)
23///  8. Derive x-coordinate of point by reducing the 512-bit integer modulo the curve's field modulus (Fq)
24///  9. Compute `y^2` from the curve formula `y^2 = x^3 + ax + b` (`a`, `b` are curve params. for BN254, `a = 0`, `b = 3`)
25///  10. IF `y^2` IS NOT A QUADRATIC RESIDUE:
26///
27///  a. increment `attempt_count` by 1 and go to step 2
28///  
29///  11. IF `y^2` IS A QUADRATIC RESIDUE:
30///
31///  a. derive y coordinate via `y = sqrt(y)`
32///
33///  b. Interpret most significant bit of 512-bit integer as a 'parity' bit
34///
35///  c. If parity bit is set AND `y`'s most significant bit is not set, invert `y`
36///
37///  d. If parity bit is not set AND `y`'s most significant bit is set, invert `y`
38///
39///  e. return (x, y)
40///
41///  N.B. steps c. and e. are because the `sqrt()` algorithm can return 2 values,
42///  we need to a way to canonically distinguish between these 2 values and select a "preferred" one
43pub(crate) fn hash_to_curve(seed: &[u8], attempt_count: u8) -> Affine {
44    let seed_size = seed.len();
45    // expand by 2 bytes to cover incremental hash attempts
46    let mut target_seed = seed.to_vec();
47    target_seed.extend_from_slice(&[0u8; 2]);
48
49    target_seed[seed_size] = attempt_count;
50    target_seed[seed_size + 1] = 0;
51    let hash_hi = blake3(&target_seed).expect("hash should succeed");
52    target_seed[seed_size + 1] = 1;
53    let hash_lo = blake3(&target_seed).expect("hash should succeed");
54
55    let mut hash = hash_hi.to_vec();
56    hash.extend_from_slice(&hash_lo);
57
58    // Here we reduce the 512 bit number modulo the base field modulus to calculate `x`
59    let x = Fq::from_be_bytes_mod_order(&hash);
60    let x = Fq::from_base_prime_field(x);
61
62    if let Some(point) = Affine::get_point_from_x_unchecked(x, false) {
63        let parity_bit = hash_hi[0] > 127;
64        let y_bit_set = point.y().unwrap().into_bigint().get_bit(0);
65        if (parity_bit && !y_bit_set) || (!parity_bit && y_bit_set) { -point } else { point }
66    } else {
67        hash_to_curve(seed, attempt_count + 1)
68    }
69}

72mod tests {
73
74    use ark_ec::AffineRepr;
75    use ark_ff::{BigInteger, PrimeField};
76
77    use super::hash_to_curve;
78
79    #[test]
80    fn smoke_test() {
81        let test_cases: [(&[u8], u8, (&str, &str)); 4] = [
82            (
83                &[],
84                0,
85                (
86                    "24c4cb9c1206ab5470592f237f1698abe684dadf0ab4d7a132c32b2134e2c12e",
87                    "0668b8d61a317fb34ccad55c930b3554f1828a0e5530479ecab4defe6bbc0b2e",
88                ),
89            ),
90            (
91                &[],
92                1,
93                (
94                    "24c4cb9c1206ab5470592f237f1698abe684dadf0ab4d7a132c32b2134e2c12e",
95                    "0668b8d61a317fb34ccad55c930b3554f1828a0e5530479ecab4defe6bbc0b2e",
96                ),
97            ),
98            (
99                &[1],
100                0,
101                (
102                    "107f1b633c6113f3222f39f6256f0546b41a4880918c86864b06471afb410454",
103                    "050cd3823d0c01590b6a50adcc85d2ee4098668fd28805578aa05a423ea938c6",
104                ),
105            ),
106            (
107                &[0x68, 0x65, 0x6c, 0x6c, 0x6f, 0x20, 0x77, 0x6f, 0x72, 0x6c, 0x64],
108                0,
109                (
110                    "037c5c229ae495f6e8d1b4bf7723fafb2b198b51e27602feb8a4d1053d685093",
111                    "10cf9596c5b2515692d930efa2cf3817607e4796856a79f6af40c949b066969f",
112                ),
113            ),
114        ];
115
116        for (seed, attempt_count, expected_point) in test_cases {
117            let point = hash_to_curve(seed, attempt_count);
118            assert!(point.is_on_curve());
119            assert_eq!(
120                hex::encode(point.x().unwrap().into_bigint().to_bytes_be()),
121                expected_point.0,
122                "Failed on x component with seed {seed:?}, attempt_count {attempt_count}"
123            );
124            assert_eq!(
125                hex::encode(point.y().unwrap().into_bigint().to_bytes_be()),
126                expected_point.1,
127                "Failed on y component with seed {seed:?}, attempt_count {attempt_count}"
128            );
129        }
130    }
