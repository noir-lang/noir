12pub(crate) const DEFAULT_DOMAIN_SEPARATOR: &[u8] = "DEFAULT_DOMAIN_SEPARATOR".as_bytes();
13const NUM_DEFAULT_GENERATORS: usize = 8;
14
15fn default_generators() -> &'static [Affine; NUM_DEFAULT_GENERATORS] {
16    static INSTANCE: OnceLock<[Affine; NUM_DEFAULT_GENERATORS]> = OnceLock::new();
17    INSTANCE.get_or_init(|| {
18        _derive_generators(DEFAULT_DOMAIN_SEPARATOR, NUM_DEFAULT_GENERATORS as u32, 0)
19            .try_into()
20            .expect("Should generate `NUM_DEFAULT_GENERATORS`")
21    })
22}

24/// Derives generator points.
25///
26/// # ALGORITHM DESCRIPTION
27///
28/// 1. Each generator has an associated "generator index" described by its location in the vector
29/// 2. a 64-byte preimage buffer is generated with the following structure:
30///     - bytes 0-31: BLAKE3 hash of domain_separator
31///     - bytes 32-63: generator index in big-endian form
32/// 3. The hash-to-curve algorithm is used to hash the above into a curve point.
33///
34/// NOTE: The domain separator is included to ensure that it is possible to derive independent sets of
35/// index-addressable generators.
36pub fn derive_generators(
37    domain_separator_bytes: &[u8],
38    num_generators: u32,
39    starting_index: u32,
40) -> Vec<Affine> {
41    // We cache a small number of the default generators so we can reuse them without needing to repeatedly recalculate them.
42    if domain_separator_bytes == DEFAULT_DOMAIN_SEPARATOR
43        && starting_index + num_generators <= NUM_DEFAULT_GENERATORS as u32
44    {
45        let start_index = starting_index as usize;
46        let end_index = (starting_index + num_generators) as usize;
47        default_generators()[start_index..end_index].to_vec()
48    } else {
49        _derive_generators(domain_separator_bytes, num_generators, starting_index)
50    }
51}
52
53fn _derive_generators(
54    domain_separator_bytes: &[u8],
55    num_generators: u32,
56    starting_index: u32,
57) -> Vec<Affine> {
58    let mut generator_preimage = [0u8; 64];
59    let domain_hash = blake3(domain_separator_bytes).expect("hash should succeed");
60    //1st 32 bytes are blake3 domain_hash
61    generator_preimage[..32].copy_from_slice(&domain_hash);
62
63    // Convert generator index in big-endian form
64    let mut res = Vec::with_capacity(num_generators as usize);
65    for i in starting_index..(starting_index + num_generators) {
66        let generator_be_bytes: [u8; 4] = i.to_be_bytes();
67        generator_preimage[32] = generator_be_bytes[0];
68        generator_preimage[33] = generator_be_bytes[1];
69        generator_preimage[34] = generator_be_bytes[2];
70        generator_preimage[35] = generator_be_bytes[3];
71        let generator = hash_to_curve(&generator_preimage, 0);
72        res.push(generator);
73    }
74    res
75}

85    #[test]
86    fn test_derive_generators() {
87        let res = derive_generators("test domain".as_bytes(), 128, 0);
88
89        let is_unique = |y: Affine, j: usize| -> bool {
90            for (i, res) in res.iter().enumerate() {
91                if i != j && *res == y {
92                    return false;
93                }
94            }
95            true
96        };
97
98        for (i, res) in res.iter().enumerate() {
99            assert!(is_unique(*res, i));
100            assert!(res.is_on_curve());
101        }
102    }
