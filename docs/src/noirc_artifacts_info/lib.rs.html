12#[derive(Debug, Serialize)]
13pub struct ProgramInfo {
14    pub package_name: String,
15    pub functions: Vec<FunctionInfo>,
16    #[serde(skip)]
17    pub unconstrained_functions_opcodes: usize,
18    pub unconstrained_functions: Vec<FunctionInfo>,

21impl From<ProgramInfo> for Vec<Row> {
22    fn from(program_info: ProgramInfo) -> Self {
23        let mut main = vecmap(program_info.functions, |function| {
24            row![
25                Fm->format!("{}", program_info.package_name),
26                Fc->format!("{}", function.name),
27                Fc->format!("{}", function.opcodes),
28                Fc->format!("{}", program_info.unconstrained_functions_opcodes),
29            ]
30        });
31        main.extend(vecmap(program_info.unconstrained_functions, |function| {
32            row![
33                Fm->format!("{}", program_info.package_name),
34                Fc->format!("{}", function.name),
35                Fc->format!("N/A"),
36                Fc->format!("{}", function.opcodes),
37            ]
38        }));
39        main
40    }
41}

49pub fn count_opcodes_and_gates_in_program(
50    compiled_program: ProgramArtifact,
51    package_name: String,
52) -> ProgramInfo {
53    let functions = compiled_program
54        .bytecode
55        .functions
56        .into_par_iter()
57        .map(|function| FunctionInfo {
58            name: function.function_name.clone(),
59            opcodes: function.opcodes.len(),
60        })
61        .collect();
62
63    let unconstrained_info: Vec<FunctionInfo> = compiled_program
64        .bytecode
65        .unconstrained_functions
66        .iter()
67        .map(|function| FunctionInfo {
68            name: function.function_name.clone(),
69            opcodes: function.bytecode.len(),
70        })
71        .collect();
72    let unconstrained_functions_opcodes = compiled_program
73        .bytecode
74        .unconstrained_functions
75        .into_par_iter()
76        .map(|function| function.bytecode.len())
77        .sum();
78
79    ProgramInfo {
80        package_name,
81        functions,
82        unconstrained_functions_opcodes,
83        unconstrained_functions: unconstrained_info,
84    }
85}

87pub fn show_info_report(info_report: InfoReport, json: bool) {
88    if json {
89        // Expose machine-readable JSON data.
90        println!("{}", serde_json::to_string(&info_report).unwrap());
91    } else {
92        // Otherwise print human-readable table.
93        if !info_report.programs.is_empty() {
94            let mut program_table = Table::new();
95            let titles =
96                row![Fm->"Package", Fm->"Function", Fm->"ACIR Opcodes", Fm->"Brillig Opcodes"];
97            let num_titles = titles.len();
98            program_table.set_titles(titles);
99
100            for program_info in info_report.programs {
101                let program_rows: Vec<Row> = program_info.into();
102                for row in program_rows {
103                    assert_eq!(row.len(), num_titles, "wrong number of columns");
104                    program_table.add_row(row);
105                }
106            }
107            program_table.printstd();
108        }
109    }
