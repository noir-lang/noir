1use std::path::PathBuf;

3use nargo::package::PackageType;
4use noirc_frontend::graph::CrateName;

7/// Errors covering situations where a package is either missing, malformed or does not pass semver
8/// validation checks.
9#[derive(Debug, Error)]
10pub enum ManifestError {
11    /// Package doesn't have a manifest file
12    #[error("cannot find a Nargo.toml for {0}")]
13    MissingFile(PathBuf),
14
15    #[error("Cannot read file {0} - does it exist?")]
16    ReadFailed(PathBuf),
17
18    #[error("Nargo.toml is missing a parent directory")]
19    MissingParent,
20
21    #[error("Missing `type` field in {0}")]
22    MissingPackageType(PathBuf),
23
24    #[error("Cannot use `{1}` for `type` field in {0}")]
25    InvalidPackageType(PathBuf, String),
26
27    /// Package manifest is unreadable.
28    #[error("Nargo.toml is badly formed, could not parse.\n\n{}", .0.message())]
29    MalformedFile(#[from] toml::de::Error),
30
31    #[error(
32        "Unexpected workspace definition found in {0}. If you're attempting to load this as a dependency, you may need to add a `directory` field to your `Nargo.toml` to show which package within the workspace to use"
33    )]
34    UnexpectedWorkspace(PathBuf),
35
36    #[error("Cannot find file {entry} which was specified as the `entry` field in {toml}")]
37    MissingEntryFile { toml: PathBuf, entry: PathBuf },
38
39    #[error(
40        r#"Cannot find file {entry} which is defaulted due to specifying `type = "{package_type}"` in {toml}"#
41    )]
42    MissingDefaultEntryFile { toml: PathBuf, entry: PathBuf, package_type: PackageType },
43
44    #[error("{} found in {toml}", if name.is_empty() { "Empty package name".into() } else { format!("Invalid package name `{name}`") })]
45    InvalidPackageName { toml: PathBuf, name: String },
46
47    #[error("{} found in {toml}", if name.is_empty() { "Empty dependency name".into() } else { format!("Invalid dependency name `{name}`") })]
48    InvalidDependencyName { toml: PathBuf, name: String },
49
50    #[error("Invalid directory path {directory} in {toml}: It must point to a subdirectory")]
51    InvalidDirectory { toml: PathBuf, directory: PathBuf },
52
53    /// Encountered error while downloading git repository.
54    #[error("{0}")]
55    GitError(String),
56
57    #[error("Selected package `{0}` was not found")]
58    MissingSelectedPackage(CrateName),
59
60    #[error("Default package was not found. Does {0} exist in your workspace?")]
61    MissingDefaultPackage(PathBuf),
62
63    #[error("Package `{0}` has type `bin` but you cannot depend on binary packages")]
64    BinaryDependency(CrateName),
65
66    #[error("No common ancestor between {root} and {current}")]
67    NoCommonAncestor { root: PathBuf, current: PathBuf },
68
69    #[error(transparent)]
70    SemverError(SemverError),
71
72    #[error("Cyclic package dependency found when processing {cycle}")]
73    CyclicDependency { cycle: String },
74}
75
76#[allow(clippy::enum_variant_names)]
77#[derive(Error, Debug, PartialEq, Eq, Clone)]
78pub enum SemverError {
79    #[error(
80        "Invalid value for `compiler_version` in package {package_name}. Requirements may only refer to full releases"
81    )]
82    InvalidCompilerVersionRequirement { package_name: CrateName, required_compiler_version: String },
83    #[error(
84        "Incompatible compiler version in package {package_name}. Required compiler version is {required_compiler_version} but the compiler version is {compiler_version_found}.\n Update the compiler_version field in Nargo.toml to >={required_compiler_version} or compile this project with version {required_compiler_version}"
85    )]
86    IncompatibleVersion {
87        package_name: CrateName,
88        required_compiler_version: String,
89        compiler_version_found: String,
90    },
91    #[error(
92        "Could not parse the required compiler version for package {package_name} in Nargo.toml. Error: {error}"
93    )]
94    CouldNotParseRequiredVersion { package_name: String, error: String },
95    #[error(
96        "Could not parse the package version for package {package_name} in Nargo.toml. Error: {error}"
97    )]
98    CouldNotParsePackageVersion { package_name: String, error: String },
99}
