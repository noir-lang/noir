5/// Creates a unique folder name for a GitHub repo
6/// by using its URL and tag
7fn resolve_folder_name(base: &url::Url, tag: &str) -> String {
8    let mut folder = PathBuf::from("");
9    for part in [base.domain().unwrap(), base.path(), tag] {
10        folder.push(part.trim_start_matches('/'));
11    }
12    folder.to_string_lossy().into_owned()
13}

15/// Path to the `nargo` directory under `$HOME`.
16fn nargo_crates() -> PathBuf {
17    dirs::home_dir().unwrap().join("nargo")
18}

20/// Target directory to download dependencies into, e.g.
21/// `$HOME/nargo/github.com/noir-lang/noir-bignum/v0.1.2`
22fn git_dep_location(base: &url::Url, tag: &str) -> PathBuf {
23    let folder_name = resolve_folder_name(base, tag);
24
25    nargo_crates().join(folder_name)
26}

32/// XXX: I'd prefer to use a GitHub library however, there
33/// does not seem to be an easy way to download a repo at a specific
34/// tag
35/// github-rs looks promising, however it seems to require an API token
36///
37/// One advantage of using "git clone" is that there is effectively no rate limit
38pub(crate) fn clone_git_repo(url: &str, tag: &str) -> Result<PathBuf, String> {
39    use std::process::Command;
40
41    let base = match url::Url::parse(url) {
42        Ok(base) => base,
43        Err(err) => return Err(err.to_string()),
44    };
45
46    let loc = git_dep_location(&base, tag);
47    if loc.exists() {
48        return Ok(loc);
49    }
50
51    Command::new("git")
52        .arg("-c")
53        .arg("advice.detachedHead=false")
54        .arg("clone")
55        .arg("--depth")
56        .arg("1")
57        .arg("--branch")
58        .arg(tag)
59        .arg(base.as_str())
60        .arg(&loc)
61        .status()
62        .expect("git clone command failed to start");
63
64    Ok(loc)
65}
