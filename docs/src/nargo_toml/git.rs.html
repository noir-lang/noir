6/// The deepest a clone root can sit below the cache root, in path components. Dependencies are
7/// downloaded to `<host>/<owner>/<name>/<tag>`, e.g. `github.com/owner/name/v1.0.0`, so a clone
8/// root is always exactly 4 components deep.
9const MAX_DEPENDENCY_CACHE_DEPTH: usize = 4;
10
11/// Lists every git dependency currently present in the global download cache, as paths relative
12/// to the cache root (e.g. `github.com/owner/name/v1.0.0`).
13///
14/// A directory is treated as a downloaded dependency when it contains a `.git` entry, which
15/// `git clone` always creates. This is host-agnostic: it doesn't assume any particular server
16/// or a fixed `owner/name` nesting depth.
17pub fn list_cached_git_dependencies() -> BTreeSet<PathBuf> {
18    collect_cached_git_dependencies(&nargo_crates())
19}
20
21/// Walks the dependency cache rooted at `cache_root`, returning the path (relative to `cache_root`)
22/// of every directory that contains a `.git` entry. Such a directory is a clone root, so we do not
23/// descend into it. Descent also stops at [`MAX_DEPENDENCY_CACHE_DEPTH`], which the contents of a
24/// clone never reach.
25fn collect_cached_git_dependencies(cache_root: &Path) -> BTreeSet<PathBuf> {
26    fn go(cache_root: &Path, dir: &Path, depth: usize, found: &mut BTreeSet<PathBuf>) {
27        let Ok(entries) = std::fs::read_dir(dir) else {
28            return;
29        };
30        for entry in entries.flatten() {
31            let path = entry.path();
32            if !path.is_dir() {
33                continue;
34            }
35            if path.join(".git").exists() {
36                if let Ok(relative) = path.strip_prefix(cache_root) {
37                    found.insert(relative.to_path_buf());
38                }
39            } else if depth + 1 < MAX_DEPENDENCY_CACHE_DEPTH {
40                go(cache_root, &path, depth + 1, found);
41            }
42        }
43    }
44
45    let mut found = BTreeSet::new();
46    go(cache_root, cache_root, 0, &mut found);
47    found
48}

50/// Creates a unique folder name for a GitHub repo
51/// by using its URL and tag
52fn resolve_folder_name(base: &url::Url, tag: &str) -> String {
53    let mut folder = PathBuf::from("");
54    for part in [base.domain().unwrap(), base.path(), tag] {
55        folder.push(part.trim_start_matches('/'));
56    }
57    folder.to_string_lossy().into_owned()
58}

60/// Path to the `nargo` directory under `$HOME`.
61fn nargo_crates() -> PathBuf {
62    dirs::home_dir().unwrap().join("nargo")
63}
64
65/// Target directory to download dependencies into, e.g.
66/// `$HOME/nargo/github.com/noir-lang/noir-bignum/v0.1.2`
67fn git_dep_location(base: &url::Url, tag: &str) -> PathBuf {
68    let folder_name = resolve_folder_name(base, tag);
69
70    nargo_crates().join(folder_name)
71}

77/// XXX: I'd prefer to use a GitHub library however, there
78/// does not seem to be an easy way to download a repo at a specific
79/// tag
80/// github-rs looks promising, however it seems to require an API token
81///
82/// One advantage of using "git clone" is that there is effectively no rate limit
83pub(crate) fn clone_git_repo(url: &str, tag: &str) -> Result<PathBuf, String> {
84    use std::process::Command;
85
86    let base = match url::Url::parse(url) {
87        Ok(base) => base,
88        Err(err) => return Err(err.to_string()),
89    };
90
91    let loc = git_dep_location(&base, tag);
92    if loc.exists() {
93        return Ok(loc);
94    }
95
96    Command::new("git")
97        .arg("-c")
98        .arg("advice.detachedHead=false")
99        .arg("clone")
100        .arg("--depth")
101        .arg("1")
102        .arg("--branch")
103        .arg(tag)
104        .arg(base.as_str())
105        .arg(&loc)
106        .status()
107        .expect("git clone command failed to start");
108
109    Ok(loc)
110}

131    /// Creates a clone root by making the directory tree `relative` under `cache_root` and
132    /// dropping a `.git` directory inside it, mirroring what `git clone` leaves behind.
133    fn make_clone_root(cache_root: &Path, relative: &str) {
134        let root = cache_root.join(relative);
135        fs::create_dir_all(root.join(".git")).unwrap();
136        // A real clone also has source files alongside `.git`; include one so the walker has to
137        // stop at the clone root rather than descend into its contents.
138        fs::create_dir_all(root.join("src")).unwrap();
139        fs::write(root.join("src/lib.nr"), "").unwrap();
140    }

142    #[test]
143    fn lists_clone_roots_relative_to_cache_and_does_not_descend_into_them() {
144        let cache = tempfile::tempdir().unwrap();
145        let cache_root = cache.path();
146
147        // Two tags of the same repo, plus a repo under a different host.
148        make_clone_root(cache_root, "github.com/noir-lang/keccak256/v0.1.2");
149        make_clone_root(cache_root, "github.com/noir-lang/keccak256/v0.1.3");
150        make_clone_root(cache_root, "example.org/owner/name/v2.0.0");

152        // A lock file at the cache root and an empty intermediate directory must be ignored.
153        fs::write(cache_root.join(".package-cache"), "").unwrap();
154        fs::create_dir_all(cache_root.join("github.com/noir-lang/not-downloaded-yet")).unwrap();
155
156        let found = collect_cached_git_dependencies(cache_root);
157
158        let expected: BTreeSet<PathBuf> = [
