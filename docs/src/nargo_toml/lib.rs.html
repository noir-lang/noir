28/// Searches for a `Nargo.toml` file in the current directory and all parent directories.
29/// For example, if the current directory is `/workspace/package/src`, then this function
30/// will search for a `Nargo.toml` file in
31/// * `/workspace/package/src`,
32/// * `/workspace/package`,
33/// * `/workspace`.
34///
35/// Returns the [PathBuf] of the `Nargo.toml` file if found, otherwise returns None.
36///
37/// It will return innermost `Nargo.toml` file, which is the one closest to the current directory.
38/// For example, if the current directory is `/workspace/package/src`, then this function
39/// will return the `Nargo.toml` file in `/workspace/package/Nargo.toml`
40pub fn find_file_manifest(current_path: &Path) -> Option<PathBuf> {
41    for path in current_path.ancestors() {
42        if let Ok(toml_path) = get_package_manifest(path) {
43            return Some(toml_path);
44        }
45    }
46    None
47}

49/// Returns the [PathBuf] of the directory containing the `Nargo.toml` by searching from `current_path` to the root of its [Path].
50/// When `workspace` is `true` it returns the topmost directory, when `false` the innermost one.
51///
52/// Returns a [ManifestError] if no parent directories of `current_path` contain a manifest file.
53pub fn find_root(current_path: &Path, workspace: bool) -> Result<PathBuf, ManifestError> {
54    if workspace { find_package_root(current_path) } else { find_file_root(current_path) }
55}

57/// Returns the [PathBuf] of the directory containing the `Nargo.toml` by searching from `current_path` to the root of its [Path],
58/// returning at the innermost directory found, i.e. the one corresponding to the package that contains the `current_path`.
59///
60/// Returns a [ManifestError] if no parent directories of `current_path` contain a manifest file.
61pub fn find_file_root(current_path: &Path) -> Result<PathBuf, ManifestError> {
62    match find_file_manifest(current_path) {
63        Some(manifest_path) => {
64            let package_root = manifest_path
65                .parent()
66                .expect("infallible: manifest file path can't be root directory");
67            Ok(package_root.to_path_buf())
68        }
69        None => Err(ManifestError::MissingFile(current_path.to_path_buf())),
70    }
71}

73/// Returns the [PathBuf] of the directory containing the `Nargo.toml` by searching from `current_path` to the root of its [Path],
74/// returning the topmost directory found, i.e. the one corresponding to the entire workspace.
75///
76/// Returns a [ManifestError] if none of the ancestor directories of `current_path` contain a manifest file.
77pub fn find_package_root(current_path: &Path) -> Result<PathBuf, ManifestError> {
78    let root = path_root(current_path);
79    let manifest_path = find_package_manifest(&root, current_path)?;
80
81    let package_root =
82        manifest_path.parent().expect("infallible: manifest file path can't be root directory");
83
84    Ok(package_root.to_path_buf())
85}

87// TODO(#2323): We are probably going to need a "filepath utils" crate soon
88/// Get the root of path, for example:
89/// * `C:\foo\bar` -> `C:\foo`
90/// * `//shared/foo/bar` -> `//shared/foo`
91/// * `/foo` -> `/foo`
92///   otherwise empty path.
93fn path_root(path: &Path) -> PathBuf {
94    let mut components = path.components();
95
96    match (components.next(), components.next()) {
97        // Preserve prefix if one exists
98        (Some(prefix @ Component::Prefix(_)), Some(root @ Component::RootDir)) => {
99            PathBuf::from(prefix.as_os_str()).join(root.as_os_str())
100        }
101        (Some(root @ Component::RootDir), _) => PathBuf::from(root.as_os_str()),
102        _ => PathBuf::new(),
103    }
104}

106/// Returns the [PathBuf] of the `Nargo.toml` file by searching from `current_path` and stopping at `root_path`.
107///
108/// Returns a [ManifestError] if no parent directories of `current_path` contain a manifest file.
109pub fn find_package_manifest(
110    root_path: &Path,
111    current_path: &Path,
112) -> Result<PathBuf, ManifestError> {
113    if current_path.starts_with(root_path) {
114        let mut found_toml_paths = Vec::new();
115        for path in current_path.ancestors() {
116            if let Ok(toml_path) = get_package_manifest(path) {
117                found_toml_paths.push(toml_path);
118            }
119            // While traversing, break once we process the root specified
120            if path == root_path {
121                break;
122            }
123        }
124
125        // Return the shallowest Nargo.toml, which will be the last in the list
126        found_toml_paths.pop().ok_or_else(|| ManifestError::MissingFile(current_path.to_path_buf()))
127    } else {
128        Err(ManifestError::NoCommonAncestor {
129            root: root_path.to_path_buf(),
130            current: current_path.to_path_buf(),
131        })
132    }
133}

135/// Returns the [PathBuf] of the `Nargo.toml` file in the `current_path` directory.
136///
137/// Returns a [ManifestError] if `current_path` does not contain a manifest file.
138pub fn get_package_manifest(current_path: &Path) -> Result<PathBuf, ManifestError> {
139    let toml_path = current_path.join("Nargo.toml");
140    if toml_path.exists() {
141        Ok(toml_path)
142    } else {
143        Err(ManifestError::MissingFile(current_path.to_path_buf()))
144    }
145}

147#[derive(Debug, Deserialize, Clone)]
148pub struct PackageConfig {
149    pub package: PackageMetadata,
150    #[serde(default)]
151    pub dependencies: BTreeMap<String, DependencyConfig>,
152}
153
154impl PackageConfig {
155    fn resolve_to_package(
156        &self,
157        root_dir: &Path,
158        processed: &mut Vec<String>,
159        assume_default_entry: bool, // assume that the 'default_entry_path' exists, e.g. src/main.nr
160    ) -> Result<Package, ManifestError> {
161        let name = &self.package.name;
162        let name: CrateName = name.parse().map_err(|_| ManifestError::InvalidPackageName {
163            toml: root_dir.join("Nargo.toml"),
164            name: name.into(),
165        })?;
166
167        let mut dependencies: BTreeMap<CrateName, Dependency> = BTreeMap::new();
168        for (name, dep_config) in self.dependencies.iter() {
169            let name = name.parse().map_err(|_| ManifestError::InvalidDependencyName {
170                toml: root_dir.join("Nargo.toml"),
171                name: name.into(),
172            })?;
173            let resolved_dep = dep_config.resolve_to_dependency(root_dir, processed)?;
174
175            dependencies.insert(name, resolved_dep);
176        }
177
178        let package_type = match self.package.package_type.as_deref() {
179            Some("lib") => PackageType::Library,
180            Some("bin") => PackageType::Binary,
181            Some("contract") => PackageType::Contract,
182            Some(invalid) => {
183                return Err(ManifestError::InvalidPackageType(
184                    root_dir.join("Nargo.toml"),
185                    invalid.to_string(),
186                ));
187            }
188            None => return Err(ManifestError::MissingPackageType(root_dir.join("Nargo.toml"))),
189        };
190
191        let entry_path = if let Some(entry_path) = &self.package.entry {
192            let custom_entry_path = root_dir.join(entry_path);
193            if custom_entry_path.exists() {
194                custom_entry_path
195            } else {
196                return Err(ManifestError::MissingEntryFile {
197                    toml: root_dir.join("Nargo.toml"),
198                    entry: custom_entry_path,
199                });
200            }
201        } else {
202            let default_entry_path = match package_type {
203                PackageType::Library => {
204                    root_dir.join("src").join("lib").with_extension(FILE_EXTENSION)
205                }
206                PackageType::Binary | PackageType::Contract => {
207                    root_dir.join("src").join("main").with_extension(FILE_EXTENSION)
208                }
209            };
210
211            if default_entry_path.exists() || assume_default_entry {
212                default_entry_path
213            } else {
214                return Err(ManifestError::MissingDefaultEntryFile {
215                    toml: root_dir.join("Nargo.toml"),
216                    entry: default_entry_path,
217                    package_type,
218                });
219            }
220        };
221
222        // If there is a package version, ensure that it is semver compatible
223        if let Some(version) = &self.package.version {
224            semver::parse_semver_compatible_version(version).map_err(|err| {
225                ManifestError::SemverError(SemverError::CouldNotParsePackageVersion {
226                    package_name: name.to_string(),
227                    error: err.to_string(),
228                })
229            })?;
230        }
231
232        // Collect any unstable features the package needs to compile.
233        // Ignore the ones that we don't recognize: maybe they are no longer unstable, but a dependency hasn't been updated.
234        let compiler_required_unstable_features =
235            self.package.compiler_unstable_features.as_ref().map_or(Vec::new(), |feats| {
236                feats.iter().flat_map(|feat| UnstableFeature::from_str(feat).ok()).collect()
237            });
238
239        Ok(Package {
240            version: self.package.version.clone(),
241            compiler_required_version: self.package.compiler_version.clone(),
242            compiler_required_unstable_features,
243            root_dir: root_dir.to_path_buf(),
244            entry_path,
245            package_type,
246            name,
247            dependencies,
248        })
249    }
250}

317/// Enum representing the different types of ways to
318/// supply a source for the dependency
319pub enum DependencyConfig {
320    Git { git: String, tag: String, directory: Option<String> },
321    Path { path: String },
322}
323
324impl DependencyConfig {
325    fn resolve_to_dependency(
326        &self,
327        pkg_root: &Path,
328        processed: &mut Vec<String>,
329    ) -> Result<Dependency, ManifestError> {
330        let dep = match self {
331            Self::Git { git, tag, directory } => {
332                let dir_path = clone_git_repo(git, tag).map_err(ManifestError::GitError)?;
333                let project_path = if let Some(directory) = directory {
334                    let internal_path = dir_path.join(directory).normalize();
335                    if !internal_path.starts_with(&dir_path) {
336                        return Err(ManifestError::InvalidDirectory {
337                            toml: pkg_root.join("Nargo.toml"),
338                            directory: directory.into(),
339                        });
340                    }
341                    internal_path
342                } else {
343                    dir_path
344                };
345                let toml_path = project_path.join("Nargo.toml");
346                let package = resolve_package_from_toml(&toml_path, processed)?;
347                Dependency::Remote { package }
348            }
349            Self::Path { path } => {
350                let dir_path = pkg_root.join(path);
351                let toml_path = dir_path.join("Nargo.toml");
352                let package = resolve_package_from_toml(&toml_path, processed)?;
353                Dependency::Local { package }
354            }
355        };
356
357        // Cannot depend on a binary
358        // TODO: Can we depend upon contracts?
359        if dep.is_binary() {
360            Err(ManifestError::BinaryDependency(dep.package_name().clone()))
361        } else {
362            Ok(dep)
363        }
364    }
365}

367fn toml_to_workspace(
368    nargo_toml: NargoToml,
369    package_selection: PackageSelection,
370    assume_default_entry: bool, // assume that the 'default_entry_path' exists, e.g. src/main.nr
371) -> Result<Workspace, ManifestError> {
372    let mut resolved = Vec::new();
373    let _lock = lock_git_deps().expect("Failed to lock git dependencies cache");
374    let workspace = match nargo_toml.config {
375        Config::Package { package_config } => {
376            let member = package_config.resolve_to_package(
377                &nargo_toml.root_dir,
378                &mut resolved,
379                assume_default_entry,
380            )?;
381            match &package_selection {
382                PackageSelection::Selected(selected_name) if selected_name != &member.name => {
383                    return Err(ManifestError::MissingSelectedPackage(member.name));
384                }
385                _ => Workspace {
386                    root_dir: nargo_toml.root_dir,
387                    selected_package_index: Some(0),
388                    members: vec![member],
389                    is_assumed: false,
390                    target_dir: None,
391                },
392            }
393        }
394        Config::Workspace { workspace_config } => {
395            let mut members = Vec::new();
396            let mut selected_package_index = None;
397            for (index, member_path) in workspace_config.members.into_iter().enumerate() {
398                let package_root_dir = nargo_toml.root_dir.join(&member_path);
399                let package_toml_path = package_root_dir.join("Nargo.toml");
400                let member = resolve_package_from_toml(&package_toml_path, &mut resolved)?;
401
402                match &package_selection {
403                    PackageSelection::Selected(selected_name) => {
404                        if &member.name == selected_name {
405                            selected_package_index = Some(index);
406                        }
407                    }
408                    PackageSelection::DefaultOrAll => {
409                        if Some(&member_path) == workspace_config.default_member.as_ref() {
410                            selected_package_index = Some(index);
411                        }
412                    }
413                    PackageSelection::All => selected_package_index = None,
414                }
415
416                members.push(member);
417            }
418
419            // If the selected_package_index is still `None` but we have see a default_member or selected package,
420            // we want to present an error to users
421            match package_selection {
422                PackageSelection::Selected(selected_name) => {
423                    if selected_package_index.is_none() {
424                        return Err(ManifestError::MissingSelectedPackage(selected_name));
425                    }
426                }
427                PackageSelection::DefaultOrAll => match workspace_config.default_member {
428                    // If `default-member` is specified but we don't have a selected_package_index, we need to fail
429                    Some(default_path) if selected_package_index.is_none() => {
430                        return Err(ManifestError::MissingDefaultPackage(default_path));
431                    }
432                    // However, if there wasn't a `default-member`, we select All, so no error is needed
433                    _ => (),
434                },
435                PackageSelection::All => (),
436            }
437
438            Workspace {
439                root_dir: nargo_toml.root_dir,
440                members,
441                selected_package_index,
442                is_assumed: false,
443                target_dir: None,
444            }
445        }
446    };
447
448    Ok(workspace)
449}

451/// Attempts to read the file at the provided `toml_path` as a `Nargo.toml`
452/// file, returning it if the read was successful.
453///
454/// # Errors
455///
456/// - [`ManifestError::ReadFailed`] if the file could not be read.
457/// - [`ManifestError::MissingParent`] if the Nargo.toml file did not have a
458///   parent directory.
459/// - [`toml::de::Error`] if the Nargo.toml file could not be converted to a
460///   valid [`Config`].
461pub fn read_toml(toml_path: &Path) -> Result<NargoToml, ManifestError> {
462    let toml_path = toml_path.normalize();
463    let toml_as_string = std::fs::read_to_string(&toml_path)
464        .map_err(|_| ManifestError::ReadFailed(toml_path.to_path_buf()))?;
465    let root_dir = toml_path.parent().ok_or(ManifestError::MissingParent)?;
466    let nargo_toml =
467        NargoToml { root_dir: root_dir.to_path_buf(), config: toml_as_string.try_into()? };
468
469    Ok(nargo_toml)
470}

472/// Resolves a Nargo.toml file into a `Package` struct as defined by our `nargo` core.
473fn resolve_package_from_toml(
474    toml_path: &Path,
475    processed: &mut Vec<String>,
476) -> Result<Package, ManifestError> {
477    // Checks for cyclic dependencies
478    let str_path = toml_path.to_str().expect("ICE - path is empty");
479    if processed.contains(&str_path.to_string()) {
480        let mut cycle = false;
481        let mut message = String::new();
482        for toml in processed {
483            cycle = cycle || toml == str_path;
484            if cycle {
485                message += &format!("{toml} referencing ");
486            }
487        }
488        message += str_path;
489        return Err(ManifestError::CyclicDependency { cycle: message });
490    }
491    // Adds the package to the set of resolved packages
492    if let Some(str) = toml_path.to_str() {
493        processed.push(str.to_string());
494    }
495
496    let nargo_toml = read_toml(toml_path)?;
497
498    let result = match nargo_toml.config {
499        Config::Package { package_config } => {
500            let assume_default_entry = false;
501            package_config.resolve_to_package(&nargo_toml.root_dir, processed, assume_default_entry)
502        }
503        Config::Workspace { .. } => {
504            Err(ManifestError::UnexpectedWorkspace(toml_path.to_path_buf()))
505        }
506    };
507    let pos =
508        processed.iter().position(|toml| toml == str_path).expect("added package must be here");
509    processed.remove(pos);
510    result
511}

538/// Resolves a Nargo.toml _ into a `Workspace` struct as defined by our `nargo` core.
539///
540/// As a side effect it downloads project dependencies as well.
541pub fn resolve_workspace_from_fixed_toml(
542    nargo_toml: NargoToml,
543    package_selection: PackageSelection,
544    current_compiler_version: Option<String>,
545    assume_default_entry: bool,
546) -> Result<Workspace, ManifestError> {

644        assert!(matches!(error, ManifestError::MissingFile(_)));

647    /// Test to demonstrate how `find_root` works.
648    #[test]
649    fn test_find_root_example() {
650        const INDENT_SIZE: usize = 4;
651        /// Create directories and files according to a YAML-like layout below
652        fn setup(layout: &str, root: &Path) {
653            fn is_dir(item: &str) -> bool {
654                !item.contains('.')
655            }
656            let mut current_dir = root.to_path_buf();
657            let mut current_indent = 0;
658            let mut last_item: Option<String> = None;
659
660            for line in layout.lines() {
661                if let Some((prefix, item)) = line.split_once('-') {
662                    let item = item.replace(std::path::MAIN_SEPARATOR, "_").trim().to_string();
663
664                    let indent = prefix.len() / INDENT_SIZE;
665
666                    if last_item.is_none() {
667                        current_indent = indent;
668                    }
669
670                    assert!(
671                        indent <= current_indent + 1,
672                        "cannot increase indent by more than {INDENT_SIZE}; item = {item}, current_dir={}",
673                        current_dir.display()
674                    );
675
676                    // Go into the last created directory
677                    if let Some(last_item) = last_item {
678                        if indent > current_indent {
679                            assert!(is_dir(&last_item), "last item was not a dir: {last_item}");
680                            current_dir.push(last_item);
681                            current_indent += 1;
682                        }
683                    }
684                    // Go back into an ancestor directory
685                    while indent < current_indent {
686                        current_dir.pop();
687                        current_indent -= 1;
688                    }
689                    // Create a file or a directory
690                    let item_path = current_dir.join(&item);
691                    if is_dir(&item) {
692                        std::fs::create_dir(&item_path).unwrap_or_else(|e| {
693                            panic!("failed to create dir {}: {e}", item_path.display())
694                        });
695                    } else {
696                        std::fs::write(&item_path, "").expect("failed to create file");
697                    }
698
699                    last_item = Some(item);
700                }
701            }
702        }
703
704        // Temporary directory to hold the project.
705        let tmp = tempfile::tempdir().unwrap();
706        // Join a string path to the tmp dir
707        let path = |p: &str| tmp.path().join(p);
708        // Check that an expected root is found
709        let assert_ok = |current_dir: &str, ws: bool, exp: &str| {
710            let root = find_root(&path(current_dir), ws).expect("should find a root");
711            assert_eq!(root, path(exp));
712        };
713        // Check that a root is not found
714        let assert_err = |current_dir: &str| {
715            find_root(&path(current_dir), true).expect_err("shouldn't find a root");
716        };
717
718        let layout = r"
719            - project
720                - docs
721                - workspace
722                    - packages
723                        - foo
724                            - Nargo.toml
725                            - Prover.toml
726                            - src
727                                - main.nr
728                        - bar
729                            - Nargo.toml
730                            - src
731                                - lib.nr
732                    - Nargo.toml
733                - examples
734                    - baz
735                        - Nargo.toml
736                        - src
737                            - main.nr
738            ";
739
740        // Set up the file system.
741        setup(layout, tmp.path());
742
743        assert_err("dummy");
744        assert_err("project/docs");
745        assert_err("project/examples");
746        assert_ok("project/workspace", true, "project/workspace");
747        assert_ok("project/workspace", false, "project/workspace");
748        assert_ok("project/workspace/packages/foo", true, "project/workspace");
749        assert_ok("project/workspace/packages/bar", false, "project/workspace/packages/bar");
750        assert_ok("project/examples/baz/src", true, "project/examples/baz");
751        assert_ok("project/examples/baz/src", false, "project/examples/baz");
752    }
