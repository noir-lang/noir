1//! Includes custom serde deserializers for Nargo.toml types that could be implemented
2//! using `serde(untagged)` but are not because using that leads to very poor error messages.
3
4use std::collections::BTreeMap;
5
6use serde::{Deserialize, Deserializer, de::Visitor};
7
8use crate::{Config, DependencyConfig, PackageConfig, PackageMetadata, WorkspaceConfig};
9
10impl<'de> Deserialize<'de> for Config {
11    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
12    where
13        D: Deserializer<'de>,
14    {
15        deserializer.deserialize_map(ConfigMapVisitor)
16    }
17}
18
19struct ConfigMapVisitor;
20
21impl<'de> Visitor<'de> for ConfigMapVisitor {
22    type Value = Config;
23
24    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
25        formatter.write_str("a map with either a [package] or a [workspace] section")
26    }
27
28    fn visit_map<A>(self, mut map: A) -> Result<Self::Value, A::Error>
29    where
30        A: serde::de::MapAccess<'de>,
31    {
32        let mut package_metadata: Option<PackageMetadata> = None;
33        let mut workspace_config: Option<WorkspaceConfig> = None;
34        let mut dependencies: BTreeMap<String, DependencyConfig> = BTreeMap::new();
35
36        while let Some(key) = map.next_key::<String>()? {
37            match key.as_str() {
38                "package" => {
39                    package_metadata = Some(map.next_value()?);
40                }
41                "workspace" => {
42                    workspace_config = Some(map.next_value()?);
43                }
44                "dependencies" => {
45                    dependencies = map.next_value()?;
46                }
47                _ => {
48                    // Skip unknown keys
49                    let _: serde::de::IgnoredAny = map.next_value()?;
50                }
51            }
52        }
53
54        match (package_metadata, workspace_config) {
55            (Some(package), None) => {
56                Ok(Config::Package { package_config: PackageConfig { package, dependencies } })
57            }
58            (None, Some(workspace_config)) => Ok(Config::Workspace { workspace_config }),
59            (Some(..), Some(..)) => Err(serde::de::Error::custom(
60                "Nargo.toml cannot have both [package] and [workspace] sections",
61            )),
62            (None, None) => Err(serde::de::Error::custom(
63                "Nargo.toml must have either a [package] or [workspace] section",
64            )),
65        }
66    }
67}

69impl<'de> Deserialize<'de> for DependencyConfig {
70    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
71    where
72        D: Deserializer<'de>,
73    {
74        deserializer.deserialize_map(DependencyConfigMapVisitor)
75    }
76}
77
78struct DependencyConfigMapVisitor;
79
80impl<'de> Visitor<'de> for DependencyConfigMapVisitor {
81    type Value = DependencyConfig;
82
83    fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
84        formatter.write_str(
85            "a dependency to be either `{ git = \"...\", tag = \"...\" }` or `{ path = \"...\" }`",
86        )
87    }
88
89    fn visit_map<A>(self, mut map: A) -> Result<Self::Value, A::Error>
90    where
91        A: serde::de::MapAccess<'de>,
92    {
93        let mut git: Option<String> = None;
94        let mut tag: Option<String> = None;
95        let mut directory: Option<String> = None;
96        let mut path: Option<String> = None;
97
98        while let Some(key) = map.next_key::<String>()? {
99            match key.as_str() {
100                "git" => {
101                    git = Some(map.next_value()?);
102                }
103                "tag" => {
104                    tag = Some(map.next_value()?);
105                }
106                "directory" => {
107                    directory = Some(map.next_value()?);
108                }
109                "path" => {
110                    path = Some(map.next_value()?);
111                }
112                _ => {
113                    // Skip unknown keys
114                    let _: serde::de::IgnoredAny = map.next_value()?;
115                }
116            }
117        }
118
119        match (git, path) {
120            (Some(git), None) => {
121                let Some(tag) = tag else {
122                    return Err(serde::de::Error::custom("Git dependencies must have a `tag` key"));
123                };
124                Ok(DependencyConfig::Git { git, tag, directory })
125            }
126            (None, Some(path)) => Ok(DependencyConfig::Path { path }),
127            (Some(..), Some(..)) => Err(serde::de::Error::custom(
128                "Dependency must have either a `git` or a `path` key, not both",
129            )),
130            (None, None) => Err(serde::de::Error::custom(
131                "Dependency must have either a `git` or a `path` key, but neither was found",
132            )),
133        }
134    }
135}
