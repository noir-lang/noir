24/// A set of [ValueId]s referring to SSA variables (not functions).
25type Variables = HashSet<ValueId>;
26/// The set variables which are dead after a given instruction (in a given block).
27type LastUses = HashMap<InstructionId, Variables>;
28/// Maps a loop (identified by its header and back-edge) to the blocks in the loop body.
29type LoopMap = HashMap<BackEdge, BTreeSet<BasicBlockId>>;
30
31/// A back edge is an edge from a node to one of its ancestors. It denotes a loop in the CFG.
32#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
33struct BackEdge {
34    /// The header of the loop.
35    header: BasicBlockId,
36    /// The back-edge of the loop.
37    start: BasicBlockId,
38}
39
40/// Check if the [Value] behind the [ValueId] requires register allocation (like a function parameter),
41/// rather than a global value like a user-defined function, intrinsic, or foreign function.
42pub(super) fn is_variable(value_id: ValueId, dfg: &DataFlowGraph) -> bool {
43    let value = &dfg[value_id];
44
45    match value {
46        Value::Instruction { .. }
47        | Value::Param { .. }
48        | Value::NumericConstant { .. }
49        | Value::Global(_) => true,
50        // Functions are not variables in a defunctionalized SSA. Only constant function values should appear.
51        Value::ForeignFunction(_) | Value::Function(_) | Value::Intrinsic(..) => false,
52    }
53}
54
55/// Collect all [ValueId]s used in an [Instruction] which refer to variables (not functions).
56pub(super) fn variables_used_in_instruction(
57    instruction: &Instruction,
58    dfg: &DataFlowGraph,
59) -> Variables {
60    let mut used = HashSet::default();
61
62    instruction.for_each_value(|value_id| {
63        if is_variable(value_id, dfg) {
64            used.insert(value_id);
65        }
66    });
67
68    used
69}
70
71/// Collect all [ValueId]s used in an [BasicBlock] which refer to [Variables].
72///
73/// Includes all the variables in the parameters, instructions and the terminator.
74fn variables_used_in_block(block: &BasicBlock, dfg: &DataFlowGraph) -> Variables {
75    let mut used: Variables = block
76        .instructions()
77        .iter()
78        .flat_map(|instruction_id| {
79            let instruction = &dfg[*instruction_id];
80            variables_used_in_instruction(instruction, dfg)
81        })
82        .collect();
83
84    // We consider block parameters used, so they live up to the block that owns them.
85    used.extend(block.parameters().iter());
86
87    if let Some(terminator) = block.terminator() {
88        terminator.for_each_value(|value_id| {
89            if is_variable(value_id, dfg) {
90                used.insert(value_id);
91            }
92        });
93    }
94
95    used
96}

98/// A struct representing the liveness of variables throughout a function.
99#[derive(Default)]
100pub(crate) struct VariableLiveness {
101    cfg: ControlFlowGraph,
102    /// The variables that are alive before the block starts executing.
103    live_in: HashMap<BasicBlockId, Variables>,
104    /// The variables that stop being alive after each specific instruction.
105    last_uses: HashMap<BasicBlockId, LastUses>,
106    /// The list of block params the given block is defining.
107    /// The order matters for the entry block, so it's a vec.
108    ///
109    /// By _defining_  we mean the allocation of variables that appear in the parameter
110    /// list of some other block which this one immediately dominates, with values to be
111    /// assigned in the terminators of the predecessors of that block.
112    param_definitions: HashMap<BasicBlockId, Vec<ValueId>>,
113}
114
115impl VariableLiveness {
116    /// Computes the liveness of variables throughout a function.
117    pub(crate) fn from_function(func: &Function, constants: &ConstantAllocation) -> Self {
118        let loops = Loops::find_all(func, LoopOrder::OutsideIn);
119
120        let back_edges: LoopMap = loops
121            .yet_to_unroll
122            .into_iter()
123            .map(|_loop| {
124                let back_edge = BackEdge { header: _loop.header, start: _loop.back_edge_start };
125                let loop_body = _loop.blocks;
126                (back_edge, loop_body)
127            })
128            .collect();
129
130        Self {
131            cfg: loops.cfg,
132            live_in: HashMap::default(),
133            last_uses: HashMap::default(),
134            param_definitions: HashMap::default(),
135        }
136        .compute_block_param_definitions(func, &loops.dom)
137        .compute_live_in_of_blocks(func, constants, back_edges)
138        .compute_last_uses(func)
139    }
140
141    /// The set of values that are alive before the block starts executing.
142    pub(crate) fn get_live_in(&self, block_id: &BasicBlockId) -> &Variables {
143        self.live_in.get(block_id).expect("live_in should have been calculated")
144    }
145
146    /// The set of values that are alive after the block has finished executed.
147    ///
148    /// By definition this is the union of variables alive in the successors of the block.
149    pub(crate) fn get_live_out(&self, block_id: &BasicBlockId) -> Variables {
150        let mut live_out = HashSet::default();
151        for successor_id in self.cfg.successors(*block_id) {
152            live_out.extend(self.get_live_in(&successor_id).iter().copied());
153        }
154        live_out
155    }
156
157    /// For a given block, get the map of instruction ID to the set of values
158    /// that die after the instruction has executed.
159    pub(crate) fn get_last_uses(&self, block_id: &BasicBlockId) -> &LastUses {
160        self.last_uses.get(block_id).expect("last_uses should have been calculated")
161    }
162
163    /// Retrieves the list of block parameters the given block is defining.
164    ///
165    /// Block parameters are defined before the block that owns them
166    /// (since they are used by the predecessor blocks to pass values).
167    /// They must be defined in the immediate dominator, which is the
168    /// last point where the block parameter can be allocated
169    /// without it being allocated in different places in different branches.
170    pub(crate) fn defined_block_params(&self, block_id: &BasicBlockId) -> Vec<ValueId> {
171        self.param_definitions.get(block_id).cloned().unwrap_or_default()
172    }
173
174    /// Compute [VariableLiveness::param_definitions].
175    ///
176    /// Append the parameters of each block to the parameter definition list of
177    /// its immediate dominator.
178    fn compute_block_param_definitions(mut self, func: &Function, dom: &DominatorTree) -> Self {
179        assert!(self.param_definitions.is_empty(), "only define parameters once");
180
181        // Going in reverse post order to process the entry block first.
182        let reverse_post_order = PostOrder::with_function(func).into_vec_reverse();
183        for block in reverse_post_order {
184            let params = func.dfg[block].parameters();
185            // If it has no dominator, it's the entry block
186            let dominator_block = dom.immediate_dominator(block).unwrap_or(func.entry_block());
187            let definitions_for_the_dominator =
188                self.param_definitions.entry(dominator_block).or_default();
189            definitions_for_the_dominator.extend(params.iter());
190        }
191
192        self
193    }
194
195    /// Compute [VariableLiveness::live_in].
196    ///
197    /// Collect the variables which are alive before each block.
198    fn compute_live_in_of_blocks(
199        mut self,
200        func: &Function,
201        constants: &ConstantAllocation,
202        back_edges: LoopMap,
203    ) -> Self {
204        // First pass, propagate up the live_ins skipping back edges.
205        self.compute_live_in(func, func.entry_block(), constants, &back_edges);
206
207        // Second pass, propagate header live_ins to the loop bodies.
208        for (back_edge, loop_body) in back_edges {
209            self.update_live_ins_within_loop(back_edge, loop_body);
210        }
211
212        self
213    }
214
215    /// Starting with the entry block, traverse down all successors to compute their `live_in`,
216    /// then propagate the information back up towards the ancestors as `live_out`.
217    ///
218    /// The variables live at the *beginning* of a block are the variables used by the block,
219    /// plus the variables used by the successors of the block, minus the variables defined
220    /// in the block (by definition not alive at the beginning).
221    ///
222    /// This is an iterative implementation to avoid stack overflows on complex programs.
223    fn compute_live_in(
224        &mut self,
225        func: &Function,
226        entry_block: BasicBlockId,
227        constants: &ConstantAllocation,
228        back_edges: &LoopMap,
229    ) {
230        // Each entry is (block_id, processing_state)
231        // processing_state: false = need to process successors, true = ready to compute live_in
232        let mut stack = vec![(entry_block, false)];
233        let mut visited = HashSet::default();
234
235        while let Some((block_id, processed)) = stack.pop() {
236            if processed {
237                // All successors have been processed, now compute live_in for this block
238                let block = &func.dfg[block_id];
239                let mut live_out = HashSet::default();
240
241                // Collect the `live_in` of successors; their union is the `live_out` of the parent.
242                for successor_id in block.successors() {
243                    // Skip back edges: do not revisit the header of the loop
244                    if back_edges.contains_key(&BackEdge { start: block_id, header: successor_id })
245                    {
246                        continue;
247                    }
248                    // Add the live_in of the successor to the union that forms the live_out of the parent.
249                    live_out.extend(
250                        self.live_in
251                            .get(&successor_id)
252                            .expect("live_in for successor should have been calculated")
253                            .iter()
254                            .copied(),
255                    );
256                }
257
258                // Based on the paper mentioned in the module docs, the definition would be:
259                // live_in[BlockId] = before_def[BlockId] union (live_out[BlockId] - killed[BlockId])
260
261                // Variables used in this block, defined in this block or before.
262                let used = variables_used_in_block(block, &func.dfg);
263
264                // Variables defined in this block are not alive at the beginning.
265                let defined = self.variables_defined_in_block(block_id, &func.dfg, constants);
266
267                // Live at the beginning are the variables used, but not defined in this block, plus the ones
268                // it passes through to its successors, which are used by them, but not defined here.
269                // (Variables used by successors and defined in this block are part of `live_out`, but not `live_in`).
270                let live_in =
271                    used.union(&live_out).filter(|v| !defined.contains(v)).copied().collect();
272
273                self.live_in.insert(block_id, live_in);
274            } else {
275                // First visit: check if we've already processed this block
276                if !visited.insert(block_id) {
277                    continue;
278                }
279
280                let block = &func.dfg[block_id];
281
282                // Check if all successors (except back edges) have been processed
283                let mut all_successors_processed = true;
284                let mut unprocessed_successors = Vec::new();
285
286                for successor_id in block.successors() {
287                    // Skip back edges
288                    if back_edges.contains_key(&BackEdge { start: block_id, header: successor_id })
289                    {
290                        continue;
291                    }
292                    // If successor hasn't been processed yet, we need to process it first
293                    if !self.live_in.contains_key(&successor_id) {
294                        all_successors_processed = false;
295                        unprocessed_successors.push(successor_id);
296                    }
297                }
298
299                // Push this block back with processed = true (for after successors)
300                stack.push((block_id, true));
301                if !all_successors_processed {
302                    // Push unprocessed successors with processed = false
303                    for successor_id in unprocessed_successors {
304                        stack.push((successor_id, false));
305                    }
306                }
307            }
308        }
309    }

311    /// Collects all the variables defined in a block, which includes:
312    /// * parameters of descendants this block immediately dominates
313    /// * the results of the instructions in the block
314    /// * constants which were allocated to this block
315    fn variables_defined_in_block(
316        &self,
317        block_id: BasicBlockId,
318        dfg: &DataFlowGraph,
319        constants: &ConstantAllocation,
320    ) -> Variables {
321        let block = &dfg[block_id];
322        let mut defined_vars = HashSet::default();
323
324        defined_vars.extend(self.defined_block_params(&block_id));
325
326        for instruction_id in block.instructions() {
327            let result_values = dfg.instruction_results(*instruction_id);
328            defined_vars.extend(result_values.iter().copied());
329        }
330
331        defined_vars.extend(constants.allocated_in_block(block_id));
332
333        defined_vars
334    }
335
336    /// Once we know which variables are alive before the loop header,
337    /// we can append those variables to all of the loop's blocks.
338    /// Since we know that we have to come back
339    /// to the beginning of the loop, none of those blocks are allowed
340    /// anything but to keep these variables alive, so that the header
341    /// can use them again.
342    fn update_live_ins_within_loop(
343        &mut self,
344        back_edge: BackEdge,
345        loop_body: BTreeSet<BasicBlockId>,
346    ) {
347        let header_live_in = self.get_live_in(&back_edge.header).clone();
348
349        for body_block_id in loop_body {
350            self.live_in
351                .get_mut(&body_block_id)
352                .expect("Live ins should have been calculated")
353                .extend(header_live_in.iter().copied());
354        }
355    }
356
357    /// Compute [VariableLiveness::last_uses].
358    ///
359    /// For each block, starting from the terminator than going backwards through the instructions,
360    /// take note of the first (technically last) instruction the value is used in.
361    fn compute_last_uses(mut self, func: &Function) -> Self {
362        for block_id in func.reachable_blocks() {
363            let block = &func.dfg[block_id];
364            let live_out = self.get_live_out(&block_id);
365
366            // Variables we have already visited, ie. they are used in "later" instructions or the terminator.
367            let mut used_after: Variables = Default::default();
368            // Variables becoming dead after each instruction.
369            let mut block_last_uses: LastUses = Default::default();
370
371            // First, handle the terminator; none of the instructions should cause these to go dead.
372            if let Some(terminator_instruction) = block.terminator() {
373                terminator_instruction.for_each_value(|value_id| {
374                    if is_variable(value_id, &func.dfg) {
375                        used_after.insert(value_id);
376                    }
377                });
378            }
379
380            // Then, handle the instructions in reverse order to find the last use.
381            for instruction_id in block.instructions().iter().rev() {
382                let instruction = &func.dfg[*instruction_id];
383                // Collect the variables which will be dead after this instruction.
384                let instruction_last_uses = variables_used_in_instruction(instruction, &func.dfg)
385                    .into_iter()
386                    .filter(|id| !used_after.contains(id) && !live_out.contains(id))
387                    .collect();
388                // Remember that we have already handled these.
389                used_after.extend(&instruction_last_uses);
390                // Remember that we can deallocate these after this instruction.
391                block_last_uses.insert(*instruction_id, instruction_last_uses);
392            }
393
394            self.last_uses.insert(block_id, block_last_uses);
395        }
396
397        self
398    }
