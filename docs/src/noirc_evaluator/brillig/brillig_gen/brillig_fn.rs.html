21/// Information required to compile an SSA [Function] into Brillig bytecode.
22///
23/// This structure is instantiated once per function and used throughout basic block code generation.
24/// It can also represent a non-function context (e.g., global instantiation) to reuse block codegen logic
25/// by leaving its `function_id` field unset.
26#[derive(Default)]
27pub(crate) struct FunctionContext {
28    /// A `FunctionContext` is necessary for using a Brillig block's code gen, but sometimes
29    /// such as with globals, we are not within a function and do not have a [FunctionId].
30    function_id: Option<FunctionId>,
31    /// Map from SSA values its allocation. Since values can be only defined once in SSA form,
32    /// we insert them here on when we allocate them at their definition.
33    ///
34    /// Multiple variables could be assigned the same slot, because this structure accumulates
35    /// historical allocations, not just the currently active ones. This is needed so that
36    /// when we start processing a block, we can always look up the allocation of the variables
37    /// which are live at the beginning of it, even if they were deemed dead by another block
38    /// we already visited.
39    ///
40    /// Note that we don't use `Allocated<BrilligVariable>` here, because we create a fresh
41    /// allocator for each block we process, and something that is allocated in e.g. block 1
42    /// might be deallocated in block 2, so it has to be done manually.
43    pub(crate) ssa_value_allocations: HashMap<ValueId, BrilligVariable>,
44    /// The block ids of the function in Reverse Post Order.
45    blocks: Vec<BasicBlockId>,
46    /// Liveness information for each variable in the function.
47    pub(crate) liveness: VariableLiveness,
48    /// Information on where to allocate constants
49    pub(crate) constant_allocation: ConstantAllocation,
50    /// True if this function is a brillig entry point
51    pub(crate) is_entry_point: bool,
52}
53
54impl FunctionContext {
55    /// Creates a new function context. It will allocate parameters for all blocks and compute the liveness of every variable.
56    pub(crate) fn new(function: &Function, is_entry_point: bool) -> Self {
57        let id = function.id();
58
59        let reverse_post_order = PostOrder::with_function(function).into_vec_reverse();
60        let constants = ConstantAllocation::from_function(function);
61        let liveness = VariableLiveness::from_function(function, &constants);
62
63        Self {
64            function_id: Some(id),
65            ssa_value_allocations: HashMap::default(),
66            blocks: reverse_post_order,
67            liveness,
68            is_entry_point,
69            constant_allocation: constants,
70        }
71    }
72
73    /// Get the ID of the function this context was created for.
74    ///
75    /// Panics if we call it when in the context created to hold
76    /// data structures for global codegen only.
77    pub(crate) fn function_id(&self) -> FunctionId {
78        self.function_id.expect("ICE: function_id should already be set")
79    }
80
81    /// Collects the return values of a given function
82    pub(crate) fn return_values(func: &Function) -> Vec<BrilligParameter> {
83        func.returns()
84            .unwrap_or_default()
85            .iter()
86            .map(|&value_id| {
87                let typ = func.dfg.type_of_value(value_id);
88                Self::ssa_type_to_parameter(&typ)
89            })
90            .collect()
91    }
92
93    /// Converts an SSA [Type] into a corresponding [BrilligParameter].
94    ///
95    /// This conversion defines the calling convention for Brillig functions,
96    /// ensuring that SSA values are correctly mapped to memory layouts understood by the VM.
97    ///
98    /// # Panics
99    /// Panics if called with a vector type, as a vector's memory layout cannot be inferred without runtime data.
100    pub(crate) fn ssa_type_to_parameter(typ: &Type) -> BrilligParameter {
101        match typ {
102            Type::Numeric(_) | Type::Reference(_) => {
103                BrilligParameter::SingleAddr(get_bit_size_from_ssa_type(typ))
104            }
105            Type::Array(item_type, size) => BrilligParameter::Array(
106                vecmap(item_type.iter(), Self::ssa_type_to_parameter),
107                *size,
108            ),
109            Type::Vector(_) => {
110                panic!("ICE: Vector parameters cannot be derived from type information")
111            }
112            // Treat functions as field values
113            Type::Function => {
114                BrilligParameter::SingleAddr(get_bit_size_from_ssa_type(&Type::field()))
115            }
116        }
117    }
118
119    /// Iterate blocks in Post Order.
120    pub(crate) fn post_order(&self) -> impl ExactSizeIterator<Item = BasicBlockId> {
121        self.blocks.iter().copied().rev()
122    }
123
124    /// Iterate blocks in Reverse Post Order.
125    pub(crate) fn reverse_post_order(&self) -> impl ExactSizeIterator<Item = BasicBlockId> {
126        self.blocks.iter().copied()
127    }
128}
