29/// Context structure for compiling a [function block][crate::ssa::ir::basic_block::BasicBlock] into Brillig bytecode.
30pub(crate) struct BrilligBlock<'block, Registers: RegisterAllocator> {
31    /// Per-function context shared across all of a function's blocks
32    pub(crate) function_context: &'block mut FunctionContext,
33    /// The basic block that is being converted
34    pub(crate) block_id: BasicBlockId,
35    /// Context for creating brillig opcodes
36    pub(crate) brillig_context: &'block mut BrilligContext<FieldElement, Registers>,
37    /// Tracks the available variable during the codegen of the block
38    pub(crate) variables: BlockVariables,
39    /// For each instruction, the set of values that are not used anymore after it.
40    pub(crate) last_uses: HashMap<InstructionId, HashSet<ValueId>>,
41
42    /// Mapping of SSA [ValueId]s to their already instantiated values in the Brillig IR.
43    pub(crate) globals: &'block HashMap<ValueId, BrilligVariable>,
44    /// Pre-instantiated constants values shared across functions which have hoisted to the global memory space.
45    pub(crate) hoisted_global_constants: &'block HoistedConstantsToBrilligGlobals,
46    /// Status variable for whether we are generating Brillig bytecode for a function or globals.
47    /// This is primarily used for gating local variable specific logic.
48    /// For example, liveness analysis for globals is unnecessary (and adds complexity),
49    /// and instead globals live throughout the entirety of the program.
50    pub(crate) building_globals: bool,
51}

53impl<'block, Registers: RegisterAllocator> BrilligBlock<'block, Registers> {
54    /// Converts an SSA basic block into a sequence of Brillig opcodes.
55    ///
56    /// This method contains the necessary initial variable and register setup for compiling
57    /// an SSA block by accessing the pre-computed liveness context.
58    pub(crate) fn compile_block(
59        function_context: &'block mut FunctionContext,
60        brillig_context: &'block mut BrilligContext<FieldElement, Registers>,
61        block_id: BasicBlockId,
62        dfg: &DataFlowGraph,
63        call_stacks: &mut CallStackHelper,
64        globals: &'block HashMap<ValueId, BrilligVariable>,
65        hoisted_global_constants: &'block HoistedConstantsToBrilligGlobals,
66    ) {
67        let live_in = function_context.liveness.get_live_in(&block_id);
68
69        let mut live_in_no_globals = HashSet::default();
70        for value in live_in {
71            if let Value::NumericConstant { constant, typ } = dfg[*value]
72                && hoisted_global_constants.contains_key(&(constant, typ))
73            {
74                continue;
75            }
76            if !dfg.is_global(*value) {
77                live_in_no_globals.insert(*value);
78            }
79        }
80
81        let variables = BlockVariables::new(live_in_no_globals);
82
83        // Replace the previous registers with a new instance, where the currently live variables are pre-allocated.
84        // These might be deallocated and reused if their last use in this block indicates they are dead,
85        // but then become pre-allocated in a new block again, depending on the order of processing.
86        brillig_context.set_allocated_registers(
87            variables
88                .get_available_variable_allocations(function_context)
89                .into_iter()
90                .map(|variable| variable.extract_register())
91                .collect(),
92        );
93        let last_uses = function_context.liveness.get_last_uses(&block_id).clone();
94
95        let mut brillig_block = BrilligBlock {
96            function_context,
97            block_id,
98            brillig_context,
99            variables,
100            last_uses,
101            globals,
102            hoisted_global_constants,
103            building_globals: false,
104        };
105
106        brillig_block.convert_block(dfg, call_stacks);
107    }

109    /// Converts SSA globals into Brillig global values.
110    ///
111    /// Global values can be:
112    /// - Numeric constants
113    /// - Instructions that compute global values
114    /// - Pre-hoisted constants (shared across functions and stored in global memory)
115    ///
116    /// This method expects SSA globals to already be converted to a [DataFlowGraph]
117    /// as to share codegen logic with standard SSA function blocks.
118    ///
119    /// This method also emits any necessary debugging initialization logic (e.g., allocating a counter used
120    /// to track array copies).
121    ///
122    /// # Returns
123    /// A map of hoisted (constant, type) pairs to their allocated Brillig variables,
124    /// which are used to resolve references to these constants throughout Brillig lowering.
125    ///
126    /// # Panics
127    /// - Globals graph contains values other than a [constant][Value::NumericConstant] or [instruction][Value::Instruction]
128    pub(crate) fn compile_globals(
129        &mut self,
130        globals: &DataFlowGraph,
131        used_globals: &HashSet<ValueId>,
132        call_stacks: &mut CallStackHelper,
133        hoisted_global_constants: &BTreeSet<(FieldElement, NumericType)>,
134    ) -> HashMap<(FieldElement, NumericType), Allocated<BrilligVariable, Registers>> {
135        // Using the end of the global memory space adds more complexity as we
136        // have to account for possible register de-allocations as part of regular global compilation.
137        // Thus, we want to allocate any reserved global slots first.
138
139        // If we want to print the array copy count in the end, we reserve the 0 slot.
140        if self.brillig_context.count_array_copies() {
141            // Detach from the register so it's never deallocated.
142            let new_variable =
143                allocate_value_with_type(self.brillig_context, Type::unsigned(32)).detach();
144            self.brillig_context
145                .const_instruction(new_variable.extract_single_addr(), FieldElement::zero());
146        }
147
148        for (id, value) in globals.values_iter() {
149            if !used_globals.contains(&id) {
150                continue;
151            }
152            match value {
153                Value::NumericConstant { .. } => {
154                    self.convert_ssa_value(id, globals);
155                }
156                Value::Instruction { instruction, .. } => {
157                    self.convert_ssa_instruction(*instruction, globals, call_stacks);
158                }
159                _ => {
160                    panic!(
161                        "Expected either an instruction or a numeric constant for a global value"
162                    )
163                }
164            }
165        }
166
167        // Allocate and initialize hoisted constants. These don't have a variable ID associated with them,
168        // so we return them explicitly, while the allocated global variables are in the `ssa_value_allocations`
169        // field of the `FunctionContext`.
170        let mut new_hoisted_constants = HashMap::default();
171        for (constant, typ) in hoisted_global_constants.iter().copied() {
172            let new_variable = allocate_value_with_type(self.brillig_context, Type::Numeric(typ));
173            self.brillig_context.const_instruction(new_variable.extract_single_addr(), constant);
174            if new_hoisted_constants.insert((constant, typ), new_variable).is_some() {
175                unreachable!("ICE: ({constant:?}, {typ:?}) was already in cache");
176            }
177        }
178
179        new_hoisted_constants
180    }

182    /// Internal method for [BrilligBlock::compile_block] that actually kicks off the Brillig compilation process.
183    ///
184    /// At this point any Brillig context should be contained in [BrilligBlock], and this function should
185    /// only need to accept external SSA and debugging structures.
186    fn convert_block(&mut self, dfg: &DataFlowGraph, call_stacks: &mut CallStackHelper) {
187        // Add a label for this block
188        let block_label = self.create_block_label_for_current_function(self.block_id);
189        self.brillig_context.enter_context(block_label);
190
191        // Allocate variables for parameter passing between blocks.
192        self.convert_block_params(dfg);
193
194        let block = &dfg[self.block_id];
195
196        // Convert all of the instructions into the block.
197        for instruction_id in block.instructions() {
198            self.convert_ssa_instruction(*instruction_id, dfg, call_stacks);
199        }
200
201        // Process the block's terminator instruction.
202        let terminator_instruction =
203            block.terminator().expect("block is expected to be constructed");
204
205        // If we are exiting the entry point, we may want to print the array copy count, for debug purposes.
206        if self.brillig_context.count_array_copies()
207            && matches!(terminator_instruction, TerminatorInstruction::Return { .. })
208            && self.function_context.is_entry_point
209        {
210            self.brillig_context.emit_println_of_array_copy_counter();
211        }
212
213        self.convert_ssa_terminator(terminator_instruction, dfg);
214    }

216    /// Creates a unique global label for a block.
217    ///
218    /// This uses the current function's function ID and the block ID,
219    /// making the assumption that the block ID passed in belongs to this function.
220    fn create_block_label_for_current_function(&self, block_id: BasicBlockId) -> Label {
221        Self::create_block_label(self.function_context.function_id(), block_id)
222    }
223    /// Creates a unique label for a block using the function Id and the block ID.
224    ///
225    /// We implicitly assume that the function ID and the block ID is enough
226    /// for us to create a unique label across functions and blocks.
227    ///
228    /// This is so that during linking there are no duplicates or labels being overwritten.
229    fn create_block_label(function_id: FunctionId, block_id: BasicBlockId) -> Label {
230        Label::block(function_id, block_id)
231    }

233    /// Converts an SSA terminator instruction into the necessary opcodes:
234    /// * allocates the hoisted constants which are used by dominated blocks
235    /// * for jumps:
236    ///   * copies the arguments to the registers allocated in [Self::convert_block_params]
237    ///   * adds jump opcodes to the labels of the destination blocks
238    /// * for return it allocates registers for the return values and copies from variables.
239    fn convert_ssa_terminator(
240        &mut self,
241        terminator_instruction: &TerminatorInstruction,
242        dfg: &DataFlowGraph,
243    ) {
244        self.initialize_constants(dfg, InstructionLocation::Terminator);
245
246        match terminator_instruction {
247            TerminatorInstruction::JmpIf {
248                condition,
249                then_destination,
250                else_destination,
251                call_stack: _,
252            } => {
253                let condition = self.convert_ssa_single_addr_value(*condition, dfg);
254                self.brillig_context.jump_if_instruction(
255                    condition.address,
256                    self.create_block_label_for_current_function(*then_destination),
257                );
258                self.brillig_context.jump_instruction(
259                    self.create_block_label_for_current_function(*else_destination),
260                );
261            }
262            TerminatorInstruction::Jmp { destination, arguments, call_stack: _ } => {
263                let destination_block = &dfg[*destination];
264                let mut moves: Vec<(MemoryAddress, MemoryAddress)> = Vec::new();
265                for (arg, param) in arguments.iter().zip(destination_block.parameters()) {
266                    let param_reg = self
267                        .variables
268                        .get_allocation(self.function_context, *param)
269                        .extract_register();
270                    let arg_reg = self.convert_ssa_value(*arg, dfg).extract_register();
271                    moves.push((arg_reg, param_reg));
272                }
273
274                // Block parameter assignments at a jmp must happen "simultaneously".
275                // A naive sequential loop can lose values when a source register
276                // is overwritten by an earlier move in the same batch. For example, with:
277                //   `jmp b1(v1, v2, u32 10)` where b1(v2, v3, v4):
278                // Sequential execution would:
279                //      1. mov reg(v2), reg(v1) — overwrites old v3
280                //      2. mov reg(v3), reg(v2) — reads the NEW v2 instead of old
281                // To prevent this, we save any source that would be overwritten into a
282                // temporary first.
283                // Filter out self-moves (e.g. from coalesced args that already share the param register).
284                moves.retain(|(src, dst)| src != dst);
285
286                let dest_set: HashSet<MemoryAddress> = moves.iter().map(|(_, d)| *d).collect();
287                // `Allocated` automatically deallocates the register when dropped,
288                // so we collect the temporaries here to keep them alive until all
289                // moves have been emitted.
290                let mut temps = Vec::new();
291                for (src, _dst) in &mut moves {
292                    if dest_set.contains(src) {
293                        let temp = self.brillig_context.allocate_register();
294                        self.brillig_context.mov_instruction(*temp, *src);
295                        *src = *temp;
296                        temps.push(temp);
297                    }
298                }
299
300                for (src, dst) in &moves {
301                    if src != dst {
302                        self.brillig_context.mov_instruction(*dst, *src);
303                    }
304                }
305
306                self.brillig_context
307                    .jump_instruction(self.create_block_label_for_current_function(*destination));
308            }
309            TerminatorInstruction::Return { return_values, .. } => {
310                let return_registers = vecmap(return_values, |value_id| {
311                    // Get the allocations of the values to be returned.
312                    self.convert_ssa_value(*value_id, dfg)
313                });
314                self.brillig_context.codegen_return(&return_registers);
315            }
316            TerminatorInstruction::Unreachable { .. } => {
317                // If we assume this is unreachable code then there's nothing to do here
318            }
319        }
320    }

322    /// Allocates the block parameters that the given block is defining.
323    ///
324    /// We don't allocate the block parameters of the block itself here, we allocate the parameters the block is defining
325    /// for the descendant blocks it immediately dominates. Since predecessors to a block have to know where the parameters
326    /// of the block are allocated to pass data to it in [Self::convert_ssa_terminator], the block parameters need to be
327    /// defined/allocated before the given block. [VariableLiveness](crate::brillig::brillig_gen::variable_liveness::VariableLiveness)
328    /// decides when the block parameters are defined.
329    ///
330    /// For the entry block, the defined block params will be the params of the function + any extra params of blocks it's the immediate dominator of.
331    fn convert_block_params(&mut self, dfg: &DataFlowGraph) {
332        for param_id in self.function_context.liveness.defined_block_params(&self.block_id) {
333            let value = &dfg[param_id];
334            let Value::Param { typ: param_type, .. } = value else {
335                unreachable!("ICE: Only Param type values should appear in block parameters");
336            };
337            match param_type {
338                Type::Numeric(_) | Type::Array(..) | Type::Vector(..) | Type::Reference(_) => {
339                    // Simple parameters and arrays are passed as already filled registers.
340                    // In the case of arrays, the values should already be in memory and the register should be a valid pointer to the array.
341                    // For vectors, two registers are passed, the pointer to the data and a register holding the size of the vector.
342                    self.variables.define_variable(
343                        self.function_context,
344                        self.brillig_context,
345                        param_id,
346                        dfg,
347                    );
348                }
349                Type::Function => unreachable!(
350                    "ICE: Type::Function Param not supported; should have been removed by defunctionalization."
351                ),
352            }
353        }
354    }

356    /// Converts an SSA instruction into a sequence of Brillig opcodes.
357    ///
358    /// If this is the last time a variable is used in this block, its memory slot gets deallocated, unless it's a global.
359    fn convert_ssa_instruction(
360        &mut self,
361        instruction_id: InstructionId,
362        dfg: &DataFlowGraph,
363        call_stacks: &mut CallStackHelper,
364    ) {
365        let instruction = &dfg[instruction_id];
366        let call_stack = dfg.get_instruction_call_stack(instruction_id);
367        let call_stack_new_id = call_stacks.get_or_insert_locations(&call_stack);
368        self.brillig_context.set_call_stack(call_stack_new_id);
369
370        self.initialize_constants(dfg, InstructionLocation::Instruction(instruction_id));
371
372        match instruction {
373            Instruction::Binary(binary) => {
374                self.codegen_binary(instruction_id, binary, dfg);
375            }
376            Instruction::Constrain(lhs, rhs, assert_message) => {
377                self.codegen_constrain(*lhs, *rhs, assert_message, dfg);
378            }
379            Instruction::ConstrainNotEqual(..) => {
380                unreachable!("only implemented in ACIR")
381            }
382            Instruction::Allocate => {
383                self.codegen_allocate(instruction_id, dfg);
384            }
385            Instruction::Store { address, value } => {
386                self.codegen_store(*address, *value, dfg);
387            }
388            Instruction::Load { address } => {
389                self.codegen_load(instruction_id, *address, dfg);
390            }
391            Instruction::Not(value) => {
392                self.codegen_not(instruction_id, *value, dfg);
393            }
394            Instruction::Call { func, arguments } => {
395                self.codegen_call(instruction_id, *func, arguments, dfg);
396            }
397            Instruction::Truncate { value, bit_size, .. } => {
398                self.codegen_truncate(instruction_id, *value, *bit_size, dfg);
399            }
400            Instruction::Cast(value, _) => {
401                self.codegen_cast(instruction_id, *value, dfg);
402            }
403            Instruction::ArrayGet { array, index } => {
404                self.codegen_array_get(instruction_id, *array, *index, dfg);
405            }
406            Instruction::ArraySet { array, index, value, mutable } => {
407                self.codegen_array_set(instruction_id, *array, *index, *value, *mutable, dfg);
408            }
409            Instruction::RangeCheck { value, max_bit_size, assert_message } => {
410                self.codegen_range_check(*value, *max_bit_size, assert_message.as_ref(), dfg);
411            }
412            Instruction::IncrementRc { value } => {
413                self.codegen_increment_rc(*value, dfg);
414            }
415            Instruction::DecrementRc { value } => {
416                self.codegen_decrement_rc(*value, dfg);
417            }
418            Instruction::EnableSideEffectsIf { .. } => {
419                unreachable!("enable_side_effects not supported by brillig")
420            }
421            Instruction::IfElse { then_condition, then_value, else_condition: _, else_value } => {
422                // The `else_condition` is assumed to be the opposite of the `then_condition`.
423                self.codegen_if_else(
424                    instruction_id,
425                    *then_condition,
426                    *then_value,
427                    *else_value,
428                    dfg,
429                );
430            }
431            Instruction::MakeArray { elements: array, typ } => {
432                self.codegen_make_array(instruction_id, array, typ, dfg);
433            }
434            Instruction::Noop => (),
435        }
436
437        if !self.building_globals {
438            let dead_variables = self
439                .last_uses
440                .get(&instruction_id)
441                .expect("Last uses for instruction should have been computed");
442
443            for dead_variable in dead_variables {
444                // Globals are reserved throughout the entirety of the program
445                let is_global = dfg.is_global(*dead_variable);
446                let is_hoisted_global = self.get_hoisted_global(dfg, *dead_variable).is_some();
447                let not_global = !is_global && !is_hoisted_global;
448                if not_global {
449                    if let Some(partner) =
450                        self.function_context.coalescing.get_partner(dead_variable)
451                        && self.variables.is_allocated(&partner)
452                    {
453                        // This value shares a register with a coalescing partner that is
454                        // still alive. We must not deallocate the register yet; it will
455                        // be freed when the partner dies (or at block boundary cleanup).
456                        self.variables.remove_variable_without_dealloc(dead_variable);
457                    } else {
458                        self.variables.remove_variable(
459                            dead_variable,
460                            self.function_context,
461                            self.brillig_context,
462                        );
463                    }
464                }
465            }
466        }
467
468        // Clear the call stack; it only applied to this instruction.
469        self.brillig_context.set_call_stack(CallStackId::root());
470    }

472    /// Converts an SSA cast to a sequence of Brillig opcodes.
473    /// Casting is only necessary when shrinking the bit size of a numeric value.
474    fn convert_cast(&mut self, destination: SingleAddrVariable, source: SingleAddrVariable) {
475        // We assume that `source` is a valid `target_type` as it's expected that a truncate instruction was emitted
476        // to ensure this is the case.
477
478        self.brillig_context.cast_instruction(destination, source);
479    }

481    /// Initializes constants allocated to a [InstructionLocation] by [ConstantAllocation](crate::brillig::brillig_gen::constant_allocation::ConstantAllocation).
482    ///
483    /// It is expected that this method is called before converting an SSA instruction to Brillig
484    /// and the constants to be initialized have been precomputed and stored in [FunctionContext::constant_allocation].
485    fn initialize_constants(&mut self, dfg: &DataFlowGraph, location: InstructionLocation) {
486        let Some(constants) = self
487            .function_context
488            .constant_allocation
489            .allocated_at_location(self.block_id, location)
490            .map(|c| c.to_vec())
491        else {
492            return;
493        };
494
495        for constant_id in constants {
496            self.convert_ssa_value(constant_id, dfg);
497        }
498    }

500    /// Converts an SSA [ValueId] into a [BrilligVariable]. Initializes if necessary, or returns an existing allocation.
501    ///
502    /// This method also first checks whether the SSA value is a hoisted global constant.
503    /// If it has already been initialized in the global space, we return the already existing variable.
504    /// If an SSA value is a [Value::Global], we check whether the value exists in the [BrilligBlock::globals] map,
505    /// otherwise the method panics.
506    pub(crate) fn convert_ssa_value(
507        &mut self,
508        value_id: ValueId,
509        dfg: &DataFlowGraph,
510    ) -> BrilligVariable {
511        let value = &dfg[value_id];
512
513        if let Some(variable) = self.get_hoisted_global(dfg, value_id) {
514            return variable;
515        }
516
517        match value {
518            Value::Global(_) => {
519                unreachable!("Expected global value to be resolve to its inner value");
520            }
521            Value::Param { .. } | Value::Instruction { .. } => {
522                // All block parameters and instruction results should have already been
523                // converted to registers so we fetch from the cache.
524                if dfg.is_global(value_id) {
525                    *self.globals.get(&value_id).unwrap_or_else(|| {
526                        panic!("ICE: Global value not found in cache {value_id}")
527                    })
528                } else {
529                    self.variables.get_allocation(self.function_context, value_id)
530                }
531            }
532            Value::NumericConstant { constant, .. } => {
533                // Constants might have been converted previously or not, so we get or create and
534                // (re)initialize the value inside.
535                if self.variables.is_allocated(&value_id) {
536                    self.variables.get_allocation(self.function_context, value_id)
537                } else if dfg.is_global(value_id) {
538                    *self.globals.get(&value_id).unwrap_or_else(|| {
539                        panic!("ICE: Global value not found in cache {value_id}")
540                    })
541                } else {
542                    let new_variable = self.variables.define_variable(
543                        self.function_context,
544                        self.brillig_context,
545                        value_id,
546                        dfg,
547                    );
548
549                    self.brillig_context
550                        .const_instruction(new_variable.extract_single_addr(), *constant);
551                    new_variable
552                }
553            }
554            Value::Function(_) => {
555                // For the debugger instrumentation we want to allow passing
556                // around values representing function pointers, even though
557                // there is no interaction with the function possible given that
558                // value.
559                let new_variable = self.variables.define_variable(
560                    self.function_context,
561                    self.brillig_context,
562                    value_id,
563                    dfg,
564                );
565
566                self.brillig_context.const_instruction(
567                    new_variable.extract_single_addr(),
568                    value_id.to_u32().into(),
569                );
570                new_variable
571            }
572            Value::Intrinsic(_) | Value::ForeignFunction(_) => {
573                unreachable!("ICE: Cannot convert value to Brillig: {value:?}")
574            }
575        }
576    }

578    /// Converts an SSA `ValueId` into a single `MemoryAddress`. Initializes if necessary.
579    ///
580    /// Panics if the value was converted to an array or vector.
581    pub(crate) fn convert_ssa_single_addr_value(
582        &mut self,
583        value_id: ValueId,
584        dfg: &DataFlowGraph,
585    ) -> SingleAddrVariable {
586        let variable = self.convert_ssa_value(value_id, dfg);
587        variable.extract_single_addr()
588    }

590    /// If the supplied value is a numeric constant check whether it is exists within
591    /// the precomputed [hoisted globals map][Self::hoisted_global_constants].
592    /// If the variable exists as a hoisted global return that value, otherwise return `None`.
593    fn get_hoisted_global(
594        &self,
595        dfg: &DataFlowGraph,
596        value_id: ValueId,
597    ) -> Option<BrilligVariable> {
598        if let Value::NumericConstant { constant, typ } = &dfg[value_id]
599            && let Some(variable) = self.hoisted_global_constants.get(&(*constant, *typ))
600        {
601            return Some(*variable);
602        }
603        None
604    }

606    /// Define the result variable, convert the value, then generate Brillig opcodes to negate the variable.
607    fn codegen_not(&mut self, instruction_id: InstructionId, value: ValueId, dfg: &DataFlowGraph) {
608        let [result_id] = dfg.instruction_result(instruction_id);
609        let condition_register = self.convert_ssa_single_addr_value(value, dfg);
610        let result_register = self.variables.define_single_addr_variable(
611            self.function_context,
612            self.brillig_context,
613            result_id,
614            dfg,
615        );
616        self.brillig_context.not_instruction(condition_register, result_register);
617    }

651    /// Convert the value, and if its bit size is larger than the maximum bit size,
652    /// then generate opcodes to constrain the value to be no greater than the maximum
653    /// value as a `Field`.
654    fn codegen_range_check(
655        &mut self,
656        value: ValueId,
657        max_bit_size: u32,
658        assert_message: Option<&String>,
659        dfg: &DataFlowGraph,
660    ) {
661        let value = self.convert_ssa_single_addr_value(value, dfg);
662        // SSA generates redundant range checks. A range check with a max bit size >= value.bit_size will always pass.
663        if value.bit_size > max_bit_size {
664            // Cast original value to field
665            let left = self.brillig_context.allocate_single_addr(FieldElement::max_num_bits());
666            self.convert_cast(*left, value);
667
668            // Create a field constant with the max
669            let max = BigUint::from(2_u128).pow(max_bit_size) - BigUint::from(1_u128);
670            let right = self.brillig_context.make_constant_instruction(
671                FieldElement::from_be_bytes_reduce(&max.to_bytes_be()),
672                FieldElement::max_num_bits(),
673            );
674
675            // Check if lte max
676            let condition = self.brillig_context.allocate_single_addr_bool();
677            self.brillig_context.binary_instruction(
678                *left,
679                *right,
680                *condition,
681                BrilligBinaryOp::LessThanEquals,
682            );
683
684            self.brillig_context.codegen_constrain(*condition, assert_message.cloned());
685        }
686    }

688    /// Convert the condition and the values, define a variable for the result,
689    /// then generate opcodes for a conditional move.
690    ///
691    /// Panics if the `then_value` and `else_value` are incompatible.
692    fn codegen_if_else(
693        &mut self,
694        instruction_id: InstructionId,
695        then_condition: ValueId,
696        then_value: ValueId,
697        else_value: ValueId,
698        dfg: &DataFlowGraph,
699    ) {
700        let then_condition = self.convert_ssa_single_addr_value(then_condition, dfg);
701        let then_value = self.convert_ssa_value(then_value, dfg);
702        let else_value = self.convert_ssa_value(else_value, dfg);
703
704        let [result_id] = dfg.instruction_result(instruction_id);
705        let result = self.variables.define_variable(
706            self.function_context,
707            self.brillig_context,
708            result_id,
709            dfg,
710        );
711
712        match (then_value, else_value) {
713            (
714                BrilligVariable::SingleAddr(then_address),
715                BrilligVariable::SingleAddr(else_address),
716            ) => {
717                self.brillig_context.conditional_move_instruction(
718                    then_condition,
719                    then_address,
720                    else_address,
721                    result.extract_single_addr(),
722                );
723            }
724            (
725                BrilligVariable::BrilligArray(then_array),
726                BrilligVariable::BrilligArray(else_array),
727            ) => {
728                // Pointer to the array which result from the if-else
729                let pointer = self.brillig_context.allocate_register();
730                self.brillig_context.conditional_move_instruction(
731                    then_condition,
732                    SingleAddrVariable::new_usize(then_array.pointer),
733                    SingleAddrVariable::new_usize(else_array.pointer),
734                    SingleAddrVariable::new_usize(*pointer),
735                );
736                let if_else_array = BrilligArray { pointer: *pointer, size: then_array.size };
737                // Copy the if-else array to the result
738                self.brillig_context
739                    .call_array_copy_procedure(if_else_array, result.extract_array());
740            }
741            (
742                BrilligVariable::BrilligVector(then_vector),
743                BrilligVariable::BrilligVector(else_vector),
744            ) => {
745                // Pointer to the vector which result from the if-else
746                let pointer = self.brillig_context.allocate_register();
747                self.brillig_context.conditional_move_instruction(
748                    then_condition,
749                    SingleAddrVariable::new_usize(then_vector.pointer),
750                    SingleAddrVariable::new_usize(else_vector.pointer),
751                    SingleAddrVariable::new_usize(*pointer),
752                );
753                let if_else_vector = BrilligVector { pointer: *pointer };
754                // Copy the if-else vector to the result
755                self.brillig_context
756                    .call_vector_copy_procedure(if_else_vector, result.extract_vector());
757            }
758            _ => unreachable!("ICE - then and else values must have the same type"),
759        }
760    }
761}

763/// Returns the type of the operation considering the types of the operands.
764pub(crate) fn type_of_binary_operation(lhs_type: &Type, rhs_type: &Type) -> Type {
765    match (lhs_type, rhs_type) {
766        (_, Type::Function) | (Type::Function, _) => {
767            unreachable!("Functions are invalid in binary operations")
768        }
769        (_, Type::Reference(_)) | (Type::Reference(_), _) => {
770            unreachable!("References are invalid in binary operations")
771        }
772        (_, Type::Array(..)) | (Type::Array(..), _) => {
773            unreachable!("Arrays are invalid in binary operations")
774        }
775        (_, Type::Vector(..)) | (Type::Vector(..), _) => {
776            unreachable!("Arrays are invalid in binary operations")
777        }
778        (Type::Numeric(lhs_type), Type::Numeric(rhs_type)) => {
779            // If both sides are numeric type, then we expect their types to be the same.
780            // For SHL and SHR they are also expected to be the same, but the SHR value itself cannot exceed the bit size.
781            assert_eq!(
782                lhs_type, rhs_type,
783                "lhs and rhs types in a binary operation are always the same but got {lhs_type} and {rhs_type}"
784            );
785            Type::Numeric(*lhs_type)
786        }
787    }
