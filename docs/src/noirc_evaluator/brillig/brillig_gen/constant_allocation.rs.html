1//! This module analyzes the usage of constants in a given function and decides an allocation point for them.
2//! The allocation point will be the common dominator of all the places where the constant is used.
3//! By allocating in the common dominator, we can cache the constants for all subsequent uses.
4
5use std::collections::{BTreeMap, BTreeSet};
6
7use crate::ssa::ir::{
8    basic_block::BasicBlockId,
9    dfg::DataFlowGraph,
10    dom::DominatorTree,
11    function::Function,
12    instruction::InstructionId,
13    value::{Value, ValueId},
14};
15
16use super::variable_liveness::{is_variable, variables_used_in_instruction};
17use crate::ssa::opt::{LoopOrder, Loops};
18
19/// Indicate where a variable was used in a block.
20#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
21pub(crate) enum InstructionLocation {
22    Instruction(InstructionId),
23    Terminator,
24}
25
26/// Decisions about which block a constant should be allocated in.
27///
28/// The allocation points can further be hoisted into the global
29/// space if they are shared across functions. That algorithm is
30/// based on these per-function results.
31#[derive(Default)]
32pub(crate) struct ConstantAllocation {
33    /// Map each constant to the blocks and instructions in which it is used.
34    constant_usage: BTreeMap<ValueId, BTreeMap<BasicBlockId, Vec<InstructionLocation>>>,
35    /// Map each block and instruction to the list of constants that should be allocated at that point.
36    allocation_points: BTreeMap<BasicBlockId, BTreeMap<InstructionLocation, Vec<ValueId>>>,
37    /// The dominator tree is used to find the common dominator of all the blocks that share a constant.
38    dominator_tree: DominatorTree,
39    /// Further to finding the common dominator, we try to find a dominator that isn't part of a loop.
40    blocks_within_loops: BTreeSet<BasicBlockId>,
41}
42
43impl ConstantAllocation {
44    /// Run the constant allocation algorithm for a [Function] and return the decisions.
45    pub(crate) fn from_function(func: &Function) -> Self {
46        let loops = Loops::find_all(func, LoopOrder::OutsideIn);
47        let blocks_within_loops =
48            loops.yet_to_unroll.into_iter().flat_map(|_loop| _loop.blocks).collect();
49
50        let mut instance = ConstantAllocation {
51            constant_usage: BTreeMap::default(),
52            allocation_points: BTreeMap::default(),
53            dominator_tree: loops.dom,
54            blocks_within_loops,
55        };
56        instance.collect_constant_usage(func);
57        instance.decide_allocation_points(func);
58
59        instance
60    }
61
62    /// Collect all constants allocated in a given block.
63    pub(crate) fn allocated_in_block(&self, block_id: BasicBlockId) -> Vec<ValueId> {
64        self.allocation_points.get(&block_id).map_or(Vec::default(), |allocations| {
65            allocations.iter().flat_map(|(_, constants)| constants).copied().collect()
66        })
67    }
68
69    /// Collect all constants allocated in a given block at a specific location.
70    pub(crate) fn allocated_at_location(
71        &self,
72        block_id: BasicBlockId,
73        location: InstructionLocation,
74    ) -> Option<&[ValueId]> {
75        let allocations = self.allocation_points.get(&block_id)?;
76        let constants = allocations.get(&location)?;
77        Some(constants.as_ref())
78    }
79
80    /// Visit all constant variables in the function and record their locations.
81    fn collect_constant_usage(&mut self, func: &Function) {
82        let mut record_if_constant =
83            |block_id: BasicBlockId, value_id: ValueId, location: InstructionLocation| {
84                if is_numeric_constant(value_id, &func.dfg) {
85                    self.constant_usage
86                        .entry(value_id)
87                        .or_default()
88                        .entry(block_id)
89                        .or_default()
90                        .push(location);
91                }
92            };
93        for block_id in func.reachable_blocks() {
94            let block = &func.dfg[block_id];
95            for &inst_id in block.instructions() {
96                let variables = variables_used_in_instruction(&func.dfg[inst_id], &func.dfg);
97                for variable in variables {
98                    record_if_constant(
99                        block_id,
100                        variable,
101                        InstructionLocation::Instruction(inst_id),
102                    );
103                }
104            }
105            if let Some(terminator_instruction) = block.terminator() {
106                terminator_instruction.for_each_value(|value_id| {
107                    if is_variable(value_id, &func.dfg) {
108                        record_if_constant(block_id, value_id, InstructionLocation::Terminator);
109                    }
110                });
111            }
112        }
113    }
114
115    /// Based on the [Self::constant_usage] collected, find the common dominator of all the block where a constant is used
116    /// and mark it as the allocation point for the constant.
117    fn decide_allocation_points(&mut self, func: &Function) {
118        for (constant_id, usage_in_blocks) in self.constant_usage.iter() {
119            let block_ids: Vec<_> = usage_in_blocks.keys().copied().collect();
120
121            let allocation_point = self.decide_allocation_point(*constant_id, &block_ids, func);
122
123            // If the allocation point is one of the places where it's used, we take the first usage in the allocation point.
124            // Otherwise, we allocate it at the terminator of the allocation point.
125            let location = if let Some(locations_in_common_dominator) =
126                usage_in_blocks.get(&allocation_point)
127            {
128                *locations_in_common_dominator
129                    .first()
130                    .expect("At least one location must have been found")
131            } else {
132                InstructionLocation::Terminator
133            };
134
135            self.allocation_points
136                .entry(allocation_point)
137                .or_default()
138                .entry(location)
139                .or_default()
140                .push(*constant_id);
141        }
142    }
143
144    /// Decide where to allocate a constant, based on the common dominator of the provided block list.
145    fn decide_allocation_point(
146        &self,
147        constant_id: ValueId,
148        used_in_blocks: &[BasicBlockId],
149        func: &Function,
150    ) -> BasicBlockId {
151        // Find the common dominator of all the blocks where the constant is used.
152        let common_dominator = used_in_blocks
153            .iter()
154            .copied()
155            .reduce(|a, b| self.dominator_tree.common_dominator(a, b))
156            .unwrap_or(used_in_blocks[0]);
157
158        // If the value only contains constants, it's safe to hoist outside of any loop.
159        // Technically we know this is going to be true, because we only collected values which are `Value::NumericConstant`.
160        if func.dfg.is_constant(constant_id) {
161            self.exit_loops(common_dominator)
162        } else {
163            common_dominator
164        }
165    }
166
167    /// Returns the nearest dominator that is outside of any loop.
168    fn exit_loops(&self, block: BasicBlockId) -> BasicBlockId {
169        let mut current_block = block;
170        while self.blocks_within_loops.contains(&current_block) {
171            current_block = self
172                .dominator_tree
173                .immediate_dominator(current_block)
174                .expect("No dominator found when trying to allocate a constant outside of a loop");
175        }
176        current_block
177    }
178
179    /// Return the SSA [ValueId] of all constants (the same numeric constant might appear with multiple IDs).
180    pub(crate) fn get_constants(&self) -> BTreeSet<ValueId> {
181        self.constant_usage.keys().copied().collect()
182    }
183}
184
185fn is_numeric_constant(id: ValueId, dfg: &DataFlowGraph) -> bool {
186    matches!(&dfg[id], Value::NumericConstant { .. })
