1//! Pre-codegen block parameter coalescing for Brillig.
2//!
3//! This module analyzes `Jmp` terminators to determine which arguments can share
4//! a register with their destination block parameter. When an argument is coalesced
5//! with a parameter, the instruction defining the argument writes directly to the
6//! parameter's register, eliminating the mov at the jmp site.
7
8use crate::ssa::ir::{
9    basic_block::BasicBlockId,
10    cfg::ControlFlowGraph,
11    function::Function,
12    instruction::TerminatorInstruction,
13    value::{Value, ValueId},
14};
15
16use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};
17
18use super::variable_liveness::VariableLiveness;
19
20/// Check if param-side coalescing is safe: the destination must have exactly
21/// one predecessor, and arg must not be live-in to dest.
22fn can_coalesce_param_side(
23    arg: &ValueId,
24    destination: &BasicBlockId,
25    dest_live_in: &HashSet<ValueId>,
26    cfg: &ControlFlowGraph,
27) -> bool {
28    cfg.predecessors(*destination).count() == 1 && !dest_live_in.contains(arg)
29}
30
31/// Maps SSA ValueIds to partners whose register they should reuse.
32///
33/// An entry `x -> y` means "when defining x, reuse y's register."
34/// This covers two cases:
35/// - Arg-side (`arg -> param`): an instruction result defined in the same block as the jmp
36///   writes directly to the parameter's register.
37/// - Param-side (`param -> arg`): a block parameter reuses the register of an already-allocated
38///   value (block param passthrough, cross-block instruction result).
39#[derive(Default)]
40pub(crate) struct CoalescingMap {
41    coalesced: HashMap<ValueId, ValueId>,
42    /// Reverse mapping: values that are targets of coalescing.
43    /// Maps a coalescing target back to its source.
44    coalesced_reverse: HashMap<ValueId, ValueId>,
45}
46
47impl CoalescingMap {
48    /// Analyze all `Jmp` terminators in the function and build the coalescing map.
49    ///
50    /// For each `(argument, parameter)` pair at a jmp, we attempt to coalesce in one
51    /// of two directions:
52    /// - Arg-side: if the arg is an instruction result defined in the source block,
53    ///   record `arg -> param` so the instruction writes to the param's register.
54    /// - Param-side: if the arg is a block parameter or instruction from another block,
55    ///   record `param -> arg` so the param reuses the arg's register.
56    pub(crate) fn from_function(func: &Function, liveness: &VariableLiveness) -> Self {
57        let mut coalesced = HashMap::default();
58        let cfg = liveness.cfg();
59
60        for block_id in func.reachable_blocks() {
61            let block = &func.dfg[block_id];
62
63            let Some(TerminatorInstruction::Jmp { destination, arguments, .. }) =
64                block.terminator()
65            else {
66                continue;
67            };
68
69            let dest_block = &func.dfg[*destination];
70            let params = dest_block.parameters();
71            let dest_live_in = liveness.get_live_in(destination);
72            let instructions = func.dfg[block_id].instructions();
73
74            // Track values used as param-side targets within this jmp to prevent
75            // two params of the same destination from reusing the same register.
76            let mut param_side_targets = HashSet::default();
77
78            for (arg, param) in arguments.iter().zip(params.iter()) {
79                if arg == param {
80                    continue;
81                }
82
83                // Skip if this arg or param is already committed to a coalescing.
84                // - Arg-side guard: the same arg must not map to multiple params across jmps.
85                // - Param-side guard: two params in the same jmp must not reuse the same arg's register.
86                if coalesced.contains_key(arg) || param_side_targets.contains(arg) {
87                    continue;
88                }
89
90                // Globals are allocated separately (in the globals map, not ssa_value_allocations),
91                // so they cannot participate in coalescing. Note: we must check via `is_global`
92                // rather than matching on `func.dfg[*arg]` because the DFG's Index impl
93                // transparently resolves Global values to their underlying instruction in the
94                // globals graph, hiding the Global wrapper.
95                if func.dfg.is_global(*arg) {
96                    continue;
97                }
98
99                match &func.dfg[*arg] {
100                    Value::Instruction { instruction: defining_inst, .. } => {
101                        if let Some(def_pos) =
102                            instructions.iter().position(|id| id == defining_inst)
103                        {
104                            // Arg-side: instruction defined in this block.
105
106                            // If arg is live-in to the destination block and the destination has
107                            // other predecessors, we must not coalesce. When the destination is a
108                            // loop header (or merge point), other predecessors will write different
109                            // values to param's register on subsequent iterations, destroying arg's
110                            // value while it is still needed.
111                            if dest_live_in.contains(arg)
112                                && cfg.predecessors(*destination).count() > 1
113                            {
114                                continue;
115                            }
116
117                            let live_in = liveness.get_live_in(&block_id);
118
119                            if !live_in.contains(param) {
120                                // param is not live-in to the source block, so the defining instruction
121                                // can safely write to param's register without clobbering anything.
122                                coalesced.insert(*arg, *param);
123                                continue;
124                            }
125
126                            // Check if param is used in the defining instruction or any subsequent
127                            // instruction, or in the block terminator.
128                            let param_used_at_or_after = instructions[def_pos..]
129                                .iter()
130                                .any(|inst_id| func.dfg[*inst_id].any_value(|v| v == *param))
131                                || func.dfg[block_id]
132                                    .unwrap_terminator()
133                                    .any_value(|v| v == *param);
134
135                            if !param_used_at_or_after {
136                                coalesced.insert(*arg, *param);
137                            }
138                        } else if can_coalesce_param_side(arg, destination, dest_live_in, cfg) {
139                            // Param-side: instruction from another block.
140                            coalesced.insert(*param, *arg);
141                            param_side_targets.insert(*arg);
142                        }
143                    }
144                    Value::Param { .. } => {
145                        // Param-side: block parameter passthrough.
146                        if can_coalesce_param_side(arg, destination, dest_live_in, cfg) {
147                            coalesced.insert(*param, *arg);
148                            param_side_targets.insert(*arg);
149                        }
150                    }
151                    _ => {} // constants, globals — skip
152                }
153            }
154        }
155
156        let coalesced_reverse = coalesced.iter().map(|(k, v)| (*v, *k)).collect::<HashMap<_, _>>();
157        Self { coalesced, coalesced_reverse }
158    }
159
160    /// Forward-only lookup: if `value_id` is a coalesced arg, returns the param
161    /// whose register it reuses. Used when defining a variable to check whether
162    /// it should share an already-allocated param's register.
163    pub(crate) fn get_coalesced(&self, value_id: &ValueId) -> Option<ValueId> {
164        self.coalesced.get(value_id).copied()
165    }
166
167    /// Check whether `value_id` is a coalesced argument (i.e., shares a register with a parameter).
168    #[cfg(test)]
169    pub(crate) fn is_coalesced(&self, value_id: &ValueId) -> bool {
170        self.coalesced.contains_key(value_id)
171    }
172
173    /// Bidirectional lookup: returns the coalescing partner of `value_id`,
174    /// regardless of whether it is the arg or the param in the pair. Used when
175    /// a value dies to check whether its partner is still alive and sharing
176    /// the same register.
177    pub(crate) fn get_partner(&self, value_id: &ValueId) -> Option<ValueId> {
178        self.coalesced.get(value_id).or_else(|| self.coalesced_reverse.get(value_id)).copied()
179    }
180
181    #[cfg(test)]
182    pub(crate) fn len(&self) -> usize {
183        self.coalesced.len()
184    }
185}

196    /// Parse SSA source and build the coalescing map once.
197    fn build_coalescing(src: &str) -> (CoalescingMap, Ssa) {
198        let ssa = Ssa::from_str(src).unwrap();
199        let func = ssa.main();
200        let constants = ConstantAllocation::from_function(func);
201        let liveness = VariableLiveness::from_function(func, &constants);
202        let coalescing = CoalescingMap::from_function(func, &liveness);
203        (coalescing, ssa)
204    }

206    /// Look up the (arg, param) pair for a specific jmp terminator.
207    fn get_jmp_pair(
208        func: &crate::ssa::ir::function::Function,
209        block_idx: usize,
210        arg_idx: usize,
211    ) -> (ValueId, ValueId) {
212        let blocks: Vec<_> = func.reachable_blocks().into_iter().collect();
213        let block = &func.dfg[blocks[block_idx]];
214        let term = block.terminator().unwrap();
215        let (dest, args) = match term {
216            crate::ssa::ir::instruction::TerminatorInstruction::Jmp {
217                destination,
218                arguments,
219                ..
220            } => (*destination, arguments),
221            _ => panic!("Expected Jmp terminator"),
222        };
223        let arg = args[arg_idx];
224        let param = func.dfg[dest].parameters()[arg_idx];
225        (arg, param)
226    }

229    fn coalesces_simple_block_parameter() {
230        // The add result in b1 is passed to b3's parameter.
231        // The parameter is NOT live-in to b1, so the add can write directly
232        // to the parameter's register.
233        // Field 42 in b2 is a constant — not coalesced.
234        let src = "
235        brillig(inline) fn main f0 {
236          b0(v0: u1, v1: Field):
237            jmpif v0 then: b1, else: b2
238          b1():
239            v2 = add v1, Field 42
240            jmp b3(v2)
241          b2():
242            jmp b3(Field 42)
243          b3(v3: Field):
244            return v3
245        }
246        ";
247        // b1 is block index 1 (b0=0, b1=1, b2=2, b3=3), arg index 0

249        let func = ssa.main();
250        let (arg, param) = get_jmp_pair(func, 1, 0);
251        assert_eq!(coalescing.get_coalesced(&arg), Some(param));

253        // Check b2's jmp — constant arg should NOT be coalesced

256        assert!(!coalescing.is_coalesced(&param));
257
258        assert_eq!(coalescing.len(), 1);

539    #[test]
540    fn does_not_double_coalesce_same_arg() {
541        // The same value v1 is passed as both arguments to b1's two parameters.
542        // The first pair (v1 -> v2) coalesces, but the second pair (v1 -> v3) must be
543        // skipped because v1 is already a key in the coalescing map.
544        let src = "
545        brillig(inline) fn main f0 {
546          b0(v0: Field):
547            v1 = add v0, Field 1
548            jmp b1(v1, v1)
549          b1(v2: Field, v3: Field):
550            v4 = add v2, v3
551            return v4
552        }
553        ";
554        // b0 is block index 0, check both arg indices
555        let (coalescing, ssa) = build_coalescing(src);
556        let func = ssa.main();
557        let (arg0, param0) = get_jmp_pair(func, 0, 0);
558        let (arg1, param1) = get_jmp_pair(func, 0, 1);
559
560        // arg0 and arg1 are the same ValueId (v1)
561        assert_eq!(arg0, arg1, "both args should be the same value");
562
563        // Exactly one of the two pairs should coalesce (the first one encountered)
564        let first_coalesced = coalescing.is_coalesced(&arg0);
565        let second_param_coalesced = coalescing.is_coalesced(&param1);
566
567        assert!(first_coalesced, "first pair (v1 -> v2) should coalesce");
568        assert_eq!(
569            coalescing.get_coalesced(&arg0),
570            Some(param0),
571            "v1 should map to v2 (the first param)"
572        );
573        assert!(
574            !second_param_coalesced,
575            "second param v3 must not coalesce: v1 is already a key from the first pair"
576        );
577        assert_eq!(coalescing.len(), 1);
578    }

580    #[test]
581    fn does_not_coalesce_cross_block_multi_predecessor() {
582        // v2 is defined in b0 and passed from b1 to b3's param v4.
583        // b3 has two predecessors (b1 and b2), so idom(b3) = b0. Since v4 is
584        // allocated at the beginning of b0 (before v2's instruction runs),
585        // we cannot guarantee v2 is allocated at that point.
586        //
587        // This could be optimized in the future by checking that arg is live-in
588        // to idom(dest) using the dominator tree, which would confirm the arg
589        // is allocated before the param's definition point.
590        let src = "
591        brillig(inline) fn main f0 {
592          b0(v0: u1, v1: Field):
593            v2 = add v1, Field 1
594            jmpif v0 then: b1, else: b2
595          b1():
596            jmp b3(v2)
597          b2():
598            v3 = mul v1, Field 2
599            jmp b3(v3)
600          b3(v4: Field):
601            return v4
602        }
603        ";
604        // b1, arg 0: v2 -> v4. Not coalesced due to multi-predecessor.
605        let (coalescing, ssa) = build_coalescing(src);
606        let func = ssa.main();
607        let (arg, param) = get_jmp_pair(func, 1, 0);
608        assert!(
609            !coalescing.is_coalesced(&arg),
610            "v2 is cross-block, falls to param-side which requires single-predecessor"
611        );
612        assert!(
613            !coalescing.is_coalesced(&param),
614            "param-side coalescing requires single-predecessor destination"
615        );
616        // b2, arg 0: v3 -> v4. v3 is defined in b2 (source block) so this is arg-side.
617        let (arg, param) = get_jmp_pair(func, 2, 0);
618        // Arg-side coalescing succeeds because v3 is not live-in to b3.
619        assert!(
620            coalescing.is_coalesced(&arg),
621            "arg-side coalescing works even with multi-predecessor dest"
622        );
623        // Param v4 is not a coalescing key on the arg-side path.
624        assert!(
625            !coalescing.is_coalesced(&param),
626            "param-side coalescing requires single-predecessor destination"
627        );
628        assert_eq!(coalescing.len(), 1);
629    }

631    #[test]
632    fn does_not_coalesce_same_arg_to_two_params_param_side() {
633        // v1 is a cross-block instruction result passed as both arguments to b2.
634        // The first pair (v2 -> v1) coalesces param-side, but the second pair (v3 -> v1)
635        // must be skipped because v1 is already a param-side target within this jmp.
636        // Without this check, both v2 and v3 would try to reuse v1's register.
637        // If v2 died early and its register got reclaimed while v3 was still live
638        // we would get corrupted values.
639        // In this specific case though even though the liveness intervals of v2 and v3 overlap
640        // they are safe to coalesce. This type of analysis will be easier once we add liveness intervals.
641        let src = "
642        brillig(inline) fn main f0 {
643          b0(v0: Field):
644            v1 = add v0, Field 1
645            jmp b1()
646          b1():
647            jmp b2(v1, v1)
648          b2(v2: Field, v3: Field):
649            v4 = add v2, v3
650            return v4
651        }
652        ";
653        // b1 is block index 1, both arg indices
654        let (coalescing, ssa) = build_coalescing(src);
655        let func = ssa.main();
656        let (_arg0, param0) = get_jmp_pair(func, 1, 0);
657        let (_arg1, param1) = get_jmp_pair(func, 1, 1);
658
659        // Exactly one of the two params should coalesce with v1
660        let first_coalesced = coalescing.is_coalesced(&param0);
661        let second_coalesced = coalescing.is_coalesced(&param1);
662        assert!(first_coalesced ^ second_coalesced, "exactly one param should coalesce with v1");
663        // The first pair encountered wins
664        assert!(first_coalesced, "first param should coalesce");
665        assert!(
666            !second_coalesced,
667            "second param must not coalesce: v1 is already a param-side target"
668        );
669    }
