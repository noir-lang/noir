1//! This module handles allocation, tracking, and lifetime management of variables
2//! within a Brillig compiled SSA basic block.
3//!
4//! [BlockVariables] maintains a set of SSA [ValueId]s that are live and available
5//! during the compilation of a single SSA block into Brillig instructions. It cooperates
6//! with the [FunctionContext] to manage the mapping from SSA values to [BrilligVariable]s
7//! and with the [BrilligContext] for allocating registers.
8//!
9//! Variables are:
10//! - Allocated when first defined in a block (if not already global or hoisted to the global space).
11//! - Cached for reuse to avoid redundant register allocation.
12//! - Deallocated explicitly when no longer needed (as determined by SSA liveness).
13use acvm::{
14    FieldElement,
15    acir::brillig::lengths::{ElementTypesLength, SemanticLength, SemiFlattenedLength},
16};
17use rustc_hash::FxHashSet as HashSet;

37/// Tracks SSA variables that are live and usable during Brillig compilation of a block.
38///
39/// This structure is meant to be instantiated per SSA basic block and initialized using the
40/// the set of live variables that must be available at the block's entry.
41///
42/// It implements:
43/// - A set of active [ValueId]s that are allocated and usable.
44/// - The interface to define new variables as needed for instructions within the block.
45/// - Utilities to remove, check, and retrieve variables during Brillig codegen.
46#[derive(Debug, Default)]
47pub(crate) struct BlockVariables {
48    available_variables: HashSet<ValueId>,
49}
50
51impl BlockVariables {
52    /// Creates a BlockVariables instance. It uses the variables that are live in to the block and the global available variables (block parameters)
53    pub(crate) fn new(live_in: HashSet<ValueId>) -> Self {
54        BlockVariables { available_variables: live_in }
55    }
56
57    /// Returns the allocations for all variables that are currently available.
58    ///
59    /// These might have been technically removed earlier in a different block,
60    /// because of the order in which we process blocks, but we expect that the
61    /// `FunctionContext` will remember the earlier allocations, and return them
62    /// so that we can restore them as pre-allocations in the next slot.
63    pub(crate) fn get_available_variable_allocations(
64        &self,
65        function_context: &FunctionContext,
66    ) -> Vec<BrilligVariable> {
67        self.available_variables
68            .iter()
69            .map(|value_id| {
70                function_context
71                    .ssa_value_allocations
72                    .get(value_id)
73                    .unwrap_or_else(|| panic!("ICE: Value not found in cache {value_id}"))
74            })
75            .copied()
76            .collect()
77    }
78
79    /// For a given SSA value id, define the variable and return the corresponding cached memory allocation.
80    ///
81    /// The allocation will be cached in [FunctionContext::ssa_value_allocations], which is how it will be
82    /// passed on to the next block as a pre-allocated register, if it's still alive at that point.
83    ///
84    /// The variable is added to [Self::available_variables] to show that it's live, where it stays until
85    /// [Self::remove_variable] deletes it.
86    pub(crate) fn define_variable<Registers: RegisterAllocator>(
87        &mut self,
88        function_context: &mut FunctionContext,
89        brillig_context: &BrilligContext<FieldElement, Registers>,
90        value_id: ValueId,
91        dfg: &DataFlowGraph,
92    ) -> BrilligVariable {
93        let allocated = allocate_value(value_id, brillig_context, dfg);
94
95        // Allocators get replaced in each block, with all visible variables becoming pre-allocated
96        // in the next block; what is allocated in one block might be deallocated in the next block,
97        // in a different allocator. Because of this we can detach from the allocator, and have to
98        // manage deallocation manually.
99        let variable = allocated.detach();
100
101        if function_context.ssa_value_allocations.insert(value_id, variable).is_some() {
102            unreachable!("ICE: ValueId {value_id:?} was already in cache");
103        }
104
105        self.available_variables.insert(value_id);
106
107        variable
108    }
109
110    /// Defines a variable that fits in a single register and returns the allocated register.
111    pub(crate) fn define_single_addr_variable<Registers: RegisterAllocator>(
112        &mut self,
113        function_context: &mut FunctionContext,
114        brillig_context: &BrilligContext<FieldElement, Registers>,
115        value: ValueId,
116        dfg: &DataFlowGraph,
117    ) -> SingleAddrVariable {
118        let variable = self.define_variable(function_context, brillig_context, value, dfg);
119        variable.extract_single_addr()
120    }
121
122    /// Removes a variable so it's not used anymore within this block.
123    pub(crate) fn remove_variable<Registers: RegisterAllocator>(
124        &mut self,
125        value_id: &ValueId,
126        function_context: &FunctionContext,
127        brillig_context: &BrilligContext<FieldElement, Registers>,
128    ) {
129        assert!(self.available_variables.remove(value_id), "ICE: Variable is not available");
130
131        // Do not remove the allocation, just get it so we can mark it as free in memory.
132        let variable = function_context
133            .ssa_value_allocations
134            .get(value_id)
135            .expect("ICE: Variable allocation not found");
136
137        // Explicitly deallocate the memory; note that this might be a different register
138        // than the one the variable was defined in, which is why don't rely on automation.
139        let register = variable.extract_register();
140        brillig_context.deallocate_register(register);
141    }
142
143    /// Checks if a variable is allocated and live.
144    pub(crate) fn is_allocated(&self, value_id: &ValueId) -> bool {
145        self.available_variables.contains(value_id)
146    }
147
148    /// For a given SSA value id, return the corresponding cached allocation.
149    ///
150    /// Panics if
151    /// * the variable is not in [Self::available_variables], which means it is no longer live
152    /// * the variable is not in [FunctionContext::ssa_value_allocations], which means it was never defined
153    pub(crate) fn get_allocation(
154        &self,
155        function_context: &FunctionContext,
156        value_id: ValueId,
157    ) -> BrilligVariable {
158        assert!(
159            self.available_variables.contains(&value_id),
160            "ICE: ValueId {value_id:?} is not available"
161        );
162
163        let variable = function_context
164            .ssa_value_allocations
165            .get(&value_id)
166            .unwrap_or_else(|| panic!("ICE: Value not found in cache {value_id}"));
167
168        *variable
169    }
170}

172/// Computes the length of an array. This will match with the indexes that SSA will issue
173pub(crate) fn compute_array_length(
174    item_typ: &CompositeType,
175    elem_count: SemanticLength,
176) -> SemiFlattenedLength {
177    ElementTypesLength(assert_u32(item_typ.len())) * elem_count
178}
179
180/// For a given [ValueId], allocates the necessary registers to hold it.
181pub(crate) fn allocate_value<F, Registers: RegisterAllocator>(
182    value_id: ValueId,
183    brillig_context: &BrilligContext<F, Registers>,
184    dfg: &DataFlowGraph,
185) -> Allocated<BrilligVariable, Registers> {
186    let typ = dfg.type_of_value(value_id);
187
188    allocate_value_with_type(brillig_context, typ)
189}
190
191/// For a given [Type], allocates the necessary registers to hold it.
192pub(crate) fn allocate_value_with_type<F, Registers: RegisterAllocator>(
193    brillig_context: &BrilligContext<F, Registers>,
194    typ: Type,
195) -> Allocated<BrilligVariable, Registers> {
196    match typ {
197        Type::Numeric(_) | Type::Reference(_) | Type::Function => brillig_context
198            .allocate_single_addr(get_bit_size_from_ssa_type(&typ))
199            .map(BrilligVariable::SingleAddr),
200        Type::Array(item_typ, elem_count) => brillig_context
201            .allocate_brillig_array(compute_array_length(&item_typ, elem_count))
202            .map(BrilligVariable::BrilligArray),
203        Type::Vector(_) => {
204            brillig_context.allocate_brillig_vector().map(BrilligVariable::BrilligVector)
205        }
206    }
