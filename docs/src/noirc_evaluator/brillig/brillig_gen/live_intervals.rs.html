2//! Liveness intervals for Brillig code generation.
3//!
4//! This module provides a global per-value interval representation complementing
5//! the per-block set-based view in [VariableLiveness]. Each SSA value is assigned
6//! a conservative `[def, last_use]` range over a total program-point ordering.
7//!
8//! Single-interval representation: we use one contiguous interval per value, which
9//! over-approximates non-contiguous live ranges. This is sound for interference
10//! checks (two values whose intervals overlap might interfere) but may report
11//! false positives. A future extension can use interval sets for precision.
12//!
13//! Current uses:
14//! - Register pressure analysis
15//! - Foundation for coalescing validation
16//! - Foundation for a future linear-scan register allocator
17//!
18//! # References
19//!
20//! The interval-building algorithm is based on the BUILDINTERVALS procedure from:
21//!
22//! - Wimmer & Franz, "Linear Scan Register Allocation on SSA Form", CGO 2010
23//!   (DOI: `10.1145/1772954.1772979`), Section 4.1.
24//!
25//! Rather than re-deriving loop liveness via Wimmer's loop-header clause, we follow
26//! the practical adaptation described at <https://bernsteinbear.com/blog/linear-scan/>
27//! and reuse the pre-existing loop-aware liveness sets already computed by
28//! [VariableLiveness]. This simplifies the implementation while preserving the
29//! same interval semantics.
30//!
31//! Long-term, these intervals are intended to feed a full linear-scan register
32//! allocator (replacing the current LRU spilling strategy), which is why the
33//! Wimmer & Franz paper is the primary reference.

43/// Monotonic index assigned to block entries, instructions, and terminators in RPO.
44///
45/// Program points form a total order over the function. A lower value means
46/// "earlier in execution" (in RPO traversal order). This is used to define
47/// intervals and check overlap.
48#[derive(Debug, Default, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
49pub(crate) struct ProgramPoint(u32);
50
51/// Conservative single-interval approximation of a value's liveness.
52///
53/// Over-approximates for non-contiguous live ranges (sound for interference).
54/// `def` is the program point where the value is defined, and `last_use` is the
55/// last program point where it is consumed.
56#[derive(Debug, Clone, Copy, PartialEq, Eq)]
57pub(crate) struct LiveInterval {
58    pub(crate) def: ProgramPoint,
59    pub(crate) last_use: ProgramPoint,
60}
61
62impl LiveInterval {
63    fn new(def: ProgramPoint, last_use: ProgramPoint) -> Self {
64        assert!(def <= last_use);
65        Self { def, last_use }
66    }
67
68    /// Check whether two intervals overlap (conservative interference).
69    #[cfg(test)]
70    pub(crate) fn interferes(&self, other: &LiveInterval) -> bool {
71        self.def <= other.last_use && other.def <= self.last_use
72    }
73
74    /// Check whether a program point falls within this interval.
75    #[cfg(test)]
76    pub(crate) fn contains(&self, point: ProgramPoint) -> bool {
77        self.def <= point && point <= self.last_use
78    }
79}

81/// Global liveness intervals for all values in a function.
82///
83/// Built from [VariableLiveness] (which provides loop-aware `live_in`/`live_out`)
84/// and [ConstantAllocation] (which provides constant allocation points).
85#[derive(Default)]
86pub(crate) struct LiveIntervals {
87    /// Map from block to the program point at its entry.
88    block_entry_points: HashMap<BasicBlockId, ProgramPoint>,
89    /// Map from instruction to its program point.
90    instruction_points: HashMap<InstructionId, ProgramPoint>,
91    /// Map from block to its terminator's program point.
92    terminator_points: HashMap<BasicBlockId, ProgramPoint>,
93    /// Per-value liveness interval.
94    intervals: HashMap<ValueId, LiveInterval>,
95    /// Maximum program point assigned (for iteration bounds).
96    max_point: ProgramPoint,
97}
98
99#[cfg(test)]
100impl LiveIntervals {
101    /// Build liveness intervals for a function.
102    ///
103    /// See the module-level docs for algorithm references and [Self::build_intervals]
104    /// for the annotated pseudocode mapping.
105    pub(crate) fn from_function(
106        func: &Function,
107        liveness: &VariableLiveness,
108        constants: &ConstantAllocation,
109        post_order: &[BasicBlockId],
110    ) -> Self {
111        let mut result = Self {
112            block_entry_points: HashMap::default(),
113            instruction_points: HashMap::default(),
114            terminator_points: HashMap::default(),
115            intervals: HashMap::default(),
116            max_point: ProgramPoint(0),
117        };
118
119        // Step 0: Assign program points in reverse post-order (RPO).
120        result.assign_program_points(func, post_order);
121
122        // Step 1: Build intervals by processing blocks in post-order.
123        result.build_intervals(func, liveness, post_order);
124
125        // Step 2: Post-process for Noir's idom allocation of block params.
126        result.adjust_block_param_defs(liveness, post_order);
127
128        // Step 3: Handle constants allocated at specific block entries.
129        result.adjust_constant_defs(constants, post_order);
130
131        result
132    }
133
134    /// Assign monotonically increasing program points to block entries,
135    /// instructions, and terminators in RPO.
136    fn assign_program_points(&mut self, func: &Function, post_order: &[BasicBlockId]) {
137        let mut index: u32 = 0;
138
139        for &block_id in post_order.iter().rev() {
140            // Block entry point.
141            self.block_entry_points.insert(block_id, ProgramPoint(index));
142            index += 1;
143
144            // Instructions.
145            let block = &func.dfg[block_id];
146            for &inst_id in block.instructions() {
147                self.instruction_points.insert(inst_id, ProgramPoint(index));
148                index += 1;
149            }
150
151            // Terminator.
152            // Every block in valid SSA must have one.
153            block.unwrap_terminator();
154            self.terminator_points.insert(block_id, ProgramPoint(index));
155            index += 1;
156        }
157
158        self.max_point = ProgramPoint(index.saturating_sub(1));
159    }
160
161    /// Build intervals by processing blocks in post-order (reverse of RPO).
162    ///
163    /// Implements BUILDINTERVALS from Wimmer & Franz 2010, Section 4.1.
164    /// The pseudocode below shows the original algorithm on the left and our
165    /// mapping on the right:
166    ///
167    /// ```text
168    /// BUILDINTERVALS
169    ///
170    /// for each block b in reverse order do          <- post_order iter (== reverse RPO)
171    ///   live = union of successor.liveIn for each   <- liveness.get_live_out() (equivalent:
172    ///          successor of b                         live_out = ∪ live_in(s) for successors s)
173    ///   for each phi function phi of successors     <- terminator.for_each_value() — adds jmp args
174    ///          of b do                                (phi inputs), jmpif conditions, return values
175    ///     live.add(phi.inputOf(b))
176    ///   for each opd in live do                     <- the `for &v in &live` loop
177    ///     intervals[opd].addRange(b.from, b.to)
178    ///   for each operation op of b in reverse do    <- block.instructions().iter().rev()
179    ///     for each output operand opd of op do      <- func.dfg.instruction_results()
180    ///       intervals[opd].setFrom(op.id)
181    ///       live.remove(opd)                        <- SKIPPED: `live` is not stored. It is only used to seed intervals
182    ///     for each input operand opd of op do       <- instruction.for_each_value()
183    ///       intervals[opd].addRange(b.from, op.id)
184    ///       live.add(opd)                           <- SKIPPED: `live` is not stored. It is only used to seed intervals
185    ///   for each phi function phi of b do           <- SKIPPED: `live` is not stored. It is only used to seed intervals
186    ///     live.remove(phi.output)
187    ///   if b is loop header then                    <- SKIPPED: VariableLiveness already propagates
188    ///     loopEnd = last block of loop at b           loop liveness through back-edges, so live_out
189    ///     for each opd in live do                     already includes loop-carried values.
190    ///       intervals[opd].addRange(b.from,           See <https://bernsteinbear.com/blog/linear-scan/>
191    ///                               loopEnd.to)
192    ///   b.liveIn = live                             <- NOT STORED: we don't need per-block liveIn
193    ///                                                 since VariableLiveness already provides it.
194    /// ```
195    ///
196    /// Key differences from the paper:
197    /// - `live_out` vs `union of successor.liveIn`: These are equivalent —
198    ///   [VariableLiveness::get_live_out()] is literally `∪ live_in(s) for each successor s`.
199    /// - Phi inputs -> terminator operands: Wimmer adds phi inputs explicitly;
200    ///   we add all terminator operands (jmp args are the phi inputs in Noir's
201    ///   block-parameter SSA form).
202    /// - Loop header clause skipped: [VariableLiveness] already handles loop
203    ///   liveness propagation, so `live_out` includes loop-carried values.
204    /// - `b.liveIn` not stored: Wimmer builds liveIn bottom-up; we already
205    ///   have it from [VariableLiveness].
206    fn build_intervals(
207        &mut self,
208        func: &Function,
209        liveness: &VariableLiveness,
210        post_order: &[BasicBlockId],
211    ) {
212        // Process blocks in post-order so that successors are
213        // processed before predecessors, matching the Wimmer algorithm.
214        for &block_id in post_order {
215            let block = &func.dfg[block_id];
216            let block_entry = self.block_entry_points[&block_id];
217            let block_end = self.terminator_points[&block_id];
218
219            // Start with live_out: values alive after the block.
220            let mut live: HashSet<ValueId> = liveness.get_live_out(&block_id);
221
222            // Add ALL terminator operands to the live set.
223            // Jmp arguments (phi inputs), jmpif conditions, return values --
224            // these must be alive at the terminator but may not be in live_out.
225            let terminator = block.unwrap_terminator();
226            terminator.for_each_value(|v| {
227                if super::variable_liveness::is_variable(v, &func.dfg) {
228                    live.insert(v);
229                }
230            });
231
232            // All live values get a range covering the entire block.
233            for &v in &live {
234                self.add_range(v, block_entry, block_end);
235            }
236
237            // Walk instructions in reverse order.
238            for &inst_id in block.instructions().iter().rev() {
239                let inst_point = self.instruction_points[&inst_id];
240
241                // For each output: set def to this instruction.
242                let results = func.dfg.instruction_results(inst_id);
243                for &result in results {
244                    self.set_from(result, inst_point);
245                }
246
247                // For each input: add range from block entry to this instruction.
248                let instruction = &func.dfg[inst_id];
249                instruction.for_each_value(|v| {
250                    if super::variable_liveness::is_variable(v, &func.dfg) {
251                        self.add_range(v, block_entry, inst_point);
252                    }
253                });
254            }
255        }
256    }

258    /// Extend each block param's def backward to its idom's entry point.
259    ///
260    /// In Brillig, block params are allocated at their immediate dominator
261    /// (the last point before the branches diverge). This matches the current
262    /// codegen behavior where param registers are reserved at the idom.
263    fn adjust_block_param_defs(
264        &mut self,
265        liveness: &VariableLiveness,
266        post_order: &[BasicBlockId],
267    ) {
268        for &block_id in post_order {
269            let entry_point = self.block_entry_points[&block_id];
270            for param in liveness.defined_block_params(&block_id) {
271                if let Some(interval) = self.intervals.get_mut(&param) {
272                    interval.def = std::cmp::min(interval.def, entry_point);
273                }
274            }
275        }
276    }
277
278    /// Extend constant defs to their allocation block's entry point.
279    ///
280    /// Constants from `ConstantAllocation::allocated_in_block` are materialized
281    /// at specific block entries. We extend their def to match.
282    fn adjust_constant_defs(
283        &mut self,
284        constants: &ConstantAllocation,
285        post_order: &[BasicBlockId],
286    ) {
287        for &block_id in post_order {
288            let entry_point = self.block_entry_points[&block_id];
289            for constant_id in constants.allocated_in_block(block_id) {
290                if let Some(interval) = self.intervals.get_mut(&constant_id) {
291                    interval.def = std::cmp::min(interval.def, entry_point);
292                }
293            }
294        }
295    }

297    /// Add a range `[from, to]` for a value. If the value already has an interval,
298    /// extend it to cover the union.
299    fn add_range(&mut self, value: ValueId, from: ProgramPoint, to: ProgramPoint) {
300        self.intervals
301            .entry(value)
302            .and_modify(|iv| {
303                iv.def = std::cmp::min(iv.def, from);
304                iv.last_use = std::cmp::max(iv.last_use, to);
305            })
306            .or_insert(LiveInterval::new(from, to));
307    }
308
309    /// Set the def point for a value, narrowing it to the actual definition.
310    fn set_from(&mut self, value: ValueId, point: ProgramPoint) {
311        self.intervals
312            .entry(value)
313            .and_modify(|iv| iv.def = point)
314            .or_insert(LiveInterval::new(point, point));
315    }

350    /// Count how many value intervals are live at a given block's entry point.
351    #[cfg(test)]
352    pub(crate) fn register_pressure_at_block(&self, block: BasicBlockId) -> usize {
353        let Some(&entry) = self.block_entry_points.get(&block) else {
354            return 0;
355        };
356        self.intervals.values().filter(|iv| iv.contains(entry)).count()
357    }
358
359    /// Return the maximum number of simultaneously live values across all program points.
360    ///
361    /// This is an approximation. We check at every assigned program point how many
362    /// intervals are alive. For large functions a sweep-line algorithm would be
363    /// more efficient, but this is adequate for diagnostic use.
364    #[cfg(test)]
365    pub(crate) fn max_register_pressure(&self) -> usize {
366        if self.intervals.is_empty() {
367            return 0;
368        }
369
370        // Collect all interval endpoints and sweep.
371        let mut events: Vec<(u32, i32)> = Vec::new();
372        for interval in self.intervals.values() {
373            events.push((interval.def.0, 1)); // interval starts
374            events.push((interval.last_use.0 + 1, -1)); // interval ends (exclusive)
375        }
376        events.sort();
377
378        let mut max_pressure = 0usize;
379        let mut current = 0i32;
380        for (_, delta) in events {
381            current += delta;
382            max_pressure = max_pressure.max(current as usize);
383        }
384        max_pressure
385    }

764    fn constants_in_diamond() {
765        // A constant used in both branches of a diamond should be allocated
766        // at the common dominator (b0), so its live interval def is extended
767        // to b0's entry.
768        let src = "
769        brillig(inline) fn main f0 {
770          b0(v0: u1, v1: Field):
771            jmpif v0 then: b1(), else: b2()
772          b1():
773            v2 = add v1, Field 42
774            jmp b3(v2)
775          b2():
776            v3 = mul v1, Field 42
777            jmp b3(v3)
778          b3(v4: Field):
779            return v4
780        }
781        ";
782        let (intervals, ssa) = build_intervals(src);
783        let func = ssa.main();
784        let rpo = PostOrder::with_function(func).into_vec_reverse();
785
786        let [b0, b1, b2, b3] = block_ids();
787        assert_eq!(rpo, vec![b0, b2, b1, b3]);
788
789        let b0_entry = intervals.block_entry_point(b0).unwrap();
790
791        let constants = ConstantAllocation::from_function(func);
792
793        // Field 42 is used in both b1 and b2, so it should be allocated at b0
794        // (the common dominator). It's the only constant in this function.
795        let b0_constants = constants.allocated_in_block(b0);
796        assert_eq!(
797            b0_constants.len(),
798            1,
799            "exactly one constant (Field 42) should be allocated in b0"
800        );
801        let constant_id = b0_constants[0];
802
803        // The constant's interval def should be at b0's entry.
804        let iv = intervals.get(constant_id).expect("constant should have an interval");
805        assert_eq!(iv.def, b0_entry, "Field 42 def should be at b0 entry (common dominator)");
806
807        // RPO is [b0, b2, b1, b3], so b1 comes after b2.
808        // Field 42's last_use should be at b1's instruction (the later use in RPO).
809        let b1_inst_pt = intervals.instruction_point(inst_id(func, b1, 0)).unwrap();
810        assert_eq!(
811            iv.last_use, b1_inst_pt,
812            "Field 42 last_use should be at b1's instruction (later in RPO)"
813        );
814    }
