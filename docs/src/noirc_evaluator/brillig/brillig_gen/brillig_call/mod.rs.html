24    /// Allocates a variable to hold the result of an external function call (e.g., foreign or black box).
25    /// For more information about foreign function calls in Brillig take a look at the [foreign call opcode][acvm::acir::brillig::Opcode::ForeignCall].
26    ///
27    /// This is typically used during Brillig codegen for calls to [Value::ForeignFunction], where
28    /// external host functions return values back into the program.
29    ///
30    /// Numeric types and fixed-sized array results are directly allocated.
31    ///
32    /// As vectors are determined at runtime they are allocated differently:
33    /// - Allocates memory for a [BrilligVariable::BrilligVector], which holds a pointer and dynamic size.
34    /// - Initializes the pointer using the free memory pointer.
35    /// - The actual size will be updated after the foreign function call returns.
36    ///
37    /// # Returns
38    /// A [BrilligVariable] representing the allocated memory structure to store the foreign call's result.
39    ///
40    /// # Panics
41    /// If there is a vector among the output variables _and_ it's followed by another vector:
42    /// when we allocate memory for a vector, we don't know its length, so it just points at the current
43    /// free memory pointer without increasing it; a second vector gets allocated at the same memory slot.
44    fn allocate_external_call_results(
45        &mut self,
46        results: &[ValueId],
47        dfg: &DataFlowGraph,
48    ) -> Vec<BrilligVariable> {
49        let mut variables = Vec::new();
50        let mut vector_allocated = None;
51
52        for result in results {
53            let result = *result;
54            let typ = dfg[result].get_type();
55            let variable = match typ.as_ref() {
56                Type::Numeric(_) => self.variables.define_variable(
57                    self.function_context,
58                    self.brillig_context,
59                    result,
60                    dfg,
61                ),
62
63                Type::Array(..) => {
64                    let variable = self.variables.define_variable(
65                        self.function_context,
66                        self.brillig_context,
67                        result,
68                        dfg,
69                    );
70                    let array = variable.extract_array();
71
72                    self.allocate_foreign_call_result_array(typ.as_ref(), array);
73
74                    variable
75                }
76                Type::Vector(_) => {
77                    let variable = self.variables.define_variable(
78                        self.function_context,
79                        self.brillig_context,
80                        result,
81                        dfg,
82                    );
83
84                    // Set its pointer to the free memory address, and expect the VM to write the data where the vector points to.
85                    // We can only support one vector output this way, otherwise the next vector would overwrite it.
86                    // The vector also has to be the last output of the function, there cannot be any arrays following it.
87                    assert!(
88                        vector_allocated.is_none(),
89                        "a previous vector has already been allocated at the free memory pointer"
90                    );
91                    // Remember the position of single vector we allocated; we will initialize it to the free memory pointer
92                    // after we have dealt with any other arrays in the output, otherwise they could overwrite it.
93                    vector_allocated = Some(variables.len());
94
95                    variable
96                }
97                _ => {
98                    unreachable!("ICE: unsupported return type for black box call {typ:?}")
99                }
100            };
101            variables.push(variable);
102        }
103
104        if let Some(idx) = vector_allocated {
105            let variable = &variables[idx];
106            let vector = variable.extract_vector();
107            self.brillig_context.load_free_memory_pointer_instruction(vector.pointer);
108        }
109
110        variables
111    }

113    /// Recursively allocates memory on the heap for a nested array returned from a foreign function call.
114    ///
115    /// # Panics
116    /// - If the provided `typ` is not an array.
117    /// - If any vector types are encountered within the nested structure, since vectors
118    ///   require runtime size information and cannot be allocated statically here.
119    fn allocate_foreign_call_result_array(&mut self, typ: &Type, array: BrilligArray) {
120        let Type::Array(types, size) = typ else {
121            unreachable!("ICE: allocate_foreign_call_array() expects an array, got {typ:?}")
122        };
123
124        // Reserve free memory on the heap and set the initial ref-count.
125        self.brillig_context.codegen_initialize_array(array);
126
127        // Go through each slot in the array: if it's a simple type then we don't need to do anything,
128        // but if it's a nested one we have to recursively allocate memory for it, and store the variable in the array.
129        // We add one since array.pointer points to [RC, ...items]
130        let mut index = offsets::ARRAY_ITEMS;
131        for _ in 0..size.0 {
132            for element_type in types.iter() {
133                match element_type {
134                    Type::Array(items, nested_size) => {
135                        // Allocate a pointer for an array on the stack.
136                        let size: SemiFlattenedLength =
137                            ElementTypesLength(assert_u32(items.len())) * *nested_size;
138                        let inner_array = self.brillig_context.allocate_brillig_array(size);
139
140                        // Recursively allocate memory for the inner array on the heap.
141                        // This sets the pointer on the stack to point at the heap.
142                        self.allocate_foreign_call_result_array(element_type, *inner_array);
143
144                        // Set the index in the outer array to be the total offset accounting for complex types.
145                        let idx =
146                            self.brillig_context.make_usize_constant_instruction(index.into());
147
148                        // Copy the inner array pointer, which points at the heap, into the outer array cell.
149                        // After this, it is okay for the `inner_array` to be deallocated from the stack,
150                        // and for its address to be reused, ie. we don't need to `.detach()` it.
151                        // What matters is that we stored the pointer to the heap.
152                        self.brillig_context.codegen_store_with_offset(
153                            array.pointer,
154                            *idx,
155                            inner_array.pointer,
156                        );
157                    }
158                    Type::Vector(_) => unreachable!(
159                        "ICE: unsupported vector type in allocate_nested_array(), expects an array or a numeric type"
160                    ),
161                    _ => (),
162                }
163                index += 1;
164            }
165        }
166    }

168    /// Internal method to codegen an [crate::ssa::ir::instruction::Instruction::Call] to a [Value::Function]
169    fn convert_ssa_function_call(
170        &mut self,
171        func_id: FunctionId,
172        arguments: &[ValueId],
173        dfg: &DataFlowGraph,
174        result_ids: &[ValueId],
175    ) {
176        let argument_variables =
177            vecmap(arguments, |argument_id| self.convert_ssa_value(*argument_id, dfg));
178        let return_variables = vecmap(result_ids, |result_id| {
179            self.variables.define_variable(
180                self.function_context,
181                self.brillig_context,
182                *result_id,
183                dfg,
184            )
185        });
186        self.brillig_context.codegen_call(func_id, &argument_variables, &return_variables);
187    }

189    /// Increase or decrease the vector length by 1.
190    ///
191    /// Vectors have a tuple structure (vector length, vector contents) to enable logic
192    /// that uses dynamic vector lengths (such as with merging vectors in the flattening pass).
193    /// This method codegens an update to the vector length.
194    ///
195    /// The binary operation performed on the vector length is always an addition or subtraction of `1`.
196    /// This is because the vector length holds the user length (length as displayed by a `.len()` call),
197    /// and not a flattened length used internally to represent arrays of tuples.
198    /// The length inside of `RegisterOrMemory::HeapVector` represents the entire flattened number
199    /// of fields in the vector.
200    ///
201    /// Note that when we subtract a value, we expect that there is a constraint in SSA
202    /// to check that the length isn't already 0. We could add a constraint opcode here,
203    /// but if it's in SSA, there is a chance it can be optimized out.
204    fn update_vector_length(
205        &mut self,
206        target_len: SingleAddrVariable,
207        source_len: SingleAddrVariable,
208        binary_op: BrilligBinaryOp,
209    ) {
210        assert!(matches!(binary_op, BrilligBinaryOp::Add | BrilligBinaryOp::Sub));
211        self.brillig_context.codegen_usize_op(source_len.address, target_len.address, binary_op, 1);
212    }

214    /// Copy the input arguments to the results.
215    fn convert_ssa_identity_call(
216        &mut self,
217        arguments: &[ValueId],
218        dfg: &DataFlowGraph,
219        result_ids: &[ValueId],
220    ) {
221        let argument_variables =
222            vecmap(arguments, |argument_id| self.convert_ssa_value(*argument_id, dfg));
223
224        let return_variables = vecmap(result_ids, |result_id| {
225            self.variables.define_variable(
226                self.function_context,
227                self.brillig_context,
228                *result_id,
229                dfg,
230            )
231        });
232
233        for (src, dst) in argument_variables.into_iter().zip(return_variables) {
234            self.brillig_context.mov_instruction(dst.extract_register(), src.extract_register());
235        }
236    }

238    /// Convert the SSA vector operations to brillig vector operations
239    fn convert_ssa_vector_intrinsic_call(
240        &mut self,
241        dfg: &DataFlowGraph,
242        intrinsic: &Value,
243        instruction_id: InstructionId,
244        arguments: &[ValueId],
245    ) {
246        // Vector operations always look like `... = call vector_<op> source_len, source_vector, ...`
247        let source_len = self.convert_ssa_value(arguments[0], dfg);
248        let source_len = source_len.extract_single_addr();
249
250        let vector_id = arguments[1];
251        let element_size = dfg.type_of_value(vector_id).element_size().to_usize();
252        let source_vector = self.convert_ssa_value(vector_id, dfg).extract_vector();
253
254        let results = dfg.instruction_results(instruction_id);
255
256        let get_target_len = |this: &mut Self, idx: usize| {
257            this.variables
258                .define_variable(this.function_context, this.brillig_context, results[idx], dfg)
259                .extract_single_addr()
260        };
261
262        let get_target_vector = |this: &mut Self, idx: usize| {
263            this.variables
264                .define_variable(this.function_context, this.brillig_context, results[idx], dfg)
265                .extract_vector()
266        };
267
268        match intrinsic {
269            Value::Intrinsic(Intrinsic::VectorPushBack) => {
270                // target_len, target_vector = vector_push_back source_len, source_vector, ...elements
271                let target_len = get_target_len(self, 0);
272                let target_vector = get_target_vector(self, 1);
273
274                let item_values = vecmap(&arguments[2..element_size + 2], |arg| {
275                    self.convert_ssa_value(*arg, dfg)
276                });
277
278                // target_len = source_len + 1
279                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Add);
280
281                self.vector_push_back_operation(
282                    target_vector,
283                    source_len,
284                    source_vector,
285                    &item_values,
286                );
287            }
288            Value::Intrinsic(Intrinsic::VectorPushFront) => {
289                let target_len = get_target_len(self, 0);
290                let target_vector = get_target_vector(self, 1);
291
292                let item_values = vecmap(&arguments[2..element_size + 2], |arg| {
293                    self.convert_ssa_value(*arg, dfg)
294                });
295
296                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Add);
297
298                self.vector_push_front_operation(
299                    target_vector,
300                    source_len,
301                    source_vector,
302                    &item_values,
303                );
304            }
305            Value::Intrinsic(Intrinsic::VectorPopBack) => {
306                let target_len = get_target_len(self, 0);
307                let target_vector = get_target_vector(self, 1);
308
309                let pop_variables = vecmap(&results[2..element_size + 2], |result| {
310                    self.variables.define_variable(
311                        self.function_context,
312                        self.brillig_context,
313                        *result,
314                        dfg,
315                    )
316                });
317
318                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Sub);
319
320                self.vector_pop_back_operation(
321                    target_vector,
322                    source_len,
323                    source_vector,
324                    &pop_variables,
325                );
326            }
327            Value::Intrinsic(Intrinsic::VectorPopFront) => {
328                // ...elements, target_len, target_vector = vector_pop_front len, vector
329                let target_len = get_target_len(self, element_size);
330                let target_vector = get_target_vector(self, element_size + 1);
331
332                let pop_variables = vecmap(&results[0..element_size], |result| {
333                    self.variables.define_variable(
334                        self.function_context,
335                        self.brillig_context,
336                        *result,
337                        dfg,
338                    )
339                });
340
341                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Sub);
342
343                self.vector_pop_front_operation(
344                    target_vector,
345                    source_len,
346                    source_vector,
347                    &pop_variables,
348                );
349            }
350            Value::Intrinsic(Intrinsic::VectorInsert) => {
351                let target_len = get_target_len(self, 0);
352                let target_vector = get_target_vector(self, 1);
353
354                // Remove if indexing in insert is changed to flattened indexing
355                // https://github.com/noir-lang/noir/issues/1889#issuecomment-1668048587
356                let user_index = self.convert_ssa_single_addr_value(arguments[2], dfg);
357
358                let converted_index =
359                    self.brillig_context.make_usize_constant_instruction(element_size.into());
360
361                // Safety: This multiplication cannot overflow because:
362                // 1. SSA generates bounds checks ensuring `user_index <= length`
363                // 2. The vector allocation is protected by FMP's checked addition
364                // 3. Therefore `element_size * user_index <= element_size * length <= allocation_size < 2^32`
365                self.brillig_context.memory_op_instruction(
366                    converted_index.address,
367                    user_index.address,
368                    converted_index.address,
369                    BrilligBinaryOp::Mul,
370                );
371
372                let items = vecmap(&arguments[3..element_size + 3], |arg| {
373                    self.convert_ssa_value(*arg, dfg)
374                });
375
376                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Add);
377
378                self.vector_insert_operation(
379                    target_vector,
380                    source_vector,
381                    *converted_index,
382                    &items,
383                );
384            }
385            Value::Intrinsic(Intrinsic::VectorRemove) => {
386                let target_len = get_target_len(self, 0);
387                let target_vector = get_target_vector(self, 1);
388
389                // Remove if indexing in remove is changed to flattened indexing
390                // https://github.com/noir-lang/noir/issues/1889#issuecomment-1668048587
391                let user_index = self.convert_ssa_single_addr_value(arguments[2], dfg);
392
393                let converted_index =
394                    self.brillig_context.make_usize_constant_instruction(element_size.into());
395
396                // Safety: This multiplication cannot overflow because:
397                // 1. SSA generates bounds checks ensuring `user_index < length`
398                // 2. The vector allocation is protected by FMP's checked addition
399                // 3. Therefore `element_size * user_index < element_size * length <= allocation_size < 2^32`
400                self.brillig_context.memory_op_instruction(
401                    converted_index.address,
402                    user_index.address,
403                    converted_index.address,
404                    BrilligBinaryOp::Mul,
405                );
406
407                let removed_items = vecmap(&results[2..element_size + 2], |result| {
408                    self.variables.define_variable(
409                        self.function_context,
410                        self.brillig_context,
411                        *result,
412                        dfg,
413                    )
414                });
415
416                self.update_vector_length(target_len, source_len, BrilligBinaryOp::Sub);
417
418                self.vector_remove_operation(
419                    target_vector,
420                    source_vector,
421                    *converted_index,
422                    &removed_items,
423                );
424            }
425            _ => unreachable!("ICE: Vector operation not supported"),
426        }
427    }
