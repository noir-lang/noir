1//! Codegen for native (black box) function calls.
2use acvm::{
3    AcirField,
4    acir::{BlackBoxFunc, brillig::BlackBoxOp},
5};
6
7use crate::brillig::brillig_ir::{
8    BrilligContext, brillig_variable::BrilligVariable, debug_show::DebugToString,
9    registers::RegisterAllocator,
10};
11
12/// Transforms SSA's black box function calls into the corresponding brillig instructions
13/// Extracting arguments and results from the SSA function call
14/// And making any necessary type conversions to adapt noir's blackbox calls to brillig's
15pub(crate) fn convert_black_box_call<F: AcirField + DebugToString, Registers: RegisterAllocator>(
16    brillig_context: &mut BrilligContext<F, Registers>,
17    bb_func: &BlackBoxFunc,
18    function_arguments: &[BrilligVariable],
19    function_results: &[BrilligVariable],
20) {
21    match bb_func {
22        BlackBoxFunc::Blake2s => {
23            if let (
24                [BrilligVariable::BrilligArray(message)],
25                [BrilligVariable::BrilligArray(result_array)],
26            ) = (function_arguments, function_results)
27            {
28                let message_array = brillig_context.codegen_brillig_array_to_heap_array(*message);
29                let output_heap_array =
30                    brillig_context.codegen_brillig_array_to_heap_array(*result_array);
31
32                brillig_context.black_box_op_instruction(BlackBoxOp::Blake2s {
33                    message: *message_array,
34                    output: *output_heap_array,
35                });
36            } else {
37                unreachable!("ICE: Blake2s expects one array argument and one array result")
38            }
39        }
40        BlackBoxFunc::Blake3 => {
41            if let (
42                [BrilligVariable::BrilligArray(message)],
43                [BrilligVariable::BrilligArray(result_array)],
44            ) = (function_arguments, function_results)
45            {
46                let message_array = brillig_context.codegen_brillig_array_to_heap_array(*message);
47                let output_heap_array =
48                    brillig_context.codegen_brillig_array_to_heap_array(*result_array);
49
50                brillig_context.black_box_op_instruction(BlackBoxOp::Blake3 {
51                    message: *message_array,
52                    output: *output_heap_array,
53                });
54            } else {
55                unreachable!("ICE: Blake3 expects one array argument and one array result")
56            }
57        }
58        BlackBoxFunc::Keccakf1600 => {
59            if let (
60                [BrilligVariable::BrilligArray(input_array)],
61                [BrilligVariable::BrilligArray(result_array)],
62            ) = (function_arguments, function_results)
63            {
64                let input_heap_array =
65                    brillig_context.codegen_brillig_array_to_heap_array(*input_array);
66                let output_heap_array =
67                    brillig_context.codegen_brillig_array_to_heap_array(*result_array);
68
69                brillig_context.black_box_op_instruction(BlackBoxOp::Keccakf1600 {
70                    input: *input_heap_array,
71                    output: *output_heap_array,
72                });
73            } else {
74                unreachable!("ICE: Keccakf1600 expects one array argument and one array result")
75            }
76        }
77        BlackBoxFunc::EcdsaSecp256k1 => {
78            if let (
79                [
80                    BrilligVariable::BrilligArray(public_key_x),
81                    BrilligVariable::BrilligArray(public_key_y),
82                    BrilligVariable::BrilligArray(signature),
83                    BrilligVariable::BrilligArray(message),
84                ],
85                [BrilligVariable::SingleAddr(result_register)],
86            ) = (function_arguments, function_results)
87            {
88                let hashed_msg = brillig_context.codegen_brillig_array_to_heap_array(*message);
89                let public_key_x =
90                    brillig_context.codegen_brillig_array_to_heap_array(*public_key_x);
91                let public_key_y =
92                    brillig_context.codegen_brillig_array_to_heap_array(*public_key_y);
93                let signature = brillig_context.codegen_brillig_array_to_heap_array(*signature);
94
95                brillig_context.black_box_op_instruction(BlackBoxOp::EcdsaSecp256k1 {
96                    hashed_msg: *hashed_msg,
97                    public_key_x: *public_key_x,
98                    public_key_y: *public_key_y,
99                    signature: *signature,
100                    result: result_register.address,
101                });
102            } else {
103                unreachable!(
104                    "ICE: EcdsaSecp256k1 expects four array arguments and one register result"
105                )
106            }
107        }
108        BlackBoxFunc::EcdsaSecp256r1 => {
109            if let (
110                [
111                    BrilligVariable::BrilligArray(public_key_x),
112                    BrilligVariable::BrilligArray(public_key_y),
113                    BrilligVariable::BrilligArray(signature),
114                    BrilligVariable::BrilligArray(message),
115                ],
116                [BrilligVariable::SingleAddr(result_register)],
117            ) = (function_arguments, function_results)
118            {
119                let hashed_msg = brillig_context.codegen_brillig_array_to_heap_array(*message);
120                let public_key_x =
121                    brillig_context.codegen_brillig_array_to_heap_array(*public_key_x);
122                let public_key_y =
123                    brillig_context.codegen_brillig_array_to_heap_array(*public_key_y);
124                let signature = brillig_context.codegen_brillig_array_to_heap_array(*signature);
125
126                brillig_context.black_box_op_instruction(BlackBoxOp::EcdsaSecp256r1 {
127                    hashed_msg: *hashed_msg,
128                    public_key_x: *public_key_x,
129                    public_key_y: *public_key_y,
130                    signature: *signature,
131                    result: result_register.address,
132                });
133            } else {
134                unreachable!(
135                    "ICE: EcdsaSecp256r1 expects four array arguments and one register result"
136                )
137            }
138        }
139
140        BlackBoxFunc::MultiScalarMul => {
141            if let (
142                [BrilligVariable::BrilligArray(points), BrilligVariable::BrilligArray(scalars)],
143                [BrilligVariable::BrilligArray(outputs)],
144            ) = (function_arguments, function_results)
145            {
146                let points = brillig_context.codegen_brillig_array_to_heap_array(*points);
147                let scalars = brillig_context.codegen_brillig_array_to_heap_array(*scalars);
148                let outputs = brillig_context.codegen_brillig_array_to_heap_array(*outputs);
149
150                brillig_context.black_box_op_instruction(BlackBoxOp::MultiScalarMul {
151                    points: *points,
152                    scalars: *scalars,
153                    outputs: *outputs,
154                });
155            } else {
156                unreachable!(
157                    "ICE: MultiScalarMul expects two register arguments and one array result"
158                )
159            }
160        }
161        BlackBoxFunc::EmbeddedCurveAdd => {
162            if let (
163                [
164                    BrilligVariable::SingleAddr(input1_x),
165                    BrilligVariable::SingleAddr(input1_y),
166                    BrilligVariable::SingleAddr(input1_infinite),
167                    BrilligVariable::SingleAddr(input2_x),
168                    BrilligVariable::SingleAddr(input2_y),
169                    BrilligVariable::SingleAddr(input2_infinite),
170                ],
171                [BrilligVariable::BrilligArray(result_array)],
172            ) = (function_arguments, function_results)
173            {
174                let result = brillig_context.codegen_brillig_array_to_heap_array(*result_array);
175
176                brillig_context.black_box_op_instruction(BlackBoxOp::EmbeddedCurveAdd {
177                    input1_x: input1_x.address,
178                    input1_y: input1_y.address,
179                    input1_infinite: input1_infinite.address,
180                    input2_x: input2_x.address,
181                    input2_y: input2_y.address,
182                    input2_infinite: input2_infinite.address,
183                    result: *result,
184                });
185            } else {
186                unreachable!(
187                    "ICE: EmbeddedCurveAdd expects four register arguments and one array result"
188                )
189            }
190        }
191        BlackBoxFunc::AND => {
192            unreachable!("ICE: `BlackBoxFunc::AND` calls should be transformed into a `BinaryOp`")
193        }
194        BlackBoxFunc::XOR => {
195            unreachable!("ICE: `BlackBoxFunc::XOR` calls should be transformed into a `BinaryOp`")
196        }
197        BlackBoxFunc::RANGE => unreachable!(
198            "ICE: `BlackBoxFunc::RANGE` calls should be transformed into a `Instruction::Cast`"
199        ),
200        BlackBoxFunc::RecursiveAggregation => unreachable!(
201            "ICE: `BlackBoxFunc::RecursiveAggregation` calls are disallowed in Brillig"
202        ),
203        BlackBoxFunc::Poseidon2Permutation => {
204            if let (
205                [BrilligVariable::BrilligArray(message)],
206                [BrilligVariable::BrilligArray(result_array)],
207            ) = (function_arguments, function_results)
208            {
209                let message_array = brillig_context.codegen_brillig_array_to_heap_array(*message);
210                let output_heap_array =
211                    brillig_context.codegen_brillig_array_to_heap_array(*result_array);
212
213                brillig_context.black_box_op_instruction(BlackBoxOp::Poseidon2Permutation {
214                    message: *message_array,
215                    output: *output_heap_array,
216                });
217            } else {
218                unreachable!(
219                    "ICE: Poseidon2Permutation expects one array argument, a length and one array result"
220                )
221            }
222        }
223        BlackBoxFunc::Sha256Compression => {
224            if let (
225                [
226                    BrilligVariable::BrilligArray(input_array),
227                    BrilligVariable::BrilligArray(hash_values),
228                ],
229                [BrilligVariable::BrilligArray(result_array)],
230            ) = (function_arguments, function_results)
231            {
232                let input = brillig_context.codegen_brillig_array_to_heap_array(*input_array);
233                let hash_values = brillig_context.codegen_brillig_array_to_heap_array(*hash_values);
234                let output = brillig_context.codegen_brillig_array_to_heap_array(*result_array);
235
236                brillig_context.black_box_op_instruction(BlackBoxOp::Sha256Compression {
237                    input: *input,
238                    hash_values: *hash_values,
239                    output: *output,
240                });
241            } else {
242                unreachable!("ICE: Sha256Compression expects two array argument, one array result")
243            }
244        }
245        BlackBoxFunc::AES128Encrypt => {
246            if let (
247                [
248                    BrilligVariable::BrilligArray(inputs),
249                    BrilligVariable::BrilligArray(iv),
250                    BrilligVariable::BrilligArray(key),
251                ],
252                [BrilligVariable::BrilligArray(outputs)],
253            ) = (function_arguments, function_results)
254            {
255                let inputs = brillig_context.codegen_brillig_array_to_heap_array(*inputs);
256                let iv = brillig_context.codegen_brillig_array_to_heap_array(*iv);
257                let key = brillig_context.codegen_brillig_array_to_heap_array(*key);
258                let outputs_array = brillig_context.codegen_brillig_array_to_heap_array(*outputs);
259
260                brillig_context.black_box_op_instruction(BlackBoxOp::AES128Encrypt {
261                    inputs: *inputs,
262                    iv: *iv,
263                    key: *key,
264                    outputs: *outputs_array,
265                });
266            } else {
267                unreachable!("ICE: AES128Encrypt expects three array arguments, one array result")
268            }
269        }
270    }
