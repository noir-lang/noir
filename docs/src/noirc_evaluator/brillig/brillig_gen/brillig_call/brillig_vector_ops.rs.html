12impl<Registers: RegisterAllocator> BrilligBlock<'_, Registers> {
13    /// Take a list of [BrilligVariable] and copy the memory they point at to the `write_pointer`,
14    /// increasing the address by 1 between each variable.
15    fn write_variables(&mut self, write_pointer: MemoryAddress, variables: &[BrilligVariable]) {
16        for (index, variable) in variables.iter().enumerate() {
17            self.brillig_context.store_instruction(write_pointer, variable.extract_register());
18            if index != variables.len() - 1 {
19                self.brillig_context.codegen_usize_op_in_place(
20                    write_pointer,
21                    BrilligBinaryOp::Add,
22                    1,
23                );
24            }
25        }
26    }
27
28    /// Prepare a vector for pushing a number of new (flattened) items to the back,
29    /// then write those variables to the returned write pointer.
30    pub(crate) fn vector_push_back_operation(
31        &mut self,
32        target_vector: BrilligVector,
33        source_len: SingleAddrVariable,
34        source_vector: BrilligVector,
35        variables_to_insert: &[BrilligVariable],
36    ) {
37        let write_pointer = self.brillig_context.allocate_register();
38        self.brillig_context.call_prepare_vector_push_procedure(
39            source_len,
40            source_vector,
41            target_vector,
42            *write_pointer,
43            variables_to_insert.len(),
44            true,
45        );
46
47        self.write_variables(*write_pointer, variables_to_insert);
48    }
49
50    /// Prepare a vector for pushing a number of new (flattened) items to the front,
51    /// then write those variables to the returned write pointer.
52    pub(crate) fn vector_push_front_operation(
53        &mut self,
54        target_vector: BrilligVector,
55        source_len: SingleAddrVariable,
56        source_vector: BrilligVector,
57        variables_to_insert: &[BrilligVariable],
58    ) {
59        let write_pointer = self.brillig_context.allocate_register();
60        self.brillig_context.call_prepare_vector_push_procedure(
61            source_len,
62            source_vector,
63            target_vector,
64            *write_pointer,
65            variables_to_insert.len(),
66            false,
67        );
68
69        self.write_variables(*write_pointer, variables_to_insert);
70    }
71
72    /// Read the memory starting at the pointer into successive variables.
73    fn read_variables(&mut self, read_pointer: MemoryAddress, variables: &[BrilligVariable]) {
74        for (index, variable) in variables.iter().enumerate() {
75            self.brillig_context.load_instruction(variable.extract_register(), read_pointer);
76            if index != variables.len() - 1 {
77                self.brillig_context.codegen_usize_op_in_place(
78                    read_pointer,
79                    BrilligBinaryOp::Add,
80                    1,
81                );
82            }
83        }
84    }
85
86    /// Read the popped number of variables from the front of the source vector,
87    /// then create the target vector from the source by skipping the number of popped items.
88    pub(crate) fn vector_pop_front_operation(
89        &mut self,
90        target_vector: BrilligVector,
91        source_len: SingleAddrVariable,
92        source_vector: BrilligVector,
93        removed_items: &[BrilligVariable],
94    ) {
95        let read_pointer = self.brillig_context.codegen_make_vector_items_pointer(source_vector);
96        self.read_variables(*read_pointer, removed_items);
97
98        self.brillig_context.call_vector_pop_front_procedure(
99            source_len,
100            source_vector,
101            target_vector,
102            removed_items.len(),
103        );
104    }
105
106    /// Create new vector with a number of (flattened) items popped from the back,
107    /// then read the popped items into the variables representing the removed items.
108    pub(crate) fn vector_pop_back_operation(
109        &mut self,
110        target_vector: BrilligVector,
111        source_len: SingleAddrVariable,
112        source_vector: BrilligVector,
113        removed_items: &[BrilligVariable],
114    ) {
115        let read_pointer = self.brillig_context.allocate_register();
116        self.brillig_context.call_vector_pop_back_procedure(
117            source_len,
118            source_vector,
119            target_vector,
120            *read_pointer,
121            removed_items.len(),
122        );
123
124        self.read_variables(*read_pointer, removed_items);
125    }
126
127    /// Prepare a vector for inserting a number of (flattened) items at a specific index
128    /// by making a hole for them, then write the variables to the returned write pointer.
129    pub(crate) fn vector_insert_operation(
130        &mut self,
131        target_vector: BrilligVector,
132        source_vector: BrilligVector,
133        index: SingleAddrVariable,
134        items: &[BrilligVariable],
135    ) {
136        let write_pointer = self.brillig_context.allocate_register();
137
138        self.brillig_context.call_prepare_vector_insert_procedure(
139            source_vector,
140            target_vector,
141            index,
142            *write_pointer,
143            items.len(),
144        );
145
146        self.write_variables(*write_pointer, items);
147    }
148
149    /// Read a number of (flattened) items at a specific index of a vector into the variables
150    /// representing the removed items, then create a new vector with the same number of
151    /// items removed and subsequent items shifted to the left.
152    ///
153    /// # Safety
154    ///
155    /// The pointer addition (`read_pointer + index`) cannot overflow because:
156    /// 1. `index` is bounds-checked at SSA level ensuring `index < length`
157    /// 2. The vector allocation is protected by FMP's checked addition
158    /// 3. Therefore `items_pointer + index < items_pointer + length <= vector_end < 2^32`
159    pub(crate) fn vector_remove_operation(
160        &mut self,
161        target_vector: BrilligVector,
162        source_vector: BrilligVector,
163        index: SingleAddrVariable,
164        removed_items: &[BrilligVariable],
165    ) {
166        let read_pointer = self.brillig_context.codegen_make_vector_items_pointer(source_vector);
167        self.brillig_context.memory_op_instruction(
168            *read_pointer,
169            index.address,
170            *read_pointer,
171            BrilligBinaryOp::Add,
172        );
173        self.read_variables(*read_pointer, removed_items);
174
175        self.brillig_context.call_vector_remove_procedure(
176            source_vector,
177            target_vector,
178            index,
179            removed_items.len(),
180        );
181    }
182}

736        ) {
737            let arguments = vec![
738                BrilligParameter::Vector(

740                    SemanticLength(array.len() as u32),

745            assert_eq!(result_length, expected_array.len());
746            let result_length_with_metadata = result_length + 2; // Leading length and capacity
747
748            let returns = vec![
749                BrilligParameter::SingleAddr(BRILLIG_MEMORY_ADDRESSING_BIT_SIZE),
750                BrilligParameter::Array(
751                    vec![BrilligParameter::SingleAddr(BRILLIG_MEMORY_ADDRESSING_BIT_SIZE)],
752                    SemanticLength(result_length_with_metadata as u32),
753                ),
754            ];
755
756            let (_, mut function_context, mut context) = create_test_environment();
757
758            // Allocate the parameters
759            let source_vector = context.allocate_brillig_vector();
760            let index_to_insert = context.allocate_single_addr_mem();
761
762            // Allocate the results
763            let target_vector = context.allocate_brillig_vector();
764            let removed_item = context.allocate_single_addr_mem();
765
766            let brillig_globals = HashMap::default();
767            let hoisted_globals = HashMap::default();

769                &mut function_context,
770                &mut context,
771                &brillig_globals,
772                &hoisted_globals,
773            );
774
775            block.vector_remove_operation(
776                *target_vector,
777                *source_vector,
778                *index_to_insert,
779                &[removed_item.to_var()],
780            );
781
782            context.codegen_return(&[removed_item.to_var(), target_vector.to_var()]);

784            let calldata: Vec<_> = array.into_iter().chain(vec![index]).collect();
