1//! Register spill management for the Brillig code generator.
2//!
3//! # When spilling kicks in
4//!
5//! Most functions fit comfortably inside the per-frame register budget, in which case the
6//! code generator just uses coalescing to reuse registers and never constructs a
7//! [`SpillManager`]. Spilling activates only when [`VariableLiveness::max_live_count`]
8//! exceeds the available registers for the current [`LayoutConfig`] — i.e. there is some
9//! program point where too many values are simultaneously live. When the spill manager is
10//! active coalescing is disabled: the two are mutually exclusive allocation strategies.
11//!
12//! # Scope of decisions
13//!
14//! The spill manager operates one basic block at a time. Inside a block, spilled values can
15//! be reloaded into any free register and their heap slot re-used by a later spill, so the
16//! manager effectively acts as a pool allocator for _transient_ spills. Across blocks this
17//! is unsafe — the next block can't predict which register a predecessor left a value in —
18//! so live-across-block values (block parameters, non-param live-ins) are promoted to
19//! _permanent_ spills: their slot is reserved for the remainder of the function and every
20//! use reloads from it.
21//!
22//! The manager owns only this *slot* bookkeeping plus an LRU eviction order. Whether a value is
23//! currently in a register is owned by the register allocator ([`BlockVariables`]) and read
24//! through the [`RegisterState`] trait; "currently spilled" is *derived* as having a slot while
25//! not in a register, never stored.
26//!
27//! Transient slots are freed as soon as the value dies; permanent slots are never freed
28//! (see <https://github.com/noir-lang/noir/issues/11695>). Each block begins with
29//! [`SpillManager::begin_block`], which (a) asserts no live-in value carries a transient slot
30//! (values living across a boundary must be permanently spilled), and (b) rebuilds the LRU from
31//! this block's live-in set. It consults no [`RegisterState`]: the block's registers are reset
32//! by the caller immediately afterwards, so a permanently-spilled value is automatically "not in
33//! a register" at entry.
34//!
35//! # Per-use flow
36//!
37//! The code generator calls into the spill manager at a small number of well-defined points:
38//!
39//! - Before making room for a fresh value, [`BrilligBlock::ensure_register_capacity`] asks
40//!   [`SpillManager::lru_victim`] for the least-recently-used value (the LRU only ever holds
41//!   register-resident values) and spills it via [`BrilligBlock::spill_value`]. That emits the
42//!   store, records the slot, and frees the register.
43//! - When a spilled value is needed, [`BrilligBlock::reload_spilled_value`] allocates a fresh
44//!   register and emits the load; re-adding the value to the register set is what marks it no
45//!   longer spilled (the slot is kept, since the load may re-execute in a loop iteration).
46//! - When a value is used, [`SpillManager::touch`] bumps it to the most-recently-used end of
47//!   the LRU, so that freshly-loaded and just-produced values aren't the first victims of the
48//!   next eviction. `touch` also asserts the value is in a register, keeping spilled values out
49//!   of the LRU at the source.
50//! - When a value dies, [`SpillManager::remove_spill`] frees its transient slot (permanent slots
51//!   are kept for the rest of the function) and drops it from the LRU.
52//!
53//! [`VariableLiveness::max_live_count`]: super::variable_liveness::VariableLiveness::max_live_count
54//! [`LayoutConfig`]: crate::brillig::brillig_ir::LayoutConfig
55//! [`BlockVariables`]: super::brillig_block_variables::BlockVariables
56//! [`BrilligBlock::ensure_register_capacity`]: super::brillig_block::BrilligBlock::ensure_register_capacity
57//! [`BrilligBlock::spill_value`]: super::brillig_block::BrilligBlock::spill_value
58//! [`BrilligBlock::reload_spilled_value`]: super::brillig_block::BrilligBlock::reload_spilled_value

68/// The register-allocation state the spill manager consults but does not own.
69///
70/// The manager tracks which values have a heap slot and whether that slot is transient or
71/// permanent; whether a value is *currently* in a register is owned elsewhere — by
72/// [`BlockVariables`] in real codegen, or by a mock in tests — and injected through this trait.
73/// Together they answer "is this value currently spilled": it has a slot and is not in a
74/// register (see [`SpillManager::is_spilled`]).
75///
76/// [`BlockVariables`]: super::brillig_block_variables::BlockVariables
77pub(crate) trait RegisterState {
78    /// Whether `value_id` currently occupies a register.
79    fn is_in_register(&self, value_id: &ValueId) -> bool;
80}

82/// Tracks register values that have been spilled to the spill region in heap memory.
83///
84/// The manager owns only *slot* bookkeeping (which values have a heap slot, and whether that
85/// slot is transient or permanent) plus the LRU. It deliberately does not track whether a value
86/// is currently in a register — that is owned by the register allocator and reached through
87/// [`RegisterState`]. "Currently spilled" is derived: `has_slot(v) && !is_in_register(v)`.
88///
89/// See the [module docs][self] for an overview of when and how the spill manager is used.
90pub(crate) struct SpillManager {
91    /// Map of all spill records
92    records: HashMap<ValueId, SpillRecord>,
93    /// Least-recently-used tracker over local SSA values (not globals, not hoisted constants).
94    lru: Lru,
95    /// Next offset within the spill region (relative to spill base register).
96    next_spill_offset: usize,
97    /// Free list of spill slots that have been reclaimed.
98    free_spill_slots: Vec<usize>,
99    /// The maximum number of spill slots needed across all blocks in this function.
100    max_spill_offset: usize,
101}
102
103/// A heap slot reserved for a spilled value.
104///
105/// # Value lifecycle
106///
107/// A value's state is the combination of its *slot* — none, transient, or permanent, tracked
108/// here — and whether it is *currently in a register*, which is owned by the register allocator
109/// ([`RegisterState`]) and not stored here. A value is "spilled" when it has a slot but is not
110/// in a register. There are five combinations; a `SpillRecord` exists only for the four with a
111/// slot:
112///
113/// | slot      | in register | meaning                                                     |
114/// |-----------|-------------|-------------------------------------------------------------|
115/// | none      | yes         | normal value — lives only in a register, has no record      |
116/// | transient | no          | spilled within the block, awaiting reload                   |
117/// | transient | yes         | reloaded, but the transient slot is still reserved          |
118/// | permanent | no          | spilled across blocks; the heap slot is the source of truth |
119/// | permanent | yes         | reloaded, but the permanent slot is still authoritative     |
120///
121/// Movement between rows:
122///
123/// - **Spill** (registers run out): a value in a register gains a slot and its register is
124///   freed. Within a block it gets a *transient* slot (`record_spill`); a value that must
125///   survive to another block gets a *permanent* slot instead (`record_permanent_spill`).
126/// - **Reload** (a spilled value is used again): it is loaded back into a register, moving to
127///   the "in register" row of the *same* slot kind. The slot is kept — the emitted load may
128///   re-run in a loop iteration — so nothing is re-allocated.
129/// - **Re-eviction**: a reloaded value the LRU picks again has its register freed, dropping back
130///   to the "not in register" row and reusing its existing slot.
131/// - **Promotion** (`ensure_permanent_spill`): a transient value that turns out to live across a
132///   block boundary is promoted to a permanent slot (transient → permanent, same residence).
133/// - **Block boundary**: the register file is reset, so every value is momentarily "not in a
134///   register" — the reloaded rows collapse onto their spilled rows. Transient slots must not
135///   survive for a live-in value (`begin_block` asserts this); permanent slots persist as the
136///   cross-block source of truth.
137/// - **Death** (`remove_spill`): at a value's last use a transient slot is freed and the value
138///   leaves the table, while a permanent slot is kept for the rest of the function.
139#[derive(Clone, Copy)]
140pub(crate) struct SpillRecord {
141    /// Offset relative to the per-frame heap-allocated spill region base.
142    pub(crate) offset: usize,
143    /// Original variable (for type/register info on reload).
144    pub(crate) variable: BrilligVariable,
145    /// Whether the slot is permanent (reserved for the rest of the function, the source of
146    /// truth across block boundaries) rather than transient (freed when the value dies).
147    permanent: bool,
148}
149
150/// Least-recently-used tracker over local SSA values, keyed by a monotonic touch clock.
151///
152/// Backs the eviction heuristic with logarithmic `touch`/`remove` instead of the linear
153/// scans and shifts a `Vec` would require (see <https://github.com/noir-lang/noir/issues/11694>).
154/// Values are ordered by the time they were last touched: the least-recently-used value is
155/// the first element of `order`, the most-recently-used the last.
156///
157/// Invariant: the LRU only ever holds register-resident values, so `SpillManager::lru_victim`
158/// never hands back a spilled one. It is guarded at both ends: `SpillManager::touch` (the only
159/// way a value enters) asserts the value is in a register, and every method that moves a value
160/// out of a register — the slot-recording methods (`record_spill`, `record_permanent_spill`,
161/// `ensure_permanent_spill`) and `remove_spill` — drops it from the LRU. `lru_victim` re-checks
162/// this as a backstop, so a future spill path that forgets to remove trips immediately rather
163/// than silently dishing out a spilled value.
164#[derive(Default)]
165struct Lru {
166    /// Logical clock, incremented once per `touch`. Strictly increasing, so no two entries
167    /// ever share a key and the ordering is total and deterministic.
168    clock: u64,
169    /// Last touch time of each tracked value.
170    last_used: HashMap<ValueId, u64>,
171    /// Tracked values ordered by touch time (ascending): first = least recently used.
172    order: BTreeSet<(u64, ValueId)>,
173}
174
175impl Lru {
176    fn new() -> Self {
177        Self::default()
178    }
179
180    /// Mark `value_id` as the most recently used value, inserting it if not already tracked.
181    fn touch(&mut self, value_id: ValueId) {
182        self.clock += 1;
183        if let Some(previous) = self.last_used.insert(value_id, self.clock) {
184            self.order.remove(&(previous, value_id));
185        }
186        self.order.insert((self.clock, value_id));
187    }
188
189    /// Stop tracking `value_id` entirely.
190    fn remove(&mut self, value_id: &ValueId) {
191        if let Some(previous) = self.last_used.remove(value_id) {
192            self.order.remove(&(previous, *value_id));
193        }
194    }
195
196    /// Iterate tracked values from least- to most-recently used.
197    fn iter(&self) -> impl Iterator<Item = ValueId> + '_ {
198        self.order.iter().map(|&(_, value_id)| value_id)
199    }
200
201    /// Forget all tracked values. The clock is left untouched so it stays monotonic.
202    fn clear(&mut self) {
203        self.last_used.clear();
204        self.order.clear();
205    }
206
207    /// Insert `values` as equally — and least — recently used (one shared clock tick).
208    ///
209    /// Because `order` is keyed by `(clock, value_id)`, entries sharing a clock are ordered by
210    /// [`ValueId`], so the caller need not pre-sort to get a deterministic ordering.
211    fn seed(&mut self, values: impl IntoIterator<Item = ValueId>) {
212        self.clock += 1;
213        let clock = self.clock;
214        for value_id in values {
215            if let Some(previous) = self.last_used.insert(value_id, clock) {
216                self.order.remove(&(previous, value_id));
217            }
218            self.order.insert((clock, value_id));
219        }
220    }
221}

223impl SpillManager {
224    pub(crate) fn new() -> Self {
225        Self {
226            records: HashMap::default(),
227            lru: Lru::new(),
228            next_spill_offset: 0,
229            free_spill_slots: Vec::new(),
230            max_spill_offset: 0,
231        }
232    }
233
234    /// Prepare spill state for a new block.
235    ///
236    /// 1. Asserts that no *live-in* value carries a transient slot. A value that lives across a
237    ///    block boundary must be permanently spilled (transient slots are within-block), so a
238    ///    transient slot for a live-in value signals a leak. A leftover transient slot for a
239    ///    value that is *not* live-in (e.g. one that died at the terminator after being
240    ///    reloaded for it) is a benign within-block artifact and is tolerated.
241    /// 2. Removes permanently-spilled values from the live-in set: the register state is reset
242    ///    at block entry, so these must be reloaded before use.
243    /// 3. Rebuilds the LRU from the surviving live-in set.
244    ///
245    /// All three steps run *before* the block's register state exists — `BrilligBlock::compile_block`
246    /// builds the new `BlockVariables` and resets the registers only after `begin_block` returns
247    /// — so they decide purely from slot state and the live-in set, never from [`RegisterState`]
248    /// (which has no valid value to consult here).
249    pub(crate) fn begin_block(&mut self, live_in: &mut HashSet<ValueId>) {
250        // A live-in value must be permanently spilled; only a transient slot for one is a leak.
251        assert!(
252            !self.records.iter().any(|(v, r)| !r.permanent && live_in.contains(v)),
253            "Transient spill leaked across block boundary: {:?}",
254            self.records
255                .iter()
256                .filter(|(v, r)| !r.permanent && live_in.contains(v))
257                .map(|(v, _)| *v)
258                .collect::<Vec<_>>()
259        );
260        live_in.retain(|v| !self.has_permanent_slot(v));
261        self.reset_lru_for_block(live_in);
262    }
263
264    /// Whether a value is currently spilled: it has a heap slot but is not in a register.
265    pub(crate) fn is_spilled(&self, value_id: &ValueId, registers: &impl RegisterState) -> bool {
266        self.has_slot(value_id) && !registers.is_in_register(value_id)
267    }
268
269    /// Whether a value was transiently spilled and has since been reloaded into a register.
270    pub(crate) fn is_transient_reloaded(
271        &self,
272        value_id: &ValueId,
273        registers: &impl RegisterState,
274    ) -> bool {
275        self.has_transient_slot(value_id) && registers.is_in_register(value_id)
276    }
277
278    /// Whether a value has any heap slot (transient or permanent).
279    fn has_slot(&self, value_id: &ValueId) -> bool {
280        self.records.contains_key(value_id)
281    }
282
283    /// Whether a value has a transient heap slot.
284    fn has_transient_slot(&self, value_id: &ValueId) -> bool {
285        self.records.get(value_id).is_some_and(|r| !r.permanent)
286    }
287
288    /// Whether a value has a permanent heap slot.
289    pub(crate) fn has_permanent_slot(&self, value_id: &ValueId) -> bool {
290        self.records.get(value_id).is_some_and(|r| r.permanent)
291    }
292
293    /// Move a value to the back of the LRU (most recently used), inserting it if absent.
294    ///
295    /// The value must currently be in a register: the LRU only ever holds register-resident
296    /// values (a definition or a reload is the only way one enters). Asserting that here keeps a
297    /// spilled value from ever entering the LRU, which is what lets [`Self::lru_victim`] hand back
298    /// its oldest entry without re-checking — a spilled value can only get in through an erroneous
299    /// touch, and this catches it at the source.
300    pub(crate) fn touch(&mut self, value_id: ValueId, registers: &impl RegisterState) {
301        assert!(
302            registers.is_in_register(&value_id),
303            "touched {value_id} which is not in a register"
304        );
305        self.lru.touch(value_id);
306    }
307
308    /// Remove a value from LRU tracking entirely.
309    fn remove_from_lru(&mut self, value_id: &ValueId) {
310        self.lru.remove(value_id);
311    }
312
313    /// Remove a (dead) value from spill tracking and from the LRU.
314    ///
315    /// A transient slot is reclaimed to the free list. A permanent slot is never freed — it must
316    /// remain valid across all blocks — so the record is kept. Either way the value is dropped
317    /// from the LRU, since a value that is gone must never be an eviction candidate.
318    ///
319    /// TODO(<https://github.com/noir-lang/noir/issues/11695>) - Free globally dead permanent spill slots
320    pub(crate) fn remove_spill(&mut self, value_id: &ValueId) {
321        // A transient slot is reclaimed; a permanent slot is kept (never freed).
322        if let Entry::Occupied(entry) = self.records.entry(*value_id)
323            && !entry.get().permanent
324        {
325            let record = entry.remove();
326            self.free_spill_slots.push(record.offset);
327        }
328        self.remove_from_lru(value_id);
329    }
330
331    /// Allocate a spill slot offset, reusing a freed slot if available.
332    pub(crate) fn allocate_spill_offset(&mut self) -> usize {
333        let offset = if let Some(offset) = self.free_spill_slots.pop() {
334            offset
335        } else {
336            let offset = self.next_spill_offset;
337            self.next_spill_offset += 1;
338            offset
339        };
340        self.max_spill_offset = self.max_spill_offset.max(offset + 1);
341        offset
342    }
343
344    /// The maximum number of spill slots used across all blocks.
345    pub(crate) fn max_spill_offset(&self) -> usize {
346        self.max_spill_offset
347    }
348
349    /// Return the least recently used value tracked in the LRU, or `None` if it is empty.
350    ///
351    /// A single-victim [`Self::lru_victims`], so the same spilled-value backstop applies.
352    pub(crate) fn lru_victim(&self, registers: &impl RegisterState) -> Option<ValueId> {
353        self.lru_victims(1, registers).first().copied()
354    }
355
356    /// Return the `k` least-recently-used values, oldest first.
357    ///
358    /// The LRU only holds register-resident values (see the [`Lru`] invariant), so these are all
359    /// valid eviction candidates. The `assert!` is a backstop: it re-checks that we never hand
360    /// back a spilled value, catching any future spill path that fails to drop it from the LRU.
361    pub(crate) fn lru_victims(&self, k: usize, registers: &impl RegisterState) -> Vec<ValueId> {
362        let victims: Vec<ValueId> = self.lru.iter().take(k).collect();
363        for victim in &victims {
364            assert!(
365                !self.is_spilled(victim, registers),
366                "lru_victim returned a spilled value: {victim:?}"
367            );
368        }
369        victims
370    }
371
372    /// Record that a value has been spilled to a transient slot, dropping it from the LRU since
373    /// a value living in its heap slot must never be an eviction candidate.
374    ///
375    /// If the value already has a slot (e.g. it was reloaded and is now being re-evicted), the
376    /// existing slot is kept — including its kind, so a permanent slot stays permanent — and only
377    /// the variable is refreshed.
378    ///
379    /// # Precondition
380    /// `value_id` must still be in a register when this is called: the caller frees its register
381    /// only *after* recording the spill. That ordering is what makes the double-spill assert
382    /// meaningful — an already-spilled value (has a slot and is not in a register) trips it, while
383    /// a legitimate re-eviction of a reloaded value passes because it is still in a register here.
384    pub(crate) fn record_spill(
385        &mut self,
386        value_id: ValueId,
387        offset: usize,
388        variable: BrilligVariable,
389        registers: &impl RegisterState,
390    ) {
391        assert!(!self.is_spilled(&value_id, registers), "Double-spill of {value_id}");
392        match self.records.entry(value_id) {
393            Entry::Occupied(mut entry) => {
394                // Always preserve the offset, so we don't leak free slots.
395                assert_eq!(
396                    offset,
397                    entry.get().offset,
398                    "Spill of {value_id} orphaned existing slot"
399                );
400                entry.get_mut().variable = variable;
401            }
402            Entry::Vacant(entry) => {
403                entry.insert(SpillRecord { offset, variable, permanent: false });
404            }
405        }
406        self.remove_from_lru(&value_id);
407    }
408
409    /// Get the spill record for a value if it is currently spilled (has a slot and is not in a
410    /// register).
411    pub(crate) fn get_spill(
412        &self,
413        value_id: &ValueId,
414        registers: &impl RegisterState,
415    ) -> Option<&SpillRecord> {
416        self.records.get(value_id).filter(|_| !registers.is_in_register(value_id))
417    }
418
419    /// Rebuild the LRU for a new block from the surviving live-in set.
420    ///
421    /// Called only from [`Self::begin_block`], after permanently-spilled values have been removed
422    /// from `live_in`. Like `begin_block`, it runs before the block's register state exists, so it
423    /// is register-free.
424    ///
425    /// The `debug_assert!` is a trip-wire: today every value live across a block boundary is
426    /// permanently spilled, so nothing live-in survives in the carried-over LRU. If a future
427    /// register allocator (<https://github.com/noir-lang/noir/issues/11638>) keeps live values in
428    /// registers across blocks it fires — a signal that re-seeding by `ValueId` discards a real
429    /// previous-block recency ordering we should be preserving instead.
430    fn reset_lru_for_block(&mut self, live_in: &HashSet<ValueId>) {
431        debug_assert!(
432            !self.lru.iter().any(|v| live_in.contains(&v)),
433            "cross-block LRU carryover is non-empty; preserve the previous block's order instead of re-seeding: {:?}",
434            self.lru.iter().collect::<Vec<_>>()
435        );
436        self.lru.clear();
437        // Seed every surviving live-in as equally least-recently-used; the LRU orders them by
438        // `ValueId`, so no explicit sort is needed.
439        self.lru.seed(live_in.iter().copied());
440    }
441
442    /// Record a value as permanently spilled, dropping it from the LRU (like [`Self::record_spill`]).
443    ///
444    /// If the value already has a transient spill slot, it is promoted to permanent.
445    /// The permanent slot ensures consistency regardless of what happens during block processing.
446    pub(crate) fn record_permanent_spill(
447        &mut self,
448        value_id: ValueId,
449        offset: usize,
450        variable: BrilligVariable,
451    ) {
452        self.records
453            .entry(value_id)
454            .and_modify(|record| {
455                assert_eq!(record.offset, offset);
456                record.permanent = true;
457                record.variable = variable;
458            })
459            .or_insert(SpillRecord { offset, variable, permanent: true });
460        self.remove_from_lru(&value_id);
461    }
462
463    /// Get the permanent spill slot offset for a value, if any.
464    pub(crate) fn get_permanent_spill_offset(&self, value_id: &ValueId) -> Option<usize> {
465        self.records.get(value_id).filter(|r| r.permanent).map(|r| r.offset)
466    }
467
468    /// Get the spill slot offset for a value, if any.
469    ///
470    /// Unlike `get_permanent_spill_offset` this could be a permanent or a transient spill;
471    /// there should be only one at any point.
472    pub(crate) fn get_spill_offset(&self, value_id: &ValueId) -> Option<usize> {
473        self.records.get(value_id).map(|r| r.offset)
474    }
475
476    /// Promote an existing spill record to a permanent slot, dropping it from the LRU.
477    ///
478    /// The permanent slot becomes the value's authoritative copy across block boundaries. This
479    /// updates slot bookkeeping only; whether the value still occupies a register is the caller's
480    /// concern — see `BrilligBlock::spill_value`.
481    ///
482    /// # Returns
483    /// * `true` — a record already existed, so the value already owns a permanent slot; the
484    ///   caller can return early without allocating a slot or emitting a store.
485    /// * `false` — no record existed: this is the value's first spill, so the caller must
486    ///   allocate a slot and record the spill itself.
487    pub(crate) fn ensure_permanent_spill(&mut self, value_id: &ValueId) -> bool {
488        let Some(record) = self.records.get_mut(value_id) else {
489            return false;
490        };
491        record.permanent = true;
492        self.remove_from_lru(value_id);
493        true
494    }
495}

516    ///
517    /// Tests drive this explicitly to mirror what `BrilligBlock` does to the real registers:
518    /// `allocate` is "a register now holds the value" (e.g. a reload, formerly `unmark_spilled`),
519    /// and `free` is "the register was released" (e.g. after a spill, or the block-entry reset
520    /// that formerly went through `restore_permanent_spills`).
521    #[derive(Default)]
522    struct MockRegisters {
523        in_register: FxHashSet<ValueId>,
524    }
525
526    impl RegisterState for MockRegisters {
527        fn is_in_register(&self, value_id: &ValueId) -> bool {
528            self.in_register.contains(value_id)
529        }
530    }
531
532    impl MockRegisters {
533        fn allocate(&mut self, value_id: ValueId) {
534            self.in_register.insert(value_id);
535        }
536
537        fn free(&mut self, value_id: ValueId) {
538            self.in_register.remove(&value_id);
539        }
540    }

761    #[test]
762    fn permanent_spill_lifecycle() {
763        let mut sm = SpillManager::new();
764        let mut regs = MockRegisters::default();
765        let v0 = Id::test_new(0);
766
767        // The value starts in a register, then is permanently spilled (register then freed).
768        regs.allocate(v0);
769        let off = sm.allocate_spill_offset();
770        sm.record_permanent_spill(v0, off, test_var(0));
771        regs.free(v0);
772        assert_eq!(off, 0);
773        assert!(sm.is_spilled(&v0, &regs));
774        assert!(sm.has_permanent_slot(&v0));
775        assert_eq!(sm.get_permanent_spill_offset(&v0), Some(0));
776
777        // Reload: a register is allocated again (was `unmark_spilled`); the slot stays.
778        regs.allocate(v0);
779        assert!(!sm.is_spilled(&v0, &regs));
780        assert!(sm.has_permanent_slot(&v0));
781        assert_eq!(sm.get_permanent_spill_offset(&v0), Some(0));
782
783        // While reloaded (in a register) the value is a normal eviction candidate.
784        sm.touch(v0, &regs);
785        assert_eq!(sm.lru_victim(&regs), Some(v0));
786
787        // Block entry resets the register state (was `restore_permanent_spills`): no longer resident.
788        regs.free(v0);
789        assert!(sm.is_spilled(&v0, &regs));
790
791        // The value dies: the permanent slot is kept, but it is dropped from the LRU.
792        sm.remove_spill(&v0);
793        assert!(sm.is_spilled(&v0, &regs));
794        // Permanent record still exists
795        assert!(sm.has_permanent_slot(&v0));
796        // Slot is NOT freed (no slot in free list)
797        assert!(sm.free_spill_slots.is_empty());
798        // The dead value was dropped from the LRU, so it is no longer an eviction candidate.
799        assert_eq!(sm.lru_victim(&regs), None);
800    }
