1//! Register spill management for the Brillig code generator.
2//!
3//! # When spilling kicks in
4//!
5//! Most functions fit comfortably inside the per-frame register budget, in which case the
6//! code generator just uses coalescing to reuse registers and never constructs a
7//! [`SpillManager`]. Spilling activates only when [`VariableLiveness::max_live_count`]
8//! exceeds the available registers for the current [`LayoutConfig`] — i.e. there is some
9//! program point where too many values are simultaneously live. When the spill manager is
10//! active coalescing is disabled: the two are mutually exclusive allocation strategies.
11//!
12//! # Scope of decisions
13//!
14//! The spill manager operates one basic block at a time. Inside a block, spilled values can
15//! be reloaded into any free register and their heap slot re-used by a later spill, so the
16//! manager effectively acts as a pool allocator for _transient_ spills. Across blocks this
17//! is unsafe — the next block can't predict which register a predecessor left a value in —
18//! so live-across-block values (block parameters, non-param live-ins) are promoted to
19//! _permanent_ spills: their slot is reserved for the remainder of the function and every
20//! use reloads from it.
21//!
22//! Transient slots are freed as soon as the value dies; permanent slots are never freed
23//! (see <https://github.com/noir-lang/noir/issues/11695>). Each block begins with
24//! [`SpillManager::begin_block`], which (a) asserts no transient spills leaked from the
25//! previous block, (b) re-marks permanent slots as "currently spilled" so their owners
26//! must reload before use, and (c) rebuilds the LRU with this block's live-in set.
27//!
28//! # Per-use flow
29//!
30//! The code generator calls into the spill manager at a small number of well-defined points:
31//!
32//! - Before making room for a fresh value, [`BrilligBlock::ensure_register_capacity`] asks
33//!   [`SpillManager::lru_victim`] for the least-recently-used non-spilled value and spills
34//!   it via [`BrilligBlock::spill_value`]. That emits the store and frees the register.
35//! - When a spilled value is needed, [`BrilligBlock::reload_spilled_value`] allocates a
36//!   fresh register, emits the load, and calls [`SpillManager::unmark_spilled`] to record
37//!   that the value is once again in a register (the slot still holds a valid copy).
38//! - When a value is used, [`SpillManager::touch`] bumps it to the most-recently-used end
39//!   of the LRU, so that freshly-loaded and just-produced values aren't the first victims
40//!   of the next eviction.
41//! - When a value dies, [`SpillManager::remove_spill`] frees its transient slot (permanent
42//!   slots are kept for the rest of the function).
43//!
44//! [`VariableLiveness::max_live_count`]: super::variable_liveness::VariableLiveness::max_live_count
45//! [`LayoutConfig`]: crate::brillig::brillig_ir::LayoutConfig
46//! [`BrilligBlock::ensure_register_capacity`]: super::brillig_block::BrilligBlock::ensure_register_capacity
47//! [`BrilligBlock::spill_value`]: super::brillig_block::BrilligBlock::spill_value
48//! [`BrilligBlock::reload_spilled_value`]: super::brillig_block::BrilligBlock::reload_spilled_value
49

57/// Lifecycle state of a spill record.
58///
59/// Transitions:
60/// - First eviction → `Transient` or `Permanent`
61/// - Reloaded into a register → `TransientReloaded` or `PermanentReloaded`
62/// - Evicted again by LRU → back to `Transient` or `Permanent`
63/// - Block boundary → `PermanentReloaded` becomes `Permanent`; `Transient` must not survive
64#[derive(Clone, Copy, PartialEq, Eq, Debug)]
65pub(crate) enum SpillStatus {
66    /// Within-block spill: value is in the spill slot, not in a register.
67    Transient,
68    /// Was transiently spilled then reloaded into a register.
69    /// The spill slot is still allocated and the register holds the live value.
70    TransientReloaded,
71    /// Permanent spill: the slot is the source of truth across block boundaries.
72    /// Value is not currently in a register.
73    Permanent,
74    /// Permanently spilled, but currently also loaded into a register.
75    /// The slot remains authoritative; the register is a transient copy.
76    PermanentReloaded,
77}

79/// Tracks register values that have been spilled to the spill region in heap memory.
80///
81/// See the [module docs][self] for an overview of when and how the spill manager is used.
82pub(crate) struct SpillManager {
83    /// Map of all spill records
84    records: HashMap<ValueId, SpillRecord>,
85    /// LRU order: front = least recently used, back = most recently used.
86    /// Only tracks local SSA values (not globals, not hoisted constants).
87    lru_order: Vec<ValueId>,
88    /// Next offset within the spill region (relative to spill base register).
89    next_spill_offset: usize,
90    /// Free list of spill slots that have been reclaimed.
91    free_spill_slots: Vec<usize>,
92    /// The maximum number of spill slots needed across all blocks in this function.
93    max_spill_offset: usize,
94}
95
96#[derive(Clone, Copy)]
97pub(crate) struct SpillRecord {
98    /// Offset relative to the per-frame heap-allocated spill region base.
99    pub(crate) offset: usize,
100    /// Original variable (for type/register info on reload).
101    pub(crate) variable: BrilligVariable,
102    /// Current lifecycle state of this spill record.
103    pub(crate) status: SpillStatus,
104}
105
106impl SpillManager {
107    pub(crate) fn new() -> Self {
108        Self {
109            records: HashMap::default(),
110            lru_order: Vec::new(),
111            next_spill_offset: 0,
112            free_spill_slots: Vec::new(),
113            max_spill_offset: 0,
114        }
115    }
116
117    /// Prepare spill state for a new block.
118    ///
119    /// 1. Asserts that no transient spills leaked from the previous block
120    ///    (only `Permanent` and `PermanentReloaded` records may survive).
121    /// 2. Restores permanently-spilled values by marking them as currently spilled.
122    /// 3. Removes spilled values from the live-in set (they have no register).
123    /// 4. Updates the LRU: retains existing entries still live-in and not spilled
124    ///    (preserving eviction hints from the previous block), then appends any
125    ///    new live-in values sorted by [ValueId] for determinism.
126    pub(crate) fn begin_block(&mut self, live_in: &mut HashSet<ValueId>) {
127        // No transient spills should survive across block boundaries.
128        assert!(
129            self.records.values().all(|r| r.status != SpillStatus::Transient),
130            "Transient spill leaked across block boundary"
131        );
132        self.restore_permanent_spills();
133        live_in.retain(|v| !self.is_spilled(v));
134        self.reset_lru_for_block(live_in);
135    }
136
137    /// Check if a value is currently spilled (in memory, not in a register).
138    pub(crate) fn is_spilled(&self, value_id: &ValueId) -> bool {
139        matches!(
140            self.records.get(value_id),
141            Some(r) if matches!(r.status, SpillStatus::Transient | SpillStatus::Permanent)
142        )
143    }
144
145    /// Check if a value was transiently spilled and has since been reloaded into a register.
146    pub(crate) fn is_transient_reloaded(&self, value_id: &ValueId) -> bool {
147        matches!(self.records.get(value_id), Some(r) if r.status == SpillStatus::TransientReloaded)
148    }
149
150    /// Check if a value has a permanent spill slot.
151    #[cfg(test)]
152    pub(crate) fn has_permanent_slot(&self, value_id: &ValueId) -> bool {
153        matches!(
154            self.records.get(value_id),
155            Some(r) if matches!(r.status, SpillStatus::Permanent | SpillStatus::PermanentReloaded)
156        )
157    }
158
159    /// Move a value to the back of the LRU (most recently used).
160    /// If the value isn't tracked yet, add it.
161    pub(crate) fn touch(&mut self, value_id: ValueId) {
162        self.remove_from_lru(&value_id);
163        self.lru_order.push(value_id);
164    }
165
166    /// Remove a value from LRU tracking entirely.
167    pub(crate) fn remove_from_lru(&mut self, value_id: &ValueId) {
168        if let Some(pos) = self.lru_order.iter().position(|v| v == value_id) {
169            self.lru_order.remove(pos);
170        }
171    }
172
173    /// Remove a value from the spill tracking.
174    ///
175    /// Permanent spill slots are never freed — they must remain valid across all blocks.
176    /// For `Permanent` records, this transitions to `PermanentReloaded` (value is live in
177    /// a register again). For transient records, the record is removed and the slot is freed.
178    ///
179    /// TODO(<https://github.com/noir-lang/noir/issues/11695>) - Free globally dead permanent spill slots
180    pub(crate) fn remove_spill(&mut self, value_id: &ValueId) {
181        if let Entry::Occupied(mut entry) = self.records.entry(*value_id) {
182            match entry.get().status {
183                SpillStatus::Permanent => {
184                    entry.get_mut().status = SpillStatus::PermanentReloaded;
185                }
186                SpillStatus::PermanentReloaded => {}
187                SpillStatus::Transient | SpillStatus::TransientReloaded => {
188                    let record = entry.remove();
189                    self.free_spill_slots.push(record.offset);
190                }
191            }
192        }
193    }
194
195    /// Allocate a spill slot offset, reusing a freed slot if available.
196    pub(crate) fn allocate_spill_offset(&mut self) -> usize {
197        let offset = if let Some(offset) = self.free_spill_slots.pop() {
198            offset
199        } else {
200            let offset = self.next_spill_offset;
201            self.next_spill_offset += 1;
202            offset
203        };
204        self.max_spill_offset = self.max_spill_offset.max(offset + 1);
205        offset
206    }
207
208    /// The maximum number of spill slots used across all blocks.
209    pub(crate) fn max_spill_offset(&self) -> usize {
210        self.max_spill_offset
211    }
212
213    /// Return the least recently used value that is not currently spilled.
214    /// Returns None if there are no eligible values in the LRU.
215    ///
216    /// Note: permanently-spilled values may appear in the LRU if they were
217    /// reloaded (which calls `touch()`). This is correct — a reloaded value
218    /// has a register and is a valid eviction candidate.
219    pub(crate) fn lru_victim(&self) -> Option<ValueId> {
220        for value_id in &self.lru_order {
221            if !self.is_spilled(value_id) {
222                return Some(*value_id);
223            }
224        }
225        None
226    }
227
228    /// Record that a value has been spilled.
229    ///
230    /// If the value already has a record (e.g., it was reloaded and then
231    /// re-evicted by LRU), this updates the existing record:
232    /// - `PermanentReloaded`: transitions back to `Permanent` (slot data is still valid).
233    /// - `TransientReloaded`: transitions back to `Transient`, updating the variable.
234    pub(crate) fn record_spill(
235        &mut self,
236        value_id: ValueId,
237        offset: usize,
238        variable: BrilligVariable,
239    ) {
240        assert!(!self.is_spilled(&value_id), "Double-spill of {value_id}");
241        let record = self.records.entry(value_id).or_insert(SpillRecord {
242            offset,
243            variable,
244            status: SpillStatus::Transient,
245        });
246        // Always preserve the offset, so we don't leak free slots.
247        assert_eq!(offset, record.offset, "Spill of {value_id} orphaned existing slot");
248        match record.status {
249            SpillStatus::PermanentReloaded => {
250                // Re-evicting a permanently-spilled value: keep permanent, slot data is valid.
251                record.status = SpillStatus::Permanent;
252            }
253            SpillStatus::TransientReloaded | SpillStatus::Transient => {
254                record.status = SpillStatus::Transient;
255                record.variable = variable;
256            }
257            SpillStatus::Permanent => {
258                unreachable!(
259                    "record_spill called on a Permanent record — is_spilled should have caught this as a double-spill"
260                );
261            }
262        }
263    }
264
265    /// Get the spill record for a value if it is currently spilled.
266    pub(crate) fn get_spill(&self, value_id: &ValueId) -> Option<&SpillRecord> {
267        self.records
268            .get(value_id)
269            .filter(|r| matches!(r.status, SpillStatus::Transient | SpillStatus::Permanent))
270    }
271
272    /// Reset the LRU for a new block, retaining ordering from the previous block.
273    ///
274    /// Entries already in `lru_order` that are still live-in and not spilled are kept
275    /// in their existing order (preserving eviction hints from the previous block).
276    /// New live-in values not yet in the LRU are appended, sorted by [ValueId] for
277    /// determinism.
278    fn reset_lru_for_block(&mut self, live_in: &HashSet<ValueId>) {
279        let records = &self.records;
280        let is_spilled = |v: &ValueId| {
281            matches!(
282                records.get(v),
283                Some(r) if matches!(r.status, SpillStatus::Transient | SpillStatus::Permanent)
284            )
285        };
286
287        // Retain existing entries that are still live-in and not spilled.
288        self.lru_order.retain(|v| live_in.contains(v) && !is_spilled(v));
289
290        // Collect live-in values not already present in LRU, sorted for determinism.
291        let existing: HashSet<ValueId> = self.lru_order.iter().copied().collect();
292        let mut new_entries: Vec<ValueId> =
293            live_in.iter().copied().filter(|v| !existing.contains(v) && !is_spilled(v)).collect();
294        new_entries.sort();
295        self.lru_order.extend(new_entries);
296    }
297
298    /// Record a value as permanently spilled.
299    ///
300    /// If the value already has a transient spill record, it is promoted to `Permanent`.
301    /// The permanent record ensures consistency regardless of what happens during
302    /// block processing.
303    pub(crate) fn record_permanent_spill(
304        &mut self,
305        value_id: ValueId,
306        offset: usize,
307        variable: BrilligVariable,
308    ) {
309        self.records
310            .entry(value_id)
311            .and_modify(|record| {
312                assert_eq!(record.offset, offset);
313                record.status = SpillStatus::Permanent;
314                record.variable = variable;
315            })
316            .or_insert(SpillRecord { offset, variable, status: SpillStatus::Permanent });
317    }
318
319    /// Get the permanent spill slot offset for a value, if any.
320    pub(crate) fn get_permanent_spill_offset(&self, value_id: &ValueId) -> Option<usize> {
321        self.records
322            .get(value_id)
323            .filter(|r| matches!(r.status, SpillStatus::Permanent | SpillStatus::PermanentReloaded))
324            .map(|r| r.offset)
325    }
326
327    /// Get the spill slot offset for a value, if any.
328    ///
329    /// Unlike `get_permanent_spill_offset` this could be a permanent or a transient spill;
330    /// there should be only one at any point.
331    pub(crate) fn get_spill_offset(&self, value_id: &ValueId) -> Option<usize> {
332        self.records.get(value_id).map(|r| r.offset)
333    }
334
335    /// Re-mark permanently-spilled values as currently spilled at block entry.
336    ///
337    /// A reload in a previous block sets the status to `PermanentReloaded`, but the
338    /// value must be reloaded again in every block that uses it (since the
339    /// permanent spill slot is always the source of truth).
340    pub(crate) fn restore_permanent_spills(&mut self) {
341        for record in self.records.values_mut() {
342            if record.status == SpillStatus::PermanentReloaded {
343                record.status = SpillStatus::Permanent;
344            }
345        }
346    }
347
348    /// Mark a spilled value as reloaded into a register, without freeing the spill slot.
349    ///
350    /// - `Transient` → `TransientReloaded`
351    /// - `Permanent` → `PermanentReloaded`
352    ///
353    /// The slot's data must remain valid because the reload code may execute
354    /// again in a loop iteration. Only `remove_spill` (used when the value is
355    /// truly dead) should free the slot.
356    pub(crate) fn unmark_spilled(&mut self, value_id: &ValueId) {
357        let record = self.records.get_mut(value_id).expect("No record for value");
358        record.status = match record.status {
359            SpillStatus::Transient => SpillStatus::TransientReloaded,
360            SpillStatus::Permanent => SpillStatus::PermanentReloaded,
361            _ => panic!("Expected record not to be spilled, but was {:?}", record.status),
362        };
363    }
364
365    /// Ensure a value has a permanent spill slot.
366    ///
367    /// Any existing record — regardless of its current state — is promoted to
368    /// `Permanent` (the slot is the source of truth and the value is not in a register).
369    ///
370    /// # Returns
371    /// * `true` if a record already existed (caller should skip further processing),
372    /// * `false` if no record exists (first encounter — caller must allocate a slot).
373    pub(crate) fn ensure_permanent_spill(&mut self, value_id: &ValueId) -> bool {
374        let Some(record) = self.records.get_mut(value_id) else {
375            return false;
376        };
377        record.status = SpillStatus::Permanent;
378        true
379    }

622    #[test]
623    fn ensure_permanent_spill_all_branches() {
624        let mut sm = SpillManager::new();
625        let v0 = Id::test_new(0);
626        let v1 = Id::test_new(1);
627        let v2 = Id::test_new(2);
628        let v3 = Id::test_new(3);
629
630        // Case 1: No record -> returns false
631        assert!(!sm.ensure_permanent_spill(&v3));
632
633        // Case 2: Permanent -> returns true, stays Permanent
634        let off0 = sm.allocate_spill_offset();
635        sm.record_permanent_spill(v0, off0, test_var(0));
636        assert_eq!(sm.records[&v0].status, SpillStatus::Permanent);
637        assert!(sm.ensure_permanent_spill(&v0));
638        assert_eq!(sm.records[&v0].status, SpillStatus::Permanent);
639
640        // Case 3: Transient -> returns true, promoted to Permanent
641        let off1 = sm.allocate_spill_offset();
642        sm.record_spill(v1, off1, test_var(1));
643        assert_eq!(sm.records[&v1].status, SpillStatus::Transient);
644        assert!(sm.ensure_permanent_spill(&v1));
645        assert_eq!(sm.records[&v1].status, SpillStatus::Permanent);
646
647        // Case 4: PermanentReloaded -> returns true, re-marked as Permanent
648        let off2 = sm.allocate_spill_offset();
649        sm.record_permanent_spill(v2, off2, test_var(2));
650        sm.unmark_spilled(&v2);
651        assert_eq!(sm.records[&v2].status, SpillStatus::PermanentReloaded);
652        assert!(sm.ensure_permanent_spill(&v2));
653        assert_eq!(sm.records[&v2].status, SpillStatus::Permanent);
654    }
