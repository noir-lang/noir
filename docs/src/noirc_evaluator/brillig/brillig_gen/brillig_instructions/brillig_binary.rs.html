15    /// Converts the Binary instruction into a sequence of Brillig opcodes,
16    /// writing the results to the `result_variable`.
17    fn convert_ssa_binary(
18        &mut self,
19        binary: &Binary,
20        dfg: &DataFlowGraph,
21        result_variable: SingleAddrVariable,
22    ) {
23        let binary_type = type_of_binary_operation(
24            dfg[binary.lhs].get_type().as_ref(),
25            dfg[binary.rhs].get_type().as_ref(),
26        );
27
28        let left = self.convert_ssa_single_addr_value(binary.lhs, dfg);
29        let right = self.convert_ssa_single_addr_value(binary.rhs, dfg);
30
31        let (is_field, is_signed) = match binary_type {
32            Type::Numeric(numeric_type) => match numeric_type {
33                NumericType::Signed { .. } => (false, true),
34                NumericType::Unsigned { .. } => (false, false),
35                NumericType::NativeField => (true, false),
36            },
37            _ => unreachable!(
38                "only numeric types are allowed in binary operations. References are handled separately"
39            ),
40        };
41
42        let brillig_binary_op = match binary.operator {
43            BinaryOp::Div => {
44                if is_signed {
45                    self.brillig_context.convert_signed_division(left, right, result_variable);
46                    return;
47                } else if is_field {
48                    BrilligBinaryOp::FieldDiv
49                } else {
50                    BrilligBinaryOp::UnsignedDiv
51                }
52            }
53            BinaryOp::Mod => {
54                if is_signed {
55                    self.convert_signed_modulo(left, right, result_variable);
56                    return;
57                } else {
58                    BrilligBinaryOp::Modulo
59                }
60            }
61            BinaryOp::Add { .. } => BrilligBinaryOp::Add,
62            BinaryOp::Sub { .. } => BrilligBinaryOp::Sub,
63            BinaryOp::Mul { .. } => BrilligBinaryOp::Mul,
64            BinaryOp::Eq => BrilligBinaryOp::Equals,
65            BinaryOp::Lt => {
66                if is_signed {
67                    self.convert_signed_less_than(left, right, result_variable);
68                    return;
69                } else {
70                    BrilligBinaryOp::LessThan
71                }
72            }
73            BinaryOp::And => BrilligBinaryOp::And,
74            BinaryOp::Or => BrilligBinaryOp::Or,
75            BinaryOp::Xor => BrilligBinaryOp::Xor,
76            BinaryOp::Shl => BrilligBinaryOp::Shl,
77            BinaryOp::Shr => {
78                if is_signed {
79                    self.convert_signed_shr(left, right, result_variable);
80                    return;
81                } else {
82                    BrilligBinaryOp::Shr
83                }
84            }
85        };
86
87        self.brillig_context.binary_instruction(left, right, result_variable, brillig_binary_op);
88
89        self.add_overflow_check(left, right, result_variable, binary, dfg, is_signed);
90    }

92    fn convert_signed_modulo(
93        &mut self,
94        left: SingleAddrVariable,
95        right: SingleAddrVariable,
96        result: SingleAddrVariable,
97    ) {
98        let scratch_var_i = self.brillig_context.allocate_single_addr(left.bit_size);
99        let scratch_var_j = self.brillig_context.allocate_single_addr(left.bit_size);
100
101        // i = left / right
102        self.brillig_context.convert_signed_division(left, right, *scratch_var_i);
103
104        // j = i * right
105        self.brillig_context.binary_instruction(
106            *scratch_var_i,
107            right,
108            *scratch_var_j,
109            BrilligBinaryOp::Mul,
110        );
111
112        // result_register = left - j
113        self.brillig_context.binary_instruction(left, *scratch_var_j, result, BrilligBinaryOp::Sub);
114    }

116    fn convert_signed_shr(
117        &mut self,
118        left: SingleAddrVariable,
119        right: SingleAddrVariable,
120        result: SingleAddrVariable,
121    ) {
122        // Check if left is negative
123        let left_is_negative = self.brillig_context.allocate_single_addr_bool();
124
125        let max_positive = self
126            .brillig_context
127            .make_constant_instruction(((1_u128 << (left.bit_size - 1)) - 1).into(), left.bit_size);
128
129        self.brillig_context.binary_instruction(
130            *max_positive,
131            left,
132            *left_is_negative,
133            BrilligBinaryOp::LessThan,
134        );
135
136        self.brillig_context.codegen_branch(left_is_negative.address, |ctx, is_negative| {
137            if is_negative {
138                let one = ctx.make_constant_instruction(1_u128.into(), left.bit_size);
139
140                // computes 2^right
141                let two_pow = ctx.allocate_single_addr(left.bit_size);
142                ctx.binary_instruction(*one, right, *two_pow, BrilligBinaryOp::Shl);
143
144                // Right shift using division on 1-complement
145                ctx.binary_instruction(left, *one, result, BrilligBinaryOp::Add);
146                ctx.convert_signed_division(result, *two_pow, result);
147                ctx.binary_instruction(result, *one, result, BrilligBinaryOp::Sub);
148            } else {
149                ctx.binary_instruction(left, right, result, BrilligBinaryOp::Shr);
150            }
151        });
152    }

154    fn convert_signed_less_than(
155        &mut self,
156        left: SingleAddrVariable,
157        right: SingleAddrVariable,
158        result: SingleAddrVariable,
159    ) {
160        let biased_left = self.brillig_context.allocate_single_addr(left.bit_size);
161        let biased_right = self.brillig_context.allocate_single_addr(right.bit_size);
162
163        let bias = self
164            .brillig_context
165            .make_constant_instruction((1_u128 << (left.bit_size - 1)).into(), left.bit_size);
166
167        self.brillig_context.binary_instruction(left, *bias, *biased_left, BrilligBinaryOp::Add);
168        self.brillig_context.binary_instruction(right, *bias, *biased_right, BrilligBinaryOp::Add);
169
170        self.brillig_context.binary_instruction(
171            *biased_left,
172            *biased_right,
173            result,
174            BrilligBinaryOp::LessThan,
175        );
176    }

178    /// Overflow checks for the following unsigned binary operations
179    /// - Checked Add/Sub/Mul
180    #[allow(clippy::too_many_arguments)]
181    fn add_overflow_check(
182        &mut self,
183        left: SingleAddrVariable,
184        right: SingleAddrVariable,
185        result: SingleAddrVariable,
186        binary: &Binary,
187        dfg: &DataFlowGraph,
188        is_signed: bool,
189    ) {
190        let bit_size = left.bit_size;
191
192        if bit_size == FieldElement::max_num_bits() || is_signed {
193            if is_signed
194                && matches!(
195                    binary.operator,
196                    BinaryOp::Add { unchecked: false }
197                        | BinaryOp::Sub { unchecked: false }
198                        | BinaryOp::Mul { unchecked: false }
199                )
200            {
201                panic!("Checked signed operations should all be removed before brillig-gen")
202            }
203            return;
204        }
205
206        match binary.operator {
207            BinaryOp::Add { unchecked: false } => {
208                self.brillig_context.codegen_add_overflow_check(left, result);
209            }
210            BinaryOp::Sub { unchecked: false } => {
211                let condition = self.brillig_context.allocate_single_addr_bool();
212                // Check that rhs <= lhs
213                self.brillig_context.binary_instruction(
214                    right,
215                    left,
216                    *condition,
217                    BrilligBinaryOp::LessThanEquals,
218                );
219                let msg = "attempt to subtract with overflow".to_string();
220                self.brillig_context.codegen_constrain(*condition, Some(msg));
221            }
222            BinaryOp::Mul { unchecked: false } => {
223                let division_by_rhs_gives_lhs = |ctx: &mut BrilligContext<
224                    FieldElement,
225                    Registers,
226                >| {
227                    let condition = ctx.allocate_single_addr_bool();
228                    let division = ctx.allocate_single_addr(bit_size);
229                    // Check that result / rhs == lhs
230                    ctx.binary_instruction(result, right, *division, BrilligBinaryOp::UnsignedDiv);
231                    ctx.binary_instruction(*division, left, *condition, BrilligBinaryOp::Equals);
232                    let msg = "attempt to multiply with overflow".to_string();
233                    ctx.codegen_constrain(*condition, Some(msg));
234                };
235
236                let rhs_may_be_zero =
237                    dfg.get_numeric_constant(binary.rhs).is_none_or(|rhs| rhs.is_zero());
238                if rhs_may_be_zero {
239                    let is_right_zero = self.brillig_context.allocate_single_addr_bool();
240                    let zero =
241                        self.brillig_context.make_constant_instruction(0_usize.into(), bit_size);
242                    self.brillig_context.binary_instruction(
243                        *zero,
244                        right,
245                        *is_right_zero,
246                        BrilligBinaryOp::Equals,
247                    );
248                    self.brillig_context
249                        .codegen_if_not(is_right_zero.address, division_by_rhs_gives_lhs);
250                } else {
251                    division_by_rhs_gives_lhs(self.brillig_context);
252                }
253            }
254            _ => {}
255        }
256    }

258    /// Define a variable for the result and convert a binary operation to Brillig opcodes.
259    pub(crate) fn codegen_binary(
260        &mut self,
261        instruction_id: InstructionId,
262        binary: &Binary,
263        dfg: &DataFlowGraph,
264    ) {
265        let [result_value] = dfg.instruction_result(instruction_id);
266        let result_var = self.variables.define_single_addr_variable(
267            self.function_context,
268            self.brillig_context,
269            result_value,
270            dfg,
271        );
272        self.convert_ssa_binary(binary, dfg, result_var);
273    }

275    /// Generate Brillig opcodes to constrain two values to be equal to each other.
276    pub(crate) fn codegen_constrain(
277        &mut self,
278        lhs: ValueId,
279        rhs: ValueId,
280        assert_message: &Option<ConstrainError>,
281        dfg: &DataFlowGraph,
282    ) {
283        // Allocate a variable to hold the result of the comparison.
284        let condition = match (
285            dfg.get_numeric_constant_with_type(lhs),
286            dfg.get_numeric_constant_with_type(rhs),
287        ) {
288            // If the constraint is of the form `x == u1 1` then we can simply constrain `x` directly
289            (Some((constant, NumericType::Unsigned { bit_size: 1 })), None)
290                if constant == FieldElement::one() =>
291            {
292                let var = self.convert_ssa_single_addr_value(rhs, dfg);
293                Allocated::pure(var)
294            }
295            (None, Some((constant, NumericType::Unsigned { bit_size: 1 })))
296                if constant == FieldElement::one() =>
297            {
298                let var = self.convert_ssa_single_addr_value(lhs, dfg);
299                Allocated::pure(var)
300            }
301
302            // Otherwise we need to perform the equality explicitly.
303            _ => {
304                let condition = self.brillig_context.allocate_single_addr_bool();
305                self.convert_ssa_binary(
306                    &Binary { lhs, rhs, operator: BinaryOp::Eq },
307                    dfg,
308                    *condition,
309                );
310                condition
311            }
312        };
313
314        match assert_message {
315            Some(ConstrainError::Dynamic(selector, is_string_type, values)) => {
316                if let Some(constant_string) =
317                    try_to_extract_string_from_error_payload(*is_string_type, values, dfg)
318                {
319                    self.brillig_context.codegen_constrain(*condition, Some(constant_string));
320                } else {
321                    let payload_values =
322                        vecmap(values, |value| self.convert_ssa_value(*value, dfg));
323                    let payload_as_params = vecmap(values, |value| {
324                        let value_type = dfg.type_of_value(*value);
325                        FunctionContext::ssa_type_to_parameter(&value_type)
326                    });
327
328                    self.brillig_context.codegen_constrain_with_error_data(
329                        *condition,
330                        payload_values,
331                        payload_as_params,
332                        Some(*selector),
333                    );
334                }
335            }
336            Some(ConstrainError::StaticString(message)) => {
337                self.brillig_context.codegen_constrain(*condition, Some(message.clone()));
338            }
339            None => {
340                self.brillig_context.codegen_constrain(*condition, None);
341            }
342        }
343    }
