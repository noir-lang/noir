15    /// Converts the Binary instruction into a sequence of Brillig opcodes,
16    /// writing the results to the `result_variable`.
17    fn convert_ssa_binary(
18        &mut self,
19        binary: &Binary,
20        dfg: &DataFlowGraph,
21        result_variable: SingleAddrVariable,
22    ) {
23        let binary_type = type_of_binary_operation(
24            dfg[binary.lhs].get_type().as_ref(),
25            dfg[binary.rhs].get_type().as_ref(),
26        );
27
28        let left = self.convert_ssa_single_addr_value(binary.lhs, dfg);
29        let right = self.convert_ssa_single_addr_value(binary.rhs, dfg);
30
31        let (is_field, is_signed) = match binary_type {
32            Type::Numeric(numeric_type) => match numeric_type {
33                NumericType::Signed { .. } => (false, true),
34                NumericType::Unsigned { .. } => (false, false),
35                NumericType::NativeField => (true, false),
36            },
37            _ => unreachable!(
38                "only numeric types are allowed in binary operations. References are handled separately"
39            ),
40        };
41
42        let brillig_binary_op = match binary.operator {
43            BinaryOp::Div => {
44                if is_signed {
45                    self.brillig_context.convert_signed_division(left, right, result_variable);
46                    return;
47                } else if is_field {
48                    BrilligBinaryOp::FieldDiv
49                } else {
50                    BrilligBinaryOp::UnsignedDiv
51                }
52            }
53            BinaryOp::Mod => {
54                if is_signed {
55                    self.convert_signed_modulo(left, right, result_variable);
56                    return;
57                } else {
58                    BrilligBinaryOp::Modulo
59                }
60            }
61            BinaryOp::Add { .. } => BrilligBinaryOp::Add,
62            BinaryOp::Sub { .. } => BrilligBinaryOp::Sub,
63            BinaryOp::Mul { .. } => BrilligBinaryOp::Mul,
64            BinaryOp::Eq => BrilligBinaryOp::Equals,
65            BinaryOp::Lt => {
66                if is_signed {
67                    self.convert_signed_less_than(left, right, result_variable);
68                    return;
69                } else {
70                    BrilligBinaryOp::LessThan
71                }
72            }
73            BinaryOp::And => BrilligBinaryOp::And,
74            BinaryOp::Or => BrilligBinaryOp::Or,
75            BinaryOp::Xor => BrilligBinaryOp::Xor,
76            BinaryOp::Shl => {
77                self.bit_shift_overflow(left, right);
78                BrilligBinaryOp::Shl
79            }
80            BinaryOp::Shr => {
81                self.bit_shift_overflow(left, right);
82                if is_signed {
83                    self.convert_signed_shr(left, right, result_variable);
84                    return;
85                } else {
86                    BrilligBinaryOp::Shr
87                }
88            }
89        };
90
91        self.brillig_context.binary_instruction(left, right, result_variable, brillig_binary_op);
92
93        self.add_overflow_check(left, right, result_variable, binary, dfg, is_signed);
94    }

96    fn convert_signed_modulo(
97        &mut self,
98        left: SingleAddrVariable,
99        right: SingleAddrVariable,
100        result: SingleAddrVariable,
101    ) {
102        let scratch_var_i = self.brillig_context.allocate_single_addr(left.bit_size);
103        let scratch_var_j = self.brillig_context.allocate_single_addr(left.bit_size);
104
105        // i = left / right
106        self.brillig_context.convert_signed_division(left, right, *scratch_var_i);
107
108        // j = i * right
109        self.brillig_context.binary_instruction(
110            *scratch_var_i,
111            right,
112            *scratch_var_j,
113            BrilligBinaryOp::Mul,
114        );
115
116        // result_register = left - j
117        self.brillig_context.binary_instruction(left, *scratch_var_j, result, BrilligBinaryOp::Sub);
118    }

120    fn bit_shift_overflow(&mut self, left: SingleAddrVariable, right: SingleAddrVariable) {
121        // Check that right is less than the bit size of left
122        // The constraint will fail also if rhs is negative, which is expected.
123        let right_overflow = self.brillig_context.allocate_single_addr_bool();
124        let left_bit_size =
125            self.brillig_context.make_constant_instruction(left.bit_size.into(), left.bit_size);
126
127        self.brillig_context.binary_instruction(
128            right,
129            *left_bit_size,
130            *right_overflow,
131            BrilligBinaryOp::LessThan,
132        );
133        let msg = "attempt to bit-shift with overflow".to_string();
134        self.brillig_context.codegen_constrain(*right_overflow, Some(msg));
135    }

137    fn convert_signed_shr(
138        &mut self,
139        left: SingleAddrVariable,
140        right: SingleAddrVariable,
141        result: SingleAddrVariable,
142    ) {
143        // Check if left is negative
144        let left_is_negative = self.brillig_context.allocate_single_addr_bool();
145
146        let max_positive = self
147            .brillig_context
148            .make_constant_instruction(((1_u128 << (left.bit_size - 1)) - 1).into(), left.bit_size);
149
150        self.brillig_context.binary_instruction(
151            *max_positive,
152            left,
153            *left_is_negative,
154            BrilligBinaryOp::LessThan,
155        );
156
157        self.brillig_context.codegen_branch(left_is_negative.address, |ctx, is_negative| {
158            if is_negative {
159                let one = ctx.make_constant_instruction(1_u128.into(), left.bit_size);
160
161                // computes 2^right
162                let two_pow = ctx.allocate_single_addr(left.bit_size);
163                ctx.binary_instruction(*one, right, *two_pow, BrilligBinaryOp::Shl);
164
165                // Right shift using division on 1-complement
166                ctx.binary_instruction(left, *one, result, BrilligBinaryOp::Add);
167                ctx.convert_signed_division(result, *two_pow, result);
168                ctx.binary_instruction(result, *one, result, BrilligBinaryOp::Sub);
169            } else {
170                ctx.binary_instruction(left, right, result, BrilligBinaryOp::Shr);
171            }
172        });
173    }

175    fn convert_signed_less_than(
176        &mut self,
177        left: SingleAddrVariable,
178        right: SingleAddrVariable,
179        result: SingleAddrVariable,
180    ) {
181        let biased_left = self.brillig_context.allocate_single_addr(left.bit_size);
182        let biased_right = self.brillig_context.allocate_single_addr(right.bit_size);
183
184        let bias = self
185            .brillig_context
186            .make_constant_instruction((1_u128 << (left.bit_size - 1)).into(), left.bit_size);
187
188        self.brillig_context.binary_instruction(left, *bias, *biased_left, BrilligBinaryOp::Add);
189        self.brillig_context.binary_instruction(right, *bias, *biased_right, BrilligBinaryOp::Add);
190
191        self.brillig_context.binary_instruction(
192            *biased_left,
193            *biased_right,
194            result,
195            BrilligBinaryOp::LessThan,
196        );
197    }

199    /// Overflow checks for the following unsigned binary operations
200    /// - Checked Add/Sub/Mul
201    #[allow(clippy::too_many_arguments)]
202    fn add_overflow_check(
203        &mut self,
204        left: SingleAddrVariable,
205        right: SingleAddrVariable,
206        result: SingleAddrVariable,
207        binary: &Binary,
208        dfg: &DataFlowGraph,
209        is_signed: bool,
210    ) {
211        let bit_size = left.bit_size;
212
213        if bit_size == FieldElement::max_num_bits() || is_signed {
214            if is_signed
215                && matches!(
216                    binary.operator,
217                    BinaryOp::Add { unchecked: false }
218                        | BinaryOp::Sub { unchecked: false }
219                        | BinaryOp::Mul { unchecked: false }
220                )
221            {
222                panic!("Checked signed operations should all be removed before brillig-gen")
223            }
224            return;
225        }
226
227        match binary.operator {
228            BinaryOp::Add { unchecked: false } => {
229                self.brillig_context.codegen_add_overflow_check(left, result);
230            }
231            BinaryOp::Sub { unchecked: false } => {
232                let condition = self.brillig_context.allocate_single_addr_bool();
233                // Check that rhs <= lhs
234                self.brillig_context.binary_instruction(
235                    right,
236                    left,
237                    *condition,
238                    BrilligBinaryOp::LessThanEquals,
239                );
240                let msg = "attempt to subtract with overflow".to_string();
241                self.brillig_context.codegen_constrain(*condition, Some(msg));
242            }
243            BinaryOp::Mul { unchecked: false } => {
244                let division_by_rhs_gives_lhs = |ctx: &mut BrilligContext<
245                    FieldElement,
246                    Registers,
247                >| {
248                    let condition = ctx.allocate_single_addr_bool();
249                    let division = ctx.allocate_single_addr(bit_size);
250                    // Check that result / rhs == lhs
251                    ctx.binary_instruction(result, right, *division, BrilligBinaryOp::UnsignedDiv);
252                    ctx.binary_instruction(*division, left, *condition, BrilligBinaryOp::Equals);
253                    let msg = "attempt to multiply with overflow".to_string();
254                    ctx.codegen_constrain(*condition, Some(msg));
255                };
256
257                let rhs_may_be_zero =
258                    dfg.get_numeric_constant(binary.rhs).is_none_or(|rhs| rhs.is_zero());
259                if rhs_may_be_zero {
260                    let is_right_zero = self.brillig_context.allocate_single_addr_bool();
261                    let zero =
262                        self.brillig_context.make_constant_instruction(0_usize.into(), bit_size);
263                    self.brillig_context.binary_instruction(
264                        *zero,
265                        right,
266                        *is_right_zero,
267                        BrilligBinaryOp::Equals,
268                    );
269                    self.brillig_context
270                        .codegen_if_not(is_right_zero.address, division_by_rhs_gives_lhs);
271                } else {
272                    division_by_rhs_gives_lhs(self.brillig_context);
273                }
274            }
275            _ => {}
276        }
277    }

279    /// Define a variable for the result and convert a binary operation to Brillig opcodes.
280    pub(crate) fn codegen_binary(
281        &mut self,
282        instruction_id: InstructionId,
283        binary: &Binary,
284        dfg: &DataFlowGraph,
285    ) {
286        let [result_value] = dfg.instruction_result(instruction_id);
287        let result_var = self.variables.define_single_addr_variable(
288            self.function_context,
289            self.brillig_context,
290            result_value,
291            dfg,
292        );
293        self.convert_ssa_binary(binary, dfg, result_var);
294    }

296    /// Generate Brillig opcodes to constrain two values to be equal to each other.
297    pub(crate) fn codegen_constrain(
298        &mut self,
299        lhs: ValueId,
300        rhs: ValueId,
301        assert_message: &Option<ConstrainError>,
302        dfg: &DataFlowGraph,
303    ) {
304        // Allocate a variable to hold the result of the comparison.
305        let condition = match (
306            dfg.get_numeric_constant_with_type(lhs),
307            dfg.get_numeric_constant_with_type(rhs),
308        ) {
309            // If the constraint is of the form `x == u1 1` then we can simply constrain `x` directly
310            (Some((constant, NumericType::Unsigned { bit_size: 1 })), None)
311                if constant == FieldElement::one() =>
312            {
313                let var = self.convert_ssa_single_addr_value(rhs, dfg);
314                Allocated::pure(var)
315            }
316            (None, Some((constant, NumericType::Unsigned { bit_size: 1 })))
317                if constant == FieldElement::one() =>
318            {
319                let var = self.convert_ssa_single_addr_value(lhs, dfg);
320                Allocated::pure(var)
321            }
322
323            // Otherwise we need to perform the equality explicitly.
324            _ => {
325                let condition = self.brillig_context.allocate_single_addr_bool();
326                self.convert_ssa_binary(
327                    &Binary { lhs, rhs, operator: BinaryOp::Eq },
328                    dfg,
329                    *condition,
330                );
331                condition
332            }
333        };
334
335        match assert_message {
336            Some(ConstrainError::Dynamic(selector, is_string_type, values)) => {
337                if let Some(constant_string) =
338                    try_to_extract_string_from_error_payload(*is_string_type, values, dfg)
339                {
340                    self.brillig_context.codegen_constrain(*condition, Some(constant_string));
341                } else {
342                    let payload_values =
343                        vecmap(values, |value| self.convert_ssa_value(*value, dfg));
344                    let payload_as_params = vecmap(values, |value| {
345                        let value_type = dfg.type_of_value(*value);
346                        FunctionContext::ssa_type_to_parameter(&value_type)
347                    });
348
349                    self.brillig_context.codegen_constrain_with_error_data(
350                        *condition,
351                        payload_values,
352                        payload_as_params,
353                        Some(*selector),
354                    );
355                }
356            }
357            Some(ConstrainError::StaticString(message)) => {
358                self.brillig_context.codegen_constrain(*condition, Some(message.clone()));
359            }
360            None => {
361                self.brillig_context.codegen_constrain(*condition, None);
362            }
363        }
364    }
