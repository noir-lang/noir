15    /// Converts the Binary instruction into a sequence of Brillig opcodes,
16    /// writing the results to the `result_variable`.
17    fn convert_ssa_binary(
18        &mut self,
19        binary: &Binary,
20        dfg: &DataFlowGraph,
21        result_variable: SingleAddrVariable,
22    ) {
23        let binary_type = type_of_binary_operation(
24            dfg[binary.lhs].get_type().as_ref(),
25            dfg[binary.rhs].get_type().as_ref(),
26        );
27
28        let left = self.convert_ssa_single_addr_value(binary.lhs, dfg);
29        let right = self.convert_ssa_single_addr_value(binary.rhs, dfg);
30
31        let (is_field, is_signed) = match binary_type {
32            Type::Numeric(numeric_type) => match numeric_type {
33                NumericType::Signed { .. } => (false, true),
34                NumericType::Unsigned { .. } => (false, false),
35                NumericType::NativeField => (true, false),
36            },
37            _ => unreachable!(
38                "only numeric types are allowed in binary operations. References are handled separately"
39            ),
40        };
41
42        let brillig_binary_op = match binary.operator {
43            BinaryOp::Div => {
44                if is_signed {
45                    self.brillig_context.convert_signed_division(
46                        left,
47                        right,
48                        result_variable,
49                        SignedDivisionOperator::Div,
50                    );
51                    return;
52                } else if is_field {
53                    BrilligBinaryOp::FieldDiv
54                } else {
55                    BrilligBinaryOp::UnsignedDiv
56                }
57            }
58            BinaryOp::Mod => {
59                // Mod is lowered to division where division by 0 gives a "divide by zero"
60                // error instead of a "mod by 0" one. This costs 4 extra opcodes to have
61                // a pre-check to change the error message.
62                if dfg.get_numeric_constant(binary.rhs).is_none_or(|rhs| rhs.is_zero()) {
63                    self.mod_by_zero_check(right);
64                }
65                if is_signed {
66                    self.convert_signed_modulo(left, right, result_variable);
67                    return;
68                } else {
69                    BrilligBinaryOp::Modulo
70                }
71            }
72            BinaryOp::Add { .. } => BrilligBinaryOp::Add,
73            BinaryOp::Sub { .. } => BrilligBinaryOp::Sub,
74            BinaryOp::Mul { .. } => BrilligBinaryOp::Mul,
75            BinaryOp::Eq => BrilligBinaryOp::Equals,
76            BinaryOp::Lt => {
77                if is_signed {
78                    self.convert_signed_less_than(left, right, result_variable);
79                    return;
80                } else {
81                    BrilligBinaryOp::LessThan
82                }
83            }
84            BinaryOp::And => BrilligBinaryOp::And,
85            BinaryOp::Or => BrilligBinaryOp::Or,
86            BinaryOp::Xor => BrilligBinaryOp::Xor,
87            BinaryOp::Shl => {
88                self.bit_shift_overflow(left, right);
89                BrilligBinaryOp::Shl
90            }
91            BinaryOp::Shr => {
92                self.bit_shift_overflow(left, right);
93                if is_signed {
94                    self.convert_signed_shr(left, right, result_variable);
95                    return;
96                } else {
97                    BrilligBinaryOp::Shr
98                }
99            }
100        };
101
102        self.brillig_context.binary_instruction(left, right, result_variable, brillig_binary_op);
103
104        self.add_overflow_check(left, right, result_variable, binary, dfg, is_signed);
105    }

107    fn convert_signed_modulo(
108        &mut self,
109        left: SingleAddrVariable,
110        right: SingleAddrVariable,
111        result: SingleAddrVariable,
112    ) {
113        let scratch_var_i = self.brillig_context.allocate_single_addr(left.bit_size);
114        let scratch_var_j = self.brillig_context.allocate_single_addr(left.bit_size);
115
116        // i = left / right
117        self.brillig_context.convert_signed_division(
118            left,
119            right,
120            *scratch_var_i,
121            SignedDivisionOperator::Mod,
122        );
123
124        // j = i * right
125        self.brillig_context.binary_instruction(
126            *scratch_var_i,
127            right,
128            *scratch_var_j,
129            BrilligBinaryOp::Mul,
130        );
131
132        // result_register = left - j
133        self.brillig_context.binary_instruction(left, *scratch_var_j, result, BrilligBinaryOp::Sub);
134    }

136    fn mod_by_zero_check(&mut self, right: SingleAddrVariable) {
137        // `LessThan` is unsigned; an all-zero bit pattern is the only encoding of zero,
138        // including for signed operands (two's complement). Negative signed values have
139        // a large unsigned bit pattern and pass `0 < right`.
140        let right_is_nonzero = self.brillig_context.allocate_single_addr_bool();
141        let zero = self.brillig_context.make_constant_instruction(0_usize.into(), right.bit_size);
142        self.brillig_context.binary_instruction(
143            *zero,
144            right,
145            *right_is_nonzero,
146            BrilligBinaryOp::LessThan,
147        );
148        let msg = "attempt to calculate the remainder with a divisor of zero".to_string();
149        self.brillig_context.codegen_constrain(*right_is_nonzero, Some(msg));
150    }

152    fn bit_shift_overflow(&mut self, left: SingleAddrVariable, right: SingleAddrVariable) {
153        // Check that right is less than the bit size of left
154        // The constraint will fail also if rhs is negative, which is expected.
155        let right_overflow = self.brillig_context.allocate_single_addr_bool();
156        let left_bit_size =
157            self.brillig_context.make_constant_instruction(left.bit_size.into(), left.bit_size);
158
159        self.brillig_context.binary_instruction(
160            right,
161            *left_bit_size,
162            *right_overflow,
163            BrilligBinaryOp::LessThan,
164        );
165        let msg = "attempt to bit-shift with overflow".to_string();
166        self.brillig_context.codegen_constrain(*right_overflow, Some(msg));
167    }

169    fn convert_signed_shr(
170        &mut self,
171        left: SingleAddrVariable,
172        right: SingleAddrVariable,
173        result: SingleAddrVariable,
174    ) {
175        // Check if left is negative
176        let left_is_negative = self.brillig_context.allocate_single_addr_bool();
177
178        let max_positive = self
179            .brillig_context
180            .make_constant_instruction(((1_u128 << (left.bit_size - 1)) - 1).into(), left.bit_size);
181
182        self.brillig_context.binary_instruction(
183            *max_positive,
184            left,
185            *left_is_negative,
186            BrilligBinaryOp::LessThan,
187        );
188
189        self.brillig_context.codegen_branch(left_is_negative.address, |ctx, is_negative| {
190            if is_negative {
191                let one = ctx.make_constant_instruction(1_u128.into(), left.bit_size);
192
193                // computes 2^right
194                let two_pow = ctx.allocate_single_addr(left.bit_size);
195                ctx.binary_instruction(*one, right, *two_pow, BrilligBinaryOp::Shl);
196
197                // Right shift using division on 1-complement
198                ctx.binary_instruction(left, *one, result, BrilligBinaryOp::Add);
199                ctx.convert_signed_division(
200                    result,
201                    *two_pow,
202                    result,
203                    SignedDivisionOperator::Shift,
204                );
205                ctx.binary_instruction(result, *one, result, BrilligBinaryOp::Sub);
206            } else {
207                ctx.binary_instruction(left, right, result, BrilligBinaryOp::Shr);
208            }
209        });
210    }

212    fn convert_signed_less_than(
213        &mut self,
214        left: SingleAddrVariable,
215        right: SingleAddrVariable,
216        result: SingleAddrVariable,
217    ) {
218        let biased_left = self.brillig_context.allocate_single_addr(left.bit_size);
219        let biased_right = self.brillig_context.allocate_single_addr(right.bit_size);
220
221        let bias = self
222            .brillig_context
223            .make_constant_instruction((1_u128 << (left.bit_size - 1)).into(), left.bit_size);
224
225        self.brillig_context.binary_instruction(left, *bias, *biased_left, BrilligBinaryOp::Add);
226        self.brillig_context.binary_instruction(right, *bias, *biased_right, BrilligBinaryOp::Add);
227
228        self.brillig_context.binary_instruction(
229            *biased_left,
230            *biased_right,
231            result,
232            BrilligBinaryOp::LessThan,
233        );
234    }

236    /// Overflow checks for the following unsigned binary operations
237    /// - Checked Add/Sub/Mul
238    #[allow(clippy::too_many_arguments)]
239    fn add_overflow_check(
240        &mut self,
241        left: SingleAddrVariable,
242        right: SingleAddrVariable,
243        result: SingleAddrVariable,
244        binary: &Binary,
245        dfg: &DataFlowGraph,
246        is_signed: bool,
247    ) {
248        let bit_size = left.bit_size;
249
250        if bit_size == FieldElement::max_num_bits() || is_signed {
251            if is_signed
252                && matches!(
253                    binary.operator,
254                    BinaryOp::Add { unchecked: false }
255                        | BinaryOp::Sub { unchecked: false }
256                        | BinaryOp::Mul { unchecked: false }
257                )
258            {
259                panic!("Checked signed operations should all be removed before brillig-gen")
260            }
261            return;
262        }
263
264        match binary.operator {
265            BinaryOp::Add { unchecked: false } => {
266                self.brillig_context.codegen_add_overflow_check(left, result);
267            }
268            BinaryOp::Sub { unchecked: false } => {
269                let condition = self.brillig_context.allocate_single_addr_bool();
270                // Check that rhs <= lhs
271                self.brillig_context.binary_instruction(
272                    right,
273                    left,
274                    *condition,
275                    BrilligBinaryOp::LessThanEquals,
276                );
277                let msg = "attempt to subtract with overflow".to_string();
278                self.brillig_context.codegen_constrain(*condition, Some(msg));
279            }
280            BinaryOp::Mul { unchecked: false } => {
281                let division_by_rhs_gives_lhs = |ctx: &mut BrilligContext<
282                    FieldElement,
283                    Registers,
284                >| {
285                    let condition = ctx.allocate_single_addr_bool();
286                    let division = ctx.allocate_single_addr(bit_size);
287                    // Check that result / rhs == lhs
288                    ctx.binary_instruction(result, right, *division, BrilligBinaryOp::UnsignedDiv);
289                    ctx.binary_instruction(*division, left, *condition, BrilligBinaryOp::Equals);
290                    let msg = "attempt to multiply with overflow".to_string();
291                    ctx.codegen_constrain(*condition, Some(msg));
292                };
293
294                let rhs_may_be_zero =
295                    dfg.get_numeric_constant(binary.rhs).is_none_or(|rhs| rhs.is_zero());
296                if rhs_may_be_zero {
297                    let is_right_zero = self.brillig_context.allocate_single_addr_bool();
298                    let zero =
299                        self.brillig_context.make_constant_instruction(0_usize.into(), bit_size);
300                    self.brillig_context.binary_instruction(
301                        *zero,
302                        right,
303                        *is_right_zero,
304                        BrilligBinaryOp::Equals,
305                    );
306                    self.brillig_context
307                        .codegen_if_not(is_right_zero.address, division_by_rhs_gives_lhs);
308                } else {
309                    division_by_rhs_gives_lhs(self.brillig_context);
310                }
311            }
312            _ => {}
313        }
314    }

316    /// Define a variable for the result and convert a binary operation to Brillig opcodes.
317    pub(crate) fn codegen_binary(
318        &mut self,
319        instruction_id: InstructionId,
320        binary: &Binary,
321        dfg: &DataFlowGraph,
322    ) {
323        let [result_value] = dfg.instruction_result(instruction_id);
324        let result_var = self.define_single_addr_variable(result_value, dfg);
325        self.convert_ssa_binary(binary, dfg, result_var);
326    }

328    /// Generate Brillig opcodes to constrain two values to be equal to each other.
329    pub(crate) fn codegen_constrain(
330        &mut self,
331        lhs: ValueId,
332        rhs: ValueId,
333        assert_message: &Option<ConstrainError>,
334        dfg: &DataFlowGraph,
335    ) {
336        // Allocate a variable to hold the result of the comparison.
337        let condition = match (
338            dfg.get_numeric_constant_with_type(lhs),
339            dfg.get_numeric_constant_with_type(rhs),
340        ) {
341            // If the constraint is of the form `x == u1 1` then we can simply constrain `x` directly
342            (Some((constant, NumericType::Unsigned { bit_size: 1 })), None)
343                if constant == FieldElement::one() =>
344            {
345                let var = self.convert_ssa_single_addr_value(rhs, dfg);
346                Allocated::pure(var)
347            }
348            (None, Some((constant, NumericType::Unsigned { bit_size: 1 })))
349                if constant == FieldElement::one() =>
350            {
351                let var = self.convert_ssa_single_addr_value(lhs, dfg);
352                Allocated::pure(var)
353            }
354
355            // Otherwise we need to perform the equality explicitly.
356            _ => {
357                let condition = self.brillig_context.allocate_single_addr_bool();
358                self.convert_ssa_binary(
359                    &Binary { lhs, rhs, operator: BinaryOp::Eq },
360                    dfg,
361                    *condition,
362                );
363                condition
364            }
365        };
366
367        match assert_message {
368            Some(ConstrainError::Dynamic(selector, is_string_type, values)) => {
369                if let Some(constant_string) =
370                    try_to_extract_string_from_error_payload(*is_string_type, values, dfg)
371                {
372                    self.brillig_context.codegen_constrain(*condition, Some(constant_string));
373                } else {
374                    let payload_values =
375                        vecmap(values, |value| self.convert_ssa_value(*value, dfg));
376                    let payload_as_params = vecmap(values, |value| {
377                        let value_type = dfg.type_of_value(*value);
378                        FunctionContext::ssa_type_to_parameter(&value_type)
379                    });
380
381                    self.brillig_context.codegen_constrain_with_error_data(
382                        *condition,
383                        payload_values,
384                        payload_as_params,
385                        Some(*selector),
386                    );
387                }
388            }
389            Some(ConstrainError::StaticString(message)) => {
390                self.brillig_context.codegen_constrain(*condition, Some(message.clone()));
391            }
392            None => {
393                self.brillig_context.codegen_constrain(*condition, None);
394            }
395        }
396    }
