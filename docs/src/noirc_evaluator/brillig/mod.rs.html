60/// Context structure for the Brillig pass.
61/// It stores Brillig-related data required for Brillig generation.
62#[derive(Default, Clone)]
63pub struct Brillig {
64    /// Maps SSA function labels to their brillig artifact
65    ssa_function_to_brillig: HashMap<FunctionId, BrilligArtifact<FieldElement>>,
66    call_stacks: CallStackHelper,
67    /// Bytecode for globals for each Brillig entry point.
68    globals: HashMap<FunctionId, BrilligArtifact<FieldElement>>,
69    /// The size of the global space for each Brillig entry point.
70    globals_memory_size: HashMap<FunctionId, usize>,

73impl Brillig {
74    /// Compiles a function into brillig and store the compilation artifacts.
75    pub(crate) fn compile(
76        &mut self,
77        func: &Function,
78        options: &BrilligOptions,
79        globals: &HashMap<ValueId, BrilligVariable>,
80        hoisted_global_constants: &HashMap<(FieldElement, NumericType), BrilligVariable>,
81        is_entry_point: bool,
82    ) {
83        let obj = self.convert_ssa_function(
84            func,
85            options,
86            globals,
87            hoisted_global_constants,
88            is_entry_point,
89        );
90        self.ssa_function_to_brillig.insert(func.id(), obj);
91    }
92
93    /// Finds a brillig artifact by its label.
94    pub(crate) fn find_by_label(
95        &self,
96        function_label: Label,
97        options: &BrilligOptions,
98        stack_start: usize,
99    ) -> Option<Cow<BrilligArtifact<FieldElement>>> {
100        match function_label.label_type {
101            LabelType::Function(function_id, _) => {
102                self.ssa_function_to_brillig.get(&function_id).map(Cow::Borrowed)
103            }
104            // Procedures are compiled as needed
105            LabelType::Procedure(procedure_id) => {
106                Some(Cow::Owned(compile_procedure(procedure_id, options, stack_start)))
107            }
108            LabelType::GlobalInit(function_id) => self.globals.get(&function_id).map(Cow::Borrowed),
109            LabelType::Entrypoint => unreachable!("ICE: Expected a function or procedure label"),
110        }
111    }

113    /// Convert an SSA function into Brillig bytecode.
114    pub(crate) fn convert_ssa_function(
115        &mut self,
116        func: &Function,
117        options: &BrilligOptions,
118        globals: &HashMap<ValueId, BrilligVariable>,
119        hoisted_global_constants: &HashMap<(FieldElement, NumericType), BrilligVariable>,
120        is_entry_point: bool,
121    ) -> BrilligArtifact<FieldElement> {
122        let mut function_context =
123            FunctionContext::new(func, is_entry_point, options.layout.max_stack_frame_size());
124
125        let mut brillig_context = BrilligContext::new(func.name(), options);
126
127        brillig_context.enter_context(Label::function(func.id()));
128
129        brillig_context.call_check_max_stack_depth_procedure();
130
131        // Only emit spill prologue placeholders when the function may need spilling.
132        if function_context.spill_enabled() {
133            brillig_context.emit_unresolved_spill_prologue();
134        }
135
136        for block in function_context.reverse_post_order().collect::<Vec<_>>() {
137            BrilligBlock::compile_block(
138                &mut function_context,
139                &mut brillig_context,
140                block,
141                &func.dfg,
142                &mut self.call_stacks,
143                globals,
144                hoisted_global_constants,
145            );
146        }
147
148        // Resolve: overwrite placeholder NOPs with real allocation if spilling occurred
149        if function_context.did_spill() {
150            brillig_context.resolve_spill_prologue(function_context.max_spill_offset());
151        }
152
153        if options.show_opcode_advisories {
154            let opcode_advisories =
155                brillig_check::opcode_advisories(func, &function_context, &brillig_context);
156
157            brillig_check::show_opcode_advisories(&opcode_advisories, brillig_context.artifact());
158        }
159
160        brillig_context.into_artifact()
161    }
162
163    pub fn call_stacks(&self) -> &CallStackHelper {
164        &self.call_stacks
165    }

175impl Ssa {
176    /// Compile Brillig functions and ACIR functions reachable from them.
177    #[tracing::instrument(level = "trace", skip_all)]
178    pub fn to_brillig(&self, options: &BrilligOptions) -> Brillig {
179        // Collect all the function ids that are reachable from Brillig.
180        // That means all the functions marked as Brillig and ACIR functions called by them,
181        // but we should already have monomorphized ACIR functions as a Brillig as well.
182        let brillig_reachable_function_ids = self
183            .functions
184            .iter()
185            .filter_map(|(id, func)| func.runtime().is_brillig().then_some(*id))
186            .collect::<BTreeSet<_>>();
187
188        let mut brillig = Brillig::default();
189
190        if brillig_reachable_function_ids.is_empty() {
191            return brillig;
192        }
193
194        // SSA Globals are computed once at compile time and shared across all functions,
195        // thus we can just fetch globals from the main function.
196        // This same globals graph will then be used to declare Brillig globals for the respective entry points.
197        let globals = (*self.functions[&self.main_id].dfg.globals).clone();
198        let globals_dfg = DataFlowGraph::from(globals);
199
200        // Produce the globals Brillig bytecode and variable allocation for each entry point.
201        let brillig_globals = BrilligGlobals::init(self, self.main_id).declare_globals(
202            &globals_dfg,
203            &mut brillig,
204            options,
205        );
206
207        for brillig_function_id in brillig_reachable_function_ids {
208            let (globals_allocations, hoisted_constant_allocations) =
209                brillig_globals.get_brillig_globals(brillig_function_id);
210
211            let func = &self.functions[&brillig_function_id];
212            let is_entry_point = brillig_globals.is_entry_point(&brillig_function_id);
213
214            brillig.compile(
215                func,
216                options,
217                globals_allocations,
218                hoisted_constant_allocations,
219                is_entry_point,
220            );
221        }
222
223        brillig
224    }

244    fn assert_equivalent_bytecode<F: AcirField>(
245        bytecode1: &[Opcode<F>],
246        bytecode2: &[Opcode<F>],
247        options1: &BrilligOptions,
248        options2: &BrilligOptions,
249    ) {
250        assert_eq!(bytecode1.len(), bytecode2.len());
251
252        // Offset where stack starts
253        // This assumes the same SSA where we have a single global register and a single call data argument.
254        let stack_start = ReservedRegisters::len() + MAX_SCRATCH_SPACE + 1 + 1;
255
256        for (op1, op2) in bytecode1.iter().zip(bytecode2) {
257            if op1 != op2 {
258                match (op1, op2) {
259                    // All opcodes are expected to be stable across memory layouts except for a few that rely
260                    // on the max stack size and stack frame size (e.g., setting of the `free_memory_pointer` and stack bounds checks).
261                    (
262                        Opcode::Const { destination: dest1, bit_size: bits1, value: val1 },
263                        Opcode::Const { destination: dest2, bit_size: bits2, value: val2 },
264                    ) => {
265                        assert_eq!(dest1, dest2);
266                        assert_eq!(bits1, bits2);
267
268                        if *dest1 == ReservedRegisters::free_memory_pointer() {
269                            // free_memory_pointer = stack_start + max_stack_size
270                            // (heap starts right after the stack, no global spill region)
271                            let expected1 = options1.layout.max_stack_size() + stack_start;
272                            let expected2 = options2.layout.max_stack_size() + stack_start;
273
274                            assert_eq!(val1.to_u128(), expected1 as u128);
275                            assert_eq!(val2.to_u128(), expected2 as u128);
276                        } else {
277                            // Stack depth bound check
278                            // This is done by the CheckMaxStackDepth procedure
279                            let bound1 = options1.layout.max_stack_size()
280                                - options1.layout.max_stack_frame_size()
281                                + stack_start;
282                            let bound2 = options2.layout.max_stack_size()
283                                - options2.layout.max_stack_frame_size()
284                                + stack_start;
285
286                            assert_eq!(val1.to_u128(), bound1 as u128);
287                            assert_eq!(val2.to_u128(), bound2 as u128);
288                        }
289                    }
290                    _ => panic!("Unexpected opcode difference {op1} != {op2}"),
291                }
292            }
293        }
294    }

296    fn compiles_to_equivalent_bytecode(
297        ssa: &Ssa,
298        options1: BrilligOptions,
299        options2: BrilligOptions,
300    ) {
301        let main = ssa.main();
302
303        let brillig1 = ssa.to_brillig(&options1);
304        let brillig2 = ssa.to_brillig(&options2);
305
306        // SSA function level comparison
307        let code1 = &brillig1.ssa_function_to_brillig[&ssa.main_id].byte_code;
308        let code2 = &brillig2.ssa_function_to_brillig[&ssa.main_id].byte_code;
309        assert_equivalent_bytecode(code1, code2, &options1, &options2);
310
311        // Entry point level comparison
312        let args = vec![BrilligParameter::SingleAddr(32)];
313        let entry1 = gen_brillig_for(main, args.clone(), &brillig1, &options1).unwrap();
314        let entry2 = gen_brillig_for(main, args, &brillig2, &options2).unwrap();
315
316        assert_equivalent_bytecode(&entry1.byte_code, &entry2.byte_code, &options1, &options2);
317    }

448    /// Minimal arithmetic spill test. Frame=6 gives 4 usable slots (start_offset=2).
449    /// Two params (v0, v1) use 2 slots, leaving 2 free. Computing v2, v3 fills the
450    /// frame. Computing v4 requires a constant temp that pushes past 4, forcing a spill.
451    ///

459            v2 = unchecked_add v0, v1
460            v3 = unchecked_add v0, u32 2
461            v4 = unchecked_add v1, u32 3
462            v5 = unchecked_add v2, v3
463            v6 = unchecked_add v5, v4
464            return v6
465        }
466        ";
467
468        let calldata = vec![FieldElement::from(3u64), FieldElement::from(5u64)];
469        let args = vec![BrilligParameter::SingleAddr(32), BrilligParameter::SingleAddr(32)];

686    /// Cross-block spill in a loop. Frame=10 gives 8 usable slots. In b0, the
687    /// immediate dominator also allocates b1's params (v5, v6), so baseline overhead is
688    /// {v0, v1, v5, v6} = 4 of 8. Computing v2..v4 plus constants fills the
689    /// frame, and v0 (LRU) gets spilled.
690    ///
691    /// In b2, v0 is reloaded from its spill slot. The spill slot must NOT be
692    /// freed on reload, because subsequent loop iterations need to reload from
693    /// the same slot.
694    ///
695    /// v0=3, v1=5:
696    /// v2=8, v3=7, v4=8+7=15
697    /// Loop 3 times: 15+3=18, 18+3=21, 21+3=24
698    #[test]
699    fn cross_block_spill_in_loop() {
700        let src = "
701        brillig(inline) fn main f0 {
702          b0(v0: u32, v1: u32):
703            v2 = unchecked_add v0, v1
704            v3 = unchecked_add v1, u32 2
705            v4 = unchecked_add v2, v3
706            jmp b1(v4, u32 0)
707          b1(v5: u32, v6: u32):
708            v7 = lt v6, u32 3
709            jmpif v7 then: b2, else: b3
710          b2():
711            v8 = unchecked_add v5, v0

714          b3():
715            return v5
716        }
717        ";
718
719        let calldata = vec![FieldElement::from(3u64), FieldElement::from(5u64)];
720        let args = vec![BrilligParameter::SingleAddr(32), BrilligParameter::SingleAddr(32)];
