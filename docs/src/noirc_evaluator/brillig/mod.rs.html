60/// Context structure for the Brillig pass.
61/// It stores Brillig-related data required for Brillig generation.
62#[derive(Default, Clone)]
63pub struct Brillig {
64    /// Maps SSA function labels to their brillig artifact
65    ssa_function_to_brillig: HashMap<FunctionId, BrilligArtifact<FieldElement>>,
66    call_stacks: CallStackHelper,
67    /// Bytecode for globals for each Brillig entry point.
68    globals: HashMap<FunctionId, BrilligArtifact<FieldElement>>,
69    /// The size of the global space for each Brillig entry point.
70    globals_memory_size: HashMap<FunctionId, usize>,
71}

73impl Brillig {
74    /// Compiles a function into brillig and store the compilation artifacts.
75    pub(crate) fn compile(
76        &mut self,
77        func: &Function,
78        options: &BrilligOptions,
79        globals: &HashMap<ValueId, BrilligVariable>,
80        hoisted_global_constants: &HashMap<(FieldElement, NumericType), BrilligVariable>,
81        is_entry_point: bool,
82    ) {
83        let obj = self.convert_ssa_function(
84            func,
85            options,
86            globals,
87            hoisted_global_constants,
88            is_entry_point,
89        );
90        self.ssa_function_to_brillig.insert(func.id(), obj);
91    }
92
93    /// Finds a brillig artifact by its label.
94    pub(crate) fn find_by_label(
95        &self,
96        function_label: Label,
97        options: &BrilligOptions,
98        stack_start: usize,
99    ) -> Option<Cow<BrilligArtifact<FieldElement>>> {
100        match function_label.label_type {
101            LabelType::Function(function_id, _) => {
102                self.ssa_function_to_brillig.get(&function_id).map(Cow::Borrowed)
103            }
104            // Procedures are compiled as needed
105            LabelType::Procedure(procedure_id) => {
106                Some(Cow::Owned(compile_procedure(procedure_id, options, stack_start)))
107            }
108            LabelType::GlobalInit(function_id) => self.globals.get(&function_id).map(Cow::Borrowed),
109            _ => unreachable!("ICE: Expected a function or procedure label"),
110        }
111    }

113    /// Convert an SSA function into Brillig bytecode.
114    pub(crate) fn convert_ssa_function(
115        &mut self,
116        func: &Function,
117        options: &BrilligOptions,
118        globals: &HashMap<ValueId, BrilligVariable>,
119        hoisted_global_constants: &HashMap<(FieldElement, NumericType), BrilligVariable>,
120        is_entry_point: bool,
121    ) -> BrilligArtifact<FieldElement> {
122        let mut brillig_context = BrilligContext::new(func.name(), options);
123
124        let mut function_context = FunctionContext::new(func, is_entry_point);
125
126        brillig_context.enter_context(Label::function(func.id()));
127
128        brillig_context.call_check_max_stack_depth_procedure();
129
130        for block in function_context.reverse_post_order().collect::<Vec<_>>() {
131            BrilligBlock::compile_block(
132                &mut function_context,
133                &mut brillig_context,
134                block,
135                &func.dfg,
136                &mut self.call_stacks,
137                globals,
138                hoisted_global_constants,
139            );
140        }
141
142        if options.show_opcode_advisories {
143            let opcode_advisories =
144                brillig_check::opcode_advisories(func, &function_context, &brillig_context);
145
146            brillig_check::show_opcode_advisories(&opcode_advisories, brillig_context.artifact());
147        }
148
149        brillig_context.into_artifact()
150    }
151
152    pub fn call_stacks(&self) -> &CallStackHelper {
153        &self.call_stacks
154    }
155}

164impl Ssa {
165    /// Compile Brillig functions and ACIR functions reachable from them.
166    #[tracing::instrument(level = "trace", skip_all)]
167    pub fn to_brillig(&self, options: &BrilligOptions) -> Brillig {
168        // Collect all the function ids that are reachable from Brillig.
169        // That means all the functions marked as Brillig and ACIR functions called by them,
170        // but we should already have monomorphized ACIR functions as a Brillig as well.
171        let brillig_reachable_function_ids = self
172            .functions
173            .iter()
174            .filter_map(|(id, func)| func.runtime().is_brillig().then_some(*id))
175            .collect::<BTreeSet<_>>();
176
177        let mut brillig = Brillig::default();
178
179        if brillig_reachable_function_ids.is_empty() {
180            return brillig;
181        }
182
183        // SSA Globals are computed once at compile time and shared across all functions,
184        // thus we can just fetch globals from the main function.
185        // This same globals graph will then be used to declare Brillig globals for the respective entry points.
186        let globals = (*self.functions[&self.main_id].dfg.globals).clone();
187        let globals_dfg = DataFlowGraph::from(globals);
188
189        // Produce the globals Brillig bytecode and variable allocation for each entry point.
190        let brillig_globals = BrilligGlobals::init(self, self.main_id).declare_globals(
191            &globals_dfg,
192            &mut brillig,
193            options,
194        );
195
196        for brillig_function_id in brillig_reachable_function_ids {
197            let (globals_allocations, hoisted_constant_allocations) =
198                brillig_globals.get_brillig_globals(brillig_function_id);
199
200            let func = &self.functions[&brillig_function_id];
201            let is_entry_point = brillig_globals.is_entry_point(&brillig_function_id);
202
203            brillig.compile(
204                func,
205                options,
206                globals_allocations,
207                hoisted_constant_allocations,
208                is_entry_point,
209            );
210        }
211
212        brillig
213    }

233    fn assert_equivalent_bytecode<F: AcirField>(
234        bytecode1: &[Opcode<F>],
235        bytecode2: &[Opcode<F>],
236        options1: &BrilligOptions,
237        options2: &BrilligOptions,
238    ) {
239        assert_eq!(bytecode1.len(), bytecode2.len());
240
241        // Offset where stack starts
242        // This assumes the same SSA where we have a single global register and a single call data argument.
243        let stack_start = ReservedRegisters::len() + MAX_SCRATCH_SPACE + 1 + 1;
244
245        for (op1, op2) in bytecode1.iter().zip(bytecode2) {
246            if op1 != op2 {
247                match (op1, op2) {
248                    // All opcodes are expected to be stable across memory layouts except for a few that rely
249                    // on the max stack size and stack frame size (e.g., setting of the `free_memory_pointer` and stack bounds checks).
250                    (
251                        Opcode::Const { destination: dest1, bit_size: bits1, value: val1 },
252                        Opcode::Const { destination: dest2, bit_size: bits2, value: val2 },
253                    ) => {
254                        assert_eq!(dest1, dest2);
255                        assert_eq!(bits1, bits2);
256
257                        if *dest1 == ReservedRegisters::free_memory_pointer() {
258                            // free_memory_pointer depends on max_stack_size + stack_start
259                            // This is where the heap begins.
260                            let expected1 = options1.layout.max_stack_size() + stack_start;
261                            let expected2 = options2.layout.max_stack_size() + stack_start;
262
263                            assert_eq!(val1.to_u128(), expected1 as u128);
264                            assert_eq!(val2.to_u128(), expected2 as u128);
265                        } else {
266                            // Stack depth bound check
267                            // This is done by the CheckMaxStackDepth procedure
268                            let bound1 = options1.layout.max_stack_size()
269                                - options1.layout.max_stack_frame_size()
270                                + stack_start;
271                            let bound2 = options2.layout.max_stack_size()
272                                - options2.layout.max_stack_frame_size()
273                                + stack_start;
274
275                            assert_eq!(val1.to_u128(), bound1 as u128);
276                            assert_eq!(val2.to_u128(), bound2 as u128);
277                        }
278                    }
279                    _ => panic!("Unexpected opcode difference {op1} != {op2}"),
280                }
281            }
282        }
283    }

285    fn compiles_to_equivalent_bytecode(
286        ssa: &Ssa,
287        options1: BrilligOptions,
288        options2: BrilligOptions,
289    ) {
290        let main = ssa.main();
291
292        let brillig1 = ssa.to_brillig(&options1);
293        let brillig2 = ssa.to_brillig(&options2);
294
295        // SSA function level comparison
296        let code1 = &brillig1.ssa_function_to_brillig[&ssa.main_id].byte_code;
297        let code2 = &brillig2.ssa_function_to_brillig[&ssa.main_id].byte_code;
298        assert_equivalent_bytecode(code1, code2, &options1, &options2);
299
300        // Entry point level comparison
301        let args = vec![BrilligParameter::SingleAddr(32)];
302        let entry1 = gen_brillig_for(main, args.clone(), &brillig1, &options1).unwrap();
303        let entry2 = gen_brillig_for(main, args, &brillig2, &options2).unwrap();
304
305        assert_equivalent_bytecode(&entry1.byte_code, &entry2.byte_code, &options1, &options2);
306    }
