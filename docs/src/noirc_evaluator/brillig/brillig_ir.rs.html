1//! This module is an abstraction layer over `Brillig`.
2//! To allow for separation of concerns, it knows nothing
3//! about SSA types, and can therefore be tested independently.
4//! `brillig_gen` is therefore the module which combines both
5//! ssa types and types in this module.
6//! A similar paradigm can be seen with the `acir_ir` module.
7//!
8//! The brillig IR provides instructions and codegens.
9//! The instructions are low level operations that are printed via `debug_show`.
10//! They should emit few opcodes. Codegens on the other hand orchestrate the
11//! low level instructions to emit the desired high level operation.

186impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
187    /// Splits a two's complement signed integer in the sign bit and the absolute value.
188    /// For example, -6 i8 (11111010) is split to 00000110 (6, absolute value) and 1 (is_negative).
189    pub(crate) fn absolute_value(
190        &mut self,
191        num: SingleAddrVariable,
192        absolute_value: SingleAddrVariable,
193        result_is_negative: SingleAddrVariable,
194    ) {
195        let max_positive = self
196            .make_constant_instruction(((1_u128 << (num.bit_size - 1)) - 1).into(), num.bit_size);
197
198        // Compute if num is negative
199        self.binary_instruction(*max_positive, num, result_is_negative, BrilligBinaryOp::LessThan);
200
201        // Two's complement of num
202        let zero = self.make_constant_instruction(0_usize.into(), num.bit_size);
203        let twos_complement = self.allocate_single_addr(num.bit_size);
204        self.binary_instruction(*zero, num, *twos_complement, BrilligBinaryOp::Sub);
205
206        // absolute_value = result_is_negative ? twos_complement : num
207        self.codegen_branch(result_is_negative.address, |ctx, is_negative| {
208            if is_negative {
209                ctx.mov_instruction(absolute_value.address, twos_complement.address);
210            } else {
211                ctx.mov_instruction(absolute_value.address, num.address);
212            }
213        });
214    }
215
216    pub(crate) fn convert_signed_division(
217        &mut self,
218        left: SingleAddrVariable,
219        right: SingleAddrVariable,
220        result: SingleAddrVariable,
221    ) {
222        let left_is_negative = self.allocate_single_addr_bool();
223        let left_abs_value = self.allocate_single_addr(left.bit_size);
224
225        let right_is_negative = self.allocate_single_addr_bool();
226        let right_abs_value = self.allocate_single_addr(right.bit_size);
227
228        let result_is_negative = self.allocate_single_addr_bool();
229
230        // Compute both absolute values
231        self.absolute_value(left, *left_abs_value, *left_is_negative);
232        self.absolute_value(right, *right_abs_value, *right_is_negative);
233
234        // Perform the division on the absolute values
235        self.binary_instruction(
236            *left_abs_value,
237            *right_abs_value,
238            result,
239            BrilligBinaryOp::UnsignedDiv,
240        );
241
242        // Compute result sign
243        self.binary_instruction(
244            *left_is_negative,
245            *right_is_negative,
246            *result_is_negative,
247            BrilligBinaryOp::Xor,
248        );
249
250        self.codegen_branch(result_is_negative.address, |ctx, is_negative| {
251            if is_negative {
252                // If result has to be negative, perform two's complement
253                let zero = ctx.make_constant_instruction(0_usize.into(), result.bit_size);
254                ctx.binary_instruction(*zero, result, result, BrilligBinaryOp::Sub);
255            } else {
256                // else the result is positive and so it must be less than '2**(bit_size-1)'
257                let max = 1_u128 << (left.bit_size - 1);
258                let max = ctx.make_constant_instruction(max.into(), left.bit_size);
259                let no_overflow = ctx.allocate_single_addr_bool();
260                ctx.binary_instruction(result, *max, *no_overflow, BrilligBinaryOp::LessThan);
261                ctx.codegen_if_not(no_overflow.address, |ctx2| {
262                    ctx2.codegen_constrain(
263                        *no_overflow,
264                        Some("Attempt to divide with overflow".to_string()),
265                    );
266                });
267            }
268        });
269    }

457        //
458        // #[oracle(get_number_sequence)]
459        // unconstrained fn get_number_sequence(size: u32) -> Vec<u32> {
460        // }
461        //
462        // unconstrained fn main() -> Vec<u32> {
463        //   let the_sequence = get_number_sequence(12);
464        //   assert(the_sequence.len() == 12);
465        // }
466
467        // Enable debug trace so we can see what the bytecode is if the test fails.
468        let options = BrilligOptions {
469            enable_debug_trace: true,
470            enable_debug_assertions: true,
471            enable_array_copy_counter: false,
472            show_opcode_advisories: false,
473            layout: Default::default(),
474        };
475        let mut context = BrilligContext::new("test", &options);
476
477        // Allocate variables
478        let r_input_size = MemoryAddress::direct(assert_u32(ReservedRegisters::len()));
479        let r_output_ptr = r_input_size.offset(1);
480        let r_output_size = r_input_size.offset(2);
481        let r_equality = r_input_size.offset(3);
482
483        let r_free = ReservedRegisters::free_memory_pointer();
484        // Set the free memory pointer after the variables allocated above.
485        let r_free_value = ReservedRegisters::len() + 4;
486        context.usize_const_instruction(r_free, FieldElement::from(r_free_value));
487
488        context.usize_const_instruction(r_input_size, FieldElement::from(12_usize));
489        // The output pointer points at the heap.
490        context.usize_const_instruction(
491            r_output_ptr,
492            FieldElement::from(r_free_value + assert_usize(offsets::VECTOR_ITEMS)),
493        );
494        context.foreign_call_instruction(
495            "make_number_sequence".into(),
496            &[ValueOrArray::MemoryAddress(r_input_size)],
497            &[HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U32))],
498            &[ValueOrArray::HeapVector(HeapVector { pointer: r_output_ptr, size: r_output_size })],
499            &[HeapValueType::Vector {
500                value_types: vec![HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U32))],
501            }],
502        );
503
504        // check r_input_size == r_output_size
505        context.memory_op_instruction(
506            r_input_size,
507            r_output_size,
508            r_equality,
509            BrilligBinaryOp::Equals,
510        );
511        // We push a JumpIf and Trap opcode directly as the constrain instruction
512        // uses unresolved jumps which requires a block to be constructed in SSA and
513        // we don't need this for Brillig IR tests
514        context.push_opcode(BrilligOpcode::Const {
515            destination: MemoryAddress::direct(0),
516            bit_size: BitSize::Integer(IntegerBitSize::U32),
517            value: FieldElement::from(0u64),
518        });
519        // If we got the expected number of items, jump to the STOP, otherwise fall through to TRAP.
520        context.push_opcode(BrilligOpcode::JumpIf { condition: r_equality, location: 8 });

550    /// Test demonstrating that `initialize_constant_array_runtime` is protected from
551    /// iterator overflow by the VM's allocation overflow check (`process_free_memory_op`).
552    ///
553    /// `codegen_for_loop` uses wrapping arithmetic for iterator increments. If the iterator
554    /// could overflow before reaching the bound, it would cause an infinite loop.
555    /// However, this cannot happen through `initialize_constant_array_runtime` because:
556    ///
557    /// 1. Array allocation adds to `free_memory_pointer` via `BinaryIntOp::Add`
558    /// 2. VM's `process_free_memory_op` intercepts adds to register 1 and uses `checked_add`
559    /// 3. If allocation would overflow, VM fails with "Out of memory" before the loop runs
560    ///
561    /// This test compiles SSA with a large constant array, then patches `free_memory_pointer`
562    /// to near u32::MAX to demonstrate that allocation triggers "Out of memory".
563    #[test]
564    fn initialize_constant_array_protected_by_allocation_check() {
565        // SSA with array of 11 identical tuples - triggers initialize_constant_array_runtime
566        let src = r#"
567            brillig(inline) predicate_pure fn main f0 {
568              b0(v0: u32):
569                v1 = make_array [
570                    u32 42, u32 42, u32 42,
571                    u32 42, u32 42, u32 42,
572                    u32 42, u32 42, u32 42,
573                    u32 42, u32 42, u32 42,
574                    u32 42, u32 42, u32 42,
575                    u32 42, u32 42, u32 42,
576                    u32 42, u32 42, u32 42,
577                    u32 42, u32 42, u32 42,
578                    u32 42, u32 42, u32 42,
579                    u32 42, u32 42, u32 42,
580                    u32 42, u32 42, u32 42
581                ] : [(u32, u32, u32); 11]
582                return
583            }
584        "#;
585
586        let ssa = Ssa::from_str(src).unwrap();
587        let main = ssa.main();
588        let options = BrilligOptions::default();
589        let brillig = ssa.to_brillig(&options);

594        // and insert a patch just before it.
595        // We patch the opcode rather than using the memory layout config because if we indicate a heap start
596        // near u32::MAX we will fail with a trap in the CheckMaxStackDepth procedure.
597        // Patching the initialization of the FMP lets us simulate prior allocations exhausting memory without having
598        // to actually allocate GBs of memory for this test.
599        let mut insert_pos = None;
600        for (i, opcode) in generated.byte_code.iter().enumerate() {
601            if let BrilligOpcode::BinaryIntOp { destination, op, .. } = opcode {

603                if *destination == ReservedRegisters::free_memory_pointer()
604                    && *op == BinaryIntOp::Add
605                {
606                    insert_pos = Some(i);
607                    break;
608                }
609            }
610        }
611        let insert_pos = insert_pos.expect("Should find BinaryIntOp::Add to FMP");
612
613        // Insert opcode to set free_memory_pointer near u32::MAX
614        // This simulates what would happen if prior allocations exhausted memory
615        let patch_opcode = BrilligOpcode::Const {
616            destination: ReservedRegisters::free_memory_pointer(), // FREE_MEMORY_POINTER_ADDRESS
617            value: FieldElement::from(u32::MAX - 10), // Near max, allocation will overflow
618            bit_size: BitSize::Integer(IntegerBitSize::U32),
619        };

622        let mut vm = VM::new(
623            vec![FieldElement::from(0u64)],
624            &generated.byte_code,
625            &DummyBlackBoxSolver,
626            false,
627            None,
628        );
629
630        let status = vm.process_opcodes();
631
632        let VMStatus::Failure {
633            reason: acvm::brillig_vm::FailureReason::RuntimeError { message },
634            ..
635        } = status
636        else {
637            panic!("Expected 'Out of memory' error from allocation overflow, got: {status:?}")
638        };

642    /// Test proving [BrilligContext::codegen_mem_copy_from_the_end] handles `num_elements=0`.
643    ///
644    /// See the function's `# Safety` documentation for why the underflow is safe.
645    /// This test directly verifies the loop exit condition is immediately `true`.
646    #[test]
647    fn mem_copy_from_end_zero_elements_condition_is_immediately_true() {
648        let options = BrilligOptions {
649            enable_debug_trace: false,
650            enable_debug_assertions: true,
651            enable_array_copy_counter: false,
652            show_opcode_advisories: false,
653            layout: Default::default(),
654        };
655        let mut context = BrilligContext::new("test", &options);
656
657        // Allocate direct memory addresses for our variables (after reserved registers)
658        let source_start = MemoryAddress::direct(assert_u32(ReservedRegisters::len()));
659        let target_start = source_start.offset(1); // Direct(4)
660        let num_elements_addr = source_start.offset(2); // Direct(5)
661        let num_elements = SingleAddrVariable::new_usize(num_elements_addr);
662
663        // Initialize reserved registers
664        // Stack pointer points to start of stack space (after our variables)
665        // After source_start, target_start, num_elements
666        let stack_start = assert_u32(ReservedRegisters::len()) + 3;
667        context.const_instruction(
668            SingleAddrVariable::new_usize(ReservedRegisters::stack_pointer()),
669            stack_start.into(),
670        );
671        // usize_one is used by codegen_usize_op for constant 1
672        context.const_instruction(
673            SingleAddrVariable::new_usize(ReservedRegisters::usize_one()),
674            1_usize.into(),
675        );
676
677        // Initialize: source_start=100, target_start=200, num_elements=0
678        // source_start > 0 is required for the transitive protection to work
679        context.usize_const_instruction(source_start, 100_usize.into());
680        context.usize_const_instruction(target_start, 200_usize.into());
681        context.const_instruction(num_elements, FieldElement::from(0_u32));
682
683        // Call the function under test
684        context.codegen_mem_copy_from_the_end(source_start, target_start, num_elements);
685
686        // Stop successfully
687        context.stop_instruction(HeapVector {
688            pointer: MemoryAddress::direct(0),
689            size: MemoryAddress::direct(0),
690        });
691
692        let bytecode = context.into_artifact().finish().byte_code;
693
694        // Find the LessThan opcode (the loop exit condition check) and its destination register.
695        // There should be exactly one LessThan in codegen_mem_copy_from_the_end.
696        let (less_than_idx, condition_dest) = bytecode
697            .iter()
698            .enumerate()
699            .find_map(|(idx, op)| {
700                if let BrilligOpcode::BinaryIntOp {
701                    op: BinaryIntOp::LessThan, destination, ..
702                } = op
703                {
704                    Some((idx, *destination))
705                } else {
706                    None
707                }
708            })
709            .expect("Should find LessThan opcode in bytecode");
710
711        // Step through opcodes until we've executed the LessThan operation
712        let mut vm = VM::new(vec![], &bytecode, &DummyBlackBoxSolver, false, None);
713        for _ in 0..=less_than_idx {
714            vm.process_opcode();
715        }
716
717        // Read the condition register value from memory.
718        // After the LessThan executes, this should be true (1) if the protection works.
719        let memory = vm.get_memory();
720        let condition_addr = match condition_dest {
721            MemoryAddress::Relative(offset) => {
722                // For relative addresses, we need to add the stack pointer value
723                let stack_ptr_address = assert_usize(ReservedRegisters::stack_pointer().to_u32());
724                let stack_ptr_value = match memory[stack_ptr_address] {
725                    MemoryValue::U32(v) => v as usize,
726                    _ => panic!("Expected stack pointer to be U32"),
727                };
728                stack_ptr_value + offset as usize
729            }
730            MemoryAddress::Direct(_) => {
731                panic!("Expected condition destination to be a relative address")
732            }
733        };
734
735        let condition_value = &memory[condition_addr];
736
737        // The condition should be true (U1(true)) because:
738        // source_pointer = source_start + (2^32 - 1) = source_start - 1 (wrapping)
739        // source_start - 1 < source_start = true
740        assert_eq!(
741            *condition_value,
742            MemoryValue::U1(true),
743            "Loop exit condition should be immediately true when num_elements=0"
744        );
745
746        // We will process a JmpIf based upon the condition which will immediately jump us to a Stop
747        for _ in less_than_idx..(less_than_idx + 2) {
748            let opcode = &bytecode[vm.program_counter()];
749            match opcode {
750                BrilligOpcode::Load { .. } | BrilligOpcode::Store { .. } => {
751                    panic!("We are performing a mem copy when it should have been skipped");
752                }
753                _ => {}
754            }
755            vm.process_opcode();
756        }
757
758        let status = vm.get_status();
759        // The VM successfully finished executing
760        assert_eq!(status, VMStatus::Finished { return_data_offset: 6, return_data_size: 6 });
761    }

905    ///   1. mov reg(v2), reg(v1) — overwrites old v3

916                v5 = lt v3, u32 17
917                jmpif v5 then: b2, else: b3
918              b2():
919                v8 = unchecked_add v3, u32 1
920                jmp b1(v8, v3, u32 10)
921              b3():
922                constrain v35 == u32 16
923                constrain v36 == u32 10
924                return
925            }
926        "#;

936        assert_eq!(
