30/// Generates a complete Brillig entry point artifact for a given SSA-level [Function], linking all dependencies.
31///
32/// This function is responsible for generating a final Brillig artifact corresponding to a compiled SSA [Function].
33/// It sets up the entry point context, registers input/output parameters, and recursively resolves and links
34/// all transitive Brillig function dependencies.
35///
36/// # Parameters
37/// - func: The SSA [Function] to compile as the entry point.
38/// - arguments: Brillig-compatible [BrilligParameter] inputs to the function
39/// - brillig: The [context structure][Brillig] of all known Brillig artifacts for dependency resolution.
40/// - options: Brillig compilation options (e.g., debug trace settings).
41///
42/// # Returns
43/// - Ok([GeneratedBrillig]): Fully linked artifact for the entry point that can be executed as a Brillig program.
44/// - Err([RuntimeError]): If the return value exceeds the witness limit or linking fails
45///
46/// # Panics
47/// - If the global memory size for the function has not been precomputed.
48pub(crate) fn gen_brillig_for(
49    func: &Function,
50    arguments: Vec<BrilligParameter>,
51    brillig: &Brillig,
52    options: &BrilligOptions,
53) -> Result<GeneratedBrillig<FieldElement>, RuntimeError> {
54    let return_parameters = FunctionContext::return_values(func);
55
56    // Check if the return value size exceeds the limit before generating the entry point.
57    // This is done early to avoid the expensive entry point codegen which iterates over
58    // each element in the return arrays.
59    func.dfg.get_num_return_witnesses(func)?;
60
61    // Create the entry point artifact
62    let globals_memory_size = brillig
63        .globals_memory_size
64        .get(&func.id())
65        .copied()
66        .expect("Should have the globals memory size specified for an entry point");
67
68    let options = BrilligOptions { enable_debug_trace: false, ..*options };
69
70    let (mut entry_point, stack_start) = BrilligContext::new_entry_point_artifact(
71        arguments,
72        return_parameters,
73        func.id(),
74        true,
75        globals_memory_size,
76        func.name(),
77        &options,
78    );
79
80    // Link the entry point with all dependencies
81    while let Some(unresolved_fn_label) = entry_point.first_unresolved_function_call() {
82        let artifact = &brillig.find_by_label(unresolved_fn_label.clone(), &options, stack_start);
83        let artifact = match artifact {
84            Some(artifact) => artifact,
85            None => {
86                return Err(InternalError::General {
87                    message: format!("Cannot find linked fn {unresolved_fn_label}"),
88                    call_stack: CallStack::new(),
89                }
90                .into());
91            }
92        };
93        entry_point.link_with(artifact);
94        // Insert the range of opcode locations occupied by a procedure
95        if let Some(procedure_id) = &artifact.procedure {
96            let num_opcodes = entry_point.byte_code.len();
97            let previous_num_opcodes = entry_point.byte_code.len() - artifact.byte_code.len();
98            // We subtract one as to keep the range inclusive on both ends
99            entry_point
100                .procedure_locations
101                .insert(procedure_id.clone(), (previous_num_opcodes, num_opcodes - 1));
102        }
103    }
104    // Generate the final bytecode
105    Ok(entry_point.finish())
106}

120        let src = "
121        brillig(inline) fn main f0 {
122          b0(v0: u32, v1: u32):
123            v2 = add v0, v1
124            return v2
125        }
126        ";
127        let ssa = Ssa::from_str(src).unwrap();
128        let options = BrilligOptions::default();
129        let brillig = ssa.to_brillig(&options);
130
131        let args = vec![BrilligParameter::SingleAddr(32), BrilligParameter::SingleAddr(32)];
132        let entry = gen_brillig_for(ssa.main(), args, &brillig, &options).unwrap();
133
134        // foo is a very simple function returning the addition of its inputs, which is done at line 18.
135        // The rest of the code is some overhead added to every brillig function for handling inputs, outputs, globals, ...
136        //
137        // Here is a breakdown of the code:
138        // Line 1-9: The function starts with the handling of the inputs via the 'calldata copy'
139        // The inputs are put in registers 1 and 2
140        // Line 10: Calling the part that deals with brillig globals. In this simple example, there are no globals so
141        // it just returns immediately.
142        // Line 11: Calling the check_max_stack_depth_procedure (Lines 24-29), via the call 24. Returning from
143        // this procedure is going to line 18 (right after the call 24 on line 17)
144        // Line 18: the actual body of the function, which is a single add instruction.
145        // Line 19-21: An overflow check on the addition, which will trap if the addition overflowed.
146        // Line 22: Moving the result of the addition into the return register (register 1)
147        // Line 23: Returning from the function body, which goes back to line 12
148        // Line 12-15: Handling the return data, moving the return value into the right place in memory and stopping execution.
149        assert_artifact_snapshot!(entry, @r"

167        16: return

172        21: call 30
