33/// Generates a complete Brillig entry point artifact for a given SSA-level [Function], linking all dependencies.
34///
35/// This function is responsible for generating a final Brillig artifact corresponding to a compiled SSA [Function].
36/// It sets up the entry point context, registers input/output parameters, and recursively resolves and links
37/// all transitive Brillig function dependencies.
38///
39/// # Parameters
40/// - func: The SSA [Function] to compile as the entry point.
41/// - arguments: Brillig-compatible [BrilligParameter] inputs to the function
42/// - brillig: The [context structure][Brillig] of all known Brillig artifacts for dependency resolution.
43/// - options: Brillig compilation options (e.g., debug trace settings).
44///
45/// # Returns
46/// - Ok([GeneratedBrillig]): Fully linked artifact for the entry point that can be executed as a Brillig program.
47/// - Err([RuntimeError]): If the return value exceeds the witness limit or linking fails
48///
49/// # Panics
50/// - If the global memory size for the function has not been precomputed.
51pub(crate) fn gen_brillig_for(
52    func: &Function,
53    arguments: Vec<BrilligParameter>,
54    brillig: &Brillig,
55    options: &BrilligOptions,
56) -> Result<GeneratedBrillig<FieldElement>, RuntimeError> {
57    let return_parameters = FunctionContext::return_values(func);
58
59    // Check if the return value size exceeds the limit before generating the entry point.
60    // This is done early to avoid the expensive entry point codegen which iterates over
61    // each element in the return arrays.
62    func.dfg.get_num_return_witnesses(func)?;
63
64    // Create the entry point artifact
65    let globals_memory_size = brillig
66        .globals_memory_size
67        .get(&func.id())
68        .copied()
69        .expect("Should have the globals memory size specified for an entry point");
70
71    let options = BrilligOptions { enable_debug_trace: false, ..*options };
72
73    let (mut entry_point, stack_start) = BrilligContext::new_entry_point_artifact(
74        arguments,
75        return_parameters,
76        func.id(),
77        globals_memory_size > 0,
78        globals_memory_size,
79        func.name(),
80        &options,
81    );
82
83    // Link the entry point with all dependencies
84    while let Some(unresolved_fn_label) = entry_point.first_unresolved_function_call() {
85        let artifact = &brillig.find_by_label(unresolved_fn_label.clone(), &options, stack_start);
86        let Some(artifact) = artifact else {
87            return Err(InternalError::General {
88                message: format!("Cannot find linked fn {unresolved_fn_label}"),
89                call_stack: CallStack::new(),
90            }
91            .into());
92        };
93        entry_point.link_with(artifact);
94        // Insert the range of opcode locations occupied by a procedure
95        if let Some(procedure_id) = &artifact.procedure {
96            let num_opcodes = entry_point.byte_code.len();
97            let previous_num_opcodes = entry_point.byte_code.len() - artifact.byte_code.len();
98            // We subtract one as to keep the range inclusive on both ends
99            entry_point
100                .procedure_locations
101                .insert(procedure_id.clone(), (previous_num_opcodes, num_opcodes - 1));
102        }
103    }
104    // Generate the final bytecode
105    Ok(entry_point.finish())
106}

122        let src = "
123        brillig(inline) fn main f0 {
124          b0(v0: u32, v1: u32):
125            v2 = add v0, v1
126            return v2
127        }
128        ";
129        let ssa = Ssa::from_str(src).unwrap();
130        let options = BrilligOptions::default();
131        let brillig = ssa.to_brillig(&options);
132
133        let args = vec![BrilligParameter::SingleAddr(32), BrilligParameter::SingleAddr(32)];
134        let entry = gen_brillig_for(ssa.main(), args, &brillig, &options).unwrap();
135
136        // A simple function returning the addition of its inputs (line 16).
137        // The rest is entry point overhead for handling inputs, outputs, and stack checks.
138        //
139        // Lines 0-2: Reserved register setup (usize_one, free_memory_pointer, stack_pointer).
140        // Lines 3-9: Calldata copy — the two u32 inputs are copied and cast into sp[2] and sp[3].
141        //            There is no globals init call because this function has no globals.
142        // Line 10: call 15 — jumps to the function body.
143        // Line 11: Moves the return value out of the function's register.
144        // Lines 12-14: Return data handling — places the result in memory and stops execution.
145        // Line 15: call 22 — check_max_stack_depth procedure (lines 22-27).
146        // Line 16: The actual add instruction.
147        // Lines 17-19: Overflow check on the addition; traps via lines 28-30 if it overflowed.
148        // Line 20: Moves the result into the return register.
