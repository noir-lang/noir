1//! Memory layout and register allocation for Brillig programs.
2//!
3//! Brillig execution splits its memory/register space into fixed and dynamic regions. The layout
4//! is strictly enforced by the register allocators defined in this module.
5//!
6//! The regions are as follows:
7//! 1. Reserved registers: Used internally by the VM for special purposes (e.g., stack pointer,
8//!    free memory pointer, etc.). These registers occupy the lowest memory addresses.
9//! 2. Scratch space: Temporary memory used for procedure arguments and return values, allowing
10//!    function calls without saving/restoring the stack. Scratch space uses direct addressing.
11//! 3. Globals: Read-only memory initialized at the beginning of the program. The size is
12//!    determined during global variable compilation and may vary per program. Starts immediately
13//!    after the scratch space.
14//! 4. Entry point: Memory region containing program input arguments and reserved space for return values.
15//!    Starts immediately after the globals and ends before the stack.
16//! 5. Stack: Dynamic stack frames used for function-local variables. Each function call
17//!    creates its own view of the stack. Stack uses relative addressing for stack frames.
18//! 6. Heap: Dynamic memory allocated after the stack for arrays, vectors, and other data
19//!    structures. Starts immediately after the stack region.
20//!
21//! This module contains:
22//! - [LayoutConfig]: Centralized configuration of maximum sizes for stack frames, total stack size,
23//!   and scratch space. All register allocators query this configuration to determine their memory bounds.
24//!   This config is meant to be immutable and provides the following benefits:
25//!   - Clear separation between memory layout policy and actual code generation.
26//!   - Unit tests to vary memory layouts and ensure bytecode remains consistent.
27//! - [RegisterAllocator]: Trait implemented by all memory region allocators. Each allocator is expected
28//!   to enforce its own bounds checks and allocation/deallocation logic.
29//! - [Stack], [ScratchSpace], and [GlobalSpace]: Register allocator implementations for each memory region.

51/// Defines the memory layout for Brillig programs.
52///
53/// Brillig execution splits its register space into fixed regions
54/// (reserved registers, scratch space, globals, calldata) and dynamic regions
55/// (stack, heap). This configuration structure centralizes the sizing rules for
56/// fixed regions and the stack.
57#[derive(Clone, Copy, Debug)]
58pub struct LayoutConfig {
59    max_stack_frame_size: usize,
60    max_stack_size: usize,
61    max_scratch_space: usize,
62}
63
64impl LayoutConfig {
65    pub fn new(
66        max_stack_frame_size: usize,
67        num_stack_frames: usize,
68        max_scratch_space: usize,
69    ) -> Self {
70        let max_stack_size = num_stack_frames * max_stack_frame_size;
71        Self { max_stack_frame_size, max_stack_size, max_scratch_space }
72    }
73
74    /// The maximum size of an individual stack frame.
75    pub(crate) fn max_stack_frame_size(&self) -> usize {
76        self.max_stack_frame_size
77    }
78
79    /// The overall maximum stack size is the maximum number of frames times the maximum size of an individual stack frame.
80    pub(crate) fn max_stack_size(&self) -> usize {
81        self.max_stack_size
82    }
83
84    pub(crate) fn max_scratch_space(&self) -> usize {
85        self.max_scratch_space
86    }
87
88    /// Start of the entry point region:
89    /// `{reserved} {scratch space} {globals} [call data]`
90    ///
91    /// Returns the point where `call data` can start.
92    pub(crate) fn entry_point_start(&self, globals_size: usize) -> usize {
93        ScratchSpace::end_with_layout(self) + globals_size
94    }
95
96    /// Start of the return data within the entry point region:
97    /// `{reserved} {scratch space} {globals} {call data} [return data]`
98    pub(crate) fn return_data_start(&self, globals_size: usize, calldata_size: usize) -> usize {
99        self.entry_point_start(globals_size) + calldata_size
100    }
101}

103// These constants represent expert chosen defaults that are appropriate for the majority of programs
104pub const NUM_STACK_FRAMES: usize = 16;
105pub const MAX_STACK_FRAME_SIZE: usize = 2048;
106pub const MAX_SCRATCH_SPACE: usize = 64;
107
108impl Default for LayoutConfig {
109    fn default() -> Self {
110        Self::new(MAX_STACK_FRAME_SIZE, NUM_STACK_FRAMES, MAX_SCRATCH_SPACE)
111    }

114pub(crate) trait RegisterAllocator {
115    /// First valid memory address
116    fn start(&self) -> usize;
117    /// Last valid memory address
118    fn end(&self) -> usize;
119    /// Allocates a new register.
120    fn allocate_register(&mut self) -> MemoryAddress;
121    /// Push a register to the deallocation list, ready for reuse.
122    fn deallocate_register(&mut self, register: MemoryAddress);
123    /// Ensures a register is allocated, allocating it if necessary.
124    fn ensure_register_is_allocated(&mut self, register: MemoryAddress);
125    /// Creates a new register context from a set of registers allocated previously.
126    fn from_preallocated_registers(
127        preallocated_registers: Vec<MemoryAddress>,
128        layout: LayoutConfig,
129    ) -> Self;
130    /// Finds the first register which is followed only by free registers.
131    ///
132    /// Always returns a [MemoryAddress::Relative] address.
133    fn empty_registers_start(&self) -> MemoryAddress;
134    /// Return the memory layout used by this allocator.
135    fn layout(&self) -> LayoutConfig;
136}

138/// Every brillig stack frame/call context has its own view of register space.
139/// This is maintained by copying these registers to the stack during calls and reading them back.
140pub(crate) struct Stack {
141    storage: DeallocationListAllocator,
142    layout: LayoutConfig,
143}
144
145impl Stack {
146    pub(crate) fn new(layout: LayoutConfig) -> Self {
147        Self { storage: DeallocationListAllocator::new(Self::start()), layout }
148    }
149
150    /// Check if a `Relative` address is within the bounds of the stack.
151    ///
152    /// Panics if the address is `Direct`.
153    fn is_within_bounds(&self, register: MemoryAddress) -> bool {
154        let offset = assert_usize(register.unwrap_relative());
155        offset >= self.start() && offset < self.end()
156    }
157
158    /// Static start address.
159    ///
160    /// The addressable space starts at offset 1; at offset 0 is the previous stack pointer.
161    pub(super) fn start() -> usize {
162        1
163    }
164}
165
166impl RegisterAllocator for Stack {
167    fn start(&self) -> usize {
168        Self::start()
169    }
170
171    fn end(&self) -> usize {
172        self.layout.max_stack_frame_size()
173    }
174
175    fn ensure_register_is_allocated(&mut self, register: MemoryAddress) {
176        assert!(self.is_within_bounds(register), "Register out of stack bounds");
177        self.storage.ensure_register_is_allocated(assert_usize(register.unwrap_relative()));
178    }
179
180    fn allocate_register(&mut self) -> MemoryAddress {
181        let allocated = MemoryAddress::relative(assert_u32(self.storage.allocate_register()));
182        assert!(self.is_within_bounds(allocated), "Stack frame too deep");
183        allocated
184    }
185
186    fn deallocate_register(&mut self, register: MemoryAddress) {
187        self.storage.deallocate_register(assert_usize(register.unwrap_relative()));
188    }
189
190    fn from_preallocated_registers(
191        preallocated_registers: Vec<MemoryAddress>,
192        layout: LayoutConfig,
193    ) -> Self {
194        let empty = Stack::new(layout);
195        for register in &preallocated_registers {
196            assert!(empty.is_within_bounds(*register), "Register out of stack bounds: {register}");
197        }
198
199        Self {
200            storage: DeallocationListAllocator::from_preallocated_registers(
201                empty.start(),
202                vecmap(preallocated_registers, |r| assert_usize(r.unwrap_relative())),
203            ),
204            layout,
205        }
206    }
207
208    fn empty_registers_start(&self) -> MemoryAddress {
209        MemoryAddress::relative(assert_u32(self.storage.empty_registers_start()))
210    }
211
212    fn layout(&self) -> LayoutConfig {
213        self.layout
214    }
215}

217/// Procedure arguments and returns are passed through scratch space.
218/// This avoids having to dump and restore the stack to call procedures.
219/// The scratch space is a much smaller set of memory cells.
220pub(crate) struct ScratchSpace {
221    storage: DeallocationListAllocator,
222    layout: LayoutConfig,
223}
224
225impl ScratchSpace {
226    pub(crate) fn new(layout: LayoutConfig) -> Self {
227        Self { storage: DeallocationListAllocator::new(Self::start()), layout }
228    }
229
230    /// Check if a `Direct` address is within the bounds of the scratch space.
231    ///
232    /// Panics if the address is `Relative`.
233    fn is_within_bounds(&self, register: MemoryAddress) -> bool {
234        let index = assert_usize(register.unwrap_direct());
235        index >= self.start() && index < self.end()
236    }
237
238    /// Return the end of `{reserved}` where `{scratch space}` can start.
239    pub(super) fn start() -> usize {
240        ReservedRegisters::len()
241    }
242
243    /// Return the end of `{reserved} {scratch space}`
244    pub(super) fn end_with_layout(layout: &LayoutConfig) -> usize {
245        ReservedRegisters::len() + layout.max_scratch_space()
246    }
247}
248
249impl RegisterAllocator for ScratchSpace {
250    fn start(&self) -> usize {
251        Self::start()
252    }
253
254    fn end(&self) -> usize {
255        Self::end_with_layout(&self.layout)
256    }
257
258    fn ensure_register_is_allocated(&mut self, register: MemoryAddress) {
259        assert!(self.is_within_bounds(register), "Register out of scratch space bounds");
260        self.storage.ensure_register_is_allocated(assert_usize(register.unwrap_direct()));
261    }
262
263    fn allocate_register(&mut self) -> MemoryAddress {
264        let allocated = MemoryAddress::direct(assert_u32(self.storage.allocate_register()));
265        assert!(self.is_within_bounds(allocated), "Scratch space too deep");
266        allocated
267    }
268
269    fn deallocate_register(&mut self, register: MemoryAddress) {
270        self.storage.deallocate_register(assert_usize(register.unwrap_direct()));
271    }
272
273    fn from_preallocated_registers(
274        preallocated_registers: Vec<MemoryAddress>,
275        layout: LayoutConfig,
276    ) -> Self {
277        let empty = Self::new(layout);
278        for register in &preallocated_registers {
279            assert!(empty.is_within_bounds(*register), "Register out of scratch space bounds");
280        }
281
282        Self {
283            storage: DeallocationListAllocator::from_preallocated_registers(
284                empty.start(),
285                vecmap(preallocated_registers, |r| assert_usize(r.unwrap_direct())),
286            ),
287            layout,
288        }
289    }
290
291    fn empty_registers_start(&self) -> MemoryAddress {
292        MemoryAddress::direct(assert_u32(self.storage.empty_registers_start()))
293    }
294
295    fn layout(&self) -> LayoutConfig {
296        self.layout
297    }

300/// Globals have a separate memory space
301/// This memory space is initialized once at the beginning of a program
302/// and is read-only.
303#[derive(Default)]
304pub(crate) struct GlobalSpace {
305    storage: DeallocationListAllocator,
306    max_memory_address: usize,
307    layout: LayoutConfig,
308}
309
310impl GlobalSpace {
311    pub(crate) fn new(layout: LayoutConfig) -> Self {
312        let start = Self::start_with_layout(&layout);
313        Self { storage: DeallocationListAllocator::new(start), max_memory_address: start, layout }
314    }
315
316    /// Expand the global space to fit a new register if necessary.
317    fn expand_max_address(&mut self, register: MemoryAddress) {
318        let index = assert_usize(register.unwrap_direct());
319        assert!(index >= self.start(), "Global space malformed");
320        if index > self.max_memory_address {
321            self.max_memory_address = index;
322        }
323    }
324
325    /// Shrink the global space to minimize unused space at the end
326    /// if a temporary register gets deallocated.
327    fn shrink_max_address(&mut self, register: MemoryAddress) {
328        let index = assert_usize(register.unwrap_direct());
329        assert!(index >= self.start(), "Global space malformed");
330        if index == self.max_memory_address {
331            let empty_start = assert_usize(self.empty_registers_start().unwrap_direct());
332            self.max_memory_address =
333                empty_start.saturating_sub(1).max(self.storage.start_register_index);
334        }
335    }
336
337    /// The maximum address occupied by global variables *after* initialization.
338    ///
339    /// Note that during initialization temporary variables may be allocated beyond this address,
340    /// so it is important that the memory after the globals is not populated with data before
341    /// the globals have been initialized, or that data could be overwritten by the init.
342    pub(super) fn max_memory_address(&self) -> usize {
343        self.max_memory_address
344    }
345
346    /// Computes the first valid memory address for global space.
347    ///
348    /// The `{global space}` starts after the `{scratch space}`.
349    pub(crate) fn start_with_layout(layout: &LayoutConfig) -> usize {
350        ScratchSpace::end_with_layout(layout)
351    }
352}
353
354impl RegisterAllocator for GlobalSpace {
355    fn start(&self) -> usize {
356        Self::start_with_layout(&self.layout)
357    }
358
359    fn end(&self) -> usize {
360        unreachable!("The global space is set by the program");
361    }
362
363    fn allocate_register(&mut self) -> MemoryAddress {
364        let allocated = MemoryAddress::direct(assert_u32(self.storage.allocate_register()));
365        self.expand_max_address(allocated);
366        allocated
367    }
368
369    fn deallocate_register(&mut self, register: MemoryAddress) {
370        self.storage.deallocate_register(assert_usize(register.unwrap_direct()));
371        self.shrink_max_address(register);
372    }
373
374    fn ensure_register_is_allocated(&mut self, register: MemoryAddress) {
375        self.expand_max_address(register);
376        self.storage.ensure_register_is_allocated(assert_usize(register.unwrap_direct()));
377    }
378
379    fn from_preallocated_registers(
380        _preallocated_registers: Vec<MemoryAddress>,
381        _layout: LayoutConfig,
382    ) -> Self {
383        unimplemented!("`GlobalSpace` does not implement `from_preallocated_registers")
384    }
385
386    fn empty_registers_start(&self) -> MemoryAddress {
387        MemoryAddress::direct(assert_u32(self.storage.empty_registers_start()))
388    }
389
390    fn layout(&self) -> LayoutConfig {
391        self.layout
392    }
393}

395/// Register allocator that tracks a contiguous range of register slots using:
396/// * the highest ever allocated register slot, and
397/// * the set of deallocated registers.
398///
399/// When asked to allocate a new register, it returns a deallocated one,
400/// or expands the high water mark of registers.
401#[derive(Default)]
402struct DeallocationListAllocator {
403    /// A free-list of registers that have been deallocated and can be used again.
404    deallocated_registers: BTreeSet<usize>,
405    /// A usize indicating the next unused register.
406    next_free_register_index: usize,
407    /// Original start index.
408    start_register_index: usize,
409}
410
411impl DeallocationListAllocator {
412    /// Create a new allocator, starting a specific slot (could be direct or relative).
413    fn new(start: usize) -> Self {
414        Self {
415            deallocated_registers: BTreeSet::new(),
416            next_free_register_index: start,
417            start_register_index: start,
418        }
419    }
420
421    /// Ensure that a register index is allocated by:
422    /// * expanding the range, if the index is beyond the current maximum, or
423    /// * removing it from the deallocated registers.
424    ///
425    /// If the register is already allocated (ie. it fits and isn't deallocated) then nothing happens.
426    fn ensure_register_is_allocated(&mut self, index: usize) {
427        if index < self.next_free_register_index {
428            // If it could be allocated, check if it's in the deallocated list and remove it from there
429            self.deallocated_registers.remove(&index);
430        } else {
431            // If it couldn't yet be, expand the register space.
432            self.next_free_register_index = index + 1;
433        }
434    }
435
436    /// Returns the first deallocated register, or expands the range to fit a new one.
437    fn allocate_register(&mut self) -> usize {
438        // If we have a register in our free list of deallocated registers,
439        // consume it first. This prioritizes reuse.
440        if let Some(register) = self.deallocated_registers.pop_first() {
441            return register;
442        }
443        // Otherwise, move to our latest register.
444        let register = self.next_free_register_index;
445        self.next_free_register_index += 1;
446        register
447    }
448
449    /// Adds the register to the deallocation list.
450    ///
451    /// Panics if:
452    /// * The register is already deallocated.
453    /// * The register is not within the tracked range.
454    fn deallocate_register(&mut self, register_index: usize) {
455        assert!(
456            !self.deallocated_registers.contains(&register_index),
457            "register already deallocated"
458        );
459        assert!(register_index < self.next_free_register_index, "deallocate after next free");
460        assert!(register_index >= self.start_register_index, "deallocate before start");
461        self.deallocated_registers.insert(register_index);
462    }
463
464    /// Create an allocator that ranges from `start` to just after the highest
465    /// preallocated register, with everything not on that list considered deallocated
466    /// and ready to be used.
467    fn from_preallocated_registers(start: usize, preallocated_registers: Vec<usize>) -> Self {
468        let next_free_register_index = preallocated_registers.iter().fold(
469            start,
470            |free_register_index, &preallocated_register| {
471                free_register_index.max(preallocated_register + 1)
472            },
473        );
474
475        let mut deallocated_registers = (start..next_free_register_index).collect::<BTreeSet<_>>();
476        for pr in preallocated_registers {
477            deallocated_registers.remove(&pr);
478        }
479
480        Self { deallocated_registers, next_free_register_index, start_register_index: start }
481    }
482
483    /// Find the first free register after which there are only free registers.
484    fn empty_registers_start(&self) -> usize {
485        let mut first_free = self.next_free_register_index;
486        // Walk backwards toward the start, as long as we step on deallocated registers.
487        while first_free > self.start_register_index {
488            // If the next slot is _not_ deallocated, this is where free starts.
489            if !self.deallocated_registers.contains(&(first_free - 1)) {
490                break;
491            }
492            first_free -= 1;
493        }
494        first_free
495    }
496}

498impl<F, Registers: RegisterAllocator> BrilligContext<F, Registers> {
499    /// Read-only reference to the registers.
500    pub(crate) fn registers(&self) -> Ref<Registers> {
501        self.registers.borrow()
502    }
503
504    /// Read-write reference to the registers.
505    pub(crate) fn registers_mut(&self) -> RefMut<Registers> {
506        self.registers.borrow_mut()
507    }
508
509    /// Manually deallocates a register which is no longer in use.
510    ///
511    /// This could be one of the pre-allocated ones which doesn't have automation.
512    pub(crate) fn deallocate_register(&self, register: MemoryAddress) {
513        self.registers_mut().deallocate_register(register);
514    }
515
516    /// Allocates an unused register.
517    pub(crate) fn allocate_register(&self) -> Allocated<MemoryAddress, Registers> {
518        let addr = self.registers_mut().allocate_register();
519        Allocated::new_addr(addr, self.registers.clone())
520    }
521
522    /// Resets the registers to a new list of allocated ones.
523    pub(crate) fn set_allocated_registers(&mut self, allocated_registers: Vec<MemoryAddress>) {
524        self.registers = Rc::new(RefCell::new(Registers::from_preallocated_registers(
525            allocated_registers,
526            self.layout(),
527        )));
528    }
529
530    /// Allocate a [SingleAddrVariable].
531    pub(crate) fn allocate_single_addr(
532        &self,
533        bit_size: u32,
534    ) -> Allocated<SingleAddrVariable, Registers> {
535        self.allocate_register().map(|a| SingleAddrVariable::new(a, bit_size))
536    }
537
538    /// Allocate a [SingleAddrVariable] with the size of a Brillig memory address.
539    pub(crate) fn allocate_single_addr_usize(&self) -> Allocated<SingleAddrVariable, Registers> {
540        self.allocate_register().map(SingleAddrVariable::new_usize)
541    }
542
543    /// Allocate a [SingleAddrVariable] with a size of 1 bit.
544    pub(crate) fn allocate_single_addr_bool(&self) -> Allocated<SingleAddrVariable, Registers> {
545        self.allocate_single_addr(1)
546    }
547
548    /// Allocate a [SingleAddrVariable] with a size of `BRILLIG_MEMORY_ADDRESSING_BIT_SIZE` bit.
549    #[allow(unused)]
550    pub(crate) fn allocate_single_addr_mem(&self) -> Allocated<SingleAddrVariable, Registers> {
551        self.allocate_single_addr(BRILLIG_MEMORY_ADDRESSING_BIT_SIZE)
552    }
553
554    /// Allocate a pointer for [BrilligVector].
555    ///
556    /// This does not include allocating memory for the data on the heap or shaping the meta-data.
557    /// That is done by [BrilligContext::codegen_initialize_vector].
558    pub(crate) fn allocate_brillig_vector(&self) -> Allocated<BrilligVector, Registers> {
559        self.allocate_register().map(|a| BrilligVector { pointer: a })
560    }
561
562    /// Allocate a pointer for [BrilligArray].
563    ///
564    /// This does not include allocating memory for the data on the heap or shaping the meta-data.
565    /// That is done by [BrilligContext::codegen_initialize_array].
566    pub(crate) fn allocate_brillig_array(
567        &self,
568        size: SemiFlattenedLength,
569    ) -> Allocated<BrilligArray, Registers> {
570        self.allocate_register().map(|a| BrilligArray { pointer: a, size })
571    }
572
573    /// Allocate a [HeapVector].
574    pub(crate) fn allocate_heap_vector(&self) -> Allocated<HeapVector, Registers> {
575        let pointer = self.allocate_register();
576        let size = self.allocate_register();
577        pointer.map2(size, |pointer, size| HeapVector { pointer, size })
578    }
579
580    /// Allocate a [HeapArray].
581    pub(crate) fn allocate_heap_array(
582        &self,
583        size: SemiFlattenedLength,
584    ) -> Allocated<HeapArray, Registers> {
585        self.allocate_register().map(|pointer| HeapArray { pointer, size })
586    }
587
588    /// Create a number of consecutive [MemoryAddress::Direct] addresses at the start of the [ScratchSpace].
589    pub(crate) fn make_scratch_registers<const N: usize>(&self) -> [MemoryAddress; N] {
590        let scratch_start = ScratchSpace::start();
591        std::array::from_fn(|i| MemoryAddress::direct(assert_u32(scratch_start + i)))
592    }
593}

605/// Wrapper for a memory address which automatically deallocates itself
606/// when it goes out of scope.
607pub(crate) struct Allocated<T, R: RegisterAllocator> {
608    /// The inner structure is optional, so that we can map it to a different value
609    /// and let the original wrapper be dropped without any effect.
610    inner: Option<AllocatedInner<T, R>>,
611}
612
613/// Address list with enough slots for `HeapVector`.
614type AddressList = SmallVec<[MemoryAddress; 2]>;
615
616struct AllocatedInner<A, R> {
617    /// The value that we actually wanted to use, holding the allocated `MemoryAddress`.
618    value: A,
619    /// Addresses allocated.
620    addresses: AddressList,
621    /// Reference to the registers, for deallocation.
622    /// Optional so that pure values don't have to clone the registers.
623    registers: Option<Rc<RefCell<R>>>,
624}
625
626impl<A, R: RegisterAllocator> Drop for Allocated<A, R> {
627    fn drop(&mut self) {
628        let Some(mut inner) = self.inner.take() else {
629            return;
630        };
631        let Some(registers) = inner.registers.take() else {
632            return;
633        };
634        let addresses = std::mem::take(&mut inner.addresses);
635        let mut registers = registers.borrow_mut();
636        for addr in addresses {
637            registers.deallocate_register(addr);
638        }
639    }
640}

642/// Construct a new allocated address.
643impl<R: RegisterAllocator> Allocated<MemoryAddress, R> {
644    /// Wrap single allocated memory address.
645    fn new_addr(addr: MemoryAddress, registers: Rc<RefCell<R>>) -> Self {
646        Self::new(addr, smallvec![addr], registers)
647    }
648}
649
650impl<A, R: RegisterAllocator> Allocated<A, R> {
651    /// Wrap a value in [Allocated] without deallocating any address at the end.
652    pub(crate) fn pure(value: A) -> Self {
653        Self::from_inner(AllocatedInner { value, addresses: smallvec![], registers: None })
654    }
655
656    /// Create an [Allocated] value that deallocates its associated addresses at the end.
657    pub(crate) fn new(value: A, addresses: AddressList, registers: Rc<RefCell<R>>) -> Self {
658        Self::from_inner(AllocatedInner { value, addresses, registers: Some(registers) })
659    }
660
661    fn from_inner(inner: AllocatedInner<A, R>) -> Self {
662        Self { inner: Some(inner) }
663    }
664
665    /// Take the inner, leaving `self` ready to be dropped without any effect.
666    fn into_inner(mut self) -> AllocatedInner<A, R> {
667        self.inner.take().expect("allocated value already taken")
668    }
669
670    fn inner(&self) -> &AllocatedInner<A, R> {
671        self.inner.as_ref().expect("allocated value already taken")
672    }
673
674    fn inner_mut(&mut self) -> &mut AllocatedInner<A, R> {
675        self.inner.as_mut().expect("allocated value already taken")
676    }
677
678    /// Map the `value` to something else.
679    ///
680    /// The deallocation of addresses is deferred to when the new value goes out of scope.
681    pub(crate) fn map<B>(self, f: impl FnOnce(A) -> B) -> Allocated<B, R> {
682        let inner = self.into_inner();
683        Allocated::from_inner(AllocatedInner {
684            value: f(inner.value),
685            addresses: inner.addresses,
686            registers: inner.registers,
687        })
688    }
689
690    /// Map two values into one.
691    ///
692    /// The addresses of the two values are merged, with the implicit assumption that the
693    /// underlying registers are the same.
694    pub(crate) fn map2<B>(self, other: Self, f: impl FnOnce(A, A) -> B) -> Allocated<B, R> {
695        let inner = self.into_inner();
696        let other = other.into_inner();
697        Allocated::from_inner(AllocatedInner {
698            value: f(inner.value, other.value),
699            addresses: Self::merge_addresses(inner.addresses, other.addresses),
700            registers: inner.registers.or(other.registers),
701        })
702    }
703
704    fn merge_addresses(mut a: AddressList, mut b: AddressList) -> AddressList {
705        a.append(&mut b);
706        a.sort();
707        a.dedup();
708        a
709    }
710
711    /// Manually deallocate a register when it's no longer needed.
712    pub(crate) fn deallocate(self) {
713        drop(self);
714    }
715
716    /// Detach the allocated value from the registers, promising to take care of ownership in a different way.
717    ///
718    /// This is used when we allocate an address in a register, then create a new register where we consider the
719    /// value pre-allocated, and finally want to deallocate the address in the new register, not the old one,
720    /// so that we can reuse the memory for something else.
721    pub(crate) fn detach(self) -> A {
722        self.into_inner().value
723    }
724}
725
726impl<A, R: RegisterAllocator> Deref for Allocated<A, R> {
727    type Target = A;
728
729    fn deref(&self) -> &Self::Target {
730        &self.inner().value
731    }
732}
733
734impl<A, R: RegisterAllocator> DerefMut for Allocated<A, R> {
735    fn deref_mut(&mut self) -> &mut Self::Target {
736        &mut self.inner_mut().value
737    }
738}

785        let start = global.storage.start_register_index;
786        assert_eq!(global.max_memory_address(), start, "max initialized to start");
787
788        let reg1 = global.allocate_register();
789        let reg2 = global.allocate_register();
790        let reg3 = global.allocate_register();
791        let reg4 = global.allocate_register();
792        let max1 = global.max_memory_address();
793        assert_eq!(max1, start + 3, "max expands during allocation");
794
795        global.deallocate_register(reg3);
796        assert_eq!(
797            global.max_memory_address(),
798            start + 3,
799            "max does not shrink during when non-max deallocated"
800        );
801
802        global.deallocate_register(reg4);
803        assert_eq!(
804            global.max_memory_address(),
805            start + 1,
806            "max shrinks to precede first free address when max deallocated"
807        );
808
809        global.deallocate_register(reg2);
810        assert_eq!(
811            global.max_memory_address(),
812            start,
813            "max shrinks to start when we only have the first register"
814        );
815
816        global.deallocate_register(reg1);
817        assert_eq!(
818            global.max_memory_address(),
819            start,
820            "max shrinks to start when we have no registers"
821        );
822
823        let reg5 = global.allocate_register();
824        let reg6 = global.allocate_register();
825        assert_eq!(reg5, reg1, "earliest free register is reallocated");
826        assert_eq!(reg6, reg2, "earliest free register is reallocated");
827        assert_eq!(global.max_memory_address(), start + 1, "max expands again");
