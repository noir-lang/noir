12impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
13    /// Copy arguments to the [`ScratchSpace`] and call [`ProcedureId::PrepareVectorPush`].
14    ///
15    /// Prepares a vector for a push operation, allocating a larger vector and copying the source vector into the destination vector if necessary.
16    /// It returns the destination vector and the write pointer to where to put the new items.
17    ///
18    /// Brillig vectors have an inherent size and capacity, but their semantic length can be different,
19    /// and it's passed as a separate variable in SSA. When ACIR flattens vectors of unequal length,
20    /// it can be different from the size and capacity in the vector data structure itself.
21    pub(crate) fn call_prepare_vector_push_procedure(
22        &mut self,
23        source_len: SingleAddrVariable,
24        source_vector: BrilligVector,
25        destination_vector: BrilligVector,
26        write_pointer: MemoryAddress,
27        item_push_count: usize,
28        back: bool,
29    ) {
30        let [
31            source_vector_length_arg,
32            source_vector_pointer_arg,
33            item_push_count_arg,
34            destination_vector_pointer_return,
35            write_pointer_return,
36            did_rc_copy_return,
37        ] = self.make_scratch_registers();
38
39        self.mov_instruction(source_vector_length_arg, source_len.address);
40        self.mov_instruction(source_vector_pointer_arg, source_vector.pointer);
41        self.usize_const_instruction(item_push_count_arg, item_push_count.into());
42
43        self.add_procedure_call_instruction(ProcedureId::PrepareVectorPush(back));
44
45        self.mov_instruction(destination_vector.pointer, destination_vector_pointer_return);
46        self.mov_instruction(write_pointer, write_pointer_return);
47
48        // Count only RC copies (not necessary reallocations) by using the flag returned by the
49        // procedure rather than comparing pointers, which would also fire for reallocations.
50        self.codegen_count_if_nonzero(did_rc_copy_return);
51    }
52}

54/// Compile [`ProcedureId::PrepareVectorPush`].
55pub(super) fn compile_prepare_vector_push_procedure<F: AcirField + DebugToString>(
56    brillig_context: &mut BrilligContext<F, ScratchSpace>,
57    push_back: bool,
58) {
59    let [
60        source_vector_length_arg,
61        source_vector_pointer_arg,
62        item_push_count_arg,
63        destination_vector_pointer_return,
64        write_pointer_return,
65    ] = brillig_context.allocate_scratch_registers();
66
67    let did_rc_copy = allocate_rc_copy_flag(brillig_context);
68
69    let source_vector = BrilligVector { pointer: source_vector_pointer_arg };
70    let target_vector = BrilligVector { pointer: destination_vector_pointer_return };
71
72    let VectorMetaData {
73        rc: source_rc,
74        size: source_size,
75        capacity: source_capacity,
76        items_pointer: source_items_pointer,
77    } = brillig_context.codegen_read_vector_metadata(
78        source_vector,
79        Some((source_vector_length_arg, item_push_count_arg)),
80    );
81
82    // The target size is the source size plus the number of items we are pushing.
83    // Uses checked addition to trap on overflow.
84    let target_size = brillig_context.allocate_single_addr_usize();
85    brillig_context.codegen_checked_add(
86        source_size.address,
87        item_push_count_arg,
88        target_size.address,
89    );
90
91    // The strategy is to reallocate first and then depending if it's push back or not, copy the items or not.
92    reallocate_vector_for_insertion(
93        brillig_context,
94        source_vector,
95        *source_rc,
96        *source_capacity,
97        target_vector,
98        *target_size,
99        did_rc_copy,
100    );
101
102    // Get the pointer to the start of the items in the target vector.
103    // This is adjusted below based on whether we push to the front or the back.
104    let target_vector_items_pointer =
105        brillig_context.codegen_make_vector_items_pointer(target_vector);
106
107    if push_back {
108        // If we are pushing to the back, we could be reusing the source vector if the RC was 1 and it had excess capacity.
109        let was_reused = brillig_context.allocate_single_addr_bool();
110        brillig_context.memory_op_instruction(
111            source_vector.pointer,
112            target_vector.pointer,
113            was_reused.address,
114            BrilligBinaryOp::Equals,
115        );
116        // If we are not reusing the source, then we need to copy its items to the target.
117        brillig_context.codegen_if_not(was_reused.address, |brillig_context| {
118            brillig_context.codegen_mem_copy(
119                source_items_pointer.address,
120                *target_vector_items_pointer,
121                *source_size,
122            );
123        });
124        // Target vector is ready for push back at this point.
125        // The write pointer returned points after source-length number of items in the target vector.
126        brillig_context.memory_op_instruction(
127            *target_vector_items_pointer,
128            source_size.address,
129            write_pointer_return,
130            BrilligBinaryOp::Add,
131        );
132    } else {
133        // If pushing to the front we need to shift the items independently of it being reused or not.
134        let target_start = brillig_context.allocate_register();
135        // Shift items by the number of items we want to push to the front.
136        brillig_context.memory_op_instruction(
137            *target_vector_items_pointer,
138            item_push_count_arg,
139            *target_start,
140            BrilligBinaryOp::Add,
141        );
142        brillig_context.codegen_mem_copy_from_the_end(
143            source_items_pointer.address,
144            *target_start,
145            *source_size,
146        );
147        // The write pointer returned is the the first (now free) item in the target vector.
148        brillig_context.mov_instruction(write_pointer_return, *target_vector_items_pointer);
149    }
150}

152/// When copy-counting is enabled, reserve the scratch slot immediately after a vector
153/// procedure's argument registers to hold the "an RC copy occurred" flag, initialised to false.
154/// [`reallocate_vector_for_insertion`] sets it to true if and only if an RC (copy-on-write) copy,
155/// rather than a necessary capacity reallocation, is performed.
156///
157/// Returns `None` when copy-counting is disabled. In that case no register is reserved and no
158/// opcode is emitted, so the procedure's bytecode — including the addresses of the temporaries
159/// allocated afterwards — is identical to when the feature is off.
160pub(super) fn allocate_rc_copy_flag<F: AcirField + DebugToString>(
161    brillig_context: &mut BrilligContext<F, ScratchSpace>,
162) -> Option<MemoryAddress> {
163    brillig_context.count_array_copies().then(|| {
164        let flag = brillig_context.allocate_register().detach();
165        brillig_context.const_instruction(SingleAddrVariable::new(flag, 1), F::from(0_usize));
166        flag
167    })
168}

170/// Reallocates the target vector for insertion, skipping reallocation if the source vector can be reused:
171/// * if the capacity accommodates the target size:
172///   * if the RC is 1, we can increase the size of the source vector as reuse it as the destination
173///   * if the RC is not 1, we allocate a new destination vector with the source capacity and target size
174/// * if the capacity is too small for the target size, we allocate the destination vector with a capacity that is double the target size.
175///
176/// Does not copy the items, only reallocates the vector.
177///
178/// Whether a copy is necessary can be gleaned by comparing the target address to the source address.
179pub(crate) fn reallocate_vector_for_insertion<
180    F: AcirField + DebugToString,
181    Registers: RegisterAllocator,
182>(
183    brillig_context: &mut BrilligContext<F, Registers>,
184    source_vector: BrilligVector,
185    source_rc: SingleAddrVariable,
186    source_capacity: SingleAddrVariable,
187    target_vector: BrilligVector,
188    target_size: SingleAddrVariable,
189    did_rc_copy: Option<MemoryAddress>,
190) {
191    // If the source capacity is at least as large than the target size, we can potentially reuse the source vector to write the new items.
192    let does_capacity_fit = brillig_context.allocate_single_addr_bool();
193    brillig_context.memory_op_instruction(
194        target_size.address,
195        source_capacity.address,
196        does_capacity_fit.address,
197        BrilligBinaryOp::LessThanEquals,
198    );
199
200    // Reallocate target vector for insertion
201    brillig_context.codegen_branch(
202        does_capacity_fit.address,
203        |brillig_context, does_capacity_fit| {
204            if does_capacity_fit {
205                // We can only reuse the source vector if the ref-count is 1.
206                let is_rc_one = brillig_context.codegen_usize_equals_one(source_rc);
207
208                brillig_context.codegen_branch(is_rc_one.address, |brillig_context, is_rc_one| {
209                    if is_rc_one {
210                        // We can insert in place, so we can just move the source pointer to the destination pointer and update the length
211                        brillig_context
212                            .mov_instruction(target_vector.pointer, source_vector.pointer);
213                        brillig_context.codegen_update_vector_size(target_vector, target_size);
214                    } else {
215                        // Increase our array copy counter if that flag is set
216                        if brillig_context.count_array_copies() {
217                            brillig_context.codegen_increment_array_copy_counter();
218                        }
219                        // Signal to the call site that an RC copy occurred (not a mere reallocation).
220                        if let Some(flag) = did_rc_copy {
221                            brillig_context.const_instruction(
222                                SingleAddrVariable::new(flag, 1),
223                                F::from(1_usize),
224                            );
225                        }
226                        // We could not reuse the source vector, because there are other references to it.
227                        // Allocate a new vector with the target size and source capacity.
228                        brillig_context.codegen_initialize_vector(
229                            target_vector,
230                            target_size,
231                            Some(source_capacity),
232                        );
233                    }
234                });
235            } else {
236                // Capacity doubling with overflow protection.
237                // Uses checked multiplication to trap if target_size * 2 > u32::MAX.
238                let double_size = brillig_context.allocate_single_addr_usize();
239                brillig_context.codegen_checked_mul_with_constant(
240                    target_size.address,
241                    double_size.address,
242                    2_usize,
243                );
244
245                // We will have to copy the vector to the new expanded memory region.
246                brillig_context.codegen_initialize_vector(
247                    target_vector,
248                    target_size,
249                    Some(*double_size),
250                );
251
252                // Don't increment the array copy counter: we don't count necessary
253                // reallocations as copies
254            }
255        },
256    );
