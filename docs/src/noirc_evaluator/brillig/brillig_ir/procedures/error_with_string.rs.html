13impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
14    /// Reverts with the error selector of the given string
15    /// This procedure is useful to deduplicate generating code for the same selector.
16    pub(crate) fn call_error_with_string_procedure(&mut self, error_string: String) {
17        self.add_procedure_call_instruction(ProcedureId::RevertWithString(error_string));
18    }
19}
20
21pub(super) fn compile_error_with_string_procedure<F: AcirField + DebugToString>(
22    brillig_context: &mut BrilligContext<F, ScratchSpace>,
23    error_string: String,
24) {
25    // Compute the error selector and register the error type
26    let error_type = ErrorType::String(error_string);
27    let error_selector = error_type.selector();
28    brillig_context.obj.error_types.insert(error_selector, error_type);
29
30    // Write the selector to the free memory pointer and trap with it as revert data
31    brillig_context.indirect_const_instruction(
32        ReservedRegisters::free_memory_pointer(),
33        64,
34        u128::from(error_selector.as_u64()).into(),
35    );
36    brillig_context.trap_instruction(HeapVector {
37        pointer: ReservedRegisters::free_memory_pointer(),
38        size: ReservedRegisters::usize_one(),
39    });
40}
