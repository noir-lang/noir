11    /// Map sources to potentially multiple destinations.
12    /// This function moves values from a set of registers to another set of registers.
13    /// It assumes that:
14    /// - every destination needs to be written at most once. Will panic if not.
15    /// - destinations are relative addresses, starting from 1 and consecutively incremented by one. Will panic if not
16    /// - sources are any relative addresses, and can also be direct address to the global space. TODO: panic if not
17    /// - sources and destinations have same length. Will panic if not.
18    pub(crate) fn codegen_mov_registers_to_registers(
19        &mut self,
20        sources: &[MemoryAddress],
21        destinations: &[MemoryAddress],
22    ) {
23        assert_eq!(sources.len(), destinations.len(), "sources and destinations length must match");
24        let n = sources.len();
25        let mut processed = 0;
26        // Compute the number of destinations for each source node that is also a destination node (i.e within 0,..n-1) in the movement graph
27        let mut num_destinations = vec![0; n];
28        for i in 0..n {
29            // Check that destinations are relatives to 0,..,n-1
30            assert_eq!(to_index(&destinations[i]), Some(i));
31            if let Some(index) = to_index(&sources[i]) {
32                if index < n && index != i {
33                    num_destinations[index] += 1;
34                }
35            }
36        }
37
38        // Process all sinks in the graph and follow their parents.
39        // Keep track of nodes having more than 2 destinations, in case they belong to a loop.
40        let mut tail_candidates = Vec::new();
41        for i in 0..n {
42            // Generate a movement for each sink in the graph
43            let mut node = i;
44            // A sink has no child
45            while num_destinations[node] == 0 {
46                if to_index(&sources[node]) == Some(node) {
47                    //no-op: mark the node as processed
48                    num_destinations[node] = usize::MAX;
49                    processed += 1;
50                    break;
51                }
52                // Generates a move instruction
53                self.perform_movement(node, sources[node], &mut num_destinations, &mut processed);
54                // Follow the parent
55                if let Some(index) = to_index(&sources[node]) {
56                    if index < n {
57                        num_destinations[index] -= 1;
58                        if num_destinations[index] > 0 {
59                            // The parent node has another child, so we cannot process it yet.
60                            tail_candidates.push((sources[node], node));
61                            break;
62                        }
63                        // process the parent node
64                        node = index;
65                        continue;
66                    }
67                }
68                // End of the path
69                break;
70            }
71            if processed == n {
72                return;
73            }
74        }
75        // All sinks and their parents have been processed, remaining nodes are part of a loop
76        // Check if a tail_candidate is a branch to a loop
77        for (entry, free) in tail_candidates {
78            let entry_idx = to_index(&entry).unwrap();
79            if entry_idx < n && num_destinations[entry_idx] == 1 {
80                // Use the branch as the temporary register for the loop
81                self.process_loop(
82                    entry_idx,
83                    &from_index(free),
84                    &mut num_destinations,
85                    sources,
86                    &mut processed,
87                );
88            }
89        }
90        if processed == n {
91            return;
92        }
93        // Now process all the remaining loops with a temporary register.
94        // Allocate one temporary per loop to avoid type confusion when reusing registers,
95        // since different loops may contain values of different types.
96        for i in 0..n {
97            if num_destinations[i] == 1 {
98                let src = from_index(i);
99                // Copy the loop entry to a temporary register.
100                // Unfortunately, we cannot use one register for all the loops
101                // when the sources do not have the same type
102                let temp_register = self.registers_mut().allocate_register();
103                self.mov_instruction(temp_register, src);
104                self.process_loop(
105                    i,
106                    &temp_register,
107                    &mut num_destinations,
108                    sources,
109                    &mut processed,
110                );
111                self.deallocate_register(temp_register);
112            } else {
113                // Nodes must have been processed, or are part of a loop.
114                assert_eq!(num_destinations[i], usize::MAX);
115            }
116        }
117    }

119    /// Generates mov opcodes corresponding to a loop, given a node from the loop (entry)
120    /// and a register not in the loop that contains its value (free)
121    fn process_loop(
122        &mut self,
123        entry: usize,
124        free: &MemoryAddress,
125        num_destinations: &mut [usize],
126        source: &[MemoryAddress],
127        processed: &mut usize,
128    ) {
129        let mut current = entry;
130        while to_index(&source[current]).unwrap() != entry {
131            self.perform_movement(current, source[current], num_destinations, processed);
132            current = to_index(&source[current]).unwrap();
133        }
134        self.perform_movement(current, *free, num_destinations, processed);
135    }
136
137    /// Generates a move opcode from 'src' to 'dest'.
138    fn perform_movement(
139        &mut self,
140        dest: usize,
141        src: MemoryAddress,
142        num_destinations: &mut [usize],
143        processed: &mut usize,
144    ) {
145        let destination = from_index(dest);
146        self.mov_instruction(destination, src);
147        // set the node as 'processed'
148        num_destinations[dest] = usize::MAX;
149        *processed += 1;
150    }
151}

153/// Map the address so that the first register of the stack will have index 0
154fn to_index(adr: &MemoryAddress) -> Option<usize> {
155    match adr {
156        MemoryAddress::Relative(size) => Some(assert_usize(*size) - Stack::start()),
157        MemoryAddress::Direct(_) => None,
158    }
159}
160
161/// Construct the register corresponding to the given mapped 'index'
162fn from_index(idx: usize) -> MemoryAddress {
163    assert!(idx != usize::MAX, "invalid index");
164    MemoryAddress::relative(assert_u32(idx + Stack::start()))
165}

190    /// Generate `Opcode::Mov` for a sequence of expected `src -> dst` moves.
191    fn generate_opcodes(movements: Vec<(usize, usize)>) -> Vec<Opcode<FieldElement>> {
192        movements
193            .into_iter()
194            .map(|(src, dst)| Opcode::Mov {
195                destination: MemoryAddress::relative(assert_u32(dst)),
196                source: MemoryAddress::relative(assert_u32(src)),
197            })
198            .collect()
199    }

201    /// Split numeric `src -> dst` movements into separate vectors and convert to `MemoryAddress`
202    fn movements_to_source_and_destinations(
203        movements: Vec<(usize, usize)>,
204    ) -> (Vec<MemoryAddress>, Vec<MemoryAddress>) {
205        let sources = movements
206            .iter()
207            .map(|(source, _)| MemoryAddress::relative(assert_u32(*source)))
208            .collect();
209        let destinations = movements
210            .iter()
211            .map(|(_, destination)| MemoryAddress::relative(assert_u32(*destination)))
212            .collect();
213        (sources, destinations)
214    }

230    /// Test that a series of `src->dst` movements results in a series of `src->dst` move opcodes.
231    fn assert_generated_opcodes(
232        movements: Vec<(usize, usize)>,
233        expected_moves: Vec<(usize, usize)>,
234    ) {
235        let mut context = create_context();
236        for _ in 0..movements.len() {
237            context.registers_mut().allocate_register();
238        }
239        let (sources, destinations) = movements_to_source_and_destinations(movements);
240        context.codegen_mov_registers_to_registers(&sources, &destinations);
241
242        let opcodes = context.into_artifact().byte_code;
243
244        assert_eq!(opcodes, generate_opcodes(expected_moves));
245    }

320    #[test]
321    fn prop_mov_registers_to_registers() {
322        const MEM_SIZE: usize = 10;
323        arbtest::arbtest(|u| {
324            // Allocate more memory to allow for temporary variables.
325            let mut memory: Vec<u32> = vec![0; MEM_SIZE * 2];
326            // Fill the memory with some random numbers.
327            for slot in memory.iter_mut() {
328                *slot = u.arbitrary()?;
329            }
330
331            // Pick a random unique subset of the slots as destinations.
332            let num_destinations = u.int_in_range(0..=MEM_SIZE)?;
333
334            // All potential memory slots; we can't address before the stack start.
335            let all_indexes = (0..MEM_SIZE).map(|i| i + Stack::start()).collect::<Vec<_>>();
336
337            let destinations: Vec<usize> = (1..num_destinations + 1).collect();
338
339            // Pick random sources for each destination (same source can be repeated).
340            let mut sources = Vec::with_capacity(num_destinations);
341            for _ in 0..num_destinations {
342                sources.push(u.choose(&all_indexes).copied()?);
343            }
344
345            // Take a snapshot of the source data; this is what we expect the destination to become.
346            let source_data = vecmap(&sources, |i| memory[*i]);
347
348            // Generate the opcodes.
349            let opcodes = {
350                // Convert to MemoryAddress
351                let sources = vecmap(&sources, |i| MemoryAddress::relative(assert_u32(*i)));
352                let destinations =
353                    vecmap(&destinations, |i| MemoryAddress::relative(assert_u32(*i)));
354
355                let mut context = create_context();
356
357                // Treat the memory we care about as pre-allocated, so temporary variables are created after them.
358                let all_registers = vecmap(all_indexes, |i| MemoryAddress::relative(assert_u32(i)));
359                context.set_allocated_registers(all_registers);
360
361                context.codegen_mov_registers_to_registers(&sources, &destinations);
362                context.into_artifact().byte_code
363            };
364
365            // Execute the opcodes.
366            for opcode in opcodes {
367                let Opcode::Mov { destination, source } = opcode else {
368                    unreachable!("only Mov expected");
369                };
370                memory[assert_usize(destination.to_u32())] = memory[assert_usize(source.to_u32())];
371            }
372
373            // Get the final values at the destination slots.
374            let destination_data = vecmap(&destinations, |i| memory[*i]);
375
376            // At the end the destination should have the same value as the source had.
377            assert_eq!(destination_data, source_data);
378
379            Ok(())
380        })
381        .run();
382    }
