15impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
16    /// Codegens a truncation of a value to the given bit size
17    pub(crate) fn codegen_truncate(
18        &mut self,
19        destination_of_truncated_value: SingleAddrVariable,
20        value_to_truncate: SingleAddrVariable,
21        bit_size: u32,
22    ) {
23        assert!(
24            bit_size <= value_to_truncate.bit_size,
25            "tried to truncate to a bit size {} greater than the variable size {}",
26            bit_size,
27            value_to_truncate.bit_size
28        );
29
30        if bit_size == value_to_truncate.bit_size {
31            self.mov_instruction(destination_of_truncated_value.address, value_to_truncate.address);
32            return;
33        }
34
35        // If we are truncating a value down to a natively supported integer, we can just use the cast instruction.
36        //
37        // This is exploiting the fact that the cast instruction will automatically truncate the value when
38        // casting to a smaller bit size. We can then avoid the more expensive modulo operation.
39        if IntegerBitSize::try_from(bit_size).is_ok() {
40            // Allocate a temporary register to hold the intermediate casted value.
41            let intermediate_register = self.allocate_single_addr(bit_size);
42            self.cast_instruction(*intermediate_register, value_to_truncate);
43
44            // We then cast back to the original type to satisfy typed memory.
45            self.cast_instruction(destination_of_truncated_value, *intermediate_register);
46
47            return;
48        }
49
50        // If the bit size we are truncating down to is not a natively supported integer, we need to use a modulo operation.
51
52        // The modulus is guaranteed to fit, since we are truncating down to a bit size that is strictly less than the value_to_truncate.bit_size
53        let modulus_var = self.make_constant_instruction(
54            F::from(2_usize).pow(&F::from(u128::from(bit_size))),
55            value_to_truncate.bit_size,
56        );
57
58        self.binary_instruction(
59            value_to_truncate,
60            *modulus_var,
61            destination_of_truncated_value,
62            BrilligBinaryOp::Modulo,
63        );
64    }
65
66    /// Issues a `to_radix` instruction. This instruction will write the modulus of the `source_field` register
67    /// and the `radix` register `target_array`, with the number of limbs given by the size of `target_array`.
68    ///
69    /// If `output_bits` is true, it generates bit limbs, otherwise it generates byte limbs.
70    /// If `little_endian` is true, then the `target_array` will contain the results in Little Endian order.
71    pub(crate) fn codegen_to_radix(
72        &mut self,
73        source_field: SingleAddrVariable,
74        target_array: BrilligArray,
75        radix: SingleAddrVariable,
76        little_endian: bool,
77        output_bits: bool,
78    ) {
79        assert!(source_field.bit_size == F::max_num_bits());
80        assert!(radix.bit_size == 32);
81
82        let bits_register = self.make_constant_instruction(output_bits.into(), 1);
83        self.codegen_initialize_array(target_array);
84        let pointer = self.codegen_make_array_items_pointer(target_array);
85        let num_limbs = self.make_usize_constant_instruction(target_array.size.0.into());
86
87        // Perform big-endian ToRadix
88        self.black_box_op_instruction(BlackBoxOp::ToRadix {
89            input: source_field.address,
90            radix: radix.address,
91            output_pointer: *pointer,
92            num_limbs: num_limbs.address,
93            output_bits: bits_register.address,
94        });
95
96        if little_endian {
97            self.codegen_array_reverse(*pointer, num_limbs.address);
98        }
99    }
