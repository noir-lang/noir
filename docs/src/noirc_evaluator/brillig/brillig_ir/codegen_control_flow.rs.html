26    pub(crate) fn codegen_generic_iteration<T>(
27        &mut self,
28        make_iterator: impl FnOnce(&mut BrilligContext<F, Registers>) -> T,
29        update_iterator: impl FnOnce(&mut BrilligContext<F, Registers>, &T),
30        make_finish_condition: impl FnOnce(
31            &mut BrilligContext<F, Registers>,
32            &T,
33        ) -> Allocated<SingleAddrVariable, Registers>,
34        on_iteration: impl FnOnce(&mut BrilligContext<F, Registers>, &T),
35        clean_iterator: impl FnOnce(&mut BrilligContext<F, Registers>, T),
36    ) {
37        let iterator = make_iterator(self);
38
39        let (loop_section, loop_label) = self.reserve_next_section_label();
40        self.enter_section(loop_section);
41
42        // Loop body
43        let should_end = make_finish_condition(self, &iterator);
44
45        let (exit_loop_section, exit_loop_label) = self.reserve_next_section_label();
46
47        self.jump_if_instruction(should_end.address, exit_loop_label);
48
49        // Call the on iteration function
50        on_iteration(self, &iterator);
51
52        // Update iterator
53        update_iterator(self, &iterator);
54        self.jump_instruction(loop_label);
55
56        // Exit the loop
57        self.enter_section(exit_loop_section);
58
59        // Free any resources held by the iterator.
60        clean_iterator(self, iterator);
61    }

63    /// This codegen will issue a loop for (let iterator_register = loop_start; i < loop_bound; i += step)
64    /// The body of the loop should be issued by the caller in the on_iteration closure.
65    ///
66    /// # Safety
67    /// Iterator increment uses wrapping 32-bit arithmetic. Callers must ensure the iterator
68    /// reaches the bound before wrapping. For `loop_start=0, step=1`, this requires
69    /// `loop_bound <= u32::MAX`. For pointer iteration, VM allocation checks ensure safety.
70    pub(crate) fn codegen_for_loop(
71        &mut self,
72        loop_start: Option<MemoryAddress>,
73        loop_bound: MemoryAddress,
74        step: Option<MemoryAddress>,
75        on_iteration: impl FnOnce(&mut BrilligContext<F, Registers>, SingleAddrVariable),
76    ) {
77        let step_register = step.unwrap_or(ReservedRegisters::usize_one());
78
79        self.codegen_generic_iteration(
80            // 1. make_iterator: Initialize counter
81            |ctx| {
82                if let Some(loop_start) = loop_start {
83                    let iterator = ctx.allocate_single_addr_usize();
84                    ctx.mov_instruction(iterator.address, loop_start);
85                    iterator
86                } else {
87                    ctx.make_usize_constant_instruction(0_usize.into())
88                }
89            },
90            // 2. update_iterator: Add step
91            |ctx, iterator| {
92                ctx.memory_op_instruction(
93                    iterator.address,
94                    step_register,
95                    iterator.address,
96                    BrilligBinaryOp::Add,
97                );
98            },
99            // 3. make_finish_condition: Check bound <= iterator
100            |ctx, iterator| {
101                let reached_bound = ctx.allocate_single_addr_bool();
102                ctx.memory_op_instruction(
103                    loop_bound,
104                    iterator.address,
105                    reached_bound.address,
106                    BrilligBinaryOp::LessThanEquals,
107                );
108                reached_bound
109            },
110            // 4. on_iteration: Call user's closure
111            |ctx, iterator| {
112                on_iteration(ctx, **iterator);
113            },
114            // 5. clean_iterator: Nothing to clean (auto-deallocated)
115            |_, _| {},
116        );
117    }

119    /// This codegen will issue a loop that will iterate from 0 to iteration_count.
120    /// The body of the loop should be issued by the caller in the on_iteration closure.
121    ///
122    /// # Safety
123    /// `iteration_count` value must not exceed u32::MAX for correct behavior.
124    /// See [BrilligContext::codegen_for_loop] for more information.     
125    pub(crate) fn codegen_loop(
126        &mut self,
127        iteration_count: MemoryAddress,
128        on_iteration: impl FnOnce(&mut BrilligContext<F, Registers>, SingleAddrVariable),
129    ) {
130        self.codegen_for_loop(None, iteration_count, None, on_iteration);
131    }

133    /// This codegen will issue an if-then branch that will check if the condition is true
134    /// and if so, perform the instructions given in `f(self, true)` and otherwise perform the
135    /// instructions given in `f(self, false)`. A boolean is passed instead of two separate
136    /// functions to allow the given function to mutably alias its environment.
137    pub(crate) fn codegen_branch(
138        &mut self,
139        condition: MemoryAddress,
140        mut f: impl FnMut(&mut BrilligContext<F, Registers>, bool),
141    ) {
142        // Reserve 3 sections
143        let (then_section, then_label) = self.reserve_next_section_label();
144        let (otherwise_section, otherwise_label) = self.reserve_next_section_label();
145        let (end_section, end_label) = self.reserve_next_section_label();
146
147        self.jump_if_instruction(condition, then_label);
148        self.jump_instruction(otherwise_label);
149
150        self.enter_section(then_section);
151        f(self, true);
152        self.jump_instruction(end_label);
153
154        self.enter_section(otherwise_section);
155        f(self, false);
156        // Fall through into `end_section`
157
158        self.enter_section(end_section);
159    }

161    /// This codegen issues a branch that jumps over the code generated by the given function if the condition is false
162    #[expect(unused)]
163    pub(crate) fn codegen_if(
164        &mut self,
165        condition: MemoryAddress,
166        f: impl FnOnce(&mut BrilligContext<F, Registers>),
167    ) {
168        let (end_section, end_label) = self.reserve_next_section_label();
169        let (then_section, then_label) = self.reserve_next_section_label();
170
171        self.jump_if_instruction(condition, then_label);
172        self.jump_instruction(end_label);
173
174        self.enter_section(then_section);
175        f(self);
176
177        self.enter_section(end_section);
178    }

195    /// Jump to a trap condition if `condition` is false.
196    /// The trap will include the given message as error data.
197    ///
198    /// If `error_selector` is None, an empty error data is generated (no error information).
199    pub(crate) fn codegen_constrain_with_error_data(
200        &mut self,
201        condition: SingleAddrVariable,
202        error_data_items: Vec<BrilligVariable>,
203        error_data_types: Vec<BrilligParameter>,
204        error_selector: Option<ErrorSelector>,
205    ) {
206        assert!(condition.bit_size == 1);
207
208        self.codegen_if_not(condition.address, |ctx| {
209            let data_size = Self::flattened_tuple_size(&error_data_types);
210
211            // Special case: No error selector means completely empty error data
212            let Some(error_selector) = error_selector else {
213                let error_data =
214                    ctx.make_usize_constant_instruction(0_usize.into()).map(|size| HeapVector {
215                        pointer: ReservedRegisters::free_memory_pointer(),
216                        size: size.address,
217                    });
218                ctx.trap_instruction(*error_data);
219                return;
220            };
221
222            // Shortcuts: empty data does not need without allocation, and can even use the procedure.
223            if data_size == 0 {
224                // use the procedure call for better code reuse
225                if ctx.can_call_procedures {
226                    if let Some(ErrorType::String(message)) =
227                        ctx.obj.error_types.get(&error_selector)
228                    {
229                        ctx.call_error_with_string_procedure(message.clone());
230                        return;
231                    }
232                }
233
234                // Fast path: Just write selector to free memory pointer, no allocation needed
235                ctx.indirect_const_instruction(
236                    ReservedRegisters::free_memory_pointer(),
237                    64,
238                    u128::from(error_selector.as_u64()).into(),
239                );
240                ctx.trap_instruction(HeapVector {
241                    pointer: ReservedRegisters::free_memory_pointer(),
242                    size: ReservedRegisters::usize_one(),
243                });
244                return;
245            }
246
247            // Allocate buffer and serialize data
248            // + 1 due to the error data id being the first item returned
249            let error_data_size = data_size + 1;
250            let error_data_size_var = ctx.make_usize_constant_instruction(error_data_size.into());
251            let error_data = ctx
252                .allocate_register()
253                .map(|pointer| HeapVector { pointer, size: error_data_size_var.address });
254            ctx.codegen_allocate_immediate_mem(error_data.pointer, error_data_size);
255
256            let current_error_data_pointer = ctx.allocate_register();
257            ctx.mov_instruction(*current_error_data_pointer, error_data.pointer);
258            ctx.indirect_const_instruction(
259                *current_error_data_pointer,
260                64,
261                u128::from(error_selector.as_u64()).into(),
262            );
263
264            ctx.codegen_usize_op_in_place(*current_error_data_pointer, BrilligBinaryOp::Add, 1);
265            for (error_variable, error_param) in
266                error_data_items.into_iter().zip(error_data_types.into_iter())
267            {
268                let flattened_size = error_param.flattened_size();
269                match error_param {
270                    BrilligParameter::SingleAddr(_) => {
271                        ctx.store_instruction(
272                            *current_error_data_pointer,
273                            error_variable.extract_single_addr().address,
274                        );
275                    }
276                    BrilligParameter::Array(item_type, item_count) => {
277                        let deflattened_items_pointer =
278                            ctx.codegen_make_array_items_pointer(error_variable.extract_array());
279
280                        ctx.flatten_array(
281                            &item_type,
282                            item_count,
283                            *current_error_data_pointer,
284                            *deflattened_items_pointer,
285                        );
286                    }
287                    BrilligParameter::Vector(_, _) => {
288                        unimplemented!("Vectors are not supported as error data")
289                    }
290                }
291                ctx.codegen_usize_op_in_place(
292                    *current_error_data_pointer,
293                    BrilligBinaryOp::Add,
294                    flattened_size,
295                );
296            }
297            ctx.trap_instruction(*error_data);
298        });
299    }

301    /// Jump to a trap condition if `condition` is false,
302    /// with any assertion message written to the error data.
303    pub(crate) fn codegen_constrain(
304        &mut self,
305        condition: SingleAddrVariable,
306        assert_message: Option<String>,
307    ) {
308        assert!(condition.bit_size == 1);
309
310        // Compute error selector if we have a message
311        let error_selector = assert_message.map(|message| {
312            let error_type = ErrorType::String(message);
313            let error_selector = error_type.selector();
314            self.obj.error_types.insert(error_selector, error_type);
315            error_selector
316        });
317
318        self.codegen_constrain_with_error_data(
319            condition,
320            vec![], // No runtime data items
321            vec![], // No runtime data types
322            error_selector,
323        );
324    }

336    // Flattens an array by recursively copying nested arrays and regular items.
337    pub(super) fn flatten_array(
338        &mut self,
339        item_type: &[BrilligParameter],
340        item_count: SemanticLength,
341        flattened_array_pointer: MemoryAddress,
342        deflattened_items_pointer: MemoryAddress,
343    ) {
344        if Self::has_nested_arrays(item_type) {
345            let movement_register = self.allocate_register();
346
347            let source_item_size = item_type.len();
348            let target_item_size = Self::flattened_tuple_size(item_type);
349
350            for item_index in 0..item_count.0 {
351                let source_item_base_index = assert_usize(item_index) * source_item_size;
352                let target_item_base_index = assert_usize(item_index) * target_item_size;
353
354                let mut target_offset = 0;
355
356                for (subitem_index, subitem) in item_type.iter().enumerate() {
357                    let source_index = self.make_usize_constant_instruction(
358                        (source_item_base_index + subitem_index).into(),
359                    );
360                    let target_index = self.make_usize_constant_instruction(
361                        (target_item_base_index + target_offset).into(),
362                    );
363
364                    match subitem {
365                        BrilligParameter::SingleAddr(_) => {
366                            self.codegen_load_with_offset(
367                                deflattened_items_pointer,
368                                *source_index,
369                                *movement_register,
370                            );
371                            self.codegen_store_with_offset(
372                                flattened_array_pointer,
373                                *target_index,
374                                *movement_register,
375                            );
376                        }
377                        BrilligParameter::Array(
378                            nested_array_item_type,
379                            nested_array_item_count,
380                        ) => {
381                            // Usually we need to pass a semi-flattened length to `allocate_brillig_array`.
382                            // However, since we are deflattening arrays, we need to allocate as many elements
383                            // as there are in this particular nested array, which is its semantic length.
384                            let deflattened_nested_array = self.allocate_brillig_array(
385                                SemiFlattenedLength(nested_array_item_count.0),
386                            );
387
388                            self.codegen_load_with_offset(
389                                deflattened_items_pointer,
390                                *source_index,
391                                deflattened_nested_array.pointer,
392                            );
393                            let deflattened_nested_array_items =
394                                self.codegen_make_array_items_pointer(*deflattened_nested_array);
395
396                            let flattened_nested_array_pointer = self.allocate_register();
397                            self.memory_op_instruction(
398                                flattened_array_pointer,
399                                target_index.address,
400                                *flattened_nested_array_pointer,
401                                BrilligBinaryOp::Add,
402                            );
403
404                            self.flatten_array(
405                                nested_array_item_type,
406                                *nested_array_item_count,
407                                *flattened_nested_array_pointer,
408                                *deflattened_nested_array_items,
409                            );
410                        }
411                        BrilligParameter::Vector(..) => unreachable!("ICE: Cannot flatten vectors"),
412                    }
413
414                    target_offset += subitem.flattened_size();
415                }
416            }
417        } else {
418            let size: SemiFlattenedLength =
419                item_count * ElementTypesLength(assert_u32(item_type.len()));
420            let item_count = self.make_usize_constant_instruction(size.0.into());
421            self.codegen_mem_copy(deflattened_items_pointer, flattened_array_pointer, *item_count);
422        }
423    }
