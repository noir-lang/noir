12impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
13    /// Generate Brillig opcodes to:
14    /// * calculate the current stack size
15    /// * copy the current stack pointer and the call arguments into a new stack frame
16    /// * update the stack pointer and execute the call
17    /// * restore the stack pointer and copy the results into the return variables
18    pub(crate) fn codegen_call(
19        &mut self,
20        func_id: FunctionId,
21        arguments: &[BrilligVariable],
22        returns: &[BrilligVariable],
23    ) {
24        // Allocate a register for the stack size. With this allocation, we have our current stack before the call.
25        let stack_size_register = self.allocate_single_addr_usize();
26
27        // Find the start of free stack memory: this is our current stack size.
28        let previous_stack_pointer = self.registers().empty_registers_start();
29        let stack_size = previous_stack_pointer.unwrap_relative();
30
31        // The first few slots in each stack frame are reserved.
32        let reserved_len = self.registers().start() as u32;
33
34        // Ensure that writing call arguments and return values won't overflow into heap memory.
35        //
36        // At compile time, we don't know the runtime stack_pointer position. The worst case
37        // is when we're in the last viable stack frame (last possible stack start in the `CheckMaxStackDepth` procedure).
38        // In that case, writing arguments could still overflow into the heap before the next `CheckMaxStackDepth` runs.
39        //
40        // Given that `CheckMaxStackDepth` ensures:
41        //   `stack_pointer <= stack_start + max_stack_size - max_frame_size`
42        //
43        // Heap corruption occurs when:
44        //   `stack_pointer + stack_size + reserved_len + max(arguments_len, returns_len) > stack_start + max_stack_size`
45        //
46        // (Note that it's okay for the LHS to equal the RHS because initially the stack_pointer equals the stack_start).
47        //
48        // Substituting worst case (`stack_pointer = stack_start + max_stack_size - max_frame_size`, ie. the last viable frame):
49        //   `(stack_start + max_stack_size - max_frame_size) + stack_size + reserved_len + max(arguments_len, returns_len) > stack_start + max_stack_size`
50        //   `max_stack_size - max_frame_size + stack_size + reserved_len + max(arguments_len, returns_len) > max_stack_size`
51        //   `stack_size + reserved_len + max(arguments_len, returns_len) > max_frame_size`
52        //
53        // Therefore, to prevent heap corruption: `stack_size + reserved_len + max(arguments_len, returns_len) <= max_frame_size`
54        //
55        // This is conservative (may reject programs that would work in the non-last viable frames),
56        // but is the only safe compile-time check without runtime pointers.
57        let max_frame_size = self.registers().layout().max_stack_frame_size();
58        let arguments_len = arguments.len();
59        let returns_len = returns.len();
60        assert!(
61            ((stack_size + reserved_len) as usize) + cmp::max(arguments_len, returns_len)
62                <= max_frame_size,
63            "Call arguments would exceed stack frame bounds: frame_size={stack_size}, arguments={arguments_len}, returns={returns_len}, max={max_frame_size}",
64        );
65
66        // Write the current stack size to a register, so we can add it to the stack pointer.
67        self.const_instruction(*stack_size_register, stack_size.into());
68
69        // Copy the current stack pointer into the 0th slot of the next stack frame.
70        // This is the previous stack pointer to return to after the call.
71        self.mov_instruction(previous_stack_pointer, ReservedRegisters::stack_pointer());
72
73        // Pass the arguments starting at the callee's start offset.
74        // Offset 0 holds the saved stack pointer;
75        // when spill support is active, offset 1 is reserved for the spill base pointer.
76        let mut current_argument_location = stack_size + reserved_len;
77        for item in arguments {
78            // Here we are still using addresses relative to the current stack pointer.
79            self.mov_instruction(
80                MemoryAddress::relative(current_argument_location),
81                item.extract_register(),
82            );
83            current_argument_location += 1;
84        }
85
86        // Increment the stack pointer for the call: stack_pointer := stack_pointer + stack_size.
87        // By increasing it with the stack size before arguments where copied, it will include the
88        // arguments at the intended relative addresses.
89        self.memory_op_instruction(
90            ReservedRegisters::stack_pointer(),
91            stack_size_register.address,
92            ReservedRegisters::stack_pointer(),
93            BrilligBinaryOp::Add,
94        );
95
96        self.add_external_call_instruction(func_id);
97
98        // Restore the previous stack pointer, which was copied into the 0th slot.
99        self.mov_instruction(ReservedRegisters::stack_pointer(), MemoryAddress::relative(0));
100
101        // Move the return values back. The return values are expected to overwrite the args.
102        let mut current_return_location = stack_size + reserved_len;
103        for item in returns {
104            self.mov_instruction(
105                item.extract_register(),
106                MemoryAddress::relative(current_return_location),
107            );
108            current_return_location += 1;
109        }
110    }
111
112    /// Codegens a return from the current function.
113    ///
114    /// Takes the variables with the addresses of the values that need to be returned.
115    /// The values are copied to the beginning of the stack space, into an equal number of slots.
116    ///
117    /// Any potential overlap between the source of the return variables and the final destination
118    /// on the beginning of the stack is handled by [Self::codegen_mov_registers_to_registers].
119    pub(crate) fn codegen_return(&mut self, return_variables: &[BrilligVariable]) {
120        let mut sources = Vec::with_capacity(return_variables.len());
121        let mut destinations = Vec::with_capacity(return_variables.len());
122
123        for (destination_index, return_variable) in return_variables.iter().enumerate() {
124            // In case we have fewer return registers than indices to write to, ensure we've allocated this register.
125            let destination_register =
126                MemoryAddress::relative(assert_u32(self.registers().start() + destination_index));
127            self.registers_mut().ensure_register_is_allocated(destination_register);
128            destinations.push(destination_register);
129            sources.push(return_variable.extract_register());
130        }
131        self.codegen_mov_registers_to_registers(&sources, &destinations);
132        self.return_instruction();
133    }
134}
