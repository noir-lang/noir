1//! Per-site array and vector copy counting for the `--count-array-copies` debugging flag.
2//!
3//! When a [`CopySiteRegistry`] is attached to the [`BrilligContext`], the copy procedures emit
4//! extra bytecode that tallies how many times each source location actually copies an
5//! array/vector (as opposed to reusing it in place). The entry-point wrapper then prints a
6//! summary of the totals once execution finishes. With no registry attached none of this
7//! bytecode is emitted, so ordinary compilation is unaffected.
8

29/// Maximum number of distinct copy sites that are tracked per-site.
30/// Sites beyond this limit are not tracked per-site (they still count toward the total).
31pub(crate) const MAX_TRACK_SITES: usize = 256;
32
33/// Number of per-site copy locations displayed at end of execution (top N by count).
34pub(crate) const MAX_DISPLAY_SITES: usize = 25;
35
36/// Inner state for [CopySiteRegistry], kept behind an `Arc<Mutex<…>>` so it can be
37/// shared across all `BrilligContext` instances that are alive during one compilation.
38#[derive(Debug, Default)]
39struct CopySiteRegistryInner {
40    /// Ordered list of call stack IDs (index = per-site counter slot).
41    sites: Vec<CallStackId>,
42    /// Map from `CallStackId` → slot index, used to deduplicate identical call sites.
43    site_to_index: FxHashMap<CallStackId, usize>,
44    /// Source-location labels resolved after compilation (set by `resolve_labels`).
45    resolved_labels: Option<Vec<String>>,
46}
47
48/// Shared registry of array/vector copy sites, accumulated during Brillig compilation.
49/// Each *unique* call site (identified by `CallStackId`) is assigned a sequential index
50/// used to address its per-site counter in global memory.
51/// Registering the same call site twice returns the same index.
52/// After compilation, call `resolve_labels` to convert `CallStackId`s to readable strings.
53#[derive(Clone, Debug, Default)]
54pub struct CopySiteRegistry(Arc<Mutex<CopySiteRegistryInner>>);
55
56impl CopySiteRegistry {
57    /// Register a copy site for the given call stack location.
58    /// Returns the site index (0-based).  Identical call sites share an index.
59    pub(crate) fn register_site(&self, id: CallStackId) -> usize {
60        let mut inner = self.0.lock().unwrap();
61        if let Some(&idx) = inner.site_to_index.get(&id) {
62            return idx;
63        }
64        let index = inner.sites.len();
65        inner.site_to_index.insert(id, index);
66        inner.sites.push(id);
67        index
68    }
69
70    /// Resolve all registered `CallStackId`s to human-readable `"filename.nr:line"` strings.
71    /// Must be called after Brillig compilation (once the `CallStackHelper` is fully populated)
72    /// but before `get_resolved_labels` is used.
73    pub(crate) fn resolve_labels(
74        &self,
75        call_stacks: &CallStackHelper,
76        files: Option<&fm::FileManager>,
77    ) {
78        use fm::codespan_files::Files;
79        let sites = self.0.lock().unwrap().sites.clone();
80        let labels: Vec<String> = sites
81            .iter()
82            .map(|&id| {
83                let stack = call_stacks.get_call_stack(id);
84                let Some(location) = stack.last() else {
85                    return "<unknown>".to_string();
86                };
87                let Some(fm) = files else {
88                    return format!("<unknown>:{}", location.span.start());
89                };
90                let file_name = fm
91                    .path(location.file)
92                    .and_then(|p| p.file_name())
93                    .and_then(|n| n.to_str())
94                    .unwrap_or("<unknown>");
95                let start_index = location.span.start() as usize;
96                match fm.as_file_map().line_index(location.file, start_index) {
97                    Ok(line_idx) => format!("{file_name}:{}", line_idx + 1),
98                    Err(_) => file_name.to_string(),
99                }
100            })
101            .collect();
102        self.0.lock().unwrap().resolved_labels = Some(labels);
103    }
104
105    /// Return a snapshot of all resolved site labels in registration order.
106    /// Returns an empty `Vec` if `resolve_labels` has not been called yet.
107    pub(crate) fn get_resolved_labels(&self) -> Vec<String> {
108        self.0.lock().unwrap().resolved_labels.clone().unwrap_or_default()
109    }
110}

112impl<F: AcirField + DebugToString, Registers: RegisterAllocator> BrilligContext<F, Registers> {
113    /// True when per-site array copy counting is enabled, i.e. a [`CopySiteRegistry`] is attached.
114    pub(crate) fn count_array_copies(&self) -> bool {
115        self.copy_site_registry.is_some()
116    }
117
118    /// Returns the address of the implicit debug variable containing the count of
119    /// implicitly copied arrays as a result of RC's copy on write semantics.
120    pub(crate) fn array_copy_counter_address(&self) -> MemoryAddress {
121        assert!(
122            self.count_array_copies(),
123            "the array copy counter is only allocated when `--count-array-copies` is set"
124        );
125
126        // The copy counter is always put in the first global slot
127        MemoryAddress::direct(assert_u32(GlobalSpace::start_with_layout(&self.layout())))
128    }
129
130    /// Returns the global memory address of the per-site copy counter for `site_index`.
131    /// Per-site counters occupy slots 1..=MAX_TRACK_SITES right after the total counter (slot 0).
132    pub(crate) fn per_site_counter_address(&self, site_index: usize) -> MemoryAddress {
133        assert!(
134            site_index < MAX_TRACK_SITES,
135            "site_index {site_index} exceeds MAX_TRACK_SITES {MAX_TRACK_SITES}"
136        );
137        MemoryAddress::direct(assert_u32(
138            GlobalSpace::start_with_layout(&self.layout()) + 1 + site_index,
139        ))
140    }
141
142    /// Registers the current location as a per-site copy site and returns the global memory
143    /// address of its counter, or `None` when copy-counting is disabled, no registry is attached,
144    /// or the site is beyond [`MAX_TRACK_SITES`].
145    ///
146    /// Sites are deduplicated by `CallStackId`, so the same call site compiled more than once
147    /// shares a single counter.
148    fn register_per_site_counter(&self) -> Option<MemoryAddress> {
149        let registry = self.copy_site_registry.clone()?;
150        let site_index = registry.register_site(self.current_call_stack_id());
151        (site_index < MAX_TRACK_SITES).then(|| self.per_site_counter_address(site_index))
152    }
153
154    /// Emit `*counter += 1` for a usize copy counter stored at `counter`.
155    fn codegen_increment_counter(&mut self, counter: MemoryAddress) {
156        self.codegen_usize_op(counter, counter, BrilligBinaryOp::Add, 1);
157    }
158
159    pub(crate) fn codegen_increment_array_copy_counter(&mut self) {
160        let array_copy_counter = self.array_copy_counter_address();
161        self.codegen_increment_counter(array_copy_counter);
162    }
163
164    /// If copy-counting is enabled, registers this as a per-site copy location and
165    /// emits runtime code that increments the per-site counter whenever a copy actually occurred
166    /// (i.e. `source_pointer != dest_pointer` after a copy procedure call).
167    pub(crate) fn codegen_count_if_copy_occurred(
168        &mut self,
169        source_pointer: MemoryAddress,
170        dest_pointer: MemoryAddress,
171    ) {
172        let Some(counter_addr) = self.register_per_site_counter() else {
173            return;
174        };
175
176        // Emit: if source_pointer != dest_pointer { counter_addr += 1 }
177        // We use: did_not_copy = (source == dest); if did_not_copy => skip increment
178        let did_not_copy = self.allocate_single_addr_bool();
179        self.memory_op_instruction(
180            source_pointer,
181            dest_pointer,
182            did_not_copy.address,
183            BrilligBinaryOp::Equals,
184        );
185        self.codegen_if_not(did_not_copy.address, |ctx| {
186            ctx.codegen_increment_counter(counter_addr);
187        });
188    }
189
190    /// Like `codegen_count_if_copy_occurred` but driven by an explicit boolean flag register
191    /// rather than a pointer comparison. Registers this as a per-site copy location and emits
192    /// runtime code that increments the per-site counter when `flag != 0`.
193    pub(crate) fn codegen_count_if_nonzero(&mut self, flag: MemoryAddress) {
194        let Some(counter_addr) = self.register_per_site_counter() else {
195            return;
196        };
197
198        // if flag != 0 { counter_addr += 1 }
199        self.codegen_if(flag, |ctx| {
200            ctx.codegen_increment_counter(counter_addr);
201        });
202    }
203
204    /// Emit print statements for the total array copy count, then for the top
205    /// [`MAX_DISPLAY_SITES`] most-copied locations sorted descending by count.
206    ///
207    /// Uses a compile-time-unrolled selection sort: [`MAX_DISPLAY_SITES`] outer iterations, each
208    /// running a runtime inner loop to find the maximum remaining counter, printing its label
209    /// via a compile-time if-else chain, then zeroing that slot in a working heap buffer.
210    ///
211    /// Multiple internal tracking slots that resolve to the same source label are merged
212    /// (their counts are summed) before sorting, so each unique source line appears once.
213    pub(crate) fn emit_println_of_array_copy_counter(&mut self) {
214        // Print total.
215        let total_addr = self.array_copy_counter_address();
216        let total_msg = format!("Total arrays copied in {}: {{}}", self.name());
217        self.emit_println_u32(&total_msg, total_addr);
218
219        // Retrieve resolved labels; nothing more to do if there are none.
220        let Some(registry) = self.copy_site_registry.clone() else {
221            return;
222        };
223        let labels = registry.get_resolved_labels();
224        if labels.is_empty() {
225            return;
226        }
227        let n = labels.len().min(MAX_TRACK_SITES);
228
229        // Merge tracking slots that share the same resolved label.
230        // `dedup_labels[j]` is the unique label for slot j in the work buffer.
231        // `dedup_groups[j]` lists the original per-site counter indices that contribute to slot j.
232        let mut dedup_labels: Vec<&str> = Vec::new();
233        let mut dedup_groups: Vec<Vec<usize>> = Vec::new();
234        for (i, label) in labels[..n].iter().enumerate() {
235            if let Some(pos) = dedup_labels.iter().position(|&l| l == label.as_str()) {
236                dedup_groups[pos].push(i);
237            } else {
238                dedup_labels.push(label.as_str());
239                dedup_groups.push(vec![i]);
240            }
241        }
242        let m = dedup_labels.len(); // number of unique locations
243
244        // Allocate a working heap buffer of M slots.
245        // Each slot holds the merged (summed) count for one unique source location.
246        let m_reg = self.make_usize_constant_instruction(F::from(m));
247        let work_ptr = self.allocate_single_addr_usize();
248        self.codegen_allocate_mem(work_ptr.address, m_reg.address);
249
250        for (j, group) in dedup_groups.iter().enumerate() {
251            // Sum all per-site counter values for this group into a temporary register.
252            let sum = self.allocate_single_addr_usize();
253            self.usize_const_instruction(sum.address, F::from(0_usize));
254            for &idx in group {
255                // counter_addr is a direct global address; mov copies the value directly.
256                let counter_addr = self.per_site_counter_address(idx);
257                let cur = self.allocate_single_addr_usize();
258                self.mov_instruction(cur.address, counter_addr);
259                self.memory_op_instruction(
260                    sum.address,
261                    cur.address,
262                    sum.address,
263                    BrilligBinaryOp::Add,
264                );
265            }
266            let j_reg = self.make_usize_constant_instruction(F::from(j));
267            self.codegen_store_with_offset(work_ptr.address, *j_reg, sum.address);
268        }
269
270        // Registers that persist across all MAX_DISPLAY_SITES outer iterations.
271        let max_val = self.allocate_single_addr_usize();
272        let max_idx = self.allocate_single_addr_usize();
273        let bound_reg = self.make_usize_constant_instruction(F::from(m));
274        let one_addr = ReservedRegisters::usize_one();
275
276        for _ in 0..MAX_DISPLAY_SITES {
277            // Initialise: max = working[0], max_idx = 0.
278            self.load_instruction(max_val.address, work_ptr.address);
279            self.usize_const_instruction(max_idx.address, F::from(0_usize));
280
281            // Inner runtime loop: scan working[1..m] and track the maximum.
282            let max_val_addr = max_val.address;
283            let max_idx_addr = max_idx.address;
284            let work_ptr_addr = work_ptr.address;
285            self.codegen_for_loop(Some(one_addr), bound_reg.address, None, |ctx, i_var| {
286                let cur_val = ctx.allocate_single_addr_usize();
287                ctx.codegen_load_with_offset(work_ptr_addr, i_var, cur_val.address);
288                let is_greater = ctx.allocate_single_addr_bool();
289                // is_greater = (max_val < cur_val)
290                ctx.memory_op_instruction(
291                    max_val_addr,
292                    cur_val.address,
293                    is_greater.address,
294                    BrilligBinaryOp::LessThan,
295                );
296                ctx.codegen_if(is_greater.address, |ctx| {
297                    ctx.mov_instruction(max_val_addr, cur_val.address);
298                    ctx.mov_instruction(max_idx_addr, i_var.address);
299                });
300            });
301
302            // Skip if the maximum counter is zero (no more nonzero sites to display).
303            let max_is_zero = self.allocate_single_addr_bool();
304            self.codegen_usize_op(max_val.address, max_is_zero.address, BrilligBinaryOp::Equals, 0);
305            let work_ptr_addr = work_ptr.address;
306            self.codegen_if_not(max_is_zero.address, |ctx| {
307                // Compile-time if-else chain: print the label for whichever slot holds the max.
308                for (j, label) in dedup_labels.iter().enumerate() {
309                    let is_match = ctx.allocate_single_addr_bool();
310                    ctx.codegen_usize_op(
311                        max_idx_addr,
312                        is_match.address,
313                        BrilligBinaryOp::Equals,
314                        j,
315                    );
316                    ctx.codegen_if(is_match.address, |ctx| {
317                        let msg = format!("  {label}: {{}}");
318                        ctx.emit_println_u32(&msg, max_val_addr);
319                    });
320                }
321                // Zero out the selected slot so it is not picked in subsequent iterations.
322                let zero = ctx.make_usize_constant_instruction(F::from(0_usize));
323                ctx.codegen_store_with_offset(
324                    work_ptr_addr,
325                    SingleAddrVariable::new_usize(max_idx_addr),
326                    zero.address,
327                );
328            });
329        }
330    }

332    /// Emit a `print` foreign call that prints `message` as a format string with one u32 substitution.
333    fn emit_println_u32(&mut self, message: &str, value_addr: MemoryAddress) {
334        let newline = ValueOrArray::MemoryAddress(ReservedRegisters::usize_one());
335        let message_len = message.len();
336        let message_value = literal_string_to_value(message, self);
337        let item_count = ValueOrArray::MemoryAddress(ReservedRegisters::usize_one());
338        let value_to_print = ValueOrArray::MemoryAddress(value_addr);
339        let type_string_metadata = literal_string_to_value(PRINT_U32_TYPE_STRING, self);
340        let is_fmt_string = ValueOrArray::MemoryAddress(ReservedRegisters::usize_one());
341
342        let inputs = [
343            newline, // true
344            *message_value,
345            item_count,     // 1
346            value_to_print, // the u32 counter value
347            *type_string_metadata,
348            is_fmt_string, // true
349        ];
350
351        let u1_type = HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U1));
352        let u8_type = HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U8));
353        let u32_type = HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U32));
354
355        let newline_type = u1_type.clone();
356        let size = SemanticLength(assert_u32(message_len));
357        let msg_type = HeapValueType::Array { value_types: vec![u8_type.clone()], size };
358        let item_count_type = HeapValueType::field();
359        let value_to_print_type = u32_type;
360        let size = SemanticLength(assert_u32(PRINT_U32_TYPE_STRING.len()));
361        let metadata_type = HeapValueType::Array { value_types: vec![u8_type], size };
362        let is_fmt_string_type = u1_type;
363
364        let input_types = [
365            newline_type,
366            msg_type,
367            item_count_type,
368            value_to_print_type,
369            metadata_type,
370            is_fmt_string_type,
371        ];
372
373        self.foreign_call_instruction("print".to_string(), &inputs, &input_types, &[], &[]);
374    }

377/// The metadata string needed to tell `print` to print out a u32
378const PRINT_U32_TYPE_STRING: &str = "{\"kind\":\"unsignedinteger\",\"width\":32}";

380/// Allocate the string `data` as a Brillig array on the heap and return it wrapped as a
381/// `HeapArray` value suitable for passing to a `print` foreign call.
382fn literal_string_to_value<F: AcirField + DebugToString, Registers: RegisterAllocator>(
383    data: &str,
384    brillig_context: &mut BrilligContext<F, Registers>,
385) -> Allocated<ValueOrArray, Registers> {
386    let brillig_array =
387        brillig_context.allocate_brillig_array(SemiFlattenedLength(assert_u32(data.len())));
388
389    // Allocate space on the heap.
390    brillig_context.codegen_initialize_array(*brillig_array);
391
392    // Get a pointer to where the items start on the heap.
393    let items_pointer = brillig_context.codegen_make_array_items_pointer(*brillig_array);
394
395    // Copy the data into the array.
396    initialize_constant_string(brillig_context, data, *items_pointer);
397
398    // Wrap the pointer into a `HeapArray`. The `BrilligArray` is no longer needed.
399    let size = SemiFlattenedLength(assert_u32(data.len()));
400    items_pointer.map(|pointer| ValueOrArray::HeapArray(HeapArray { pointer, size }))
401}

403/// Generate opcodes to initialize the memory at `pointer` to the bytes in the `data` string.
404fn initialize_constant_string<F: AcirField + DebugToString, Registers: RegisterAllocator>(
405    brillig_context: &mut BrilligContext<F, Registers>,
406    data: &str,
407    pointer: MemoryAddress,
408) {
409    // Allocate a register for the iterator
410    let write_pointer_register = brillig_context.allocate_register();
411    brillig_context.mov_instruction(*write_pointer_register, pointer);
412
413    for (element_idx, byte) in data.bytes().enumerate() {
414        let byte_field = AcirField::from_le_bytes_reduce(&[byte]);
415        // Store the item in memory
416        brillig_context.indirect_const_instruction(*write_pointer_register, 32, byte_field);
417
418        if element_idx != data.len() - 1 {
419            // Increment the write_pointer_register
420            brillig_context.memory_op_inc_by_usize_one(*write_pointer_register);
421        }
422    }
