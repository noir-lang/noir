12/// Represents a parameter or a return value of an entry point function.
13#[derive(Debug, Clone, Eq, PartialEq, Hash, PartialOrd, Ord)]
14pub(crate) enum BrilligParameter {
15    /// A single address parameter or return value. Holds the bit size of the parameter.
16    SingleAddr(u32),
17    /// An array parameter or return value. Holds the type of an array item and its size.
18    Array(Vec<BrilligParameter>, SemanticLength),
19    /// A vector parameter or return value. Holds the type of a vector item.
20    /// Only known-length vectors can be passed to brillig entry points, so the size is available as well.
21    Vector(Vec<BrilligParameter>, SemanticLength),
22}
23
24impl BrilligParameter {
25    /// Computes the size of a parameter if it was flattened
26    pub(crate) fn flattened_size(&self) -> usize {
27        match self {
28            BrilligParameter::SingleAddr(_) => 1,
29            BrilligParameter::Array(item_types, item_count)
30            | BrilligParameter::Vector(item_types, item_count) => {
31                let size_of_item: usize =
32                    item_types.iter().map(|param| param.flattened_size()).sum();
33                assert_usize(item_count.0) * size_of_item
34            }
35        }
36    }
37}

39/// The result of compiling and linking brillig artifacts.
40/// This is ready to run bytecode with attached metadata.
41#[derive(Debug, Default, Clone)]
42pub(crate) struct GeneratedBrillig<F> {
43    pub(crate) byte_code: Vec<BrilligOpcode<F>>,
44    pub(crate) locations: BTreeMap<OpcodeLocation, CallStackId>,
45    pub(crate) error_types: BTreeMap<ErrorSelector, ErrorType>,
46    pub(crate) name: String,
47    pub(crate) procedure_locations: BTreeMap<ProcedureId, (OpcodeLocation, OpcodeLocation)>,
48}

50impl<F: std::fmt::Display> std::fmt::Display for GeneratedBrillig<F> {
51    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {

53        let width = self.byte_code.len().to_string().len();

62/// Artifacts resulting from the compilation of a function into brillig byte code.
63/// It includes the bytecode of the function and all the metadata that allows linking with other functions.
64pub struct BrilligArtifact<F> {
65    pub(crate) byte_code: Vec<BrilligOpcode<F>>,
66    pub(crate) error_types: BTreeMap<ErrorSelector, ErrorType>,
67    /// The set of jumps that need to have their locations
68    /// resolved.
69    unresolved_jumps: Vec<(JumpInstructionPosition, UnresolvedJumpLocation)>,
70    /// A map of labels to their position in byte code.
71    pub(crate) labels: HashMap<Label, OpcodeLocation>,
72    /// Set of labels which are external to the bytecode.
73    ///
74    /// This will most commonly contain the labels of functions
75    /// which are defined in other bytecode, that this bytecode has called.
76    /// TODO: perhaps we should combine this with the `unresolved_jumps` field
77    /// TODO: and have an enum which indicates whether the jump is internal or external
78    unresolved_external_call_labels: Vec<(JumpInstructionPosition, Label)>,
79    /// Maps the opcodes that are associated with a callstack to it.
80    locations: BTreeMap<OpcodeLocation, CallStackId>,
81    /// The current call stack. All opcodes that are pushed will be associated with this call stack.
82    call_stack_id: CallStackId,
83    /// Name of the function, only used for debugging purposes.
84    pub(crate) name: String,
85
86    /// This field contains the given procedure id if this artifact originates from as procedure
87    pub(crate) procedure: Option<ProcedureId>,
88    /// Procedure ID mapped to the range of its opcode locations
89    /// This is created as artifacts are linked together and allows us to determine
90    /// which opcodes originate from reusable procedures.s
91    /// The range is inclusive for both start and end opcode locations.
92    pub(crate) procedure_locations: BTreeMap<ProcedureId, (OpcodeLocation, OpcodeLocation)>,
93}

109#[derive(Debug, Clone, Eq, PartialEq, Hash)]
110pub(crate) enum LabelType {
111    /// Labels for the entry point bytecode
112    Entrypoint,
113    /// Labels for user defined functions
114    Function(FunctionId, Option<BasicBlockId>),
115    /// Labels for intrinsic procedures
116    Procedure(ProcedureId),
117    /// Label for initialization of globals
118    /// Stores a function ID referencing the entry point
119    GlobalInit(FunctionId),
120}
121
122impl std::fmt::Display for LabelType {
123    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
124        match self {
125            LabelType::Function(function_id, block_id) => {
126                if let Some(block_id) = block_id {
127                    write!(f, "Function({function_id:?}, {block_id:?})")
128                } else {
129                    write!(f, "Function({function_id:?})")
130                }
131            }
132            LabelType::Entrypoint => write!(f, "Entrypoint"),
133            LabelType::Procedure(procedure_id) => write!(f, "Procedure({procedure_id:?})"),
134            LabelType::GlobalInit(function_id) => {
135                write!(f, "Globals Initialization({function_id:?})")
136            }
137        }
138    }
139}

143/// It is assumed that an entity will keep a map

146pub(crate) struct Label {

158    }

190/// When constructing the bytecode, there may be instructions
191/// which require one to jump to a specific region of code (function)
192///
193/// The position of a function cannot always be known
194/// at this point in time, so Jumps are unresolved
195/// until all functions/all of the bytecode has been processed.
196/// `Label` is used as the jump location and once all of the bytecode
197/// has been processed, the jumps are resolved using a map from Labels
198/// to their position in the bytecode.
199pub(crate) type UnresolvedJumpLocation = Label;

201impl<F: Clone + std::fmt::Debug> BrilligArtifact<F> {
202    /// Resolves all jumps and generates the final bytecode
203    pub(crate) fn finish(mut self) -> GeneratedBrillig<F> {
204        self.resolve_jumps();
205        GeneratedBrillig {
206            byte_code: self.byte_code,
207            locations: self.locations,
208            error_types: self.error_types,
209            name: self.name,
210            procedure_locations: self.procedure_locations,
211        }
212    }

214    /// Gets the first unresolved function call of this artifact.
215    pub(crate) fn first_unresolved_function_call(&self) -> Option<Label> {
216        self.unresolved_external_call_labels.first().map(|(_, label)| label.clone())
217    }
218
219    /// Link with an external brillig artifact called from this artifact.
220    ///
221    /// This method will offset the positions in the Brillig artifact to
222    /// account for the fact that it is being appended to the end of this
223    /// Brillig artifact (self).
224    pub(crate) fn link_with(&mut self, obj: &BrilligArtifact<F>) {
225        // Add the unresolved jumps of the linked function to this artifact.
226        self.add_unresolved_jumps_and_calls(obj);
227
228        for (error_selector, error_type) in &obj.error_types {
229            self.error_types.insert(*error_selector, error_type.clone());
230        }
231
232        self.byte_code.append(&mut obj.byte_code.clone());
233
234        // Remove all resolved external calls and transform them to jumps
235        let is_resolved = |label: &Label| self.labels.contains_key(label);
236
237        let resolved_external_calls = self
238            .unresolved_external_call_labels
239            .iter()
240            .filter(|(_, label)| is_resolved(label))
241            .cloned()
242            .collect::<Vec<_>>();
243
244        for resolved_external_call in resolved_external_calls {
245            self.unresolved_jumps.push(resolved_external_call);
246        }
247
248        self.unresolved_external_call_labels.retain(|(_, label)| !is_resolved(label));
249    }
250
251    /// Adds unresolved jumps & function calls from another artifact offset by the current opcode count in the artifact.
252    fn add_unresolved_jumps_and_calls(&mut self, obj: &BrilligArtifact<F>) {
253        let offset = self.index_of_next_opcode();
254        for (jump_label, jump_location) in &obj.unresolved_jumps {
255            self.unresolved_jumps.push((jump_label + offset, jump_location.clone()));
256        }
257
258        for (label_id, position_in_bytecode) in &obj.labels {
259            let old_value = self.labels.insert(label_id.clone(), position_in_bytecode + offset);
260            assert!(old_value.is_none(), "overwriting label {label_id} {old_value:?}");
261        }
262
263        for (position_in_bytecode, label_id) in &obj.unresolved_external_call_labels {
264            self.unresolved_external_call_labels
265                .push((position_in_bytecode + offset, label_id.clone()));
266        }
267
268        for (position_in_bytecode, call_stack) in obj.locations.iter() {
269            self.locations.insert(position_in_bytecode + offset, *call_stack);
270        }
271    }
272
273    /// Adds a brillig instruction to the brillig byte code
274    pub(crate) fn push_opcode(&mut self, opcode: BrilligOpcode<F>) {
275        if !self.call_stack_id.is_root() {
276            self.locations.insert(self.index_of_next_opcode(), self.call_stack_id);
277        }
278        self.byte_code.push(opcode);
279    }
280
281    /// Adds a unresolved jump to be fixed at the end of bytecode processing.
282    pub(crate) fn add_unresolved_jump(
283        &mut self,
284        jmp_instruction: BrilligOpcode<F>,
285        destination: UnresolvedJumpLocation,
286    ) {
287        assert!(
288            Self::is_jmp_instruction(&jmp_instruction),
289            "expected a jump instruction, but found {jmp_instruction:?}"
290        );
291
292        self.unresolved_jumps.push((self.index_of_next_opcode(), destination));
293        self.push_opcode(jmp_instruction);
294    }
295    /// Adds a unresolved external call that will be fixed once linking has been done.
296    pub(crate) fn add_unresolved_external_call(
297        &mut self,
298        call_instruction: BrilligOpcode<F>,
299        destination: UnresolvedJumpLocation,
300    ) {
301        // TODO: Add a check to ensure that the opcode is a call instruction
302
303        self.unresolved_external_call_labels.push((self.index_of_next_opcode(), destination));
304        self.push_opcode(call_instruction);
305    }

307    /// Returns true if the opcode is a jump instruction
308    fn is_jmp_instruction(instruction: &BrilligOpcode<F>) -> bool {
309        matches!(instruction, BrilligOpcode::JumpIf { .. } | BrilligOpcode::Jump { .. })
310    }
311
312    /// Adds a label in the bytecode to specify where this block's
313    /// opcodes will start.
314    pub(crate) fn add_label_at_position(&mut self, label: Label, position: OpcodeLocation) {
315        let old_value = self.labels.insert(label.clone(), position);
316        assert!(
317            old_value.is_none(),
318            "overwriting label {label}. old_value = {old_value:?}, new_value = {position}"
319        );
320    }

322    /// Returns the index of the next opcode.
323    ///
324    /// This is useful for labelling regions of code
325    /// before we start generating the opcodes for the region.
326    pub(crate) fn index_of_next_opcode(&self) -> OpcodeLocation {
327        self.byte_code.len()
328    }
329
330    /// Resolves all of the unresolved jumps in the program.
331    ///
332    /// Note: This should only be called once all blocks are processed and
333    /// linkage with other bytecode has happened.
334    fn resolve_jumps(&mut self) {
335        for (location_of_jump, unresolved_location) in &self.unresolved_jumps {
336            let resolved_location = self.labels[unresolved_location];
337
338            let jump_instruction = self.byte_code[*location_of_jump].clone();
339            match jump_instruction {
340                BrilligOpcode::Jump { location } => {
341                    assert_eq!(
342                        location, 0,
343                        "location is not zero, which means that the jump label does not need resolving"
344                    );
345
346                    self.byte_code[*location_of_jump] =
347                        BrilligOpcode::Jump { location: resolved_location };
348                }
349                BrilligOpcode::JumpIf { condition, location } => {
350                    assert_eq!(
351                        location, 0,
352                        "location is not zero, which means that the jump label does not need resolving"
353                    );
354
355                    self.byte_code[*location_of_jump] =
356                        BrilligOpcode::JumpIf { condition, location: resolved_location };
357                }
358                BrilligOpcode::Call { location } => {
359                    assert_eq!(
360                        location, 0,
361                        "location is not zero, which means that the call label does not need resolving"
362                    );
363
364                    self.byte_code[*location_of_jump] =
365                        BrilligOpcode::Call { location: resolved_location };
366                }
367                _ => unreachable!(
368                    "all jump labels should point to a jump instruction in the bytecode"
369                ),
370            }
371        }
372    }
