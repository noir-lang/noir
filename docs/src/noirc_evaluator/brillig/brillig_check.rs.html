33pub(crate) enum OpcodeAdvisory {
34    /// A memory address is being written by the opcode that no other following opcode reads.
35    NeverRead { addr: MemoryAddress },
36    /// A memory address is being written by the opcode that will get overwritten before being read.
37    OverwrittenBeforeRead { addr: MemoryAddress, write_at: OpcodeLocation, read_at: OpcodeLocation },
38}

40pub(crate) type OpcodeAdvisories = HashMap<OpcodeLocation, Vec<OpcodeAdvisory>>;

42/// Go through opcodes and collect advisories, indicating opcodes which are potentially unused.
43///
44/// At the moment this relies on the CFG still available from the SSA via the [FunctionContext].
45/// Unfortunately this doesn't work for the artifacts created by `compile_procedures`,
46/// which use looping in the bytecode level. To handle them in isolation, we would need to
47/// reconstruct the CFG from unresolved jump labels, and then establish a post order from that.
48pub(crate) fn opcode_advisories<F: AcirField>(
49    function: &Function,
50    function_context: &FunctionContext,
51    brillig_context: &BrilligContext<F, Stack>,
52) -> OpcodeAdvisories {
53    // Find where each block start and ends in the bytecode.
54    let block_opcode_ranges = block_opcode_ranges(brillig_context);
55
56    // Accumulate stack addresses read. These are all in the same function context,
57    // so they should fit in one stack frame, which makes relative addresses comparable.
58    let registers = brillig_context.registers();
59    let addr_range = Range { start: registers.start(), end: registers.end() };
60
61    // First collect all reads per block.
62    let reads_in_blocks: HashMap<BasicBlockId, im::HashSet<MemoryAddress>> = function_context
63        .reverse_post_order()
64        .map(|block_id| {
65            let mut read_collector = ReadCollector::new(&addr_range);
66            let opcode_range = block_opcode_ranges[&block_id].clone();
67            for loc in opcode_range {
68                let opcode = &brillig_context.artifact().byte_code[loc];
69                read_collector.visit_opcode(opcode, loc);
70            }
71            (block_id, read_collector.into_reads())
72        })
73        .collect();
74
75    // Find each loop in the function. If a block is part of a loop and writes to an address,
76    // we need to consider all blocks that we can get to from the header as a potential descendant.
77    let blocks_in_loops: HashMap<BasicBlockId, _> = Loops::find_all(function, LoopOrder::OutsideIn)
78        .yet_to_unroll
79        .into_iter()
80        .map(|loop_| (loop_.header, loop_.blocks))
81        .collect();
82
83    // Collect all the reads that happen in the descendants of blocks.
84    let mut reads_in_descendants: HashMap<BasicBlockId, im::HashSet<MemoryAddress>> =
85        HashMap::default();
86
87    // Going in Post Order, so successors will have been processed already,
88    // except for back-edges: the header of the loop they return to comes later.
89    for block_id in function_context.post_order() {
90        // The reads are a union of addresses read:
91        // * directly by the successors
92        // * indirectly by the descendants of the successors
93        let mut acc = im::HashSet::new();
94        for successor_id in function.dfg[block_id].successors() {
95            acc = acc.union(reads_in_blocks[&successor_id].clone());
96            // If this is a back-edge block, the header will not have been done yet.
97            if let Some(reads) = reads_in_descendants.get(&successor_id) {
98                acc = acc.union(reads.clone());
99            }
100        }
101        // If the block is a header of a loop, then anything we can reach from the header,
102        // we can reach from any block in the loop too.
103        // For example say we have a CFG like this, b1 is the header of a loop [b1, b2, b3]
104        //      +-------+
105        //      v       |
106        // b0--b1--b2--b3
107        //      +----------b4
108        // An address can be written in b3 and read in b1 (e.g. the loop variable),
109        // or be written in b2 and read in b4 (e.g. a reference allocated in b0).
110        if let Some(blocks_in_loop) = blocks_in_loops.get(&block_id) {
111            for member_id in blocks_in_loop {
112                if *member_id == block_id {
113                    continue;
114                }
115                // Blocks in the loop are successors, they should be done already.
116                let member_reads = reads_in_descendants[member_id].clone();
117                reads_in_descendants.insert(*member_id, member_reads.union(acc.clone()));
118            }
119        }
120        // We are done with this node, can move on to its predecessors.
121        reads_in_descendants.insert(block_id, acc);
122    }
123
124    // Now that we have the reads in the blocks and their descendants, look at the writes.
125    let mut advisories = HashMap::new();
126    for block_id in function_context.reverse_post_order() {
127        let mut advisory_collector =
128            AdvisoryCollector::new(&addr_range, &reads_in_descendants[&block_id]);
129        let opcode_range = block_opcode_ranges[&block_id].clone();
130
131        // Going backwards, so reads at the end are recorded before earlier writes.
132        for loc in opcode_range.rev() {
133            let opcode = &brillig_context.artifact().byte_code[loc];
134            advisory_collector.visit_opcode(opcode, loc);
135        }
136
137        advisories.extend(advisory_collector.into_advisories());
138    }
139    advisories
140}

142/// Display the opcodes and their corresponding advisories.
143pub(crate) fn show_opcode_advisories<F: Display>(
144    advisories: &OpcodeAdvisories,
145    artifact: &BrilligArtifact<F>,
146) {
147    println!(
148        "\n// There are {} Brillig opcode advisories for function '{}'",
149        advisories.len(),
150        artifact.name
151    );
152
153    if advisories.is_empty() {
154        return;
155    }
156
157    let advisory = |opcode: &Opcode<F>, msg: &str| {
158        let mut a = format!("{opcode}");
159        if msg.is_empty() {
160            return a;
161        }
162        for _ in a.len()..40 {
163            a.push(' ');
164        }
165        a.push_str(" // ");
166        a.push_str(msg);
167        a
168    };
169
170    println!("fn {}", artifact.name);
171    let index_width = artifact.byte_code.len().to_string().len();
172    for (index, opcode) in artifact.byte_code.iter().enumerate() {
173        let msg = if let Some(ads) = advisories.get(&index) {
174            &ads.iter()
175                .map(|a| match a {
176                    OpcodeAdvisory::NeverRead { addr } => format!("{addr} is never read"),
177                    OpcodeAdvisory::OverwrittenBeforeRead { addr, write_at, read_at } => {
178                        format!("{addr} is overwritten at {write_at} before read at {read_at}")
179                    }
180                })
181                .collect::<Vec<_>>()
182                .join("; ")
183        } else {
184            ""
185        };
186        println!("{index:>index_width$}: {}", advisory(opcode, msg));
187    }
188}

190/// Go through the labels in the [BrilligContext] and collect the opcode locations
191/// where each block starts and ends.
192///
193/// # Assumptions on the label set
194///
195/// This function operates on a per-function Brillig artifact, *before* linking.
196/// At that point the artifact's `labels` map is expected to only contain
197/// `LabelType::Function(_, _)` entries:
198///
199/// 1. All entries share a single `FunctionId` — the function this artifact
200///    belongs to.
201/// 2. When sorted by opcode location, labels for one `block_id` form a
202///    contiguous run. Section markers (`Label { section: Some(n), .. }`) sit
203///    between a block's primary label and the next block's first label.
204///    A given `block_id` never reappears after another block's labels.
205/// 3. No `Procedure`, `GlobalInit`, or `Entrypoint` labels appear here.
206///
207/// Function-entry labels (`Function(_, None)`) frequently share an opcode
208/// location with the first block label (both land at offset 0 when there is
209/// no `check_max_stack_depth` call and no spill prologue). They are not block
210/// boundaries — only block labels are — so we drop them up-front. Otherwise
211/// the non-deterministic iteration order of equal-location entries in the
212/// underlying `HashMap` could land a function-entry label between block
213/// labels of the same `block_id` and produce an empty or truncated range.
214///
215/// `debug_assert!`s catch regressions in invariants 1 and 3 directly, and the
216/// `close_block` helper catches a violation of invariant 2 by refusing to
217/// overwrite an existing range.
218fn block_opcode_ranges<F, R: RegisterAllocator>(
219    brillig_context: &BrilligContext<F, R>,
220) -> HashMap<BasicBlockId, Range<OpcodeLocation>> {
221    let mut ranges = HashMap::new();
222
223    // Close out a block range. Debug-asserts that the block hasn't already
224    // been closed: this is how invariant 2 (contiguity) is enforced — a
225    // non-contiguous label sequence would close the same `block_id` twice.
226    let close_block = |ranges: &mut HashMap<BasicBlockId, Range<OpcodeLocation>>,
227                       block_id: BasicBlockId,
228                       range: Range<OpcodeLocation>| {
229        let previous = ranges.insert(block_id, range);
230        debug_assert!(
231            previous.is_none(),
232            "block_opcode_ranges: labels for block {block_id:?} are not contiguous when sorted \
233             by location"
234        );
235    };
236
237    // Used only by the debug assertion below: the artifact must contain labels
238    // for a single function.
239    let mut function_id = None;
240
241    // Project to (block_id, location) pairs, dropping non-block (function-entry)
242    // labels and asserting invariants 1 and 3.
243    let mut block_labels: Vec<(BasicBlockId, OpcodeLocation)> = brillig_context
244        .artifact()
245        .labels
246        .iter()
247        .filter_map(|(label, location)| match &label.label_type {
248            LabelType::Function(fid, block_id) => {
249                match function_id {
250                    None => function_id = Some(*fid),
251                    Some(expected) => debug_assert_eq!(
252                        *fid, expected,
253                        "block_opcode_ranges expects all labels to share one FunctionId"
254                    ),
255                }
256                block_id.map(|bid| (bid, *location))
257            }
258            other => {
259                debug_assert!(
260                    false,
261                    "block_opcode_ranges expects only Function labels in a per-function \
262                     artifact, found {other:?}"
263                );
264                None
265            }
266        })
267        .collect();
268    block_labels.sort_by_key(|(_, location)| *location);
269
270    let mut last_start: Option<(BasicBlockId, OpcodeLocation)> = None;
271    for (block_id, location) in block_labels {
272        match last_start {
273            None => last_start = Some((block_id, location)),
274            Some((current_block, start)) => {
275                if current_block != block_id {
276                    close_block(&mut ranges, current_block, Range { start, end: location });
277                    last_start = Some((block_id, location));
278                }
279            }
280        }
281    }
282    if let Some((block_id, start)) = last_start {
283        close_block(
284            &mut ranges,
285            block_id,
286            Range { start, end: brillig_context.artifact().byte_code.len() },
287        );
288    }
289
290    ranges
291}

293/// Collect all the relative addresses read in a range of relative address offsets.
294struct ReadCollector<'a> {
295    addr_range: &'a Range<Offset>,
296    reads: im::HashSet<MemoryAddress>,
297}
298
299impl<'a> ReadCollector<'a> {
300    fn new(addr_range: &'a Range<Offset>) -> Self {
301        Self { addr_range, reads: im::HashSet::new() }
302    }
303
304    fn into_reads(self) -> im::HashSet<MemoryAddress> {
305        self.reads
306    }
307}
308
309impl OpcodeAddressVisitor for ReadCollector<'_> {
310    fn read(&mut self, addr: &MemoryAddress, _location: OpcodeLocation) {
311        if addr.is_relative() && self.addr_range.contains(&assert_usize(addr.to_u32())) {
312            self.reads.insert(*addr);
313        }
314    }
315
316    fn write(&mut self, _addr: &MemoryAddress, _location: OpcodeLocation) {
317        // Ignore writes.
318    }
319
320    fn should_visit_opcode<F>(&mut self, _opcode: &Opcode<F>) -> bool {
321        true
322    }
323}

325/// Collect advisories about the opcodes in a single block.
326///
327/// Assumes that we are visiting the opcodes going backwards.
328struct AdvisoryCollector<'a> {
329    addr_range: &'a Range<Offset>,
330    /// Addresses read in any of the descendants of the block.
331    reads_in_descendants: &'a im::HashSet<MemoryAddress>,
332    /// Last location the address was read from in this block.
333    reads: HashMap<MemoryAddress, OpcodeLocation>,
334    /// Last location the address was written to in this block.
335    writes: HashMap<MemoryAddress, OpcodeLocation>,
336    /// Advisories collected in this block.
337    advisories: OpcodeAdvisories,
338    /// Position relative to a call's argument-passing region, walking backwards.
339    in_call_region: CallRegion,
340    /// Indicate that we are in the region where we are passing out return parameters.
341    in_return_region: bool,
342    /// Indicate whether the current opcode can fail or has side effects, which
343    /// means it needs to be kept even if its result is never read.
344    is_fallible: bool,
345}
346
347impl<'a> AdvisoryCollector<'a> {
348    fn new(
349        addr_range: &'a Range<Offset>,
350        reads_in_descendants: &'a im::HashSet<MemoryAddress>,
351    ) -> Self {
352        Self {
353            addr_range,
354            reads_in_descendants,
355            reads: HashMap::new(),
356            writes: HashMap::new(),
357            advisories: HashMap::new(),
358            in_call_region: CallRegion::Outside,
359            in_return_region: false,
360            is_fallible: false,
361        }
362    }
363
364    fn into_advisories(self) -> OpcodeAdvisories {
365        self.advisories
366    }
367
368    fn addr_in_range(&self, addr: &MemoryAddress) -> bool {
369        addr.is_relative() && self.addr_range.contains(&assert_usize(addr.to_u32()))
370    }
371
372    fn add_advisory(&mut self, location: OpcodeLocation, advisory: OpcodeAdvisory) {
373        self.advisories.entry(location).or_default().push(advisory);
374    }
375}

377impl OpcodeAddressVisitor for AdvisoryCollector<'_> {
378    /// Maintain some contextual information about regions.
379    ///
380    /// This is assuming that we are visiting the opcodes in reverse.
381    fn should_visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>) -> bool {
382        // The return instruction can be preceded by a number of moves from various parts of the stack
383        // to the beginning, depending on the number of return values. They destinations can be shuffled.
384        if matches!(opcode, Opcode::Return) {
385            self.in_return_region = true;
386        } else if !matches!(opcode, Opcode::Mov { .. } | Opcode::Load { .. }) {
387            self.in_return_region = false;
388        }
389
390        self.is_fallible = is_fallible_opcode(opcode);
391
392        match opcode {
393            // Handle the conventions of `codegen_call`: we are passing call arguments by coping them
394            // to new stack registers, then we set the stack pointer @0, which changes the meaning of
395            // relative addresses. The parameters are going to be read by another function, not visible
396            // in the scope of a single function we are analyzing here.
397            Opcode::Mov { destination, source } => {
398                if *destination == ReservedRegisters::stack_pointer()
399                    && *source == MemoryAddress::relative(0)
400                {
401                    // This is the restore of the stack pointer, the end of a call.
402                    self.in_call_region = CallRegion::Inside;
403                    false
404                } else if self.in_call_region == CallRegion::Inside
405                    && *source == ReservedRegisters::stack_pointer()
406                {
407                    // This is where we store the stack pointer in the first address of the next stack frame.
408                    // The next opcode is a const that contains the current stack size.
409                    self.in_call_region = CallRegion::AtStackSize;
410                    false
411                } else {
412                    // We can visit the Mov instructions that copy parameters and return values;
413                    // we may need to ignore the destination, but we can still register the reads.
414                    true
415                }
416            }
417            Opcode::Const { .. } if self.in_call_region == CallRegion::AtStackSize => {
418                // This is the instruction where we set the stack size at the beginning of the call.
419                self.in_call_region = CallRegion::Outside;
420                true
421            }
422            // A `constrain` of some arbitrary expression is usually broken up into an instruction with
423            // a boolean result, and then a constrain on the result being equal to 1. Constant allocation
424            // creates a constant for that 1, but the codegen later might not use it, but rather check
425            // the variable directly, since it's already a bool value.
426            Opcode::Const {
427                destination: _,
428                bit_size: BitSize::Integer(IntegerBitSize::U1),
429                value,
430            } => !value.is_one(),
431            _ => true,
432        }
433    }
434
435    /// Remember the opcode location where a memory address was last read.
436    fn read(&mut self, addr: &MemoryAddress, location: OpcodeLocation) {
437        if !self.addr_in_range(addr) {
438            return;
439        }
440        self.reads.insert(*addr, location);
441    }
442
443    /// Remember the opcode location where a memory address was last written.
444    ///
445    /// Insert an advisory if:
446    /// * the address is not read after this opcode
447    /// * the address is read after this, but before that there is another write
448    fn write(&mut self, addr: &MemoryAddress, location: OpcodeLocation) {
449        if !self.addr_in_range(addr) {
450            return;
451        }
452        if self.in_call_region.is_active() || self.in_return_region {
453            // Ignore the write, it's a parameter meant for another function.
454            // The reads can be inspected, as they should consume data we have prepared.
455            return;
456        }
457
458        // Some opcodes can have side effects, in which case we need to run them even
459        // if we don't use their results.
460        let ignore_unused = self.is_fallible;
461
462        if !ignore_unused {
463            if let Some(read_at) = self.reads.get(addr) {
464                if let Some(write_at) = self.writes.get(addr)
465                    && write_at < read_at
466                {
467                    self.add_advisory(
468                        location,
469                        OpcodeAdvisory::OverwrittenBeforeRead {
470                            addr: *addr,
471                            write_at: *write_at,
472                            read_at: *read_at,
473                        },
474                    );
475                }
476            } else if !self.reads_in_descendants.contains(addr) {
477                self.add_advisory(location, OpcodeAdvisory::NeverRead { addr: *addr });
478            }
479        }
480
481        self.writes.insert(*addr, location);
482    }
483}

485trait OpcodeAddressVisitor {
486    /// Decide if an opcode should be visited.
487    ///
488    /// Can mutate self to set flags that allows it to ignore upcoming opcodes as well.
489    fn should_visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>) -> bool;
490
491    /// Called with all the addresses read by the opcode.
492    fn read(&mut self, addr: &MemoryAddress, location: OpcodeLocation);
493
494    /// Called with all addresses written by the opcode.
495    fn write(&mut self, addr: &MemoryAddress, location: OpcodeLocation);
496
497    /// Call with each opcode to record reads and writes.
498    ///
499    /// The expected order of visit depends on the purpose of the visitor implementation.
500    fn visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>, location: OpcodeLocation) {
501        if !self.should_visit_opcode(opcode) {
502            return;
503        }
504        match opcode {
505            Opcode::BinaryFieldOp { destination, lhs, rhs, .. }
506            | Opcode::BinaryIntOp { destination, lhs, rhs, .. } => {
507                self.read(lhs, location);
508                self.read(rhs, location);
509                self.write(destination, location);
510            }
511            Opcode::Cast { destination, source, .. }
512            | Opcode::Mov { destination, source }
513            | Opcode::Not { destination, source, .. } => {
514                self.read(source, location);
515                self.write(destination, location);
516            }
517            Opcode::Load { destination, source_pointer } => {
518                self.read(source_pointer, location);
519                self.write(destination, location);
520            }
521            Opcode::Store { destination_pointer, source, .. } => {
522                self.read(destination_pointer, location); // the write goes where the pointer points
523                self.read(source, location);
524            }
525            Opcode::JumpIf { condition, .. } => {
526                self.read(condition, location);
527            }
528            Opcode::CalldataCopy { destination_address, size_address, offset_address } => {
529                self.read(size_address, location);
530                self.read(offset_address, location);
531                self.write(destination_address, location);
532            }
533            Opcode::Const { destination, .. } => {
534                self.write(destination, location);
535            }
536            Opcode::IndirectConst { destination_pointer, .. } => {
537                self.read(destination_pointer, location);
538            }
539            Opcode::ForeignCall { destinations, inputs, .. } => {
540                for input in inputs {
541                    self.read_value_or_array(input, location);
542                }
543                for destination in destinations {
544                    match destination {
545                        ValueOrArray::HeapArray(_) | ValueOrArray::HeapVector(_) => {
546                            // Consider the destinations as something the VM will read;
547                            // these are pointers to addresses on the heap to write directly into,
548                            // we don't need to read their values afterwards.
549                            self.read_value_or_array(destination, location);
550                        }
551                        ValueOrArray::MemoryAddress(_) => {
552                            // Simple addresses should be used read later to make use of them.
553                        }
554                    }
555
556                    // We can also consider them being written to.
557                    self.write_value_or_array(destination, location);
558                }
559            }
560            Opcode::ConditionalMov { source_a, source_b, condition, destination } => {
561                self.read(condition, location);
562                self.read(source_a, location);
563                self.read(source_b, location);
564                self.write(destination, location);
565            }
566            Opcode::BlackBox(black_box_op) => self.visit_black_box_op(black_box_op, location),
567            Opcode::Trap { revert_data } => self.read_heap_vector(revert_data, location),
568            Opcode::Stop { return_data } => self.read_heap_vector(return_data, location),
569            Opcode::Jump { .. } | Opcode::Call { .. } | Opcode::Return => {}
570        }
571    }
572
573    fn read_heap_array(&mut self, array: &HeapArray, location: OpcodeLocation) {
574        self.read(&array.pointer, location);
575    }
576
577    fn read_heap_vector(&mut self, vector: &HeapVector, location: OpcodeLocation) {
578        self.read(&vector.pointer, location);
579        self.read(&vector.size, location);
580    }
581
582    fn read_value_or_array(&mut self, value: &ValueOrArray, location: OpcodeLocation) {
583        match value {
584            ValueOrArray::MemoryAddress(memory_address) => self.read(memory_address, location),
585            ValueOrArray::HeapArray(heap_array) => self.read_heap_array(heap_array, location),
586            ValueOrArray::HeapVector(heap_vector) => self.read_heap_vector(heap_vector, location),
587        }
588    }
589
590    fn write_heap_array(&mut self, array: &HeapArray, location: OpcodeLocation) {
591        // The pointer register is read by the VM to find the heap location
592        // it should write through. Marking it as read ensures upstream
593        // instructions that set up this pointer aren't mistakenly flagged
594        // as dead writes. The accompanying `write` tracks the
595        // heap-side effect for `OverwrittenBeforeRead` purposes.
596        self.read(&array.pointer, location);
597        self.write(&array.pointer, location);
598    }
599
600    fn write_heap_vector(&mut self, vector: &HeapVector, location: OpcodeLocation) {
601        // Same reasoning as `write_heap_array`: the pointer is read by the
602        // VM, so mark it as both read and written.
603        self.read(&vector.pointer, location);
604        self.write(&vector.pointer, location);
605        self.write(&vector.size, location);
606    }
607
608    fn write_value_or_array(&mut self, value: &ValueOrArray, location: OpcodeLocation) {
609        match value {
610            ValueOrArray::MemoryAddress(memory_address) => self.write(memory_address, location),
611            ValueOrArray::HeapArray(heap_array) => self.write_heap_array(heap_array, location),
612            ValueOrArray::HeapVector(heap_vector) => self.write_heap_vector(heap_vector, location),
613        }
614    }
615
616    fn visit_black_box_op(&mut self, op: &BlackBoxOp, location: OpcodeLocation) {
617        match op {
618            BlackBoxOp::AES128Encrypt { inputs, iv, key, outputs } => {
619                self.read_heap_array(inputs, location);
620                self.read_heap_array(iv, location);
621                self.read_heap_array(key, location);
622                self.write_heap_array(outputs, location);
623            }
624            BlackBoxOp::Blake2s { message, output } | BlackBoxOp::Blake3 { message, output } => {
625                self.read_heap_array(message, location);
626                self.write_heap_array(output, location);
627            }
628            BlackBoxOp::Keccakf1600 { input, output } => {
629                self.read_heap_array(input, location);
630                self.write_heap_array(output, location);
631            }
632            BlackBoxOp::EcdsaSecp256k1 {
633                hashed_msg,
634                public_key_x,
635                public_key_y,
636                signature,
637                result,
638            }
639            | BlackBoxOp::EcdsaSecp256r1 {
640                hashed_msg,
641                public_key_x,
642                public_key_y,
643                signature,
644                result,
645            } => {
646                self.read_heap_array(hashed_msg, location);
647                self.read_heap_array(public_key_x, location);
648                self.read_heap_array(public_key_y, location);
649                self.read_heap_array(signature, location);
650                self.write(result, location);
651            }
652            BlackBoxOp::MultiScalarMul { points, scalars, outputs } => {
653                self.read_heap_array(points, location);
654                self.read_heap_array(scalars, location);
655                self.write_heap_array(outputs, location);
656            }
657            BlackBoxOp::EmbeddedCurveAdd { input1_x, input1_y, input2_x, input2_y, result } => {
658                self.read(input1_x, location);
659                self.read(input1_y, location);
660                self.read(input2_x, location);
661                self.read(input2_y, location);
662                self.write_heap_array(result, location);
663            }
664            BlackBoxOp::Poseidon2Permutation { message, output } => {
665                self.read_heap_array(message, location);
666                self.write_heap_array(output, location);
667            }
668            BlackBoxOp::Sha256Compression { input, hash_values, output } => {
669                self.read_heap_array(input, location);
670                self.read_heap_array(hash_values, location);
671                self.write_heap_array(output, location);
672            }
673            BlackBoxOp::ToRadix { input, radix, output_pointer, num_limbs, output_bits } => {
674                self.read(input, location);
675                self.read(radix, location);
676                self.read(num_limbs, location);
677                self.read(output_bits, location);
678                self.read(output_pointer, location); // indirect
679            }
680        }
681    }
682}

684/// The highest relative (stack-frame) register index referenced anywhere in `byte_code`.
685///
686/// Reuses [`OpcodeAddressVisitor`] so it stays in sync with the opcode set. For a call-free
687/// function this is the peak stack-frame register usage, i.e. the allocator high-water mark
688/// that the spill decision is meant to predict.
689#[cfg(test)]
690pub(crate) fn max_relative_register<F: AcirField>(byte_code: &[Opcode<F>]) -> usize {
691    struct MaxRelative(usize);
692
693    impl OpcodeAddressVisitor for MaxRelative {
694        fn should_visit_opcode<G: AcirField>(&mut self, _opcode: &Opcode<G>) -> bool {
695            true
696        }
697
698        fn read(&mut self, addr: &MemoryAddress, _location: OpcodeLocation) {
699            if let MemoryAddress::Relative(index) = addr {
700                self.0 = self.0.max(*index as usize);
701            }
702        }
703
704        fn write(&mut self, addr: &MemoryAddress, _location: OpcodeLocation) {
705            if let MemoryAddress::Relative(index) = addr {
706                self.0 = self.0.max(*index as usize);
707            }
708        }
709    }
710
711    let mut visitor = MaxRelative(0);
712    for (location, opcode) in byte_code.iter().enumerate() {
713        visitor.visit_opcode(opcode, location);
714    }
715    visitor.0
716}

718/// Whether the opcode's result must be preserved even if it is never read.
719///
720/// True for opcodes that can trap on user input (division by zero) or have
721/// externally observable side effects (foreign calls, black-box ops). Opcodes
722/// whose only failure modes come from miscompilation (e.g. bit-size mismatches,
723/// non-boolean conditions in `JumpIf`/`ConditionalMov`) are not considered
724/// fallible here, because correctly generated bytecode cannot exercise them.
725///
726/// Notable opcodes deliberately *excluded*:
727/// * `Shl`/`Shr` — the Brillig VM saturates to zero on out-of-range shift
728///   counts rather than trapping. The user-facing range check is emitted as a
729///   separate `JumpIf` + `Trap` sequence whose result register is independent
730///   of the shift's destination.
731/// * `Add`/`Sub`/`Mul` — wrap in the VM; the overflow check is again a
732///   separate `JumpIf` + `Trap` sequence.
733/// * `Cast` — truncates and never traps.
734/// * `Trap`/`Stop`/`Call`/`Return`/`Jump`/`JumpIf` — control-flow opcodes that
735///   do not write to any tracked memory address, so this flag would not affect
736///   their handling either way.
737fn is_fallible_opcode<F>(opcode: &Opcode<F>) -> bool {
738    match opcode {
739        // Division by zero traps in the VM.
740        Opcode::BinaryFieldOp { op: BinaryFieldOp::Div | BinaryFieldOp::IntegerDiv, .. }
741        | Opcode::BinaryIntOp { op: BinaryIntOp::Div, .. } => true,
742        // Mirror the classification used by ACIR's
743        // [`acvm::acir::BlackBoxFunc::has_side_effects`]: only black-box ops that
744        // validate inputs, alter circuit state, or produce constraint-relevant
745        // side effects are treated as fallible. Pure hashes and ciphers can be
746        // removed when their result is never observed.
747        Opcode::BlackBox(black_box_op) => is_fallible_black_box_op(black_box_op),
748        // Foreign calls invoke an oracle which is externally observable and
749        // may fail on bad inputs.
750        Opcode::ForeignCall { .. } => true,
751        _ => false,
752    }
753}
754
755/// Side-effecting / failing black-box ops. Delegates to ACIR's
756/// [`acvm::acir::BlackBoxFunc::has_side_effects`] for the ACIR-mapped
757/// variants (so any future change to that classification flows through here
758/// automatically), with [`BlackBoxOp::ToRadix`] handled explicitly because
759/// the brillig VM validates its `radix` argument against `[2, 256]` at
760/// runtime and ToRadix has no ACIR analog.
761fn is_fallible_black_box_op(op: &BlackBoxOp) -> bool {
762    if let Some(func) = black_box_op_to_acir_func(op) {
763        func.has_side_effects()
764    } else {
765        matches!(op, BlackBoxOp::ToRadix { .. })
766    }
767}

769/// Map a brillig [`BlackBoxOp`] to its ACIR [`BlackBoxFunc`] counterpart.
770/// `None` for brillig-only ops with no ACIR analog (currently
771/// [`BlackBoxOp::ToRadix`]).
772fn black_box_op_to_acir_func(op: &BlackBoxOp) -> Option<BlackBoxFunc> {
773    match op {
774        BlackBoxOp::AES128Encrypt { .. } => Some(BlackBoxFunc::AES128Encrypt),
775        BlackBoxOp::Blake2s { .. } => Some(BlackBoxFunc::Blake2s),
776        BlackBoxOp::Blake3 { .. } => Some(BlackBoxFunc::Blake3),
777        BlackBoxOp::Keccakf1600 { .. } => Some(BlackBoxFunc::Keccakf1600),
778        BlackBoxOp::EcdsaSecp256k1 { .. } => Some(BlackBoxFunc::EcdsaSecp256k1),
779        BlackBoxOp::EcdsaSecp256r1 { .. } => Some(BlackBoxFunc::EcdsaSecp256r1),
780        BlackBoxOp::MultiScalarMul { .. } => Some(BlackBoxFunc::MultiScalarMul),
781        BlackBoxOp::EmbeddedCurveAdd { .. } => Some(BlackBoxFunc::EmbeddedCurveAdd),
782        BlackBoxOp::Poseidon2Permutation { .. } => Some(BlackBoxFunc::Poseidon2Permutation),
783        BlackBoxOp::Sha256Compression { .. } => Some(BlackBoxFunc::Sha256Compression),
784        BlackBoxOp::ToRadix { .. } => None,
785    }
786}

788/// Position of the visitor relative to a call's argument-passing region while
789/// walking opcodes in reverse.
790///
791/// `codegen_call` emits a fixed prologue/epilogue around an external `Call`. Going
792/// backwards through the bytecode, the boundary opcodes are:
793///
794/// * `Mov { destination: stack_pointer, source: relative(0) }` — restores the
795///   caller's stack pointer (encountered first when walking backwards;
796///   marks the end of the call region).
797/// * `Mov { destination: <next-frame-slot-0>, source: stack_pointer }` — saves
798///   the caller's stack pointer into the next frame (encountered second).
799/// * `Const { .. }` — sets the stack-size register (encountered last;
800///   marks the start of the call region).
801///
802/// Writes inside this region target the *callee*'s stack frame, so the local
803/// advisory logic ignores them. Reads inside the region still count, since they
804/// consume values prepared by the current function.
805#[derive(Debug, Clone, Copy, PartialEq, Eq)]
806enum CallRegion {
807    /// Not in a call's argument-passing region.
808    Outside,
809    /// Inside the call region — between the stack-pointer-restore Mov and the
810    /// stack-pointer-save Mov, walking backwards.
811    Inside,
812    /// One opcode away from leaving the call region. The next visited opcode is
813    /// expected to be the Const that holds the saved stack size; once seen the
814    /// visitor transitions back to [`CallRegion::Outside`].
815    AtStackSize,
816}
817
818impl CallRegion {
819    /// Whether the visitor is currently traversing the argument-passing region
820    /// of a call (or its boundary opcodes).
821    fn is_active(self) -> bool {
822        !matches!(self, CallRegion::Outside)
823    }
824}

939    /// Assert the canonical "predicate-gated trap" shape that brillig-gen emits
940    /// for runtime checks (bit-shift range check, arithmetic overflow check, …):
941    ///
942    /// * A `target` opcode produces a value (the operation being checked).
943    /// * A `predicate` opcode writes a boolean into a register that is
944    ///   *different* from the target's destination — proving the trap branch
945    ///   does not depend on the target's result register.
946    /// * A `JumpIf` reads the predicate register and skips over the trap. It
947    ///   appears *after* the predicate in opcode order.
948    /// * A `Call` (to the `ErrorWithString` procedure) or a `Trap` opcode
949    ///   provides the trap branch and appears *after* the gating `JumpIf`.
950    ///
951    /// Both shift and arithmetic-overflow checks share this shape; only the
952    /// concrete `target` / `predicate` opcodes differ (and whether the
953    /// predicate is computed before or after the target — that order is *not*
954    /// asserted here).
955    fn assert_predicate_gated_trap<F>(
956        bytecode: &[Opcode<F>],
957        target_kind: &str,
958        is_target: impl Fn(&Opcode<F>) -> bool,
959        is_predicate: impl Fn(&Opcode<F>) -> bool,
960    ) {
961        let target_dst = bytecode
962            .iter()
963            .find_map(|op| if is_target(op) { binary_destination(op) } else { None })
964            .unwrap_or_else(|| panic!("brillig should contain a {target_kind} opcode"));
965
966        let (predicate_idx, predicate_dst) = bytecode
967            .iter()
968            .enumerate()
969            .find_map(|(i, op)| {
970                if is_predicate(op) { binary_destination(op).map(|dst| (i, dst)) } else { None }
971            })
972            .unwrap_or_else(|| panic!("brillig should contain the predicate for {target_kind}"));
973
974        assert_ne!(
975            predicate_dst, target_dst,
976            "{target_kind} predicate register must differ from its target destination"
977        );
978
979        // The JumpIf reading the predicate's result must appear *after* the
980        // predicate; otherwise it would be gating on a stale value.
981        let jmpif_idx = bytecode
982            .iter()
983            .enumerate()
984            .find_map(|(i, op)| match op {
985                Opcode::JumpIf { condition, .. } if *condition == predicate_dst => Some(i),
986                _ => None,
987            })
988            .unwrap_or_else(|| panic!("expected a JumpIf gating the trap for {target_kind}"));
989        assert!(
990            jmpif_idx > predicate_idx,
991            "{target_kind} JumpIf at {jmpif_idx} must follow the predicate at {predicate_idx}"
992        );
993
994        // The trap branch (Call to the error procedure or a direct Trap) must
995        // sit between the JumpIf and the next opcode the JumpIf skips to.
996        let trap_idx = bytecode
997            .iter()
998            .enumerate()
999            .skip(jmpif_idx + 1)
1000            .find_map(|(i, op)| match op {
1001                Opcode::Call { .. } | Opcode::Trap { .. } => Some(i),
1002                _ => None,
1003            })
1004            .unwrap_or_else(|| {
1005                panic!("expected a Trap or Call (ErrorWithString) for {target_kind}")
1006            });
1007        assert!(
1008            trap_idx > jmpif_idx,
1009            "{target_kind} trap at {trap_idx} must follow the gating JumpIf at {jmpif_idx}"
1010        );
1011    }
