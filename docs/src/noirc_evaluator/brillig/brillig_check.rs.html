29pub(crate) enum OpcodeAdvisory {
30    /// A memory address is being written by the opcode that no other following opcode reads.
31    NeverRead { addr: MemoryAddress },
32    /// A memory address is being written by the opcode that will get overwritten before being read.
33    OverwrittenBeforeRead { addr: MemoryAddress, write_at: OpcodeLocation, read_at: OpcodeLocation },
34}
35
36pub(crate) type OpcodeAdvisories = HashMap<OpcodeLocation, Vec<OpcodeAdvisory>>;

38/// Go through opcodes and collect advisories, indicating opcodes which are potentially unused.
39///
40/// At the moment this relies on the CFG still available from the SSA via the [FunctionContext].
41/// Unfortunately this doesn't work for the artifacts created by `compile_procedures`,
42/// which use looping in the bytecode level. To handle them in isolation, we would need to
43/// reconstruct the CFG from unresolved jump labels, and then establish a post order from that.
44pub(crate) fn opcode_advisories<F: AcirField>(
45    function: &Function,
46    function_context: &FunctionContext,
47    brillig_context: &BrilligContext<F, Stack>,
48) -> OpcodeAdvisories {
49    // Find where each block start and ends in the bytecode.
50    let block_opcode_ranges = block_opcode_ranges(brillig_context);
51
52    // Accumulate stack addresses read. These are all in the same function context,
53    // so they should fit in one stack frame, which makes relative addresses comparable.
54    let registers = brillig_context.registers();
55    let addr_range = Range { start: registers.start(), end: registers.end() };
56
57    // First collect all reads per block.
58    let reads_in_blocks: HashMap<BasicBlockId, im::HashSet<MemoryAddress>> = function_context
59        .reverse_post_order()
60        .map(|block_id| {
61            let mut read_collector = ReadCollector::new(&addr_range);
62            let opcode_range = block_opcode_ranges[&block_id].clone();
63            for loc in opcode_range {
64                let opcode = &brillig_context.artifact().byte_code[loc];
65                read_collector.visit_opcode(opcode, loc);
66            }
67            (block_id, read_collector.into_reads())
68        })
69        .collect();
70
71    // Find each loop in the function. If a block is part of a loop and writes to an address,
72    // we need to consider all blocks that we can get to from the header as a potential descendant.
73    let blocks_in_loops: HashMap<BasicBlockId, _> = Loops::find_all(function, LoopOrder::OutsideIn)
74        .yet_to_unroll
75        .into_iter()
76        .map(|loop_| (loop_.header, loop_.blocks))
77        .collect();
78
79    // Collect all the reads that happen in the descendants of blocks.
80    let mut reads_in_descendants: HashMap<BasicBlockId, im::HashSet<MemoryAddress>> =
81        HashMap::default();
82
83    // Going in Post Order, so successors will have been processed already,
84    // except for back-edges: the header of the loop they return to comes later.
85    for block_id in function_context.post_order() {
86        // The reads are a union of addresses read:
87        // * directly by the successors
88        // * indirectly by the descendants of the successors
89        let mut acc = im::HashSet::new();
90        for successor_id in function.dfg[block_id].successors() {
91            acc = acc.union(reads_in_blocks[&successor_id].clone());
92            // If this is a back-edge block, the header will not have been done yet.
93            if let Some(reads) = reads_in_descendants.get(&successor_id) {
94                acc = acc.union(reads.clone());
95            }
96        }
97        // If the block is a header of a loop, then anything we can reach from the header,
98        // we can reach from any block in the loop too.
99        // For example say we have a CFG like this, b1 is the header of a loop [b1, b2, b3]
100        //      +-------+
101        //      v       |
102        // b0--b1--b2--b3
103        //      +----------b4
104        // An address can be written in b3 and read in b1 (e.g. the loop variable),
105        // or be written in b2 and read in b4 (e.g. a reference allocated in b0).
106        if let Some(blocks_in_loop) = blocks_in_loops.get(&block_id) {
107            for member_id in blocks_in_loop {
108                if *member_id == block_id {
109                    continue;
110                }
111                // Blocks in the loop are successors, they should be done already.
112                let member_reads = reads_in_descendants[member_id].clone();
113                reads_in_descendants.insert(*member_id, member_reads.union(acc.clone()));
114            }
115        }
116        // We are done with this node, can move on to its predecessors.
117        reads_in_descendants.insert(block_id, acc);
118    }
119
120    // Now that we have the reads in the blocks and their descendants, look at the writes.
121    let mut advisories = HashMap::new();
122    for block_id in function_context.reverse_post_order() {
123        let mut advisory_collector =
124            AdvisoryCollector::new(&addr_range, &reads_in_descendants[&block_id]);
125        let opcode_range = block_opcode_ranges[&block_id].clone();
126
127        // Going backwards, so reads at the end are recorded before earlier writes.
128        for loc in opcode_range.rev() {
129            let opcode = &brillig_context.artifact().byte_code[loc];
130            advisory_collector.visit_opcode(opcode, loc);
131        }
132
133        advisories.extend(advisory_collector.into_advisories());
134    }
135    advisories
136}

138/// Display the opcodes and their corresponding advisories.
139pub(crate) fn show_opcode_advisories<F: Display>(
140    advisories: &OpcodeAdvisories,
141    artifact: &BrilligArtifact<F>,
142) {
143    println!(
144        "\n// There are {} Brillig opcode advisories for function '{}'",
145        advisories.len(),
146        artifact.name
147    );
148
149    if advisories.is_empty() {
150        return;
151    }
152
153    let advisory = |opcode: &Opcode<F>, msg: &str| {
154        let mut a = format!("{opcode}");
155        if msg.is_empty() {
156            return a;
157        }
158        for _ in a.len()..40 {
159            a.push(' ');
160        }
161        a.push_str(" // ");
162        a.push_str(msg);
163        a
164    };
165
166    println!("fn {}", artifact.name);
167    let index_width = artifact.byte_code.len().to_string().len();
168    for (index, opcode) in artifact.byte_code.iter().enumerate() {
169        let msg = if let Some(ads) = advisories.get(&index) {
170            &ads.iter()
171                .map(|a| match a {
172                    OpcodeAdvisory::NeverRead { addr } => format!("{addr} is never read"),
173                    OpcodeAdvisory::OverwrittenBeforeRead { addr, write_at, read_at } => {
174                        format!("{addr} is overwritten at {write_at} before read at {read_at}")
175                    }
176                })
177                .collect::<Vec<_>>()
178                .join("; ")
179        } else {
180            ""
181        };
182        println!("{index:>index_width$}: {}", advisory(opcode, msg));
183    }
184}

186/// Go through the labels in the [BrilligContext] and collect the opcode locations
187/// where each block starts and ends.
188fn block_opcode_ranges<F, R: RegisterAllocator>(
189    brillig_context: &BrilligContext<F, R>,
190) -> HashMap<BasicBlockId, Range<OpcodeLocation>> {
191    let mut ranges = HashMap::new();
192
193    let mut labels = brillig_context.artifact().labels.iter().collect::<Vec<_>>();
194    labels.sort_by_key(|(_, location)| **location);
195
196    let mut last_start = None;
197    for (label, location) in labels {
198        match (&label.label_type, last_start) {
199            (LabelType::Function(_, Some(block_id)), None) => {
200                last_start = Some((*block_id, *location));
201            }
202            (LabelType::Function(_, Some(new_block_id)), Some((block_id, start))) => {
203                if *new_block_id != block_id {
204                    ranges.insert(block_id, Range { start, end: *location });
205                    last_start = Some((*new_block_id, *location));
206                }
207            }
208            (_, Some((block_id, start))) => {
209                ranges.insert(block_id, Range { start, end: *location });
210                last_start = None;
211            }
212            (_, None) => {}
213        }
214    }
215    if let Some((block_id, start)) = last_start {
216        ranges.insert(block_id, Range { start, end: brillig_context.artifact().byte_code.len() });
217    }
218
219    ranges
220}

222/// Collect all the relative addresses read in a range.
223struct ReadCollector<'a> {
224    addr_range: &'a Range<OpcodeLocation>,
225    reads: im::HashSet<MemoryAddress>,
226}
227
228impl<'a> ReadCollector<'a> {
229    fn new(addr_range: &'a Range<usize>) -> Self {
230        Self { addr_range, reads: im::HashSet::new() }
231    }
232
233    fn into_reads(self) -> im::HashSet<MemoryAddress> {
234        self.reads
235    }
236}
237
238impl OpcodeAddressVisitor for ReadCollector<'_> {
239    fn read(&mut self, addr: &MemoryAddress, _location: OpcodeLocation) {
240        if addr.is_relative() && self.addr_range.contains(&assert_usize(addr.to_u32())) {
241            self.reads.insert(*addr);
242        }
243    }
244
245    fn write(&mut self, _addr: &MemoryAddress, _location: OpcodeLocation) {
246        // Ignore writes.
247    }
248
249    fn should_visit_opcode<F>(&mut self, _opcode: &Opcode<F>) -> bool {
250        true
251    }
252}

254/// Collect advisories about the opcodes in a single block.
255///
256/// Assumes that we are visiting the opcodes going backwards.
257struct AdvisoryCollector<'a> {
258    addr_range: &'a Range<OpcodeLocation>,
259    /// Addresses read in any of the descendants of the block.
260    reads_in_descendants: &'a im::HashSet<MemoryAddress>,
261    /// Last location the address was read from in this block.
262    reads: HashMap<MemoryAddress, OpcodeLocation>,
263    /// Last location the address was written to in this block.
264    writes: HashMap<MemoryAddress, OpcodeLocation>,
265    /// Advisories collected in this block.
266    advisories: OpcodeAdvisories,
267    /// Indicate that we are in a region where call parameters are being passed.
268    /// * 2 means we are in a call region
269    /// * 1 means we are 1 step before the start of the call region
270    /// * 0 means we are not in a call region
271    in_call_region: u8,
272    /// Indicate that we are in the region where we are passing out return parameters.
273    in_return_region: bool,
274    /// Indicate whether the current opcode can fail, which means it needs to be kept
275    /// even if its value is unused.
276    is_fallible: bool,
277}
278
279impl<'a> AdvisoryCollector<'a> {
280    fn new(
281        addr_range: &'a Range<usize>,
282        reads_in_descendants: &'a im::HashSet<MemoryAddress>,
283    ) -> Self {
284        Self {
285            addr_range,
286            reads_in_descendants,
287            reads: HashMap::new(),
288            writes: HashMap::new(),
289            advisories: HashMap::new(),
290            in_call_region: 0,
291            in_return_region: false,
292            is_fallible: false,
293        }
294    }
295
296    fn into_advisories(self) -> OpcodeAdvisories {
297        self.advisories
298    }
299
300    fn addr_in_range(&self, addr: &MemoryAddress) -> bool {
301        addr.is_relative() && self.addr_range.contains(&assert_usize(addr.to_u32()))
302    }
303
304    fn add_advisory(&mut self, location: OpcodeLocation, advisory: OpcodeAdvisory) {
305        self.advisories.entry(location).or_default().push(advisory);
306    }
307}

309impl OpcodeAddressVisitor for AdvisoryCollector<'_> {
310    /// Maintain some contextual information about regions.
311    fn should_visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>) -> bool {
312        // The return instruction can be preceded by a number of moves from various parts of the stack
313        // to the beginning, depending on the number of return values. They destinations can be shuffled.
314        if matches!(opcode, Opcode::Return) {
315            self.in_return_region = true;
316        } else if !matches!(opcode, Opcode::Mov { .. } | Opcode::Load { .. }) {
317            self.in_return_region = false;
318        }
319
320        // Add here anything where an unused result can be ignored, as the operation can fail the circuit.
321        self.is_fallible = match opcode {
322            Opcode::BinaryFieldOp {
323                op: BinaryFieldOp::Div | BinaryFieldOp::IntegerDiv, ..
324            }
325            | Opcode::BinaryIntOp { op: BinaryIntOp::Div, .. } => true,
326            // I'm sure there are many others.
327            _ => false,
328        };
329
330        match opcode {
331            // Handle the conventions of `codegen_call`: we are passing call arguments by coping them
332            // to new stack registers, then we set the stack pointer @0, which changes the meaning of
333            // relative addresses. The parameters are going to be read by another function, not visible
334            // in the scope of a single function we are analyzing here.
335            Opcode::Mov { destination, source } => {
336                if *destination == ReservedRegisters::stack_pointer()
337                    && *source == MemoryAddress::relative(0)
338                {
339                    // This is the restore of the stack pointer, the end of a call.
340                    self.in_call_region = 2;
341                    false
342                } else if self.in_call_region == 2 && *source == ReservedRegisters::stack_pointer()
343                {
344                    // This is where we store the stack pointer in the first address of the next stack frame.
345                    // The next opcode is a const that contains the current stack size.
346                    self.in_call_region = 1;
347                    false
348                } else {
349                    // We can visit the Mov instructions that copy parameters and return values;
350                    // we may need to ignore the destination, but we can still register the reads.
351                    true
352                }
353            }
354            Opcode::Const { .. } if self.in_call_region == 1 => {
355                // This is the instruction where we set the stack size at the beginning of the call.
356                self.in_call_region = 0;
357                true
358            }
359            // A `constrain` of some arbitrary expression is usually broken up into an instruction with
360            // a boolean result, and then a constrain on the result being equal to 1. Constant allocation
361            // creates a constant for that 1, but the codegen later might not use it, but rather check
362            // the variable directly, since it's already a bool value.
363            Opcode::Const {
364                destination: _,
365                bit_size: BitSize::Integer(IntegerBitSize::U1),
366                value,
367            } => !value.is_one(),
368            _ => true,
369        }
370    }
371
372    /// Remember the opcode location where a memory address was last read.
373    fn read(&mut self, addr: &MemoryAddress, location: OpcodeLocation) {
374        if !self.addr_in_range(addr) {
375            return;
376        }
377        self.reads.insert(*addr, location);
378    }
379
380    /// Remember the opcode location where a memory address was last written.
381    ///
382    /// Insert an advisory if:
383    /// * the address is not read after this opcode
384    /// * the address is read after this, but before that there is another write
385    fn write(&mut self, addr: &MemoryAddress, location: OpcodeLocation) {
386        if !self.addr_in_range(addr) {
387            return;
388        }
389        if self.in_call_region != 0 || self.in_return_region {
390            // Ignore the write, it's a parameter meant for another function.
391            // The reads can be inspected, as they should consume data we have prepared.
392            return;
393        }
394
395        // Some opcodes can have side effects, in which case we need to run them even
396        // if we don't use their results.
397        let ignore_unused = self.is_fallible;
398
399        if !ignore_unused {
400            if let Some(read_at) = self.reads.get(addr) {
401                if let Some(write_at) = self.writes.get(addr) {
402                    if write_at < read_at {
403                        self.add_advisory(
404                            location,
405                            OpcodeAdvisory::OverwrittenBeforeRead {
406                                addr: *addr,
407                                write_at: *write_at,
408                                read_at: *read_at,
409                            },
410                        );
411                    }
412                }
413            } else if !self.reads_in_descendants.contains(addr) {
414                self.add_advisory(location, OpcodeAdvisory::NeverRead { addr: *addr });
415            }
416        }
417
418        self.writes.insert(*addr, location);
419    }
420}

422trait OpcodeAddressVisitor {
423    /// Decide if an opcode should be visited.
424    ///
425    /// Can mutate self to set flags that allows it to ignore upcoming opcodes as well.
426    fn should_visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>) -> bool;
427
428    /// Called with all the addresses read by the opcode.
429    fn read(&mut self, addr: &MemoryAddress, location: OpcodeLocation);
430
431    /// Called with all addresses written by the opcode.
432    fn write(&mut self, addr: &MemoryAddress, location: OpcodeLocation);
433
434    /// Call with each opcode to record reads and writes.
435    ///
436    /// The expected order of visit depends on the purpose of the visitor implementation.
437    fn visit_opcode<F: AcirField>(&mut self, opcode: &Opcode<F>, location: OpcodeLocation) {
438        if !self.should_visit_opcode(opcode) {
439            return;
440        }
441        match opcode {
442            Opcode::BinaryFieldOp { destination, lhs, rhs, .. }
443            | Opcode::BinaryIntOp { destination, lhs, rhs, .. } => {
444                self.read(lhs, location);
445                self.read(rhs, location);
446                self.write(destination, location);
447            }
448            Opcode::Cast { destination, source, .. }
449            | Opcode::Mov { destination, source }
450            | Opcode::Not { destination, source, .. } => {
451                self.read(source, location);
452                self.write(destination, location);
453            }
454            Opcode::Load { destination, source_pointer } => {
455                self.read(source_pointer, location);
456                self.write(destination, location);
457            }
458            Opcode::Store { destination_pointer, source, .. } => {
459                self.read(destination_pointer, location); // the write goes where the pointer points
460                self.read(source, location);
461            }
462            Opcode::JumpIf { condition, .. } => {
463                self.read(condition, location);
464            }
465            Opcode::CalldataCopy { destination_address, size_address, offset_address } => {
466                self.read(size_address, location);
467                self.read(offset_address, location);
468                self.write(destination_address, location);
469            }
470            Opcode::Const { destination, .. } => {
471                self.write(destination, location);
472            }
473            Opcode::IndirectConst { destination_pointer, .. } => {
474                self.read(destination_pointer, location);
475            }
476            Opcode::ForeignCall { destinations, inputs, .. } => {
477                for input in inputs {
478                    self.read_value_or_array(input, location);
479                }
480                for destination in destinations {
481                    match destination {
482                        ValueOrArray::HeapArray(_) | ValueOrArray::HeapVector(_) => {
483                            // Consider the destinations as something the VM will read;
484                            // these are pointers to addresses on the heap to write directly into,
485                            // we don't need to read their values afterwards.
486                            self.read_value_or_array(destination, location);
487                        }
488                        ValueOrArray::MemoryAddress(_) => {
489                            // Simple addresses should be used read later to make use of them.
490                        }
491                    }
492
493                    // We can also consider them being written to.
494                    self.write_value_or_array(destination, location);
495                }
496            }
497            Opcode::ConditionalMov { source_a, source_b, condition, .. } => {
498                self.read(condition, location);
499                self.read(source_a, location);
500                self.read(source_b, location);
501            }
502            Opcode::BlackBox(black_box_op) => self.visit_black_box_op(black_box_op, location),
503            Opcode::Trap { revert_data } => self.read_heap_vector(revert_data, location),
504            Opcode::Stop { return_data } => self.read_heap_vector(return_data, location),
505            Opcode::Jump { .. } | Opcode::Call { .. } | Opcode::Return => {}
506        }
507    }
508
509    fn read_heap_array(&mut self, array: &HeapArray, location: OpcodeLocation) {
510        self.read(&array.pointer, location);
511    }
512
513    fn read_heap_vector(&mut self, vector: &HeapVector, location: OpcodeLocation) {
514        self.read(&vector.pointer, location);
515        self.read(&vector.size, location);
516    }
517
518    fn read_value_or_array(&mut self, value: &ValueOrArray, location: OpcodeLocation) {
519        match value {
520            ValueOrArray::MemoryAddress(memory_address) => self.read(memory_address, location),
521            ValueOrArray::HeapArray(heap_array) => self.read_heap_array(heap_array, location),
522            ValueOrArray::HeapVector(heap_vector) => self.read_heap_vector(heap_vector, location),
523        }
524    }
525
526    fn write_heap_array(&mut self, array: &HeapArray, location: OpcodeLocation) {
527        self.write(&array.pointer, location);
528    }
529
530    fn write_heap_vector(&mut self, vector: &HeapVector, location: OpcodeLocation) {
531        self.write(&vector.pointer, location);
532        self.write(&vector.size, location);
533    }
534
535    fn write_value_or_array(&mut self, value: &ValueOrArray, location: OpcodeLocation) {
536        match value {
537            ValueOrArray::MemoryAddress(memory_address) => self.write(memory_address, location),
538            ValueOrArray::HeapArray(heap_array) => self.write_heap_array(heap_array, location),
539            ValueOrArray::HeapVector(heap_vector) => self.write_heap_vector(heap_vector, location),
540        }
541    }
542
543    fn visit_black_box_op(&mut self, op: &BlackBoxOp, location: OpcodeLocation) {
544        match op {
545            BlackBoxOp::AES128Encrypt { inputs, iv, key, outputs } => {
546                self.read_heap_array(inputs, location);
547                self.read_heap_array(iv, location);
548                self.read_heap_array(key, location);
549                self.write_heap_array(outputs, location);
550            }
551            BlackBoxOp::Blake2s { message, output } | BlackBoxOp::Blake3 { message, output } => {
552                self.read_heap_array(message, location);
553                self.write_heap_array(output, location);
554            }
555            BlackBoxOp::Keccakf1600 { input, output } => {
556                self.read_heap_array(input, location);
557                self.write_heap_array(output, location);
558            }
559            BlackBoxOp::EcdsaSecp256k1 {
560                hashed_msg,
561                public_key_x,
562                public_key_y,
563                signature,
564                result,
565            }
566            | BlackBoxOp::EcdsaSecp256r1 {
567                hashed_msg,
568                public_key_x,
569                public_key_y,
570                signature,
571                result,
572            } => {
573                self.read_heap_array(hashed_msg, location);
574                self.read_heap_array(public_key_x, location);
575                self.read_heap_array(public_key_y, location);
576                self.read_heap_array(signature, location);
577                self.write(result, location);
578            }
579            BlackBoxOp::MultiScalarMul { points, scalars, outputs } => {
580                self.read_heap_array(points, location);
581                self.read_heap_array(scalars, location);
582                self.write_heap_array(outputs, location);
583            }
584            BlackBoxOp::EmbeddedCurveAdd {
585                input1_x,
586                input1_y,
587                input1_infinite,
588                input2_x,
589                input2_y,
590                input2_infinite,
591                result,
592            } => {
593                self.read(input1_x, location);
594                self.read(input1_y, location);
595                self.read(input1_infinite, location);
596                self.read(input2_x, location);
597                self.read(input2_y, location);
598                self.read(input2_infinite, location);
599                self.write_heap_array(result, location);
600            }
601            BlackBoxOp::Poseidon2Permutation { message, output } => {
602                self.read_heap_array(message, location);
603                self.write_heap_array(output, location);
604            }
605            BlackBoxOp::Sha256Compression { input, hash_values, output } => {
606                self.read_heap_array(input, location);
607                self.read_heap_array(hash_values, location);
608                self.write_heap_array(output, location);
609            }
