1//! Call graph analysis
2//!
3//! This module provides a `CallGraph` structure that builds a directed call graph from SSA functions.
4//!
5//! It enables:
6//! - Construction of the call graph from SSA or explicit function dependencies
7//! - Detection of directly and mutually recursive functions using Kosaraju's algorithm for finding SCCs
8//!
9//! This utility is used by SSA passes such as inlining, which need to avoid recursive functions,
10//! and purity analysis which needs to unify the purities of all functions called within another function.
11use std::collections::{BTreeMap, BTreeSet};
12

14    algo::kosaraju_scc,
15    graph::{DiGraph, NodeIndex as PetGraphIndex},
16    visit::{Dfs, EdgeRef, Walker},
17};
18use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};

28/// Represents a function call graph built from the SSA
29/// Internally, this is a directed graph where each node is a [FunctionId] and each edge
30/// represents a call from one function to another.
31/// Each edge has an associated weight representing the number of times a descendant node
32/// was called by an ancestor node.
33pub(crate) struct CallGraph {
34    graph: DiGraph<FunctionId, usize>,
35    ids_to_indices: HashMap<FunctionId, PetGraphIndex>,
36    indices_to_ids: HashMap<PetGraphIndex, FunctionId>,
37}
38
39impl CallGraph {
40    /// Construct a [CallGraph] from the [Ssa]
41    /// The edges in this graph are unweighted. Thus, it is a pure dependency map.
42    pub(crate) fn from_ssa(ssa: &Ssa) -> Self {
43        let function_deps = ssa
44            .functions
45            .iter()
46            .map(|(id, func)| {
47                let called_functions = called_functions(func);
48                (*id, called_functions)
49            })
50            .collect();
51        Self::from_deps(function_deps)
52    }
53
54    /// Construct a [CallGraph] from the [Ssa] with its edges weighted.
55    /// An edges weight refers to the numbers of times a descendant node was called by its ancestor.
56    pub(crate) fn from_ssa_weighted(ssa: &Ssa) -> Self {
57        let dependencies = compute_callees(ssa);
58        Self::from_deps_weighted(dependencies)
59    }
60
61    fn from_deps_weighted(dependencies: BTreeMap<FunctionId, BTreeMap<FunctionId, usize>>) -> Self {
62        let mut graph = DiGraph::new();
63        let mut ids_to_indices = HashMap::default();
64        let mut indices_to_ids = HashMap::default();
65
66        for function in dependencies.keys() {
67            let index = graph.add_node(*function);
68            ids_to_indices.insert(*function, index);
69            indices_to_ids.insert(index, *function);
70        }
71
72        // Create edges from caller -> called
73        for (function, dependencies) in dependencies {
74            let function_index = ids_to_indices[&function];
75            for (callee, weight) in dependencies {
76                let dependency_index = ids_to_indices[&callee];
77                graph.add_edge(function_index, dependency_index, weight);
78            }
79        }
80
81        Self { graph, ids_to_indices, indices_to_ids }
82    }
83
84    /// Construct a [CallGraph] from an explicit dependency mapping of (caller -> callees)
85    /// The edges in this graph all have a weight of one. Thus, it is a pure dependency graph.
86    pub(crate) fn from_deps(dependencies: HashMap<FunctionId, BTreeSet<FunctionId>>) -> Self {
87        let mut graph = DiGraph::new();
88        let mut ids_to_indices = HashMap::default();
89        let mut indices_to_ids = HashMap::default();
90
91        for function in dependencies.keys() {
92            let index = graph.add_node(*function);
93            ids_to_indices.insert(*function, index);
94            indices_to_ids.insert(index, *function);
95        }
96
97        // Create edges from caller -> called
98        for (function, dependencies) in dependencies {
99            let function_index = ids_to_indices[&function];
100            for dependency in dependencies {
101                let dependency_index = ids_to_indices[&dependency];
102                graph.add_edge(function_index, dependency_index, 1);
103            }
104        }
105
106        Self { graph, ids_to_indices, indices_to_ids }
107    }

109    /// Returns the set of all recursive functions.
110    ///
111    /// A function is considered recursive if:
112    /// - It is self-recursive (calls itself), or
113    /// - It is part of a mutual recursion cycle with other functions.
114    pub(crate) fn get_recursive_functions(&self) -> HashSet<FunctionId> {
115        let (_, recursive_functions) = self.sccs();
116        recursive_functions
117    }

119    /// Returns both the call graph's strongly connected components (SCCs)
120    /// as well as a utility set of all recursive functions.
121    pub(crate) fn sccs(&self) -> (Vec<Vec<FunctionId>>, HashSet<FunctionId>) {
122        let mut sccs_ids = Vec::new();
123        let mut recursive_functions = HashSet::default();
124
125        let sccs = kosaraju_scc(&self.graph);
126        for scc in sccs {
127            let scc_ids: Vec<_> = scc.iter().map(|idx| self.indices_to_ids[idx]).collect();
128            if scc.len() > 1 {
129                // Mutual recursion
130                for idx in scc {
131                    recursive_functions.insert(self.indices_to_ids[&idx]);
132                }
133            } else {
134                // Check for self-recursion
135                let idx = scc[0];
136                if self.graph.neighbors(idx).any(|n| n == idx) {
137                    recursive_functions.insert(self.indices_to_ids[&idx]);
138                }
139            }
140            sccs_ids.push(scc_ids);
141        }
142        (sccs_ids, recursive_functions)
143    }

157    pub(crate) fn build_acyclic_subgraph(
158        &self,
159        recursive_functions: &HashSet<FunctionId>,
160    ) -> CallGraph {
161        let mut graph = DiGraph::new();
162        let mut ids_to_indices = HashMap::default();
163        let mut indices_to_ids = HashMap::default();
164
165        // Add all non-recursive nodes
166        for (&function, _) in self.ids_to_indices.iter() {
167            if recursive_functions.contains(&function) {
168                continue;
169            }
170            let index = graph.add_node(function);
171            ids_to_indices.insert(function, index);
172            indices_to_ids.insert(index, function);
173        }
174
175        // Create edges from caller -> called between non-recursive nodes
176        for (&func_id, &old_index) in self.ids_to_indices.iter() {
177            if recursive_functions.contains(&func_id) {
178                continue;
179            }
180            let new_src = ids_to_indices[&func_id];
181            for edge in self.graph.edges(old_index) {
182                let dst_index = edge.target();
183                let callee_id = self.indices_to_ids[&dst_index];
184                if recursive_functions.contains(&callee_id) {
185                    continue;
186                }
187                if let Some(&new_dst) = ids_to_indices.get(&callee_id) {
188                    graph.add_edge(new_src, new_dst, *edge.weight());
189                }
190            }
191        }
192
193        Self { graph, ids_to_indices, indices_to_ids }
194    }

196    /// Fetch for each function the set of functions called by it, and how many times it does so.
197    pub(crate) fn callees(&self) -> BTreeMap<FunctionId, BTreeMap<FunctionId, usize>> {
198        let mut callees: BTreeMap<FunctionId, BTreeMap<FunctionId, usize>> = BTreeMap::new();
199
200        for (caller_id, caller_index) in self.ids_to_indices.iter() {
201            // Ensure all entries exist, even if they have no callers
202            let entry = callees.entry(*caller_id).or_default();
203
204            for edge in self.graph.edges(*caller_index) {
205                let callee_index = edge.target();
206                let callee_id = self.indices_to_ids[&callee_index];
207                entry.insert(callee_id, *edge.weight());
208            }
209        }
210
211        callees
212    }

214    /// Fetch for each function the set of functions that call it, and how many times they do so.
215    pub(crate) fn callers(&self) -> BTreeMap<FunctionId, BTreeMap<FunctionId, usize>> {
216        let mut callers: BTreeMap<FunctionId, BTreeMap<FunctionId, usize>> = BTreeMap::new();
217
218        for (caller_id, caller_index) in self.ids_to_indices.iter() {
219            // Ensure all entries exist, even if they have no callers
220            callers.entry(*caller_id).or_default();
221
222            for edge in self.graph.edges(*caller_index) {
223                let callee_idx = edge.target();
224                let callee_id = self.indices_to_ids[&callee_idx];
225                callers.entry(callee_id).or_default().insert(*caller_id, *edge.weight());
226            }
227        }
228
229        callers
230    }

232    /// Compute the times each function is called from any other function.
233    pub(crate) fn times_called(&self) -> im::HashMap<FunctionId, usize> {
234        let mut counts = im::HashMap::default();
235
236        for edge in self.graph.edge_references() {
237            let callee_idx = edge.target();
238            let callee_id = self.indices_to_ids[&callee_idx];
239            *counts.entry(callee_id).or_default() += *edge.weight();
240        }
241
242        // Ensure all nodes appear even if they are never called
243        for function_id in self.ids_to_indices.keys() {
244            counts.entry(*function_id).or_insert(0);
245        }
246
247        counts
248    }

250    /// Returns all functions reachable from the provided root(s).
251    ///
252    /// This function uses DFS internally to find all nodes reachable from the provided root(s).
253    pub(crate) fn reachable_from(
254        &self,
255        roots: impl IntoIterator<Item = FunctionId>,
256    ) -> HashSet<FunctionId> {
257        let mut reachable = HashSet::default();
258
259        for root in roots {
260            // If the root does not exist, skip it.
261            let Some(&start_index) = self.ids_to_indices.get(&root) else {
262                continue;
263            };
264            // Use DFS to determine all reachable nodes from the root
265            let dfs = Dfs::new(&self.graph, start_index);
266            reachable.extend(dfs.iter(&self.graph).map(|index| self.indices_to_ids[&index]));
267        }
268
269        reachable
270    }

273/// Utility function to find out the direct calls of a function.
274///
275/// Returns the function IDs from all `Call` instructions without deduplication.
276pub(crate) fn called_functions_vec(func: &Function) -> Vec<FunctionId> {
277    let mut called_function_ids = Vec::new();
278    for block_id in func.reachable_blocks() {
279        for instruction_id in func.dfg[block_id].instructions() {
280            let Instruction::Call { func: called_value_id, .. } = &func.dfg[*instruction_id] else {
281                continue;
282            };
283
284            if let Value::Function(function_id) = func.dfg[*called_value_id] {
285                called_function_ids.push(function_id);
286            }
287        }
288    }
289
290    called_function_ids
291}
292
293/// Utility function to find out the deduplicated direct calls made from a function.
294pub(crate) fn called_functions(func: &Function) -> BTreeSet<FunctionId> {
295    called_functions_vec(func).into_iter().collect()
296}

298/// Compute for each function the set of functions called by it, and how many times it does so.
299fn compute_callees(ssa: &Ssa) -> BTreeMap<FunctionId, BTreeMap<FunctionId, usize>> {
300    ssa.functions
301        .iter()
302        .flat_map(|(caller_id, function)| {
303            let called_functions = called_functions_vec(function);
304            called_functions.into_iter().map(|callee_id| (*caller_id, callee_id))
305        })
306        .fold(
307            // Make sure an entry exists even for ones that don't call anything.
308            ssa.functions.keys().map(|id| (*id, BTreeMap::new())).collect(),
309            |mut acc, (caller_id, callee_id)| {
310                let callees = acc.entry(caller_id).or_default();
311                *callees.entry(callee_id).or_default() += 1;
312                acc
313            },
314        )
315}
