43/// The DataFlowGraph contains most of the actual data in a function including
44/// its blocks, instructions, and values. This struct is largely responsible for
45/// owning most data in a function and handing out Ids to this data that can be
46/// shared without worrying about ownership.
47#[serde_as]
48#[derive(Debug, Clone, Serialize, Deserialize, Default)]
49pub(crate) struct DataFlowGraph {
50    /// Runtime of the [function][super::function::Function] that owns this [DataFlowGraph].
51    /// This might change during the `runtime_separation` pass where
52    /// ACIR functions are cloned as Brillig functions.
53    runtime: RuntimeType,
54
55    /// All of the instructions in a function
56    instructions: DenseMap<Instruction>,
57
58    /// Stores the results for a particular instruction.
59    ///
60    /// An instruction may return multiple values
61    /// and for this, we will also use the cranelift strategy
62    /// to fetch them via indices.
63    ///
64    /// Currently, we need to define them in a better way
65    /// Call instructions require the func signature, but
66    /// other instructions may need some more reading on my part
67    #[serde_as(as = "HashMap<DisplayFromStr, _>")]
68    results: HashMap<InstructionId, smallvec::SmallVec<[ValueId; 1]>>,
69
70    /// Storage for all of the values defined in this
71    /// function.
72    values: DenseMap<Value>,
73
74    /// Each constant is unique, attempting to insert the same constant
75    /// twice will return the same ValueId.
76    #[serde(skip)]
77    constants: HashMap<(FieldElement, NumericType), ValueId>,
78
79    /// Contains each function that has been imported into the current function.
80    /// A unique `ValueId` for each function's [`Value::Function`] is stored so any given FunctionId
81    /// will always have the same ValueId within this function.
82    #[serde(skip)]
83    functions: HashMap<FunctionId, ValueId>,
84
85    /// Contains each intrinsic that has been imported into the current function.
86    /// This map is used to ensure that the ValueId for any given intrinsic is always
87    /// represented by only 1 ValueId within this function.
88    #[serde(skip)]
89    intrinsics: HashMap<Intrinsic, ValueId>,
90
91    /// Contains each foreign function that has been imported into the current function.
92    /// This map is used to ensure that the ValueId for any given foreign function is always
93    /// represented by only 1 ValueId within this function.
94    #[serde(skip)]
95    foreign_functions: HashMap<String, ValueId>,
96
97    /// All blocks in a function
98    blocks: DenseMap<BasicBlock>,
99
100    /// Source location of each instruction for debugging and issuing errors.
101    ///
102    /// The `CallStack` here corresponds to the entire callstack of locations. Initially this
103    /// only contains the actual location of the instruction. During inlining, a new location
104    /// will be pushed to each instruction for the location of the function call of the function
105    /// the instruction was originally located in. Once inlining is complete, the locations Vec
106    /// here should contain the entire callstack for each instruction.
107    ///
108    /// Instructions inserted by internal SSA passes that don't correspond to user code
109    /// may not have a corresponding location.
110    #[serde(skip)]
111    locations: HashMap<InstructionId, CallStackId>,
112
113    pub(crate) call_stack_data: CallStackHelper,
114
115    #[serde(skip)]
116    pub(crate) data_bus: DataBus,
117
118    pub(crate) globals: Arc<GlobalsGraph>,
119
120    #[serde(skip)]
121    pub(crate) function_purities: Arc<FunctionPurities>,
122
123    /// Indicate whether the Brillig array index offset optimizations have been performed.
124    pub(crate) brillig_arrays_offset: bool,
125}

196    /// Create a new block with the same parameter count and parameter
197    /// types from the given block.
198    /// This is a somewhat niche operation used in loop unrolling but is included
199    /// here as doing it outside the DataFlowGraph would require cloning the parameters.
200    pub(crate) fn make_block_with_parameters_from_block(
201        &mut self,
202        block: BasicBlockId,
203    ) -> BasicBlockId {
204        let new_block = self.make_block();
205        let parameters = self.blocks[block].parameters();
206
207        let parameters = vecmap(parameters.iter().enumerate(), |(position, param)| {
208            let typ = self.values[*param].get_type().into_owned();
209            self.values.insert(Value::Param { block: new_block, position, typ })
210        });
211
212        self.blocks[new_block].set_parameters(parameters);
213        new_block
214    }

243    /// Check if the function runtime would simply ignore this instruction.
244    pub(crate) fn is_handled_by_runtime(&self, instruction: &Instruction) -> bool {
245        match self.runtime() {
246            RuntimeType::Acir(_) => !matches!(
247                instruction,
248                Instruction::IncrementRc { .. } | Instruction::DecrementRc { .. }
249            ),
250            RuntimeType::Brillig(_) => {
251                !matches!(instruction, Instruction::EnableSideEffectsIf { .. })
252            }
253        }
254    }

269    pub(crate) fn insert_instruction_and_results_without_simplification(
270        &mut self,
271        instruction: Instruction,
272        block: BasicBlockId,
273        ctrl_typevars: Option<Vec<Type>>,
274        call_stack: CallStackId,
275    ) -> InsertInstructionResult<'_> {
276        if !self.is_handled_by_runtime(&instruction) {
277            // Panicking to raise attention. If we're not supposed to simplify it immediately,
278            // pushing the instruction would just cause a potential panic later on.
279            panic!("Attempted to insert instruction not handled by runtime: {instruction:?}");
280        }
281        let id = self.insert_instruction_without_simplification(
282            instruction,
283            block,
284            ctrl_typevars,
285            call_stack,
286        );
287
288        InsertInstructionResult::Results(id, self.instruction_results(id))
289    }

291    /// Simplifies a new instruction and inserts it at the end of the given block and returns its results.
292    /// If the instruction is not handled by the current runtime, `InstructionRemoved` is returned.
293    pub(crate) fn insert_instruction_and_results(
294        &mut self,
295        instruction: Instruction,
296        block: BasicBlockId,
297        ctrl_typevars: Option<Vec<Type>>,
298        call_stack: CallStackId,
299    ) -> InsertInstructionResult<'_> {
300        self.insert_instruction_and_results_if_simplified(
301            instruction,
302            block,
303            ctrl_typevars,
304            call_stack,
305            None,
306        )
307    }

309    /// Simplifies a potentially existing instruction and inserts it only if it changed.
310    pub(crate) fn insert_instruction_and_results_if_simplified(
311        &mut self,
312        instruction: Instruction,
313        block: BasicBlockId,
314        ctrl_typevars: Option<Vec<Type>>,
315        call_stack: CallStackId,
316        existing_id: Option<InstructionId>,
317    ) -> InsertInstructionResult<'_> {
318        if !self.is_handled_by_runtime(&instruction) {
319            // BUG: With panicking it fails to build the `token_contract`; see:
320            // https://github.com/AztecProtocol/aztec-packages/pull/11294#issuecomment-2624379102
321            // panic!("Attempted to insert instruction not handled by runtime: {instruction:?}");
322            return InsertInstructionResult::InstructionRemoved;
323        }
324
325        let simplify_result =
326            simplify(&instruction, self, block, ctrl_typevars.clone(), call_stack);
327
328        match simplify_result {
329            SimplifyResult::SimplifiedTo(simplification) => {
330                InsertInstructionResult::SimplifiedTo(simplification)
331            }
332            SimplifyResult::SimplifiedToMultiple(simplification) => {
333                InsertInstructionResult::SimplifiedToMultiple(simplification)
334            }
335            SimplifyResult::Remove => InsertInstructionResult::InstructionRemoved,
336            result @ (SimplifyResult::SimplifiedToInstruction(_)
337            | SimplifyResult::SimplifiedToInstructionMultiple(_)
338            | SimplifyResult::None) => {
339                let is_simplified = match &result {
340                    SimplifyResult::SimplifiedToInstruction(i) => {
341                        // `Binary` can simplify to itself instead of None.
342                        *i != instruction
343                    }
344                    SimplifyResult::SimplifiedToInstructionMultiple(is) => {
345                        // `Constrain` can simplify to a Multiple, with a single item of itself.
346                        is.len() != 1 || is[0] != instruction
347                    }
348                    SimplifyResult::None => false,
349                    _ => unreachable!("matched specific SimplifyResult types"),
350                };
351                if !is_simplified
352                    && let Some(id) = existing_id
353                    && self[id] == instruction
354                {
355                    // Just (re)insert into the block, no need to redefine.
356                    self.blocks[block].insert_instruction(id);
357                    let results = self.instruction_results(id);
358                    return InsertInstructionResult::Results(id, results);
359                }
360                let mut instructions = result.instructions().unwrap_or(vec![instruction]);
361                assert!(
362                    !instructions.is_empty(),
363                    "`SimplifyResult::SimplifiedToInstructionMultiple` must not return empty vector"
364                );
365
366                if instructions.len() > 1 {
367                    // There's currently no way to pass results from one instruction in `instructions` on to the next.
368                    // We then restrict this to only support multiple instructions if they're all `Instruction::Constrain`
369                    // as this instruction type does not have any results.
370                    assert!(
371                        instructions
372                            .iter()
373                            .all(|instruction| matches!(instruction, Instruction::Constrain(..))),
374                        "`SimplifyResult::SimplifiedToInstructionMultiple` only supports `Constrain` instructions"
375                    );
376                }
377
378                // Pull off the last instruction as we want to return its results.
379                let last_instruction = instructions.pop().expect("`instructions` can't be empty");
380                for instruction in instructions {
381                    self.insert_instruction_without_simplification(
382                        instruction,
383                        block,
384                        ctrl_typevars.clone(),
385                        call_stack,
386                    );
387                }
388                self.insert_instruction_and_results_without_simplification(
389                    last_instruction,
390                    block,
391                    ctrl_typevars,
392                    call_stack,
393                )
394            }
395        }
396    }

502    /// Attaches results to the instruction, clearing any previous results.
503    ///
504    /// This does not normally need to be called manually as it is called within
505    /// make_instruction automatically.
506    ///
507    /// Returns the results of the instruction
508    pub(crate) fn make_instruction_results(
509        &mut self,
510        instruction: InstructionId,
511        ctrl_typevars: Option<Vec<Type>>,
512    ) {
513        let mut results = smallvec::SmallVec::new();
514        let mut position = 0;
515        self.for_each_instruction_result_type(instruction, ctrl_typevars, |this, typ| {
516            let result = this.values.insert(Value::Instruction { typ, position, instruction });
517            position += 1;
518            results.push(result);
519        });
520
521        self.results.insert(instruction, results);
522    }

524    /// Return the result types of this instruction.
525    ///
526    /// In the case of Load, Call, and Intrinsic, the function's result
527    /// type may be unknown. In this case, the given ctrl_typevars are returned instead.
528    /// ctrl_typevars is taken in as an Option since it is common to omit them when getting
529    /// the type of an instruction that does not require them. Compared to passing an empty Vec,
530    /// Option has the benefit of panicking if it is accidentally used for a Call instruction,
531    /// rather than silently returning the empty Vec and continuing.
532    fn for_each_instruction_result_type(
533        &mut self,
534        instruction_id: InstructionId,
535        ctrl_typevars: Option<Vec<Type>>,
536        mut f: impl FnMut(&mut Self, Type),
537    ) {
538        let instruction = &self.instructions[instruction_id];
539        match instruction.result_type() {
540            InstructionResultType::Known(typ) => f(self, typ),
541            InstructionResultType::Operand(value) => f(self, self.type_of_value(value)),
542            InstructionResultType::None => (),
543            InstructionResultType::Unknown => {
544                for typ in ctrl_typevars.expect("Control typevars required but not given") {
545                    f(self, typ);
546                }
547            }
548        }
549    }

556    /// Returns the maximum possible number of bits that `value` can potentially be.
557    ///
558    /// Should `value` be a numeric constant then this function will return the exact number of bits required,
559    /// otherwise it will return the minimum number of bits based on type information.
560    pub(crate) fn get_value_max_num_bits(&self, value: ValueId) -> u32 {
561        match self[value] {
562            Value::Instruction { instruction, .. } => {
563                let value_bit_size = self.type_of_value(value).bit_size();
564                if let Instruction::Cast(original_value, _) = self[instruction] {
565                    let original_bit_size = self.get_value_max_num_bits(original_value);
566                    // We might have cast e.g. `u1` to `u8` to be able to do arithmetic,
567                    // in which case we want to recover the original smaller bit size;
568                    // OTOH if we cast down, then we don't need the higher original size.
569                    value_bit_size.min(original_bit_size)
570                } else {
571                    value_bit_size
572                }
573            }
574
575            Value::NumericConstant { constant, .. } => constant.num_bits(),
576            _ => self.type_of_value(value).bit_size(),
577        }
578    }

591    /// Returns N results, asserting that there are exactly N items.
592    pub(crate) fn instruction_result<const N: usize>(
593        &self,
594        instruction_id: InstructionId,
595    ) -> [ValueId; N] {
596        let results = self.instruction_results(instruction_id);
597        if results.len() != N {

657    /// Returns the item values in with this ValueId if it refers to an array constant, along with the type of the array item.
658    /// Otherwise, this returns None.
659    pub(crate) fn get_array_constant(&self, value: ValueId) -> Option<(im::Vector<ValueId>, Type)> {
660        match self.get_local_or_global_instruction(value)? {

666    /// If this value is an array, return the length of the array as indicated by its type.
667    /// Otherwise, return None.
668    pub(crate) fn try_get_array_length(&self, value: ValueId) -> Option<SemanticLength> {
669        match self.type_of_value(value) {
670            Type::Array(_, length) => Some(length),
671            _ => None,
672        }
673    }
674    pub(crate) fn try_get_vector_capacity(&self, value: ValueId) -> Option<SemanticLength> {
675        // For arrays we know the size statically
676        if let Some(length) = self.try_get_array_length(value) {
677            return Some(length);
678        }
679
680        // Check if the value was made by a MakeArray instruction, which can create vectors as well.
681        let (array, typ) = self.get_array_constant(value)?;
682        let elements_size = typ.element_size();
683
684        let length = if elements_size.0 == 0 {
685            SemanticLength(assert_u32(array.len()))
686        } else {
687            SemiFlattenedLength(assert_u32(array.len())) / elements_size
688        };
689
690        Some(length)
691    }

693    /// If this value points to an array of constant bytes, returns a string
694    /// consisting of those bytes if they form a valid UTF-8 string.
695    pub(crate) fn get_string(&self, value: ValueId) -> Option<String> {
696        let (value_ids, _typ) = self.get_array_constant(value)?;
697
698        let mut bytes = Vec::new();
699        for value_id in value_ids {
700            let field_value = self.get_numeric_constant(value_id)?;
701            let u64_value = field_value.try_to_u64()?;
702            if u64_value > 255 {
703                return None;
704            }
705            let byte = u64_value as u8;
706            bytes.push(byte);
707        }
708        String::from_utf8(bytes).ok()
709    }

711    /// A constant index less than the array length is safe
712    pub(crate) fn is_safe_index(&self, index: ValueId, array: ValueId) -> bool {
713        #[allow(clippy::match_like_matches_macro)]
714        match (self.type_of_value(array), self.get_numeric_constant(index)) {
715            (Type::Array(elements, len), Some(index)) => {
716                let elements_length = ElementTypesLength(assert_u32(elements.len()));
717                let semi_flattened_length = len * elements_length;
718                index.to_u128() < u128::from(semi_flattened_length.0)
719            }
720            _ => false,
721        }
722    }

733    /// Moves the entirety of the given block's contents into the destination block.
734    /// The source block afterward will be left in a valid but emptied state. The
735    /// destination block will also have its terminator overwritten with that of the
736    /// source block.
737    pub(crate) fn inline_block(&mut self, source: BasicBlockId, destination: BasicBlockId) {
738        let source = &mut self.blocks[source];
739        let mut instructions = source.take_instructions();
740        let terminator = source.take_terminator();
741
742        let destination = &mut self.blocks[destination];
743        destination.instructions_mut().append(&mut instructions);
744        destination.set_terminator(terminator);
745    }

776    /// True if the given [ValueId] refers to a constant value.
777    /// A [MakeArray][Instruction::MakeArray] instruction is considered constant if all its elements are constants.
778    pub(crate) fn is_constant(&self, argument: ValueId) -> bool {
779        match &self[argument] {
780            Value::Param { .. } => false,
781            Value::Instruction { .. } => {
782                let Some(instruction) = self.get_local_or_global_instruction(argument) else {
783                    return false;
784                };
785                match &instruction {
786                    Instruction::MakeArray { elements, .. } => {
787                        elements.iter().all(|element| self.is_constant(*element))
788                    }
789                    _ => false,
790                }
791            }
792            Value::Global(_) => {
793                unreachable!("The global value should have been indexed from the global space");
794            }
795            _ => true,
796        }
797    }

818    /// Uses value information to determine whether an instruction is from
819    /// this function's DFG or the global space's DFG, including the instruction ID.
820    pub(crate) fn get_local_or_global_instruction_with_id(
821        &self,
822        value: ValueId,
823    ) -> Option<(&Instruction, InstructionId)> {
824        match &self[value] {
825            Value::Instruction { instruction: instruction_id, .. } => {
826                let instruction = if self.is_global(value) {
827                    let instruction = &self.globals[*instruction_id];
828                    // We expect to only have MakeArray instructions in the global space
829                    assert!(matches!(instruction, Instruction::MakeArray { .. }));
830                    instruction
831                } else {
832                    &self[*instruction_id]
833                };
834                Some((instruction, *instruction_id))
835            }
836            _ => None,
837        }
838    }

848    /// Determine the appropriate [ArrayOffset] to use for indexing an array or vector.
849    pub(crate) fn array_offset(&self, array: ValueId, index: ValueId) -> ArrayOffset {
850        if !self.runtime.is_brillig()
851            || !self.brillig_arrays_offset
852            || self.get_numeric_constant(index).is_none()

855        }
856        match self.type_of_value(array) {

859            _ => ArrayOffset::None,
860        }

874    /// Computes the number of flattened values returned by the SSA terminator.
875    ///
876    /// Returns [RuntimeError::ReturnLimitExceeded] if it exceeds [MAX_ELEMENTS].
877    pub(crate) fn get_num_return_witnesses(&self, func: &Function) -> Result<usize, RuntimeError> {
878        if let Some(TerminatorInstruction::Return { return_values, call_stack }) =
879            func.return_instruction()
880        {
881            let num_return_values: usize = return_values
882                .iter()
883                .map(|result_id| self.type_of_value(*result_id).flattened_size().to_usize())
884                .sum();
885
886            if num_return_values > MAX_ELEMENTS {
887                let call_stack = func.dfg.call_stack_data.get_call_stack(*call_stack);
888                return Err(RuntimeError::ReturnLimitExceeded {
889                    num_witnesses: num_return_values,
890                    max_witnesses: MAX_ELEMENTS,
891                    call_stack,
892                });
893            }
894            return Ok(num_return_values);
895        }
896
897        Ok(0)
898    }

953// The result of calling DataFlowGraph::insert_instruction can
954// be a list of results or a single ValueId if the instruction was simplified
955// to an existing value.
956#[derive(Debug)]
957pub enum InsertInstructionResult<'dfg> {
958    /// Results is the standard case containing the instruction id and the results of that instruction.
959    Results(InstructionId, &'dfg [ValueId]),
960    SimplifiedTo(ValueId),
961    SimplifiedToMultiple(Vec<ValueId>),
962    InstructionRemoved,
963}
964
965impl<'dfg> InsertInstructionResult<'dfg> {
966    /// Retrieve the first (and expected to be the only) result.
967    pub(crate) fn first(&self) -> ValueId {
968        match self {
969            InsertInstructionResult::SimplifiedTo(value) => *value,
970            InsertInstructionResult::SimplifiedToMultiple(values) => values[0],
971            InsertInstructionResult::Results(_, results) => {
972                assert_eq!(results.len(), 1);
973                results[0]
974            }
975            InsertInstructionResult::InstructionRemoved => {
976                panic!("Instruction was removed, no results")
977            }
978        }
979    }
980
981    /// Return all the results contained in the internal results array.
982    /// This is used for instructions returning multiple results like function calls.
983    pub(crate) fn results(self) -> Cow<'dfg, [ValueId]> {
984        match self {
985            InsertInstructionResult::Results(_, results) => Cow::Borrowed(results),
986            InsertInstructionResult::SimplifiedTo(result) => Cow::Owned(vec![result]),
987            InsertInstructionResult::SimplifiedToMultiple(results) => Cow::Owned(results),
988            InsertInstructionResult::InstructionRemoved => Cow::Owned(vec![]),
989        }
990    }
991
992    /// Returns the amount of ValueIds contained
993    pub(crate) fn len(&self) -> usize {
994        match self {
995            InsertInstructionResult::SimplifiedTo(_) => 1,
996            InsertInstructionResult::SimplifiedToMultiple(results) => results.len(),
997            InsertInstructionResult::Results(_, results) => results.len(),
998            InsertInstructionResult::InstructionRemoved => 0,
999        }
1000    }
1001}
1002
1003impl std::ops::Index<usize> for InsertInstructionResult<'_> {
1004    type Output = ValueId;
1005
1006    fn index(&self, index: usize) -> &Self::Output {
1007        match self {
1008            InsertInstructionResult::Results(_, results) => &results[index],
1009            InsertInstructionResult::SimplifiedTo(result) => {
1010                assert_eq!(index, 0);
1011                result
1012            }
1013            InsertInstructionResult::SimplifiedToMultiple(results) => &results[index],
1014            InsertInstructionResult::InstructionRemoved => {
1015                panic!("Cannot index into InsertInstructionResult::InstructionRemoved")
1016            }
1017        }
1018    }
1019}
