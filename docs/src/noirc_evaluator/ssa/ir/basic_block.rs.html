1use super::{
2    instruction::{InstructionId, TerminatorInstruction},
3    map::Id,
4    value::ValueId,

6use noirc_errors::call_stack::CallStackId;
7use serde::{Deserialize, Serialize};

9/// A Basic block is a maximal collection of instructions
10/// such that there are only jumps at the end of block
11/// and one can only enter the block from the beginning.
12///
13/// This means that if one instruction is executed in a basic
14/// block, then all instructions are executed. ie single-entry single-exit.
15#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
16pub struct BasicBlock {
17    /// Parameters to the basic block.
18    parameters: Vec<ValueId>,
19
20    /// Instructions in the basic block.
21    instructions: Vec<InstructionId>,
22
23    /// The terminating instruction for the basic block.
24    ///
25    /// This will be a control flow instruction. This is only
26    /// None if the block is still being constructed.
27    terminator: Option<TerminatorInstruction>,
28}
29
30/// An identifier for a Basic Block.
31pub type BasicBlockId = Id<BasicBlock>;
32
33impl BasicBlock {
34    /// Create a new BasicBlock with the given parameters.
35    /// Parameters can also be added later via BasicBlock::add_parameter
36    pub(crate) fn new() -> Self {
37        Self { parameters: Vec::new(), instructions: Vec::new(), terminator: None }
38    }
39
40    /// Returns the parameters of this block
41    pub(crate) fn parameters(&self) -> &[ValueId] {
42        &self.parameters
43    }
44
45    /// Removes all the parameters of this block
46    pub(crate) fn take_parameters(&mut self) -> Vec<ValueId> {
47        std::mem::take(&mut self.parameters)
48    }
49
50    /// Adds a parameter to this BasicBlock.
51    /// Expects that the ValueId given should refer to a Value::Param
52    /// instance with its position equal to self.parameters.len().
53    pub(crate) fn add_parameter(&mut self, parameter: ValueId) {
54        self.parameters.push(parameter);
55    }
56
57    /// Replace this block's current parameters with that of the given Vec.
58    /// This does not perform any checks that any previous parameters were unused.
59    pub(crate) fn set_parameters(&mut self, parameters: Vec<ValueId>) {
60        self.parameters = parameters;
61    }
62
63    /// Insert an instruction at the end of this block
64    pub(crate) fn insert_instruction(&mut self, instruction: InstructionId) {
65        self.instructions.push(instruction);
66    }
67
68    /// Retrieve a reference to all instructions in this block.
69    pub(crate) fn instructions(&self) -> &[InstructionId] {
70        &self.instructions
71    }
72
73    /// Retrieve a mutable reference to all instructions in this block.
74    pub(crate) fn instructions_mut(&mut self) -> &mut Vec<InstructionId> {
75        &mut self.instructions
76    }
77
78    /// Take the instructions in this block, replacing it with an empty Vec
79    pub(crate) fn take_instructions(&mut self) -> Vec<InstructionId> {
80        std::mem::take(&mut self.instructions)
81    }
82
83    /// Sets the terminator instruction of this block.
84    ///
85    /// A properly-constructed block will always terminate with a TerminatorInstruction -
86    /// which either jumps to another block or returns from the current function. A block
87    /// will only have no terminator if it is still under construction.
88    pub(crate) fn set_terminator(&mut self, terminator: TerminatorInstruction) {
89        self.terminator = Some(terminator);
90    }
91
92    /// Returns the terminator of this block.
93    ///
94    /// Once this block has finished construction, this is expected to always be Some.
95    pub(crate) fn terminator(&self) -> Option<&TerminatorInstruction> {
96        self.terminator.as_ref()
97    }
98
99    /// Returns the terminator of this block, panics if there is None.
100    ///
101    /// Once this block has finished construction, this is expected to always be Some.
102    pub(crate) fn unwrap_terminator(&self) -> &TerminatorInstruction {
103        self.terminator().expect("Expected block to have terminator instruction")
104    }
105
106    /// Returns a mutable reference to the terminator of this block.
107    ///
108    /// Once this block has finished construction, this is expected to always be Some.
109    pub(crate) fn unwrap_terminator_mut(&mut self) -> &mut TerminatorInstruction {
110        self.terminator.as_mut().expect("Expected block to have terminator instruction")
111    }
112
113    /// Take ownership of this block's terminator, replacing it with an empty return terminator
114    /// so that no clone is needed.
115    ///
116    /// It is expected that this function is used as an optimization on blocks that are no longer
117    /// reachable or will have their terminator overwritten afterwards. Using this on a reachable
118    /// block without setting the terminator afterward will result in the empty return terminator
119    /// being kept, which is likely unwanted.
120    pub(crate) fn take_terminator(&mut self) -> TerminatorInstruction {
121        let terminator = self.terminator.as_mut().expect("Expected block to have a terminator");
122        std::mem::replace(
123            terminator,
124            TerminatorInstruction::Return {
125                return_values: Vec::new(),
126                call_stack: CallStackId::root(),
127            },
128        )
129    }
130
131    /// Return the jmp arguments, if any, of this block's TerminatorInstruction.
132    ///
133    /// If this block has no terminator, or a Return terminator this will be empty.
134    pub(crate) fn terminator_arguments(&self) -> &[ValueId] {
135        match &self.terminator {
136            Some(TerminatorInstruction::Jmp { arguments, .. }) => arguments,
137            _ => &[],
138        }
139    }
140
141    /// Iterate over all the successors of the currently block, as determined by
142    /// the blocks jumped to in the terminator instruction. If there is no terminator
143    /// instruction yet, this will iterate 0 times.
144    pub(crate) fn successors(
145        &self,
146    ) -> impl ExactSizeIterator<Item = BasicBlockId> + DoubleEndedIterator {
147        match &self.terminator {
148            Some(TerminatorInstruction::Jmp { destination, .. }) => vec![*destination].into_iter(),
149            Some(TerminatorInstruction::JmpIf { then_destination, else_destination, .. }) => {
150                vec![*then_destination, *else_destination].into_iter()
151            }
152            Some(TerminatorInstruction::Return { .. })
153            | Some(TerminatorInstruction::Unreachable { .. })
154            | None => vec![].into_iter(),
155        }
156    }
157}
