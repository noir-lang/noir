18/// Value is the most basic type allowed in the IR.
19/// Transition Note: A `Id<Value>` is similar to `NodeId` in our previous IR.
20#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
21pub enum Value {
22    /// This value was created due to an instruction
23    ///
24    /// * `instruction`: This is the instruction which defined it
25    /// * `typ`: This is the `Type` of the instruction
26    /// * `position`: Returns the position in the results vector that this `Value` is located.
27    ///
28    /// Example, if you add two numbers together, then the resulting
29    /// value would have position `0`, the typ would be the type
30    /// of the operands, and the instruction would map to an add instruction.
31    Instruction { instruction: InstructionId, position: usize, typ: Type },
32
33    /// This Value originates from a block parameter. Since function parameters
34    /// are also represented as block parameters, this includes function parameters as well.
35    ///
36    /// position -- the index of this Value in the block parameters list
37    Param { block: BasicBlockId, position: usize, typ: Type },
38
39    /// This Value originates from a numeric constant
40    NumericConstant { constant: FieldElement, typ: NumericType },
41
42    /// This Value refers to a function in the IR.
43    /// Functions always have the type Type::Function.
44    /// If the argument or return types are needed, users should retrieve
45    /// their types via the Call instruction's arguments or the Call instruction's
46    /// result types respectively.
47    Function(FunctionId),
48
49    /// An Intrinsic is a special kind of builtin function that may be handled internally
50    /// or optimized into a special form.
51    Intrinsic(Intrinsic),
52
53    /// This Value refers to an external function in the IR.
54    /// ForeignFunction's always have the type Type::Function and have similar semantics to Function,
55    /// other than generating different backend operations and being only accessible through Brillig.
56    ForeignFunction(String),
57
58    /// This Value indicates we have a reserved slot that needs to be accessed in a separate global context
59    Global(Type),
60}
61
62impl Value {
63    /// Retrieves the type of this Value
64    pub(crate) fn get_type(&self) -> Cow<Type> {
65        match self {
66            Value::Instruction { typ, .. } | Value::Param { typ, .. } => Cow::Borrowed(typ),
67            Value::NumericConstant { typ, .. } => Cow::Owned(Type::Numeric(*typ)),
68            Value::Function { .. } | Value::Intrinsic { .. } | Value::ForeignFunction { .. } => {
69                Cow::Owned(Type::Function)
70            }
71            Value::Global(typ) => Cow::Borrowed(typ),
72        }
73    }
74}

76/// Like `HashMap<ValueId, ValueId>` but handles:
77/// 1. recursion (if v0 -> v1 and v1 -> v2, then v0 -> v2)
78/// 2. self-mapping values (a value mapped to itself won't be inserted into the HashMap)
79#[derive(Default, Debug)]
80pub(crate) struct ValueMapping {
81    map: HashMap<ValueId, ValueId>,
82}
83
84impl ValueMapping {
85    pub(crate) fn insert(&mut self, from: ValueId, to: ValueId) {
86        if from == to {
87            return;
88        }
89
90        // If `to` is mapped to something, directly map `from` to that value
91        let to = self.get(to);
92        self.map.insert(from, to);
93    }
94
95    pub(crate) fn batch_insert(&mut self, from: &[ValueId], to: &[ValueId]) {
96        assert_eq!(from.len(), to.len(), "Lengths of arrays of values being mapped must match");
97        for (from_value, to_value) in from.iter().zip(to) {
98            self.insert(*from_value, *to_value);
99        }
100    }
101
102    pub(crate) fn get(&self, value: ValueId) -> ValueId {
103        if let Some(replacement) = self.map.get(&value) { self.get(*replacement) } else { value }
104    }
105
106    pub(crate) fn is_empty(&self) -> bool {
107        self.map.is_empty()
108    }
109
110    /// Returns true if all [`ValueId`]s are mapped to a [`ValueId`] of the same type.
111    ///
112    /// Mapping a [`ValueId`] to one of a different type implies a compilation error.
113    #[must_use]
114    #[cfg(debug_assertions)]
115    pub(crate) fn value_types_are_consistent(&self, dfg: &super::dfg::DataFlowGraph) -> bool {
116        self.map.iter().all(|(from, to)| dfg.type_of_value(*from) == dfg.type_of_value(*to))
117    }
118}
