1//! Union-Find (disjoint set) data structure for efficiently computing connected components.
2//! Uses path compression and union by rank for near-O(1) amortized operations.
3
4use rustc_hash::FxHashMap as HashMap;
5
6use std::hash::Hash;
7
8pub(crate) struct UnionFind<K: Copy + Eq + Hash> {
9    parent: HashMap<K, K>,
10    rank: HashMap<K, u32>,
11}
12
13impl<K: Copy + Eq + Hash> Default for UnionFind<K> {
14    fn default() -> Self {
15        Self::new()
16    }
17}
18
19impl<K: Copy + Eq + Hash> UnionFind<K> {
20    pub(crate) fn new() -> Self {
21        Self { parent: HashMap::default(), rank: HashMap::default() }
22    }
23
24    /// Ensure a value exists in the union-find.
25    pub(crate) fn make_set(&mut self, v: K) {
26        self.parent.entry(v).or_insert(v);
27    }
28
29    /// Find the root representative of the set containing `v`, with path compression.
30    pub(crate) fn find(&mut self, v: K) -> K {
31        // Lazily initialize the set for `v` as himself if it doesn't exist.
32        let p = *self.parent.entry(v).or_insert(v);
33        if p == v {
34            return v;
35        }
36        let root = self.find(p);
37        self.parent.insert(v, root);
38        root
39    }
40
41    /// Find the root representative of the set containing `v`, without path
42    /// compression.
43    fn find_immutable(&self, v: K) -> Option<K> {
44        let mut current = v;
45        loop {
46            let parent = *self.parent.get(&current)?;
47            if parent == current {
48                return Some(current);
49            }
50            current = parent;
51        }
52    }
53
54    /// Return a map from each class representative to the number of values
55    /// in that class.
56    pub(crate) fn class_sizes(&self) -> HashMap<K, u32> {
57        let mut sizes = HashMap::default();
58        for &v in self.parent.keys() {
59            if let Some(root) = self.find_immutable(v) {
60                *sizes.entry(root).or_insert(0) += 1;
61            }
62        }
63        sizes
64    }
65
66    /// Union the sets containing `a` and `b`. Uses union by rank.
67    pub(crate) fn union(&mut self, a: K, b: K) {
68        let ra = self.find(a);
69        let rb = self.find(b);
70        if ra == rb {
71            return;
72        }
73        let rank_a = *self.rank.entry(ra).or_insert(0);
74        let rank_b = *self.rank.entry(rb).or_insert(0);
75        if rank_a < rank_b {
76            self.parent.insert(ra, rb);
77        } else {
78            self.parent.insert(rb, ra);
79            if rank_a == rank_b {
80                *self.rank.entry(ra).or_insert(0) += 1;
81            }
82        }
83    }
84}
85
86/// Compute connected components from a directed edge map.
87///
88/// Returns:
89/// - A map from each value to its group ID.
90/// - A vec of all members in each group.
91pub(crate) fn connected_components<K: Copy + Eq + Hash>(
92    edges: &HashMap<K, K>,
93) -> (HashMap<K, usize>, Vec<Vec<K>>) {
94    let mut uf = UnionFind::new();
95    for (&k, &v) in edges {
96        uf.make_set(k);
97        uf.make_set(v);
98        uf.union(k, v);
99    }
100
101    let all_values: Vec<K> = uf.parent.keys().copied().collect();
102    let mut root_to_group: HashMap<K, usize> = HashMap::default();
103    let mut groups: HashMap<K, usize> = HashMap::default();
104    let mut group_members: Vec<Vec<K>> = Vec::new();
105
106    for value in all_values {
107        let root = uf.find(value);
108        let group_id = *root_to_group.entry(root).or_insert_with(|| {
109            group_members.push(Vec::new());
110            group_members.len() - 1
111        });
112        groups.insert(value, group_id);
113        group_members[group_id].push(value);
114    }
115
116    (groups, group_members)
117}
