1//! The post-order for a given function represented as a vector of basic block ids.
2//!
3//! This ordering is beneficial to the efficiency of various algorithms, such as those for dead
4//! code elimination and calculating dominance trees.
5
6use std::collections::{BTreeMap, BTreeSet, HashSet};

8use crate::ssa::ir::{basic_block::BasicBlockId, function::Function};

10use super::cfg::ControlFlowGraph;

12/// Depth-first traversal stack state marker for computing the cfg post-order.
13enum Visit {
14    First,
15    Last,
16}
17
18/// In the post-order, each block is visited after all of its successors.
19#[derive(Default, Clone)]
20pub(crate) struct PostOrder(Vec<BasicBlockId>);
21
22impl PostOrder {
23    pub(crate) fn as_slice(&self) -> &[BasicBlockId] {
24        self.0.as_slice()
25    }
26}

28impl PostOrder {
29    /// Allocate and compute a function's block post-order.
30    pub(crate) fn with_function(func: &Function) -> Self {
31        let cfg = ControlFlowGraph::with_function(func);
32        Self::with_cfg(&cfg)
33    }

35    /// Allocate and compute a function's block post-order.
36    pub(crate) fn with_cfg(cfg: &ControlFlowGraph) -> Self {
37        let roots = cfg.compute_entry_blocks();
38        PostOrder(Self::compute_post_order(cfg, roots))
39    }

41    /// Allocate and compute a function's block post-order, always rooted at the
42    /// function's entry block.
43    ///
44    /// Unlike [`Self::with_function`], this includes the entry block even when
45    /// it has incoming edges (e.g. malformed SSA with a back-edge to entry).
46    pub(crate) fn with_function_from_entry(func: &Function) -> Self {
47        let cfg = ControlFlowGraph::with_function(func);
48        PostOrder(Self::compute_post_order(&cfg, vec![func.entry_block()]))
49    }

51    /// Return blocks in post-order.
52    pub(crate) fn into_vec(self) -> Vec<BasicBlockId> {
53        self.0
54    }

56    /// Return blocks in reverse-post-order (RPO).
57    ///
58    /// In RPO, each block is visited only after all of its predecessors, except for the
59    /// back-edge predecessors of loop headers. Blocks are ordered by the maximum cost to reach
60    /// them from the entry block (the longest path once back-edges are removed), with ties
61    /// broken by block id. The cost of each loop's exit blocks is then raised above the loop
62    /// body so that every loop's blocks are contiguous and precede the blocks reached by
63    /// exiting the loop.
64    ///
65    /// Take this CFG for example:
66    /// ```text
67    ///      b0
68    ///      |
69    ///      b1<-+
70    ///     /  \ |
71    ///    b3   b2
72    /// ```
73    /// The RPO is `[b0, b1, b2, b3]`: the loop body `b2` is ordered before the loop exit `b3`.
74    pub(crate) fn into_vec_reverse(self) -> Vec<BasicBlockId> {
75        let mut blocks = self.into_vec();
76        blocks.reverse();
77        blocks
78    }

80    /// Computes the post-order of the CFG so that, in the reverse-post-order, each block is
81    /// visited only after all of its non-back-edge predecessors.
82    ///
83    /// A plain depth-first post-order already has this property for back-edge-free CFGs, but it
84    /// can order a loop exit ahead of the loop body (see
85    /// <https://github.com/noir-lang/noir/issues/9771>). To get a deterministic, well-behaved
86    /// order we rank each block by the *maximum cost to reach it* from a root: the length of the
87    /// longest path once back-edges are removed. Sorting by `(cost, block id)` is always a valid
88    /// topological order of the back-edge-free CFG, since any forward edge `u -> v` forces
89    /// `cost(v) >= cost(u) + 1`.
90    ///
91    /// Each block is visited at most once during the depth-first traversal; the cost of each
92    /// block is then computed in a single linear scan before a final sort.
93    ///
94    /// After the base costs are computed, the cost of each loop's exit blocks is raised above
95    /// the loop body (see [`Self::raise_loop_exit_costs`]) so that loops stay contiguous and
96    /// the blocks reached by exiting a loop are ordered after the entire loop body.
97    fn compute_post_order(cfg: &ControlFlowGraph, roots: Vec<BasicBlockId>) -> Vec<BasicBlockId> {
98        let post_order = Self::depth_first_post_order(cfg, roots);
99
100        // The per-block scratch data below is held in dense arrays indexed by `block.to_u32()`,
101        // sized to cover every reachable block. Block ids are dense, so this avoids the per-call
102        // hashing and allocation churn of a map on what is a very hot path.
103        let capacity =
104            post_order.iter().map(|block| block.to_u32() as usize + 1).max().unwrap_or(0);
105
106        // A block's index in the depth-first reverse-post-order. Crucially, every non-back edge
107        // points forwards in this order (smaller index -> larger index) while every back edge
108        // points backwards. So a predecessor `u` of `v` is a non-back-edge ("forward")
109        // predecessor exactly when `rpo_index[u] < rpo_index[v]`; no explicit edge classification
110        // is needed. Blocks unreachable from the roots keep the `u32::MAX` sentinel and are
111        // treated as having no index.
112        let mut rpo_index = vec![u32::MAX; capacity];
113        for (index, block) in post_order.iter().rev().enumerate() {
114            rpo_index[block.to_u32() as usize] = index as u32;
115        }
116
117        // The maximum cost to reach each block: `cost(v) = max(cost(u) + 1)` over the forward
118        // predecessors `u` of `v`, or 0 for a root. Iterating in reverse-post-order guarantees
119        // every forward predecessor has already been assigned its cost. Predecessors that are
120        // unreachable from the roots have no `rpo_index` and are ignored, as are back-edge
121        // predecessors (which have a larger-or-equal `rpo_index`). A block's `cost` entry is
122        // meaningful only once it has an `rpo_index`, which holds for every block read below.
123        //
124        // A predecessor with `rpo_index >= block_index` is a back-edge, so the loop here also
125        // records whether the CFG has any back-edge at all. When it does not (e.g. ACIR after loop
126        // unrolling), the loop-exit pass is skipped entirely: it would discover no loops, but only
127        // after another full scan over every block's predecessors.
128        let mut has_back_edge = false;
129        let mut cost = vec![0u32; capacity];
130        for &block in post_order.iter().rev() {
131            let block_index = rpo_index[block.to_u32() as usize];
132            let mut block_cost = 0;
133            for predecessor in cfg.predecessors(block) {
134                let Some(predecessor_index) = Self::index_of(&rpo_index, predecessor) else {
135                    continue;
136                };
137                if predecessor_index < block_index {
138                    block_cost = block_cost.max(cost[predecessor.to_u32() as usize] + 1);
139                } else {
140                    has_back_edge = true;
141                }
142            }
143            cost[block.to_u32() as usize] = block_cost;
144        }
145
146        if has_back_edge {
147            Self::raise_loop_exit_costs(cfg, &post_order, &rpo_index, &mut cost);
148        }
149
150        // Order by ascending cost (ties broken by block id) to get the reverse-post-order, then
151        // reverse it to recover the post-order. The cost key is computed once per block via
152        // `sort_by_cached_key`, so the sort itself does no per-comparison array indexing.
153        let mut reverse_post_order = post_order;
154        reverse_post_order.sort_by_cached_key(|block| (cost[block.to_u32() as usize], *block));
155        reverse_post_order.reverse();
156        reverse_post_order
157    }
158
159    /// The reverse-post-order index of `block`, or `None` when `block` is unreachable from the
160    /// roots (absent from the post-order and so still carrying the `u32::MAX` sentinel, or beyond
161    /// the dense array entirely).
162    fn index_of(rpo_index: &[u32], block: BasicBlockId) -> Option<u32> {
163        match rpo_index.get(block.to_u32() as usize).copied() {
164            None | Some(u32::MAX) => None,
165            Some(index) => Some(index),
166        }
167    }
168
169    /// Re-establish the invariant `cost[v] >= cost[u] + 1` for every forward edge `u -> v` with a
170    /// single relaxation pass in reverse-post-order, making the sorted order a valid
171    /// reverse-post-order (every block ends up after a predecessor) as the dominator tree requires.
172    ///
173    /// This is the irreducible-CFG fallback for [`Self::raise_loop_exit_costs`]: it is invoked only
174    /// when the loop-exit longest path could not be ordered topologically (a cycle in the augmented
175    /// graph). Because a forward edge always points to a strictly larger reverse-post-order index,
176    /// processing blocks in that order finalizes every forward predecessor before the block, so a
177    /// single pass suffices to repair the invariant.
178    fn enforce_forward_order(
179        cfg: &ControlFlowGraph,
180        post_order: &[BasicBlockId],
181        rpo_index: &[u32],
182        cost: &mut [u32],
183    ) {
184        for &block in post_order.iter().rev() {
185            let block_slot = block.to_u32() as usize;
186            let block_index = rpo_index[block_slot];
187            let mut new_cost = cost[block_slot];
188            for predecessor in cfg.predecessors(block) {
189                if Self::index_of(rpo_index, predecessor).is_some_and(|index| index < block_index) {
190                    new_cost = new_cost.max(cost[predecessor.to_u32() as usize] + 1);
191                }
192            }
193            cost[block_slot] = new_cost;
194        }
195    }

197    /// Raise the cost of every loop's exit blocks above the maximum cost within the loop. After
198    /// this, sorting by `(cost, block id)` keeps each natural loop's blocks contiguous and orders
199    /// the blocks reached by exiting a loop after the entire loop body. Without it a loop exit
200    /// branches off the loop header (a low-cost block) and so inherits a low cost, letting it sort
201    /// ahead of deeper loop-body blocks (see <https://github.com/noir-lang/noir/issues/9771>).
202    ///
203    /// An exit is a successor `e` of a loop block `b` that is not itself in the loop and is
204    /// reached by a *forward* edge (`rpo_index[e] > rpo_index[b]`); a successor reached by a
205    /// back-edge is an enclosing loop header, which legitimately precedes the loop.
206    ///
207    /// The final cost of every block is the longest path to it in the directed acyclic graph of:
208    ///
209    /// * the forward CFG edges (`u -> v` with `rpo_index[u] < rpo_index[v]`, weight 1), and
210    /// * for each loop `L`, a virtual loop node with an edge `b -> L` (weight 0) from every block
211    ///   `b` of `L` and an edge `L -> e` (weight 1) to every exit `e` of `L`. Routing the exit
212    ///   constraint through one node per loop keeps the edge count linear (`O(|L| + exits)` rather
213    ///   than `O(|L| * exits)`) and forces `cost[e] >= max_{b in L} cost[b] + 1`.
214    ///
215    /// This graph is acyclic — a loop's exits are strictly outside it — so a single longest-path
216    /// pass in topological order (Kahn's algorithm) computes every cost in linear time. The result
217    /// is the unique least fixpoint of the above constraints; the previous block costs (the longest
218    /// path over forward edges alone) are a valid lower bound and so seed the block nodes directly.
219    fn raise_loop_exit_costs(
220        cfg: &ControlFlowGraph,
221        post_order: &[BasicBlockId],
222        rpo_index: &[u32],
223        cost: &mut [u32],
224    ) {
225        let loops = Self::collect_loop_block_sets(cfg, post_order, rpo_index);
226        if loops.is_empty() {
227            return;
228        }
229
230        // Nodes `0..block_node_count` are blocks (indexed by `block.to_u32()`); nodes
231        // `block_node_count + loop_index` are the virtual loop nodes.
232        let block_node_count = rpo_index.len();
233        let node_count = block_node_count + loops.len();
234        let mut successors: Vec<Vec<(u32, u32)>> = vec![Vec::new(); node_count];
235        let mut in_degree = vec![0u32; node_count];
236        let mut add_edge = |from: usize, to: usize, weight: u32| {
237            successors[from].push((to as u32, weight));
238            in_degree[to] += 1;
239        };
240
241        // Loop-node edges: `b -> L` (weight 0) for each block, and `L -> e` (weight 1) for each
242        // distinct forward exit. `BTreeSet` keeps the exit edges deterministic.
243        for (loop_index, blocks) in loops.iter().enumerate() {
244            let loop_node = block_node_count + loop_index;
245            let mut exits = BTreeSet::new();
246            for &block in blocks {
247                let Some(block_index) = Self::index_of(rpo_index, block) else {
248                    continue;
249                };
250                add_edge(block.to_u32() as usize, loop_node, 0);
251                for successor in cfg.successors(block) {
252                    if blocks.contains(&successor) {
253                        continue;
254                    }
255                    if Self::index_of(rpo_index, successor).is_some_and(|index| index > block_index)
256                    {
257                        exits.insert(successor);
258                    }
259                }
260            }
261            for exit in exits {
262                add_edge(loop_node, exit.to_u32() as usize, 1);
263            }
264        }
265
266        // Forward CFG edges among reachable blocks.
267        for &block in post_order {
268            let block_index = rpo_index[block.to_u32() as usize];
269            for successor in cfg.successors(block) {
270                if Self::index_of(rpo_index, successor).is_some_and(|index| index > block_index) {
271                    add_edge(block.to_u32() as usize, successor.to_u32() as usize, 1);
272                }
273            }
274        }
275
276        // Longest path in topological order. Block costs live in `cost`; loop-node costs are held
277        // separately. Seed the queue with the sources (blocks with no forward predecessor, plus any
278        // loop node with no reachable member).
279        let mut loop_cost = vec![0u32; loops.len()];
280        let mut queue: Vec<usize> = Vec::new();
281        for &block in post_order {
282            if in_degree[block.to_u32() as usize] == 0 {
283                queue.push(block.to_u32() as usize);
284            }
285        }
286        for loop_index in 0..loops.len() {
287            if in_degree[block_node_count + loop_index] == 0 {
288                queue.push(block_node_count + loop_index);
289            }
290        }
291
292        let mut processed = 0usize;
293        while let Some(node) = queue.pop() {
294            processed += 1;
295            let node_cost = if node < block_node_count {
296                cost[node]
297            } else {
298                loop_cost[node - block_node_count]
299            };
300            for &(successor, weight) in &successors[node] {
301                let successor = successor as usize;
302                let candidate = node_cost + weight;
303                if successor < block_node_count {
304                    cost[successor] = cost[successor].max(candidate);
305                } else {
306                    let slot = successor - block_node_count;
307                    loop_cost[slot] = loop_cost[slot].max(candidate);
308                }
309                in_degree[successor] -= 1;
310                if in_degree[successor] == 0 {
311                    queue.push(successor);
312                }
313            }
314        }
315
316        // The augmented graph is acyclic for reducible CFGs, so Kahn's algorithm processes every
317        // node and the longest path above is exact. An irreducible CFG, however, can yield an
318        // over-approximate loop set whose exit feeds back into the loop, leaving a cycle that Kahn
319        // cannot order; the nodes on it keep partial costs that may violate `cost[v] >= cost[u] + 1`
320        // for a forward edge. That is purely a contiguity heuristic gone wrong, but it would make
321        // the sort produce an invalid reverse-post-order, so repair the forward-edge invariant. This
322        // runs only when a cycle was actually left unprocessed, so reducible CFGs pay nothing.
323        if processed < post_order.len() + loops.len() {
324            Self::enforce_forward_order(cfg, post_order, rpo_index, cost);
325        }
326    }

328    /// The natural loop block sets of the CFG, one per loop header.
329    ///
330    /// A retreating edge `p -> h` (`rpo_index[p] >= rpo_index[h]`) is a back-edge to header `h`.
331    /// The blocks of `h`'s natural loop are `h` together with every block that can reach a
332    /// back-edge source without passing through `h`. When a header has several back-edges (e.g. a
333    /// loop with `break`/`continue` arms that each jump back) the natural loop is the *union* of
334    /// the per-edge sets — otherwise a block internal to the larger arm would look like an exit of
335    /// the smaller one, and the two would fight over its position.
336    fn collect_loop_block_sets(
337        cfg: &ControlFlowGraph,
338        post_order: &[BasicBlockId],
339        rpo_index: &[u32],
340    ) -> Vec<BTreeSet<BasicBlockId>> {
341        // `BTreeMap`/`BTreeSet` keep header discovery and the merged sets deterministic.
342        let mut loops: BTreeMap<BasicBlockId, BTreeSet<BasicBlockId>> = BTreeMap::new();
343        for &header in post_order {
344            let header_index = rpo_index[header.to_u32() as usize];
345            for predecessor in cfg.predecessors(header) {
346                let Some(predecessor_index) = Self::index_of(rpo_index, predecessor) else {
347                    continue;
348                };
349                if predecessor_index >= header_index {
350                    let blocks = cfg.find_blocks_in_loop(header, predecessor);
351                    loops.entry(header).or_default().extend(blocks);
352                }
353            }
354        }
355        loops.into_values().collect()
356    }

358    // Computes the post-order of the CFG by doing a depth-first traversal of the given
359    // root blocks' previously unvisited children. Each block is sequenced according
360    // to when the traversal exits it.
361    fn depth_first_post_order(
362        cfg: &ControlFlowGraph,
363        roots: Vec<BasicBlockId>,
364    ) -> Vec<BasicBlockId> {
365        let mut stack = vec![];
366        let mut visited: HashSet<BasicBlockId> = HashSet::new();
367        let mut post_order: Vec<BasicBlockId> = Vec::new();
368
369        // Set root blocks
370        stack.extend(roots.into_iter().map(|root| (Visit::First, root)));
371
372        while let Some((visit, block_id)) = stack.pop() {
373            match visit {
374                Visit::First => {
375                    if !visited.contains(&block_id) {
376                        // This is the first time we pop the block, so we need to scan its
377                        // successors and then revisit it.
378                        visited.insert(block_id);
379                        stack.push((Visit::Last, block_id));
380                        // Stack successors for visiting. Because items are taken from the top of the
381                        // stack, we push the item that's due for a visit first to the top.
382                        for successor_id in cfg.successors(block_id).rev() {
383                            if !visited.contains(&successor_id) {
384                                // This not visited check would also be covered by the next
385                                // iteration, but checking here too saves an iteration per successor.
386                                stack.push((Visit::First, successor_id));
387                            }
388                        }
389                    }
390                }
391
392                Visit::Last => {
393                    // We've finished all this node's successors.
394                    post_order.push(block_id);
395                }
396            }
397        }
398        post_order
399    }

506        //        V
507        //       b3

519        builder.terminate_with_jmpif_no_args(zero, b2, b3);

522        builder.terminate_with_jmp(b1, Vec::new());
523

526        let ssa = builder.finish();
527
528        let func = ssa.main();
529        let post_order = PostOrder::with_function(func);
530
531        // Costs (longest path from b0, ignoring the b2->b1 back-edge): b0=0, b1=1, b2=2, b3=2.
532        // The reverse-post-order is [b0, b1, b2, b3] (b2 before b3 by block-id tie-break), so

538    /// loop exit branches off the header, so under a plain max-cost order it is hoisted ahead
539    /// of the deeper latch block. The order must keep the whole loop body before the exit.
540    #[test]
541    fn loop_with_separate_latch() {
542        // b0 -> b1 <--+
543        //       |\    |
544        //       | b2  |
545        //       |  \  |
546        //       |  b4-+   (b4 is the latch: jumps back to the header b1)
547        //       V
548        //      b3        (loop exit)
549        let src = "
550        acir(inline) fn main f0 {
551          b0(v0: u1):
552            jmp b1()
553          b1():
554            jmpif v0 then: b2(), else: b3()
555          b2():
556            jmp b4()
557          b3():
558            return
559          b4():
560            jmp b1()
561        }";
562
563        let ssa = Ssa::from_str(src).unwrap();
564        let func = ssa.main();
565        let post_order = PostOrder::with_function(func);
566
567        // Loop = {b1, b2, b4}, exit = b3. The exit b3 must come after the entire loop body in
568        // the forward order, so it must appear *first* in the post-order:
569        // forward order [b0, b1, b2, b4, b3] -> post-order [b3, b4, b2, b1, b0].
570        assert_eq!(post_order.0, [b(3), b(4), b(2), b(1), b(0)]);
571    }

599    #[test]
600    fn nested_loop() {
601        // b0 -> b1 -> b3
602        //      / ^
603        //     V   \
604        //     b2   |
605        //     |    |
606        //     V    |
607        //     b4->b6
608        //     | ^
609        //     V  \
610        //     b5->b8
611        //     |   ^
612        //     V  /
613        //     b7
614        //
615        // Expected topological sort:
616        // [b0, {loop blocks}, b3]  (see below for more complete explanation)
617        let src = "
618            acir(inline) fn main f0 {
619              b0(v0: u1):
620                jmp b1(v0)
621              b1(v1: u1):
622                jmpif v1 then: b2(), else: b3()
623              b2():
624                jmp b4(v1)
625              b3():
626                return
627              b4(v2: u1):
628                jmpif v2 then: b5(), else: b6()
629              b5():
630                jmpif v2 then: b7(), else: b8()
631              b6():
632                jmp b1(v1)
633              b7():
634                jmp b8()
635              b8():
636                jmp b4(v1)
637            }
638            ";
639
640        // We can break the CFG above into CFG's for each loop where the start
641        // node is where the loop was entered from. Break all incoming edges
642        // from this node, and remove outgoing edges that lead out of the loop:
643        //
644        //       b1
645        //      /
646        //     V
647        //     b2
648        //     |
649        //     V
650        //     b4->b6
651        //     | ^
652        //     V  \
653        //     b5->b8
654        //     |   ^
655        //     V  /
656        //     b7
657        //
658        // Here the expected topological sort is more clear:
659        // [ b1, b2, { loop blocks }, b6 ]
660        //
661        // We can do this again for the inner loop starting from `b4`:
662        //
663        //     b4
664        //     |
665        //     V
666        //     b5->b8
667        //     |   ^
668        //     V  /
669        //     b7
670        //
671        // Where the topological sort for this is unambiguously
672        // [b4, b5, b7, b8]
673        //
674        // Now we can slot this in to the unknown {loop blocks} from the first loop's
675        // topological sort to get:
676        //
677        // [b1, b2, b4, b5, b7, b8, b6]
678        //
679        // Blocks are first ordered by their maximum cost to reach them from the entry block,
680        // i.e. the longest path from b0 once the back-edges (b6->b1 and b8->b4) are removed:
681        //
682        //   b0=0, b1=1, b2=2, b3=2, b4=3, b5=4, b6=4, b7=5, b8=6
683        //
684        // Each loop's exit costs are then raised above the loop body. The inner loop
685        // {b4, b5, b7, b8} (max cost 6) pushes its exit b6 to 7; the outer loop
686        // {b1, b2, b4, b5, b6, b7, b8} (now max cost 7) pushes its exit b3 to 8:
687        //
688        //   b0=0, b1=1, b2=2, b4=3, b5=4, b7=5, b8=6, b6=7, b3=8
689        //
690        // Sorting by (cost, block-id) ascending gives the reverse-post-order
691        // [b0, b1, b2, b4, b5, b7, b8, b6, b3]. Every block still comes after all of its
692        // (non-back-edge) predecessors, and now each loop is contiguous with its exit ordered
693        // after the entire loop body. The expected post-order is the reverse:
694        let expected_post_order = [b(3), b(6), b(8), b(7), b(5), b(4), b(2), b(1), b(0)];
695
696        let ssa = Ssa::from_str(src).unwrap();
697
698        let func = ssa.main();
699        let post_order = PostOrder::with_function(func);
700        assert_eq!(post_order.0, expected_post_order);
701    }

703    /// Regression test for the dominator-tree panic on irreducible control flow. The loop-exit
704    /// cost raising builds an auxiliary graph that is acyclic only for reducible CFGs; an
705    /// irreducible loop can leave a cycle there, and a naive longest-path over it once ordered a
706    /// block ahead of its own predecessor — an invalid reverse-post-order that made
707    /// `DominatorTree::compute_immediate_dominator` panic with "block node must have one reachable
708    /// predecessor". The order must remain a valid reverse-post-order regardless.
709    #[test]
710    fn irreducible_cfg_produces_valid_reverse_post_order() {
711        use crate::ssa::ir::{cfg::ControlFlowGraph, dom::DominatorTree};
712
713        // This CFG was reduced from a program the AST fuzzer generated for the dominator-tree
714        // panic. It contains nested irreducible loops (e.g. `b6`'s loop is entered both at its
715        // header and, via `b14 -> b15`, in its middle), so the natural-loop approximation produces
716        // a loop whose exit re-enters it — the cycle that the loop-exit longest path cannot order
717        // topologically, and which made the unguarded version place a block before its predecessor.
718        // The condition `v0` is threaded through every block only to keep the SSA well-formed.
719        let src = "
720        acir(inline) fn main f0 {
721          b0(v0: u1):
722            jmpif v0 then: b1(v0), else: b2(v0)
723          b1(v1: u1):
724            jmp b12(v1)
725          b2(v2: u1):
726            jmp b3(v2)
727          b3(v3: u1):
728            jmpif v3 then: b5(v3), else: b4(v3)
729          b4(v4: u1):
730            jmp b3(v4)
731          b5(v5: u1):
732            jmpif v5 then: b7(v5), else: b8(v5)
733          b6(v6: u1):
734            jmpif v6 then: b9(v6), else: b10(v6)
735          b7(v7: u1):
736            jmp b6(v7)
737          b8(v8: u1):
738            jmp b11(v8)
739          b9(v9: u1):
740            jmp b12(v9)
741          b10(v10: u1):
742            jmp b15(v10)
743          b11(v11: u1):
744            jmpif v11 then: b14(v11), else: b13(v11)
745          b12(v12: u1):
746            return
747          b13(v13: u1):
748            jmp b11(v13)
749          b14(v14: u1):
750            jmp b15(v14)
751          b15(v15: u1):
752            jmp b6(v15)
753        }";
754
755        let ssa = Ssa::from_str(src).unwrap();
756        let func = ssa.main();
757
758        // Every reachable block other than the entry must appear after one of its predecessors in
759        // the reverse-post-order; otherwise the order is not a valid RPO.
760        let forward = PostOrder::with_function(func).into_vec_reverse();
761        let cfg = ControlFlowGraph::with_function(func);
762        for (position, &block) in forward.iter().enumerate() {
763            if position == 0 {
764                continue;
765            }
766            let has_earlier_predecessor = cfg.predecessors(block).any(|predecessor| {
767                forward.iter().take(position).any(|&earlier| earlier == predecessor)
768            });
769            assert!(
770                has_earlier_predecessor,
771                "block {block:?} at reverse-post-order position {position} has no earlier \
772                 predecessor; order = {forward:?}",
773            );
774        }
775
776        // Building the dominator tree exercises that invariant directly and must not panic.
777        let _dominator_tree = DominatorTree::with_function(func);
778    }

781/// Verifies, over real programs compiled through the full SSA pipeline, that the forward order
782/// places every block reached by exiting a loop after the entire loop body (see
783/// [`PostOrder::raise_loop_exit_costs`]).
784#[cfg(test)]
785mod loop_ordering_property_tests {
786    use std::collections::{BTreeMap, BTreeSet};
787
788    use rustc_hash::FxHashMap;
789
790    use super::PostOrder;
791    use crate::ssa::ir::{basic_block::BasicBlockId, cfg::ControlFlowGraph, function::Function};
792    use crate::ssa::opt::{LoopOrder, Loops};
793    use crate::ssa::ssa_gen::generate_ssa;
794    use crate::ssa::{Ssa, SsaBuilder, SsaEvaluatorOptions, primary_passes};
795    use noirc_frontend::test_utils::get_monomorphized;
796
797    /// A place where a loop's exit block is ordered at or before a block of that loop in the
798    /// function's forward (reverse-post) order. The fields are reported through `Debug` in the
799    /// failure message.
800    #[derive(Debug)]
801    #[allow(dead_code)]
802    struct Violation {
803        loop_header: BasicBlockId,
804        exit_block: BasicBlockId,
805        offending_loop_block: BasicBlockId,
806        exit_position: usize,
807        offending_position: usize,
808    }
809
810    /// For each natural loop in `func`, check that every block reached by exiting the loop comes
811    /// after the entire loop body in the forward order.
812    fn find_loop_exit_ordering_violations(func: &Function) -> Vec<Violation> {
813        let forward = PostOrder::with_function(func).into_vec_reverse();
814        let position: FxHashMap<BasicBlockId, usize> =
815            forward.iter().copied().enumerate().map(|(index, block)| (block, index)).collect();
816
817        let cfg = ControlFlowGraph::with_function(func);
818        // `find_all` does not mutate the function; it only reads the CFG to discover loops. It
819        // returns one entry per back-edge, so merge entries that share a header into the header's
820        // natural loop (the union of its back-edge block sets).
821        let loops = Loops::find_all(func, LoopOrder::InsideOut);
822        let mut natural_loops: BTreeMap<BasicBlockId, BTreeSet<BasicBlockId>> = BTreeMap::new();
823        for loop_ in &loops.yet_to_unroll {
824            natural_loops.entry(loop_.header).or_default().extend(loop_.blocks.iter().copied());
825        }
826
827        let mut violations = Vec::new();
828        for (header, blocks) in &natural_loops {
829            // The latest position among the loop's blocks, and which block holds it. Loop blocks
830            // unreachable from the entry have no position and are ignored.
831            let Some((max_position, offending_loop_block)) = blocks
832                .iter()
833                .filter_map(|block| position.get(block).map(|pos| (*pos, *block)))
834                .max_by_key(|(pos, _)| *pos)
835            else {
836                continue;
837            };
838
839            // The blocks reached by *forward*-exiting the loop: successors of a loop block that
840            // are not themselves in the loop and are not an enclosing header reached by a
841            // back-edge (such a header dominates the block and legitimately precedes the loop).
842            // `BTreeSet` keeps the reported order deterministic.
843            let mut exits: BTreeSet<BasicBlockId> = BTreeSet::new();
844            for &block in blocks {
845                for successor in cfg.successors(block) {
846                    if !blocks.contains(&successor) && !loops.dom.dominates(successor, block) {
847                        exits.insert(successor);
848                    }
849                }
850            }
851
852            for exit_block in exits {
853                let Some(&exit_position) = position.get(&exit_block) else {
854                    continue;
855                };
856                if exit_position <= max_position {
857                    violations.push(Violation {
858                        loop_header: *header,
859                        exit_block,
860                        offending_loop_block,
861                        exit_position,
862                        offending_position: max_position,
863                    });
864                }
865            }
866        }
867        violations
868    }

870    /// Compile a Noir source string to fully-optimized SSA, mirroring the primary pass pipeline.

873    /// so the frontend test helper, which does not link the standard library, can compile them.
874    /// Broad coverage over real `test_programs/` (which use the full stdlib) is exercised
875    /// separately by running the property check inside the `nargo_cli` execution suite.

881            ssa,

883            options.ssa_logging_hide_unchanged,
884            false,
885            None,
886        );
887        builder.run_passes(&primary_passes(&options)).expect("passes should run").finish()
888    }
