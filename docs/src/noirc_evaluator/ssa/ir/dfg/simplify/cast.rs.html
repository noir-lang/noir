12/// Try to simplify this cast instruction. If the instruction can be simplified to a known value,
13/// that value is returned. Otherwise None is returned.
14pub(super) fn simplify_cast(
15    value: ValueId,
16    dst_typ: NumericType,
17    dfg: &mut DataFlowGraph,
18) -> SimplifyResult {
19    use SimplifyResult::*;
20    assert!(
21        dfg.type_of_value(value).is_numeric(),
22        "Can only cast numeric types, got {:?}",
23        dfg.type_of_value(value)
24    );
25
26    if Type::Numeric(dst_typ) == dfg.type_of_value(value) {
27        return SimplifiedTo(value);
28    }
29
30    if let Value::Instruction { instruction, .. } = &dfg[value]
31        && let Instruction::Cast(original_value, _) = &dfg[*instruction]
32    {
33        let original_value = *original_value;
34        return match simplify_cast(original_value, dst_typ, dfg) {
35            None => SimplifiedToInstruction(Instruction::Cast(original_value, dst_typ)),
36            simpler => simpler,
37        };
38    }
39
40    if let Some(constant) = dfg.get_numeric_constant(value) {
41        let src_typ = dfg.type_of_value(value).unwrap_numeric();
42        match (src_typ, dst_typ) {
43            (
44                NumericType::Unsigned { .. } | NumericType::Signed { .. },
45                NumericType::NativeField,
46            ) => {
47                // Unsigned/Signed -> Field: redefine same constant as Field
48                SimplifiedTo(dfg.make_constant(constant, dst_typ))
49            }
50            (
51                NumericType::NativeField
52                | NumericType::Unsigned { .. }
53                | NumericType::Signed { .. },
54                NumericType::Unsigned { bit_size },
55            ) => {
56                // Field/Unsigned -> unsigned: truncate
57                let integer_modulus = BigUint::from(2u128).pow(bit_size);
58                let constant: BigUint = BigUint::from_bytes_be(&constant.to_be_bytes());
59                let truncated = constant % integer_modulus;
60                let truncated = FieldElement::from_be_bytes_reduce(&truncated.to_bytes_be());
61                SimplifiedTo(dfg.make_constant(truncated, dst_typ))
62            }
63            (
64                NumericType::NativeField | NumericType::Unsigned { .. },
65                NumericType::Signed { bit_size },
66            ) => {
67                // `IntegerConstant::from_numeric_constant` cannot handle i128
68                if bit_size == 128 {
69                    return None;
70                }
71
72                // Field/Unsigned -> signed
73                // We could only simplify to signed when we are below the maximum integer of the destination type.
74                // However, we expect that overflow constraints have been generated appropriately that enforce correctness.
75                let integer_constant = IntegerConstant::from_numeric_constant(constant, dst_typ);
76                if integer_constant.is_some() {
77                    SimplifiedTo(dfg.make_constant(constant, dst_typ))
78                } else {
79                    None
80                }
81            }
82            (NumericType::Signed { .. }, NumericType::Signed { bit_size }) => {
83                // `IntegerConstant::from_numeric_constant` cannot handle i128
84                if bit_size == 128 {
85                    return None;
86                }
87
88                // When going from signed to signed, we first need to interpret the constant as the signed source type,
89                // and then convert it to the signed destination type.
90                // For example, when going from `i8 -1` to `i16`, `i8 -1` is represented as the FieldElement 255,
91                // and it would be incorrect to use `IntegerConstant::from_numeric_constant(constant, dst_typ)` as
92                // that would give `i16 255` instead of the desired `i16 -1`.
93                if let Some(src_constant) =
94                    IntegerConstant::from_numeric_constant(constant, src_typ)
95                {
96                    let dst_constant = IntegerConstant::Signed {
97                        value: src_constant.apply(|v| v, |v| v as i128),
98                        bit_size,
99                    };
100                    let (dst_constant, dst_typ) = dst_constant.into_numeric_constant();
101                    SimplifiedTo(dfg.make_constant(dst_constant, dst_typ))
102                } else {
103                    None
104                }
105            }
106            (NumericType::NativeField, NumericType::NativeField) => {
107                unreachable!("This should be covered in previous if-branch")
108            }
109        }
110    } else {
111        None
112    }
113}

119    #[test]
120    fn unsigned_u8_to_i8_safe() {
121        let src = "
122        brillig(inline) predicate_pure fn main f0 {
123          b0():
124            v2 = cast u8 135 as i8
125            v3 = truncate v2 to 8 bits, max_bit_size: 9
126            v4 = cast v3 as u8
127            v6 = lt v4, u8 128
128            constrain v6 == u1 0
129            return v3
130        }
131        ";
132        let ssa = Ssa::from_str_simplifying(src).unwrap();
133
134        assert_ssa_snapshot!(ssa, @r"
135        brillig(inline) predicate_pure fn main f0 {
136          b0():
137            return i8 -121
138        }
139        ");
140    }

187    #[test]
188    fn simplifies_out_casting_there_and_back() {
189        // Casting from e.g. i8 to u64 used to go through sign extending to i64,
190        // which itself first cast to u8, then u64 to do some arithmetic, then
191        // the result was cast to i64 and back to u64.
192        let src = "
193        acir(inline) fn main f0 {
194          b0(v0: u64, v1: u64):
195            v2 = unchecked_add v0, v1
196            v3 = cast v2 as i64
197            v4 = cast v3 as u64
198            return v4
199        }
200        ";
201
202        let ssa = Ssa::from_str_simplifying(src).unwrap();
203
204        assert_ssa_snapshot!(ssa, @r"
205        acir(inline) fn main f0 {
206          b0(v0: u64, v1: u64):
207            v2 = unchecked_add v0, v1
208            v3 = cast v2 as i64
209            return v2
210        }
211        ");
212    }
