31/// Try to simplify this call instruction. If the instruction can be simplified to a known value,
32/// that value is returned. Otherwise None is returned.
33///
34/// The `block` parameter indicates the block any new instructions that are part of a call's
35/// simplification will be inserted into. For example, all vector intrinsics require updates
36/// to the vector length, which requires inserting a binary instruction. This update instruction
37/// must be inserted into the same block that the call itself is being simplified into.
38pub(super) fn simplify_call(
39    func: ValueId,
40    arguments: &[ValueId],
41    dfg: &mut DataFlowGraph,
42    block: BasicBlockId,
43    ctrl_typevars: Option<Vec<Type>>,
44    call_stack: CallStackId,
45) -> SimplifyResult {
46    let intrinsic = match &dfg[func] {
47        Value::Intrinsic(intrinsic) => *intrinsic,
48        _ => return SimplifyResult::None,
49    };
50
51    let return_type = ctrl_typevars.and_then(|return_types| return_types.first().cloned());
52
53    let constant_args: Option<Vec<_>> =
54        arguments.iter().map(|value_id| dfg.get_numeric_constant(*value_id)).collect();
55
56    let simplified_result = match intrinsic {
57        Intrinsic::ToBits(endian) => {
58            // TODO: simplify to a range constraint if `limb_count == 1`
59            if let (Some(constant_args), Some(return_type)) = (constant_args, return_type.clone()) {
60                let field = constant_args[0];
61                let Type::Array(_, limb_count) = return_type else {
62                    unreachable!("ICE: Intrinsic::ToRadix return type must be array")
63                };
64                simplify_constant_to_radix(endian, field, 2, limb_count.0, |values| {
65                    make_constant_array(
66                        dfg,
67                        values.into_iter(),
68                        NumericType::bool(),
69                        block,
70                        call_stack,
71                    )
72                })
73            } else {
74                SimplifyResult::None
75            }
76        }
77        Intrinsic::ToRadix(endian) => {
78            // TODO: simplify to a range constraint if `limb_count == 1`
79            if let (Some(constant_args), Some(return_type)) = (constant_args, return_type.clone()) {
80                let field = constant_args[0];
81                let radix = constant_args[1].to_u128() as u32;
82                let Type::Array(_, limb_count) = return_type else {
83                    unreachable!("ICE: Intrinsic::ToRadix return type must be array")
84                };
85                simplify_constant_to_radix(endian, field, radix, limb_count.0, |values| {
86                    make_constant_array(
87                        dfg,
88                        values.into_iter(),
89                        NumericType::Unsigned { bit_size: 8 },
90                        block,
91                        call_stack,
92                    )
93                })
94            } else {
95                SimplifyResult::None
96            }
97        }
98        Intrinsic::ArrayLen => {
99            let length = match *dfg.type_of_value(arguments[0]) {
100                Type::Array(_, length) => {
101                    dfg.make_constant(FieldElement::from(length.0), NumericType::length_type())
102                }
103                Type::Numeric(NumericType::Unsigned { bit_size: 32 }) => {
104                    assert!(matches!(*dfg.type_of_value(arguments[1]), Type::Vector(_)));
105                    arguments[0]
106                }
107                _ => panic!("First argument to ArrayLen must be an array or a vector length"),
108            };
109            SimplifyResult::SimplifiedTo(length)
110        }
111        // Strings are already arrays of bytes in SSA
112        Intrinsic::ArrayAsStrUnchecked => SimplifyResult::SimplifiedTo(arguments[0]),
113        Intrinsic::AsVector => {
114            if let Some(result) =
115                simplify_as_vector_for_zero_sized_vector(arguments, dfg, block, call_stack)
116            {
117                return result;
118            }
119
120            if let Some((array, array_type)) = dfg.get_array_constant(arguments[0]) {
121                // Compute the resulting vector length
122                let inner_element_types = array_type.element_types();
123                let vector_length_value = dfg.try_get_vector_capacity(arguments[0]).unwrap();
124                let vector_length =
125                    dfg.make_constant(vector_length_value.0.into(), NumericType::length_type());
126                let new_vector =
127                    make_array(dfg, array, Type::Vector(inner_element_types), block, call_stack);
128                SimplifyResult::SimplifiedToMultiple(vec![vector_length, new_vector])
129            } else {
130                SimplifyResult::None
131            }
132        }
133        Intrinsic::VectorPushBack => {
134            if let Some(result) = simplify_vector_push_back_or_front_for_zero_sized_vector(
135                arguments, dfg, block, call_stack,
136            ) {
137                return result;
138            }
139
140            let vector = dfg.get_array_constant(arguments[1]);
141            if let Some((mut vector, vector_type)) = vector {
142                if let Some(IntegerConstant::Unsigned { value: vector_len, .. }) =
143                    dfg.get_integer_constant(arguments[0])
144                {
145                    let elements_size = vector_type.element_size();
146                    let semi_flattened_vector_len =
147                        SemanticLength(vector_len as u32) * elements_size;
148
149                    // This simplification, which push back directly on the vector, only works if the real vector_len is the
150                    // the length of the vector (taking the elements size into account).
151                    if semi_flattened_vector_len == SemiFlattenedLength(vector.len() as u32) {
152                        // Old code before implementing multiple vector mergers
153                        for elem in &arguments[2..] {
154                            vector.push_back(*elem);
155                        }
156
157                        let new_vector_length =
158                            increment_vector_length(arguments[0], dfg, block, call_stack);
159
160                        let new_vector = make_array(dfg, vector, vector_type, block, call_stack);
161                        return SimplifyResult::SimplifiedToMultiple(vec![
162                            new_vector_length,
163                            new_vector,
164                        ]);
165                    }
166                }
167
168                simplify_vector_push_back(vector, vector_type, arguments, dfg, block, call_stack)
169            } else {
170                SimplifyResult::None
171            }
172        }
173        Intrinsic::VectorPushFront => {
174            if let Some(result) = simplify_vector_push_back_or_front_for_zero_sized_vector(
175                arguments, dfg, block, call_stack,
176            ) {
177                return result;
178            }
179
180            let vector = dfg.get_array_constant(arguments[1]);
181            if let Some((mut vector, vector_type)) = vector {
182                for elem in arguments[2..].iter().rev() {
183                    vector.push_front(*elem);
184                }
185
186                let new_vector_length =
187                    increment_vector_length(arguments[0], dfg, block, call_stack);
188
189                let new_vector = make_array(dfg, vector, vector_type, block, call_stack);
190                SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, new_vector])
191            } else {
192                SimplifyResult::None
193            }
194        }
195        Intrinsic::VectorPopBack => {
196            if let Some(result) = simplify_vector_pop_back_or_front_for_zero_sized_vector(
197                arguments, dfg, block, call_stack,
198            ) {
199                return result;
200            }
201
202            let length = dfg.get_numeric_constant(arguments[0]);
203            if length.is_none_or(|length| length.is_zero()) {
204                // If the length is zero then we're trying to pop the last element from an empty vector.
205                // Defer the error to acir_gen.
206                return SimplifyResult::None;
207            }
208
209            let vector = dfg.get_array_constant(arguments[1]);
210            if let Some((vector, typ)) = vector {
211                simplify_vector_pop_back(vector, typ, arguments, dfg, block, call_stack)
212            } else {
213                SimplifyResult::None
214            }
215        }
216        Intrinsic::VectorPopFront => {
217            if let Some(result) = simplify_vector_pop_back_or_front_for_zero_sized_vector(
218                arguments, dfg, block, call_stack,
219            ) {
220                return result;
221            }
222
223            let length = dfg.get_numeric_constant(arguments[0]);
224            if length.is_none_or(|length| length.is_zero()) {
225                // If the length is zero then we're trying to pop the first element from an empty vector.
226                // Defer the error to acir_gen.
227                return SimplifyResult::None;
228            }
229
230            let vector = dfg.get_array_constant(arguments[1]);
231            if let Some((mut vector, typ)) = vector {
232                let element_count = typ.element_size();
233
234                // We must pop multiple elements in the case of a vector of tuples
235                let mut results = vecmap(0..element_count.to_usize(), |_| {
236                    vector.pop_front().expect("There are no elements in this vector to be removed")
237                });
238
239                let new_vector_length =
240                    decrement_vector_length(arguments[0], dfg, block, call_stack);
241
242                results.push(new_vector_length);
243
244                let new_vector = make_array(dfg, vector, typ, block, call_stack);
245
246                // The vector is the last item returned for pop_front
247                results.push(new_vector);
248                SimplifyResult::SimplifiedToMultiple(results)
249            } else {
250                SimplifyResult::None
251            }
252        }
253        Intrinsic::VectorInsert => {
254            if let Some(result) =
255                simplify_vector_insert_for_zero_sized_vector(arguments, dfg, block, call_stack)
256            {
257                return result;
258            }
259
260            let vector = dfg.get_array_constant(arguments[1]);
261            let index = dfg.get_numeric_constant(arguments[2]);
262            if let (Some((mut vector, typ)), Some(index)) = (vector, index) {
263                let elements = &arguments[3..];
264                let mut index = index.to_u128() as usize * elements.len();
265
266                // Do not simplify the index is greater than the vector capacity
267                // or else we will panic inside of the im::Vector insert method
268                // Constraints should be generated during SSA gen to tell the user
269                // they are attempting to insert at too large of an index
270                if index > vector.len() {
271                    return SimplifyResult::None;
272                }
273
274                for elem in &arguments[3..] {
275                    vector.insert(index, *elem);
276                    index += 1;
277                }
278
279                let new_vector_length =
280                    increment_vector_length(arguments[0], dfg, block, call_stack);
281
282                let new_vector = make_array(dfg, vector, typ, block, call_stack);
283                SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, new_vector])
284            } else {
285                SimplifyResult::None
286            }
287        }
288        Intrinsic::VectorRemove => {
289            if let Some(result) =
290                simplify_vector_remove_for_zero_sized_vector(arguments, dfg, block, call_stack)
291            {
292                return result;
293            }
294
295            let length = dfg.get_numeric_constant(arguments[0]);
296            if length.is_none_or(|length| length.is_zero()) {
297                // If the length is zero then we're trying to remove an element from an empty vector.
298                // Defer the error to acir_gen.
299                return SimplifyResult::None;
300            }
301
302            let vector = dfg.get_array_constant(arguments[1]);
303            let index = dfg.get_numeric_constant(arguments[2]);
304            if let (Some((mut vector, typ)), Some(index)) = (vector, index) {
305                let element_count = typ.element_size().to_usize();
306                let mut results = Vec::with_capacity(element_count + 1);
307                let index = index.to_u128() as usize * element_count;
308
309                // Do not simplify if the index is not less than the vector capacity
310                // or else we will panic inside of the im::Vector remove method.
311                // Constraints should be generated during SSA gen to tell the user
312                // they are attempting to remove at too large of an index.
313                if index >= vector.len() {
314                    return SimplifyResult::None;
315                }
316
317                for _ in 0..element_count {
318                    results.push(vector.remove(index));
319                }
320
321                let new_vector = make_array(dfg, vector, typ, block, call_stack);
322                results.insert(0, new_vector);
323
324                let new_vector_length =
325                    decrement_vector_length(arguments[0], dfg, block, call_stack);
326
327                results.insert(0, new_vector_length);
328
329                SimplifyResult::SimplifiedToMultiple(results)
330            } else {
331                SimplifyResult::None
332            }
333        }
334        Intrinsic::StrAsBytes => {
335            // Strings are already represented as bytes internally
336            SimplifyResult::SimplifiedTo(arguments[0])
337        }
338        Intrinsic::AssertConstant => {
339            if arguments.iter().all(|argument| dfg.is_constant(*argument)) {
340                SimplifyResult::Remove
341            } else {
342                SimplifyResult::None
343            }
344        }
345        Intrinsic::StaticAssert => {
346            if arguments.len() < 2 {
347                panic!("ICE: static_assert called with wrong number of arguments")
348            }
349
350            // Arguments at positions `1..` form the message and they must all be constant.
351            if arguments.iter().skip(1).any(|argument| !dfg.is_constant(*argument)) {
352                return SimplifyResult::None;
353            }
354
355            if dfg.is_constant_true(arguments[0]) {
356                SimplifyResult::Remove
357            } else {
358                SimplifyResult::None
359            }
360        }
361        Intrinsic::ApplyRangeConstraint => {
362            let value = arguments[0];
363            let max_bit_size = dfg
364                .get_numeric_constant(arguments[1])
365                .expect("ApplyRangeConstraint bit-size must be a numeric constant")
366                .to_u128() as u32;
367            let max_potential_bits = dfg.get_value_max_num_bits(value);
368            if max_potential_bits < max_bit_size {
369                SimplifyResult::Remove
370            } else {
371                SimplifyResult::SimplifiedToInstruction(Instruction::RangeCheck {
372                    value,
373                    max_bit_size,
374                    assert_message: Some("call to assert_max_bit_size".to_owned()),
375                })
376            }
377        }
378        Intrinsic::Hint(Hint::BlackBox) => SimplifyResult::None,
379        Intrinsic::BlackBox(bb_func) => {
380            simplify_black_box_func(bb_func, arguments, dfg, block, call_stack)
381        }
382        Intrinsic::AsWitness => SimplifyResult::None,
383        Intrinsic::IsUnconstrained => {
384            let result = dfg.runtime().is_brillig().into();
385            SimplifyResult::SimplifiedTo(dfg.make_constant(result, NumericType::bool()))
386        }
387        Intrinsic::DerivePedersenGenerators => {
388            if let Some(Type::Array(_, len)) = return_type.clone() {
389                simplify_derive_generators(dfg, arguments, len.0, block, call_stack)
390            } else {
391                unreachable!("Derive Pedersen Generators must return an array");
392            }
393        }
394        Intrinsic::FieldLessThan => {
395            if let Some(constants) = constant_args {
396                let lhs = constants[0];
397                let rhs = constants[1];
398                let result = dfg.make_constant((lhs < rhs).into(), NumericType::bool());
399                SimplifyResult::SimplifiedTo(result)
400            } else {
401                SimplifyResult::None
402            }
403        }
404        Intrinsic::ArrayRefCount | Intrinsic::VectorRefCount => {
405            if dfg.runtime.is_acir() {
406                // In ACIR, ref counts are not tracked so we always simplify them to zero.
407                let zero = dfg.make_constant(FieldElement::zero(), NumericType::unsigned(32));
408                SimplifyResult::SimplifiedTo(zero)
409            } else {
410                SimplifyResult::None
411            }
412        }
413    };
414
415    if let (Some(expected_types), SimplifyResult::SimplifiedTo(result)) =
416        (return_type, &simplified_result)
417    {
418        assert_eq!(
419            *dfg.type_of_value(*result),
420            expected_types,
421            "Simplification should not alter return type"
422        );
423    }
424
425    simplified_result
426}

428fn simplify_as_vector_for_zero_sized_vector(
429    arguments: &[ValueId],
430    dfg: &mut DataFlowGraph,
431    block: BasicBlockId,
432    call_stack: CallStackId,
433) -> Option<SimplifyResult> {
434    let array_type = dfg.type_of_value(arguments[0]);
435    let Type::Array(element_types, length) = array_type.as_ref() else {
436        unreachable!("ICE: AsVector should only be called on arrays")
437    };
438    if !element_types.is_empty() {
439        return None;
440    }
441    // If this is a zero-sized arrays it can never have values in it, so we can always simplify
442    // it to (length, @[])
443    let element_types = element_types.clone();
444    let vector_length = dfg.make_constant(length.0.into(), NumericType::length_type());
445    let new_vector =
446        make_array(dfg, im::Vector::new(), Type::Vector(element_types), block, call_stack);
447    Some(SimplifyResult::SimplifiedToMultiple(vec![vector_length, new_vector]))
448}

450fn simplify_vector_push_back_or_front_for_zero_sized_vector(
451    arguments: &[ValueId],
452    dfg: &mut DataFlowGraph,
453    block: BasicBlockId,
454    call_stack: CallStackId,
455) -> Option<SimplifyResult> {
456    let vector_type = dfg.type_of_value(arguments[1]);
457    let Type::Vector(element_types) = vector_type.as_ref() else {
458        unreachable!("ICE: VectorInsert should only be called on vectors")
459    };
460    if !element_types.is_empty() {
461        return None;
462    }
463
464    // If this is a zero-sized vector then it can never have values in it, so we can just
465    // return an incremented length and return the same vector.
466    let length = arguments[0];
467    let new_vector_length = increment_vector_length(length, dfg, block, call_stack);
468    Some(SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, arguments[1]]))
469}

471fn simplify_vector_pop_back_or_front_for_zero_sized_vector(
472    arguments: &[ValueId],
473    dfg: &mut DataFlowGraph,
474    block: BasicBlockId,
475    call_stack: CallStackId,
476) -> Option<SimplifyResult> {
477    let vector_type = dfg.type_of_value(arguments[1]);
478    let Type::Vector(element_types) = vector_type.as_ref() else {
479        unreachable!("ICE: VectorInsert should only be called on vectors")
480    };
481    if !element_types.is_empty() {
482        return None;
483    }
484
485    // If this is a zero-sized vector then it can never have values in it.
486    // We do need to check that the length is not zero, though, but only in ACIR
487    // because in Brillig we already insert such check in FunctionContext::codegen_intrinsic_call_checks.
488    let length = arguments[0];
489
490    if dfg.runtime().is_acir() {
491        let zero_u32 = dfg.make_constant(FieldElement::zero(), NumericType::length_type());
492        let length_eq_zero = dfg
493            .insert_instruction_and_results(
494                Instruction::Binary(Binary { lhs: length, operator: BinaryOp::Eq, rhs: zero_u32 }),
495                block,
496                None,
497                call_stack,
498            )
499            .first();
500        let zero = dfg.make_constant(FieldElement::zero(), NumericType::bool());
501        let message = Some(ConstrainError::StaticString("Cannot pop from an empty vector".into()));
502        dfg.insert_instruction_and_results(
503            Instruction::Constrain(length_eq_zero, zero, message),
504            block,
505            None,
506            call_stack,
507        );
508    }
509
510    let new_vector_length = decrement_vector_length(length, dfg, block, call_stack);
511    Some(SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, arguments[1]]))
512}

558/// Returns a vector (represented by a tuple (len, vector)) of constants corresponding to the limbs of the radix decomposition.
559fn simplify_constant_to_radix(
560    endian: Endian,
561    field: FieldElement,
562    radix: u32,
563    limb_count: u32,
564    mut make_array: impl FnMut(Vec<FieldElement>) -> ValueId,
565) -> SimplifyResult {
566    match constant_to_radix(endian, field, radix, limb_count) {
567        Some(result) => SimplifyResult::SimplifiedTo(make_array(result)),
568        None => SimplifyResult::None,
569    }
570}

572pub(crate) fn constant_to_radix(
573    endian: Endian,
574    field: FieldElement,
575    radix: u32,
576    limb_count: u32,
577) -> Option<Vec<FieldElement>> {
578    let bit_size = u32::BITS - (radix - 1).leading_zeros();
579    let radix_big = BigUint::from(radix);
580    let radix_range = BigUint::from(2u128)..=BigUint::from(256u128);
581    if !radix_range.contains(&radix_big) || BigUint::from(2u128).pow(bit_size) != radix_big {
582        // NOTE: expect an error to be thrown later in
583        // acir::generated_acir::radix_le_decompose
584        return None;
585    }
586    let big_integer = BigUint::from_bytes_be(&field.to_be_bytes());
587
588    // Decompose the integer into its radix digits in little endian form.
589    let decomposed_integer = big_integer.to_radix_le(radix);
590    if limb_count < decomposed_integer.len() as u32 {
591        // `field` cannot be represented as `limb_count` bits.
592        // defer error to acir_gen.
593        None
594    } else {
595        let mut limbs = vecmap(0..limb_count, |i| match decomposed_integer.get(i as usize) {
596            Some(digit) => FieldElement::from_be_bytes_reduce(&[*digit]),
597            None => FieldElement::zero(),
598        });
599        if endian == Endian::Big {
600            limbs.reverse();
601        }
602        Some(limbs)
603    }
604}

606fn make_constant_array(
607    dfg: &mut DataFlowGraph,
608    results: impl Iterator<Item = FieldElement>,
609    typ: NumericType,
610    block: BasicBlockId,
611    call_stack: CallStackId,
612) -> ValueId {
613    let result_constants: im::Vector<_> =
614        results.map(|element| dfg.make_constant(element, typ)).collect();
615
616    let typ = Type::Array(
617        Arc::new(vec![Type::Numeric(typ)]),
618        SemanticLength(assert_u32(result_constants.len())),
619    );
620    make_array(dfg, result_constants, typ, block, call_stack)
621}

623fn make_array(
624    dfg: &mut DataFlowGraph,
625    elements: im::Vector<ValueId>,
626    typ: Type,
627    block: BasicBlockId,
628    call_stack: CallStackId,
629) -> ValueId {
630    let instruction = Instruction::MakeArray { elements, typ };
631    dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
632}

634/// Vectors have a tuple structure (vector length, vector contents) to enable logic
635/// that uses dynamic vector lengths (such as with merging vectors in the flattening pass).
636/// This method codegens an update to the vector length.
637///
638/// The binary operation performed on the vector length is always an addition or subtraction of `1`.
639/// This is because the vector length holds the user length (length as displayed by a `.len()` call),
640/// and not a flattened length used internally to represent arrays of tuples.
641fn update_vector_length(
642    vector_len: ValueId,
643    dfg: &mut DataFlowGraph,
644    operator: BinaryOp,
645    block: BasicBlockId,
646    call_stack: CallStackId,
647) -> ValueId {
648    let one = dfg.make_constant(FieldElement::one(), NumericType::length_type());
649    let instruction = Instruction::Binary(Binary { lhs: vector_len, operator, rhs: one });
650    dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
651}

653fn increment_vector_length(
654    vector_len: ValueId,
655    dfg: &mut DataFlowGraph,
656    block: BasicBlockId,
657    call_stack: CallStackId,
658) -> ValueId {
659    update_vector_length(vector_len, dfg, BinaryOp::Add { unchecked: false }, block, call_stack)
660}
661
662fn decrement_vector_length(
663    vector_len: ValueId,
664    dfg: &mut DataFlowGraph,
665    block: BasicBlockId,
666    call_stack: CallStackId,
667) -> ValueId {
668    // Simplifications only run if the length is a known non-zero constant, so the subtraction should never overflow.
669    update_vector_length(vector_len, dfg, BinaryOp::Sub { unchecked: true }, block, call_stack)
670}

672/// Simplify a vector push back when the length is not known to equal capacity, ie. we don't
673/// know whether we to push new items and grow the capacity of the vector, or overwrite the
674/// next padding item.
675///
676/// The strategy is to:
677/// 1. Create a new vector where the new item is pushed to the end, extending its capacity
678/// 2. Set the item at the original semantic length as well
679///
680/// There result is that the vector will physically always be extended by 1, with the pushed
681/// item appearing at the end, and potentially in the middle of the vector if we weren't at capacity.
682fn simplify_vector_push_back(
683    mut vector: im::Vector<ValueId>,
684    element_type: Type,
685    arguments: &[ValueId],
686    dfg: &mut DataFlowGraph,
687    block: BasicBlockId,
688    call_stack: CallStackId,
689) -> SimplifyResult {
690    // TODO(#2752): We need to handle the element_type size to appropriately handle vectors of complex types.
691    // This is reliant on dynamic indices of non-homogenous vectors also being implemented.
692    if element_type.element_size() != ElementTypesLength(1) {
693        return SimplifyResult::None;
694    }
695    assert_eq!(arguments.len(), 3, "should only push a single item");
696
697    let new_vector_length = increment_vector_length(arguments[0], dfg, block, call_stack);
698
699    vector.push_back(arguments[2]);
700
701    let extended_vector = make_array(dfg, vector, element_type, block, call_stack);
702
703    // Set the value at the semantic length: if the vector had extra capacity, this will set the first
704    // padding to the item we wanted to push. By doing this on the extended vector, we guarantee that
705    // there will be extra capacity. If we tried to do this on the original, we could get Index OOB if
706    // the capacity and the size were the same.
707    let set_last_vector_instr = Instruction::ArraySet {
708        array: extended_vector,
709        index: arguments[0],
710        value: arguments[2],
711        mutable: false,
712    };
713
714    let set_last_vector =
715        dfg.insert_instruction_and_results(set_last_vector_instr, block, None, call_stack).first();
716
717    SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, set_last_vector])
718}
719
720fn simplify_vector_pop_back(
721    mut vector: im::Vector<ValueId>,
722    vector_type: Type,
723    arguments: &[ValueId],
724    dfg: &mut DataFlowGraph,
725    block: BasicBlockId,
726    call_stack: CallStackId,
727) -> SimplifyResult {
728    let element_types = vector_type.element_types();
729    let element_count = element_types.len();
730    let mut results = VecDeque::with_capacity(element_count + 1);
731
732    let new_vector_length = decrement_vector_length(arguments[0], dfg, block, call_stack);
733
734    let element_size =
735        dfg.make_constant((element_count as u128).into(), NumericType::length_type());
736    // Compute the flattened length doing an unchecked mul
737    // (it shouldn't overflow because it would have overflowed before when the vector was created)
738    let flattened_len_instr =
739        Instruction::binary(BinaryOp::Mul { unchecked: true }, arguments[0], element_size);
740    let mut flattened_len =
741        dfg.insert_instruction_and_results(flattened_len_instr, block, None, call_stack).first();
742
743    // We must pop multiple elements in the case of a vector of tuples
744    // Iterating through element types in reverse here since we're popping from the end
745    for element_type in element_types.iter().rev() {
746        flattened_len = decrement_vector_length(flattened_len, dfg, block, call_stack);
747        let get_last_elem_instr =
748            Instruction::ArrayGet { array: arguments[1], index: flattened_len };
749
750        let element_type = Some(vec![element_type.clone()]);
751        let get_last_elem = dfg
752            .insert_instruction_and_results(get_last_elem_instr, block, element_type, call_stack)
753            .first();
754        results.push_front(get_last_elem);
755        vector.pop_back();
756    }
757
758    let new_vector = make_array(dfg, vector, vector_type, block, call_stack);
759    results.push_front(new_vector);
760
761    results.push_front(new_vector_length);
762    SimplifyResult::SimplifiedToMultiple(results.into())
763}

765/// Try to simplify this black box call. If the call can be simplified to a known value,
766/// that value is returned. Otherwise [`SimplifyResult::None`] is returned.
767fn simplify_black_box_func(
768    bb_func: BlackBoxFunc,
769    arguments: &[ValueId],
770    dfg: &mut DataFlowGraph,
771    block: BasicBlockId,
772    call_stack: CallStackId,
773) -> SimplifyResult {
774    cfg_if::cfg_if! {
775        if #[cfg(feature = "bn254")] {
776            let solver = bn254_blackbox_solver::Bn254BlackBoxSolver;
777        } else {
778            let solver = acvm::blackbox_solver::StubbedBlackBoxSolver;
779        }
780    };
781    match bb_func {
782        BlackBoxFunc::Blake2s => blackbox::simplify_hash(
783            dfg,
784            arguments,
785            acvm::blackbox_solver::blake2s,
786            block,
787            call_stack,
788        ),
789        BlackBoxFunc::Blake3 => blackbox::simplify_hash(
790            dfg,
791            arguments,
792            acvm::blackbox_solver::blake3,
793            block,
794            call_stack,
795        ),
796        BlackBoxFunc::Keccakf1600 => {
797            if let Some((array_input, _)) = dfg.get_array_constant(arguments[0]) {
798                if array_is_constant(dfg, &array_input) {
799                    let const_input: Vec<u64> = array_input
800                        .iter()
801                        .map(|id| {
802                            let field = dfg
803                                .get_numeric_constant(*id)
804                                .expect("value id from array should point at constant");
805                            field.to_u128() as u64
806                        })
807                        .collect();
808
809                    let state = acvm::blackbox_solver::keccakf1600(
810                        const_input.try_into().expect("Keccakf1600 input should have length of 25"),
811                    )
812                    .expect("Rust solvable black box function should not fail");
813                    let state_values = state.iter().map(|x| FieldElement::from(u128::from(*x)));
814                    let result_array = make_constant_array(
815                        dfg,
816                        state_values,
817                        NumericType::Unsigned { bit_size: 64 },
818                        block,
819                        call_stack,
820                    );
821                    SimplifyResult::SimplifiedTo(result_array)
822                } else {
823                    SimplifyResult::None
824                }
825            } else {
826                SimplifyResult::None
827            }
828        }
829        BlackBoxFunc::Poseidon2Permutation => {
830            blackbox::simplify_poseidon2_permutation(dfg, solver, arguments, block, call_stack)
831        }
832        BlackBoxFunc::EcdsaSecp256k1 => blackbox::simplify_signature(
833            dfg,
834            arguments,
835            acvm::blackbox_solver::ecdsa_secp256k1_verify,
836        ),
837        BlackBoxFunc::EcdsaSecp256r1 => blackbox::simplify_signature(
838            dfg,
839            arguments,
840            acvm::blackbox_solver::ecdsa_secp256r1_verify,
841        ),
842
843        BlackBoxFunc::MultiScalarMul => {
844            blackbox::simplify_msm(dfg, solver, arguments, block, call_stack)
845        }
846        BlackBoxFunc::EmbeddedCurveAdd => {
847            blackbox::simplify_ec_add(dfg, solver, arguments, block, call_stack)
848        }
849
850        BlackBoxFunc::RecursiveAggregation => SimplifyResult::None,
851
852        BlackBoxFunc::AND => {
853            unreachable!("ICE: `BlackBoxFunc::AND` calls should be transformed into a `BinaryOp`")
854        }
855        BlackBoxFunc::XOR => {
856            unreachable!("ICE: `BlackBoxFunc::XOR` calls should be transformed into a `BinaryOp`")
857        }
858        BlackBoxFunc::RANGE => {
859            unreachable!(
860                "ICE: `BlackBoxFunc::RANGE` calls should be transformed into a `Instruction::Cast`"
861            )
862        }
863        BlackBoxFunc::Sha256Compression => {
864            blackbox::simplify_sha256_compression(dfg, arguments, block, call_stack)
865        }
866        BlackBoxFunc::AES128Encrypt => SimplifyResult::None,
867    }
868}

886/// Replaces a call to `derive_pedersen_generators` with the results of the computation.
887///
888/// It only works if the arguments to the call are both constants, which means that the
889/// function which contains this call needs to be inlined into its caller, where the
890/// arguments are known. This is taken care of by the `#[no_predicates]` attribute,
891/// which forces inlining after flattening.
892///
893/// This intrinsic must not reach Brillig-gen.
894fn simplify_derive_generators(
895    dfg: &mut DataFlowGraph,
896    arguments: &[ValueId],
897    num_generators: u32,
898    block: BasicBlockId,
899    call_stack: CallStackId,
900) -> SimplifyResult {
901    if arguments.len() == 2 {
902        let domain_separator_string = dfg.get_array_constant(arguments[0]);
903        let starting_index = dfg.get_numeric_constant(arguments[1]);
904        if let (Some(domain_separator_string), Some(starting_index)) =
905            (domain_separator_string, starting_index)
906        {
907            let domain_separator_bytes = domain_separator_string
908                .0
909                .iter()
910                .map(|&x| dfg.get_numeric_constant(x).unwrap().to_u128() as u8)
911                .collect::<Vec<u8>>();
912            let generators = derive_generators(
913                &domain_separator_bytes,
914                num_generators,
915                starting_index.try_to_u32().expect("argument is declared as u32"),
916            );
917            let mut results = Vec::new();
918            for generator in generators {
919                let x_big: BigUint = generator.x.into();
920                let x = FieldElement::from_be_bytes_reduce(&x_big.to_bytes_be());
921                let y_big: BigUint = generator.y.into();
922                let y = FieldElement::from_be_bytes_reduce(&y_big.to_bytes_be());
923                results.push(dfg.make_constant(x, NumericType::NativeField));
924                results.push(dfg.make_constant(y, NumericType::NativeField));
925            }
926            let len = results.len() as u32;
927            assert!(
928                len.is_multiple_of(2),
929                "The number of results from derive_generators must be a multiple of 2"
930            );
931            let typ =
932                Type::Array(vec![Type::field(), Type::field()].into(), SemanticLength(len / 2));
933            let result = make_array(dfg, results.into(), typ, block, call_stack);
934            SimplifyResult::SimplifiedTo(result)
935        } else {
936            SimplifyResult::None
937        }
938    } else {
939        unreachable!("Unexpected number of arguments to derive_generators");
940    }
941}
