32/// Try to simplify this call instruction. If the instruction can be simplified to a known value,
33/// that value is returned. Otherwise None is returned.
34///
35/// The `block` parameter indicates the block any new instructions that are part of a call's
36/// simplification will be inserted into. For example, all vector intrinsics require updates
37/// to the vector length, which requires inserting a binary instruction. This update instruction
38/// must be inserted into the same block that the call itself is being simplified into.
39pub(super) fn simplify_call(
40    func: ValueId,
41    arguments: &[ValueId],
42    dfg: &mut DataFlowGraph,
43    block: BasicBlockId,
44    ctrl_typevars: Option<Vec<Type>>,
45    call_stack: CallStackId,
46) -> SimplifyResult {
47    let intrinsic = match &dfg[func] {
48        Value::Intrinsic(intrinsic) => *intrinsic,
49        _ => return SimplifyResult::None,
50    };
51
52    let return_type = ctrl_typevars.and_then(|return_types| return_types.first().cloned());
53
54    let constant_args: Option<Vec<_>> =
55        arguments.iter().map(|value_id| dfg.get_numeric_constant(*value_id)).collect();
56
57    let simplified_result = match intrinsic {
58        Intrinsic::ToBits(endian) => {
59            // TODO: simplify to a range constraint if `limb_count == 1`
60            if let (Some(constant_args), Some(return_type)) = (constant_args, return_type.clone()) {
61                let field = constant_args[0];
62                let limb_count = if let Type::Array(_, array_len) = return_type {
63                    array_len
64                } else {
65                    unreachable!("ICE: Intrinsic::ToRadix return type must be array")
66                };
67                simplify_constant_to_radix(endian, field, 2, limb_count.0, |values| {
68                    make_constant_array(
69                        dfg,
70                        values.into_iter(),
71                        NumericType::bool(),
72                        block,
73                        call_stack,
74                    )
75                })
76            } else {
77                SimplifyResult::None
78            }
79        }
80        Intrinsic::ToRadix(endian) => {
81            // TODO: simplify to a range constraint if `limb_count == 1`
82            if let (Some(constant_args), Some(return_type)) = (constant_args, return_type.clone()) {
83                let field = constant_args[0];
84                let radix = constant_args[1].to_u128() as u32;
85                let limb_count = if let Type::Array(_, array_len) = return_type {
86                    array_len
87                } else {
88                    unreachable!("ICE: Intrinsic::ToRadix return type must be array")
89                };
90                simplify_constant_to_radix(endian, field, radix, limb_count.0, |values| {
91                    make_constant_array(
92                        dfg,
93                        values.into_iter(),
94                        NumericType::Unsigned { bit_size: 8 },
95                        block,
96                        call_stack,
97                    )
98                })
99            } else {
100                SimplifyResult::None
101            }
102        }
103        Intrinsic::ArrayLen => {
104            let length = match dfg.type_of_value(arguments[0]) {
105                Type::Array(_, length) => {
106                    dfg.make_constant(FieldElement::from(length.0), NumericType::length_type())
107                }
108                Type::Numeric(NumericType::Unsigned { bit_size: 32 }) => {
109                    assert!(matches!(dfg.type_of_value(arguments[1]), Type::Vector(_)));
110                    arguments[0]
111                }
112                _ => panic!("First argument to ArrayLen must be an array or a vector length"),
113            };
114            SimplifyResult::SimplifiedTo(length)
115        }
116        // Strings are already arrays of bytes in SSA
117        Intrinsic::ArrayAsStrUnchecked => SimplifyResult::SimplifiedTo(arguments[0]),
118        Intrinsic::AsVector => {
119            let array = dfg.get_array_constant(arguments[0]);
120            if let Some((array, array_type)) = array {
121                // Compute the resulting vector length
122                let inner_element_types = array_type.element_types();
123                let vector_length_value = dfg.try_get_vector_capacity(arguments[0]).unwrap();
124                let vector_length =
125                    dfg.make_constant(vector_length_value.0.into(), NumericType::length_type());
126                let new_vector =
127                    make_array(dfg, array, Type::Vector(inner_element_types), block, call_stack);
128                SimplifyResult::SimplifiedToMultiple(vec![vector_length, new_vector])
129            } else {
130                SimplifyResult::None
131            }
132        }
133        Intrinsic::VectorPushBack => {
134            let vector = dfg.get_array_constant(arguments[1]);
135            if let Some((mut vector, element_type)) = vector {
136                // TODO(#2752): We need to handle the element_type size to appropriately handle vectors of complex types.
137                // This is reliant on dynamic indices of non-homogenous vectors also being implemented.
138                if element_type.element_size() != ElementTypesLength(1) {
139                    if let Some(IntegerConstant::Unsigned { value: vector_len, .. }) =
140                        dfg.get_integer_constant(arguments[0])
141                    {
142                        // This simplification, which push back directly on the vector, only works if the real vector_len is the
143                        // the length of the vector.
144                        if vector_len as usize == vector.len() {
145                            // Old code before implementing multiple vector mergers
146                            for elem in &arguments[2..] {
147                                vector.push_back(*elem);
148                            }
149
150                            let new_vector_length =
151                                increment_vector_length(arguments[0], dfg, block, call_stack);
152
153                            let new_vector =
154                                make_array(dfg, vector, element_type, block, call_stack);
155                            return SimplifyResult::SimplifiedToMultiple(vec![
156                                new_vector_length,
157                                new_vector,
158                            ]);
159                        }
160                    }
161                    return SimplifyResult::None;
162                }
163
164                simplify_vector_push_back(vector, element_type, arguments, dfg, block, call_stack)
165            } else {
166                SimplifyResult::None
167            }
168        }
169        Intrinsic::VectorPushFront => {
170            let vector = dfg.get_array_constant(arguments[1]);
171            if let Some((mut vector, element_type)) = vector {
172                for elem in arguments[2..].iter().rev() {
173                    vector.push_front(*elem);
174                }
175
176                let new_vector_length =
177                    increment_vector_length(arguments[0], dfg, block, call_stack);
178
179                let new_vector = make_array(dfg, vector, element_type, block, call_stack);
180                SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, new_vector])
181            } else {
182                SimplifyResult::None
183            }
184        }
185        Intrinsic::VectorPopBack => {
186            let length = dfg.get_numeric_constant(arguments[0]);
187            if length.is_none_or(|length| length.is_zero()) {
188                // If the length is zero then we're trying to pop the last element from an empty vector.
189                // Defer the error to acir_gen.
190                return SimplifyResult::None;
191            }
192
193            let vector = dfg.get_array_constant(arguments[1]);
194            if let Some((vector, typ)) = vector {
195                simplify_vector_pop_back(vector, typ, arguments, dfg, block, call_stack)
196            } else {
197                SimplifyResult::None
198            }
199        }
200        Intrinsic::VectorPopFront => {
201            let length = dfg.get_numeric_constant(arguments[0]);
202            if length.is_none_or(|length| length.is_zero()) {
203                // If the length is zero then we're trying to pop the first element from an empty vector.
204                // Defer the error to acir_gen.
205                return SimplifyResult::None;
206            }
207
208            let vector = dfg.get_array_constant(arguments[1]);
209            if let Some((mut vector, typ)) = vector {
210                let element_count = typ.element_size();
211
212                // We must pop multiple elements in the case of a vector of tuples
213                let mut results = vecmap(0..element_count.to_usize(), |_| {
214                    vector.pop_front().expect("There are no elements in this vector to be removed")
215                });
216
217                let new_vector_length =
218                    decrement_vector_length(arguments[0], dfg, block, call_stack);
219
220                results.push(new_vector_length);
221
222                let new_vector = make_array(dfg, vector, typ, block, call_stack);
223
224                // The vector is the last item returned for pop_front
225                results.push(new_vector);
226                SimplifyResult::SimplifiedToMultiple(results)
227            } else {
228                SimplifyResult::None
229            }
230        }
231        Intrinsic::VectorInsert => {
232            let vector = dfg.get_array_constant(arguments[1]);
233            let index = dfg.get_numeric_constant(arguments[2]);
234            if let (Some((mut vector, typ)), Some(index)) = (vector, index) {
235                let elements = &arguments[3..];
236                let mut index = index.to_u128() as usize * elements.len();
237
238                // Do not simplify the index is greater than the vector capacity
239                // or else we will panic inside of the im::Vector insert method
240                // Constraints should be generated during SSA gen to tell the user
241                // they are attempting to insert at too large of an index
242                if index > vector.len() {
243                    return SimplifyResult::None;
244                }
245
246                for elem in &arguments[3..] {
247                    vector.insert(index, *elem);
248                    index += 1;
249                }
250
251                let new_vector_length =
252                    increment_vector_length(arguments[0], dfg, block, call_stack);
253
254                let new_vector = make_array(dfg, vector, typ, block, call_stack);
255                SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, new_vector])
256            } else {
257                SimplifyResult::None
258            }
259        }
260        Intrinsic::VectorRemove => {
261            let length = dfg.get_numeric_constant(arguments[0]);
262            if length.is_none_or(|length| length.is_zero()) {
263                // If the length is zero then we're trying to remove an element from an empty vector.
264                // Defer the error to acir_gen.
265                return SimplifyResult::None;
266            }
267
268            let vector = dfg.get_array_constant(arguments[1]);
269            let index = dfg.get_numeric_constant(arguments[2]);
270            if let (Some((mut vector, typ)), Some(index)) = (vector, index) {
271                let element_count = typ.element_size().to_usize();
272                let mut results = Vec::with_capacity(element_count + 1);
273                let index = index.to_u128() as usize * element_count;
274
275                // Do not simplify if the index is not less than the vector capacity
276                // or else we will panic inside of the im::Vector remove method.
277                // Constraints should be generated during SSA gen to tell the user
278                // they are attempting to remove at too large of an index.
279                if index >= vector.len() {
280                    return SimplifyResult::None;
281                }
282
283                for _ in 0..element_count {
284                    results.push(vector.remove(index));
285                }
286
287                let new_vector = make_array(dfg, vector, typ, block, call_stack);
288                results.insert(0, new_vector);
289
290                let new_vector_length =
291                    decrement_vector_length(arguments[0], dfg, block, call_stack);
292
293                results.insert(0, new_vector_length);
294
295                SimplifyResult::SimplifiedToMultiple(results)
296            } else {
297                SimplifyResult::None
298            }
299        }
300        Intrinsic::StrAsBytes => {
301            // Strings are already represented as bytes internally
302            SimplifyResult::SimplifiedTo(arguments[0])
303        }
304        Intrinsic::AssertConstant => {
305            if arguments.iter().all(|argument| dfg.is_constant(*argument)) {
306                SimplifyResult::Remove
307            } else {
308                SimplifyResult::None
309            }
310        }
311        Intrinsic::StaticAssert => {
312            if arguments.len() < 2 {
313                panic!("ICE: static_assert called with wrong number of arguments")
314            }
315
316            // Arguments at positions `1..` form the message and they must all be constant.
317            if arguments.iter().skip(1).any(|argument| !dfg.is_constant(*argument)) {
318                return SimplifyResult::None;
319            }
320
321            if dfg.is_constant_true(arguments[0]) {
322                SimplifyResult::Remove
323            } else {
324                SimplifyResult::None
325            }
326        }
327        Intrinsic::ApplyRangeConstraint => {
328            let value = arguments[0];
329            let max_bit_size = dfg.get_numeric_constant(arguments[1]);
330            if let Some(max_bit_size) = max_bit_size {
331                let max_bit_size = max_bit_size.to_u128() as u32;
332                let max_potential_bits = dfg.get_value_max_num_bits(value);
333                if max_potential_bits < max_bit_size {
334                    SimplifyResult::Remove
335                } else {
336                    SimplifyResult::SimplifiedToInstruction(Instruction::RangeCheck {
337                        value,
338                        max_bit_size,
339                        assert_message: Some("call to assert_max_bit_size".to_owned()),
340                    })
341                }
342            } else {
343                SimplifyResult::None
344            }
345        }
346        Intrinsic::Hint(Hint::BlackBox) => SimplifyResult::None,
347        Intrinsic::BlackBox(bb_func) => {
348            simplify_black_box_func(bb_func, arguments, dfg, block, call_stack)
349        }
350        Intrinsic::AsWitness => SimplifyResult::None,
351        Intrinsic::IsUnconstrained => {
352            let result = dfg.runtime().is_brillig().into();
353            SimplifyResult::SimplifiedTo(dfg.make_constant(result, NumericType::bool()))
354        }
355        Intrinsic::DerivePedersenGenerators => {
356            if let Some(Type::Array(_, len)) = return_type.clone() {
357                simplify_derive_generators(dfg, arguments, len.0, block, call_stack)
358            } else {
359                unreachable!("Derive Pedersen Generators must return an array");
360            }
361        }
362        Intrinsic::FieldLessThan => {
363            if let Some(constants) = constant_args {
364                let lhs = constants[0];
365                let rhs = constants[1];
366                let result = dfg.make_constant((lhs < rhs).into(), NumericType::bool());
367                SimplifyResult::SimplifiedTo(result)
368            } else {
369                SimplifyResult::None
370            }
371        }
372        Intrinsic::ArrayRefCount | Intrinsic::VectorRefCount => {
373            if dfg.runtime.is_acir() {
374                // In ACIR, ref counts are not tracked so we always simplify them to zero.
375                let zero = dfg.make_constant(FieldElement::zero(), NumericType::unsigned(32));
376                SimplifyResult::SimplifiedTo(zero)
377            } else {
378                SimplifyResult::None
379            }
380        }
381    };
382
383    if let (Some(expected_types), SimplifyResult::SimplifiedTo(result)) =
384        (return_type, &simplified_result)
385    {
386        assert_eq!(
387            dfg.type_of_value(*result),
388            expected_types,
389            "Simplification should not alter return type"
390        );
391    }
392
393    simplified_result
394}

396/// Returns a vector (represented by a tuple (len, vector)) of constants corresponding to the limbs of the radix decomposition.
397fn simplify_constant_to_radix(
398    endian: Endian,
399    field: FieldElement,
400    radix: u32,
401    limb_count: u32,
402    mut make_array: impl FnMut(Vec<FieldElement>) -> ValueId,
403) -> SimplifyResult {
404    match constant_to_radix(endian, field, radix, limb_count) {
405        Some(result) => SimplifyResult::SimplifiedTo(make_array(result)),
406        None => SimplifyResult::None,
407    }
408}

410pub(crate) fn constant_to_radix(
411    endian: Endian,
412    field: FieldElement,
413    radix: u32,
414    limb_count: u32,
415) -> Option<Vec<FieldElement>> {
416    let bit_size = u32::BITS - (radix - 1).leading_zeros();
417    let radix_big = BigUint::from(radix);
418    let radix_range = BigUint::from(2u128)..=BigUint::from(256u128);
419    if !radix_range.contains(&radix_big) || BigUint::from(2u128).pow(bit_size) != radix_big {
420        // NOTE: expect an error to be thrown later in
421        // acir::generated_acir::radix_le_decompose
422        return None;
423    }
424    let big_integer = BigUint::from_bytes_be(&field.to_be_bytes());
425
426    // Decompose the integer into its radix digits in little endian form.
427    let decomposed_integer = big_integer.to_radix_le(radix);
428    if limb_count < decomposed_integer.len() as u32 {
429        // `field` cannot be represented as `limb_count` bits.
430        // defer error to acir_gen.
431        None
432    } else {
433        let mut limbs = vecmap(0..limb_count, |i| match decomposed_integer.get(i as usize) {
434            Some(digit) => FieldElement::from_be_bytes_reduce(&[*digit]),
435            None => FieldElement::zero(),
436        });
437        if endian == Endian::Big {
438            limbs.reverse();
439        }
440        Some(limbs)
441    }
442}

444fn make_constant_array(
445    dfg: &mut DataFlowGraph,
446    results: impl Iterator<Item = FieldElement>,
447    typ: NumericType,
448    block: BasicBlockId,
449    call_stack: CallStackId,
450) -> ValueId {
451    let result_constants: im::Vector<_> =
452        results.map(|element| dfg.make_constant(element, typ)).collect();
453
454    let typ = Type::Array(
455        Arc::new(vec![Type::Numeric(typ)]),
456        SemanticLength(assert_u32(result_constants.len())),
457    );
458    make_array(dfg, result_constants, typ, block, call_stack)
459}

461fn make_array(
462    dfg: &mut DataFlowGraph,
463    elements: im::Vector<ValueId>,
464    typ: Type,
465    block: BasicBlockId,
466    call_stack: CallStackId,
467) -> ValueId {
468    let instruction = Instruction::MakeArray { elements, typ };
469    dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()

472/// Vectors have a tuple structure (vector length, vector contents) to enable logic
473/// that uses dynamic vector lengths (such as with merging vectors in the flattening pass).
474/// This method codegens an update to the vector length.
475///
476/// The binary operation performed on the vector length is always an addition or subtraction of `1`.
477/// This is because the vector length holds the user length (length as displayed by a `.len()` call),
478/// and not a flattened length used internally to represent arrays of tuples.
479fn update_vector_length(
480    vector_len: ValueId,
481    dfg: &mut DataFlowGraph,
482    operator: BinaryOp,
483    block: BasicBlockId,
484    call_stack: CallStackId,
485) -> ValueId {
486    let one = dfg.make_constant(FieldElement::one(), NumericType::length_type());
487    let instruction = Instruction::Binary(Binary { lhs: vector_len, operator, rhs: one });
488    dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
489}

491fn increment_vector_length(
492    vector_len: ValueId,
493    dfg: &mut DataFlowGraph,
494    block: BasicBlockId,
495    call_stack: CallStackId,
496) -> ValueId {
497    update_vector_length(vector_len, dfg, BinaryOp::Add { unchecked: false }, block, call_stack)
498}
499
500fn decrement_vector_length(
501    vector_len: ValueId,
502    dfg: &mut DataFlowGraph,
503    block: BasicBlockId,
504    call_stack: CallStackId,
505) -> ValueId {
506    // Simplifications only run if the length is a known non-zero constant, so the subtraction should never overflow.
507    update_vector_length(vector_len, dfg, BinaryOp::Sub { unchecked: true }, block, call_stack)
508}

510fn simplify_vector_push_back(
511    mut vector: im::Vector<ValueId>,
512    element_type: Type,
513    arguments: &[ValueId],
514    dfg: &mut DataFlowGraph,
515    block: BasicBlockId,
516    call_stack: CallStackId,
517) -> SimplifyResult {
518    // The capacity must be an integer so that we can compare it against the vector length
519    let capacity = dfg.make_constant((vector.len() as u128).into(), NumericType::length_type());
520    let len_equals_capacity_instr =
521        Instruction::Binary(Binary { lhs: arguments[0], operator: BinaryOp::Eq, rhs: capacity });
522    let len_equals_capacity = dfg
523        .insert_instruction_and_results(len_equals_capacity_instr, block, None, call_stack)
524        .first();
525    let len_not_equals_capacity_instr = Instruction::Not(len_equals_capacity);
526    let len_not_equals_capacity = dfg
527        .insert_instruction_and_results(len_not_equals_capacity_instr, block, None, call_stack)
528        .first();
529
530    let new_vector_length = increment_vector_length(arguments[0], dfg, block, call_stack);
531
532    for elem in &arguments[2..] {
533        vector.push_back(*elem);
534    }
535    let vector_size = SemiFlattenedLength(assert_u32(vector.len()));
536    let element_size = element_type.element_size();
537    let new_vector = make_array(dfg, vector, element_type, block, call_stack);
538
539    let set_last_vector_value_instr = Instruction::ArraySet {
540        array: new_vector,
541        index: arguments[0],
542        value: arguments[2],
543        mutable: false,
544    };
545
546    let set_last_vector_value = dfg
547        .insert_instruction_and_results(set_last_vector_value_instr, block, None, call_stack)
548        .first();
549
550    let mut vector_sizes = HashMap::default();
551    vector_sizes.insert(set_last_vector_value, vector_size / element_size);
552    vector_sizes.insert(new_vector, vector_size / element_size);
553
554    let mut value_merger = ValueMerger::new(dfg, block, &vector_sizes, call_stack);
555
556    let Ok(new_vector) = value_merger.merge_values(
557        len_not_equals_capacity,
558        len_equals_capacity,
559        set_last_vector_value,
560        new_vector,
561    ) else {
562        // If we were to percolate up the error here, it'd get to insert_instruction and eventually
563        // all of ssa. Instead we just choose not to simplify the vector call since this should
564        // be a rare case.
565        return SimplifyResult::None;
566    };
567
568    SimplifyResult::SimplifiedToMultiple(vec![new_vector_length, new_vector])
569}

571fn simplify_vector_pop_back(
572    mut vector: im::Vector<ValueId>,
573    vector_type: Type,
574    arguments: &[ValueId],
575    dfg: &mut DataFlowGraph,
576    block: BasicBlockId,
577    call_stack: CallStackId,
578) -> SimplifyResult {
579    let element_types = vector_type.element_types();
580    let element_count = element_types.len();
581    let mut results = VecDeque::with_capacity(element_count + 1);
582
583    let new_vector_length = decrement_vector_length(arguments[0], dfg, block, call_stack);
584
585    let element_size =
586        dfg.make_constant((element_count as u128).into(), NumericType::length_type());
587    // Compute the flattened length doing an unchecked mul
588    // (it shouldn't overflow because it would have overflowed before when the vector was created)
589    let flattened_len_instr =
590        Instruction::binary(BinaryOp::Mul { unchecked: true }, arguments[0], element_size);
591    let mut flattened_len =
592        dfg.insert_instruction_and_results(flattened_len_instr, block, None, call_stack).first();
593
594    // We must pop multiple elements in the case of a vector of tuples
595    // Iterating through element types in reverse here since we're popping from the end
596    for element_type in element_types.iter().rev() {
597        flattened_len = decrement_vector_length(flattened_len, dfg, block, call_stack);
598        let get_last_elem_instr =
599            Instruction::ArrayGet { array: arguments[1], index: flattened_len };
600
601        let element_type = Some(vec![element_type.clone()]);
602        let get_last_elem = dfg
603            .insert_instruction_and_results(get_last_elem_instr, block, element_type, call_stack)
604            .first();
605        results.push_front(get_last_elem);
606        vector.pop_back();
607    }
608
609    let new_vector = make_array(dfg, vector, vector_type, block, call_stack);
610    results.push_front(new_vector);
611
612    results.push_front(new_vector_length);
613    SimplifyResult::SimplifiedToMultiple(results.into())
614}

616/// Try to simplify this black box call. If the call can be simplified to a known value,
617/// that value is returned. Otherwise [`SimplifyResult::None`] is returned.
618fn simplify_black_box_func(
619    bb_func: BlackBoxFunc,
620    arguments: &[ValueId],
621    dfg: &mut DataFlowGraph,
622    block: BasicBlockId,
623    call_stack: CallStackId,
624) -> SimplifyResult {
625    cfg_if::cfg_if! {
626        if #[cfg(feature = "bn254")] {
627            let solver = bn254_blackbox_solver::Bn254BlackBoxSolver;
628        } else {
629            let solver = acvm::blackbox_solver::StubbedBlackBoxSolver;
630        }
631    };
632    match bb_func {
633        BlackBoxFunc::Blake2s => blackbox::simplify_hash(
634            dfg,
635            arguments,
636            acvm::blackbox_solver::blake2s,
637            block,
638            call_stack,
639        ),
640        BlackBoxFunc::Blake3 => blackbox::simplify_hash(
641            dfg,
642            arguments,
643            acvm::blackbox_solver::blake3,
644            block,
645            call_stack,
646        ),
647        BlackBoxFunc::Keccakf1600 => {
648            if let Some((array_input, _)) = dfg.get_array_constant(arguments[0]) {
649                if array_is_constant(dfg, &array_input) {
650                    let const_input: Vec<u64> = array_input
651                        .iter()
652                        .map(|id| {
653                            let field = dfg
654                                .get_numeric_constant(*id)
655                                .expect("value id from array should point at constant");
656                            field.to_u128() as u64
657                        })
658                        .collect();
659
660                    let state = acvm::blackbox_solver::keccakf1600(
661                        const_input.try_into().expect("Keccakf1600 input should have length of 25"),
662                    )
663                    .expect("Rust solvable black box function should not fail");
664                    let state_values = state.iter().map(|x| FieldElement::from(u128::from(*x)));
665                    let result_array = make_constant_array(
666                        dfg,
667                        state_values,
668                        NumericType::Unsigned { bit_size: 64 },
669                        block,
670                        call_stack,
671                    );
672                    SimplifyResult::SimplifiedTo(result_array)
673                } else {
674                    SimplifyResult::None
675                }
676            } else {
677                SimplifyResult::None
678            }
679        }
680        BlackBoxFunc::Poseidon2Permutation => {
681            blackbox::simplify_poseidon2_permutation(dfg, solver, arguments, block, call_stack)
682        }
683        BlackBoxFunc::EcdsaSecp256k1 => blackbox::simplify_signature(
684            dfg,
685            arguments,
686            acvm::blackbox_solver::ecdsa_secp256k1_verify,
687        ),
688        BlackBoxFunc::EcdsaSecp256r1 => blackbox::simplify_signature(
689            dfg,
690            arguments,
691            acvm::blackbox_solver::ecdsa_secp256r1_verify,
692        ),
693
694        BlackBoxFunc::MultiScalarMul => {
695            blackbox::simplify_msm(dfg, solver, arguments, block, call_stack)
696        }
697        BlackBoxFunc::EmbeddedCurveAdd => {
698            blackbox::simplify_ec_add(dfg, solver, arguments, block, call_stack)
699        }
700
701        BlackBoxFunc::RecursiveAggregation => SimplifyResult::None,
702
703        BlackBoxFunc::AND => {
704            unreachable!("ICE: `BlackBoxFunc::AND` calls should be transformed into a `BinaryOp`")
705        }
706        BlackBoxFunc::XOR => {
707            unreachable!("ICE: `BlackBoxFunc::XOR` calls should be transformed into a `BinaryOp`")
708        }
709        BlackBoxFunc::RANGE => {
710            unreachable!(
711                "ICE: `BlackBoxFunc::RANGE` calls should be transformed into a `Instruction::Cast`"
712            )
713        }
714        BlackBoxFunc::Sha256Compression => {
715            blackbox::simplify_sha256_compression(dfg, arguments, block, call_stack)
716        }
717        BlackBoxFunc::AES128Encrypt => SimplifyResult::None,
718    }
719}

721fn to_u8_vec(dfg: &DataFlowGraph, values: im::Vector<ValueId>) -> Vec<u8> {
722    values
723        .iter()
724        .map(|id| {
725            let field = dfg
726                .get_numeric_constant(*id)
727                .expect("value id from array should point at constant");
728            *field.to_be_bytes().last().unwrap()
729        })
730        .collect()
731}
732
733fn array_is_constant(dfg: &DataFlowGraph, values: &im::Vector<ValueId>) -> bool {
734    values.iter().all(|value| dfg.get_numeric_constant(*value).is_some())
735}

737/// Replaces a call to `derive_pedersen_generators` with the results of the computation.
738///
739/// It only works if the arguments to the call are both constants, which means that the
740/// function which contains this call needs to be inlined into its caller, where the
741/// arguments are known. This is taken care of by the `#[no_predicates]` attribute,
742/// which forces inlining after flattening.
743///
744/// This intrinsic must not reach Brillig-gen.
745fn simplify_derive_generators(
746    dfg: &mut DataFlowGraph,
747    arguments: &[ValueId],
748    num_generators: u32,
749    block: BasicBlockId,
750    call_stack: CallStackId,
751) -> SimplifyResult {
752    if arguments.len() == 2 {
753        let domain_separator_string = dfg.get_array_constant(arguments[0]);
754        let starting_index = dfg.get_numeric_constant(arguments[1]);
755        if let (Some(domain_separator_string), Some(starting_index)) =
756            (domain_separator_string, starting_index)
757        {
758            let domain_separator_bytes = domain_separator_string
759                .0
760                .iter()
761                .map(|&x| dfg.get_numeric_constant(x).unwrap().to_u128() as u8)
762                .collect::<Vec<u8>>();
763            let generators = derive_generators(
764                &domain_separator_bytes,
765                num_generators,
766                starting_index.try_to_u32().expect("argument is declared as u32"),
767            );
768            let is_infinite = dfg.make_constant(FieldElement::zero(), NumericType::bool());
769            let mut results = Vec::new();
770            for generator in generators {
771                let x_big: BigUint = generator.x.into();
772                let x = FieldElement::from_be_bytes_reduce(&x_big.to_bytes_be());
773                let y_big: BigUint = generator.y.into();
774                let y = FieldElement::from_be_bytes_reduce(&y_big.to_bytes_be());
775                results.push(dfg.make_constant(x, NumericType::NativeField));
776                results.push(dfg.make_constant(y, NumericType::NativeField));
777                results.push(is_infinite);
778            }
779            let len = results.len() as u32;
780            assert!(
781                len % 3 == 0,
782                "The number of results from derive_generators must be a multiple of 3"
783            );
784            let typ = Type::Array(
785                vec![Type::field(), Type::field(), Type::unsigned(1)].into(),
786                SemanticLength(len / 3),
787            );
788            let result = make_array(dfg, results.into(), typ, block, call_stack);
789            SimplifyResult::SimplifiedTo(result)
790        } else {
791            SimplifyResult::None
792        }
793    } else {
794        unreachable!("Unexpected number of arguments to derive_generators");
795    }
796}
