26pub(crate) struct ValueMerger<'a> {
27    dfg: &'a mut DataFlowGraph,
28    block: BasicBlockId,
29
30    /// Maps SSA array values with a vector type to their size.
31    /// This must be computed before merging values.
32    vector_sizes: &'a HashMap<ValueId, SemanticLength>,
33
34    call_stack: CallStackId,
35
36    /// Optional information about the current side effects variable, and what
37    /// side effects variables are applied to each `array_set` instruction that happen
38    /// before the values being merged.
39    ///
40    /// When arrays or vectors are merged as a result of an `if ... else` instruction,
41    /// a new array will be built that does `array_get` on the arrays on the "then" and
42    /// "else" branches, and combine those values. These newly inserted `array_get` could
43    /// be optimized by reusing previously inserted instructions, such as an `array_set` at
44    /// the same index as the one in the `array_get`. However, this is only safe to do
45    /// if we know the side effects var of those two instructions is the same. Hence, that
46    /// information can optionally be specified here.
47    array_get_optimization_side_effects: Option<ArrayGetOptimizationSideEffects<'a>>,
48}

50impl<'a> ValueMerger<'a> {
51    pub(crate) fn new(
52        dfg: &'a mut DataFlowGraph,
53        block: BasicBlockId,
54        vector_sizes: &'a HashMap<ValueId, SemanticLength>,
55        call_stack: CallStackId,
56        array_get_optimization_side_effects: Option<ArrayGetOptimizationSideEffects<'a>>,
57    ) -> Self {
58        ValueMerger { dfg, block, vector_sizes, call_stack, array_get_optimization_side_effects }
59    }
60
61    /// Choose a call stack to return with the [RuntimeError].
62    ///
63    /// If the call stack of the value is empty, it returns the call stack of the if-then-else itself.
64    fn get_call_stack(&self, value: ValueId) -> Vec<Location> {
65        // The value points at one of the problematic references, while the instruction would
66        // point at where we got the if-then-else; it's not clear which one is more useful.
67        let call_stack = self.dfg.get_value_call_stack(value);
68        if call_stack.is_empty() { self.dfg.get_call_stack(self.call_stack) } else { call_stack }
69    }
70
71    /// Merge two values a and b to a single value.
72    /// If these two values are numeric, the result will be
73    /// `then_condition * (then_value - else_value) + else_value`.
74    /// Otherwise, if the values being merged are arrays, a new array will be made
75    /// recursively from combining each element of both input arrays.
76    ///
77    /// Returns an error if called with a function value or a reference or function values
78    /// as it is less clear how to merge these.
79    pub(crate) fn merge_values(
80        &mut self,
81        then_condition: ValueId,
82        else_condition: ValueId,
83        then_value: ValueId,
84        else_value: ValueId,
85    ) -> RtResult<ValueId> {
86        if then_value == else_value {
87            return Ok(then_value);
88        }
89
90        match self.dfg.type_of_value(then_value) {
91            Type::Numeric(_) => Ok(Self::merge_numeric_values(
92                self.dfg,
93                self.block,
94                then_condition,
95                else_condition,
96                then_value,
97                else_value,
98            )),
99            typ @ Type::Array(_, _) => {
100                self.merge_array_values(typ, then_condition, else_condition, then_value, else_value)
101            }
102            typ @ Type::Vector(_) => self.merge_vector_values(
103                typ,
104                then_condition,
105                else_condition,
106                then_value,
107                else_value,
108            ),
109            Type::Reference(_) => {
110                let call_stack = self.get_call_stack(then_value);
111                Err(RuntimeError::ReturnedReferenceFromDynamicIf { call_stack })
112            }
113            Type::Function => {
114                let call_stack = self.get_call_stack(then_value);
115                Err(RuntimeError::ReturnedFunctionFromDynamicIf { call_stack })
116            }
117        }
118    }
119
120    /// Merge two numeric values a and b from separate basic blocks to a single value. This
121    /// function would return the result of `if c { a } else { b }` as  `c*a + (!c)*b`.
122    pub(crate) fn merge_numeric_values(
123        dfg: &mut DataFlowGraph,
124        block: BasicBlockId,
125        then_condition: ValueId,
126        else_condition: ValueId,
127        then_value: ValueId,
128        else_value: ValueId,
129    ) -> ValueId {
130        let then_type = dfg.type_of_value(then_value).unwrap_numeric();
131        let else_type = dfg.type_of_value(else_value).unwrap_numeric();
132        assert_eq!(
133            then_type, else_type,
134            "Expected values merged to be of the same type but found {then_type} and {else_type}"
135        );
136
137        if then_value == else_value {
138            return then_value;
139        }
140
141        let then_call_stack = dfg.get_value_call_stack_id(then_value);
142        let else_call_stack = dfg.get_value_call_stack_id(else_value);
143
144        let call_stack = if then_call_stack.is_root() { else_call_stack } else { then_call_stack };
145
146        // We must cast the bool conditions to the actual numeric type used by each value.
147        let cast = Instruction::Cast(then_condition, then_type);
148        let then_condition =
149            dfg.insert_instruction_and_results(cast, block, None, call_stack).first();
150
151        let cast = Instruction::Cast(else_condition, else_type);
152        let else_condition =
153            dfg.insert_instruction_and_results(cast, block, None, call_stack).first();
154
155        // Unchecked mul because `then_condition` will be 1 or 0
156        let mul =
157            Instruction::binary(BinaryOp::Mul { unchecked: true }, then_condition, then_value);
158        let then_value = dfg.insert_instruction_and_results(mul, block, None, call_stack).first();
159
160        // Unchecked mul because `else_condition` will be 1 or 0
161        let mul =
162            Instruction::binary(BinaryOp::Mul { unchecked: true }, else_condition, else_value);
163        let else_value = dfg.insert_instruction_and_results(mul, block, None, call_stack).first();
164
165        // Unchecked add because one of the values will always be 0
166        let add = Instruction::binary(BinaryOp::Add { unchecked: true }, then_value, else_value);
167        dfg.insert_instruction_and_results(add, block, None, call_stack).first()
168    }

170    /// Given an if expression that returns an array: `if c { array1 } else { array2 }`,
171    /// this function will recursively merge array1 and array2 into a single resulting array
172    /// by creating a new array containing the result of `self.merge_values` for each element.
173    fn merge_array_values(
174        &mut self,
175        typ: Type,
176        then_condition: ValueId,
177        else_condition: ValueId,
178        then_value: ValueId,
179        else_value: ValueId,
180    ) -> Result<ValueId, RuntimeError> {
181        let mut merged = im::Vector::new();
182
183        let (element_types, len) = match &typ {
184            Type::Array(elements, len) => (elements.as_slice(), *len),
185            _ => panic!("Expected array type"),
186        };
187
188        let element_count = element_types.len() as u32;
189
190        for i in 0..len.0 {
191            for (element_index, element_type) in element_types.iter().enumerate() {
192                let index_value = u128::from(i * element_count + element_index as u32).into();
193                let index = self.dfg.make_constant(index_value, NumericType::length_type());
194
195                let typevars = Some(vec![element_type.clone()]);
196
197                let mut get_element = |array, typevars| {
198                    self.maybe_optimized_array_get(array, index, index_value, typevars)
199                };
200
201                let then_element = get_element(then_value, typevars.clone());
202                let else_element = get_element(else_value, typevars);
203
204                merged.push_back(self.merge_values(
205                    then_condition,
206                    else_condition,
207                    then_element,
208                    else_element,
209                )?);
210            }
211        }
212
213        let instruction = Instruction::MakeArray { elements: merged, typ };
214        let result =
215            self.dfg.insert_instruction_and_results(instruction, self.block, None, self.call_stack);
216        Ok(result.first())
217    }

219    fn merge_vector_values(
220        &mut self,
221        typ: Type,
222        then_condition: ValueId,
223        else_condition: ValueId,
224        then_value_id: ValueId,
225        else_value_id: ValueId,
226    ) -> Result<ValueId, RuntimeError> {
227        let mut merged = im::Vector::new();
228
229        let element_types = match &typ {
230            Type::Vector(elements) => elements.as_slice(),
231            _ => panic!("Expected vector type"),
232        };
233
234        let then_len = self.vector_sizes.get(&then_value_id).copied().unwrap_or_else(|| {
235            panic!("ICE: Merging values during flattening encountered vector {then_value_id} without a preset size");
236        });
237
238        let else_len = self.vector_sizes.get(&else_value_id).copied().unwrap_or_else(|| {
239            panic!("ICE: Merging values during flattening encountered vector {else_value_id} without a preset size");
240        });
241
242        let len = then_len.max(else_len);
243        let element_count = ElementTypesLength(assert_u32(element_types.len()));
244
245        let semi_flat_then_length = then_len * element_count;
246        let semi_flat_else_length = else_len * element_count;
247
248        for i in 0..len.0 {
249            for (element_index, element_type) in element_types.iter().enumerate() {
250                let index_u32 = i * element_count.0 + element_index as u32;
251                let index_value = u128::from(index_u32).into();
252                let index = self.dfg.make_constant(index_value, NumericType::length_type());
253
254                let typevars = Some(vec![element_type.clone()]);
255
256                let mut get_element = |array, typevars, len: SemiFlattenedLength| {
257                    assert!(index_u32 < len.0, "get_element invoked with an out of bounds index");
258
259                    self.maybe_optimized_array_get(array, index, index_value, typevars)
260                };
261
262                // If it's out of bounds for the "then" vector, a value in the "else" *must* exist.
263                // We can use that value directly as accessing it is always checked against the actual
264                // vector length.
265                if index_u32 >= semi_flat_then_length.0 {
266                    let else_element = get_element(else_value_id, typevars, semi_flat_else_length);
267                    merged.push_back(else_element);
268                    continue;
269                }
270
271                // Same for if it's out of bounds for the "else" vector.
272                if index_u32 >= semi_flat_else_length.0 {
273                    let then_element = get_element(then_value_id, typevars, semi_flat_then_length);
274                    merged.push_back(then_element);
275                    continue;
276                }
277
278                let then_element =
279                    get_element(then_value_id, typevars.clone(), semi_flat_then_length);
280                let else_element = get_element(else_value_id, typevars, semi_flat_else_length);
281
282                merged.push_back(self.merge_values(
283                    then_condition,
284                    else_condition,
285                    then_element,
286                    else_element,
287                )?);
288            }
289        }
290
291        let instruction = Instruction::MakeArray { elements: merged, typ };
292        let call_stack = self.call_stack;
293        let result =
294            self.dfg.insert_instruction_and_results(instruction, self.block, None, call_stack);
295        Ok(result.first())
296    }
297
298    fn maybe_optimized_array_get(
299        &mut self,
300        array: ValueId,
301        index: ValueId,
302        index_value: FieldElement,
303        typevars: Option<Vec<Type>>,
304    ) -> ValueId {
305        let side_effects = self.array_get_optimization_side_effects.as_ref();
306        match try_optimize_array_get_from_previous_instructions(
307            array,
308            index_value,
309            self.dfg,
310            side_effects,
311        ) {
312            Some(ArrayGetOptimizationResult::Value(value)) => value,
313            Some(ArrayGetOptimizationResult::ArrayGet(new_array)) => {
314                assert_ne!(
315                    new_array, array,
316                    "ArrayGetOptimizationResult::ArrayGet returned the same array_id"
317                );
318
319                let get = Instruction::ArrayGet { array: new_array, index };
320                self.dfg
321                    .insert_instruction_and_results(get, self.block, typevars, self.call_stack)
322                    .first()
323            }
324            None => {
325                let get = Instruction::ArrayGet { array, index };
326                self.dfg
327                    .insert_instruction_and_results(get, self.block, typevars, self.call_stack)
328                    .first()
329            }
330        }
331    }
