27/// Reference to an instruction
28///
29/// Note that InstructionIds are not unique. That is, two InstructionIds
30/// may refer to the same Instruction data. This is because, although
31/// identical, instructions may have different results based on their
32/// placement within a block.
33pub(crate) type InstructionId = Id<Instruction>;

35/// These are similar to built-ins in other languages.
36/// These can be classified under two categories:
37/// - Opcodes which the IR knows the target machine has
38///   special support for. (LowLevel)
39/// - Opcodes which have no function definition in the
40///   source code and must be processed by the IR.
41#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
42pub enum Intrinsic {
43    /// ArrayLen - returns the length of the input array
44    /// argument: array (value id)
45    /// result: length of the array, panic if the input is not an array
46    ArrayLen,
47    /// ArrayAsStrUnchecked - Converts a byte array of type `[u8; N]` to a string
48    /// argument: array (value id)
49    /// result: str
50    ArrayAsStrUnchecked,
51    /// AsVector
52    /// argument: value id
53    /// result: a vector containing the elements of the argument. Panic if the value id does not correspond to an `array` type
54    AsVector,
55    /// AssertConstant - Enforce the argument to be a constant value, at compile time.
56    /// argument: value id
57    /// result: (), panic if the argument does not resolve to a constant value
58    AssertConstant,
59    /// StaticAssert - Enforce the first argument to be true, at compile time
60    /// arguments: boolean (value id), ...message. The message can be a `format string` of several arguments
61    /// result: (), panic if the arguments do not resolve to constant values or if the first one is false.
62    StaticAssert,
63    /// VectorPushBack - Add elements at the end of a vector
64    /// arguments:  vector length, vector contents, ...elements_to_push
65    /// result: a vector containing `vector contents,..elements_to_push`
66    VectorPushBack,
67    /// VectorPushFront - Add elements at the start of a vector
68    /// arguments:  vector length, vector contents, ...elements_to_push
69    /// result: a vector containing `..elements_to_push, vector contents`
70    VectorPushFront,
71    /// VectorPopBack - Removes the last element of a vector
72    /// arguments: vector length, vector contents
73    /// result: a vector without the last element of `vector contents`
74    VectorPopBack,
75    /// VectorPopFront - Removes the first element of a vector
76    /// arguments: vector length, vector contents
77    /// result: a vector without the first element of `vector contents`
78    VectorPopFront,
79    /// VectorInsert - Insert elements inside a vector.
80    /// arguments: vector length, vector contents, insert index, ...elements_to_insert
81    /// result: a vector with ...elements_to_insert inserted at the `insert index`
82    VectorInsert,
83    /// VectorRemove - Removes an element from a vector
84    /// arguments: vector length, vector contents, remove index
85    /// result: a vector with without the element at `remove index`
86    VectorRemove,
87    /// ApplyRangeConstraint - Enforces the `bit size` of the first argument via a range check.
88    /// arguments: value id, bit size (constant)
89    /// result: applies a range check constraint to the input. It is replaced by a RangeCheck instruction during simplification.
90    ApplyRangeConstraint,
91    /// StrAsBytes - Convert a `str` into a byte array of type `[u8; N]`
92    /// arguments: value id
93    /// result: the argument. Internally a `str` is a byte array.
94    StrAsBytes,
95    /// ToBits(Endian) - Computes the bit decomposition of the argument.
96    /// argument: a field element (value id)
97    /// result: an array whose elements are the bit decomposition of the argument, in the endian order depending on the chosen variant.
98    /// The type of the result gives the number of limbs to use for the decomposition.
99    ToBits(Endian),
100    /// ToRadix(Endian) - Decompose the first argument over the `radix` base
101    /// arguments: a field element (value id), the radix to use (constant, a power of 2 between 2 and 256)
102    /// result: an array whose elements are the decomposition of the argument into the `radix` base, in the endian order depending on the chosen variant.
103    /// The type of the result gives the number of limbs to use for the decomposition.
104    ToRadix(Endian),
105    /// BlackBox(BlackBoxFunc) - Calls a blackbox function. More details can be found here: [acvm-repo::acir::::circuit::opcodes::BlackBoxFuncCall]
106    BlackBox(BlackBoxFunc),
107    /// Hint(Hint) - Avoid its arguments to be removed by DIE.
108    /// arguments: ... value id
109    /// result: the arguments. Hint does not layout any constraint but avoid its arguments to be simplified out during SSA transformations
110    Hint(Hint),
111    /// AsWitness - Adds a new witness constrained to be equal to the argument
112    /// arguments: value id
113    /// result: the argument
114    AsWitness,
115    /// IsUnconstrained - Indicates if the execution context is constrained or unconstrained
116    /// argument: ()
117    /// result: true if execution is under unconstrained context, false else.
118    IsUnconstrained,
119    /// DerivePedersenGenerators - Computes the Pedersen generators
120    /// arguments: domain_separator_string (constant string), starting_index (constant)
121    /// result: array of elliptic curve points (Grumpkin) containing the generators.
122    /// The type of the result gives the number of generators to compute.
123    DerivePedersenGenerators,
124    /// FieldLessThan - Compare the arguments: `lhs` < `rhs`
125    /// arguments: lhs, rhs. Field elements
126    /// result: true if `lhs` mod p < `rhs` mod p (p being the field characteristic), false else
127    FieldLessThan,
128    /// ArrayRefCount - Gives the reference count of the array
129    /// argument: array (value id)
130    /// result: reference count of `array`. In unconstrained context, the reference count is stored alongside the array.
131    /// in constrained context, it will be 0.
132    ArrayRefCount,
133    /// VectorRefCount - Gives the reference count of the vector
134    /// arguments: vector length, vector contents (value id)
135    /// result: reference count of `vector`. In unconstrained context, the reference count is stored alongside the vector.
136    /// in constrained context, it will be 0.
137    VectorRefCount,
138}

172impl Intrinsic {
173    /// Returns whether the `Intrinsic` has side effects.
174    ///
175    /// If there are no side effects then the `Intrinsic` can be removed if the result is unused.
176    ///
177    /// An example of a side effect is increasing the reference count of an array, but functions
178    /// which can fail due to implicit constraints are also considered to have a side effect.
179    pub(crate) fn has_side_effects(&self) -> bool {
180        match self {
181            Intrinsic::AssertConstant
182            | Intrinsic::StaticAssert
183            | Intrinsic::ApplyRangeConstraint
184            // Array & vector ref counts are treated as having side effects since they operate
185            // on hidden variables on otherwise identical array values.
186            | Intrinsic::ArrayRefCount
187            | Intrinsic::VectorRefCount
188            | Intrinsic::AsWitness => true,
189
190            // These apply a constraint that the input must fit into a specified number of limbs.
191            Intrinsic::ToBits(_) | Intrinsic::ToRadix(_) => true,
192
193            // These imply a check that the vector is non-empty and should fail otherwise.
194            Intrinsic::VectorPopBack | Intrinsic::VectorPopFront | Intrinsic::VectorRemove | Intrinsic::VectorInsert => true,
195
196            Intrinsic::ArrayLen
197            | Intrinsic::ArrayAsStrUnchecked
198            | Intrinsic::AsVector
199            | Intrinsic::VectorPushBack
200            | Intrinsic::VectorPushFront
201            | Intrinsic::StrAsBytes
202            | Intrinsic::IsUnconstrained
203            | Intrinsic::DerivePedersenGenerators
204            | Intrinsic::FieldLessThan => false,
205
206            // Treat the black_box hint as-if it could potentially have side effects.
207            Intrinsic::Hint(Hint::BlackBox) => true,
208
209            // Some black box functions have side-effects
210            Intrinsic::BlackBox(func) => func.has_side_effects(),
211        }
212    }
213
214    /// Returns true if this intrinsic can modify its input array in Brillig
215    /// due to copy-on-write optimization when the reference count is 1.
216    ///
217    /// This is used to ensure we don't skip clones for these operations,
218    /// even though they're technically "pure" (no observable side effects).
219    /// Without proper reference counting, the caller's array could be corrupted.
220    pub(crate) fn modifies_input_array_in_brillig(&self) -> bool {
221        matches!(
222            self,
223            Intrinsic::VectorPushBack
224                | Intrinsic::VectorPushFront
225                | Intrinsic::VectorPopBack
226                | Intrinsic::VectorPopFront
227                | Intrinsic::VectorInsert
228                | Intrinsic::VectorRemove
229        )
230    }
231
232    pub(crate) fn purity(&self) -> Purity {
233        match self {
234            // These apply a constraint in the form of ACIR opcodes, but they can be deduplicated
235            // if the inputs are the same. If they depend on a side effect variable (e.g. because
236            // they were in an if-then-else) then `handle_instruction_side_effects` in `flatten_cfg`
237            // will have attached the condition variable to their inputs directly, so they don't
238            // directly depend on the corresponding `enable_side_effect` instruction any more.
239            // However, to conform with the expectations of `Instruction::can_be_deduplicated` and
240            // `constant_folding` we only use this information if the caller shows interest in it.
241            Intrinsic::ToBits(_) | Intrinsic::ToRadix(_) => Purity::PureWithPredicate,
242            Intrinsic::BlackBox(func) if func.has_side_effects() => Purity::PureWithPredicate,
243
244            // Operations that remove items from a vector don't modify the vector, they just assert it's non-empty.
245            // Vector insert also reads from its input vector, thus needing to assert that it is non-empty.
246            Intrinsic::VectorPopBack
247            | Intrinsic::VectorPopFront
248            | Intrinsic::VectorRemove
249            | Intrinsic::VectorInsert => Purity::PureWithPredicate,
250
251            Intrinsic::AssertConstant
252            | Intrinsic::StaticAssert
253            | Intrinsic::ApplyRangeConstraint
254            | Intrinsic::AsWitness => Purity::PureWithPredicate,
255
256            _ if self.has_side_effects() => Purity::Impure,
257            _ => Purity::Pure,
258        }
259    }

261    /// Lookup an Intrinsic by name and return it if found.
262    /// If there is no such intrinsic by that name, None is returned.
263    pub(crate) fn lookup(name: &str) -> Option<Intrinsic> {

265            "array_len" => Some(Intrinsic::ArrayLen),
266            "array_as_str_unchecked" => Some(Intrinsic::ArrayAsStrUnchecked),

268            "assert_constant" => Some(Intrinsic::AssertConstant),
269            "static_assert" => Some(Intrinsic::StaticAssert),
270            "apply_range_constraint" => Some(Intrinsic::ApplyRangeConstraint),
271            "vector_push_back" => Some(Intrinsic::VectorPushBack),
272            "vector_push_front" => Some(Intrinsic::VectorPushFront),
273            "vector_pop_back" => Some(Intrinsic::VectorPopBack),
274            "vector_pop_front" => Some(Intrinsic::VectorPopFront),
275            "vector_insert" => Some(Intrinsic::VectorInsert),
276            "vector_remove" => Some(Intrinsic::VectorRemove),
277            "str_as_bytes" => Some(Intrinsic::StrAsBytes),
278            "to_le_radix" => Some(Intrinsic::ToRadix(Endian::Little)),
279            "to_be_radix" => Some(Intrinsic::ToRadix(Endian::Big)),
280            "to_le_bits" => Some(Intrinsic::ToBits(Endian::Little)),
281            "to_be_bits" => Some(Intrinsic::ToBits(Endian::Big)),
282            "as_witness" => Some(Intrinsic::AsWitness),
283            "is_unconstrained" => Some(Intrinsic::IsUnconstrained),
284            "derive_pedersen_generators" => Some(Intrinsic::DerivePedersenGenerators),

290            other => BlackBoxFunc::lookup(other).map(Intrinsic::BlackBox),
291        }

313/// Instructions are used to perform tasks.
314/// The instructions that the IR is able to specify are listed below.
315pub enum Instruction {
316    /// Binary Operations like +, -, *, /, ==, !=
317    Binary(Binary),
318
319    /// Converts `Value` into the given `NumericType`
320    ///
321    /// This operation only changes the type of the value, it does not change the value itself.
322    /// It is expected that the value can fit into the target type.
323    /// For instance a value of type `u32` casted to `u8` must already fit into 8 bits
324    /// A value of type `i8` cannot be casted to 'i16' since the value would need to include the sign bit (which is the MSB)
325    /// Ssa code-gen must ensure that the necessary truncation or sign extension is performed when emitting a Cast instruction.
326    Cast(ValueId, NumericType),
327
328    /// Computes a bit wise not
329    Not(ValueId),
330
331    /// Truncates `value` to `bit_size`
332    Truncate { value: ValueId, bit_size: u32, max_bit_size: u32 },
333
334    /// Constrains two values to be equal to one another.
335    Constrain(ValueId, ValueId, Option<ConstrainError>),
336
337    /// Constrains two values to not be equal to one another.
338    ConstrainNotEqual(ValueId, ValueId, Option<ConstrainError>),
339
340    /// Range constrain `value` to `max_bit_size`
341    RangeCheck { value: ValueId, max_bit_size: u32, assert_message: Option<String> },
342
343    /// Performs a function call with a list of its arguments.
344    Call { func: ValueId, arguments: Vec<ValueId> },
345
346    /// Allocates a region of memory. Note that this is not concerned with
347    /// the type of memory, the type of element is determined when loading this memory.
348    /// This is used for representing mutable variables and references.
349    Allocate,
350
351    /// Loads a value from memory.
352    Load { address: ValueId },
353
354    /// Writes a value to memory.
355    Store { address: ValueId, value: ValueId },
356
357    /// Provides a context for all instructions that follow up until the next
358    /// `EnableSideEffectsIf` is encountered, for stating a condition that determines whether
359    /// such instructions are allowed to have side-effects.
360    ///
361    /// For example,
362    /// ```text
363    /// EnableSideEffectsIf condition0;
364    /// code0;
365    /// EnableSideEffectsIf condition1;
366    /// code1;
367    /// ```
368    /// - `code0` will have side effects iff `condition0` evaluates to `true`
369    /// - `code1` will have side effects iff `condition1` evaluates to `true`
370    ///
371    /// This instruction is only emitted after the cfg flattening pass, and is used to annotate
372    /// instruction regions with a condition that corresponds to their position in the CFG's
373    /// if-branching structure.
374    EnableSideEffectsIf { condition: ValueId },
375
376    /// Retrieve a value from an array at the given index
377    ArrayGet { array: ValueId, index: ValueId },
378
379    /// Creates a new array with the new value at the given index. All other elements are identical
380    /// to those in the given array. This will not modify the original array unless `mutable` is
381    /// set. This flag is off by default and only enabled when optimizations determine it is safe.
382    ArraySet { array: ValueId, index: ValueId, value: ValueId, mutable: bool },
383
384    /// An instruction to increment the reference count of a value.
385    ///
386    /// This currently only has an effect in Brillig code where array sharing and copy on write is
387    /// implemented via reference counting. In ACIR code this is done with im::Vector and these
388    /// IncrementRc instructions are ignored.
389    IncrementRc { value: ValueId },
390
391    /// An instruction to decrement the reference count of a value.
392    ///
393    /// This currently only has an effect in Brillig code where array sharing and copy on write is
394    /// implemented via reference counting. In ACIR code this is done with im::Vector and these
395    /// DecrementRc instructions are ignored.
396    DecrementRc { value: ValueId },
397
398    /// Merge two values returned from opposite branches of a conditional into one.
399    ///
400    /// ```text
401    /// if then_condition {
402    ///     then_value
403    /// } else {   // else_condition = !then_condition
404    ///     else_value
405    /// }
406    /// ```
407    IfElse {
408        then_condition: ValueId,
409        then_value: ValueId,
410        else_condition: ValueId,
411        else_value: ValueId,
412    },
413
414    /// Creates a new array or vector.
415    ///
416    /// `typ` should be an array or vector type with an element type
417    /// matching each of the `elements` values' types.
418    MakeArray { elements: im::Vector<ValueId>, typ: Type },
419
420    /// A No-op instruction. These are intended to replace other instructions in a block's
421    /// instructions vector without having to move each instruction afterward.
422    ///
423    /// A No-op has no results and is always removed when Instruction::simplify is called.
424    /// When replacing another instruction, the instruction's results should always be mapped to a
425    /// new value since they will not be able to refer to their original instruction value any more.
426    Noop,
427}

430    /// Returns a binary instruction with the given operator, lhs, and rhs
431    pub(crate) fn binary(operator: BinaryOp, lhs: ValueId, rhs: ValueId) -> Instruction {
432        Instruction::Binary(Binary { lhs, operator, rhs })

435    /// Returns the type that this instruction will return.
436    pub(crate) fn result_type(&self) -> InstructionResultType {
437        match self {
438            Instruction::Binary(binary) => binary.result_type(),
439            Instruction::Cast(_, typ) => InstructionResultType::Known(Type::Numeric(*typ)),
440            Instruction::MakeArray { typ, .. } => InstructionResultType::Known(typ.clone()),
441            Instruction::Not(value)
442            | Instruction::Truncate { value, .. }
443            | Instruction::ArraySet { array: value, .. }
444            | Instruction::IfElse { then_value: value, .. } => {
445                InstructionResultType::Operand(*value)
446            }
447            Instruction::Constrain(..)
448            | Instruction::ConstrainNotEqual(..)
449            | Instruction::Store { .. }
450            | Instruction::IncrementRc { .. }
451            | Instruction::DecrementRc { .. }
452            | Instruction::RangeCheck { .. }
453            | Instruction::Noop
454            | Instruction::EnableSideEffectsIf { .. } => InstructionResultType::None,
455            Instruction::Allocate
456            | Instruction::Load { .. }
457            | Instruction::ArrayGet { .. }
458            | Instruction::Call { .. } => InstructionResultType::Unknown,
459        }
460    }

462    /// True if this instruction requires specifying the control type variables when
463    /// inserting this instruction into a DataFlowGraph.
464    pub(crate) fn requires_ctrl_typevars(&self) -> bool {
465        matches!(self.result_type(), InstructionResultType::Unknown)
466    }

468    /// If true the instruction will depend on `enable_side_effects` context during acir-gen.
469    pub(crate) fn requires_acir_gen_predicate(&self, dfg: &DataFlowGraph) -> bool {
470        match self {
471            Instruction::Binary(binary) => binary.requires_acir_gen_predicate(dfg),
472
473            Instruction::ArrayGet { array, index } => {
474                // `ArrayGet`s which read from "known good" indices from an array should not need a predicate.
475                // This extra out of bounds (OOB) check is only inserted in the ACIR runtime.
476                // Thus, in Brillig an `ArrayGet` is always a pure operation in isolation and
477                // it is expected that OOB checks are inserted separately. However, it would
478                // not be safe to separate the `ArrayGet` from the OOB constraints that precede it,
479                // because while it could read an array index, the returned data could be invalid,
480                // and fail at runtime if we tried using it in the wrong context.
481                !dfg.is_safe_index(*index, *array)
482            }
483
484            Instruction::EnableSideEffectsIf { .. }
485            | Instruction::ArraySet { .. }
486            | Instruction::ConstrainNotEqual(..) => true,
487
488            Instruction::Call { func, .. } => match dfg[*func] {
489                // All user-defined function calls are predicated during ACIR generation
490                // (both ACIR and Brillig calls unconditionally pass the side effects predicate),
491                // so this must return true regardless of purity.
492                Value::Function(_) => true,
493                Value::Intrinsic(intrinsic) => {
494                    match intrinsic {
495                        // These utilize `noirc_evaluator::acir::Context::get_flattened_index` internally
496                        // which uses the side effects predicate.
497                        Intrinsic::VectorInsert | Intrinsic::VectorRemove => true,
498                        // Technically these don't use the side effects predicate, but they fail on empty vectors,
499                        // and by pretending that they require the predicate, we can preserve any current side
500                        // effect variable in the SSA and use it to optimize out memory operations that we know
501                        // would fail, but they shouldn't because they might be disabled.
502                        Intrinsic::VectorPopFront | Intrinsic::VectorPopBack => true,
503                        // RecursiveAggregation's predicate is injected implicitly from
504                        // `current_side_effects_enabled_var` during ACIR generation, so we
505                        // must preserve the EnableSideEffectsIf that sets it.
506                        Intrinsic::BlackBox(BlackBoxFunc::RecursiveAggregation) => true,
507                        _ => false,
508                    }
509                }
510                _ => false,
511            },
512            Instruction::Cast(_, _)
513            | Instruction::Not(_)
514            | Instruction::Truncate { .. }
515            | Instruction::Constrain(_, _, _)
516            | Instruction::RangeCheck { .. }
517            | Instruction::Allocate
518            | Instruction::Load { .. }
519            | Instruction::Store { .. }
520            | Instruction::IfElse { .. }
521            | Instruction::IncrementRc { .. }
522            | Instruction::DecrementRc { .. }
523            | Instruction::Noop
524            | Instruction::MakeArray { .. } => false,
525        }
526    }

528    /// Indicates if the instruction has a side effect, ie. it can fail, or it interacts with memory.
529    pub(crate) fn has_side_effects(&self, dfg: &DataFlowGraph) -> bool {
530        use Instruction::*;
531
532        match self {
533            // These either have side-effects or interact with memory
534            EnableSideEffectsIf { .. }
535            | Allocate
536            | Load { .. }
537            | Store { .. }
538            | IncrementRc { .. }
539            | DecrementRc { .. } => true,
540
541            Call { func, .. } => match dfg[*func] {
542                Value::Intrinsic(intrinsic) => intrinsic.has_side_effects(),
543                // Functions known to be pure have no side effects.
544                // `PureWithPredicates` functions may still have side effects.
545                Value::Function(function) => dfg.purity_of(function) != Some(Purity::Pure),
546                _ => true, // Be conservative and assume other functions can have side effects.
547            },
548
549            // These can fail.
550            Constrain(..) | ConstrainNotEqual(..) | RangeCheck { .. } => true,
551
552            // This should never be side-effectual
553            MakeArray { .. } | Noop => false,
554
555            // Some binary math can overflow or underflow.
556            Binary(binary) => match binary.operator {
557                BinaryOp::Add { unchecked: false }
558                | BinaryOp::Sub { unchecked: false }
559                | BinaryOp::Mul { unchecked: false } => {
560                    let typ = dfg.type_of_value(binary.lhs);
561                    !matches!(typ, Type::Numeric(NumericType::NativeField))
562                }
563                BinaryOp::Div | BinaryOp::Mod => {
564                    // If we don't know rhs at compile time, it might be zero or -1
565                    let Some(rhs) = dfg.get_numeric_constant(binary.rhs) else {
566                        return true;
567                    };
568
569                    // Div or mod by zero is a side effect (failure)
570                    if rhs.is_zero() {
571                        return true;
572                    }
573
574                    // For signed types, division or modulo by -1 can overflow.
575                    let typ = dfg.type_of_value(binary.rhs).unwrap_numeric();
576                    let NumericType::Signed { bit_size } = typ else {
577                        return false;
578                    };
579
580                    let minus_one = IntegerConstant::Signed { value: -1, bit_size };
581                    if IntegerConstant::from_numeric_constant(rhs, typ) == Some(minus_one) {
582                        return true;
583                    }
584
585                    false
586                }
587                BinaryOp::Shl | BinaryOp::Shr => {
588                    // Bit-shifts which are known to be by a number of bits less than the bit size of the type have no side effects.
589                    dfg.get_numeric_constant(binary.rhs).is_none_or(|c| {
590                        let typ = dfg.type_of_value(binary.lhs);
591                        c >= typ.bit_size().into()
592                    })
593                }
594                BinaryOp::Add { unchecked: true }
595                | BinaryOp::Sub { unchecked: true }
596                | BinaryOp::Mul { unchecked: true }
597                | BinaryOp::Eq
598                | BinaryOp::Lt
599                | BinaryOp::And
600                | BinaryOp::Or
601                | BinaryOp::Xor => false,
602            },
603
604            // These don't have side effects
605            Cast(_, _) | Not(_) | Truncate { .. } | IfElse { .. } => false,
606
607            // `ArrayGet`s which read from "known good" indices from an array have no side effects
608            // This extra out of bounds (OOB) check is only inserted in the ACIR runtime.
609            // Thus, in Brillig an `ArrayGet` is always a pure operation in isolation and
610            // it is expected that OOB checks are inserted separately. However, it would not
611            // be safe to separate the `ArrayGet` from its corresponding OOB constraints in Brillig,
612            // as a value read from an array at an invalid index could cause failures when subsequently
613            // used in the wrong context. Since we use this information to decide whether to hoist
614            // instructions during deduplication, we consider unsafe values as potentially having
615            // indirect side effects.
616            ArrayGet { array, index } => !dfg.is_safe_index(*index, *array),
617
618            // ArraySet has side effects
619            ArraySet { .. } => true,
620        }
621    }

624    pub(crate) fn replace_values(&mut self, mapping: &ValueMapping) {
625        if !mapping.is_empty() {
626            self.map_values_mut(|value_id| mapping.get(value_id));
627        }

630    /// Maps each ValueId inside this instruction to a new ValueId, returning the new instruction.
631    /// Note that the returned instruction is fresh and will not have an assigned InstructionId
632    /// until it is manually inserted in a DataFlowGraph later.
633    pub(crate) fn map_values(&self, mut f: impl FnMut(ValueId) -> ValueId) -> Instruction {
634        match self {
635            Instruction::Binary(binary) => Instruction::Binary(Binary {
636                lhs: f(binary.lhs),
637                rhs: f(binary.rhs),
638                operator: binary.operator,
639            }),
640            Instruction::Cast(value, typ) => Instruction::Cast(f(*value), *typ),
641            Instruction::Not(value) => Instruction::Not(f(*value)),
642            Instruction::Truncate { value, bit_size, max_bit_size } => Instruction::Truncate {
643                value: f(*value),
644                bit_size: *bit_size,
645                max_bit_size: *max_bit_size,
646            },
647            Instruction::Constrain(lhs, rhs, assert_message) => {
648                // Must map the `lhs` and `rhs` first as the value `f` is moved with the closure
649                let lhs = f(*lhs);
650                let rhs = f(*rhs);
651                let assert_message = assert_message.as_ref().map(|error| match error {
652                    ConstrainError::Dynamic(selector, is_string, payload_values) => {
653                        ConstrainError::Dynamic(
654                            *selector,
655                            *is_string,
656                            payload_values.iter().map(|&value| f(value)).collect(),
657                        )
658                    }
659                    _ => error.clone(),
660                });
661                Instruction::Constrain(lhs, rhs, assert_message)
662            }
663            Instruction::ConstrainNotEqual(lhs, rhs, assert_message) => {
664                // Must map the `lhs` and `rhs` first as the value `f` is moved with the closure
665                let lhs = f(*lhs);
666                let rhs = f(*rhs);
667                let assert_message = assert_message.as_ref().map(|error| match error {
668                    ConstrainError::Dynamic(selector, is_string, payload_values) => {
669                        ConstrainError::Dynamic(
670                            *selector,
671                            *is_string,
672                            payload_values.iter().map(|&value| f(value)).collect(),
673                        )
674                    }
675                    _ => error.clone(),
676                });
677                Instruction::ConstrainNotEqual(lhs, rhs, assert_message)
678            }
679            Instruction::Call { func, arguments } => Instruction::Call {
680                func: f(*func),
681                arguments: vecmap(arguments.iter().copied(), f),
682            },
683            Instruction::Allocate => Instruction::Allocate,
684            Instruction::Load { address } => Instruction::Load { address: f(*address) },
685            Instruction::Store { address, value } => {
686                Instruction::Store { address: f(*address), value: f(*value) }
687            }
688            Instruction::EnableSideEffectsIf { condition } => {
689                Instruction::EnableSideEffectsIf { condition: f(*condition) }
690            }
691            Instruction::ArrayGet { array, index } => {
692                Instruction::ArrayGet { array: f(*array), index: f(*index) }
693            }
694            Instruction::ArraySet { array, index, value, mutable } => Instruction::ArraySet {
695                array: f(*array),
696                index: f(*index),
697                value: f(*value),
698                mutable: *mutable,
699            },
700            Instruction::IncrementRc { value } => Instruction::IncrementRc { value: f(*value) },
701            Instruction::DecrementRc { value } => Instruction::DecrementRc { value: f(*value) },
702            Instruction::RangeCheck { value, max_bit_size, assert_message } => {
703                Instruction::RangeCheck {
704                    value: f(*value),
705                    max_bit_size: *max_bit_size,
706                    assert_message: assert_message.clone(),
707                }
708            }
709            Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
710                Instruction::IfElse {
711                    then_condition: f(*then_condition),
712                    then_value: f(*then_value),
713                    else_condition: f(*else_condition),
714                    else_value: f(*else_value),
715                }
716            }
717            Instruction::MakeArray { elements, typ } => {
718                let mut elements = elements.clone();
719                im_vec_map_values_mut(&mut elements, f);
720                Instruction::MakeArray { elements, typ: typ.clone() }
721            }
722            Instruction::Noop => Instruction::Noop,
723        }
724    }

726    /// Maps each ValueId inside this instruction to a new ValueId in place.
727    pub(crate) fn map_values_mut(&mut self, mut f: impl FnMut(ValueId) -> ValueId) {
728        match self {
729            Instruction::Binary(binary) => {
730                binary.lhs = f(binary.lhs);
731                binary.rhs = f(binary.rhs);
732            }
733            Instruction::Cast(value, _) => *value = f(*value),
734            Instruction::Not(value) => *value = f(*value),
735            Instruction::Truncate { value, bit_size: _, max_bit_size: _ } => {
736                *value = f(*value);
737            }
738            Instruction::Constrain(lhs, rhs, assert_message)
739            | Instruction::ConstrainNotEqual(lhs, rhs, assert_message) => {
740                *lhs = f(*lhs);
741                *rhs = f(*rhs);
742                if let Some(ConstrainError::Dynamic(_, _, payload_values)) = assert_message {
743                    for value in payload_values {
744                        *value = f(*value);
745                    }
746                }
747            }
748            Instruction::Call { func, arguments } => {
749                *func = f(*func);
750                for argument in arguments {
751                    *argument = f(*argument);
752                }
753            }
754            Instruction::Allocate => (),
755            Instruction::Load { address } => *address = f(*address),
756            Instruction::Store { address, value } => {
757                *address = f(*address);
758                *value = f(*value);
759            }
760            Instruction::EnableSideEffectsIf { condition } => {
761                *condition = f(*condition);
762            }
763            Instruction::ArrayGet { array, index } => {
764                *array = f(*array);
765                *index = f(*index);
766            }
767            Instruction::ArraySet { array, index, value, mutable: _ } => {
768                *array = f(*array);
769                *index = f(*index);
770                *value = f(*value);
771            }
772            Instruction::IncrementRc { value } => *value = f(*value),
773            Instruction::DecrementRc { value } => {
774                *value = f(*value);
775            }
776            Instruction::RangeCheck { value, max_bit_size: _, assert_message: _ } => {
777                *value = f(*value);
778            }
779            Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
780                *then_condition = f(*then_condition);
781                *then_value = f(*then_value);
782                *else_condition = f(*else_condition);
783                *else_value = f(*else_value);
784            }
785            Instruction::MakeArray { elements, typ: _ } => {
786                im_vec_map_values_mut(elements, f);
787            }
788            Instruction::Noop => (),
789        }
790    }
791
792    /// Applies a function to each input value this instruction holds.
793    pub(crate) fn for_each_value<T>(&self, mut f: impl FnMut(ValueId) -> T) {
794        match self {
795            Instruction::Binary(binary) => {
796                f(binary.lhs);
797                f(binary.rhs);
798            }
799            Instruction::Call { func, arguments } => {
800                f(*func);
801                for argument in arguments {
802                    f(*argument);
803                }
804            }
805            Instruction::Cast(value, _)
806            | Instruction::Not(value)
807            | Instruction::Truncate { value, .. }
808            | Instruction::Load { address: value } => {
809                f(*value);
810            }
811            Instruction::Constrain(lhs, rhs, assert_error)
812            | Instruction::ConstrainNotEqual(lhs, rhs, assert_error) => {
813                f(*lhs);
814                f(*rhs);
815                if let Some(ConstrainError::Dynamic(_, _, values)) = assert_error.as_ref() {
816                    values.iter().for_each(|&val| {
817                        f(val);
818                    });
819                }
820            }
821
822            Instruction::Store { address, value } => {
823                f(*address);
824                f(*value);
825            }
826            Instruction::Allocate => (),
827            Instruction::ArrayGet { array, index } => {
828                f(*array);
829                f(*index);
830            }
831            Instruction::ArraySet { array, index, value, mutable: _ } => {
832                f(*array);
833                f(*index);
834                f(*value);
835            }
836            Instruction::EnableSideEffectsIf { condition } => {
837                f(*condition);
838            }
839            Instruction::IncrementRc { value }
840            | Instruction::DecrementRc { value }
841            | Instruction::RangeCheck { value, .. } => {
842                f(*value);
843            }
844            Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
845                f(*then_condition);
846                f(*then_value);
847                f(*else_condition);
848                f(*else_value);
849            }
850            Instruction::MakeArray { elements, typ: _ } => {
851                for element in elements {
852                    f(*element);
853                }
854            }
855            Instruction::Noop => (),
856        }
857    }

860/// Determines whether an ArrayGet or ArraySet index has been shifted by a given value.
861/// Offsets are set during `crate::ssa::opt::brillig_array_gets` for brillig arrays
862/// and vectors with constant indices.
863#[derive(Debug, PartialEq, Eq, Hash, Copy, Clone, Serialize, Deserialize)]
864pub enum ArrayOffset {
865    None,
866    Array,
867    Vector,
868}
869
870impl ArrayOffset {
871    pub fn from_u32(value: u32) -> Option<Self> {
872        match value {
873            0 => Some(Self::None),
874            1 => Some(Self::Array),
875            3 => Some(Self::Vector),
876            _ => None,
877        }
878    }
879
880    pub fn to_u32(self) -> u32 {
881        match self {
882            Self::None => 0,
883            // Arrays in brillig are represented as [RC, ...items]
884            Self::Array => 1,
885            // Vectors in brillig are represented as [RC, Size, Capacity, ...items]
886            Self::Vector => 3,
887        }
888    }

891impl Binary {
892    pub(crate) fn requires_acir_gen_predicate(&self, dfg: &DataFlowGraph) -> bool {
893        match self.operator {
894            BinaryOp::Add { unchecked: false }
895            | BinaryOp::Sub { unchecked: false }
896            | BinaryOp::Mul { unchecked: false } => {
897                match dfg.type_of_value(self.rhs).unwrap_numeric() {
898                    NumericType::NativeField => false,
899                    // Some binary math can overflow or underflow for non-field types.
900                    NumericType::Unsigned { .. } => true,
901                    // However, we assume that signed types should have already been expanded using unsigned operations.
902                    NumericType::Signed { .. } => {
903                        unreachable!("signed instructions should have been already expanded")
904                    }
905                }
906            }
907            BinaryOp::Shl | BinaryOp::Shr => {
908                // Bit-shifts which are known to be by a number of bits less than the bit size of the type have no side effects.
909                dfg.get_numeric_constant(self.rhs).is_none_or(|c| {
910                    let typ = dfg.type_of_value(self.lhs);
911                    c >= typ.bit_size().into()
912                })
913            }
914            BinaryOp::Div | BinaryOp::Mod => true,
915            BinaryOp::Add { unchecked: true }
916            | BinaryOp::Sub { unchecked: true }
917            | BinaryOp::Mul { unchecked: true }
918            | BinaryOp::Eq
919            | BinaryOp::Lt
920            | BinaryOp::And
921            | BinaryOp::Or
922            | BinaryOp::Xor => false,
923        }
924    }
925}

927#[derive(Debug, PartialEq, Eq, Hash, Clone)]
928pub enum ErrorType {
929    String(String),
930    Dynamic(HirType),
931}
932
933impl ErrorType {
934    /// Hash the error type to get a unique selector for it.
935    pub fn selector(&self) -> ErrorSelector {
936        struct U64(pub u64);
937
938        impl FromStableHash for U64 {
939            type Hash = SipHasher128Hash;
940
941            fn from(hash: Self::Hash) -> Self {
942                Self(hash.0[0])
943            }
944        }
945
946        // We explicitly do not use `rustc-hash` here as we require hashes to be stable across 32- and 64-bit architectures.
947        let mut hasher =
948            rustc_stable_hash::StableHasher::<rustc_stable_hash::hashers::SipHasher128>::new();
949        self.hash(&mut hasher);
950        let hash = hasher.finish::<U64>();
951        ErrorSelector::new(hash.0)
952    }
953}

976/// The possible return values for Instruction::return_types
977pub(crate) enum InstructionResultType {
978    /// The result type of this instruction matches that of this operand
979    Operand(ValueId),
980
981    /// The result type of this instruction is known to be this type - independent of its operands.
982    Known(Type),
983
984    /// The result type of this function is unknown and separate from its operand types.
985    /// This occurs for function calls and load operations.
986    Unknown,
987
988    /// This instruction does not return any results.
989    None,
990}

992/// These are operations which can exit a basic block
993/// ie control flow type operations
994///
995/// Since our IR needs to be in SSA form, it makes sense
996/// to split up instructions like this, as we are sure that these instructions
997/// will not be in the list of instructions for a basic block.
998#[derive(Debug, PartialEq, Eq, Hash, Clone, Serialize, Deserialize)]
999pub(crate) enum TerminatorInstruction {
1000    /// Control flow
1001    ///
1002    /// Jump If
1003    ///
1004    /// If the condition is true: jump to the specified `then_destination`.
1005    /// Otherwise, jump to the specified `else_destination`.
1006    JmpIf {
1007        condition: ValueId,
1008        then_destination: BasicBlockId,
1009        else_destination: BasicBlockId,
1010        call_stack: CallStackId,
1011    },
1012
1013    /// Unconditional Jump
1014    ///
1015    /// Jumps to specified `destination` with `arguments`.
1016    /// The CallStack here is expected to be used to issue an error when the start range of
1017    /// a for loop cannot be deduced at compile-time.
1018    Jmp { destination: BasicBlockId, arguments: Vec<ValueId>, call_stack: CallStackId },
1019
1020    /// Return from the current function with the given return values.
1021    ///
1022    /// All finished functions should have exactly 1 return instruction.
1023    /// Functions with early returns should instead be structured to
1024    /// unconditionally jump to a single exit block with the return values
1025    /// as the block arguments. Then the exit block can terminate in a return
1026    /// instruction returning these values.
1027    Return { return_values: Vec<ValueId>, call_stack: CallStackId },
1028
1029    /// A terminator that will never be reached because an instruction in its block
1030    /// will always produce an assertion failure.
1031    Unreachable { call_stack: CallStackId },
1032}
1033
1034impl TerminatorInstruction {
1035    /// Mutate each ValueId to a new ValueId using the given mapping function
1036    pub(crate) fn map_values_mut(&mut self, mut f: impl FnMut(ValueId) -> ValueId) {
1037        use TerminatorInstruction::*;
1038        match self {
1039            JmpIf { condition, .. } => {
1040                *condition = f(*condition);
1041            }
1042            Jmp { arguments, .. } => {
1043                for argument in arguments {
1044                    *argument = f(*argument);
1045                }
1046            }
1047            Return { return_values, .. } => {
1048                for return_value in return_values {
1049                    *return_value = f(*return_value);
1050                }
1051            }
1052            Unreachable { .. } => (),
1053        }
1054    }
1055
1056    /// Apply a function to each value
1057    pub(crate) fn for_each_value<T>(&self, mut f: impl FnMut(ValueId) -> T) {
1058        use TerminatorInstruction::*;
1059        match self {
1060            JmpIf { condition, .. } => {
1061                f(*condition);
1062            }
1063            Jmp { arguments, .. } => {
1064                for argument in arguments {
1065                    f(*argument);
1066                }
1067            }
1068            Return { return_values, .. } => {
1069                for return_value in return_values {
1070                    f(*return_value);
1071                }
1072            }
1073            Unreachable { .. } => (),
1074        }
1075    }
1076
1077    /// Apply a function to each value along with its index
1078    pub(crate) fn for_eachi_value<T>(&self, mut f: impl FnMut(usize, ValueId) -> T) {
1079        use TerminatorInstruction::*;
1080        match self {
1081            JmpIf { condition, .. } => {
1082                f(0, *condition);
1083            }
1084            Jmp { arguments, .. } => {
1085                for (index, argument) in arguments.iter().enumerate() {
1086                    f(index, *argument);
1087                }
1088            }
1089            Return { return_values, .. } => {
1090                for (index, return_value) in return_values.iter().enumerate() {
1091                    f(index, *return_value);
1092                }
1093            }
1094            Unreachable { .. } => (),
1095        }
1096    }
1097
1098    /// Mutate each BlockId to a new BlockId specified by the given mapping function.
1099    pub(crate) fn mutate_blocks(&mut self, mut f: impl FnMut(BasicBlockId) -> BasicBlockId) {
1100        use TerminatorInstruction::*;
1101        match self {
1102            JmpIf { then_destination, else_destination, .. } => {
1103                *then_destination = f(*then_destination);
1104                *else_destination = f(*else_destination);
1105            }
1106            Jmp { destination, .. } => {
1107                *destination = f(*destination);
1108            }
1109            Return { .. } | Unreachable { .. } => (),
1110        }
1111    }
1112
1113    pub(crate) fn call_stack(&self) -> CallStackId {
1114        match self {
1115            TerminatorInstruction::JmpIf { call_stack, .. }
1116            | TerminatorInstruction::Jmp { call_stack, .. }
1117            | TerminatorInstruction::Return { call_stack, .. }
1118            | TerminatorInstruction::Unreachable { call_stack } => *call_stack,
1119        }
1120    }
1121
1122    pub(crate) fn set_call_stack(&mut self, new_call_stack: CallStackId) {
1123        match self {
1124            TerminatorInstruction::JmpIf { call_stack, .. }
1125            | TerminatorInstruction::Jmp { call_stack, .. }
1126            | TerminatorInstruction::Return { call_stack, .. }
1127            | TerminatorInstruction::Unreachable { call_stack } => *call_stack = new_call_stack,
1128        }
1129    }
1130}

1132/// Try to avoid mutation until we know something changed, to take advantage of
1133/// structural sharing, and avoid needlessly calling `Arc::make_mut` which clones
1134/// the content and increases memory use by allocating more pointers on the heap.
1135fn im_vec_map_values_mut<T, F>(xs: &mut im::Vector<T>, mut f: F)
1136where
1137    T: Copy + PartialEq,
1138    F: FnMut(T) -> T,
1139{
1140    // Even `xs.iter_mut()` calls `get_mut` on each element, regardless of whether there is actual mutation.
1141    // If we go index-by-index, get the item, put it back only if it changed, then we can avoid
1142    // allocating memory unless we need to, however we incur O(n * log(n)) complexity.
1143    // Collecting changes first and then updating only those positions proved to be the
1144    // fastest among some alternatives that didn't sacrifice memory for speed or vice versa.
1145    let mut changes = Vec::new();
1146    for (i, x) in xs.iter().enumerate() {
1147        let y = f(*x);
1148        if *x != y {
1149            changes.push((i, y));
1150        }
1151    }
1152    if changes.is_empty() {
1153        return;
1154    }
1155    // Using `Focus` allows us to only make mutable what is needed,
1156    // and should be faster for batches than indexing individual items.
1157    let mut focus = xs.focus_mut();
1158    for (i, y) in changes {
1159        focus.set(i, y);
1160    }
