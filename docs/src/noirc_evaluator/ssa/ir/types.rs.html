17/// A numeric type in the Intermediate representation
18/// Note: we class NativeField as a numeric type
19/// though we also apply limitations to it, such as not
20/// being able to compare two native fields, whereas this is
21/// something that you can do with a signed/unsigned integer.
22///
23/// Fields do not have a notion of ordering, so this distinction
24/// is reasonable.
25#[derive(Copy, Clone, Debug, PartialEq, Eq, Hash, Ord, PartialOrd, Serialize, Deserialize)]
26pub enum NumericType {
27    Signed { bit_size: u32 },
28    Unsigned { bit_size: u32 },
29    NativeField,
30}
31
32impl NumericType {
33    /// Returns the bit size of the provided numeric type.
34    pub(crate) fn bit_size<F: AcirField>(self: &NumericType) -> u32 {
35        match self {
36            NumericType::NativeField => F::max_num_bits(),
37            NumericType::Unsigned { bit_size } | NumericType::Signed { bit_size } => *bit_size,
38        }
39    }
40
41    /// Creates a NumericType::Signed type
42    pub(crate) fn signed(bit_size: u32) -> NumericType {
43        NumericType::Signed { bit_size }
44    }
45
46    /// Creates a NumericType::Unsigned type
47    pub(crate) fn unsigned(bit_size: u32) -> NumericType {
48        NumericType::Unsigned { bit_size }
49    }
50
51    /// Creates the u1 type
52    pub(crate) fn bool() -> NumericType {
53        NumericType::Unsigned { bit_size: 1 }
54    }
55
56    /// Creates the char type, represented as u8.
57    pub(crate) fn char() -> NumericType {
58        NumericType::Unsigned { bit_size: 8 }
59    }
60
61    /// Creates the type of an array's length.
62    pub(crate) fn length_type() -> NumericType {
63        NumericType::Unsigned { bit_size: SSA_WORD_SIZE }
64    }
65
66    /// Returns None if the given Field value is within the numeric limits
67    /// for the current NumericType. Otherwise returns a string describing
68    /// the limits, as a range.
69    pub(crate) fn value_is_outside_limits(self, value: SignedField) -> Option<String> {
70        match self {
71            NumericType::Unsigned { bit_size } => {
72                let max = if bit_size == 128 { u128::MAX } else { 2u128.pow(bit_size) - 1 };
73                if value.is_negative() || value > SignedField::positive(max) {
74                    Some(format!("0..={max}"))
75                } else {
76                    None
77                }
78            }
79            NumericType::Signed { bit_size } => {
80                let min = 2u128.pow(bit_size - 1);
81                let max = 2u128.pow(bit_size - 1) - 1;
82                if value > SignedField::positive(max) || value < SignedField::negative(min) {
83                    Some(format!("-{min}..={max}"))
84                } else {
85                    None
86                }
87            }
88            NumericType::NativeField => None,
89        }
90    }
91
92    pub(crate) fn is_field(&self) -> bool {
93        matches!(self, NumericType::NativeField)
94    }
95
96    pub(crate) fn is_unsigned(&self) -> bool {
97        matches!(self, NumericType::Unsigned { .. })
98    }
99
100    pub(crate) fn is_signed(&self) -> bool {
101        matches!(self, NumericType::Signed { .. })
102    }
103
104    pub(crate) fn max_value(&self) -> Result<FieldElement, String> {
105        match self {
106            NumericType::Unsigned { bit_size } => match bit_size {
107                bit_size if *bit_size > 128 => {
108                    Err("Cannot get max value for unsigned type: bit size is greater than 128"
109                        .to_string())
110                }
111                128 => Ok(FieldElement::from(u128::MAX)),
112                _ => Ok(FieldElement::from(2u128.pow(*bit_size) - 1)),
113            },
114            other => Err(format!("Cannot get max value for type: {other}")),
115        }
116    }
117}

145/// All types representable in the IR.
146#[derive(Clone, Debug, PartialEq, Eq, Hash, Ord, PartialOrd, Serialize, Deserialize)]
147pub enum Type {
148    /// Represents numeric types in the IR, including field elements
149    Numeric(NumericType),
150
151    /// A reference to some value, such as an array
152    Reference(Arc<Type>),
153
154    /// An immutable array value with the given element type and length
155    Array(Arc<CompositeType>, SemanticLength),
156
157    /// An immutable vector value with a given element type
158    Vector(Arc<CompositeType>),
159
160    /// A function that may be called directly
161    Function,
162}
163
164impl Type {
165    /// Returns whether the `Type` represents an unsigned numeric type.
166    pub fn is_unsigned(&self) -> bool {
167        matches!(self, Type::Numeric(NumericType::Unsigned { .. }))
168    }
169
170    /// Returns whether the `Type` represents an signed numeric type.
171    pub fn is_signed(&self) -> bool {
172        matches!(self, Type::Numeric(NumericType::Signed { .. }))
173    }
174
175    /// Create a new signed integer type with the given amount of bits.
176    pub fn signed(bit_size: u32) -> Type {
177        Type::Numeric(NumericType::Signed { bit_size })
178    }
179
180    /// Create a new unsigned integer type with the given amount of bits.
181    pub fn unsigned(bit_size: u32) -> Type {
182        Type::Numeric(NumericType::Unsigned { bit_size })
183    }
184
185    /// Creates the boolean type, represented as u1.
186    pub fn bool() -> Type {
187        Type::unsigned(1)
188    }
189
190    /// Creates the char type, represented as u8.
191    pub fn char() -> Type {
192        Type::unsigned(8)
193    }
194
195    /// Creates the `str<N>` type, of the given length N
196    pub fn str(length: u32) -> Type {
197        Type::Array(Arc::new(vec![Type::char()]), SemanticLength(length))
198    }
199
200    /// Creates the native field type.
201    pub fn field() -> Type {
202        Type::Numeric(NumericType::NativeField)
203    }
204
205    /// Creates the type of an array's length.
206    pub fn length_type() -> Type {
207        Type::unsigned(SSA_WORD_SIZE)
208    }
209
210    /// True if this type is a numeric primitive type.
211    pub(crate) fn is_numeric(&self) -> bool {
212        matches!(self, Type::Numeric(_))
213    }
214
215    /// Returns the inner NumericType if this is one, or panics otherwise
216    pub(crate) fn unwrap_numeric(&self) -> NumericType {
217        match self {
218            Type::Numeric(numeric) => *numeric,
219            other => panic!("Expected NumericType, found {other}"),
220        }
221    }
222
223    /// Returns the bit size of the provided numeric type.
224    ///
225    /// # Panics
226    ///
227    /// Panics if `self` is not a [`Type::Numeric`]
228    pub(crate) fn bit_size(&self) -> u32 {
229        match self {
230            Type::Numeric(numeric_type) => numeric_type.bit_size::<FieldElement>(),
231            other => panic!("bit_size: Expected numeric type, found {other}"),
232        }
233    }
234
235    /// Returns the size of the element type for this array/vector.
236    /// The size of a type is defined as representing how many Fields are needed
237    /// to represent the type. This is 1 for every primitive type, and is the number of fields
238    /// for any flattened tuple type.
239    ///
240    /// Equivalent to `self.element_types().len()`.
241    ///
242    /// Panics if `self` is not a [`Type::Array`] or [`Type::Vector`].
243    pub(crate) fn element_size(&self) -> ElementTypesLength {
244        match self {
245            Type::Array(elements, _) | Type::Vector(elements) => {
246                ElementTypesLength(assert_u32(elements.len()))
247            }
248            other => panic!("element_size: Expected array or vector, found {other}"),
249        }
250    }
251
252    /// Return the types of items in this array/vector.
253    ///
254    /// Panics if `self` is not a [`Type::Array`] or [`Type::Vector`].
255    pub(crate) fn element_types(&self) -> Arc<Vec<Type>> {
256        match self {
257            Type::Array(element_types, _) | Type::Vector(element_types) => element_types.clone(),
258            other => panic!("element_types: Expected array or vector, found {other}"),
259        }
260    }
261
262    pub(crate) fn contains_vector_element(&self) -> bool {
263        match self {
264            Type::Array(elements, _) => {
265                elements.iter().any(|element| element.contains_vector_element())
266            }
267            Type::Vector(_) => true,
268            Type::Numeric(_) => false,
269            Type::Reference(element) => element.contains_vector_element(),
270            Type::Function => false,
271        }
272    }
273
274    /// Returns the flattened size of a Type.
275    ///
276    /// The flattened type is mostly useful in ACIR, where nested arrays are also flattened,
277    /// as opposed to SSA, where only tuples get flattened into the array they are in,
278    /// but nested arrays appear as a value ID.
279    pub(crate) fn flattened_size(&self) -> FlattenedLength {
280        match self {
281            Type::Array(elements, len) => {
282                let elements_flattened_length: FlattenedLength =
283                    elements.iter().map(|elem| elem.flattened_size()).sum();
284                ElementsFlattenedLength::from(elements_flattened_length) * *len
285            }
286            Type::Vector(_) => {
287                unimplemented!("ICE: cannot fetch flattened vector size");
288            }
289            _ => FlattenedLength(1),
290        }
291    }
292
293    /// True if this type is an array (or vector)
294    pub(crate) fn is_array(&self) -> bool {
295        matches!(self, Type::Array(_, _) | Type::Vector(_))
296    }
297
298    pub(crate) fn is_nested_vector(&self) -> bool {
299        if let Type::Vector(element_types) | Type::Array(element_types, _) = self {
300            element_types.as_ref().iter().any(|typ| typ.contains_vector_element())
301        } else {
302            false
303        }
304    }
305
306    /// True if this type is an array (or vector) or internally contains an array (or vector)
307    pub(crate) fn contains_an_array(&self) -> bool {
308        match self {
309            Type::Numeric(_) | Type::Function => false,
310            Type::Array(_, _) | Type::Vector(_) => true,
311            Type::Reference(element) => element.contains_an_array(),
312        }
313    }
314
315    pub(crate) fn first(&self) -> Type {
316        match self {
317            Type::Numeric(_) | Type::Function => self.clone(),
318            Type::Reference(typ) => typ.first(),
319            Type::Vector(element_types) | Type::Array(element_types, _) => element_types[0].first(),
320        }
321    }
322
323    /// True if this is a reference type or if it is a composite type which contains a reference.
324    pub(crate) fn contains_reference(&self) -> bool {
325        match self {
326            Type::Reference(_) => true,
327            Type::Numeric(_) | Type::Function => false,
328            Type::Array(elements, _) | Type::Vector(elements) => {
329                elements.iter().any(|elem| elem.contains_reference())
330            }
331        }
332    }
333
334    /// True if this is a function type or if it is a composite type which contains a function.
335    pub(crate) fn contains_function(&self) -> bool {
336        match self {
337            Type::Reference(element_type) => element_type.contains_function(),
338            Type::Function => true,
339            Type::Numeric(_) => false,
340            Type::Array(elements, _) | Type::Vector(elements) => {
341                elements.iter().any(|elem| elem.contains_function())
342            }
343        }
344    }
345}

352impl std::fmt::Display for Type {
353    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
354        match self {
355            Type::Numeric(numeric) => numeric.fmt(f),
356            Type::Reference(element) => write!(f, "&mut {element}"),
357            Type::Array(element, length) => {
358                let elements = vecmap(element.iter(), |element| element.to_string());
359                if elements.len() == 1 {
360                    write!(f, "[{}; {length}]", elements.join(", "))
361                } else {
362                    write!(f, "[({}); {length}]", elements.join(", "))
363                }
364            }
365            Type::Vector(element) => {
366                let elements = vecmap(element.iter(), |element| element.to_string());
367                if elements.len() == 1 {
368                    write!(f, "[{}]", elements.join(", "))
369                } else {
370                    write!(f, "[({})]", elements.join(", "))
371                }
372            }
373            Type::Function => write!(f, "function"),
374        }
375    }
376}

378impl std::fmt::Display for NumericType {
379    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
380        match self {
381            NumericType::Signed { bit_size } => write!(f, "i{bit_size}"),
382            NumericType::Unsigned { bit_size } => write!(f, "u{bit_size}"),
383            NumericType::NativeField => write!(f, "Field"),
384        }
385    }
