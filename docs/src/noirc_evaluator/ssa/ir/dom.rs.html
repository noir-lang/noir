1//! The dominator tree of a function, represented as a hash map of each reachable block id to its
2//! immediate dominator.
3//!
4//! Dominator trees are useful for tasks such as identifying back-edges in loop analysis or
5//! calculating dominance frontiers.
6
7use std::cmp::Ordering;
8
9use super::{
10    basic_block::BasicBlockId, cfg::ControlFlowGraph, function::Function, post_order::PostOrder,
11};
12use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};
13
14/// Dominator tree node. We keep one of these per reachable block.
15#[derive(Clone, Default)]
16struct DominatorTreeNode {
17    /// The block's idx in the control flow graph's reverse post-order
18    reverse_post_order_idx: u32,
19
20    /// The block that immediately dominated that of the node in question.
21    ///
22    /// This will be None for the entry block, which has no immediate dominator.
23    immediate_dominator: Option<BasicBlockId>,
24}
25
26impl DominatorTreeNode {
27    /// Updates the immediate dominator estimate, returning true if it has changed.
28    ///
29    /// This is used internally as a shorthand during `compute_dominator_tree`.
30    pub(self) fn update_estimate(&mut self, immediate_dominator: BasicBlockId) -> bool {
31        let immediate_dominator = Some(immediate_dominator);
32        if self.immediate_dominator == immediate_dominator {
33            false
34        } else {
35            self.immediate_dominator = immediate_dominator;
36            true
37        }
38    }
39}
40
41/// The dominator tree for a single function.
42#[derive(Default)]
43pub(crate) struct DominatorTree {
44    /// The nodes of the dominator tree
45    ///
46    /// After dominator tree computation has complete, this will contain a node for every
47    /// reachable block, and no nodes for unreachable blocks.
48    nodes: HashMap<BasicBlockId, DominatorTreeNode>,
49
50    /// Subsequent calls to `dominates` are cached to speed up access
51    cache: HashMap<(BasicBlockId, BasicBlockId), bool>,
52}
53
54/// Methods for querying the dominator tree.
55impl DominatorTree {
56    /// Is `block_id` reachable from the entry block?
57    pub(crate) fn is_reachable(&self, block_id: BasicBlockId) -> bool {
58        self.nodes.contains_key(&block_id)
59    }
60
61    /// Returns the immediate dominator of `block_id`.
62    ///
63    /// A block is said to *dominate* `block_id` if all control flow paths from the function
64    /// entry to `block_id` must go through the block.
65    ///
66    /// The *immediate dominator* is the dominator that is closest to `block_id`. All other
67    /// dominators also dominate the immediate dominator.
68    ///
69    /// This returns `None` if `block_id` is not reachable from the entry block, or if it is the
70    /// entry block which has no dominators.
71    pub(crate) fn immediate_dominator(&self, block_id: BasicBlockId) -> Option<BasicBlockId> {
72        self.nodes.get(&block_id).and_then(|node| node.immediate_dominator)
73    }
74
75    /// Compare two blocks relative to the reverse post-order.
76    pub(crate) fn reverse_post_order_cmp(&self, a: BasicBlockId, b: BasicBlockId) -> Ordering {
77        match (self.reverse_post_order_idx(a), self.reverse_post_order_idx(b)) {
78            (Some(a), Some(b)) => a.cmp(&b),
79            _ => unreachable!("Post order for unreachable block is undefined"),
80        }
81    }
82
83    /// Position in the Reverse Post-Order.
84    pub(crate) fn reverse_post_order_idx(&self, block_id: BasicBlockId) -> Option<u32> {
85        self.nodes.get(&block_id).map(|n| n.reverse_post_order_idx)
86    }
87
88    /// Returns `true` if `block_a_id` dominates `block_b_id`.
89    ///
90    /// This means that every control-flow path from the function entry to `block_b_id` must go
91    /// through `block_a_id`.
92    ///
93    /// This function panics if either of the blocks are unreachable.
94    ///
95    /// A block is considered to dominate itself.
96    pub(crate) fn dominates(&mut self, block_a_id: BasicBlockId, block_b_id: BasicBlockId) -> bool {
97        if let Some(res) = self.cache.get(&(block_a_id, block_b_id)) {
98            return *res;
99        }
100
101        let result = self.dominates_helper(block_a_id, block_b_id);
102        self.cache.insert((block_a_id, block_b_id), result);
103        result
104    }
105
106    pub(crate) fn dominates_helper(
107        &self,
108        block_a_id: BasicBlockId,
109        mut block_b_id: BasicBlockId,
110    ) -> bool {
111        // Walk up the dominator tree from "b" until we encounter or pass "a". Doing the
112        // comparison on the reverse post-order may allow to test whether we have passed "a"
113        // without waiting until we reach the root of the tree.
114        loop {
115            match self.reverse_post_order_cmp(block_a_id, block_b_id) {
116                Ordering::Less => {
117                    block_b_id = match self.immediate_dominator(block_b_id) {
118                        Some(immediate_dominator) => immediate_dominator,
119                        None => return false, // a is unreachable, so we climbed past the entry
120                    }
121                }
122                Ordering::Greater => return false,
123                Ordering::Equal => return true,
124            }
125        }
126    }
127
128    /// Walk up the dominator tree until we find a block for which `f` returns `Some` value.
129    /// Otherwise return `None` when we reach the top.
130    ///
131    /// Similar to `Iterator::filter_map` but only returns the first hit.
132    pub(crate) fn find_map_dominator<T>(
133        &self,
134        mut block_id: BasicBlockId,
135        f: impl Fn(BasicBlockId) -> Option<T>,
136    ) -> Option<T> {
137        if !self.is_reachable(block_id) {
138            return None;
139        }
140        loop {
141            if let Some(value) = f(block_id) {
142                return Some(value);
143            }
144            block_id = self.immediate_dominator(block_id)?;
145        }
146    }

148    /// Allocate and compute a dominator tree from a pre-computed control flow graph and
149    /// post-order counterpart.
150    ///
151    /// This method should be used for when we want to compute a post-dominator tree.
152    /// A post-dominator tree just expects the control flow graph to be reversed.
153    pub(crate) fn with_cfg_and_post_order(cfg: &ControlFlowGraph, post_order: &PostOrder) -> Self {
154        let mut dom_tree = DominatorTree { nodes: HashMap::default(), cache: HashMap::default() };
155        dom_tree.compute_dominator_tree(cfg, post_order);
156        dom_tree
157    }
158
159    /// Allocate and compute a dominator tree for the given function.
160    ///
161    /// This approach computes the control flow graph and post-order internally and then
162    /// discards them. If either should be retained reuse it is better to instead pre-compute them
163    /// and build the dominator tree with `DominatorTree::with_cfg_and_post_order`.
164    pub(crate) fn with_function(func: &Function) -> Self {
165        let cfg = ControlFlowGraph::with_function(func);
166        let post_order = PostOrder::with_function(func);
167        Self::with_cfg_and_post_order(&cfg, &post_order)
168    }
169
170    /// Allocate and compute a post-dominator tree for the given function.
171    ///
172    /// This approach computes the reversed control flow graph and post-order internally and then
173    /// discards them. If either should be retained for reuse, it is better to instead pre-compute them
174    /// and build the dominator tree with `DominatorTree::with_cfg_and_post_order`.
175    #[cfg(test)]
176    pub(crate) fn with_function_post_dom(func: &Function) -> Self {
177        let reversed_cfg = ControlFlowGraph::with_function(func).reverse();
178        let post_order = PostOrder::with_cfg(&reversed_cfg);
179        Self::with_cfg_and_post_order(&reversed_cfg, &post_order)
180    }

182    /// Build a dominator tree from a control flow graph using Keith D. Cooper's
183    /// "Simple, Fast Dominator Algorithm."
184    fn compute_dominator_tree(&mut self, cfg: &ControlFlowGraph, post_order: &PostOrder) {
185        // We'll be iterating over a reverse post-order of the CFG, skipping the entry block.
186        let Some((entry_block_id, entry_free_post_order)) = post_order.as_slice().split_last()
187        else {
188            return;
189        };
190
191        // Do a first pass where we assign reverse post-order indices to all reachable nodes. The
192        // entry block will be the only node with no immediate dominator.
193        self.nodes.insert(
194            *entry_block_id,
195            DominatorTreeNode { reverse_post_order_idx: 0, immediate_dominator: None },
196        );
197        for (i, &block_id) in entry_free_post_order.iter().rev().enumerate() {
198            // Indices have been displaced by 1 by the removal of the entry node
199            let reverse_post_order_idx = i as u32 + 1;
200
201            // Due to the nature of the post-order traversal, every node we visit will have at
202            // least one predecessor that has previously been assigned during this loop.
203            let immediate_dominator = self.compute_immediate_dominator(block_id, cfg);
204            self.nodes.insert(
205                block_id,
206                DominatorTreeNode {
207                    immediate_dominator: Some(immediate_dominator),
208                    reverse_post_order_idx,
209                },
210            );
211        }
212
213        // Now that we have reverse post-order indices for everything and initial immediate
214        // dominator estimates, iterate until convergence.
215        //
216        // If the function is free of irreducible control flow, this will exit after one iteration.
217        let mut changed = true;
218        while changed {
219            changed = false;
220            for &block_id in entry_free_post_order.iter().rev() {
221                let immediate_dominator = self.compute_immediate_dominator(block_id, cfg);
222                changed = self
223                    .nodes
224                    .get_mut(&block_id)
225                    .expect("Assigned in first pass")
226                    .update_estimate(immediate_dominator);
227            }
228        }
229    }
230
231    // Compute the immediate dominator for `block_id` using the pre-calculate immediate dominators
232    // of reachable nodes.
233    fn compute_immediate_dominator(
234        &self,
235        block_id: BasicBlockId,
236        cfg: &ControlFlowGraph,
237    ) -> BasicBlockId {
238        // Get an iterator with just the reachable, already visited predecessors to `block_id`.
239        // Note that during the first pass `node` was pre-populated with all reachable blocks.
240        let mut reachable_predecessors =
241            cfg.predecessors(block_id).filter(|pred_id| self.nodes.contains_key(pred_id));
242
243        // This function isn't called on unreachable blocks or the entry block, so the reverse
244        // post-order will contain at least one predecessor to this block.
245        let mut immediate_dominator =
246            reachable_predecessors.next().expect("block node must have one reachable predecessor");
247
248        for predecessor in reachable_predecessors {
249            immediate_dominator = self.common_dominator(immediate_dominator, predecessor);
250        }
251
252        immediate_dominator
253    }

255    /// Compute the common dominator of two basic blocks.
256    ///
257    /// Both basic blocks are assumed to be reachable.
258    pub(crate) fn common_dominator(
259        &self,
260        mut block_a_id: BasicBlockId,
261        mut block_b_id: BasicBlockId,
262    ) -> BasicBlockId {
263        loop {
264            match self.reverse_post_order_cmp(block_a_id, block_b_id) {
265                Ordering::Less => {
266                    // "a" comes before "b" in the reverse post-order. Move "b" up.
267                    block_b_id = self.nodes[&block_b_id]
268                        .immediate_dominator
269                        .expect("Unreachable basic block?");
270                }
271                Ordering::Greater => {
272                    // "b" comes before "a" in the reverse post-order. Move "a" up.
273                    block_a_id = self.nodes[&block_a_id]
274                        .immediate_dominator
275                        .expect("Unreachable basic block?");
276                }
277                Ordering::Equal => break,
278            }
279        }
280
281        assert_eq!(block_a_id, block_b_id, "Unreachable block passed to common_dominator?");
282        block_a_id
283    }

285    /// Computes the dominance frontier for all blocks in the dominator tree.
286    ///
287    /// The Dominance Frontier of a basic block X is the set of all blocks that are immediate
288    /// successors to blocks dominated by X, but which aren’t themselves strictly dominated by X.
289    /// It is the set of blocks that are not dominated X, and which are “first reached” on paths from X.
290    ///
291    /// For example in the following CFG the DF of B is {E}, because B dominates {C},
292    /// but it's just one edge away from dominating E, as there is another path to E through D.
293    /// ```text
294    ///    A
295    ///   / \
296    ///  B   D
297    ///  |   |
298    ///  C   |
299    ///   \ /
300    ///    E
301    /// ```
302    ///
303    /// This method uses the algorithm specified in Cooper, Keith D. et al. “A Simple, Fast Dominance Algorithm.” (1999).
304    /// As referenced in the paper a dominance frontier is the set of all CFG nodes, y, such that
305    /// b dominates a predecessor of y but does not strictly dominate y.
306    ///
307    /// This method expects the appropriate CFG depending on whether we are operating over
308    /// a dominator tree (standard CFG) or a post-dominator tree (reversed CFG).
309    /// Calling this method on a dominator tree will return a function's dominance frontiers,
310    /// while on a post-dominator tree the method will return the function's reverse (or post) dominance frontiers.
311    pub(crate) fn compute_dominance_frontiers(
312        &mut self,
313        cfg: &ControlFlowGraph,
314    ) -> HashMap<BasicBlockId, HashSet<BasicBlockId>> {
315        let mut dominance_frontiers: HashMap<BasicBlockId, HashSet<BasicBlockId>> =
316            HashMap::default();
317
318        let nodes = self.nodes.keys().copied().collect::<Vec<_>>();
319        // Find out about each block which dominance frontiers they belong to, if any.
320        for block_id in nodes {
321            let predecessors = cfg.predecessors(block_id);
322            // Dominance frontier nodes must have more than one predecessor. They are join points in the CFG.
323            if predecessors.len() <= 1 {
324                continue;
325            }
326            let Some(immediate_dominator) = self.immediate_dominator(block_id) else {
327                continue;
328            };
329            // Iterate over the predecessors of the current block and walk backwards from them in the dominator tree.
330            for pred_id in predecessors {
331                let mut runner = pred_id;
332                loop {
333                    // Once we reach the immediate dominator of the current block, we know the current block
334                    // won't be in the frontier of any further blocks (frontier blocks are *not* dominated by them).
335                    if immediate_dominator == runner {
336                        break;
337                    }
338                    // Checking if the current block dominates the predecessor;
339                    // for example a loop header has the loop body as one of its predecessors, which it dominates,
340                    // but we don't consider following back-edges as alternative paths on which we reach the header first.
341                    if self.dominates(block_id, runner) {
342                        break;
343                    }
344                    dominance_frontiers.entry(runner).or_default().insert(block_id);
345                    // Continue walking backwards to the dominators of the runner, which also have the
346                    // current block in their frontier, unless they dominate it.
347                    let Some(runner_immediate_dom) = self.immediate_dominator(runner) else {
348                        break;
349                    };
350                    runner = runner_immediate_dom;
351                }
352            }
353        }
354
355        dominance_frontiers
356    }

726
727        let blocks = vecmap(0..6, Id::<BasicBlock>::test_new);
728
729        // b0 is the entry node, thus it does not post-dominate anything except itself
730        //
731        // b2 and b4 are leaves in the post-dominator tree. There are no nodes that must pass through
732        // those blocks to reach the exit node.
733        // The dominator tree computation does not recognize that the loop has constant bounds,
734        // so it will still account for the jmpif in b1 and the possibility of skipping b2.
735        //
736        // All nodes except the exit node b3, must pass through b1 to reach the exit node.
737        //
738        // Starting from the exit node b3 which should be the root of the post-dominator tree
739        // Every block except for the loop header b1, the exit node b3, and the entry node b0,
740        // must pass through the loop exit, b5, to reach the exit node.
741        //
742        // Post-dominance matrix:
743        // ✓: Row item post-dominates column item
744        //    b0  b1  b2  b3  b4  b5
745        // b0 ✓
746        // b1 ✓   ✓   ✓       ✓   ✓
747        // b2         ✓
748        // b3 ✓   ✓   ✓   ✓   ✓   ✓
749        // b4                 ✓
750        // b5         ✓       ✓   ✓
751
752        let post_dominance_matrix = vec![
753            vec![true, false, false, false, false, false],
754            vec![true, true, true, false, true, true],
755            vec![false, false, true, false, false, false],
756            vec![true, true, true, true, true, true],
757            vec![false, false, false, false, true, false],
758            vec![false, false, true, false, true, true],
759        ];
760

765    fn dom_frontiers() {
766        let ssa = loop_with_cond();
767        let main = ssa.main();
768
769        let cfg = ControlFlowGraph::with_function(main);
770        let post_order = PostOrder::with_cfg(&cfg);
771
772        let mut dt = DominatorTree::with_cfg_and_post_order(&cfg, &post_order);
773        let dom_frontiers = dt.compute_dominance_frontiers(&cfg);
774
775        let blocks = vecmap(0..6, Id::<BasicBlock>::test_new);
776
777        // b0 is the entry block which dominates all other blocks
778        // Thus, it has an empty set for its dominance frontier
779        assert!(!dom_frontiers.contains_key(&blocks[0]));
780        assert!(!dom_frontiers.contains_key(&blocks[1]));
781        assert!(!dom_frontiers.contains_key(&blocks[2]));
782        // b3 is the exit block which does not dominate any blocks
783        assert!(!dom_frontiers.contains_key(&blocks[3]));
784
785        // b4 has DF { b5 } because b4 jumps to b5, thus being a predecessor to b5.
786        // b5 dominates itself but b5 does not strictly dominate b4.
787        let b4_df = &dom_frontiers[&blocks[4]];
788        assert_eq!(b4_df.len(), 1);
789        assert!(b4_df.contains(&blocks[5]));
790
791        assert!(!dom_frontiers.contains_key(&blocks[5]));
792    }

823    fn post_dom_frontiers() {
824        let ssa = loop_with_cond();
825        let main = ssa.main();
826
827        let cfg = ControlFlowGraph::with_function(main);
828        let reversed_cfg = cfg.reverse();
829        let post_order = PostOrder::with_cfg(&reversed_cfg);
830
831        let mut post_dom = DominatorTree::with_cfg_and_post_order(&reversed_cfg, &post_order);
832        let post_dom_frontiers = post_dom.compute_dominance_frontiers(&reversed_cfg);
833
834        let blocks = vecmap(0..6, Id::<BasicBlock>::test_new);
835
836        // Another way to think about the post-dominator frontier (PDF) for a node n,
837        // is that we can reach a block in the PDF during execution without going through n.
838
839        // b0 is the entry node of the program and the exit block of the post-dominator tree.
840        // Thus, it has an empty set for its PDF
841        assert!(!post_dom_frontiers.contains_key(&blocks[0]));
842        // We must go through b1 and b2 to reach the exit node
843        assert!(!post_dom_frontiers.contains_key(&blocks[1]));
844        assert!(!post_dom_frontiers.contains_key(&blocks[2]));
845
846        // b3 is the exit block of the program, but the starting node of the post-dominator tree
847        // Thus, it has an empty PDF
848        assert!(!post_dom_frontiers.contains_key(&blocks[3]));
849
850        // b4 has DF { b2 } because b2 post-dominates itself and is a predecessor to b4.
851        // b2 does not strictly post-dominate b4.
852        let b4_pdf = &post_dom_frontiers[&blocks[4]];
853        assert_eq!(b4_pdf.len(), 1);
854        assert!(b4_pdf.contains(&blocks[2]));
855
856        // Must go through b5 to reach the exit node
857        assert!(!post_dom_frontiers.contains_key(&blocks[5]));
858    }
