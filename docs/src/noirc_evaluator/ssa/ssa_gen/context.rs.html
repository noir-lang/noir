29/// The FunctionContext is the main context object for translating a
30/// function into SSA form during the SSA-gen pass.
31///
32/// This context can be used to build any amount of functions,
33/// so long as it is cleared out in between each function via
34/// calling self.new_function().
35///
36/// If compiling many functions across multiple threads, there should
37/// be a separate FunctionContext for each thread. Each FunctionContext
38/// can communicate via the SharedContext field which as its name suggests
39/// is the only part of the context that needs to be shared between threads.
40pub(super) struct FunctionContext<'a> {
41    definitions: HashMap<LocalId, Values>,
42    pub(super) redefinitions_allowed: bool,
43
44    pub(super) builder: FunctionBuilder,
45    shared_context: &'a SharedContext,
46
47    /// Contains any loops we're currently in the middle of translating.
48    /// These are ordered such that an inner loop is at the end of the vector and
49    /// outer loops are at the beginning. When a loop is finished, it is popped.
50    loops: Vec<Loop>,
51}
52
53/// Shared context for all functions during ssa codegen. This is the only
54/// object that is shared across all threads when generating ssa in multiple threads.
55///
56/// The main job of the SharedContext is to remember which functions are already
57/// compiled, what their IDs are, and keep a queue of which functions still need to
58/// be compiled.
59///
60/// SSA can be generated by continuously popping from this function_queue and using
61/// FunctionContext to generate from the popped function id. Once the queue is empty,
62/// no other functions are reachable and the SSA generation is finished.
63pub(super) struct SharedContext {
64    /// All currently known functions which have already been assigned function ids.
65    /// These functions are all either currently having their SSA generated or are
66    /// already finished.
67    functions: RwLock<HashMap<FuncId, IrFunctionId>>,
68
69    /// Queue of which functions still need to be compiled.
70    ///
71    /// The queue is currently Last-in First-out (LIFO) but this is an
72    /// implementation detail that can be trivially changed and should
73    /// not impact the resulting SSA besides changing which IDs are assigned
74    /// to which functions.
75    function_queue: Mutex<FunctionQueue>,
76
77    /// Shared counter used to assign the ID of the next function
78    function_counter: AtomicCounter<Function>,
79
80    /// A pseudo function that represents global values.
81    /// Globals are only concerned with the values and instructions (due to Instruction::MakeArray)
82    /// in a function's DataFlowGraph. However, in order to re-use various codegen methods
83    /// we need to use the same `Function` type.
84    pub(super) globals_context: Function,
85
86    pub(super) globals: BTreeMap<GlobalId, Values>,
87
88    /// The entire monomorphized source program
89    pub(super) program: Program,
90}

93pub(super) struct Loop {
94    pub(super) loop_entry: BasicBlockId,
95    /// The loop index will be `Some` for a `for` and `None` for a `loop`
96    pub(super) loop_index: Option<ValueId>,
97    pub(super) loop_end: BasicBlockId,
98    /// A variable that tracks whether a `break` was hit or not:
99    /// `false` if a `break` was hit, `true` if not.
100    /// This is only `Some` in the case of an inclusive for loop which is
101    /// generated as an exclusive for loop with an extra iteration for the
102    /// end of the loop. This extra iteration is only done if no `break` was hit
103    /// in the exclusive iterations (and if `start <= end`).
104    /// We track the negated value because we execute the last iteration
105    /// if we did not hit a break, in the end being `did_not_hit_break && (start <= end)`.
106    /// If we tracked whether we hit a break or not, the condition to execute
107    /// the last iteration would be `(not hit_break) && (start <= end)`, which is larger
108    /// by one instruction.
109    pub(super) did_not_hit_break_var: Option<ValueId>,
110}

116    /// Create a new FunctionContext to compile the first function in the shared_context's
117    /// function queue.
118    ///
119    /// This will pop from the function queue, so it is expected the shared_context's function
120    /// queue is non-empty at the time of calling this function. This can be ensured by calling
121    /// `shared_context.get_or_queue_function(function_to_queue)` before calling this constructor.
122    ///
123    /// `function_name` and `parameters` are expected to be the name and parameters of the function
124    /// this constructor will pop from the function queue.
125    pub(super) fn new(
126        function_name: String,
127        parameters: &Parameters,
128        runtime: RuntimeType,
129        shared_context: &'a SharedContext,
130        globals: GlobalsGraph,
131    ) -> Self {
132        let function_id = shared_context
133            .pop_next_function_in_queue()
134            .expect("No function in queue for the FunctionContext to compile")
135            .1;
136
137        let mut builder = FunctionBuilder::new(function_name, function_id);
138        builder.set_globals(Arc::new(globals));
139        builder.set_runtime(runtime);
140
141        let definitions = HashMap::default();
142        let mut this = Self {
143            definitions,
144            builder,
145            shared_context,
146            loops: Vec::new(),
147            redefinitions_allowed: false,
148        };
149        this.add_parameters_to_scope(parameters);
150        this
151    }

153    /// Finish building the current function and switch to building a new function with the
154    /// given name, id, and parameters.
155    ///
156    /// Note that the previous function cannot be resumed after calling this. Developers should
157    /// avoid calling new_function until the previous function is completely finished with ssa-gen.
158    pub(super) fn new_function(&mut self, id: IrFunctionId, func: &ast::Function) {
159        self.definitions.clear();
160
161        if func.unconstrained {
162            self.builder.new_brillig_function(func.name.clone(), id, func.inline_type);
163        } else {
164            self.builder.new_function(func.name.clone(), id, func.inline_type);
165        }
166
167        self.add_parameters_to_scope(&func.parameters);
168    }

170    /// Add each parameter to the current scope, and return the list of parameter types.
171    ///
172    /// The returned parameter type list will be flattened, so any struct parameters will
173    /// be returned as one entry for each field (recursively).
174    fn add_parameters_to_scope(&mut self, parameters: &Parameters) {
175        for (id, mutable, _, typ, _visibility) in parameters {
176            self.add_parameter_to_scope(*id, typ, *mutable);
177        }
178    }

180    /// Adds a "single" parameter to scope.
181    ///
182    /// Single is in quotes here because in the case of tuple parameters, the tuple is flattened
183    /// into a new parameter for each field recursively.
184    fn add_parameter_to_scope(
185        &mut self,
186        parameter_id: LocalId,
187        parameter_type: &ast::Type,
188        mutable: bool,
189    ) {
190        // Add a separate parameter for each field type in 'parameter_type'
191        let parameter_value = Self::map_type(parameter_type, |typ| {
192            let value = self.builder.add_parameter(typ);
193            if mutable {
194                // This will wrap any `mut var: T` in a reference
195                self.new_mutable_variable(value)
196            } else {
197                value.into()
198            }
199        });
200
201        self.definitions.insert(parameter_id, parameter_value);
202    }

204    /// Allocate a single slot of memory and store into it the given initial value of the variable.
205    /// Always returns a Value::Mutable wrapping the allocate instruction.
206    pub(super) fn new_mutable_variable(&mut self, value_to_store: ValueId) -> Value {
207        let element_type = self.builder.current_function.dfg.type_of_value(value_to_store);
208
209        let alloc = self.builder.insert_allocate(element_type);
210        self.builder.insert_store(alloc, value_to_store);
211        let typ = self.builder.type_of_value(value_to_store);
212        Value::Mutable(alloc, typ)
213    }

215    /// Maps the given type to a Tree of the result type.
216    ///
217    /// This can be used to (for example) flatten a tuple type, creating
218    /// and returning a new parameter for each field type.
219    pub(super) fn map_type<T>(typ: &ast::Type, mut f: impl FnMut(Type) -> T) -> Tree<T> {
220        Self::map_type_helper(typ, &mut f)
221    }

223    // This helper is needed because we need to take f by mutable reference,
224    // otherwise we cannot move it multiple times each loop of vecmap.
225    fn map_type_helper<T>(typ: &ast::Type, f: &mut dyn FnMut(Type) -> T) -> Tree<T> {
226        match typ {
227            ast::Type::Tuple(fields) => {
228                Tree::Branch(vecmap(fields, |field| Self::map_type_helper(field, f)))
229            }
230            ast::Type::Unit => Tree::empty(),
231            // A mutable reference wraps each element into a reference.
232            // This can be multiple values if the element type is a tuple.
233            ast::Type::Reference(element, _) => {
234                Self::map_type_helper(element, &mut |typ| f(Type::Reference(Arc::new(typ))))
235            }
236            ast::Type::FmtString(len, fields) => {
237                // A format string is represented by multiple values
238                // The message string, the number of fields to be formatted, and
239                // then the encapsulated fields themselves
240                let final_fmt_str_fields =
241                    vec![ast::Type::String(*len), ast::Type::Field, *fields.clone()];
242                let fmt_str_tuple = ast::Type::Tuple(final_fmt_str_fields);
243                Self::map_type_helper(&fmt_str_tuple, f)
244            }
245            ast::Type::Vector(elements) => {
246                let element_types = Self::convert_type(elements).flatten();
247                Tree::Branch(vec![
248                    Tree::Leaf(f(Type::length_type())),
249                    Tree::Leaf(f(Type::Vector(Arc::new(element_types)))),
250                ])
251            }
252            other => Tree::Leaf(f(Self::convert_non_tuple_type(other))),
253        }
254    }

256    /// Convert a monomorphized type to an SSA type, preserving the structure
257    /// of any tuples within.
258    pub(super) fn convert_type(typ: &ast::Type) -> Tree<Type> {
259        // Do nothing in the closure here - map_type_helper already calls
260        // convert_non_tuple_type internally.
261        Self::map_type_helper(typ, &mut |x| x)
262    }

264    /// Converts a non-tuple type into an SSA type. Panics if a tuple type is passed.
265    ///
266    /// This function is needed since this SSA IR has no concept of tuples and thus no type for
267    /// them. Use `convert_type` if tuple types need to be handled correctly.
268    pub(super) fn convert_non_tuple_type(typ: &ast::Type) -> Type {
269        match typ {
270            ast::Type::Field => Type::field(),
271            ast::Type::Array(len, element) => {
272                let element_types = Self::convert_type(element).flatten();
273                Type::Array(Arc::new(element_types), SemanticLength(*len))
274            }
275            ast::Type::Integer(Signedness::Signed, bits) => Type::signed((*bits).into()),
276            ast::Type::Integer(Signedness::Unsigned, bits) => Type::unsigned((*bits).into()),
277            ast::Type::Bool => Type::unsigned(1),
278            ast::Type::String(len) => Type::str(*len),
279            ast::Type::FmtString(_, _) => {
280                panic!("convert_non_tuple_type called on a fmt string: {typ}")
281            }
282            ast::Type::Unit => panic!("convert_non_tuple_type called on a unit type"),
283            ast::Type::Tuple(_) => panic!("convert_non_tuple_type called on a tuple: {typ}"),
284            ast::Type::Function(_, _, _, _) => Type::Function,
285            ast::Type::Vector(_) => panic!("convert_non_tuple_type called on a vector: {typ}"),
286            ast::Type::Reference(element, _) => {
287                // Recursive call to panic if element is a tuple
288                let element = Self::convert_non_tuple_type(element);
289                Type::Reference(Arc::new(element))
290            }
291        }
292    }

299    /// Insert a numeric constant into the current function
300    ///
301    /// Unlike FunctionBuilder::numeric_constant, this version checks the given constant
302    /// is within the range of the given type. This is needed for user provided values where
303    /// otherwise values like 2^128 can be assigned to a u8 without error or wrapping.
304    pub(super) fn checked_numeric_constant(
305        &mut self,
306        value: SignedField,
307        numeric_type: NumericType,
308    ) -> Result<ValueId, RuntimeError> {
309        if let Some(range) = numeric_type.value_is_outside_limits(value) {
310            let call_stack = self.builder.get_call_stack();
311            return Err(RuntimeError::IntegerOutOfBounds {
312                value,
313                typ: numeric_type,
314                range,
315                call_stack,
316            });
317        }
318
319        let value = if value.is_negative() {
320            match numeric_type {
321                NumericType::NativeField => -value.absolute_value(),
322                NumericType::Signed { bit_size } | NumericType::Unsigned { bit_size } => {
323                    assert!(bit_size < 128);
324                    let base = 1_u128 << bit_size;
325                    FieldElement::from(base) - value.absolute_value()
326                }
327            }
328        } else {
329            value.absolute_value()
330        };
331
332        Ok(self.builder.numeric_constant(value, numeric_type))
333    }

335    /// Insert a binary instruction at the end of the current block.
336    /// Converts the form of the binary instruction as necessary
337    /// (e.g. swapping arguments, inserting a not) to represent it in the IR.
338    /// For example, (a <= b) is represented as !(b < a)
339    pub(super) fn insert_binary(
340        &mut self,
341        mut lhs: ValueId,
342        operator: BinaryOpKind,
343        mut rhs: ValueId,
344        location: Location,
345    ) -> Values {
346        let mut op = convert_operator(operator);
347        if operator_requires_swapped_operands(operator) {
348            std::mem::swap(&mut lhs, &mut rhs);
349        }
350
351        self.builder.set_location(location);
352
353        // Field operations are inherently unchecked (finite field arithmetic wraps naturally)
354        let lhs_type = self.builder.type_of_value(lhs);
355        if matches!(lhs_type, Type::Numeric(NumericType::NativeField)) {
356            op = op.into_unchecked();
357        }
358
359        let mut result = self.builder.insert_binary(lhs, op, rhs);
360
361        if operator_requires_not(operator) {
362            result = self.builder.insert_not(result);
363        }
364
365        result.into()
366    }

368    /// Inserts a call instruction at the end of the current block and returns the results
369    /// of the call.
370    ///
371    /// Compared to self.builder.insert_call, this version will reshape the returned `Vec<ValueId>`
372    /// back into a Values tree of the proper shape.
373    pub(super) fn insert_call(
374        &mut self,
375        function: ValueId,
376        arguments: Vec<ValueId>,
377        result_type: &ast::Type,
378        location: Location,
379    ) -> Values {
380        let result_types = Self::convert_type(result_type).flatten();
381        let results =
382            self.builder.set_location(location).insert_call(function, arguments, result_types);
383
384        let mut i = 0;
385        let reshaped_return_values = Self::map_type(result_type, |_| {
386            let result = results[i].into();
387            i += 1;
388            result
389        });
390        assert_eq!(i, results.len());
391        reshaped_return_values
392    }

394    /// Inserts a cast instruction at the end of the current block and returns the results
395    /// of the cast.
396    ///
397    /// Compared to `self.builder.insert_cast`, this version will automatically truncate `value` to be a valid `typ`.
398    pub(super) fn insert_safe_cast(
399        &mut self,
400        value: ValueId,
401        typ: NumericType,
402        location: Location,
403    ) -> ValueId {
404        self.builder.set_location(location);
405        let incoming_type = self.builder.type_of_value(value);
406
407        let result = match (&incoming_type, typ) {
408            // Casting to field is safe
409            (_, NumericType::NativeField) => value,
410            (
411                Type::Numeric(NumericType::Signed { bit_size: incoming_type_size }),
412                NumericType::Signed { bit_size: target_type_size },
413            ) => {
414                match target_type_size.cmp(incoming_type_size) {
415                    std::cmp::Ordering::Less => {
416                        // If target size is smaller, we do a truncation
417                        self.builder.insert_truncate(value, target_type_size, *incoming_type_size)
418                    }
419                    std::cmp::Ordering::Equal => value,
420                    std::cmp::Ordering::Greater => {
421                        self.sign_extend(value, *incoming_type_size, target_type_size, location)
422                    }
423                }
424            }
425            (
426                Type::Numeric(NumericType::Unsigned { bit_size: incoming_type_size }),
427                NumericType::Unsigned { bit_size: target_type_size },
428            ) => {
429                // If target size is smaller, we do a truncation
430                if target_type_size < *incoming_type_size {
431                    self.builder.insert_truncate(value, target_type_size, *incoming_type_size)
432                } else {
433                    value
434                }
435            }
436            // When casting a signed value to u1 we can truncate then cast
437            (
438                Type::Numeric(NumericType::Signed { bit_size: incoming_type_size }),
439                NumericType::Unsigned { bit_size: 1 },
440            ) => self.builder.insert_truncate(value, 1, *incoming_type_size),
441
442            // For mixed signed to unsigned:
443            (
444                Type::Numeric(NumericType::Signed { bit_size: incoming_type_size }),
445                NumericType::Unsigned { bit_size: target_type_size },
446            ) => {
447                // when going from lower to higher bit size:
448                // 1. we sign-extend to the target bits
449                // 2. we are already in the target signedness
450                if *incoming_type_size < target_type_size {
451                    // By not the casting to a signed type with the target bit size, we avoid potentially going
452                    // through i128, which is not a type we support in the frontend, and would be strange in SSA.
453                    self.sign_extend(value, *incoming_type_size, target_type_size, location)
454                }
455                // when the target bit size is not higher than the source:
456                // 1. we cast to the required type using the same signedness
457                // 2. then we switch the signedness
458                else if *incoming_type_size != target_type_size {
459                    self.insert_safe_cast(value, NumericType::signed(target_type_size), location)
460                } else {
461                    value
462                }
463            }
464
465            // For mixed unsigned to signed:
466            // 1. we cast to the required type using the same signedness
467            // 2. then we switch the signedness
468            (
469                Type::Numeric(NumericType::Unsigned { bit_size: incoming_type_size }),
470                NumericType::Signed { bit_size: target_type_size },
471            ) => {
472                if *incoming_type_size != target_type_size {
473                    self.insert_safe_cast(value, NumericType::unsigned(target_type_size), location)
474                } else {
475                    value
476                }
477            }
478
479            // Field to signed/unsigned:
480            (
481                Type::Numeric(NumericType::NativeField),
482                NumericType::Unsigned { bit_size: target_type_size },
483            )
484            | (
485                Type::Numeric(NumericType::NativeField),
486                NumericType::Signed { bit_size: target_type_size },
487            ) => {
488                self.builder.insert_truncate(value, target_type_size, FieldElement::max_num_bits())
489            }
490            _ => unreachable!("Invalid cast from {} to {}", incoming_type, typ),
491        };
492        self.builder.insert_cast(result, typ)
493    }

495    /// During casting signed values, if target size is bigger, we do a sign extension:
496    ///
497    /// When the value is negative, it is represented in 2-complement form; `2^s-v`, where `s` is the incoming bit size and `v` is the absolute value.
498    /// Sign extension in this case will give `2^t-v`, where `t` is the target bit size.
499    /// So we simply convert `2^s-v` into `2^t-v` by adding `2^t-2^s` to the value when the value is negative.
500    ///
501    /// Casting s-bits signed v0 to t-bits will add the following instructions:
502    /// ```ssa
503    /// v1 = cast v0 to 's-bits unsigned'
504    /// v2 = lt v1, 2**(s-1)
505    /// v3 = not(v1)
506    /// v4 = cast v3 to 't-bits unsigned'
507    /// v5 = v3 * (2**t - 2**s)
508    /// v6 = cast v1 to 't-bits unsigned'
509    /// return v6 + v5
510    /// ```
511    ///
512    /// Return an unsigned value that we can cast back to the signed type if we want,
513    /// or keep it as it is, if we did the sign extension as part of casting e.g. `i8` to `u64`.
514    fn sign_extend(
515        &mut self,
516        value: ValueId,
517        incoming_type_size: u32,
518        target_type_size: u32,
519        location: Location,
520    ) -> ValueId {
521        let value_as_unsigned =
522            self.insert_safe_cast(value, NumericType::unsigned(incoming_type_size), location);
523        let half_width = self.builder.numeric_constant(
524            FieldElement::from(2_u128.pow(incoming_type_size - 1)),
525            NumericType::unsigned(incoming_type_size),
526        );
527        // value_sign is 1 if the value is positive, 0 otherwise
528        let value_sign = self.builder.insert_binary(value_as_unsigned, BinaryOp::Lt, half_width);
529        let max_for_incoming_type_size =
530            if incoming_type_size == 128 { u128::MAX } else { 2_u128.pow(incoming_type_size) - 1 };
531        let max_for_target_type_size =
532            if target_type_size == 128 { u128::MAX } else { 2_u128.pow(target_type_size) - 1 };
533        let patch = self.builder.numeric_constant(
534            FieldElement::from(max_for_target_type_size - max_for_incoming_type_size),
535            NumericType::unsigned(target_type_size),
536        );
537        let mut is_negative_predicate = self.builder.insert_not(value_sign);
538        is_negative_predicate = self.insert_safe_cast(
539            is_negative_predicate,
540            NumericType::unsigned(target_type_size),
541            location,
542        );
543        // multiplication by a boolean cannot overflow
544        let patch_with_sign_predicate = self.builder.insert_binary(
545            patch,
546            BinaryOp::Mul { unchecked: true },
547            is_negative_predicate,
548        );
549        let value_as_unsigned =
550            self.builder.insert_cast(value_as_unsigned, NumericType::unsigned(target_type_size));
551        // Patch the bit sign, which gives a `target_type_size` bit size value, so it does not overflow.
552        self.builder.insert_binary(
553            patch_with_sign_predicate,
554            BinaryOp::Add { unchecked: true },
555            value_as_unsigned,
556        )
557    }

559    /// Create a const offset of an address for an array load or store
560    pub(super) fn make_offset(
561        &mut self,
562        mut address: ValueId,
563        offset: u128,
564        unchecked: bool,
565    ) -> ValueId {
566        if offset != 0 {
567            let typ = self.builder.type_of_value(address).unwrap_numeric();
568            let offset = self.builder.numeric_constant(offset, typ);
569            address = self.builder.insert_binary(address, BinaryOp::Add { unchecked }, offset);
570        }
571        address
572    }

648    /// Extracts the current value out of an LValue.
649    ///
650    /// Goal: Handle the case of assigning to nested expressions such as `foo.bar[i1].baz[i2] = e`
651    ///       while also noting that assigning to arrays will create a new array rather than mutate
652    ///       the original.
653    ///
654    /// Method: First `extract_current_value` must recurse on the lvalue to extract the current
655    ///         value contained:
656    ///
657    /// ```text
658    /// v0 = foo.bar                 ; allocate instruction for bar
659    /// v1 = load v0                 ; loading the bar array
660    /// v2 = add i1, baz_index       ; field offset for index i1, field baz
661    /// v3 = array_get v1, index v2  ; foo.bar[i1].baz
662    /// ```
663    ///
664    /// Method (part 2): Then, `assign_new_value` will recurse in the opposite direction to
665    ///                  construct the larger value as needed until we can `store` to the nearest
666    ///                  allocation.
667    ///
668    /// ```text
669    /// v4 = array_set v3, index i2, e   ; finally create a new array setting the desired value
670    /// v5 = array_set v1, index v2, v4  ; now must also create the new bar array
671    /// store v5 in v0                   ; and store the result in the only mutable reference
672    /// ```
673    ///
674    /// The returned `LValueRef` tracks the current value at each step of the lvalue.
675    /// This is later used by `assign_new_value` to construct a new updated value that
676    /// can be assigned to an allocation within the LValueRef::Ident.
677    ///
678    /// This is operationally equivalent to extract_current_value_recursive, but splitting these
679    /// into two separate functions avoids cloning the outermost `Values` returned by the recursive
680    /// version, as it is only needed for recursion.
681    pub(super) fn extract_current_value(
682        &mut self,
683        lvalue: &ast::LValue,
684    ) -> Result<LValue, RuntimeError> {
685        Ok(match lvalue {
686            ast::LValue::Ident(ident) => {
687                let (reference, should_auto_deref) = self.ident_lvalue(ident);
688                if should_auto_deref { LValue::Dereference { reference } } else { LValue::Ident }
689            }
690            ast::LValue::Index { array, index, location, .. } => {
691                self.index_lvalue(array, index, location)?.2
692            }
693            ast::LValue::MemberAccess { object, field_index } => {
694                let (old_object, object_lvalue) = self.extract_current_value_recursive(object)?;
695                let object_lvalue = Box::new(object_lvalue);
696                LValue::MemberAccess { old_object, object_lvalue, index: *field_index }
697            }
698            ast::LValue::Dereference { reference, .. } => {
699                let (reference, _) = self.extract_current_value_recursive(reference)?;
700                LValue::Dereference { reference }
701            }
702            ast::LValue::Clone(lvalue) => self.extract_current_value(lvalue)?,
703        })
704    }
705
706    fn dereference_lvalue(&mut self, values: &Values, element_type: &ast::Type) -> Values {
707        let element_types = Self::convert_type(element_type);
708        values.map_both(element_types, |value, element_type| {
709            let reference = value.eval_reference();
710            self.builder.insert_load(reference, element_type).into()
711        })
712    }

714    /// Compile the given identifier as a reference - ie. avoid calling .eval().
715    /// Returns the variable's value and whether the variable is mutable.
716    fn ident_lvalue(&self, ident: &ast::Ident) -> (Values, bool) {
717        match &ident.definition {
718            ast::Definition::Local(id) => (self.lookup(*id), ident.mutable),
719            other => panic!("Unexpected definition found for mutable value: {other}"),
720        }
721    }

723    /// Compile the given `array[index]` expression as a reference.
724    /// This will return a triple of (array, index, lvalue_ref, `Option<length>`) where the lvalue_ref records the
725    /// structure of the lvalue expression for use by `assign_new_value`.
726    /// The optional length is for indexing vectors rather than arrays since vectors
727    /// are represented as a tuple in the form: (length, vector contents).
728    fn index_lvalue(
729        &mut self,
730        array: &ast::LValue,
731        index: &ast::Expression,
732        location: &Location,
733    ) -> Result<(ValueId, ValueId, LValue, Option<ValueId>), RuntimeError> {
734        let (old_array, array_lvalue) = self.extract_current_value_recursive(array)?;
735        let index = self.codegen_non_tuple_expression(index)?;
736        let array_lvalue = Box::new(array_lvalue);
737        let array_values = old_array.clone().into_value_list(self);
738
739        let location = *location;
740        // A vector is represented as a tuple (length, vector contents).
741        // We need to fetch the second value.
742        Ok(if array_values.len() > 1 {
743            let vector_lvalue = LValue::VectorIndex {
744                old_vector: old_array,
745                index,
746                vector_lvalue: array_lvalue,
747                location,
748            };
749            (array_values[1], index, vector_lvalue, Some(array_values[0]))
750        } else {
751            let array_lvalue =
752                LValue::Index { old_array: array_values[0], index, array_lvalue, location };
753            (array_values[0], index, array_lvalue, None)
754        })
755    }
756
757    fn extract_current_value_recursive(
758        &mut self,
759        lvalue: &ast::LValue,
760    ) -> Result<(Values, LValue), RuntimeError> {
761        match lvalue {
762            ast::LValue::Ident(ident) => {
763                let (variable, should_auto_deref) = self.ident_lvalue(ident);
764                if should_auto_deref {
765                    let dereferenced = self.dereference_lvalue(&variable, &ident.typ);
766                    Ok((dereferenced, LValue::Dereference { reference: variable }))
767                } else {
768                    Ok((variable.clone(), LValue::Ident))
769                }
770            }
771            ast::LValue::Index { array, index, element_type, location } => {
772                let (old_array, index, index_lvalue, max_length) =
773                    self.index_lvalue(array, index, location)?;
774                let element = self.codegen_array_index(
775                    old_array,
776                    index,
777                    element_type,
778                    *location,
779                    max_length,
780                )?;
781                Ok((element, index_lvalue))
782            }
783            ast::LValue::MemberAccess { object, field_index: index } => {
784                let (old_object, object_lvalue) = self.extract_current_value_recursive(object)?;
785                let object_lvalue = Box::new(object_lvalue);
786                let element = Self::get_field_ref(&old_object, *index).clone();
787                Ok((element, LValue::MemberAccess { old_object, object_lvalue, index: *index }))
788            }
789            ast::LValue::Dereference { reference, element_type } => {
790                let (reference, _) = self.extract_current_value_recursive(reference)?;
791                let dereferenced = self.dereference_lvalue(&reference, element_type);
792                Ok((dereferenced, LValue::Dereference { reference }))
793            }
794            ast::LValue::Clone(lvalue) => {
795                let (values, lvalue) = self.extract_current_value_recursive(lvalue)?;
796                values.clone().for_each(|value| {
797                    let value = value.eval(self);
798                    self.builder.increment_array_reference_count(value);
799                });
800                Ok((values, lvalue))
801            }
802        }
803    }

805    /// Assigns a new value to the given LValue.
806    /// The LValue can be created via a previous call to extract_current_value.
807    /// This method recurs on the given LValue to create a new value to assign an allocation
808    /// instruction within an LValue::Ident or LValue::Dereference - see the comment on
809    /// `extract_current_value` for more details.
810    pub(super) fn assign_new_value(&mut self, lvalue: LValue, new_value: Values) {
811        match lvalue {
812            LValue::Ident => unreachable!("Cannot assign to a variable without a reference"),
813            LValue::Index { old_array: mut array, index, array_lvalue, location } => {
814                let array_type = &self.builder.type_of_value(array);
815
816                // Checks for index Out-of-bounds
817                match array_type {
818                    Type::Array(_, len) => {
819                        // Out of bounds array accesses are guaranteed to fail in ACIR so this check is performed implicitly.
820                        // We then only need to inject it for brillig functions.
821                        if self.builder.current_function.runtime().is_brillig() {
822                            let len = self
823                                .builder
824                                .numeric_constant(u128::from(len.0), NumericType::length_type());
825                            self.codegen_access_check(index, len);
826                        }
827                    }
828                    _ => unreachable!("must have array or vector but got {array_type}"),
829                }
830
831                array = self.assign_lvalue_index(new_value, array, index, location);
832                self.assign_new_value(*array_lvalue, array.into());
833            }
834            LValue::VectorIndex { old_vector: vector, index, vector_lvalue, location } => {
835                let mut vector_values = vector.into_value_list(self);
836
837                let array_type = &self.builder.type_of_value(vector_values[1]);
838
839                // Checks for index Out-of-bounds
840                match array_type {
841                    Type::Vector(_) => {
842                        self.codegen_access_check(index, vector_values[0]);
843                    }
844                    _ => unreachable!("must have array or vector but got {array_type}"),
845                }
846
847                vector_values[1] =
848                    self.assign_lvalue_index(new_value, vector_values[1], index, location);
849
850                // The size of the vector does not change in a vector index assignment so we can reuse the same length value
851                let new_vector =
852                    Tree::Branch(vec![vector_values[0].into(), vector_values[1].into()]);
853                self.assign_new_value(*vector_lvalue, new_vector);
854            }
855            LValue::MemberAccess { old_object, index, object_lvalue } => {
856                let new_object = Self::replace_field(old_object, index, new_value);
857                self.assign_new_value(*object_lvalue, new_object);
858            }
859            LValue::Dereference { reference } => {
860                self.assign(reference, new_value);
861            }
862        }
863    }
864
865    fn assign_lvalue_index(
866        &mut self,
867        new_value: Values,
868        mut array: ValueId,
869        index: ValueId,
870        location: Location,
871    ) -> ValueId {
872        let index = self.make_array_index(index);
873        let element_size =
874            self.builder.numeric_constant(self.element_size(array), NumericType::length_type());
875
876        // The actual base index is the user's index * the array element type's size
877        // Unchecked mul because we are reaching for an array element: if it overflows here
878        // it would have overflowed when creating the array.
879        let mut index = self.builder.set_location(location).insert_binary(
880            index,
881            BinaryOp::Mul { unchecked: true },
882            element_size,
883        );
884        let one = self.builder.numeric_constant(FieldElement::one(), NumericType::length_type());
885
886        new_value.for_each(|value| {
887            let value = value.eval(self);
888            let mutable = false;
889            array = self.builder.insert_array_set(array, index, value, mutable);
890            // Unchecked add because this can't overflow (it would have overflowed when creating the array)
891            index = self.builder.insert_binary(index, BinaryOp::Add { unchecked: true }, one);
892        });
893        array
894    }
895
896    fn element_size(&self, array: ValueId) -> FieldElement {
897        let size = self.builder.type_of_value(array).element_size();
898        FieldElement::from(size.0)
899    }

901    /// Given an lhs containing only references, create a store instruction to store each value of
902    /// rhs into its corresponding value in lhs.
903    fn assign(&mut self, lhs: Values, rhs: Values) {
904        match (lhs, rhs) {
905            (Tree::Branch(lhs_branches), Tree::Branch(rhs_branches)) => {
906                assert_eq!(lhs_branches.len(), rhs_branches.len());
907
908                for (lhs, rhs) in lhs_branches.into_iter().zip(rhs_branches) {
909                    self.assign(lhs, rhs);
910                }
911            }
912            (Tree::Leaf(lhs), Tree::Leaf(rhs)) => {
913                let (lhs, rhs) = (lhs.eval_reference(), rhs.eval(self));
914                self.builder.insert_store(lhs, rhs);
915            }
916            (lhs, rhs) => {
917                unreachable!(
918                    "assign: Expected lhs and rhs values to match but found {lhs:?} and {rhs:?}"
919                )
920            }
921        }
922    }

938/// True if the given operator cannot be encoded directly and needs
939/// to be represented as !(some other operator)
940fn operator_requires_not(op: BinaryOpKind) -> bool {
941    use BinaryOpKind::*;
942    matches!(op, NotEqual | LessEqual | GreaterEqual)
943}
944
945/// True if the given operator cannot be encoded directly and needs
946/// to have its lhs and rhs swapped to be represented with another operator.
947/// Example: (a > b) needs to be represented as (b < a)
948fn operator_requires_swapped_operands(op: BinaryOpKind) -> bool {
949    use BinaryOpKind::*;
950    matches!(op, Greater | LessEqual)
951}
952
953/// Converts the given operator to the appropriate BinaryOp.
954/// Take care when using this to insert a binary instruction: this requires
955/// checking operator_requires_not and operator_requires_swapped_operands
956/// to represent the full operation correctly.
957fn convert_operator(op: BinaryOpKind) -> BinaryOp {
958    match op {
959        BinaryOpKind::Add => BinaryOp::Add { unchecked: false },
960        BinaryOpKind::Subtract => BinaryOp::Sub { unchecked: false },
961        BinaryOpKind::Multiply => BinaryOp::Mul { unchecked: false },
962        BinaryOpKind::Divide => BinaryOp::Div,
963        BinaryOpKind::Modulo => BinaryOp::Mod,
964        BinaryOpKind::Equal => BinaryOp::Eq,
965        BinaryOpKind::NotEqual => BinaryOp::Eq, // Requires not
966        BinaryOpKind::Less => BinaryOp::Lt,
967        BinaryOpKind::Greater => BinaryOp::Lt, // Requires operand swap
968        BinaryOpKind::LessEqual => BinaryOp::Lt, // Requires operand swap and not
969        BinaryOpKind::GreaterEqual => BinaryOp::Lt, // Requires not
970        BinaryOpKind::And => BinaryOp::And,
971        BinaryOpKind::Or => BinaryOp::Or,
972        BinaryOpKind::Xor => BinaryOp::Xor,
973        BinaryOpKind::ShiftLeft => BinaryOp::Shl,
974        BinaryOpKind::ShiftRight => BinaryOp::Shr,
975    }
976}

978impl SharedContext {
979    /// Create a new SharedContext for the given monomorphized program.
980    pub(super) fn new(program: Program) -> Self {
981        let globals_shared_context = SharedContext::new_for_globals();
982
983        let globals_id = Program::global_space_id();
984
985        // Queue the function representing the globals space for compilation
986        globals_shared_context.get_or_queue_function(globals_id);
987
988        let mut context = FunctionContext::new(
989            "globals".to_owned(),
990            &vec![],
991            RuntimeType::Brillig(InlineType::default()),
992            &globals_shared_context,
993            GlobalsGraph::default(),
994        );
995        let mut globals = BTreeMap::default();
996        for (id, (_, _, global)) in program.globals.iter() {
997            let values = context.codegen_expression(global).unwrap();
998            globals.insert(*id, values);
999        }
1000
1001        Self {
1002            functions: Default::default(),
1003            function_queue: Default::default(),
1004            function_counter: Default::default(),
1005            program,
1006            globals_context: context.builder.current_function,
1007            globals,
1008        }
1009    }

1011    pub(super) fn new_for_globals() -> Self {
1012        let globals_context = Function::new_for_globals();
1013
1014        Self {
1015            functions: Default::default(),
1016            function_queue: Default::default(),
1017            function_counter: Default::default(),
1018            program: Default::default(),
1019            globals_context,
1020            globals: Default::default(),
1021        }

1024    /// Pops the next function from the shared function queue, returning None if the queue is empty.
1025    pub(super) fn pop_next_function_in_queue(&self) -> Option<(FuncId, IrFunctionId)> {
1026        self.function_queue.lock().expect("Failed to lock function_queue").pop()
1027    }
1028
1029    /// Return the matching id for the given function if known. If it is not known this
1030    /// will add the function to the queue of functions to compile, assign it a new id,
1031    /// and return this new id.
1032    pub(super) fn get_or_queue_function(&self, id: FuncId) -> IrFunctionId {
1033        // Start a new block to guarantee the destructor for the map lock is released
1034        // before map needs to be acquired again in self.functions.write() below
1035        {
1036            let map = self.functions.read().expect("Failed to read self.functions");
1037            if let Some(existing_id) = map.get(&id) {
1038                return *existing_id;
1039            }
1040        }
1041
1042        let next_id = self.function_counter.next();
1043
1044        let mut queue = self.function_queue.lock().expect("Failed to lock function queue");
1045        queue.push((id, next_id));
1046
1047        self.functions.write().expect("Failed to write to self.functions").insert(id, next_id);
1048
1049        next_id
1050    }
1051}
