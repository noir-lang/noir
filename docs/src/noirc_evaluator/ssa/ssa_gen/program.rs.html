18/// Contains the entire SSA representation of the program.
19#[serde_as]
20#[derive(Serialize, Deserialize)]
21pub struct Ssa {
22    #[serde_as(as = "Vec<(_, _)>")]
23    pub functions: BTreeMap<FunctionId, Function>,
24    pub main_id: FunctionId,
25    #[serde(skip)]
26    pub next_id: AtomicCounter<Function>,
27    /// Maps SSA entry point function ID -> Final generated ACIR artifact index.
28    /// There can be functions specified in SSA which do not act as ACIR entry points.
29    /// This mapping is necessary to use the correct function pointer for an ACIR call,
30    /// as the final program artifact will be a list of only entry point functions.
31    #[serde(skip)]
32    entry_point_to_generated_index: BTreeMap<FunctionId, u32>,
33    // We can skip serializing this field as the error selector types end up as part of the
34    // ABI not the actual SSA IR.
35    #[serde(skip)]
36    pub error_selector_to_type: BTreeMap<ErrorSelector, HirType>,
37}
38
39impl Ssa {
40    /// Create a new Ssa object from the given SSA functions.
41    /// The first function in this vector is expected to be the main function.
42    pub fn new(functions: Vec<Function>, error_types: BTreeMap<ErrorSelector, HirType>) -> Self {
43        let main_id = functions.first().expect("Expected at least 1 SSA function").id();
44        let mut max_id = main_id;
45
46        let functions = btree_map(functions, |f| {
47            max_id = std::cmp::max(max_id, f.id());
48            (f.id(), f)
49        });
50
51        Self {
52            functions,
53            main_id,
54            next_id: AtomicCounter::starting_after(max_id),
55            entry_point_to_generated_index: BTreeMap::new(),
56            error_selector_to_type: error_types,
57        }
58    }
59
60    /// Returns the entry-point function of the program
61    pub(crate) fn main(&self) -> &Function {
62        &self.functions[&self.main_id]
63    }
64
65    /// Returns the entry-point function of the program as a mutable reference
66    #[cfg(test)]
67    pub(crate) fn main_mut(&mut self) -> &mut Function {
68        self.functions.get_mut(&self.main_id).expect("ICE: Ssa should have a main function")
69    }
70
71    /// Adds a new function to the program
72    pub(crate) fn add_fn(
73        &mut self,
74        build_with_id: impl FnOnce(FunctionId) -> Function,
75    ) -> FunctionId {
76        let new_id = self.next_id.next();
77        let function = build_with_id(new_id);
78        self.functions.insert(new_id, function);
79        new_id
80    }
81
82    pub(crate) fn generate_entry_point_index(mut self) -> Self {
83        let entry_points =
84            self.functions.keys().filter(|function| self.is_entry_point(**function)).enumerate();
85        self.entry_point_to_generated_index = btree_map(entry_points, |(i, id)| (*id, i as u32));
86        self
87    }
88
89    pub(crate) fn get_entry_point_index(&self, func_id: &FunctionId) -> Option<u32> {
90        // Ensure the map has been initialized
91        assert!(
92            !self.entry_point_to_generated_index.is_empty(),
93            "Trying to read uninitialized entry point index"
94        );
95        self.entry_point_to_generated_index.get(func_id).copied()
96    }
97
98    pub(crate) fn is_entry_point(&self, function: FunctionId) -> bool {
99        function == self.main_id || self.functions[&function].runtime().is_entry_point()
100    }
101
102    /// Collect all the global value IDs used per function.
103    pub(crate) fn used_globals_in_functions(&self) -> HashMap<FunctionId, HashSet<ValueId>> {
104        fn add_value_to_globals_if_global(
105            function: &Function,
106            value_id: ValueId,
107            used_globals: &mut HashSet<ValueId>,
108        ) {
109            if !function.dfg.is_global(value_id) {
110                return;
111            }
112
113            if !used_globals.insert(value_id) {
114                return;
115            }
116
117            // If we found a new global, its value could be an instruction that points to other globals.
118            let globals = &function.dfg.globals;
119            if let Value::Instruction { instruction, .. } = globals[value_id] {
120                let instruction = &globals[instruction];
121                instruction.for_each_value(|value_id| {
122                    add_value_to_globals_if_global(function, value_id, used_globals);
123                });
124            }
125        }
126
127        let mut used_globals = HashMap::default();
128
129        for (function_id, function) in &self.functions {
130            let mut used_globals_in_function = HashSet::default();
131
132            for call_data in &function.dfg.data_bus.call_data {
133                add_value_to_globals_if_global(
134                    function,
135                    call_data.array_id,
136                    &mut used_globals_in_function,
137                );
138            }
139
140            for block_id in function.reachable_blocks() {
141                let block = &function.dfg[block_id];
142                for instruction_id in block.instructions() {
143                    let instruction = &function.dfg[*instruction_id];
144                    instruction.for_each_value(|value_id| {
145                        add_value_to_globals_if_global(
146                            function,
147                            value_id,
148                            &mut used_globals_in_function,
149                        );
150                    });
151                }
152
153                block.unwrap_terminator().for_each_value(|value_id| {
154                    add_value_to_globals_if_global(
155                        function,
156                        value_id,
157                        &mut used_globals_in_function,
158                    );
159                });
160            }
161
162            used_globals.insert(*function_id, used_globals_in_function);
163        }
164
165        used_globals
166    }
167}

175#[cfg(test)]
176mod tests {
177    use crate::ssa::ir::map::Id;
178
179    use crate::ssa::ssa_gen::Ssa;
180    use crate::ssa::{
181        function_builder::FunctionBuilder,
182        ir::{instruction::BinaryOp, types::Type},
183    };
184
185    #[test]
186    fn serialization_roundtrip() {
187        let main_id = Id::test_new(0);
188
189        // Compiling main
190        let mut builder = FunctionBuilder::new("main".into(), main_id);
191        let v0 = builder.add_parameter(Type::field());
192
193        let one = builder.field_constant(1u128);
194        let three = builder.field_constant(3u128);
195
196        let v1 = builder.insert_binary(v0, BinaryOp::Add { unchecked: false }, one);
197        let v2 = builder.insert_binary(v1, BinaryOp::Mul { unchecked: false }, three);
198        builder.terminate_with_return(vec![v2]);
199
200        let ssa = builder.finish();
201        let serialized_ssa = &serde_json::to_string(&ssa).unwrap();
202        let deserialized_ssa: Ssa = serde_json::from_str(serialized_ssa).unwrap();
203        let actual_string = format!("{}", deserialized_ssa.print_without_locations());
204
205        let expected_string = "acir(inline) fn main f0 {\n  \
206        b0(v0: Field):\n    \
207            v3 = add v0, Field 1\n    \
208            v4 = mul v3, Field 3\n    \
209            return v4\n\
210        }\n";
211        assert_eq!(actual_string, expected_string);
212    }
