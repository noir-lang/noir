8/// A general Tree structure which is used in the SSA generation pass
9/// to represent both values and types which may be tuples.
10///
11/// Since the underlying SSA intermediate representation (IR) does not
12/// support tuples directly, they're instead represented as Tree::Branch
13/// nodes. For example, a single ssa value may be a Tree::Leaf(Value),
14/// while a tuple would be a Tree::Branch(values).
15#[derive(Debug, Clone)]
16pub(super) enum Tree<T> {
17    Branch(Vec<Tree<T>>),
18    Leaf(T),
19}
20
21/// A single value in ssa form. This wrapper enum is needed mostly to enable
22/// us to automatically create a Instruction::Load whenever a mutable variable
23/// is referenced.
24///
25/// Note that these values wrap the ValueIds
26/// used internally by functions in the ssa ir and should thus be isolated
27/// to a given function. If used outside their function of origin, the IDs
28/// would be invalid.
29#[derive(Debug, Clone)]
30pub(super) enum Value {
31    Normal(IrValueId),
32
33    /// A mutable variable that must be loaded as the given type before being used
34    Mutable(IrValueId, Type),
35}
36
37impl Value {
38    /// Evaluate a value, returning an IrValue from it.
39    /// This has no effect on Value::Normal, but any variables will
40    /// need to be loaded from memory
41    pub(super) fn eval(self, ctx: &mut FunctionContext) -> IrValueId {
42        match self {
43            Value::Normal(value) => value,
44            Value::Mutable(address, typ) => ctx.builder.insert_load(address, typ),
45        }
46    }
47
48    /// Evaluates the value, returning a reference to the mutable variable found within
49    /// if possible. Compared to .eval, this method will not load from self if it is Value::Mutable.
50    pub(super) fn eval_reference(self) -> IrValueId {
51        match self {
52            Value::Normal(value) => value,
53            Value::Mutable(address, _) => address,
54        }
55    }
56}

58/// A tree of values.
59///
60/// Compared to Value alone, the addition of being able to represent structs/tuples as
61/// a Tree::Branch means this type can hold any kind of value a frontend expression may return.
62/// This is why it is used as the return type for every codegen_* function in ssa_gen/mod.rs.
63pub(super) type Values = Tree<Value>;
64
65impl<T> Tree<T> {
66    /// Returns an empty tree node represented by a Branch with no branches
67    pub(super) fn empty() -> Self {
68        Tree::Branch(vec![])
69    }
70
71    /// Flattens the tree into a vector of each leaf value
72    pub(super) fn flatten(self) -> Vec<T> {
73        match self {
74            Tree::Branch(values) => values.into_iter().flat_map(Tree::flatten).collect(),
75            Tree::Leaf(value) => vec![value],
76        }
77    }
78
79    /// Returns the total amount of leaves in this tree
80    pub(super) fn count_leaves(&self) -> usize {
81        match self {
82            Tree::Branch(trees) => trees.iter().map(|tree| tree.count_leaves()).sum(),
83            Tree::Leaf(_) => 1,
84        }
85    }
86
87    /// Iterates over each Leaf node, calling f on each value within.
88    pub(super) fn for_each(self, mut f: impl FnMut(T)) {
89        self.for_each_helper(&mut f);
90    }
91
92    fn for_each_helper(self, f: &mut impl FnMut(T)) {
93        match self {
94            Tree::Branch(trees) => trees.into_iter().for_each(|tree| tree.for_each_helper(f)),
95            Tree::Leaf(value) => f(value),
96        }
97    }
98
99    /// Calls the given function on each leaf node, mapping this tree into a new one.
100    ///
101    /// Because the given function returns a `Tree<U>` rather than a U, it is possible
102    /// to use this function to turn Leaf nodes into either other Leaf nodes or even Branch nodes.
103    pub(super) fn map<U>(self, mut f: impl FnMut(T) -> Tree<U>) -> Tree<U> {
104        self.map_helper(&mut f)
105    }
106
107    fn map_helper<U>(self, f: &mut impl FnMut(T) -> Tree<U>) -> Tree<U> {
108        match self {
109            Tree::Branch(trees) => Tree::Branch(vecmap(trees, |tree| tree.map_helper(f))),
110            Tree::Leaf(value) => f(value),
111        }
112    }
113
114    /// Map two trees alongside each other.
115    /// This asserts each tree has the same internal structure.
116    pub(super) fn map_both<U, R>(
117        &self,
118        other: Tree<U>,
119        mut f: impl FnMut(T, U) -> Tree<R>,
120    ) -> Tree<R>
121    where
122        T: std::fmt::Debug + Clone,
123        U: std::fmt::Debug,
124    {
125        self.map_both_helper(other, &mut f)
126    }
127
128    fn map_both_helper<U, R>(&self, other: Tree<U>, f: &mut impl FnMut(T, U) -> Tree<R>) -> Tree<R>
129    where
130        T: std::fmt::Debug + Clone,
131        U: std::fmt::Debug,
132    {
133        match (self, other) {
134            (Tree::Branch(self_trees), Tree::Branch(other_trees)) => {
135                assert_eq!(self_trees.len(), other_trees.len());
136                let trees = self_trees.iter().zip(other_trees);
137                Tree::Branch(vecmap(trees, |(l, r)| l.map_both_helper(r, f)))
138            }
139            (Tree::Leaf(self_value), Tree::Leaf(other_value)) => f(self_value.clone(), other_value),
140            other => panic!("Found unexpected tree combination during SSA: {other:?}"),
141        }
142    }
143
144    /// Unwraps this Tree into the value of the leaf node. Panics if
145    /// this Tree is a Branch
146    pub(super) fn into_leaf(self) -> T {
147        match self {
148            Tree::Branch(_) => panic!("into_leaf called on a Tree::Branch"),
149            Tree::Leaf(value) => value,
150        }
151    }
152}

166// Specialize this impl just to give a better name for this function
167impl Tree<Type> {
168    /// Returns the size of the type in terms of the number of FieldElements it contains.
169    /// Non-field types like functions and references are also counted as 1 FieldElement.
170    pub(super) fn size_of_type(&self) -> usize {
171        self.count_leaves()
172    }
173}

175impl Tree<Value> {
176    /// Flattens and evaluates this `Tree<Value>` into a list of ir values
177    /// for return statements, branching instructions, or function parameters.
178    pub(super) fn into_value_list(self, ctx: &mut FunctionContext) -> Vec<IrValueId> {
179        vecmap(self.flatten(), |value| value.eval(ctx))
180    }
