1//! The goal of the "remove enable side effects" optimization pass is to delay any
2//! [Instruction::EnableSideEffectsIf] instructions in ACIR functions such that they cover
3//! the minimum number of instructions possible.
4//!
5//! The pass works as follows:
6//! - Insert instructions until an [Instruction::EnableSideEffectsIf] is encountered, save this
7//!   [InstructionId][crate::ssa::ir::instruction::InstructionId].
8//! - Continue inserting instructions until either
9//!     - Another [Instruction::EnableSideEffectsIf] is encountered, if so then drop the previous
10//!       [InstructionId][crate::ssa::ir::instruction::InstructionId] in favour of this one.
11//!     - An [Instruction] that is affected by the side-effects variable is encountered, if so
12//!       then insert the currently saved [Instruction::EnableSideEffectsIf] before the
13//!       [Instruction]. Continue inserting instructions until the next
14//!       [Instruction::EnableSideEffectsIf] is encountered.
15//!
16//! The pass will also remove redundant [Instruction::EnableSideEffectsIf] instructions,
17//! for example if two consecutive [Instruction::EnableSideEffectsIf] instructions have the same
18//! condition.
19//!
20//! This pass doesn't run in Brillig functions as [Instruction::EnableSideEffectsIf] is not allowed
21//! in Brillig functions.
22//!
23//! ## Preconditions:
24//! - This pass must run once ACIR functions only have one basic block.
25
26use acvm::{FieldElement, acir::AcirField};
27
28use crate::ssa::{
29    ir::{dfg::DataFlowGraph, function::Function, instruction::Instruction, types::NumericType},
30    ssa_gen::Ssa,
31};
32
33impl Ssa {
34    /// See [`remove_enable_side_effects`][self] module for more information.
35    #[tracing::instrument(level = "trace", skip(self))]
36    pub(crate) fn remove_enable_side_effects(mut self) -> Ssa {
37        for function in self.functions.values_mut() {
38            #[cfg(debug_assertions)]
39            remove_enable_side_effects_pre_check(function);
40
41            function.remove_enable_side_effects();
42        }
43        self
44    }
45}
46
47impl Function {
48    fn remove_enable_side_effects(&mut self) {
49        if self.runtime().is_brillig() {
50            // Brillig functions do not make use of the `EnableSideEffects` instruction so are unaffected by this pass.
51            return;
52        }
53
54        let one = self.dfg.make_constant(FieldElement::one(), NumericType::bool());
55        let mut active_condition = one;
56        let mut last_side_effects_enabled_instruction = None;
57
58        let mut previous_block = None;
59
60        self.simple_optimization(|context| {
61            if Some(context.block_id) != previous_block {
62                active_condition = one;
63                last_side_effects_enabled_instruction = None;
64            }
65            previous_block = Some(context.block_id);
66
67            let instruction = context.instruction();
68
69            // If we run into another `Instruction::EnableSideEffectsIf` before encountering any
70            // instructions with side effects then we can drop the instruction we're holding and
71            // continue with the new `Instruction::EnableSideEffectsIf`.
72            if let Instruction::EnableSideEffectsIf { condition } = instruction {
73                // If this instruction isn't changing the currently active condition then we can ignore it.
74                if active_condition == *condition {
75                    context.remove_current_instruction();
76                    return;
77                }
78                active_condition = *condition;
79
80                // If we're seeing an `enable_side_effects u1 1` then we want to insert it immediately.
81                // This is because we want to maximize the effect it will have.
82                let condition_is_one = context
83                    .dfg
84                    .get_numeric_constant(*condition)
85                    .is_some_and(|condition| condition.is_one());
86
87                if condition_is_one {
88                    last_side_effects_enabled_instruction = None;
89                    context.insert_current_instruction();
90                } else {
91                    last_side_effects_enabled_instruction = Some(context.instruction_id);
92                    context.remove_current_instruction();
93                }
94                return;
95            }
96
97            if should_insert_side_effects_before_instruction(instruction, context.dfg)
98                && let Some(enable_side_effects_instruction_id) =
99                    last_side_effects_enabled_instruction.take()
100            {
101                context.insert_instruction_by_id(enable_side_effects_instruction_id);
102            }
103        });
104    }
105}
106
107/// Decide we should insert any pending side effect variable before we insert
108/// a particular instruction into the SSA.
109fn should_insert_side_effects_before_instruction(
110    instruction: &Instruction,
111    dfg: &DataFlowGraph,
112) -> bool {
113    // Constrain instructions don't need ACIR predicates, because the variables
114    // they operate on have the side effects incorporated into them,
115    // however they can later be turned into a ConstrainNotEqual instruction,
116    // which _does_ need an ACIR predicate. If we don't require the side effect
117    // variable for Constrain, then we might lose it and end up with a disabled
118    // constrain, as it could inherit some unintended side effect.
119    if matches!(instruction, Instruction::Constrain(..)) {
120        return true;
121    }
122    // If we hit an instruction which is affected by the side effects var then we must insert the
123    // `Instruction::EnableSideEffectsIf` before we insert this new instruction.
124    instruction.requires_acir_gen_predicate(dfg)
125}

127/// Pre-check condition for remove_enable_side_effects.
128///
129/// Panics if:
130///   - The CFG has not been flattened for ACIR functions.
131#[cfg(debug_assertions)]
132fn remove_enable_side_effects_pre_check(function: &Function) {
133    if function.runtime().is_acir() {
134        // flatten_cfg must have run
135        super::checks::assert_cfg_is_flattened(function);
136    }
137}

561    #[test]
562    fn keep_enable_side_effects_for_recursive_aggregation() {
563        // RecursiveAggregation uses the `current_side_effects_enabled_var` as its predicate
564        // during ACIR generation. If `remove_enable_side_effects` drops the `EnableSideEffectsIf`
565        // before a recursive_aggregation call, ACIR gen injects a stale predicate, which can
566        // silently disable recursive verification constraints.
567        let src = r#"
568        acir(inline) predicate_pure fn main f0 {
569          b0(v0: u1, v1: [Field; 1], v2: [Field; 1], v3: [Field; 1], v4: Field):
570            v5 = not v0
571            enable_side_effects v0
572            constrain Field 1 != Field 0
573            enable_side_effects v5
574            call recursive_aggregation(v1, v2, v3, v4, u32 0)
575            enable_side_effects u1 1
576            return
577        }
578        "#;
579        assert_ssa_does_not_change(src, Ssa::remove_enable_side_effects);
580    }
