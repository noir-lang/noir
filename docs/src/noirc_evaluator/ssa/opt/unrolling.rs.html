1//! This file contains the loop unrolling pass for the new SSA IR.
2//!
3//! This pass is divided into a few steps:
4//! 1. Find all loops in the program (`find_all_loops`)
5//! 2. For each loop:
6//!    1. If the loop is in our list of loops that previously failed to unroll, skip it.
7//!    2. If we have previously modified any of the blocks in the loop,
8//!       restart from step 1 to refresh the context.
9//!    3. If not, try to unroll the loop. If successful, remember the modified
10//!       blocks. If unsuccessful either error if the abort_on_error flag is set,
11//!       or otherwise remember that the loop failed to unroll and leave it unmodified.
12//!
13//! Note that this pass also often creates superfluous jmp instructions in the
14//! program that will need to be removed by a later simplify CFG pass.
15//!
16//! ACIR/Brillig differences within this pass:
17//!   - Brillig functions may contain loops using `continue` or `break` which this pass does not
18//!     support the unrolling of (running the pass on such functions is not an error).
19//!   - Brillig functions only have small loops unrolled, where a small loop is defined as a loop
20//!     which, when unrolled, is estimated to have the same or fewer total instructions as it
21//!     has when not unrolled.
22//!   - Unrolling may be reverted for brillig functions if the increase in instruction count is
23//!     greater than `max_bytecode_increase_percent` (if set).
24//!   - Differing post-conditions (see below).
25//!
26//! Relevance to other passes:
27//!   - Loop unrolling is a required pass for constrained code (ACIR functions) since ACIR itself
28//!     does not contain any branching constructs.
29//!   - Loop unrolling must occur before flattening on ACIR functions
30//!   - Since unrolling may fail on loops that fail to unroll, any simplification passes before it
31//!     which affect loop bounds may affect which code fails to compile. One important optimization
32//!     in this category is mem2reg which may simplify mutable variables, including those potentially
33//!     used as loop bounds, into a form which loop unrolling may better identify.
34//!
35//! Conditions:
36//!   - Pre-condition: All loop headers have a single induction variable.
37//!   - Pre-condition: The SSA must be optimized to a point at which loop bounds are known.
38//!     Some passes such as inlining and mem2reg are de-facto required before running this pass on arbitrary noir code.
39//!   - Post-condition (ACIR-only): All loops in ACIR functions should be unrolled when this pass is
40//!     completed successfully. Any loops that are not unrolled (e.g. because of a mutable variable
41//!     used in the loop condition whose value is unknown) will result in an error.
42//!   - Post-condition (Brillig-only): If `max_bytecode_increase_percent` is set, the instruction count
43//!     of each function should increase by no more than that percentage compared to before the pass.

69/// Maximum number of instructions (after unrolling) for Brillig loops to be
70/// force-unrolled regardless of the cost model. Loops with no function calls,
71/// constant bounds, and no breaks whose unrolled instruction count is at or
72/// below this threshold will always be unrolled.
73pub const FORCE_UNROLL_THRESHOLD: usize = 32;

75impl Ssa {
76    /// Loop unrolling can return errors, since ACIR functions need to be fully unrolled.
77    /// This meta-pass will keep trying to unroll loops and simplifying the SSA until no more errors are found.
78    ///
79    /// The `max_bytecode_incr_pct`, when given, is used to limit the growth of the Brillig bytecode size
80    /// after unrolling small loops to some percentage of the original loop. For example a value of 150 would
81    /// mean the new loop can be 150% (ie. 2.5 times) larger than the original loop. It will still contain
82    /// fewer SSA instructions, but that can still result in more Brillig opcodes.
83    ///
84    /// The `force_unroll_threshold` overrides the default threshold for force-unrolling
85    /// small Brillig loops. Set to 0 to disable force-unrolling.
86    #[tracing::instrument(level = "trace", skip(self))]
87    pub(crate) fn unroll_loops_iteratively(
88        mut self,
89        max_bytecode_increase_percent: Option<i32>,
90        force_unroll_threshold: usize,
91    ) -> Result<Ssa, RuntimeError> {
92        for function in self.functions.values_mut() {
93            let is_brillig = function.runtime().is_brillig();
94
95            // Take a snapshot in case we have to restore it.
96            let orig_function =
97                (max_bytecode_increase_percent.is_some() && is_brillig).then(|| function.clone());
98
99            // We must be able to unroll ACIR loops at this point, so exit on failure to unroll.
100            let has_unrolled = function.unroll_loops_iteratively(force_unroll_threshold)?;
101
102            // Check if the size increase is acceptable
103            // This is here now instead of in `Function::unroll_loops_iteratively` because we'd need
104            // more finessing to convince the borrow checker that it's okay to share a read-only reference
105            // to the globals and a mutable reference to the function at the same time, both part of the `Ssa`.
106            if has_unrolled
107                && is_brillig
108                && let Some(max_incr_pct) = max_bytecode_increase_percent
109            {
110                let orig_function = orig_function.expect("took snapshot to compare");
111                let new_size = function.num_instructions();
112                let orig_size = orig_function.num_instructions();
113                if !is_new_size_ok(orig_size, new_size, max_incr_pct) {
114                    *function = orig_function;
115                }
116            }
117        }
118        Ok(self)
119    }
120}

122impl Function {
123    /// Try to unroll loops in the function.
124    ///
125    /// Returns an `Err` if it cannot be done, for example because the loop bounds
126    /// cannot be determined at compile time. This can happen during pre-processing,
127    /// but it should still leave the function in a partially unrolled, but valid state.
128    ///
129    /// If successful, returns a flag indicating whether any loops have been unrolled.
130    ///
131    /// The `force_unroll_threshold` overrides the default threshold for
132    /// force-unrolling small Brillig loops.
133    pub(super) fn unroll_loops_iteratively(
134        &mut self,
135        force_unroll_threshold: usize,
136    ) -> Result<bool, RuntimeError> {
137        // Try to unroll loops first:
138        let (mut has_unrolled, mut unroll_errors) = self.try_unroll_loops(force_unroll_threshold);
139
140        // Keep unrolling until no more errors are found
141        while !unroll_errors.is_empty() {
142            let prev_unroll_err_count = unroll_errors.len();
143
144            // Simplify the SSA before retrying
145            simplify_between_unrolls(self);
146
147            // Unroll again
148            let (new_unrolled, new_errors) = self.try_unroll_loops(force_unroll_threshold);
149            unroll_errors = new_errors;
150            has_unrolled |= new_unrolled;
151
152            // If we didn't manage to unroll any more loops, exit
153            if unroll_errors.len() >= prev_unroll_err_count {
154                return Err(unroll_errors.swap_remove(0));
155            }
156        }
157
158        Ok(has_unrolled)
159    }
160
161    /// Unroll all loops within the function.
162    /// Any loops which fail to be unrolled (due to using non-constant indices) will be unmodified.
163    /// Returns a flag indicating whether any blocks have been modified.
164    ///
165    /// Loop unrolling in brillig can lead to a code explosion currently.
166    /// This can also be true for ACIR, but we have no alternative to unrolling in ACIR.
167    /// Brillig also generally prefers smaller code rather than faster code,
168    /// so we only attempt to unroll small loops, which we decide on a case-by-case basis.
169    fn try_unroll_loops(&mut self, force_unroll_threshold: usize) -> (bool, Vec<RuntimeError>) {
170        // The loops that failed to be unrolled so that we do not try to unroll them again.
171        // Each loop is identified by its header block id.
172        let mut failed_to_unroll = HashSet::new();
173        // The reasons why loops in the above set failed to unroll.
174        let mut unroll_errors = vec![];
175        let mut has_unrolled = false;
176
177        // Repeatedly find all loops as we unroll outer loops and go towards nested ones.
178        loop {
179            let order = if self.runtime().is_brillig() {
180                LoopOrder::InsideOut
181            } else {
182                LoopOrder::OutsideIn
183            };
184            let mut loops = Loops::find_all(self, order);
185
186            // Blocks which were part of loops we unrolled. Nested loops are included in the
187            // outer loops, so if an outer loop is unrolled, we have to restart looking for
188            // the nested ones.
189            let mut modified_blocks = HashSet::new();
190            let mut needs_refresh = false;
191
192            while let Some(next_loop) = loops.yet_to_unroll.pop() {
193                // If we've previously modified a block in this loop we need to refresh.
194                // This happens any time we have nested loops.
195                if next_loop.blocks.iter().any(|block| modified_blocks.contains(block)) {
196                    needs_refresh = true;
197                    continue;
198                }
199
200                // Don't try to unroll the loop again if it is known to fail
201                let result = if failed_to_unroll.contains(&next_loop.header) {
202                    LoopUnrollResult::Skipped
203                } else {
204                    self.try_unroll_loop(next_loop, &loops, force_unroll_threshold)
205                };
206                match result {
207                    LoopUnrollResult::Skipped => continue,
208                    LoopUnrollResult::Failed(header, error) => {
209                        failed_to_unroll.insert(header);
210                        unroll_errors.push(error);
211                    }
212                    LoopUnrollResult::Unrolled(blocks) => {
213                        has_unrolled = true;
214                        modified_blocks.extend(blocks);
215                    }
216                }
217            }
218
219            // If we didn't need to refresh, we're done
220            if !needs_refresh {
221                break;
222            }
223        }
224        (has_unrolled, unroll_errors)
225    }
226
227    /// Try to unroll a single loop.
228    ///
229    /// Returns the result: whether the loop was skipped, failed, or unrolled.
230    fn try_unroll_loop(
231        &mut self,
232        loop_: Loop,
233        loops: &Loops,
234        force_unroll_threshold: usize,
235    ) -> LoopUnrollResult {
236        // Only unroll small loops in Brillig.
237        if self.runtime().is_brillig()
238            && !loop_.should_unroll_in_brillig(self, &loops.cfg, force_unroll_threshold)
239        {
240            return LoopUnrollResult::Skipped;
241        }
242
243        // Check if we will be able to unroll this loop, before starting to modify the blocks.
244        if loop_.has_const_back_edge_induction_value(&self.dfg) {
245            // Don't try to unroll this.
246            // If this is Brillig, we can still evaluate this loop at runtime.
247            if self.runtime().is_acir() {
248                return LoopUnrollResult::Failed(
249                    loop_.header,
250                    RuntimeError::UnknownLoopBound { call_stack: CallStack::new() },
251                );
252            }
253            return LoopUnrollResult::Skipped;
254        }
255
256        // Try to unroll.
257        match loop_.unroll(self, &loops.cfg) {
258            Ok(_) => LoopUnrollResult::Unrolled(loop_.blocks),
259            Err(call_stack) => LoopUnrollResult::Failed(
260                loop_.header,
261                RuntimeError::UnknownLoopBound { call_stack },
262            ),
263        }
264    }
265}

277/// Describe the blocks that constitute up a loop.
278#[derive(Debug)]
279pub(crate) struct Loop {
280    /// The header block of a loop is the block which dominates all the
281    /// other blocks in the loop.
282    pub(crate) header: BasicBlockId,
283
284    /// The start of the back_edge n -> d is the block n at the end of
285    /// the loop that jumps back to the header block d which restarts the loop.
286    pub(crate) back_edge_start: BasicBlockId,
287
288    /// All the blocks contained within the loop, including `header` and `back_edge_start`.
289    pub(crate) blocks: BTreeSet<BasicBlockId>,
290}

292/// Order in which loops should be processed.
293#[derive(Debug, Clone, Copy, PartialEq, Eq)]
294pub(crate) enum LoopOrder {
295    /// Process inner (smaller) loops first, then outer loops.
296    /// Used for Brillig which can tolerate inner loops that reference outer induction variables.
297    InsideOut,
298    /// Process outer (larger) loops first, then inner loops.
299    /// Used for ACIR which cannot tolerate inner loops that reference outer induction variables,
300    /// so outer loops must be unrolled first.
301    OutsideIn,
302}

304/// All the unrolled loops in the SSA.
305pub(crate) struct Loops {
306    /// Loops that haven't been unrolled yet, which is all the loops currently in the CFG.
307    pub(crate) yet_to_unroll: Vec<Loop>,
308    /// The CFG so we can query the predecessors of blocks when needed.
309    pub(crate) cfg: ControlFlowGraph,
310    /// The [DominatorTree] used during the discovery of loops.
311    pub(crate) dom: DominatorTree,
312}

314impl Loops {
315    /// Find all loops in the program by finding a node that dominates any predecessor node.
316    /// The edge where this happens will be the back-edge of the loop.
317    ///
318    /// For example consider the following SSA of a basic loop:
319    /// ```text
320    /// main():
321    ///   v0 = ... start ...
322    ///   v1 = ... end ...
323    ///   jmp loop_entry(v0)
324    /// loop_entry(i: Field):
325    ///   v2 = lt i v1
326    ///   jmpif v2, then: loop_body, else: loop_end
327    /// loop_body():
328    ///   v3 = ... body ...
329    ///   v4 = add 1, i
330    ///   jmp loop_entry(v4)
331    /// loop_end():
332    /// ```
333    ///
334    /// The CFG will look something like this:
335    /// ```text
336    /// main
337    ///   ↓
338    /// loop_entry ←---↰
339    ///   ↓        ↘   |
340    /// loop_end    loop_body
341    /// ```
342    /// `loop_entry` has two predecessors: `main` and `loop_body`, and it dominates `loop_body`.
343    ///
344    /// Returns all groups of blocks that look like a loop, even if we might not be able to unroll them,
345    /// which we can use to check whether we were able to unroll all blocks.
346    pub(crate) fn find_all(function: &Function, order: LoopOrder) -> Self {
347        let cfg = ControlFlowGraph::with_function(function);
348        let post_order = PostOrder::with_function(function);
349        let mut dom_tree = DominatorTree::with_cfg_and_post_order(&cfg, &post_order);
350
351        let mut loops = vec![];
352
353        // Iterating over blocks in reverse-post-order, ie. forward order, just because it's already available.
354        for block in post_order.into_vec_reverse() {
355            for predecessor in cfg.predecessors(block) {
356                // In the above example, we're looking for when `block` is `loop_entry` and `predecessor` is `loop_body`.
357                if dom_tree.dominates(block, predecessor) {
358                    // predecessor -> block is the back-edge of a loop
359                    loops.push(Loop::find_blocks_in_loop(block, predecessor, &cfg));
360                }
361            }
362        }
363
364        match order {
365            LoopOrder::InsideOut => {
366                // Sort by block size descending so we pop and unroll smaller, inner loops first.
367                // This is safe for Brillig because if inner loop bounds depend on an outer
368                // induction variable, `get_const_bounds` returns None, `is_small_loop` returns
369                // false, and we skip it. After unrolling inner loops, outer loops have simpler
370                // bodies and more accurate cost estimates for the `is_small_loop` heuristic.
371                loops.sort_by_key(|loop_| std::cmp::Reverse(loop_.blocks.len()));
372            }
373            LoopOrder::OutsideIn => {
374                // Sort by block size ascending so we unroll larger, outer loops of nested loops first.
375                // This is needed because inner loops may use the induction variable from their
376                // outer loops in their loop range.
377                loops.sort_by_key(|loop_| loop_.blocks.len());
378            }
379        }
380
381        Self { yet_to_unroll: loops, cfg, dom: dom_tree }
382    }
383}

385impl Loop {
386    /// Return each block that is in a loop starting in the given header block.
387    /// Expects back_edge_start -> header to be the back edge of the loop.
388    pub(crate) fn find_blocks_in_loop(
389        header: BasicBlockId,
390        back_edge_start: BasicBlockId,
391        cfg: &ControlFlowGraph,
392    ) -> Self {
393        let mut blocks = BTreeSet::default();
394        // Insert the header so we don't go past it when traversing backwards from the back-edge.
395        blocks.insert(header);
396
397        let mut insert = |block, stack: &mut Vec<BasicBlockId>| {
398            if !blocks.contains(&block) {
399                blocks.insert(block);
400                stack.push(block);
401            }
402        };
403
404        // Starting from the back edge of the loop, enqueue each predecessor of this block until we reach the header.
405        let mut stack = vec![];
406        insert(back_edge_start, &mut stack);
407
408        while let Some(block) = stack.pop() {
409            for predecessor in cfg.predecessors(block) {
410                insert(predecessor, &mut stack);
411            }
412        }
413
414        Self { header, back_edge_start, blocks }
415    }
416
417    /// Check that the loop does not end with a constant value passed to the header
418    /// from the back-edge, which would result in a loop we would never finish unrolling.
419    ///
420    /// This can happen if constant folding replaces a variable with a constant it is
421    /// constrained to equal (which doesn't even have to fall into the loop bounds).
422    ///
423    /// For example:
424    /// ```text
425    /// brillig(inline) predicate_pure fn main f0 {
426    ///   b0():
427    ///     jmp b1(u32 1)                // Pre-header
428    ///   b1(v0: u32):                   // Header
429    ///     v3 = lt v0, u32 20
430    ///     jmpif v3 then: b2, else: b3
431    ///   b2():                          // Back edge
432    ///     constrain v0 == u32 1        // Constrain the induction variable to a known value.
433    ///     jmp b1(u32 2)                // `v1 = unchecked_add v0, u32 1; jmp b1(v1)` replaced by `jmp b1 (1+1)`
434    ///   b3():
435    ///     return
436    /// }
437    /// ```
438    fn has_const_back_edge_induction_value(&self, dfg: &DataFlowGraph) -> bool {
439        let back_edge = &dfg[self.back_edge_start];
440        let Some(TerminatorInstruction::Jmp { destination, arguments, .. }) =
441            back_edge.terminator()
442        else {
443            unreachable!("the back edge is expected to end in a `Jmp`");
444        };
445        assert_eq!(*destination, self.header, "back edge goes to the header");
446        assert_eq!(arguments.len(), 1);
447        dfg.get_numeric_constant(arguments[0]).is_some()
448    }
449
450    /// Check if the loop header has a constant zero jump condition, which indicates an empty loop.
451    ///
452    /// This can happen if a jump condition has been simplified out.
453    ///
454    /// For example:
455    /// ```text
456    /// brillig(inline) predicate_pure fn main f0 {
457    ///   b0():
458    ///     jmp b1(u32 10)               // Pre-header
459    ///   b1(v0: u32):                   // Header
460    ///     // v3 = lt v0, u32 0         // Simplified to `u1 0`
461    ///     jmpif u1 0 then: b2, else: b3
462    ///   b2():                          // Back edge
463    ///     v1 = unchecked_add v0, u32 1 // Increment induction value
464    ///     jmp b1(v1)
465    ///   b3():
466    ///     return
467    /// }
468    /// ```
469    fn has_const_zero_jump_condition(&self, dfg: &DataFlowGraph) -> bool {
470        let header = &dfg[self.header];
471        let Some(TerminatorInstruction::JmpIf { condition, .. }) = header.terminator() else {
472            return false;
473        };
474        let Some(condition) = dfg.get_numeric_constant(*condition) else {
475            return false;
476        };
477        condition.is_zero()
478    }
479
480    /// Find the lower bound of the loop in the pre-header and return it
481    /// if it's a numeric constant, which it will be if the previous SSA
482    /// steps managed to inline it.
483    ///
484    /// Consider the following example of a `for i in 0..4` loop:
485    /// ```text
486    /// brillig(inline) fn main f0 {
487    ///   b0(v0: u32):                  // Pre-header
488    ///     ...
489    ///     jmp b1(u32 0)               // Lower-bound
490    ///   b1(v1: u32):                  // Induction variable
491    ///     v5 = lt v1, u32 4
492    ///     jmpif v5 then: b3, else: b2
493    /// ```
494    fn get_const_lower_bound(
495        &self,
496        dfg: &DataFlowGraph,
497        pre_header: BasicBlockId,
498    ) -> Option<IntegerConstant> {
499        let jump_value = get_induction_variable(dfg, pre_header).ok()?;
500        dfg.get_integer_constant(jump_value)
501    }
502
503    /// Find the upper bound of the loop in the loop header and return it
504    /// if it's a numeric constant, which it will be if the previous SSA
505    /// steps managed to inline it.
506    ///
507    /// Consider the following example of a `for i in 0..4` loop:
508    /// ```text
509    /// brillig(inline) fn main f0 {
510    ///   b0(v0: u32):
511    ///     ...
512    ///     jmp b1(u32 0)
513    ///   b1(v1: u32):                  // Loop header
514    ///     v5 = lt v1, u32 4           // Upper bound
515    ///     jmpif v5 then: b3, else: b2
516    /// ```
517    fn get_const_upper_bound(
518        &self,
519        dfg: &DataFlowGraph,
520        pre_header: BasicBlockId,
521    ) -> Option<IntegerConstant> {
522        let header = &dfg[self.header];
523
524        // If the header has no parameters then this must be a `loop` or `while`.
525        if header.parameters().is_empty() {
526            return None;
527        }
528
529        let instructions = header.instructions();
530        if instructions.is_empty() {
531            // If the loop condition is constant, the loop header will be
532            // simplified to a simple jump.
533            if self.has_const_zero_jump_condition(dfg) {
534                // There are cases where the upper bound jmpif degenerates into a constant `false`;
535                // in that case we can just return the `lower` to emulate a known empty loop.
536                return self.get_const_lower_bound(dfg, pre_header);
537            } else {
538                return None;
539            };
540        }
541
542        if instructions.len() != 1 {
543            // The header should just compare the induction variable and jump.
544            // If that's not the case, this might be a `loop` and not a `for` loop.
545            return None;
546        }
547
548        match &dfg[instructions[0]] {
549            Instruction::Binary(Binary { lhs: _, operator: BinaryOp::Lt, rhs }) => {
550                dfg.get_integer_constant(*rhs)
551            }
552            Instruction::Binary(Binary { lhs: _, operator: BinaryOp::Eq, rhs }) => {
553                // `for i in 0..1` is turned into:
554                // b1(v0: u32):
555                //   v12 = eq v0, u32 0
556                //   jmpif v12 then: b2, else: b3
557                dfg.get_integer_constant(*rhs).map(|c| c.inc())
558            }
559            Instruction::Not(_) => {
560                // We simplify equality operations with booleans like `(boolean == false)` into `!boolean`.
561                // Thus, using a u1 in a loop bound can possibly lead to a Not instruction
562                // as a loop header's jump condition.
563                //
564                // `for i in 0..1` is turned into:
565                //  b1(v0: u1):
566                //    v2 = eq v0, u32 0
567                //    jmpif v2 then: b2, else: b3
568                //
569                // Which is further simplified into:
570                //  b1(v0: u1):
571                //    v2 = not v0
572                //    jmpif v2 then: b2, else: b3
573                Some(IntegerConstant::Unsigned { value: 1, bit_size: 1 })
574            }
575            Instruction::Cast(_, _) => {
576                // A cast of a constant would already be simplified
577                None
578            }
579            other => panic!("Unexpected instruction in header: {other:?}"),
580        }
581    }

583    /// Get the lower and upper bounds of the loop if both are constant numeric values.
584    pub(super) fn get_const_bounds(
585        &self,
586        dfg: &DataFlowGraph,
587        pre_header: BasicBlockId,
588    ) -> Option<(IntegerConstant, IntegerConstant)> {
589        let lower = self.get_const_lower_bound(dfg, pre_header)?;
590        let upper = self.get_const_upper_bound(dfg, pre_header)?;
591        Some((lower, upper))
592    }

594    /// Unroll a single loop in the function.
595    /// Returns Ok(()) if it succeeded, Err(callstack) if it failed,
596    /// where the callstack indicates the location of the instruction
597    /// that could not be processed, or empty if such information was
598    /// not available.
599    ///
600    /// Consider this example:
601    /// ```text
602    /// main():
603    ///   v0 = 0
604    ///   v1 = 2
605    ///   jmp loop_entry(v0)
606    /// loop_entry(i: Field):
607    ///   v2 = lt i, v1
608    ///   jmpif v2, then: loop_body, else: loop_end
609    /// ```
610    ///
611    /// The first step is to unroll the header by recognizing that jump condition
612    /// is a constant, which means it will go to `loop_body`:
613    /// ```text
614    /// main():
615    ///   v0 = 0
616    ///   v1 = 2
617    ///   v2 = lt v0, v1
618    ///   // jmpif v2, then: loop_body, else: loop_end
619    ///   jmp dest: loop_body
620    /// ```
621    ///
622    /// Following that we unroll the loop body, which is the next source, replace
623    /// the induction variable with the new value created in the body, and have
624    /// another go at the header.
625    /// ```text
626    /// main():
627    ///   v0 = 0
628    ///   v1 = 2
629    ///   v2 = lt v0, v1
630    ///   v3 = ... body ...
631    ///   v4 = add v0, u32 1
632    ///   jmp loop_entry(v4)
633    /// ```
634    ///
635    /// At the end we reach a point where the condition evaluates to 0 and we jump to the end.
636    /// ```text
637    /// main():
638    ///   v0 = 0
639    ///   v1 = 2
640    ///   v2 = lt v0, v1
641    ///   v3 = ... body ...
642    ///   v4 = add u32 1, v0
643    ///   v5 = lt v4, v1
644    ///   v6 = ... body ...
645    ///   v7 = add v4, u32 1
646    ///   v8 = lt v7, v1
647    ///   jmp loop_end
648    /// ```
649    ///
650    /// When e.g. `v8 = lt v7, v1` cannot be evaluated to a constant, the loop signals by returning `Err`
651    /// that a few SSA passes are required to evaluate and simplify these values.
652    fn unroll(&self, function: &mut Function, cfg: &ControlFlowGraph) -> Result<(), CallStack> {
653        let mut unroll_into = self.get_pre_header(function, cfg)?;
654        let mut jump_value = get_induction_variable(&function.dfg, unroll_into)?;
655
656        while let Some((context, loop_header_id)) =
657            self.unroll_header(function, unroll_into, jump_value)?
658        {
659            (unroll_into, jump_value) = context.unroll_loop_iteration(loop_header_id);
660        }
661
662        Ok(())
663    }

665    /// The loop pre-header is the block that comes before the loop begins. Generally a header block
666    /// is expected to have 2 predecessors: the pre-header and the final block of the loop which jumps
667    /// back to the beginning. Other predecessors can come from `break` or `continue`.
668    pub(super) fn get_pre_header(
669        &self,
670        function: &Function,
671        cfg: &ControlFlowGraph,
672    ) -> Result<BasicBlockId, CallStack> {
673        let mut pre_header = cfg
674            .predecessors(self.header)
675            .filter(|predecessor| *predecessor != self.back_edge_start)
676            .collect::<Vec<_>>();
677
678        if function.runtime().is_acir() {
679            assert_eq!(pre_header.len(), 1);
680            Ok(pre_header.remove(0))
681        } else if pre_header.len() == 1 {
682            Ok(pre_header.remove(0))
683        } else {
684            // We can come back into the header from multiple blocks, so we can't unroll this.
685            Err(CallStack::new())
686        }
687    }
688
689    /// Unrolls the header block of the loop. This is the block that dominates all other blocks in the
690    /// loop and contains the jmpif instruction that lets us know if we should continue looping.
691    /// Returns Some((iteration context, loop_header_id)) if we should perform another iteration.
692    fn unroll_header<'a>(
693        &'a self,
694        function: &'a mut Function,
695        unroll_into: BasicBlockId,
696        induction_value: ValueId,
697    ) -> Result<Option<(LoopIteration<'a>, BasicBlockId)>, CallStack> {
698        // We insert into a fresh block first and move instructions into the unroll_into block later
699        // only once we verify the jmpif instruction has a constant condition. If it does not, we can
700        // just discard this fresh block and leave the loop unmodified.
701        let fresh_block = function.dfg.make_block();
702
703        let mut context = LoopIteration::new(function, self, fresh_block, self.header);
704        let loop_header_id = context.source_block;
705        let source_block = &context.dfg()[loop_header_id];
706        assert_eq!(source_block.parameters().len(), 1, "Expected only 1 argument in loop header");
707
708        // Insert the current value of the loop induction variable into our context.
709        let first_param = source_block.parameters()[0];
710        context.inserter.try_map_value(first_param, induction_value);
711        // Copy over all instructions and a fresh terminator.
712        context.inline_instructions_from_block();
713        context.visited_blocks.insert(loop_header_id);
714
715        // Mutate the terminator if possible so that it points at the iteration block.
716        match context.dfg()[fresh_block].unwrap_terminator() {
717            TerminatorInstruction::JmpIf {
718                condition,
719                then_destination,
720                else_destination,
721                call_stack,
722            } => {
723                let condition = *condition;
724                let next_blocks = context.handle_jmpif(
725                    condition,
726                    *then_destination,
727                    *else_destination,
728                    *call_stack,
729                );
730
731                // If there is only 1 next block the jmpif evaluated to a single known block.
732                // This is the expected case and lets us know if we should loop again or not.
733                if next_blocks.len() == 1 {
734                    context.dfg_mut().inline_block(fresh_block, unroll_into);
735
736                    // The fresh block is gone now so we're committing to insert into the original
737                    // unroll_into block from now on.
738                    context.insert_block = unroll_into;
739
740                    // In the last iteration, `handle_jmpif` will have replaced `context.source_block`
741                    // with the `else_destination`, that is, the `loop_end`, which signals that we
742                    // have no more loops to unroll, because that block was not part of the loop itself,
743                    // ie. it wasn't between `loop_header` and `loop_body`. Otherwise we have the `loop_body`
744                    // in `source_block` and can unroll that into the destination.
745                    Ok(self
746                        .blocks
747                        .contains(&context.source_block)
748                        .then_some(context)
749                        .map(|iteration_context| (iteration_context, loop_header_id)))
750                } else {
751                    // If this case is reached the loop either uses non-constant indices or we need
752                    // another pass, such as mem2reg to resolve them to constants.
753                    Err(context.inserter.function.dfg.get_value_call_stack(condition))
754                }
755            }
756            other => unreachable!(
757                "Expected loop header to terminate in a JmpIf to the loop body, but found {other:?} instead"
758            ),
759        }
760    }

762    /// Find all reference values which were allocated before the pre-header.
763    ///
764    /// These are accessible inside the loop body, and they can be involved
765    /// in load/store operations that could be eliminated if we unrolled the
766    /// body into the pre-header.
767    ///
768    /// Consider this loop:
769    /// ```text
770    /// let mut sum = 0;
771    /// let mut arr = &[];
772    /// for i in 0..3 {
773    ///     sum = sum + i;
774    ///     arr.push_back(sum)
775    /// }
776    /// sum
777    /// ```
778    ///
779    /// The SSA has a load+store for the `sum` and a load+push for the `arr`:
780    /// ```text
781    /// b0(v0: u32):
782    ///   v2 = allocate -> &mut u32     // reference allocated for `sum`
783    ///   store u32 0 at v2             // initial value for `sum`
784    ///   v4 = allocate -> &mut u32     // reference allocated for the length of `arr`
785    ///   store u32 0 at v4             // initial length of `arr`
786    ///   inc_rc [] of u32              // storage for `arr`
787    ///   v6 = allocate -> &mut [u32]   // reference allocated to point at the storage of `arr`
788    ///   store [] of u32 at v6         // initial value for the storage of `arr`
789    ///   jmp b1(u32 0)                 // start looping from 0
790    /// b1(v1: u32):                    // `i` induction variable
791    ///   v8 = lt v1, u32 3             // loop until 3
792    ///   jmpif v8 then: b3, else: b2
793    /// b3():
794    ///   v11 = load v2 -> u32          // load `sum`
795    ///   v12 = add v11, v1             // add `i` to `sum`
796    ///   store v12 at v2               // store updated `sum`
797    ///   v13 = load v4 -> u32          // load length of `arr`
798    ///   v14 = load v6 -> [u32]        // load storage of `arr`
799    ///   v16, v17 = call vector_push_back(v13, v14, v12) -> (u32, [u32]) // builtin to push, will store to storage and length references
800    ///   v19 = add v1, u32 1           // increase `arr`
801    ///   jmp b1(v19)                   // back-edge of the loop
802    /// b2():                           // after the loop
803    ///   v9 = load v2 -> u32           // read final value of `sum`
804    /// ```
805    ///
806    /// We won't always find load _and_ store ops (e.g. the push above doesn't come with a store),
807    /// but it's likely that mem2reg could eliminate a lot of the loads we can find, so we can
808    /// use this as an approximation of the gains we would see.
809    fn find_pre_header_reference_values(
810        &self,
811        function: &Function,
812        cfg: &ControlFlowGraph,
813    ) -> Option<HashSet<ValueId>> {
814        // We need to traverse blocks from the pre-header up to the block entry point.
815        let pre_header = self.get_pre_header(function, cfg).ok()?;
816        let function_entry = function.entry_block();
817
818        // The algorithm in `find_blocks_in_loop` expects to collect the blocks between the header and the back-edge of the loop,
819        // but technically works the same if we go from the pre-header up to the function entry as well.
820        let blocks = Self::find_blocks_in_loop(function_entry, pre_header, cfg).blocks;
821
822        // Collect allocations in all blocks above the header.
823        let allocations = blocks.iter().flat_map(|block| {
824            let instructions = function.dfg[*block].instructions().iter();
825            instructions
826                .filter(|i| matches!(&function.dfg[**i], Instruction::Allocate))
827                // Get the value into which the allocation was stored.
828                .map(|i| function.dfg.instruction_result::<1>(*i)[0])
829        });
830
831        // Collect reference parameters of the function itself.
832        let params =
833            function.parameters().iter().filter(|p| function.dfg.value_is_reference(**p)).copied();
834
835        Some(params.chain(allocations).collect())
836    }

838    /// Count the number of load and store instructions of specific variables in the loop.
839    ///
840    /// Returns `(loads, stores)` in case we want to differentiate in the estimates.
841    fn count_loads_and_stores(
842        &self,
843        function: &Function,
844        refs: &HashSet<ValueId>,
845    ) -> (usize, usize) {
846        let mut loads = 0;
847        let mut stores = 0;
848        for block in &self.blocks {
849            for instruction in function.dfg[*block].instructions() {
850                match &function.dfg[*instruction] {
851                    Instruction::Load { address } if refs.contains(address) => {
852                        loads += 1;
853                    }
854                    Instruction::Store { address, .. } if refs.contains(address) => {
855                        stores += 1;
856                    }
857                    _ => {}
858                }
859            }
860        }
861        (loads, stores)
862    }
863
864    /// Count the number of instructions in the loop, including the terminating jumps.
865    fn count_all_instructions(&self, function: &Function) -> usize {
866        let iter = self.blocks.iter().map(|block| {
867            let block = &function.dfg[*block];
868            block.instructions().len() + usize::from(block.terminator().is_some())
869        });
870        iter.sum()
871    }

873    /// Count the number of increments to the induction variable.
874    /// It should be one, but it can be duplicated.
875    /// The increment should be in the block where the back-edge was found.
876    fn count_induction_increments(&self, function: &Function) -> usize {
877        let back = &function.dfg[self.back_edge_start];
878        let header = &function.dfg[self.header];
879        let induction_var = header.parameters()[0];
880
881        back.instructions()
882            .iter()
883            .filter(|instruction| {
884                let instruction = &function.dfg[**instruction];
885                matches!(instruction,
886                    Instruction::Binary(Binary { lhs, operator: BinaryOp::Add { .. }, rhs: _ })
887                        if *lhs == induction_var
888                )
889            })
890            .count()
891    }

893    /// Whether this loop should be unrolled when compiling to Brillig.
894    ///
895    /// A loop is unrolled if:
896    /// 1. It has constant bounds and no breaks (`boilerplate_stats` + `is_fully_executed`)
897    /// 2. AND either:
898    ///    a. The cost model predicts unrolling reduces code size (`is_small`), OR
899    ///    b. The total unrolled instruction count is within the force-unroll threshold
900    fn should_unroll_in_brillig(
901        &self,
902        function: &Function,
903        cfg: &ControlFlowGraph,
904        force_unroll_threshold: usize,
905    ) -> bool {
906        let threshold = force_unroll_threshold;
907        self.boilerplate_stats(function, cfg)
908            .map(|s| {
909                let force_unroll = s.unrolled_instructions() <= threshold;
910                (force_unroll || s.is_small()) && self.is_fully_executed(cfg)
911            })
912            .unwrap_or_default()
913    }

915    /// Collect boilerplate stats if we can figure out the upper and lower bounds of the loop,
916    /// and the loop doesn't have multiple back-edges from breaks and continues.
917    fn boilerplate_stats(
918        &self,
919        function: &Function,
920        cfg: &ControlFlowGraph,
921    ) -> Option<BoilerplateStats> {
922        let pre_header = self.get_pre_header(function, cfg).ok()?;
923        let (lower, upper) = self.get_const_bounds(&function.dfg, pre_header)?;
924        let refs = self.find_pre_header_reference_values(function, cfg)?;
925
926        let (loads, stores) = self.count_loads_and_stores(function, &refs);
927        let increments = self.count_induction_increments(function);
928        let all_instructions = self.count_all_instructions(function);
929
930        // Currently we don't iterate in reverse, so if upper <= lower it means 0 iterations.
931        let iterations: usize = upper
932            .reduce(
933                lower,
934                |u, l| u.saturating_sub(l).max(0) as usize,
935                |u, l| u.saturating_sub(l) as usize,
936            )
937            .unwrap_or_default();
938
939        Some(BoilerplateStats {
940            iterations,
941            loads,
942            stores,
943            increments,
944            all_instructions,
945            has_const_zero_jump_condition: self.has_const_zero_jump_condition(&function.dfg),
946        })
947    }
948}

950/// All the instructions in the following example are boilerplate:
951/// ```text
952/// brillig(inline) fn main f0 {
953///   b0(v0: u32):
954///     ...
955///     jmp b1(u32 0)
956///   b1(v1: u32):
957///     v5 = lt v1, u32 4
958///     jmpif v5 then: b3, else: b2
959///   b3():
960///     ...
961///     v11 = add v1, u32 1
962///     jmp b1(v11)
963///   b2():
964///     ...
965/// }
966/// ```
967#[derive(Debug)]
968struct BoilerplateStats {
969    /// Number of iterations in the loop.
970    iterations: usize,
971    /// Number of loads  pre-header references in the loop.
972    loads: usize,
973    /// Number of stores into pre-header references in the loop.
974    stores: usize,
975    /// Number of increments to the induction variable (might be duplicated).
976    increments: usize,
977    /// Number of instructions in the loop, including boilerplate,
978    /// but excluding the boilerplate which is outside the loop.
979    all_instructions: usize,
980    /// Indicate whether the comparison with the upper bound has been simplified out.
981    has_const_zero_jump_condition: bool,
982}
983
984impl BoilerplateStats {
985    /// Instruction count if we leave the loop as-is.
986    /// It's the instructions in the loop, plus the one to kick it off in the pre-header.
987    fn baseline_instructions(&self) -> usize {
988        self.all_instructions + 1
989    }
990
991    /// Estimated number of _useful_ instructions, which is the ones in the loop
992    /// minus all in-loop boilerplate.
993    fn useful_instructions(&self) -> usize {
994        // Two jumps + plus the comparison with the upper bound.
995        // This could be just 2 if the comparison has been simplified out.
996        let boilerplate = if self.has_const_zero_jump_condition { 2 } else { 3 };
997        // Be conservative and only assume that mem2reg gets rid of load followed by store.
998        // NB we have not checked that these are actual pairs.
999        let load_and_store = self.loads.min(self.stores) * 2;
1000        let total_boilerplate = self.increments + load_and_store + boilerplate;
1001        assert!(
1002            total_boilerplate <= self.all_instructions,
1003            "Boilerplate instructions exceed total instructions in loop"
1004        );
1005        self.all_instructions.saturating_sub(total_boilerplate)
1006    }
1007
1008    /// Estimated number of instructions if we unroll the loop.
1009    fn unrolled_instructions(&self) -> usize {
1010        self.useful_instructions() * self.iterations
1011    }
1012
1013    /// A small loop is where if we unroll it into the pre-header then considering the
1014    /// number of iterations we still end up with a smaller bytecode than if we leave
1015    /// the blocks in tact with all the boilerplate involved in jumping, and the extra
1016    /// reference access instructions.
1017    fn is_small(&self) -> bool {
1018        self.unrolled_instructions() < self.baseline_instructions()
1019    }
1020}

1022/// Return the induction value of the current iteration of the loop, from the given block's jmp arguments.
1023///
1024/// Expects the current block to terminate in `jmp h(N)` where h is the loop header and N is
1025/// a Field value. Returns an `Err` if this isn't the case.
1026///
1027/// Consider the following example:
1028/// ```text
1029/// main():
1030///   v0 = ... start ...
1031///   v1 = ... end ...
1032///   jmp loop_entry(v0)
1033/// loop_entry(i: Field):
1034///   ...
1035/// ```
1036/// We're looking for the terminating jump of the `main` predecessor of `loop_entry`.
1037fn get_induction_variable(dfg: &DataFlowGraph, block: BasicBlockId) -> Result<ValueId, CallStack> {
1038    match dfg[block].terminator() {
1039        Some(TerminatorInstruction::Jmp { arguments, call_stack: location, .. }) => {
1040            // This assumption will no longer be valid if e.g. mutable variables are represented as
1041            // block parameters. If that becomes the case we'll need to figure out which variable
1042            // is generally constant and increasing to guess which parameter is the induction
1043            // variable.
1044            if arguments.len() != 1 {
1045                // It is expected that a loop's induction variable is the only block parameter of the loop header.
1046                // If there's no variable this might be a `loop`.
1047                let call_stack = dfg.get_call_stack(*location);
1048                return Err(call_stack);
1049            }
1050
1051            let value = arguments[0];
1052            if dfg.get_numeric_constant(value).is_some() {
1053                Ok(value)
1054            } else {
1055                let call_stack = dfg.get_call_stack(*location);
1056                Err(call_stack)
1057            }
1058        }
1059        Some(terminator) => Err(dfg.get_call_stack(terminator.call_stack())),
1060        None => Err(CallStack::new()),
1061    }
1062}

1064/// The context object for each loop iteration.
1065/// Notably each loop iteration maps each loop block to a fresh, unrolled block.
1066struct LoopIteration<'f> {
1067    inserter: FunctionInserter<'f>,
1068    loop_: &'f Loop,
1069
1070    /// Maps pre-unrolled block ids from within the loop to new block ids of each loop
1071    /// block for each loop iteration.
1072    blocks: HashMap<BasicBlockId, BasicBlockId>,
1073
1074    /// Maps unrolled block ids back to the original source block ids
1075    original_blocks: HashMap<BasicBlockId, BasicBlockId>,
1076    visited_blocks: HashSet<BasicBlockId>,
1077
1078    /// Has `unroll_loop_iteration` reached the `loop_header_id`?
1079    encountered_loop_header: bool,
1080
1081    insert_block: BasicBlockId,
1082    source_block: BasicBlockId,
1083
1084    /// The induction value (and the block it was found in) is the new value for
1085    /// the variable traditionally called `i` on each iteration of the loop.
1086    /// This is None until we visit the block which jumps back to the start of the
1087    /// loop, at which point we record its value and the block it was found in.
1088    induction_value: Option<(BasicBlockId, ValueId)>,
1089}
1090
1091impl<'f> LoopIteration<'f> {
1092    fn new(
1093        function: &'f mut Function,
1094        loop_: &'f Loop,
1095        insert_block: BasicBlockId,
1096        source_block: BasicBlockId,
1097    ) -> Self {
1098        Self {
1099            inserter: FunctionInserter::new(function),
1100            loop_,
1101            insert_block,
1102            source_block,
1103            blocks: HashMap::default(),
1104            original_blocks: HashMap::default(),
1105            visited_blocks: HashSet::default(),
1106            encountered_loop_header: false,
1107
1108            induction_value: None,
1109        }
1110    }
1111
1112    /// Unroll a single iteration of the loop.
1113    ///
1114    /// Note that after unrolling a single iteration, the loop is _not_ in a valid state.
1115    /// It is expected the terminator instructions are set up to branch into an empty block
1116    /// for further unrolling. When the loop is finished this will need to be mutated to
1117    /// jump to the end of the loop instead.
1118    fn unroll_loop_iteration(mut self, loop_header_id: BasicBlockId) -> (BasicBlockId, ValueId) {
1119        // Kick off the unrolling from the initial source block.
1120        let mut next_blocks = self.unroll_loop_block();
1121
1122        while let Some(block) = next_blocks.pop() {
1123            self.insert_block = block;
1124            self.source_block = self.get_original_block(block);
1125            self.encountered_loop_header |= loop_header_id == self.source_block;
1126
1127            if !self.visited_blocks.contains(&self.source_block) {
1128                let mut blocks = self.unroll_loop_block();
1129                next_blocks.append(&mut blocks);
1130            }
1131        }
1132        // After having unrolled all blocks in the loop body, we must know how to get back to the header;
1133        // this is also the block into which we have to unroll into next.
1134        let (end_block, induction_value) = self
1135            .induction_value
1136            .expect("Expected to find the induction variable by end of loop iteration");
1137
1138        assert!(
1139            self.encountered_loop_header,
1140            "expected to encounter loop header when visiting blocks"
1141        );
1142
1143        (end_block, induction_value)
1144    }
1145
1146    /// Unroll a single block in the current iteration of the loop.
1147    ///
1148    /// Returns the next blocks to unroll, based on whether the jmp terminator has 1 or 2 destinations.
1149    fn unroll_loop_block(&mut self) -> Vec<BasicBlockId> {
1150        self.visited_blocks.insert(self.source_block);
1151
1152        // Copy instructions from the loop body to the unroll destination, replacing the terminator.
1153        self.inline_instructions_from_block();
1154
1155        let terminator = self.inserter.function.dfg[self.insert_block].unwrap_terminator();
1156
1157        let next_blocks = match terminator {
1158            TerminatorInstruction::JmpIf {
1159                condition,
1160                then_destination,
1161                else_destination,
1162                call_stack,
1163            } => self.handle_jmpif(*condition, *then_destination, *else_destination, *call_stack),
1164            TerminatorInstruction::Jmp { destination, arguments, call_stack: _ } => {
1165                if self.get_original_block(*destination) == self.loop_.header {
1166                    // We found the back-edge of the loop.
1167                    assert_eq!(arguments.len(), 1, "back-edge should only have 1 argument");
1168                    assert!(self.induction_value.is_none(), "there should be only one back-edge");
1169                    self.induction_value = Some((self.insert_block, arguments[0]));
1170                }
1171                vec![*destination]
1172            }
1173            TerminatorInstruction::Return { .. } => {
1174                // Early returns from loops are not implemented.
1175                unreachable!("unexpected return terminator in loop body");
1176            }
1177            TerminatorInstruction::Unreachable { .. } => {
1178                // The SSA pass that adds unreachable terminators must come after unrolling.
1179                unreachable!("unexpected unreachable terminator in loop body");
1180            }
1181        };
1182
1183        // Guarantee that the next blocks we set up to be unrolled, are actually part of the loop,
1184        // which we recorded while inlining the instructions of the blocks already processed.
1185        // Since we only call `unroll_loop_block` from `unroll_loop_iteration`, which we only call
1186        // if the single destination in `unroll_header` is *not* outside the loop, this should hold.
1187        next_blocks.iter().for_each(|block| {
1188            let b = self.get_original_block(*block);
1189            assert!(self.loop_.blocks.contains(&b), "destination not in original loop");
1190        });
1191
1192        next_blocks
1193    }
1194
1195    /// Find the next branch(es) to take from a jmpif terminator and return them.
1196    /// If only one block is returned, it means the jmpif condition evaluated to a known
1197    /// constant and we can safely take only the given branch. In this case the method
1198    /// also replaces the terminator of the insert block (a.k.a fresh block) to be a `Jmp`,
1199    /// and changes the source block in the context for the next iteration to be the
1200    /// destination indicated by the constant condition (ie. the `then` or the `else`).
1201    fn handle_jmpif(
1202        &mut self,
1203        condition: ValueId,
1204        then_destination: BasicBlockId,
1205        else_destination: BasicBlockId,
1206        call_stack: CallStackId,
1207    ) -> Vec<BasicBlockId> {
1208        let condition = self.inserter.resolve(condition);
1209
1210        match self.dfg().get_numeric_constant(condition) {
1211            Some(constant) => {
1212                let destination =
1213                    if constant.is_zero() { else_destination } else { then_destination };
1214
1215                self.source_block = self.get_original_block(destination);
1216
1217                let arguments = Vec::new();
1218                let jmp = TerminatorInstruction::Jmp { destination, arguments, call_stack };
1219                self.inserter.function.dfg.set_block_terminator(self.insert_block, jmp);
1220                vec![destination]
1221            }
1222            None => vec![then_destination, else_destination],
1223        }
1224    }

1226    /// Translate a block id to a block id in the unrolled loop.
1227    ///
1228    /// If the given block id is not within the loop, it is returned as-is,
1229    /// which is the case for when the header jumps to the block following the loop.
1230    fn get_or_insert_block(&mut self, block: BasicBlockId) -> BasicBlockId {
1231        if let Some(new_block) = self.blocks.get(&block) {
1232            return *new_block;
1233        }
1234
1235        // If the block is in the loop we create a fresh block for each iteration
1236        if self.loop_.blocks.contains(&block) {
1237            let new_block = self.dfg_mut().make_block_with_parameters_from_block(block);
1238            self.inserter.remember_block_params_from_block(block, new_block);
1239
1240            self.blocks.insert(block, new_block);
1241            self.original_blocks.insert(new_block, block);
1242            new_block
1243        } else {
1244            block
1245        }
1246    }
1247
1248    /// Find the original ID of a block that replaced it.
1249    fn get_original_block(&self, block: BasicBlockId) -> BasicBlockId {
1250        self.original_blocks.get(&block).copied().unwrap_or(block)
1251    }

1253    /// Copy over instructions from the source into the insert block,
1254    /// while simplifying instructions and keeping track of original block IDs.
1255    fn inline_instructions_from_block(&mut self) {
1256        let source_block = &self.dfg()[self.source_block];
1257        let instructions = source_block.instructions().to_vec();
1258
1259        // We cannot directly append each instruction since we need to substitute any
1260        // instances of the induction variable or any values that were changed as a result
1261        // of the new induction variable value.
1262        for instruction in instructions {
1263            self.inserter.push_instruction(instruction, self.insert_block, false);
1264        }
1265        let mut terminator = self.dfg()[self.source_block].unwrap_terminator().clone();
1266
1267        terminator.map_values_mut(|value| self.inserter.resolve(value));
1268
1269        // Replace the blocks in the terminator with fresh one with the same parameters,
1270        // while remembering which were the original block IDs.
1271        terminator.mutate_blocks(|block| self.get_or_insert_block(block));
1272        self.inserter.function.dfg.set_block_terminator(self.insert_block, terminator);
1273    }
1274
1275    fn dfg(&self) -> &DataFlowGraph {
1276        &self.inserter.function.dfg
1277    }
1278
1279    fn dfg_mut(&mut self) -> &mut DataFlowGraph {
1280        &mut self.inserter.function.dfg
1281    }
1282}

1284/// Unrolling leaves some duplicate instructions which can potentially be removed.
1285fn simplify_between_unrolls(function: &mut Function) {
1286    // Do a mem2reg after the last unroll to aid simplify_cfg
1287    function.mem2reg();
1288    function.simplify_function();
1289    // Do another mem2reg after simplify_cfg to aid the next unroll
1290    function.mem2reg();
1291}

1293/// Decide if the new bytecode size is acceptable, compared to the original.
1294///
1295/// The maximum increase can be expressed as a negative value if we demand a decrease.
1296/// (Values -100 and under mean the new size should be 0).
1297fn is_new_size_ok(orig_size: usize, new_size: usize, max_incr_pct: i32) -> bool {
1298    let max_size_pct = 100i32.saturating_add(max_incr_pct).max(0) as usize;
1299    let max_size = orig_size.saturating_mul(max_size_pct);
1300    new_size.saturating_mul(100) <= max_size
1301}
