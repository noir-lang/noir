14/// HashMap from `(Instruction, side_effects_enabled_var)` to the results of the instruction.
15/// Stored as a two-level map to avoid cloning Instructions during the `.get` call.
16///
17/// The `side_effects_enabled_var` is optional because we only use them when `Instruction::requires_acir_gen_predicate`
18/// is true _and_ the constraint information is also taken into account.
19///
20/// In addition to each result, the original BasicBlockId is stored as well. This allows us
21/// to deduplicate instructions across blocks as long as the new block dominates the original.
22#[derive(Default)]
23pub(super) struct InstructionResultCache(
24    HashMap<Instruction, HashMap<Option<ValueId>, ResultCache>>,
25);
26
27impl InstructionResultCache {
28    /// Get a cached result if it can be used in this context.
29    pub(super) fn get(
30        &self,
31        dfg: &DataFlowGraph,
32        dom: &mut DominatorTree,
33        id: InstructionId,
34        instruction: &Instruction,
35        predicate: Option<ValueId>,
36        block: BasicBlockId,
37    ) -> Option<CacheResult> {
38        let results_for_instruction = self.0.get(instruction)?;
39
40        let cached_results = results_for_instruction.get(&predicate)?.get(
41            block,
42            dom,
43            instruction.has_side_effects(dfg),
44        );
45
46        cached_results.filter(|results| {
47            // This is a hacky solution to https://github.com/noir-lang/noir/issues/9477
48            // We explicitly check that the cached result values are of the same type as expected by the instruction
49            // being checked against the cache and reject if they differ.
50            if let CacheResult::Cached { results, .. } = results {
51                let old_results = dfg.instruction_results(id);
52
53                results.len() == old_results.len()
54                    && old_results
55                        .iter()
56                        .zip(results.iter())
57                        .all(|(old, new)| dfg.type_of_value(*old) == dfg.type_of_value(*new))
58            } else {
59                true
60            }
61        })
62    }
63
64    pub(super) fn cache(
65        &mut self,
66        dom: &mut DominatorTree,
67        instruction: Instruction,
68        predicate: Option<ValueId>,
69        block: BasicBlockId,
70        results: Vec<ValueId>,
71    ) {
72        self.0
73            .entry(instruction)
74            .or_default()
75            .entry(predicate)
76            .or_default()
77            .cache(block, dom, results);
78    }
79
80    pub(super) fn remove(
81        &mut self,
82        instruction: &Instruction,
83    ) -> Option<HashMap<Option<ValueId>, ResultCache>> {
84        self.0.remove(instruction)
85    }
86
87    /// Remove all cached MakeArray instructions that produce the given type.
88    /// Used when we encounter a mutation of an array value that we can't trace back
89    /// to a specific instruction (e.g. block parameters), so we must conservatively
90    /// invalidate all cached MakeArrays that could be the source.
91    fn remove_make_arrays_of_type(&mut self, typ: &Type) {
92        self.0.retain(|instruction, _| {
93            !matches!(instruction, Instruction::MakeArray { typ: make_array_typ, .. } if make_array_typ == typ)
94        });
95    }
96
97    /// Remove previously cached instructions that created arrays,
98    /// if the current instruction is such that it could modify that array.
99    pub(super) fn remove_possibly_mutated_cached_make_arrays(
100        &mut self,
101        instruction: &Instruction,
102        dfg: &DataFlowGraph,
103    ) {
104        use Instruction::{ArraySet, Call, MakeArray, Store};
105
106        /// Recursively remove from the cache any array values.
107        fn go(
108            dfg: &DataFlowGraph,
109            cached_instruction_results: &mut InstructionResultCache,
110            value: &ValueId,
111        ) {
112            // We expect globals to be immutable, so we can cache those results indefinitely.
113            if dfg.is_global(*value) {
114                return;
115            }
116
117            let value_type = dfg.type_of_value(*value);
118
119            // We only care about arrays and vectors. (`Store` can act on non-array values as well)
120            if !value_type.is_array() {
121                return;
122            }
123
124            // Look up the original instruction that created the value, which is the cache key.
125            let instruction = match &dfg[*value] {
126                Value::Instruction { instruction, .. } => &dfg[*instruction],
127                _ => {
128                    // If we can't trace back to a creating instruction (e.g. block parameters),
129                    // conservatively remove all cached MakeArrays of the same type since any
130                    // of them could be the source of this value.
131                    cached_instruction_results.remove_make_arrays_of_type(&value_type);
132                    return;
133                }
134            };
135
136            // Remove the creator instruction from the cache.
137            if matches!(instruction, MakeArray { .. } | Call { .. }) {
138                cached_instruction_results.remove(instruction);
139            }
140
141            // For arrays, we also want to invalidate the values, because multi-dimensional arrays
142            // can be passed around, and through them their sub-arrays might be modified.
143            if let MakeArray { elements, .. } = instruction {
144                for elem in elements {
145                    go(dfg, cached_instruction_results, elem);
146                }
147            }
148        }
149
150        let mut remove_if_array = |value| go(dfg, self, value);
151
152        // Should we consider calls to vector_push_back and similar to be mutating operations as well?
153        match instruction {
154            Store { value, .. } | ArraySet { array: value, .. } => {
155                // If we write to a value, it's not safe for reuse, as its value has changed since its creation.
156                remove_if_array(value);
157            }
158            Call { arguments, func } if dfg.runtime().is_brillig() => {
159                // If we pass a value to a function, it might get modified, making it unsafe for reuse after the call.
160                let Value::Function(func_id) = &dfg[*func] else { return };
161                if matches!(dfg.purity_of(*func_id), None | Some(Purity::Impure)) {
162                    // Arrays passed to functions might be mutated by them if there are no `inc_rc` instructions
163                    // placed *before* the call to protect them. Currently we don't track the ref count in this
164                    // context, so be conservative and do not reuse any array shared with a callee.
165                    // In ACIR we don't track refcounts, so it should be fine.
166                    for arg in arguments {
167                        remove_if_array(arg);
168                    }
169                }
170            }
171            _ => {}
172        }
173    }
174}

176/// Records the results of all duplicate [`Instruction`]s along with the blocks in which they sit.
177///
178/// For more information see [`InstructionResultCache`].
179#[derive(Default, Debug)]
180pub(super) struct ResultCache {
181    result: Option<(BasicBlockId, Vec<ValueId>)>,
182}
183impl ResultCache {
184    /// Records that an `Instruction` in block `block` produced the result values `results`.
185    fn cache(&mut self, block: BasicBlockId, dom: &mut DominatorTree, results: Vec<ValueId>) {
186        let overwrite = match self.result {
187            None => true,
188            Some((origin, _)) => origin != block && dom.dominates(block, origin),
189        };
190
191        if overwrite {
192            self.result = Some((block, results));
193        }
194    }
195
196    /// Returns a set of [`ValueId`]s produced from a copy of this [`Instruction`] which sits
197    /// within a block which dominates `block`.
198    ///
199    /// We require that the cached instruction's block dominates `block` in order to avoid
200    /// cycles causing issues (e.g. two instructions being replaced with the results of each other
201    /// such that neither instruction exists anymore.)
202    pub(super) fn get(
203        &self,
204        block: BasicBlockId,
205        dom: &mut DominatorTree,
206        has_side_effects: bool,
207    ) -> Option<CacheResult> {
208        let (origin, results) = self.result.as_ref()?;
209        if dom.dominates(*origin, block) {
210            Some(CacheResult::Cached { results })
211        } else if !has_side_effects {
212            // Insert a copy of this instruction in the common dominator
213            let dominator = dom.common_dominator(*origin, block);
214            Some(CacheResult::NeedToHoistToCommonBlock { dominator })
215        } else {
216            None
217        }
218    }
219}
220
221#[derive(Debug)]
222pub(super) enum CacheResult<'a> {
223    /// The result of an earlier instruction can be readily reused, because it was found
224    /// in a block that dominates the one where the current instruction is. We can drop
225    /// the current instruction and redefine its results in terms of the existing values.
226    Cached {
227        /// The value IDs we can reuse.
228        results: &'a [ValueId],
229    },
230    /// We found an identical instruction in a non-dominating block, so we cannot directly
231    /// reuse its results, because they are not visible in the current block. However, we
232    /// can hoist the instruction into the common dominator, and deduplicate later.
233    NeedToHoistToCommonBlock {
234        /// The common dominator where we can hoist the current instruction.
235        dominator: BasicBlockId,
236    },
