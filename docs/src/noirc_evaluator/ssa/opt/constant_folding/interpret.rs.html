17/// Checks if the given instruction is a call to a function with all constant arguments.
18/// If so, we can try to evaluate that function and replace the results with the evaluation results.
19pub(super) fn try_interpret_call(
20    instruction: &Instruction,
21    block: BasicBlockId,
22    dfg: &mut DataFlowGraph,
23    interpreter: Option<&mut Interpreter<Empty>>,
24) -> Option<Vec<ValueId>> {
25    let evaluation_result = evaluate_const_argument_call(instruction, interpreter?, dfg);
26
27    match evaluation_result {
28        EvaluationResult::NotABrilligCall | EvaluationResult::CannotEvaluate => None,
29        EvaluationResult::Evaluated(const_results) => {
30            let new_results = vecmap(const_results, |const_result| {
31                interpreter_value_to_ir_value(const_result, dfg, block)
32            });
33            Some(new_results)
34        }
35    }
36}

38/// Result of trying to evaluate an instruction (any instruction) in this pass.
39enum EvaluationResult {
40    /// Nothing was done because the instruction wasn't a call to a brillig function,
41    /// or some arguments to it were not constants.
42    NotABrilligCall,
43    /// The instruction was a call to a brillig function, but we couldn't evaluate it.
44    /// This can occur in the situation where the brillig function reaches a "trap" or a foreign call opcode.
45    CannotEvaluate,
46    /// The instruction was a call to a brillig function and we were able to evaluate it,
47    /// returning [SSA interpreter][Interpreter] [values][InterpreterValue].
48    Evaluated(Vec<InterpreterValue>),
49}

51/// Tries to evaluate an instruction if it's a call where all its arguments are constant.
52/// We do this by interpreting the function's SSA to calculate the result.
53fn evaluate_const_argument_call(
54    instruction: &Instruction,
55    interpreter: &mut Interpreter<Empty>,
56    dfg: &DataFlowGraph,
57) -> EvaluationResult {
58    let Instruction::Call { func: func_id, arguments } = instruction else {
59        return EvaluationResult::NotABrilligCall;
60    };
61
62    let func_value = &dfg[*func_id];
63    let Value::Function(func_id) = func_value else {
64        return EvaluationResult::NotABrilligCall;
65    };
66
67    let Some(_) = interpreter.functions().get(func_id) else {
68        return EvaluationResult::NotABrilligCall;
69    };
70
71    // Ensure all arguments to the call are constant
72    if !arguments.iter().all(|argument| dfg.is_constant(*argument)) {
73        return EvaluationResult::CannotEvaluate;
74    }
75
76    let interpreter_args =
77        arguments.iter().map(|arg| const_ir_value_to_interpreter_value(*arg, dfg)).collect();
78
79    let Ok(result_values) = interpreter.interpret_function(*func_id, interpreter_args) else {
80        return EvaluationResult::CannotEvaluate;
81    };
82
83    EvaluationResult::Evaluated(result_values)
84}

86/// Converts a constant [SSA value][Value] into an [interpreter value][InterpreterValue] for execution.
87fn const_ir_value_to_interpreter_value(value_id: ValueId, dfg: &DataFlowGraph) -> InterpreterValue {
88    let typ = dfg.type_of_value(value_id);
89    match typ {
90        Type::Numeric(numeric_type) => {
91            let constant =
92                dfg.get_numeric_constant(value_id).expect("Should have a numeric constant");
93            InterpreterValue::from_constant(constant, numeric_type)
94                .expect("Should be a valid constant")
95        }
96        Type::Reference(_) => unreachable!("References cannot be constant values"),
97        Type::Array(element_types, _) => {
98            let (array_constant, _) =
99                dfg.get_array_constant(value_id).expect("Should have an array constant");
100            let mut elements = Vec::new();
101            for element in array_constant {
102                elements.push(const_ir_value_to_interpreter_value(element, dfg));
103            }
104            InterpreterValue::array(elements, element_types.to_vec())
105        }
106        Type::Vector(element_types) => {
107            let (array_constant, _) =
108                dfg.get_array_constant(value_id).expect("Should have an array constant");
109            let mut elements = Vec::new();
110            for element in array_constant {
111                elements.push(const_ir_value_to_interpreter_value(element, dfg));
112            }
113            InterpreterValue::vector(elements, element_types)
114        }
115        Type::Function => unreachable!("Functions cannot be constant values"),
116    }
117}

119/// Converts a constant [interpreter value][InterpreterValue] back into an SSA constant.
120fn interpreter_value_to_ir_value(
121    value: InterpreterValue,
122    dfg: &mut DataFlowGraph,
123    block_id: BasicBlockId,
124) -> ValueId {
125    let typ = value.get_type();
126    match typ {
127        Type::Numeric(numeric_type) => {
128            let constant = value.as_numeric().expect("Should be numeric").convert_to_field();
129            dfg.make_constant(constant, numeric_type)
130        }
131        Type::Array(element_types, length) => {
132            let array = match value {
133                InterpreterValue::ArrayOrVector(array) => array,
134                _ => unreachable!("Expected an ArrayOrVector"),
135            };
136
137            let mut elements = Vector::new();
138            for element in array.elements.unwrap_or_clone() {
139                elements.push_back(interpreter_value_to_ir_value(element, dfg, block_id));
140            }
141
142            let instruction =
143                Instruction::MakeArray { elements, typ: Type::Array(element_types, length) };
144
145            let instruction_id = dfg.make_instruction(instruction, None);
146            dfg[block_id].instructions_mut().push(instruction_id);
147            dfg.instruction_result::<1>(instruction_id)[0]
148        }
149        Type::Vector(element_types) => {
150            let array = match value {
151                InterpreterValue::ArrayOrVector(array) => array,
152                _ => unreachable!("Expected an ArrayOrVector"),
153            };
154
155            let mut elements = Vector::new();
156            for element in array.elements.unwrap_or_clone() {
157                elements.push_back(interpreter_value_to_ir_value(element, dfg, block_id));
158            }
159
160            let instruction = Instruction::MakeArray { elements, typ: Type::Vector(element_types) };
161
162            let instruction_id = dfg.make_instruction(instruction, None);
163            dfg[block_id].instructions_mut().push(instruction_id);
164            dfg.instruction_result::<1>(instruction_id)[0]
165        }
166        Type::Function | Type::Reference(_) => unreachable!("Cannot be a constant value"),
167    }
