7use rustc_hash::FxHashMap as HashMap;

9/// Records a simplified equivalents of an [`Instruction`][crate::ssa::ir::instruction::Instruction] in the blocks
10/// where the constraint that advised the simplification has been encountered.
11///
12/// For more information see [`ConstraintSimplificationCache`].
13#[derive(Default)]
14pub(super) struct SimplificationCache {
15    /// Simplified expressions where we found them.
16    ///
17    /// It will always have at least one value because `add` is called
18    /// after the default is constructed.
19    simplifications: HashMap<BasicBlockId, ValueId>,
20}
21
22impl SimplificationCache {
23    /// Try to find a simplification in a visible block.
24    pub(super) fn get(&self, block: BasicBlockId, dom: &DominatorTree) -> Option<ValueId> {
25        // Deterministically walk up the dominator chain until we encounter a block that contains a simplification.
26        dom.find_map_dominator(block, |b| self.simplifications.get(&b).cloned())
27    }
28
29    /// Add a newly encountered simplification to the cache.
30    fn add(&mut self, dfg: &DataFlowGraph, simple: ValueId, block: BasicBlockId) {
31        self.simplifications
32            .entry(block)
33            .and_modify(|existing| {
34                // `SimplificationCache` may already hold a simplification in this block
35                // so we check whether `simple` is a better simplification than the current one.
36                if let Some((_, simpler)) = simplify(dfg, *existing, simple) {
37                    *existing = simpler;
38                };
39            })
40            .or_insert(simple);
41    }
42}
43
44/// HashMap from `(side_effects_enabled_var, Instruction)` to a simplified expression that it can
45/// be replaced with based on constraints that testify to their equivalence, stored together
46/// with the set of blocks at which this constraint has been observed.
47///
48/// Only blocks dominated by one in the cache should have access to this information, otherwise
49/// we create a sort of time paradox where we replace an instruction with a constant we believe
50/// it _should_ equal to, without ever actually producing and asserting the value.
51#[derive(Default)]
52pub(super) struct ConstraintSimplificationCache(
53    HashMap<ValueId, HashMap<ValueId, SimplificationCache>>,
54);
55
56impl ConstraintSimplificationCache {
57    pub(super) fn cache(
58        &mut self,
59        dfg: &DataFlowGraph,
60        predicate: ValueId,
61        block: BasicBlockId,
62        lhs: ValueId,
63        rhs: ValueId,
64    ) {
65        if let Some((complex, simple)) = simplify(dfg, lhs, rhs) {
66            let predicate_cache = self.0.entry(predicate).or_default();
67            predicate_cache.entry(complex).or_default().add(dfg, simple, block);
68        }
69    }
70
71    /// Get the simplification mapping from complex to simpler instructions,
72    /// which all depend on the same side effect condition variable.
73    pub(super) fn get(&self, predicate: ValueId) -> Option<&HashMap<ValueId, SimplificationCache>> {
74        self.0.get(&predicate)
75    }
76}
77
78/// Check if one expression is simpler than the other.
79/// Returns `Some((complex, simple))` if a simplification was found, otherwise `None`.
80/// Expects the `ValueId`s to be fully resolved.
81fn simplify(dfg: &DataFlowGraph, lhs: ValueId, rhs: ValueId) -> Option<(ValueId, ValueId)> {
82    match (&dfg[lhs], &dfg[rhs]) {
83        // Ignore trivial constraints
84        (Value::NumericConstant { .. }, Value::NumericConstant { .. }) => None,
85
86        // Prefer replacing with constants where possible.
87        (Value::NumericConstant { .. }, _) => Some((rhs, lhs)),
88        (_, Value::NumericConstant { .. }) => Some((lhs, rhs)),
89        // Otherwise prefer block parameters over instruction results.
90        // This is as block parameters are more likely to be a single witness rather than a full expression.
91        (Value::Param { .. }, Value::Instruction { .. }) => Some((rhs, lhs)),
92        (Value::Instruction { .. }, Value::Param { .. }) => Some((lhs, rhs)),
93        (_, _) => None,
94    }
95}
