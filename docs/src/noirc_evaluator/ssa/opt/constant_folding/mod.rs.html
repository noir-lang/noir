1//! The goal of the constant folding optimization pass is to propagate any constants forwards into
2//! later [`Instruction`]s to maximize the impact of [compile-time simplifications][crate::ssa::ir::dfg::simplify::simplify()].
3//!
4//! The pass works as follows:
5//! - Re-insert each instruction in order to apply the instruction simplification performed
6//!   by the [`DataFlowGraph`] automatically as new instructions are pushed.
7//! - Check whether any input values have been constrained to be equal to a value of a simpler form
8//!   by a [constrain instruction][Instruction::Constrain]. If so, replace the input value with the simpler form.
9//! - Check whether the instruction [`can_be_deduplicated`]
10//!   by duplicate instruction earlier in the same block.
11//!
12//! These operations are done in parallel so that they can each benefit from each other
13//! without the need for multiple passes.
14//!
15//! This is the only pass which removes duplicated pure [`Instruction`]s however and so is needed when
16//! different blocks are merged, i.e. after the [`flatten_cfg`][super::flatten_cfg] pass.

57impl Ssa {
58    /// Performs constant folding on each instruction.
59    ///
60    /// It will not look at constraints to inform simplifications
61    /// based on the stated equivalence of two instructions.
62    ///
63    /// See [`constant_folding`][self] module for more information.
64    #[tracing::instrument(level = "trace", skip(self))]
65    pub(crate) fn fold_constants(mut self, max_iter: usize) -> Ssa {
66        for function in self.functions.values_mut() {
67            function.constant_fold(false, max_iter, &mut None);
68        }
69        self
70    }
71
72    /// Performs constant folding on each instruction.
73    ///
74    /// Also uses constraint information to inform more optimizations.
75    ///
76    /// See [`constant_folding`][self] module for more information.
77    #[tracing::instrument(level = "trace", skip(self))]
78    pub(crate) fn fold_constants_using_constraints(mut self, max_iter: usize) -> Ssa {
79        for function in self.functions.values_mut() {
80            function.constant_fold(true, max_iter, &mut None);
81        }
82        self
83    }

85    /// Performs constant folding on each instruction while also replacing calls to brillig functions
86    /// with all constant arguments by trying to evaluate those calls.
87    #[tracing::instrument(level = "trace", skip(self))]
88    pub fn fold_constants_with_brillig(mut self, max_iter: usize) -> Ssa {
89        // Collect all brillig functions so that later we can find them when processing a call instruction
90        let mut brillig_functions: BTreeMap<FunctionId, Function> = BTreeMap::new();
91        for (func_id, func) in &self.functions {
92            if func.runtime().is_brillig() {
93                let cloned_function = Function::clone_with_id(*func_id, func);
94                brillig_functions.insert(*func_id, cloned_function);
95            };
96        }
97        let mut interpreter = if brillig_functions.is_empty() {
98            None
99        } else {
100            let mut interpreter = Interpreter::new_from_functions(
101                &brillig_functions,
102                InterpreterOptions {
103                    no_foreign_calls: true,
104                    step_limit: Some(DEFAULT_INTERPRETER_STEP_LIMIT),
105                    ..Default::default()
106                },
107                std::io::empty(),
108            );
109            // Interpret globals once so that we do not have to repeat this computation on every Brillig call.
110            interpreter.interpret_globals().expect("ICE: Interpreter failed to interpret globals");
111            Some(interpreter)
112        };
113
114        for function in self.functions.values_mut() {
115            function.constant_fold(false, max_iter, &mut interpreter);
116        }
117
118        self
119    }
120}

122impl Function {
123    /// The structure of this pass is simple:
124    /// Go through each block and re-insert all instructions.
125    pub(crate) fn constant_fold(
126        &mut self,
127        use_constraint_info: bool,
128        max_iter: usize,
129        interpreter: &mut Option<Interpreter<Empty>>,
130    ) {
131        let mut dom = DominatorTree::with_function(self);
132        let mut context = Context::new(use_constraint_info);
133
134        context.enqueue(&dom, [self.entry_block()]);
135
136        for _ in 0..max_iter {
137            while let Some(block) = context.block_queue.pop_front() {
138                context.fold_constants_in_block(&mut self.dfg, &mut dom, block, interpreter);
139                context.enqueue(&dom, self.dfg[block].successors());
140            }
141
142            #[cfg(debug_assertions)]
143            constant_folding_post_check(&context, &self.dfg);
144
145            // Rebuild the cache and deduplicate the blocks we hoisted into with the origins.
146            let blocks_to_revisit = context.blocks_to_revisit;
147
148            // Preserve the values_to_replace mapping across iterations.
149            // This is necessary because instructions that were simplified or deduplicated in earlier
150            // iterations may still be referenced by instructions in blocks that are revisited.
151            // Without preserving this mapping, those references could point to orphaned instructions.
152            let values_to_replace = context.values_to_replace;
153
154            // Create a fresh context, so values cached towards the end are not visible to blocks during a revisit.
155            // For example reusing the cache could be problematic when using constraint info, as it could make the
156            // original content simplify out based on its own prior assertion of a value being a constant.
157            context = Context::new(use_constraint_info);
158            context.values_to_replace = values_to_replace;
159            context.enqueue(&dom, blocks_to_revisit);
160        }
161
162        // After all iterations, apply the final values_to_replace mapping to all instructions
163        // in all reachable blocks. This is necessary in case an instruction uses a value re-mapped
164        // in another block that is processed after the instruction's block.
165        // By applying the final mapping here, we ensure consistency.
166        if !context.values_to_replace.is_empty() {
167            for block_id in self.reachable_blocks() {
168                self.dfg.replace_values_in_block(block_id, &context.values_to_replace);
169            }
170            self.dfg.data_bus.replace_values(&context.values_to_replace);
171        }
172    }
173}

183struct Context {
184    /// Keeps track of visited blocks and blocks to visit.
185    /// Prioritizes them based on their Reverse Post Order rank, which ensures
186    /// that we see them in a consistent order even during restarts.
187    block_queue: VisitOncePriorityQueue<u32, BasicBlockId>,
188
189    /// Blocks which we hoisted instructions into. We can make another folding iteration
190    /// starting from these blocks and revisiting all their descendants to:
191    /// 1. Deduplicate the original instruction we found in the cache
192    /// 2. Unlock further instructions that can be hoisted after deduplication.
193    blocks_to_revisit: BTreeSet<BasicBlockId>,
194
195    /// Whether to use [constraints][Instruction::Constrain] to inform simplifications later on in the program.
196    ///
197    /// For example, this allows simplifying the instructions below to determine that `v2 == Field 2` without
198    /// laying down constraints for the addition:
199    ///
200    /// ```ssa
201    /// constrain v1 == Field 0
202    /// v2 = add v1, Field 2
203    /// ```
204    use_constraint_info: bool,
205
206    /// Contains sets of values which are constrained to be equivalent to each other.
207    ///
208    /// The mapping's structure is `side_effects_enabled_var => (constrained_value => simplified_value)`.
209    ///
210    /// We partition the maps of constrained values according to the side-effects flag at the point
211    /// at which the values are constrained. This prevents constraints which are only sometimes enforced
212    /// being used to modify the rest of the program.
213    constraint_simplification_mappings: ConstraintSimplificationCache,
214
215    /// Cache of instructions along with their outputs which are safe to reuse.
216    ///
217    /// See [`can_be_deduplicated`] for more information
218    cached_instruction_results: InstructionResultCache,
219
220    /// Maps pre-folded ValueIds to the new ValueIds obtained by re-inserting the instruction.
221    values_to_replace: ValueMapping,
222}

224impl Context {
225    fn new(use_constraint_info: bool) -> Self {
226        Self {
227            use_constraint_info,
228            block_queue: Default::default(),
229            constraint_simplification_mappings: Default::default(),
230            cached_instruction_results: Default::default(),
231            values_to_replace: Default::default(),
232            blocks_to_revisit: Default::default(),
233        }
234    }

236    fn enqueue(&mut self, dom: &DominatorTree, blocks: impl IntoIterator<Item = BasicBlockId>) {
237        for block in blocks {
238            let rank = dom.reverse_post_order_idx(block).unwrap();
239            self.block_queue.push(rank, block);
240        }
241    }

243    fn fold_constants_in_block(
244        &mut self,
245        dfg: &mut DataFlowGraph,
246        dom: &mut DominatorTree,
247        block_id: BasicBlockId,
248        interpreter: &mut Option<Interpreter<Empty>>,
249    ) {
250        let instructions = dfg[block_id].take_instructions();
251
252        // Default side effect condition variable with an enabled state.
253        let mut side_effects_enabled_var =
254            dfg.make_constant(FieldElement::one(), NumericType::bool());
255
256        for instruction_id in instructions {
257            let instruction = &mut dfg[instruction_id];
258
259            instruction.replace_values(&self.values_to_replace);
260
261            self.fold_constants_into_instruction(
262                dfg,
263                dom,
264                block_id,
265                instruction_id,
266                &mut side_effects_enabled_var,
267                interpreter,
268            );
269        }
270
271        // Map the block terminator, resolving any values in the terminator with the
272        // internal value mapping generated by this pass.
273        dfg.replace_values_in_block_terminator(block_id, &self.values_to_replace);
274        dfg.data_bus.replace_values(&self.values_to_replace);
275
276        // Map a terminator in place, replacing any ValueId in the terminator with the
277        // resolved version of that value id from the simplification cache's internal value mapping.
278        // We need this in addition to the value replacement above in order to take advantage
279        // of constraints that may have advised simplifications.
280        // The value mapping (`self.values_to_replace`) only maps old instruction results to new instruction results.
281        // However, constraints do not have "results" like other instructions, thus are not included in `self.values_to_replace`.
282        // To take advantage of constraint simplification we need to still resolve its cache.
283        let mut terminator = dfg[block_id].take_terminator();
284        let constraint_simplification_cache =
285            self.constraint_simplification_mappings.get(side_effects_enabled_var);
286        let mut resolve_cache =
287            |value| resolve_cache(block_id, dom, constraint_simplification_cache, value);
288
289        terminator.map_values_mut(&mut resolve_cache);
290        dfg[block_id].set_terminator(terminator);
291        dfg.data_bus.map_values_mut(resolve_cache);
292    }

294    fn fold_constants_into_instruction(
295        &mut self,
296        dfg: &mut DataFlowGraph,
297        dom: &mut DominatorTree,
298        block: BasicBlockId,
299        id: InstructionId,
300        side_effects_enabled_var: &mut ValueId,
301        interpreter: &mut Option<Interpreter<Empty>>,
302    ) {
303        let constraint_simplification_mapping =
304            self.constraint_simplification_mappings.get(*side_effects_enabled_var);
305
306        let instruction =
307            Self::resolve_instruction(id, block, dfg, dom, constraint_simplification_mapping);
308
309        let old_results = dfg.instruction_results(id).to_vec();
310        let mut target_block = block;
311
312        // If a copy of this instruction exists earlier in the block, then reuse the previous results.
313        let runtime_is_brillig = dfg.runtime().is_brillig();
314        let predicate = self.cache_predicate(*side_effects_enabled_var, &instruction, dfg);
315        if let Some(cache_result) =
316            self.cached_instruction_results.get(dfg, dom, id, &instruction, predicate, block)
317        {
318            match cache_result {
319                CacheResult::Cached { results: cached, .. } => {
320                    // Guard against self-deduplication: if the cached results are exactly
321                    // our own results, the instruction was hoisted here in an earlier visit.
322                    // The cache still points to itself, so mapping old→cached is a no-op that
323                    // would cause us to skip re-inserting the instruction, orphaning its result values.
324                    // In that case, fall through and re-insert instead of deduplicating.
325                    // The `!is_empty` guard is needed because Constrain instructions have
326                    // no results and trivially match.
327                    if !old_results.is_empty() && old_results == cached {
328                        // Fall through to re-insert the instruction normally.
329                    } else {
330                        // We track whether we may mutate `MakeArray` instructions before we deduplicate
331                        // them but we still need to issue an extra inc_rc in case they're mutated afterward.
332                        //
333                        // This also applies to calls that return arrays.
334                        if runtime_is_brillig {
335                            Self::increase_rc(id, cached, block, dfg);
336                        }
337
338                        self.values_to_replace.batch_insert(&old_results, cached);
339
340                        return;
341                    }
342                }
343                CacheResult::NeedToHoistToCommonBlock { dominator } => {
344                    // During revisits we can visit a block which dominates something we already cached instructions from,
345                    // if we restarted from a hoist point that this block also dominates. Most likely it is pointless to
346                    // schedule a revisit of *this* block after again, because something must have prevented this instruction
347                    // from being reused already (e.g. an array mutation).
348                    if dominator != block {
349                        self.blocks_to_revisit.insert(dominator);
350                    }
351
352                    // Just change the block to insert in the common dominator instead.
353                    // This will only move the current instance of the instruction right now.
354                    // When constant folding is run a second time later on, it'll catch
355                    // that the previous instance can be deduplicated to this instance.
356                    // Another effect is going to be that the cache should be updated to
357                    // point at the dominator, so subsequent blocks can use the result.
358                    target_block = dominator;
359                }
360            }
361        };
362
363        // First try to inline a call to a brillig function with all constant arguments.
364        let new_results = if runtime_is_brillig {
365            Self::push_instruction(id, instruction.clone(), &old_results, target_block, dfg)
366        } else {
367            // We only want to try to inline Brillig calls for Brillig entry points (functions called from an ACIR runtime).
368            try_interpret_call(&instruction, target_block, dfg, interpreter.as_mut())
369                // Otherwise, try inserting the instruction again to apply any optimizations using the newly resolved inputs.
370                .unwrap_or_else(|| {
371                    Self::push_instruction(id, instruction.clone(), &old_results, target_block, dfg)
372                })
373        };
374        // If the target_block is distinct than the original block
375        // that means that the current instruction is not added in the original block
376        // so it is deduplicated by the one in the target block.
377        // In case it refers to an array that is mutated, we need to increment
378        // its reference count.
379        if target_block != block && runtime_is_brillig {
380            Self::increase_rc(id, &new_results, block, dfg);
381        }
382
383        self.values_to_replace.batch_insert(&old_results, &new_results);
384
385        self.cache_instruction(
386            &instruction,
387            new_results,
388            dfg,
389            dom,
390            *side_effects_enabled_var,
391            target_block,
392        );
393
394        // If we just inserted an `Instruction::EnableSideEffectsIf`, we need to update `side_effects_enabled_var`
395        // so that we use the correct set of constrained values in future.
396        if let Instruction::EnableSideEffectsIf { condition } = instruction {
397            *side_effects_enabled_var = condition;
398        };
399    }

401    fn increase_rc(
402        id: InstructionId,
403        results: &[ValueId],
404        block: BasicBlockId,
405        dfg: &mut DataFlowGraph,
406    ) {
407        let instruction = &dfg[id];
408        if matches!(instruction, Instruction::MakeArray { .. } | Instruction::Call { .. }) {
409            let call_stack = dfg.get_instruction_call_stack_id(id);
410            for value in results {
411                let value_type = dfg.type_of_value(*value);
412                if value_type.is_array() {
413                    let inc_rc = Instruction::IncrementRc { value: *value };
414                    dfg.insert_instruction_and_results(inc_rc, block, None, call_stack);
415                }
416            }
417        }
418    }

420    /// Fetches an [`Instruction`] by its [`InstructionId`] and fully resolves its inputs.
421    fn resolve_instruction(
422        instruction_id: InstructionId,
423        block: BasicBlockId,
424        dfg: &DataFlowGraph,
425        dom: &mut DominatorTree,
426        constraint_simplification_mapping: Option<&HashMap<ValueId, SimplificationCache>>,
427    ) -> Instruction {
428        let mut instruction = dfg[instruction_id].clone();
429
430        // Resolve any inputs to ensure that we're comparing like-for-like instructions.
431        instruction.map_values_mut(|value_id| {
432            resolve_cache(block, dom, constraint_simplification_mapping, value_id)
433        });
434        instruction
435    }

437    /// Pushes a new [`Instruction`] into the [`DataFlowGraph`] which applies any optimizations
438    /// based on newly resolved values for its inputs.
439    ///
440    /// This may result in the [`Instruction`] being optimized away or replaced with a constant value.
441    fn push_instruction(
442        id: InstructionId,
443        instruction: Instruction,
444        old_results: &[ValueId],
445        block: BasicBlockId,
446        dfg: &mut DataFlowGraph,
447    ) -> Vec<ValueId> {
448        let ctrl_typevars = instruction
449            .requires_ctrl_typevars()
450            .then(|| vecmap(old_results, |result| dfg.type_of_value(*result)));
451
452        let call_stack = dfg.get_instruction_call_stack_id(id);
453        let results = dfg.insert_instruction_and_results_if_simplified(
454            instruction,
455            block,
456            ctrl_typevars,
457            call_stack,
458            Some(id),
459        );
460        let new_results = results.results().to_vec();
461        // Optimizations while inserting the instruction should not change the number of results.
462        assert_eq!(old_results.len(), new_results.len());
463
464        new_results
465    }

467    #[allow(clippy::too_many_arguments)]
468    fn cache_instruction(
469        &mut self,
470        instruction: &Instruction,
471        instruction_results: Vec<ValueId>,
472        dfg: &DataFlowGraph,
473        dom: &mut DominatorTree,
474        side_effects_enabled_var: ValueId,
475        block: BasicBlockId,
476    ) {
477        if self.use_constraint_info {
478            match instruction {
479                // If the instruction was a constraint, then create a link between the two `ValueId`s
480                // to map from the more complex to the simpler value.
481                Instruction::Constrain(lhs, rhs, _) => {
482                    // These `ValueId`s should be fully resolved now.
483                    self.constraint_simplification_mappings.cache(
484                        dfg,
485                        side_effects_enabled_var,
486                        block,
487                        *lhs,
488                        *rhs,
489                    );
490                }
491
492                // If we have an array get whose value is from an array set on the same array at the same index,
493                // we can simplify that array get to the value of the previous array set.
494                //
495                // For example:
496                // v3 = array_set v0, index v1, value v2
497                // v4 = array_get v3, index v1 -> Field
498                //
499                // We know that `v4` can be simplified to `v2`.
500                // Thus, even if the index is dynamic (meaning the array get would have side effects),
501                // we can simplify the operation when we take into account the predicate.
502                //
503                // Note that this does not work in brillig where array sets may actually mutate the arrays.
504                Instruction::ArraySet { index, value, .. } if dfg.runtime().is_acir() => {
505                    let array_get =
506                        Instruction::ArrayGet { array: instruction_results[0], index: *index };
507
508                    // If we encounter an array_get for this address, we know what the result will be.
509                    self.cached_instruction_results.cache(
510                        dom,
511                        array_get,
512                        Some(side_effects_enabled_var),
513                        block,
514                        vec![*value],
515                    );
516                }
517                _ => (),
518            }
519        }
520
521        self.cached_instruction_results
522            .remove_possibly_mutated_cached_make_arrays(instruction, dfg);
523
524        // If the instruction doesn't have side-effects and if it won't interact with enable_side_effects during acir_gen,
525        // we cache the results so we can reuse them if the same instruction appears again later in the block.
526        // Others have side effects representing failure, which are implicit in the ACIR code and can also be deduplicated.
527        let can_be_deduplicated = can_be_deduplicated(instruction, dfg);
528
529        let use_constraint_info = self.use_constraint_info;
530        let is_make_array = matches!(instruction, Instruction::MakeArray { .. });
531
532        let cache_instruction = || {
533            let predicate = self.cache_predicate(side_effects_enabled_var, instruction, dfg);
534            // If we see this make_array again, we can reuse the current result.
535            self.cached_instruction_results.cache(
536                dom,
537                instruction.clone(),
538                predicate,
539                block,
540                instruction_results,
541            );
542        };
543
544        match can_be_deduplicated {
545            CanBeDeduplicated::Always => cache_instruction(),
546            CanBeDeduplicated::UnderSamePredicate if use_constraint_info => cache_instruction(),
547            // We also allow deduplicating MakeArray instructions that we have tracked which haven't been mutated.
548            _ if is_make_array => cache_instruction(),
549
550            CanBeDeduplicated::UnderSamePredicate | CanBeDeduplicated::Never => {}
551        }
552    }

554    /// Returns the predicate value to be used when looking up this [`Instruction`] in the cache.
555    ///
556    /// We sometimes remove the predicate in situations where an instruction is infallible as it allows us to
557    /// deduplicate more aggressively.
558    fn cache_predicate(
559        &self,
560        side_effects_enabled_var: ValueId,
561        instruction: &Instruction,
562        dfg: &DataFlowGraph,
563    ) -> Option<ValueId> {
564        let use_predicate =
565            self.use_constraint_info && instruction.requires_acir_gen_predicate(dfg);
566        use_predicate.then_some(side_effects_enabled_var)
567    }

570// Alternate between resolving `value_id` in the `dfg` and checking to see if the resolved value
571// has been constrained to be equal to some simpler value in the current block.
572//
573// This allows us to reach a stable final `ValueId` for each instruction input as we add more
574// constraints to the cache.
575fn resolve_cache(
576    block: BasicBlockId,
577    dom: &mut DominatorTree,
578    cache: Option<&HashMap<ValueId, SimplificationCache>>,
579    value_id: ValueId,
580) -> ValueId {
581    match cache.and_then(|cache| cache.get(&value_id)) {
582        Some(simplification_cache) => {
583            if let Some(simplified) = simplification_cache.get(block, dom) {
584                resolve_cache(block, dom, cache, simplified)
585            } else {
586                value_id
587            }
588        }
589        None => value_id,
590    }
591}

593#[derive(Debug)]
594enum CanBeDeduplicated {
595    /// This instruction has no side effects so we can substitute the results for those of the same instruction elsewhere.
596    Always,
597    /// This instruction has some side effects such as potentially fallible constraints which could halt execution.
598    ///
599    /// This means that if this instruction passes under a given predicate, we can reuse its results across all
600    /// later instances of this instruction under the same predicate.
601    UnderSamePredicate,
602    /// This instruction has side effects which prevent all deduplication.
603    ///
604    /// An example is `EnableSideEffects` where a "duplicate" of this instruction has an important effect on later instructions
605    /// which is not implied by the existence of the original `EnableSideEffects` instruction. For example:
606    ///
607    /// ```ssa
608    /// enable_side_effects u1 1
609    /// enable_side_effects u1 0
610    /// enable_side_effects u1 1 <-- deduplicating this instruction results in side effects being disabled rather than enabled.
611    /// ```
612    Never,
613}
614
615/// Indicates if the instruction can be safely replaced with the results of another instruction with the same inputs.
616/// If `deduplicate_with_predicate` is set, we assume we're deduplicating with the instruction
617/// and its predicate, rather than just the instruction. Setting this means instructions that
618/// rely on predicates can be deduplicated as well.
619///
620/// Some instructions get the predicate attached to their inputs by `handle_instruction_side_effects` in `flatten_cfg`.
621/// These can be deduplicated because they implicitly depend on the predicate, not only when the caller uses the
622/// predicate variable as a key to cache results. However, to avoid tight coupling between passes, we make the deduplication
623/// conditional on whether the caller wants the predicate to be taken into account or not.
624fn can_be_deduplicated(instruction: &Instruction, dfg: &DataFlowGraph) -> CanBeDeduplicated {
625    use Instruction::*;
626
627    match instruction {
628        // These either have side-effects or interact with memory
629        EnableSideEffectsIf { .. }
630        | Allocate
631        | Load { .. }
632        | Store { .. }
633        | IncrementRc { .. }
634        | DecrementRc { .. } => CanBeDeduplicated::Never,
635
636        Call { func, .. } => {
637            let purity = match dfg[*func] {
638                Value::Intrinsic(intrinsic) => Some(intrinsic.purity()),
639                Value::Function(id) => dfg.purity_of(id),
640                _ => None,
641            };
642            match purity {
643                Some(Purity::Pure) => CanBeDeduplicated::Always,
644                Some(Purity::PureWithPredicate) => CanBeDeduplicated::UnderSamePredicate,
645                Some(Purity::Impure) => CanBeDeduplicated::Never,
646                None => CanBeDeduplicated::Never,
647            }
648        }
649
650        // We can deduplicate these instructions if we know the predicate is also the same.
651        Constrain(..) | ConstrainNotEqual(..) | RangeCheck { .. } => {
652            CanBeDeduplicated::UnderSamePredicate
653        }
654
655        // Noop instructions can always be deduplicated, although they're more likely to be
656        // removed entirely.
657        Noop => CanBeDeduplicated::Always,
658
659        // These instructions can always be deduplicated
660        Cast(_, _) | Not(_) | Truncate { .. } => CanBeDeduplicated::Always,
661        IfElse { then_value, .. } => {
662            if dfg.type_of_value(*then_value).is_array() && dfg.runtime().is_brillig() {
663                CanBeDeduplicated::Never
664            } else {
665                CanBeDeduplicated::Always
666            }
667        }
668        // Arrays can be mutated in unconstrained code so code that handles this case must
669        // take care to track whether the array was possibly mutated or not before
670        // deduplicating. Since we don't know if the containing pass checks for this, we
671        // can only assume these are safe to deduplicate in constrained code.
672        MakeArray { .. } => {
673            if dfg.runtime().is_acir() {
674                CanBeDeduplicated::Always
675            } else {
676                CanBeDeduplicated::Never
677            }
678        }
679
680        // These can have different behavior depending on the EnableSideEffectsIf context.
681        // Replacing them with a similar instruction potentially enables replacing an instruction
682        // with one that was disabled. See
683        // https://github.com/noir-lang/noir/pull/4716#issuecomment-2047846328.
684        Binary(_) | ArrayGet { .. } => {
685            if instruction.requires_acir_gen_predicate(dfg) {
686                CanBeDeduplicated::UnderSamePredicate
687            } else {
688                CanBeDeduplicated::Always
689            }
690        }
691
692        // ArraySet has different behaviors based on both the EnableSideEffectsIf instruction
693        // (ACIR) and the underlying dynamic RC value of the array (Brillig). In the later case,
694        // we can never deduplicate
695        ArraySet { .. } => {
696            if dfg.runtime().is_acir() {
697                CanBeDeduplicated::UnderSamePredicate
698            } else {
699                CanBeDeduplicated::Never
700            }
701        }
702    }
703}
