1//! An SSA pass that operates on ACIR functions that checks that multiplying two u128 doesn't
2//! overflow because both operands are greater or equal than 2^64.
3//! If both are, then the result is surely greater or equal than 2^128 so it would overflow.
4//! The operands can still overflow if just one of them is less than 2^64, but in that case
5//! the result will be less than 2^192 so it fits in a Field value, and acir will check that
6//! it fits in a u128.
7//!
8//! In Brillig an overflow check is automatically performed on unsigned binary operations
9//! so this SSA pass has no effect for Brillig functions.
10use acvm::{AcirField, FieldElement};
11use num_bigint::BigUint;
12
13use crate::ssa::{
14    ir::{
15        function::Function,
16        instruction::{Binary, BinaryOp, ConstrainError, Instruction},
17        types::NumericType,
18        value::ValueId,
19    },
20    ssa_gen::Ssa,
21};
22
23use super::simple_optimization::SimpleOptimizationContext;

25impl Ssa {
26    /// See [`check_u128_mul_overflow`][self] module for more information.
27    #[tracing::instrument(level = "trace", skip(self))]
28    pub(crate) fn check_u128_mul_overflow(mut self) -> Ssa {
29        for function in self.functions.values_mut() {
30            function.check_u128_mul_overflow();
31        }
32        self
33    }
34}
35
36impl Function {
37    fn check_u128_mul_overflow(&mut self) {
38        if !self.runtime().is_acir() {
39            return;
40        }
41
42        self.simple_optimization(|context| {
43            context.insert_current_instruction();
44
45            let Instruction::Binary(Binary {
46                lhs,
47                rhs,
48                operator: BinaryOp::Mul { unchecked: false },
49            }) = context.instruction()
50            else {
51                return;
52            };
53
54            let binary_type = context.dfg.type_of_value(*lhs).unwrap_numeric();
55            let NumericType::Unsigned { bit_size: 128 } = binary_type else {
56                return;
57            };
58
59            check_u128_mul_overflow(*lhs, *rhs, context);
60        });
61    }
62}
63
64/// MAX_NON_OVERFLOWING_CONST_ARG is expected to be [p/U],
65/// where U=U128::max() and p is the field modulus.
66///
67/// Then x<=[p/U]<p/U, so x*U<p
68static MAX_NON_OVERFLOWING_CONST_ARG: std::sync::LazyLock<u128> = std::sync::LazyLock::new(|| {
69    let max_non_overflowing_const_arg = u128::try_from(FieldElement::modulus() / u128::MAX)
70        .expect("expected max_const_value_that_does_not_overflow to fit into a u128");
71    assert!(BigUint::from(u128::MAX) * max_non_overflowing_const_arg < FieldElement::modulus());
72    assert!(
73        (max_non_overflowing_const_arg + 1)
74            .checked_mul(max_non_overflowing_const_arg + 1)
75            .is_none(),
76        "(expected max_non_overflowing_const_arg + 1) * (max_non_overflowing_const_arg + 1) to overflow u128"
77    );
78    max_non_overflowing_const_arg
79});
80
81fn max_non_overflowing_const_arg() -> u128 {
82    *MAX_NON_OVERFLOWING_CONST_ARG
83}
84
85fn check_u128_mul_overflow(
86    lhs: ValueId,
87    rhs: ValueId,
88    context: &mut SimpleOptimizationContext<'_, '_>,
89) {
90    let dfg = &mut context.dfg;
91    let lhs_value = dfg.get_numeric_constant(lhs);
92    let rhs_value = dfg.get_numeric_constant(rhs);
93
94    assert!(
95        lhs_value.map(|value| value.fits_in_u128()).unwrap_or(true),
96        "expected lhs_value to fit in a u128, but found {lhs_value:?}"
97    );
98    assert!(
99        rhs_value.map(|value| value.fits_in_u128()).unwrap_or(true),
100        "expected rhs_value to fit in a u128, but found {rhs_value:?}"
101    );
102
103    if lhs_value.is_some_and(|value| value.to_u128() <= max_non_overflowing_const_arg())
104        || rhs_value.is_some_and(|value| value.to_u128() <= max_non_overflowing_const_arg())
105    {
106        return;
107    }
108
109    let block = context.block_id;
110    let call_stack = dfg.get_instruction_call_stack_id(context.instruction_id);
111
112    let u128 = NumericType::unsigned(128);
113    let two_pow_64 = 1_u128 << 64;
114    let two_pow_64 = dfg.make_constant(two_pow_64.into(), u128);
115    let mul = BinaryOp::Mul { unchecked: true };
116
117    // To check if a value is less than 2^64 we divide it by 2^64 and expect the result to be zero.
118    let res = match (lhs_value, rhs_value) {
119        (Some(_), Some(_)) => {
120            // If both values are known at compile time, at this point we know it overflows
121            dfg.make_constant(FieldElement::one(), u128)
122        }
123        (Some(_), None) => {
124            // If only the left-hand side is known we just need to check that the right-hand side
125            // isn't greater than 2^64
126            let instruction =
127                Instruction::Binary(Binary { lhs: rhs, rhs: two_pow_64, operator: BinaryOp::Div });
128            dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
129        }
130        (None, Some(_)) => {
131            // Same goes for the other side
132            let instruction =
133                Instruction::Binary(Binary { lhs, rhs: two_pow_64, operator: BinaryOp::Div });
134            dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
135        }
136        (None, None) => {
137            // Check both sides
138            let instruction =
139                Instruction::Binary(Binary { lhs, rhs: two_pow_64, operator: BinaryOp::Div });
140            let divided_lhs =
141                dfg.insert_instruction_and_results(instruction, block, None, call_stack).first();
142
143            let instruction =
144                Instruction::Binary(Binary { lhs: rhs, rhs: two_pow_64, operator: BinaryOp::Div });
145            let divided_rhs =
146                dfg.insert_instruction_and_results(instruction, block, None, call_stack).first();
147
148            // Unchecked as operands are restricted to be less than 2^64 so multiplying them cannot overflow.
149            let instruction =
150                Instruction::Binary(Binary { lhs: divided_lhs, rhs: divided_rhs, operator: mul });
151            dfg.insert_instruction_and_results(instruction, block, None, call_stack).first()
152        }
153    };
154
155    // We must only check for overflow if the side effects var is active
156    let predicate = Instruction::Cast(context.enable_side_effects, u128);
157    let predicate = dfg.insert_instruction_and_results(predicate, block, None, call_stack).first();
158    let res = Instruction::Binary(Binary { lhs: res, rhs: predicate, operator: mul });
159    let res = dfg.insert_instruction_and_results(res, block, None, call_stack).first();
160
161    let zero = dfg.make_constant(FieldElement::zero(), u128);
162    let instruction = Instruction::Constrain(
163        res,
164        zero,
165        Some(ConstrainError::StaticString("attempt to multiply with overflow".to_string())),
166    );
167    dfg.insert_instruction_and_results(instruction, block, None, call_stack);
168}
