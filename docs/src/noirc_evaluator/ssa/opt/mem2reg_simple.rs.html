1//! Mem2reg algorithm adapted from the paper: <https://bernsteinbear.com/assets/img/bebenita-ssa.pdf>
2//!
3//! The goal is for this new, simpler to eventually replace our existing mem2reg algorithm in `ssa/opt/mem2reg.rs`.
4//! The pre-existing pass however can optimize in more/different cases than this pass. For example,
5//! it can still optimize out stores/loads in some cases even when the reference is aliased. That
6//! other pass has a larger surface area for bugs though and this one is simpler so the goal is to
7//! replace the old pass with this one plus any other, separate passes needed for the features
8//! unhandled here (such as alias analysis).
9//!
10//! ## Block parameter placement
11//!
12//! When a variable is stored in multiple blocks, this pass places block parameters only at the
13//! **iterated dominance frontier** (IDF) of those definition sites. This is the minimal set of
14//! blocks where values from different control-flow paths could merge, following the standard
15//! algorithm from Cytron et al. (1991). For variables stored in a single block, the dominance
16//! frontier is often empty, meaning no block parameters are needed at all — the value simply
17//! propagates to all loads.
18//!
19//! For blocks not in the IDF (the common case in unrolled code with single-predecessor chains),
20//! the variable's value is inherited directly from the predecessor's exit state. This avoids
21//! the O(variables × blocks) cost of adding block parameters everywhere.
22use iter_extended::vecmap;
23use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};

43    ///
44    /// ACIR functions have no variable limit since they benefit more from full promotion.
45    /// Brillig keeps the limit to avoid regressions in loop-heavy code.

89    fn mem2reg_simple(&mut self, cleanup: bool) {
90        let cfg = ControlFlowGraph::with_function(self);
91        let post_order = PostOrder::with_cfg(&cfg);
92        let mut dom_tree = DominatorTree::with_cfg_and_post_order(&cfg, &post_order);
93        let mut inserter = FunctionInserter::new(self);
94
95        let blocks = post_order.into_vec_reverse();
96
97        // Note that `variables` and `entry_values` in variable_states are all keyed by the original
98        // ValueId of the `allocate` instruction result. These are all iterated over at some point
99        // so it is important we use a deterministic order so that block arguments always correspond
100        // to block parameters in the same order.
101        let (variables, def_sites) =
102            collect_eligible_variables_and_def_sites(inserter.function, &blocks);
103
104        if variables.is_empty() {
105            return;
106        }
107
108        // Compute where block parameters are needed using iterated dominance frontiers.
109        // A variable only needs a block parameter at blocks where values from different
110        // control-flow paths could merge (its IDF). For variables stored in a single block,
111        // this is typically empty — no block parameters needed at all.
112        let dom_frontiers = dom_tree.compute_dominance_frontiers_with_back_edges(&cfg);
113        let param_locations = compute_param_locations(&variables, &def_sites, &dom_frontiers);
114
115        // Precompute which variables are visible at each block by walking the dominator tree.
116        // A variable declared in block D is visible at block B iff D dominates B.
117        // Instead of checking dominates() for each (variable, block) pair — O(blocks × variables) —
118        // we inherit the visible set from the immediate dominator: O(blocks) tree walk.
119        // This completes the Cytron-style SSA construction (the IDF placement above is phase 1;
120        // this visibility propagation replaces the per-variable dominance checks in phase 2).
121        let visible_vars = compute_visible_vars(&blocks, &variables, &dom_tree);
122
123        let mut block_states = BlockStates::default();
124        add_block_params_and_find_exit_states(
125            &blocks,
126            &visible_vars,
127            &param_locations,
128            &mut inserter,
129            &mut block_states,
130            &cfg,
131        );
132        add_terminator_arguments(
133            &blocks,
134            &variables,
135            &param_locations,
136            &mut inserter,
137            &block_states,
138            &cfg,
139        );
140        if cleanup {
141            remove_params_from_blocks_with_identical_terminator_args(&blocks, &mut inserter, &cfg);
142        }
143        commit(&mut inserter, &variables, blocks);
144    }
145
146    /// Like `mem2reg_simple` but skips the cleanup pass that removes block parameters
147    /// whose arguments are all identical. This reveals whether the IDF-based placement
148    /// avoided unnecessary parameters at source, rather than relying on cleanup.
149    #[cfg(test)]
150    fn mem2reg_simple_without_cleanup(&mut self) {
151        self.mem2reg_simple(false);
152    }
153}

161/// Contains the starting & ending values of each variable in one block
162#[derive(Default)]
163struct BlockState {
164    /// Maps each variable visible in this block to its starting value in the block. This is always
165    /// a block parameter or a forwarded value from a previous block.
166    entry_state: BTreeMap<ValueId, ValueId>,
167
168    /// Maps each variable modified within this block to the value it is set to at the end of
169    /// the block. Note that to save on memory, we do not store variables which were not modified
170    /// within the block. Their values will be the same as the value in `entry_states`.
171    exit_state: BTreeMap<ValueId, ValueId>,
172}

174/// For each eligible variable, the set of blocks where a block parameter must be inserted.
175/// Computed as the iterated dominance frontier of the variable's definition sites.
176type ParamLocations = BTreeMap<ValueId, HashSet<BasicBlockId>>;
177
178/// Compute where block parameters are needed for each variable.
179///
180/// A variable needs a block parameter at block B if B is in the iterated dominance frontier
181/// of the blocks where the variable is stored to. This is the minimal set of blocks where
182/// values from different control-flow paths could merge.
183fn compute_param_locations(
184    variables: &BTreeMap<ValueId, BasicBlockId>,
185    def_sites: &HashMap<ValueId, HashSet<BasicBlockId>>,
186    dom_frontiers: &HashMap<BasicBlockId, HashSet<BasicBlockId>>,
187) -> ParamLocations {
188    let mut result = BTreeMap::new();
189    for var in variables.keys() {
190        let sites = def_sites.get(var).cloned().unwrap_or_default();
191        result.insert(*var, iterated_dominance_frontier(&sites, dom_frontiers));
192    }
193    result
194}

196/// Compute the iterated dominance frontier of a set of blocks.
197///
198/// Starting from the given definition sites, this computes the closure of the dominance
199/// frontier: DF+(S) = DF(S) ∪ DF(DF(S)) ∪ ... This is the standard worklist algorithm.
200fn iterated_dominance_frontier(
201    def_sites: &HashSet<BasicBlockId>,
202    dom_frontiers: &HashMap<BasicBlockId, HashSet<BasicBlockId>>,
203) -> HashSet<BasicBlockId> {
204    let mut result = HashSet::default();
205    let mut worklist: Vec<BasicBlockId> = def_sites.iter().copied().collect();
206
207    while let Some(block) = worklist.pop() {
208        if let Some(frontier) = dom_frontiers.get(&block) {
209            for &df_block in frontier {
210                if result.insert(df_block) {
211                    worklist.push(df_block);
212                }
213            }
214        }
215    }
216    result
217}

219/// Precompute which variables are visible at each block by walking the dominator tree.
220///
221/// A variable declared in block D is visible at block B iff D dominates B. Instead of
222/// checking `dominates(D, B)` for every (variable, block) pair — O(variables × blocks) —
223/// we walk blocks in RPO and inherit the visible set from the immediate dominator.
224/// Each block's visible set is its idom's visible set plus any variables declared locally.
225fn compute_visible_vars(
226    blocks: &[BasicBlockId],
227    variables: &BTreeMap<ValueId, BasicBlockId>,
228    dom_tree: &DominatorTree,
229) -> HashMap<BasicBlockId, BTreeMap<ValueId, BasicBlockId>> {
230    // Group variables by their declaration block
231    let mut vars_by_decl_block: HashMap<BasicBlockId, Vec<ValueId>> = HashMap::default();
232    for (var, decl_block) in variables {
233        vars_by_decl_block.entry(*decl_block).or_default().push(*var);
234    }
235
236    let mut visible: HashMap<BasicBlockId, BTreeMap<ValueId, BasicBlockId>> = HashMap::default();
237    for &block in blocks {
238        let mut vars = match dom_tree.immediate_dominator(block) {
239            Some(idom) => visible[&idom].clone(),
240            None => BTreeMap::new(),
241        };
242        if let Some(declared_here) = vars_by_decl_block.get(&block) {
243            for var in declared_here {
244                vars.insert(*var, block);
245            }
246        }
247        visible.insert(block, vars);
248    }
249    visible
250}

252/// Find the starting & ending states of each variable in each block.
253///
254/// Block parameters are only added at blocks in the variable's IDF (param_locations).
255/// For all other blocks, the entry value is inherited from the predecessor's exit state.
256fn add_block_params_and_find_exit_states(
257    blocks: &[BasicBlockId],
258    visible_vars: &HashMap<BasicBlockId, BTreeMap<ValueId, BasicBlockId>>,
259    param_locations: &ParamLocations,
260    inserter: &mut FunctionInserter,
261    block_states: &mut BlockStates,
262    cfg: &ControlFlowGraph,
263) {
264    for block in blocks.iter().copied() {
265        let entry_state = compute_entry_state(
266            &visible_vars[&block],
267            param_locations,
268            block,
269            &mut inserter.function.dfg,
270            block_states,
271            cfg,
272        );
273        let exit_state = abstract_interpret_block(inserter, block, &entry_state);
274        block_states.blocks.insert(block, BlockState { entry_state, exit_state });
275    }
276}

278/// Compute the entry state for a block.
279///
280/// `visible_vars` contains only the variables whose declaration dominates this block
281/// (precomputed via dominator tree walk in `compute_visible_vars`).
282///
283/// For each visible variable:
284/// - If this is the declaration block: use the original allocate result
285/// - If this block is in the variable's IDF: add a fresh block parameter
286/// - Otherwise: inherit the value from a visited predecessor's exit state
287fn compute_entry_state(
288    visible_vars: &BTreeMap<ValueId, BasicBlockId>,
289    param_locations: &ParamLocations,
290    block: BasicBlockId,
291    dfg: &mut DataFlowGraph,
292    block_states: &BlockStates,
293    cfg: &ControlFlowGraph,
294) -> StateVec {
295    visible_vars
296        .iter()
297        .filter_map(|(var, decl_block)| {
298            let value = if block == *decl_block {
299                // Declaration block: use original allocate result
300                *var
301            } else if param_locations[var].contains(&block) {
302                // IDF block: add a block parameter for this variable
303                let typ = dfg
304                    .type_of_value(*var)
305                    .reference_element_type()
306                    .expect("All variables should be references")
307                    .clone();
308                dfg.add_block_parameter(block, typ)
309            } else {
310                // Non-IDF block: inherit from a visited predecessor.
311                // Since we process in RPO, at least one non-back-edge predecessor
312                // has been visited. All visited predecessors must agree on this
313                // variable's value (otherwise this block would be in the IDF).
314                get_value_from_visited_predecessor(*var, block, cfg, block_states)?
315            };
316
317            Some((*var, value))
318        })
319        .collect()
320}

322/// Get a variable's exit value from a visited predecessor.
323///
324/// Returns None if no predecessor has been visited yet (can happen for unreachable blocks).
325fn get_value_from_visited_predecessor(
326    var: ValueId,
327    block: BasicBlockId,
328    cfg: &ControlFlowGraph,
329    block_states: &BlockStates,
330) -> Option<ValueId> {
331    for predecessor in cfg.predecessors(block) {
332        if let Some(pred_state) = block_states.blocks.get(&predecessor) {
333            return Some(pred_state.get_exit_value(var));
334        }
335    }
336    None
337}

339/// Link entry & exit states by adding terminator arguments for variables at IDF blocks.
340///
341/// Only blocks in a variable's IDF have block parameters that need arguments wired.
342fn add_terminator_arguments(
343    blocks: &[BasicBlockId],
344    variables: &BTreeMap<ValueId, BasicBlockId>,
345    param_locations: &ParamLocations,
346    inserter: &mut FunctionInserter,
347    block_states: &BlockStates,
348    cfg: &ControlFlowGraph,
349) {
350    for block in blocks.iter().copied() {
351        let block_state = &block_states.blocks[&block];
352
353        for predecessor in cfg.predecessors(block) {
354            let pred_state = &block_states.blocks[&predecessor];
355            let args = get_terminator_args_mut(&mut inserter.function.dfg, predecessor, block);
356            for address in block_state.entry_state.keys() {
357                // Only wire arguments for IDF blocks (those with block parameters).
358                // Declaration blocks and inherited-value blocks don't have params to wire.
359                if block != variables[address] && param_locations[address].contains(&block) {
360                    args.push(pred_state.get_exit_value(*address));
361                }
362            }
363        }
364    }
365}

367/// For every block, remove any block parameters whose arguments (from predecessors' terminators) are all identical
368fn remove_params_from_blocks_with_identical_terminator_args(
369    blocks: &[BasicBlockId],
370    inserter: &mut FunctionInserter,
371    cfg: &ControlFlowGraph,
372) {
373    // Sort blocks such that we remove parameters from blocks with multiple predecessors last.
374    // This helps remove some dependency on block ordering for optimizations. E.g. the
375    // `read_only_loop` test requires 2 passes of mem2reg_simple without this.
376    let mut blocks = blocks.to_vec();
377    blocks.sort_unstable_by_key(|block| cfg.predecessors(*block).len());
378
379    for block in blocks {
380        remove_params_from_block_with_identical_terminator_args(block, inserter, cfg);
381    }
382}

384/// Removes block parameters whose arguments (from predecessors' terminators) are all identical
385fn remove_params_from_block_with_identical_terminator_args(
386    block: BasicBlockId,
387    inserter: &mut FunctionInserter,
388    cfg: &ControlFlowGraph,
389) {
390    let parameters = inserter.function.dfg.block_parameters(block).to_vec();
391
392    // Mask of whether each parameter has non-identical arguments,
393    // E.g. if parameter 2's arguments are all identical, then `mask[2]` would be false
394    let mask = keep_argument_mask(inserter, cfg, block, &parameters);
395
396    // Remove unneeded parameters from the block
397    retain_items_from_mask(inserter.function.dfg[block].parameters_mut(), &mask);
398
399    // And remove the corresponding parameter's arguments from each predecessor
400    for predecessor in cfg.predecessors(block) {
401        let arguments = get_terminator_args_mut(&mut inserter.function.dfg, predecessor, block);
402        retain_items_from_mask(arguments, &mask);
403    }
404}

406/// Mapping from a variable to its value at a point in time.
407type StateVec = BTreeMap<ValueId, ValueId>;

409impl BlockState {
410    /// Gets the "exit value" of a variable, which is its value at the end of the given block.
411    ///
412    /// `variable` should always be a reference.
413    fn get_exit_value(&self, variable: ValueId) -> ValueId {
414        *self
415            .exit_state
416            .get(&variable)
417            // To save memory, `exit_states` only contains the value if it changed within the block.
418            // So we have to check `entry_states` for the value if it went unchanged.
419            .unwrap_or_else(|| &self.entry_state[&variable])
420    }
421}

423/// Return a mask indicating whether to keep or remove the corresponding parameter.
424///
425/// For a given parameter at index i, index i of the resulting mask is false (indicating to remove the parameter)
426/// if all arguments passed to that parameter from predecessor blocks are identical.
427///
428/// Each parameter that should be removed will be mapped to its single argument's value.
429fn keep_argument_mask(
430    inserter: &mut FunctionInserter,
431    cfg: &ControlFlowGraph,
432    block: BasicBlockId,
433    parameters: &[ValueId],
434) -> Vec<bool> {
435    let entry = inserter.function.entry_block();
436    if block == entry {
437        // Entry block has no predecessors, so keep all parameters
438        return vec![true; parameters.len()];
439    }
440
441    vecmap(parameters.iter().enumerate(), |(i, parameter)| {
442        let mut args = cfg
443            .predecessors(block)
444            .map(|predecessor| get_terminator_args(&inserter.function.dfg, predecessor, block)[i])
445            .map(|arg| inserter.resolve(arg))
446            // Filtering here optimizes away cases where we have the parameter changed in one block
447            // and unchanged in another (argument can only equal parameter in the case of back-edges)
448            .filter(|arg| arg != parameter);
449
450        let first_arg = args
451            .next()
452            .expect("Entry block is excluded so there should always be >= 1 predecessor");
453
454        // keep the parameter if the arguments do not all match
455        // unwrap safety: for all multi-predecessor blocks, each predecessor should end in a jmp, not jmpif
456        let keep_param = !args.all(|arg| arg == first_arg);
457        if !keep_param {
458            // All arguments are identical, so the choice to map to the first is arbitrary
459            // Do not `inserter.resolve()` this parameter! We want the original block parameter, but it
460            // may already be mapped away in the inserter.
461            inserter.map_value(*parameter, first_arg);
462        }
463        keep_param
464    })
465}

467/// Get the terminator arguments for block `block` jumping to block `jmp_target`.
468/// The `jmp_target` is relevant if `block` terminates in a jmpif terminator and may jmp to
469/// multiple blocks. Panics if the given block does not have block arguments.
470fn get_terminator_args(
471    dfg: &DataFlowGraph,
472    block: BasicBlockId,
473    jmp_target: BasicBlockId,
474) -> &[ValueId] {
475    match dfg[block].unwrap_terminator() {
476        TerminatorInstruction::Jmp { arguments, .. } => arguments,
477        TerminatorInstruction::JmpIf {
478            then_destination, then_arguments, else_arguments, ..
479        } => {
480            if jmp_target == *then_destination {
481                then_arguments
482            } else {
483                else_arguments
484            }
485        }
486        TerminatorInstruction::Return { .. } | TerminatorInstruction::Unreachable { .. } => panic!(
487            "get_terminator_args called on block edge {block} -> {jmp_target} but {block} does not have any arguments"
488        ),
489    }
490}

493/// The `jmp_target` is relevant if `block` terminates in a jmpif terminator and may jmp to
494/// multiple blocks. Panics if the given block does not have block arguments.
495fn get_terminator_args_mut(
496    dfg: &mut DataFlowGraph,
497    block: BasicBlockId,
498    jmp_target: BasicBlockId,
499) -> &mut Vec<ValueId> {
500    match dfg[block].unwrap_terminator_mut() {

502        TerminatorInstruction::JmpIf {

512            "get_terminator_args called on block edge {block} -> {jmp_target} but {block} does not have any arguments"
513        ),
514    }

517/// For each index i of `items`, keep `items[i]` iff `mask[i]`
518fn retain_items_from_mask(items: &mut Vec<ValueId>, mask: &[bool]) {
519    debug_assert_eq!(items.len(), mask.len());
520    let mut mask_iter = mask.iter();
521    items.retain(|_| *mask_iter.next().unwrap());
522    // Reclaim some memory, important in larger programs
523    items.shrink_to_fit();
524}

526/// Abstractly interpret a block, collecting the value of each reference at the end of a block.
527/// Any references not included in the result are assumed to be unchanged.
528///
529/// This function is very simple and will produce incorrect results for first-class references
530/// that are aliased, passed to functions, or stored in arrays, etc.
531///
532/// Note that this function will also replace any instances of the reference being stored to
533/// with its current value in the block. Most of the time, this locally renames the reference
534/// in the block so that we can map its value in the block independent of other blocks.
535fn abstract_interpret_block(
536    inserter: &mut FunctionInserter,
537    block: BasicBlockId,
538    entry_state: &StateVec,
539) -> StateVec {
540    // Any variables not in the exit_state by function end are assumed to be unchanged from the entry_state
541    let mut exit_state = StateVec::new();
542    let instructions = inserter.function.dfg[block].take_instructions();
543
544    for instruction_id in &instructions {
545        match &inserter.function.dfg[*instruction_id] {
546            Instruction::Store { address, value } => {
547                // Only update the value if the address is already in the entry state, since only
548                // those addresses are eligible for mem2reg optimization. If the address is
549                // in the state, we remove it.
550                if entry_state.contains_key(address) {
551                    exit_state.insert(*address, *value);
552                }
553            }
554            Instruction::Load { address } => {
555                if let Some(value) = exit_state.get(address).or_else(|| entry_state.get(address)) {
556                    let result = inserter.function.dfg.instruction_results(*instruction_id)[0];
557                    inserter.map_value(result, *value);
558                }
559            }
560            _ => (),
561        }
562    }
563
564    *inserter.function.dfg[block].instructions_mut() = instructions;
565    exit_state
566}

568/// Return a map from each eligible variable to the block it was declared in,
569/// along with the set of blocks where each variable is stored to (definition sites).
570///
571/// Only includes variables that are eligible for mem2reg optimization,
572/// i.e. those that are allocated but never used in a first-class manner.
573fn collect_eligible_variables_and_def_sites(
574    function: &Function,
575    blocks: &[BasicBlockId],
576) -> (BTreeMap<ValueId, BasicBlockId>, HashMap<ValueId, HashSet<BasicBlockId>>) {
577    // Map each variable to the block it was declared in
578    let mut variables = BTreeMap::default();
579    // Map each variable to the set of blocks that contain stores to it
580    let mut def_sites: HashMap<ValueId, HashSet<BasicBlockId>> = HashMap::default();
581
582    // Workaround for https://github.com/noir-lang/noir/issues/11482
583    // If the declaration block of an allocate has no starting store then it isn't eligible for mem2reg_simple.
584    let mut variables_with_stores_in_decl_block = HashSet::default();
585
586    for block_id in blocks.iter().copied() {
587        let block = &function.dfg[block_id];
588        for instruction_id in block.instructions() {
589            let instruction = &function.dfg[*instruction_id];
590            match instruction {
591                Instruction::Allocate => {
592                    let address = function.dfg.instruction_results(*instruction_id)[0];
593                    variables.insert(address, block_id);
594                }
595                Instruction::Load { .. } => (),
596                // Storing to an address is fine, but storing an address prevents optimizing it out.
597                Instruction::Store { address, value } => {
598                    variables.remove(value);
599
600                    if variables.contains_key(address) {
601                        def_sites.entry(*address).or_default().insert(block_id);
602                    }
603
604                    if variables.get(address) == Some(&block_id) {
605                        variables_with_stores_in_decl_block.insert(*address);
606                    }
607                }
608                // Any other use of an address (in arrays, functions, etc) is also first-class and prevents optimization.
609                _ => instruction.for_each_value(|value| {
610                    variables.remove(&value);
611                    def_sites.remove(&value);
612                }),
613            }
614        }
615
616        block.unwrap_terminator().for_each_value(|value| {
617            variables.remove(&value);
618            def_sites.remove(&value);
619        });
620    }
621
622    variables.retain(|address, _| variables_with_stores_in_decl_block.contains(address));
623    def_sites.retain(|address, _| variables.contains_key(address));
624    (variables, def_sites)
625}

627/// Commit to all changes made by the pass:
628/// - Map any values mapped from the inserter to their new values in the function
629/// - Remove all Allocate, Load, and Store instructions from the eligible variables
630fn commit(
631    inserter: &mut FunctionInserter,
632    variables: &BTreeMap<ValueId, BasicBlockId>,
633    blocks: Vec<BasicBlockId>,
634) {
635    for block in blocks {
636        let mut instructions = inserter.function.dfg[block].take_instructions();
637
638        // Remove any allocate, load, or store instructions for variables which were optimized out
639        instructions.retain(|instruction_id| {
640            let instruction = &inserter.function.dfg[*instruction_id];
641            let keep = match instruction {
642                Instruction::Allocate => {
643                    let address = inserter.function.dfg.instruction_results(*instruction_id)[0];
644                    !variables.contains_key(&address)
645                }
646                Instruction::Load { address } | Instruction::Store { address, value: _ } => {
647                    !variables.contains_key(address)
648                }
649                _ => true,
650            };
651
652            if keep {
653                inserter.map_instruction_in_place(*instruction_id);
654            }
655            keep
656        });
657
658        *inserter.function.dfg[block].instructions_mut() = instructions;
659
660        let mut terminator = inserter.function.dfg[block].take_terminator();
661        terminator.map_values_mut(|value| inserter.resolve(value));
662        inserter.function.dfg[block].set_terminator(terminator);
663    }
664}
