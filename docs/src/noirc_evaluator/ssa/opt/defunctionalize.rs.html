1//! This module defines the defunctionalization pass for the SSA IR.
2//! Certain IR targets (e.g., Brillig and ACIR) do not support higher-order functions directly.
3//!
4//! The pass eliminates higher-order functions (a function which accepts function values as arguments or returns functions)
5//! by transforming functions used as values (i.e., first-class functions)
6//! into constant numbers (fields) that represent their function IDs.
7//!
8//! Defunctionalization handles higher-order functions by lowering function values into
9//! constant identifiers and replacing calls of function values with calls to a single
10//! dispatch `apply` function.
11//!
12//! ## How the pass works:
13//! - Every function used as a value (e.g., passed as a parameter) is assigned a unique [NumericType::NativeField] value.
14//!   This value now represents the first-class function's ID.
15//! - All call instructions with non-literal targets are replaced by calls to an `apply` function.
16//! - The `apply` function is a dispatcher. It takes the function ID as its first argument
17//!   and calls the appropriate function based on that ID.
18//!
19//! Pseudocode of an `apply` function is given below:
20//! ```text
21//! fn apply(function_id: Field, arg1: Field, arg2: Field) -> Field {
22//!     match function_id {
23//!         0 -> function0(arg1, arg2),
24//!         1 -> function1(arg1, arg2),
25//!         ...
26//!         N -> functionN(arg1, arg2),
27//!     }
28//! }
29//! ```
30//!
31//! After this pass all first-class functions are replaced with numeric IDs
32//! and calls are routed via the newly generated `apply` functions.

56/// Represents an 'apply' function created by this pass to dispatch higher order functions to.
57/// Pseudocode of an `apply` function is given below:
58/// ```text
59/// fn apply(function_id: Field, arg1: Field, arg2: Field) -> Field {
60///     match function_id {
61///         0 -> function0(arg1, arg2),
62///         1 -> function1(arg1, arg2),
63///         ...
64///         N -> functionN(arg1, arg2),
65///     }
66/// }
67/// ```
68/// Apply functions generally take the function to apply as their first parameter. This is a Field value
69/// obtained by converting the FunctionId into a Field. The remaining parameters of apply are the
70/// arguments to forward to this function when calling it internally.
71#[derive(Debug, Clone, Copy)]
72struct ApplyFunction {
73    id: FunctionId,
74    dispatches_to_multiple_functions: bool,
75}

77/// All functions used as a value that share the same signature and runtime type
78/// Maps ([Signature], Caller [RuntimeType]) -> Vec<([FunctionId], Callee [RuntimeType])>
79type Variants = BTreeMap<(Signature, RuntimeType), Vec<(FunctionId, RuntimeType)>>;
80/// All generated apply functions for each grouping of function variants.
81/// Each apply function is handles a specific ([Signature], [RuntimeType]) group.
82/// Maps ([Signature], [RuntimeType]) -> [ApplyFunction]
83type ApplyFunctions = HashMap<(Signature, RuntimeType), ApplyFunction>;

85/// Performs defunctionalization on all functions
86/// This is done by changing all functions as value to be a number (FieldElement)
87/// And creating apply functions that dispatch to the correct target by runtime comparisons with constants
88#[derive(Debug, Clone)]
89struct DefunctionalizationContext {
90    apply_functions: ApplyFunctions,
91}

93impl Ssa {
94    /// See [`defunctionalize`][self] module for more information.
95    #[tracing::instrument(level = "trace", skip(self))]
96    pub(crate) fn defunctionalize(mut self) -> Ssa {
97        // Check that we have removed all cases we don't handle in this pass.
98        #[cfg(debug_assertions)]
99        self.functions.values().for_each(defunctionalize_pre_check);
100
101        // Find all functions used as value that share the same signature and runtime type
102        let variants = find_variants(&self);
103
104        // Generate the apply functions for the provided variants
105        let (apply_functions, purities) = create_apply_functions(&mut self, variants);
106
107        // Setup the pass context
108        let context = DefunctionalizationContext { apply_functions };
109
110        // Run defunctionalization over all functions in the SSA
111        context.defunctionalize_all(&mut self);
112
113        let purities = Arc::new(purities);
114        for function in self.functions.values_mut() {
115            function.dfg.set_function_purities(purities.clone());
116        }
117
118        // Check that we have established the properties expected from this pass.
119        #[cfg(debug_assertions)]
120        self.functions.values().for_each(defunctionalize_post_check);
121
122        self
123    }
124}

126impl DefunctionalizationContext {
127    /// Defunctionalize all functions in the SSA
128    fn defunctionalize_all(self, ssa: &mut Ssa) {
129        for function in ssa.functions.values_mut() {
130            // We mutate value types in `defunctionalize`, so to prevent that from affecting which
131            // apply functions are chosen we replace all first-class function calls with calls to
132            // the appropriate apply function beforehand.
133            self.replace_first_class_calls_with_apply_function(function);
134
135            // Replace any first-class function values with field values. This will also mutate the
136            // type of some values, such as block arguments
137            self.defunctionalize(function);
138        }
139    }
140
141    /// Replaces any function calls using first-class function values with calls to the
142    /// appropriate `apply` function. Note that this must be done before types are mutated
143    /// in `defunctionalize` since this uses the pre-mutated types to query apply functions.
144    fn replace_first_class_calls_with_apply_function(&self, func: &mut Function) {
145        for block_id in func.reachable_blocks() {
146            let block = &mut func.dfg[block_id];
147
148            #[allow(clippy::unnecessary_to_owned)] // clippy is wrong here
149            for instruction_id in block.instructions().to_vec() {
150                let instruction = &func.dfg[instruction_id];
151
152                // Operate on call instructions
153                let (target_func_id, arguments) = match &instruction {
154                    Instruction::Call { func: target_func_id, arguments } => {
155                        (*target_func_id, arguments)
156                    }
157                    _ => continue,
158                };
159
160                // If the target is a function used as value
161                use Value::Param;
162                if matches!(&func.dfg[target_func_id], Param { .. } | Value::Instruction { .. }) {
163                    let mut arguments = arguments.clone();
164                    let results = func.dfg.instruction_results(instruction_id);
165                    let signature = Signature {
166                        params: vecmap(&arguments, |param| func.dfg.type_of_value(*param)),
167                        returns: vecmap(results, |result| func.dfg.type_of_value(*result)),
168                    };
169
170                    // Find the correct apply function
171                    let Some(apply_function) = self.get_apply_function(signature, func.runtime())
172                    else {
173                        // We should have generated an apply function for all function's used as a value,
174                        // even if there are no variants for that function call.
175                        panic!(
176                            "ICE: It is expected to have an apply function for every function used as a value"
177                        );
178                    };
179
180                    // Replace the instruction with a call to apply
181                    let apply_function_value_id = func.dfg.import_function(apply_function.id);
182                    if apply_function.dispatches_to_multiple_functions {
183                        arguments.insert(0, target_func_id);
184                    }
185                    let func_id = apply_function_value_id;
186                    let replacement_instruction = Instruction::Call { func: func_id, arguments };
187                    func.dfg[instruction_id] = replacement_instruction;
188                }
189            }
190        }
191    }

193    /// Defunctionalize a single function
194    fn defunctionalize(&self, func: &mut Function) {
195        for block_id in func.reachable_blocks() {
196            let block = &mut func.dfg[block_id];
197
198            // Temporarily take the parameters here just to avoid cloning them
199            let parameters = block.take_parameters();
200            for parameter in &parameters {
201                let typ = &func.dfg.type_of_value(*parameter);
202                if let Some(rep) = replacement_type(typ) {
203                    func.dfg.set_type_of_value(*parameter, rep);
204                }
205            }
206
207            let block = &mut func.dfg[block_id];
208            block.set_parameters(parameters);
209
210            // Do the same for the terminator
211            let mut terminator = block.take_terminator();
212            terminator.map_values_mut(|value| map_function_to_field(func, value).unwrap_or(value));
213
214            let block = &mut func.dfg[block_id];
215            block.set_terminator(terminator);
216
217            // Now we can finally change each instruction, replacing
218            // each first class function with a field value.
219            #[allow(clippy::unnecessary_to_owned)] // clippy is wrong here
220            for instruction_id in block.instructions().to_vec() {
221                let mut instruction = func.dfg[instruction_id].clone();
222
223                if remove_first_class_functions_in_instruction(func, &mut instruction) {
224                    func.dfg[instruction_id] = instruction;
225                }
226
227                #[allow(clippy::unnecessary_to_owned)] // clippy is wrong here
228                for result in func.dfg.instruction_results(instruction_id).to_vec() {
229                    let typ = &func.dfg.type_of_value(result);
230                    if let Some(rep) = replacement_type(typ) {
231                        func.dfg.set_type_of_value(result, rep);
232                    }
233                }
234            }
235        }
236    }

238    /// Returns the apply function for the given signature
239    fn get_apply_function(
240        &self,
241        signature: Signature,
242        runtime: RuntimeType,
243    ) -> Option<ApplyFunction> {
244        self.apply_functions.get(&(signature, runtime)).copied()
245    }

248/// Replace any first class functions used in an instruction with a field value.
249/// This applies to any function used anywhere else other than the function position
250/// of a call instruction. Returns true if the instruction was modified
251fn remove_first_class_functions_in_instruction(
252    func: &mut Function,
253    instruction: &mut Instruction,
254) -> bool {
255    let mut modified = false;
256    let mut map_value = |value: ValueId| {
257        if let Some(new_value) = map_function_to_field(func, value) {
258            modified = true;
259            new_value
260        } else {
261            value
262        }
263    };
264
265    if let Instruction::Call { func: _, arguments } = instruction {
266        for arg in arguments {
267            *arg = map_value(*arg);
268        }
269    } else if let Instruction::MakeArray { typ, .. } = instruction {
270        let mut modified_type = false;
271        if let Some(rep) = replacement_type(typ) {
272            *typ = rep;
273            modified_type = true;
274        }
275
276        instruction.map_values_mut(map_value);
277
278        if modified_type {
279            modified = true;
280        }
281    } else {
282        instruction.map_values_mut(map_value);
283    }
284
285    modified
286}

288/// Try to map the given function literal to a field, returning Some(field) on success.
289/// Returns none if the given value was not a function or doesn't need to be mapped.
290fn map_function_to_field(func: &mut Function, value: ValueId) -> Option<ValueId> {
291    let typ = func.dfg[value].get_type();
292    if let Some(rep) = replacement_type(typ.as_ref()) {
293        match &func.dfg[value] {
294            // If the value is a static function, transform it to the function id
295            Value::Function(id) => {
296                let new_value = function_id_to_field(*id);
297                return Some(func.dfg.make_constant(new_value, NumericType::NativeField));
298            }
299            // If the value is a function used as value, just change the type of it
300            Value::Instruction { .. } | Value::Param { .. } => {
301                func.dfg.set_type_of_value(value, rep);
302            }
303            _ => (),
304        }
305    }
306    None
307}

309/// Collects all functions used as values that can be called by their signatures.
310///
311/// Groups all [FunctionId]s used as values by their [Signature] and caller [RuntimeType],
312/// producing a mapping from these tuples to the list of target functions that may be dynamically dispatched.
313///
314/// # Arguments
315/// - `ssa`: The full [Ssa] structure
316///
317/// # Returns
318/// [Variants] that should then be used to generate apply functions for dispatching
319fn find_variants(ssa: &Ssa) -> Variants {
320    let mut dynamic_dispatches: BTreeSet<(Signature, RuntimeType)> = BTreeSet::new();
321    let mut functions_as_values: BTreeSet<FunctionId> = BTreeSet::new();
322
323    for function in ssa.functions.values() {
324        functions_as_values.extend(find_functions_as_values(function));
325        dynamic_dispatches.extend(
326            find_dynamic_dispatches(function).into_iter().map(|sig| (sig, function.runtime())),
327        );
328    }
329
330    // Group function variant candidates by their signature
331    let mut signature_to_functions_as_value: BTreeMap<Signature, Vec<FunctionId>> = BTreeMap::new();
332
333    for function_id in functions_as_values {
334        let signature = ssa.functions[&function_id].signature();
335        signature_to_functions_as_value.entry(signature).or_default().push(function_id);
336    }
337
338    let mut variants: Variants = BTreeMap::new();
339
340    // Further group function variant candidates by their caller runtime.
341    // The target functions are never tested against their runtime, which is assumed to be correct,
342    // but there is nothing which guarantees this.
343    // Filtering the functions by their runtime at this stage will not work because if the variant
344    // is empty, the apply function will create a dummy function. So instead of calling a function with a wrong runtime,
345    // we would end-up calling a dummy function.
346    // The solution is to add the runtime information to the variants map
347    // and defer to create_apply_function for handling the runtime checks.
348    for (dispatch_signature, caller_runtime) in dynamic_dispatches {
349        let target_fns = signature_to_functions_as_value
350            .get(&dispatch_signature)
351            .cloned()
352            .unwrap_or_default()
353            .into_iter()
354            .map(|f| (f, ssa.functions[&f].runtime()));
355        let target_fns: Vec<(FunctionId, RuntimeType)> = target_fns.collect();
356        variants.insert((dispatch_signature, caller_runtime), target_fns);
357    }
358
359    // We will now have fully constructed our variants map and can return it
360    variants
361}

363/// Finds all literal functions used as values in the given function
364fn find_functions_as_values(func: &Function) -> BTreeSet<FunctionId> {
365    let mut functions_as_values: BTreeSet<FunctionId> = BTreeSet::new();
366
367    visit_values_other_than_call_target(func, |value| {
368        if let Value::Function(id) = value {
369            functions_as_values.insert(*id);
370        }
371    });
372
373    functions_as_values
374}

376/// Visit all values which are *not* targets of a `Call`.
377fn visit_values_other_than_call_target(func: &Function, mut f: impl FnMut(&Value)) {
378    let mut process_value = |value_id: ValueId| {
379        f(&func.dfg[value_id]);
380    };
381
382    for block_id in func.reachable_blocks() {
383        let block = &func.dfg[block_id];
384        for instruction_id in block.instructions() {
385            let instruction = &func.dfg[*instruction_id];
386
387            // Handle call instructions separately. Functions used in their function field
388            // don't have to be first-class values.
389            if let Instruction::Call { arguments, .. } = instruction {
390                arguments.iter().for_each(|value_id| process_value(*value_id));
391            } else {
392                instruction.for_each_value(&mut process_value);
393            }
394        }
395
396        block.unwrap_terminator().for_each_value(&mut process_value);
397    }
398}

400/// Finds all dynamic dispatch signatures in the given function.
401/// Note that these are the signatures before function types are mutated to turn into field types.
402///
403/// A dynamic dispatch is defined as a call into a function value where that
404/// value comes from a parameter (i.e., calling a function passed as a function parameter
405/// or another instruction (i.e., calling a function returned from another function call).
406fn find_dynamic_dispatches(func: &Function) -> BTreeSet<Signature> {
407    let mut dispatches = BTreeSet::new();
408
409    for block_id in func.reachable_blocks() {
410        let block = &func.dfg[block_id];
411        for instruction_id in block.instructions() {
412            let instruction = &func.dfg[*instruction_id];
413            match instruction {
414                Instruction::Call { func: target, arguments } => {
415                    if let Value::Param { .. } | Value::Instruction { .. } = &func.dfg[*target] {
416                        let results = func.dfg.instruction_results(*instruction_id);
417                        dispatches.insert(Signature {
418                            params: vecmap(arguments, |param| func.dfg.type_of_value(*param)),
419                            returns: vecmap(results, |result| func.dfg.type_of_value(*result)),
420                        });
421                    }
422                }
423                _ => continue,
424            }
425        }
426    }
427    dispatches
428}

430/// Creates all apply functions needed for dispatch of function values.
431///
432/// This function maintains the grouping set in [Variants], meaning an apply
433/// function is grouped by functions that share a target signature and caller runtime.
434///
435/// An apply function is only created if there are multiple function variants
436/// for a specific ([Signature], [RuntimeType]) group.
437/// Otherwise, if there is a single variant that function is simply reused.
438///
439/// If there are no variants a dummy function is created.
440/// A dummy function acts as a safe no-op to continue compilation even though there are no variants
441/// for a first-class function call. For more information you can reference [create_dummy_function].
442///
443/// # Arguments
444/// - `ssa`: A mutable reference to the full [Ssa] structure containing all functions.
445/// - `variants_map`:  [Variants]
446///
447/// # Returns
448/// - [ApplyFunctions] keyed by each function's signature _before_ functions are changed
449///   into field types. The inner apply function itself will have its defunctionalized type,
450///   with function values represented as field values.
451/// - [HashMap<FunctionId, Purity>] with purities that must be set to all functions in the SSA,
452///   as this function might have created dummy pure functions.
453fn create_apply_functions(
454    ssa: &mut Ssa,
455    variants_map: Variants,
456) -> (ApplyFunctions, HashMap<FunctionId, Purity>) {
457    let mut apply_functions = HashMap::default();
458    let mut purities = if ssa.functions.is_empty() {
459        HashMap::default()
460    } else {
461        (*ssa.functions.iter().next().unwrap().1.dfg.function_purities).clone()
462    };
463
464    for ((signature, caller_runtime), variants) in variants_map.into_iter() {
465        // Calling an ACIR function from a Brillig runtime is not allowed.
466        // We expect all ACIR functions called from Brillig to be specialized
467        // as Brillig functions at compile time (e.g., before SSA generation).
468        // Calling a Brillig function from ACIR is allowed, but we avoid creating dispatch functions to variants
469        // which would pass invalid types across runtime boundaries.
470        // Closures are expected to be duplicated by their runtime.
471        // It is expected that the frontend has appropriately extracted the correct closure
472        // based on the current runtime of the caller function.
473        // However, during defunctionalization we are unable to tell that the correct target is being called so all
474        // variants with different runtimes will be included in the apply function.
475        // Including a closure variant with a different runtime in the closure variant will cause an ill-formed SSA.
476        // Thus, to avoid a broken SSA we simply do not include calls to variants which differ from their callers.
477        // It is expected that if the new ID associated with the bad variant is used, that we will still fail as the last
478        // function to dispatch constrains that we have an expected ID.
479
480        let pre_runtime_filter_len = variants.len();
481        let variants: Vec<(FunctionId, RuntimeType)> = variants
482            .into_iter()
483            .filter(|(_, callee_runtime)| {
484                // Note that the Inline property is ignored.
485                caller_runtime.is_brillig() && callee_runtime.is_brillig()
486                    || caller_runtime.is_acir() && callee_runtime.is_acir()
487                    || caller_runtime.is_acir()
488                        && callee_runtime.is_brillig()
489                        && is_valid_across_boundaries(&signature)
490            })
491            .collect();
492
493        let dispatches_to_multiple_functions = variants.len() > 1;
494
495        // This will be the same signature but with each function type replaced with
496        // a Field type.
497        let mut defunctionalized_signature = signature.clone();
498
499        // Update the shared function signature of the higher-order function variants
500        // to replace any function passed as a value to a numeric field type.
501        for typ in defunctionalized_signature
502            .params
503            .iter_mut()
504            .chain(&mut defunctionalized_signature.returns)
505        {
506            if let Some(rep) = replacement_type(typ) {
507                *typ = rep;
508            }
509        }
510
511        let id = if dispatches_to_multiple_functions {
512            // If we have multiple variants for this signature and runtime type group
513            // we need to generate an apply function.
514            create_apply_function(ssa, defunctionalized_signature, caller_runtime, variants)
515        } else if !variants.is_empty() {
516            // If there is only variant, we can use it directly rather than creating a new apply function.
517            variants[0].0
518        } else if pre_runtime_filter_len != 0 && caller_runtime.is_brillig() {
519            // We had variants, but they were all filtered out.
520            // Frontend bug: only ACIR variants in a Brillig group.
521            panic!("ICE: invalid defunctionalization: only ACIR variants for a Brillig runtime");
522        } else {
523            // If no variants exist for a dynamic call we leave removing those dead calls and parameters to DIE.
524            // However, we have to construct a dummy function for these dead calls as to keep a well formed SSA
525            // and to not break the semantics of other SSA passes before DIE is reached.
526            create_dummy_function(ssa, defunctionalized_signature, caller_runtime, &mut purities)
527        };
528        apply_functions.insert(
529            (signature, caller_runtime),
530            ApplyFunction { id, dispatches_to_multiple_functions },
531        );
532    }
533
534    (apply_functions, purities)
535}

542/// Check if a [Signature] is valid when the caller is ACIR and the callee is Brillig.
543fn is_valid_across_boundaries(signature: &Signature) -> bool {
544    !signature.params.iter().any(|typ| typ.contains_reference())
545        && !signature.returns.iter().any(|typ| {
546            typ.contains_reference() || typ.contains_vector_element() || typ.contains_function()
547        })

550/// Creates a single apply function to enable dispatch across multiple function variants
551/// that share the same [Signature] and [RuntimeType].
552///
553/// This function is responsible for generating an entry point that dispatches between several
554/// concrete functions at runtime based on a target field value. It builds a sequence of
555/// conditional checks (if-else chain) to compare the target against each
556/// function's ID, and calls the matching function.
557///
558/// These apply functions are to be aggressively inlined as it is assumed that they will be optimized
559/// away by the constants at the call site.
560///
561/// # Arguments
562/// - `ssa`: A mutable reference to the full [Ssa] structure containing all functions.
563/// - `signature`: The shared [Signature] of all variants but with each `Type::Function` replaced with a field type.
564/// - `caller_runtime`: The runtime in which the apply function will be called, used to update inlining policies.
565/// - `function_ids`: A non-empty list of [FunctionId]s representing concrete functions to dispatch between.
566///   This method will panic if `function_ids` is empty.
567///
568/// # Returns
569/// The [FunctionId] of the new apply function
570///
571/// # Panics
572/// If the `function_ids` argument has fewer than two elements, implying that no apply function is necessary.
573fn create_apply_function(
574    ssa: &mut Ssa,
575    signature: Signature,
576    caller_runtime: RuntimeType,
577    function_ids: Vec<(FunctionId, RuntimeType)>,
578) -> FunctionId {
579    assert!(
580        function_ids.len() > 1,
581        "create_apply_function is expected to be called with two or more FunctionIds"
582    );
583    // Clone the user-defined globals and the function purities mapping,
584    // which are shared across all functions.
585    // We will be borrowing `ssa` mutably so we need to fetch this shared information
586    // before attempting to add a new function to the SSA.
587    let globals = ssa.main().dfg.globals.clone();
588    ssa.add_fn(|id| {
589        let mut function_builder = FunctionBuilder::new("apply".to_string(), id);
590        function_builder.set_globals(globals);
591
592        // We want to push for apply functions to be inlined more aggressively;
593        // they are expected to be optimized away by constants visible at the call site.
594        let runtime = match caller_runtime {
595            RuntimeType::Acir(_) => RuntimeType::Acir(InlineType::InlineAlways),
596            RuntimeType::Brillig(_) => RuntimeType::Brillig(InlineType::InlineAlways),
597        };
598        function_builder.set_runtime(runtime);
599        // Set up the parameters of the apply function
600        // The first argument is the target function ID for which are dispatching a call
601        let target_id = function_builder.add_parameter(Type::field());
602        // The remaining apply function parameters are the actual parameters of the variants for which we are dispatching calls
603        let params_ids = vecmap(signature.params, |typ| function_builder.add_parameter(typ));
604
605        let entry_block = function_builder.current_block();
606
607        let make_end_block = |builder: &mut FunctionBuilder| -> (BasicBlockId, Vec<ValueId>) {
608            let block = builder.insert_block();
609            // The values that will be ultimately returned from the function.
610            let params =
611                vecmap(&signature.returns, |typ| builder.add_block_parameter(block, typ.clone()));
612            (block, params)
613        };
614
615        let (end_block, end_results) = make_end_block(&mut function_builder);
616        let mut blocks_to_merge = Vec::with_capacity(function_ids.len());
617
618        // Switch back to the entry block to build the rest of the dispatch function
619        function_builder.switch_to_block(entry_block);
620
621        for (index, (function_id, _)) in function_ids.iter().enumerate() {
622            let is_last = index == function_ids.len() - 1;
623            let mut next_function_block = None;
624
625            let function_id_constant = function_builder
626                .numeric_constant(function_id_to_field(*function_id), NumericType::NativeField);
627
628            // If it's not the last function to dispatch, create an if statement
629            if !is_last {
630                next_function_block = Some(function_builder.insert_block());
631                let executor_block = function_builder.insert_block();
632
633                let condition =
634                    function_builder.insert_binary(target_id, BinaryOp::Eq, function_id_constant);
635                function_builder.terminate_with_jmpif(
636                    condition,
637                    executor_block,
638                    next_function_block.unwrap(),
639                );
640                function_builder.switch_to_block(executor_block);
641            } else {
642                // Else just constrain the condition
643                function_builder.insert_constrain(target_id, function_id_constant, None);
644            }
645
646            // Call the function variant
647            let target_function_value = function_builder.import_function(*function_id);
648            let call_results = function_builder
649                .insert_call(target_function_value, params_ids.clone(), signature.returns.clone())
650                .to_vec();
651
652            let local_end_block = make_end_block(&mut function_builder);
653            // Jump to the end block
654            function_builder.terminate_with_jmp(local_end_block.0, call_results);
655            blocks_to_merge.push(local_end_block);
656
657            if let Some(next_block) = next_function_block {
658                // Switch to the next block for the else branch
659                function_builder.switch_to_block(next_block);
660            }
661        }
662
663        // Merge blocks as last-in first-out:
664        //
665        // local_end_block0-----------------------------------------\
666        //                                                           end block
667        //                                                          /
668        // local_end_block1---------------------\                  /
669        //                                       new merge block2-/
670        // local_end_block2--\                  /
671        //                    new merge block1-/
672        // local_end_block3--/
673        //
674        // This is necessary since SSA panics during flattening if we immediately
675        // try to jump directly to end block instead
676        // (see https://github.com/noir-lang/noir/issues/7323 for a case where this happens).
677        //
678        // It'd also be more efficient to merge them tournament-bracket style but that
679        // also leads to panics during flattening for similar reasons.
680        while let Some((block, results)) = blocks_to_merge.pop() {
681            function_builder.switch_to_block(block);
682
683            if let Some((block2, results2)) = blocks_to_merge.pop() {
684                // Merge two blocks in the queue together
685                let (new_merge, new_merge_results) = make_end_block(&mut function_builder);
686                blocks_to_merge.push((new_merge, new_merge_results));
687
688                function_builder.terminate_with_jmp(new_merge, results);
689
690                function_builder.switch_to_block(block2);
691                function_builder.terminate_with_jmp(new_merge, results2);
692            } else {
693                // Finally done, jump to the end
694                function_builder.terminate_with_jmp(end_block, results);
695            }
696        }
697
698        function_builder.switch_to_block(end_block);
699        function_builder.terminate_with_return(end_results);
700
701        // The above code can result in a suboptimal CFG so we simplify it here.
702        let mut function = function_builder.current_function;
703        function.simplify_function();
704        function
705    })
706}

708/// Creates a placeholder (dummy) function to replace calls to invalid function references.
709/// An example of a possible invalid function reference is an out-of-bounds access on a zero-length function array.
710///
711/// This prevents the compiler from crashing by ensuring that the IR always has a valid function to call.
712/// The dummy function is created using the supplied function's signature as to maintain a well formed SSA IR.
713/// The dummy function is pure, contains no logic, and just returns zeroed out values for its return types.
714///
715/// This is especially useful in cases where we cannot statically resolve the function reference,
716/// but want to continue compiling the rest of the program safely.
717///
718/// Returns the [FunctionId] of the newly created dummy function.
719fn create_dummy_function(
720    ssa: &mut Ssa,
721    signature: Signature,
722    caller_runtime: RuntimeType,
723    purities: &mut HashMap<FunctionId, Purity>,
724) -> FunctionId {
725    ssa.add_fn(|id| {
726        let mut function_builder = FunctionBuilder::new("apply_dummy".to_string(), id);
727
728        // Set the runtime of the dummy function. The dummy function is expect to always be simplified out
729        // but we let the caller set the runtime here as to match Noir's runtime semantics.
730        let runtime = match caller_runtime {
731            RuntimeType::Acir(_) => RuntimeType::Acir(InlineType::InlineAlways),
732            RuntimeType::Brillig(_) => RuntimeType::Brillig(InlineType::InlineAlways),
733        };
734        function_builder.set_runtime(runtime);
735
736        // The remaining dummy function parameters are the actual parameters of the function call without any variants.
737        // We generate these just to maintain a well-formed IR. Not doing this could result in errors if the dummy function
738        // was set to be inlined before the call to it was removed by DIE.
739        vecmap(signature.params, |typ| function_builder.add_parameter(typ));
740
741        // We can mark the dummy function pure as all it does is return.
742        // As the dummy function is just meant to be a placeholder for any calls to
743        // higher-order functions without variants, we want the function to be marked pure
744        // so that dead instruction elimination can remove any calls to it.
745        purities.insert(id, Purity::Pure);
746
747        let results =
748            vecmap(signature.returns, |typ| make_dummy_return_data(&mut function_builder, &typ));
749
750        function_builder.terminate_with_return(results);
751
752        function_builder.current_function
753    })
754}

756/// Construct a dummy value to be returned from the placeholder function for calls to invalid lambda references.
757/// We need to construct the appropriate value so that our SSA is well formed even if the function
758/// pointer has no variants.
759fn make_dummy_return_data(function_builder: &mut FunctionBuilder, typ: &Type) -> ValueId {
760    match typ {
761        Type::Numeric(numeric_type) => function_builder.numeric_constant(0_u128, *numeric_type),
762        Type::Array(element_types, len) => {
763            let mut array = im::Vector::new();
764            for _ in 0..len.0 {
765                for typ in element_types.iter() {
766                    array.push_back(make_dummy_return_data(function_builder, typ));
767                }
768            }
769            function_builder.insert_make_array(array, typ.clone())
770        }
771        Type::Vector(_) => {
772            let array = im::Vector::new();
773            // The contents of a vector do not matter for a dummy function, we simply
774            // desire to have a well formed SSA by returning the correct value for a type.
775            // Thus, we return an empty vector here.
776            function_builder.insert_make_array(array, typ.clone())
777        }
778        Type::Reference(element_type) => function_builder.insert_allocate((**element_type).clone()),
779        Type::Function => {
780            unreachable!(
781                "ICE: Any function passed as a value should have already been converted to a field type"
782            )
783        }
784    }
785}

820/// Return what type a function value type should be replaced with:
821/// * Global functions are replaced with a `Field`.
822/// * Function references are replaced with a reference to the replacement type of the underlying type, recursively.
823/// * Array and vectors that contain function types are handled recursively.
824///
825/// If the type doesn't need replacement, `None` is returned.
826fn replacement_type(typ: &Type) -> Option<Type> {
827    match typ {
828        Type::Function => Some(Type::field()),
829        Type::Reference(typ) => {
830            replacement_type(typ.as_ref()).map(|typ| Type::Reference(Arc::new(typ)))
831        }
832        Type::Numeric(_) => None,
833        Type::Array(items, size) => {
834            replacement_types(items.as_ref()).map(|types| Type::Array(Arc::new(types), *size))
835        }
836        Type::Vector(items) => {
837            replacement_types(items.as_ref()).map(|types| Type::Vector(Arc::new(types)))
838        }
839    }
840}
841
842/// Take a list of types that might need replacement.
843/// Replaces the ones that need replacement, leaving all others as-is.
844/// If no type needs replacement, `None` is returned.
845fn replacement_types(types: &[Type]) -> Option<Vec<Type>> {
846    let mut reps = Vec::new();
847    let mut has_rep = false;
848    for typ in types {
849        let rep = replacement_type(typ);
850        has_rep |= rep.is_some();
851        reps.push(rep);
852    }
853    if !has_rep {
854        None
855    } else {
856        Some(
857            reps.into_iter()
858                .zip(types)
859                .map(|(rep, typ)| rep.unwrap_or_else(|| typ.clone()))
860                .collect(),
861        )
862    }
863}
