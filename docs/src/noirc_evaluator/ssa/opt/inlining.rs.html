35/// An arbitrary limit to the maximum number of recursive call
36/// frames at any point in time.
37const RECURSION_LIMIT: u32 = 1000;
38
39impl Ssa {
40    /// Inline all functions within the IR.
41    ///
42    /// In the case of recursive Acir functions, this will attempt
43    /// to recursively inline until the RECURSION_LIMIT is reached.
44    ///
45    /// Functions are recursively inlined into main until either we finish
46    /// inlining all functions or we encounter a function whose function id is not known.
47    /// When the later happens, the call instruction is kept in addition to the function
48    /// it refers to. The function it refers to is kept unmodified without any inlining
49    /// changes. This is because if the function's id later becomes known by a later
50    /// pass, we would need to re-run all of inlining anyway to inline it, so we might
51    /// as well save the work for later instead of performing it twice.
52    ///
53    /// There are some attributes that allow inlining a function at a different step of codegen.
54    /// Currently this is just `InlineType::NoPredicates` for which we have a flag indicating
55    /// whether treating that inline functions. The default is to treat these functions as entry points.
56    ///
57    /// This step should run after runtime separation, since it relies on the runtime of the called functions being final.
58    #[tracing::instrument(level = "trace", skip(self))]
59    pub(crate) fn inline_functions(
60        self,
61        aggressiveness: i64,
62        small_function_max_instructions: usize,
63    ) -> Result<Ssa, RuntimeError> {
64        self.inline_until_fixed_point(aggressiveness, small_function_max_instructions, false)
65    }
66
67    /// Run the inlining pass where functions marked with `InlineType::NoPredicates` as not entry points
68    pub(crate) fn inline_functions_with_no_predicates(
69        self,
70        aggressiveness: i64,
71        small_function_max_instructions: usize,
72    ) -> Result<Ssa, RuntimeError> {
73        self.inline_until_fixed_point(aggressiveness, small_function_max_instructions, true)
74    }
75
76    /// Inline functions repeatedly until no new functions are inlined.
77    pub(crate) fn inline_until_fixed_point(
78        mut self,
79        aggressiveness: i64,
80        small_function_max_instructions: usize,
81        inline_no_predicates_functions: bool,
82    ) -> Result<Ssa, RuntimeError> {
83        loop {
84            let num_functions_before = self.functions.len();
85
86            let call_graph = CallGraph::from_ssa_weighted(&self);
87
88            let inline_infos = compute_inline_infos(
89                &self,
90                &call_graph,
91                inline_no_predicates_functions,
92                small_function_max_instructions,
93                aggressiveness,
94            );
95
96            // Bottom-up order, starting with the "leaf" functions.
97            let bottom_up = compute_bottom_up_order(&self, &call_graph);
98            let bottom_up = vecmap(bottom_up, |(id, _)| id);
99
100            self = Self::inline_functions_inner(self, &inline_infos, &bottom_up)?;
101
102            let num_functions_after = self.functions.len();
103            if num_functions_after == num_functions_before {
104                break;
105            }
106        }
107
108        Ok(self)
109    }
110
111    /// Inline entry points in the order of appearance in `inline_infos`, assuming it goes in bottom-up order.
112    fn inline_functions_inner(
113        mut self,
114        inline_infos: &InlineInfos,
115        bottom_up: &[FunctionId],
116    ) -> Result<Ssa, RuntimeError> {
117        let inline_targets = bottom_up
118            .iter()
119            .filter_map(|id| {
120                let info = inline_infos.get(id)?;
121                let dfg = &self.functions[id].dfg;
122                info.is_inline_target(dfg).then_some(*id)
123            })
124            .collect::<Vec<_>>();
125
126        let should_inline_call = |callee: &Function| -> bool {
127            // We defer to the inline info computation to determine whether a function should be inlined
128            InlineInfo::should_inline(inline_infos, callee.id())
129        };
130
131        // We are going bottom up, so hopefully we can inline leaf functions into their callers and retain less memory.
132        let mut new_functions = HashSet::new();
133        for entry_point in inline_targets {
134            let function = &self.functions[&entry_point];
135            let inlined = function.inlined(&self, &should_inline_call)?;
136            assert_eq!(inlined.id(), entry_point);
137            self.functions.insert(entry_point, inlined);
138            new_functions.insert(entry_point);
139        }
140
141        // Drop functions that weren't inline targets.
142        self.functions.retain(|id, _| new_functions.contains(id));
143
144        Ok(self)
145    }
146}

159/// The context for the function inlining pass.
160///
161/// This works using an internal FunctionBuilder to build a new main function from scratch.
162/// Doing it this way properly handles importing instructions between functions and lets us
163/// reuse the existing API at the cost of essentially cloning each of main's instructions.
164struct InlineContext {
165    recursion_level: u32,
166    builder: FunctionBuilder,
167
168    call_stack: CallStackId,
169
170    // The FunctionId of the entry point function we're inlining into in the old, unmodified Ssa.
171    entry_point: FunctionId,
172}
173
174/// The per-function inlining context contains information that is only valid for one function.
175/// For example, each function has its own DataFlowGraph, and thus each function needs a translation
176/// layer to translate between BlockId to BlockId for the current function and the function to
177/// inline into. The same goes for ValueIds, InstructionIds, and for storing other data like
178/// parameter to argument mappings.
179struct PerFunctionContext<'function> {
180    /// The function that we are inlining calls into.
181    entry_function: &'function Function,
182
183    /// The source function is the function we're currently inlining into the function being built.
184    source_function: &'function Function,
185
186    /// The shared inlining context for all functions. This notably contains the FunctionBuilder used
187    /// to build the function we're inlining into.
188    context: &'function mut InlineContext,
189
190    /// Maps ValueIds in the function being inlined to the new ValueIds to use in the function
191    /// being inlined into. This mapping also contains the mapping from parameter values to
192    /// argument values.
193    values: HashMap<ValueId, ValueId>,
194
195    /// Maps blocks in the source function to blocks in the function being inlined into, where
196    /// each mapping is from the start of a source block to an inlined block in which the
197    /// analogous program point occurs.
198    ///
199    /// Note that the starts of multiple source blocks can map into a single inlined block.
200    /// Conversely the whole of a source block is not guaranteed to map into a single inlined
201    /// block.
202    blocks: HashMap<BasicBlockId, BasicBlockId>,
203
204    /// True if we're currently working on the entry point function.
205    inlining_entry: bool,
206}
207
208impl InlineContext {
209    /// Create a new context object for the function inlining pass.
210    /// This starts off with an empty mapping of instructions for main's parameters.
211    /// The function being inlined into will always be the main function, although it is
212    /// actually a copy that is created in case the original main is still needed from a function
213    /// that could not be inlined calling it.
214    fn new(ssa: &Ssa, entry_point: FunctionId) -> Self {
215        let source = &ssa.functions[&entry_point];
216        let builder = FunctionBuilder::from_existing(source, entry_point);
217        Self { builder, recursion_level: 0, entry_point, call_stack: CallStackId::root() }
218    }
219
220    /// Start inlining the entry point function and all functions reachable from it.
221    fn inline_all(
222        mut self,
223        ssa: &Ssa,
224        should_inline_call: &impl Fn(&Function) -> bool,
225    ) -> Result<Function, RuntimeError> {
226        let entry_point = &ssa.functions[&self.entry_point];
227
228        let mut context = PerFunctionContext::new(&mut self, entry_point, entry_point);
229        context.inlining_entry = true;
230
231        // The entry block is already inserted so we have to add it to context.blocks and add
232        // its parameters here. Failing to do so would cause context.translate_block() to add
233        // a fresh block for the entry block rather than use the existing one.
234        let entry_block = context.context.builder.current_function.entry_block();
235        context.translate_block_parameters(context.source_function.entry_block(), entry_block);
236        context.blocks.insert(context.source_function.entry_block(), entry_block);
237        context.inline_blocks(ssa, should_inline_call)?;
238        // translate databus values
239        let databus = entry_point.dfg.data_bus.map_values(|t| context.translate_value(t));
240
241        // Finally, we should have 1 function left representing the inlined version of the target function.
242        let mut new_ssa = self.builder.finish();
243        assert_eq!(new_ssa.functions.len(), 1);
244        let mut new_func = new_ssa.functions.pop_first().unwrap().1;
245        new_func.dfg.data_bus = databus;
246        Ok(new_func)
247    }
248
249    /// Inlines a function into the current function and returns the translated return values
250    /// of the inlined function.
251    fn inline_function(
252        &mut self,
253        ssa: &Ssa,
254        id: FunctionId,
255        arguments: &[ValueId],
256        should_inline_call: &impl Fn(&Function) -> bool,
257    ) -> Result<Vec<ValueId>, RuntimeError> {
258        self.recursion_level += 1;
259
260        let source_function = &ssa.functions[&id];
261        if self.recursion_level > RECURSION_LIMIT {
262            return Err(RuntimeError::RecursionLimit {
263                function_name: source_function.name().to_string(),
264                limit: RECURSION_LIMIT,
265                call_stack: self.builder.current_function.dfg.get_call_stack(self.call_stack),
266            });
267        }
268
269        let entry_point = &ssa.functions[&self.entry_point];
270        let mut context = PerFunctionContext::new(self, entry_point, source_function);
271
272        let parameters = source_function.parameters();
273        assert_eq!(parameters.len(), arguments.len());
274        context.values = parameters.iter().copied().zip(arguments.iter().copied()).collect();
275
276        let current_block = context.context.builder.current_block();
277        context.blocks.insert(source_function.entry_block(), current_block);
278
279        let return_values = context.inline_blocks(ssa, should_inline_call)?;
280        self.recursion_level -= 1;
281        Ok(return_values)
282    }
283}

286    /// Create a new PerFunctionContext from the source function.
287    /// The value and block mappings for this context are initially empty except
288    /// for containing the mapping between parameters in the source_function and
289    /// the arguments of the destination function.
290    fn new(
291        context: &'function mut InlineContext,
292        entry_function: &'function Function,
293        source_function: &'function Function,
294    ) -> Self {
295        Self {
296            context,
297            entry_function,
298            source_function,
299            blocks: HashMap::default(),
300            values: HashMap::default(),
301            inlining_entry: false,
302        }
303    }

305    /// Translates a ValueId from the function being inlined to a ValueId of the function
306    /// being inlined into. Note that this expects value ids for all Value::Instruction and
307    /// Value::Param values are already handled as a result of previous inlining of instructions
308    /// and blocks respectively. If these assertions trigger it means a value is being used before
309    /// the instruction or block that defines the value is inserted.
310    fn translate_value(&mut self, id: ValueId) -> ValueId {
311        if let Some(value) = self.values.get(&id) {
312            return *value;
313        }
314
315        let new_value = match &self.source_function.dfg[id] {
316            value @ Value::Instruction { .. } => {
317                if self.source_function.dfg.is_global(id) {
318                    return id;
319                }
320                unreachable!(
321                    "All Value::Instructions should already be known during inlining after creating the original inlined instruction. Unknown value {id} = {value:?}"
322                )
323            }
324            value @ Value::Param { .. } => {
325                unreachable!(
326                    "All Value::Params should already be known from previous calls to translate_block. Unknown value {id} = {value:?}"
327                )
328            }
329            Value::NumericConstant { constant, typ } => {
330                // The dfg indexes a global's inner value directly, so we need to check here
331                // whether we have a global.
332                if self.source_function.dfg.is_global(id) {
333                    id
334                } else {
335                    self.context.builder.numeric_constant(*constant, *typ)
336                }
337            }
338            Value::Function(function) => self.context.builder.import_function(*function),
339            Value::Intrinsic(intrinsic) => self.context.builder.import_intrinsic_id(*intrinsic),
340            Value::ForeignFunction(function) => {
341                self.context.builder.import_foreign_function(function)
342            }
343            Value::Global(_) => {
344                panic!("Expected a global to be resolved to its inner value");
345            }
346        };
347
348        self.values.insert(id, new_value);
349        new_value
350    }
351
352    /// Translates the program point representing the start of the given `source_block` to the
353    /// inlined block in which the analogous program point occurs. (Once inlined, the source
354    /// block's analogous program region may span multiple inlined blocks.)
355    ///
356    /// If the block isn't already known, this will insert a new block into the target function
357    /// with the same parameter types as the source block.
358    fn translate_block(
359        &mut self,
360        source_block: BasicBlockId,
361        block_queue: &mut VisitOnceDeque,
362    ) -> BasicBlockId {
363        if let Some(block) = self.blocks.get(&source_block) {
364            return *block;
365        }
366
367        // The block is not yet inlined, queue it
368        block_queue.push_back(source_block);
369
370        // The block is not already present in the function being inlined into so we must create it.
371        // The block's instructions are not copied over as they will be copied later in inlining.
372        let new_block = self.context.builder.insert_block();
373        self.translate_block_parameters(source_block, new_block);
374        self.blocks.insert(source_block, new_block);
375        new_block
376    }
377
378    /// Copy block parameters from `source_block` into `target_block`.
379    fn translate_block_parameters(
380        &mut self,
381        source_block: BasicBlockId,
382        target_block: BasicBlockId,
383    ) {
384        let original_parameters = self.source_function.dfg.block_parameters(source_block);
385        for parameter in original_parameters {
386            let typ = self.source_function.dfg.type_of_value(*parameter);
387            let new_parameter = self.context.builder.add_block_parameter(target_block, typ);
388            self.values.insert(*parameter, new_parameter);
389        }
390    }

392    /// Try to retrieve the function referred to by the given Id.
393    /// Expects that the given ValueId belongs to the source_function.
394    ///
395    /// Returns None if the id is not known to refer to a function.
396    fn get_function(&mut self, mut id: ValueId) -> Option<FunctionId> {
397        id = self.translate_value(id);
398        match self.context.builder[id] {
399            Value::Function(id) => Some(id),
400            // We don't set failed_to_inline_a_call for intrinsics since those
401            // don't correspond to actual functions in the SSA program that would
402            // need to be removed afterward.
403            Value::Intrinsic(_) => None,
404            _ => None,
405        }
406    }

408    /// Inline all reachable blocks within the source_function into the destination function.
409    fn inline_blocks(
410        &mut self,
411        ssa: &Ssa,
412        should_inline_call: &impl Fn(&Function) -> bool,
413    ) -> Result<Vec<ValueId>, RuntimeError> {
414        let mut block_queue = VisitOnceDeque::default();
415        block_queue.push_back(self.source_function.entry_block());
416
417        // This Vec will contain each block with a Return instruction along with the
418        // returned values of that block.
419        let mut function_returns = vec![];
420
421        while let Some(source_block_id) = block_queue.pop_front() {
422            let translated_block_id = self.translate_block(source_block_id, &mut block_queue);
423            self.context.builder.switch_to_block(translated_block_id);
424
425            self.inline_block_instructions(ssa, source_block_id, should_inline_call)?;
426
427            if let Some((block, values)) =
428                self.handle_terminator_instruction(source_block_id, &mut block_queue)
429            {
430                function_returns.push((block, values));
431            }
432        }
433
434        Ok(self.handle_function_returns(function_returns))
435    }

437    /// Handle inlining a function's possibly multiple return instructions.
438    /// If there is only 1 return we can just continue inserting into that block.
439    /// If there are multiple, we'll need to create a join block to jump to with each value.
440    fn handle_function_returns(
441        &mut self,
442        mut returns: Vec<(BasicBlockId, Vec<ValueId>)>,
443    ) -> Vec<ValueId> {
444        match returns.len() {
445            0 => Vec::new(),
446            1 => {
447                let (return_block, return_values) = returns.remove(0);
448                self.context.builder.switch_to_block(return_block);
449                return_values
450            }
451            _ => {
452                panic!(
453                    "ICE: found a function with multiple return terminators, but that should not happen"
454                )
455            }
456        }
457    }

459    /// Inline each instruction in the given block into the function being inlined into.
460    /// This may recurse if it finds another function to inline if a call instruction is within this block.
461    fn inline_block_instructions(
462        &mut self,
463        ssa: &Ssa,
464        block_id: BasicBlockId,
465        should_inline_call: &impl Fn(&Function) -> bool,
466    ) -> Result<(), RuntimeError> {
467        let mut side_effects_enabled: Option<ValueId> = None;
468
469        let block = &self.source_function.dfg[block_id];
470        for id in block.instructions() {
471            match &self.source_function.dfg[*id] {
472                Instruction::Call { func, arguments } => match self.get_function(*func) {
473                    Some(func_id) => {
474                        let call_stack = self.source_function.dfg.get_instruction_call_stack(*id);
475                        let callee = &ssa.functions[&func_id];
476
477                        // Sanity check to validate runtime compatibility
478                        self.validate_callee(callee, call_stack)?;
479
480                        // Do not inline self-recursive functions on the top level.
481                        // Inlining a self-recursive function works when there is something to inline into
482                        // by importing all the recursive blocks, but for the entry function there is no wrapper.
483                        // We must do this check here as inlining can be can triggered on a non-inline target (e.g., non-entry point).
484                        let inlining_self_recursion_at_top_level =
485                            self.entry_function.id() == func_id;
486                        if !inlining_self_recursion_at_top_level && should_inline_call(callee) {
487                            self.inline_function(ssa, *id, func_id, arguments, should_inline_call)?;
488
489                            // This is only relevant during handling functions with `InlineType::NoPredicates` as these
490                            // can pollute the function they're being inlined into with `Instruction::EnabledSideEffects`,
491                            // resulting in predicates not being applied properly.
492                            //
493                            // Note that this doesn't cover the case in which there exists an `Instruction::EnableSideEffectsIf`
494                            // within the function being inlined whilst the source function has not encountered one yet.
495                            // In practice this isn't an issue as the last `Instruction::EnableSideEffectsIf` in the
496                            // function being inlined will be to turn off predicates rather than to create one.
497                            if let Some(condition) = side_effects_enabled {
498                                self.context.builder.insert_enable_side_effects_if(condition);
499                            }
500                        } else {
501                            self.push_instruction(*id);
502                        }
503                    }
504                    None => self.push_instruction(*id),
505                },
506                Instruction::EnableSideEffectsIf { condition } => {
507                    side_effects_enabled = Some(self.translate_value(*condition));
508                    self.push_instruction(*id);
509                }
510                _ => self.push_instruction(*id),
511            }
512        }
513        Ok(())
514    }

516    /// Extra error check where given a caller's runtime its callee runtime is valid.
517    /// We determine validity as the following (where we have caller -> callee).
518    /// Valid:
519    /// - ACIR -> ACIR
520    /// - ACIR -> Brillig
521    /// - Brillig -> Brillig
522    ///
523    /// Invalid:
524    /// - Brillig -> ACIR
525    ///
526    /// Whether a valid callee should be inlined is determined separately by the inline info computation.
527    fn validate_callee(
528        &self,
529        callee: &Function,
530        call_stack: Vec<Location>,
531    ) -> Result<(), RuntimeError> {
532        if self.entry_function.runtime().is_brillig() && callee.runtime().is_acir() {
533            // If the caller is Brillig and the called function is ACIR,
534            // it cannot be inlined because runtimes do not share the same semantics
535            return Err(RuntimeError::UnconstrainedCallingConstrained {
536                call_stack,
537                constrained: callee.name().to_string(),
538                unconstrained: self.entry_function.name().to_string(),
539            });
540        }
541
542        Ok(())
543    }

545    /// Inline a function call and remember the inlined return values in the values map
546    fn inline_function(
547        &mut self,
548        ssa: &Ssa,
549        call_id: InstructionId,
550        function: FunctionId,
551        arguments: &[ValueId],
552        should_inline_call: &impl Fn(&Function) -> bool,
553    ) -> Result<(), RuntimeError> {
554        let old_results = self.source_function.dfg.instruction_results(call_id);
555        let arguments = vecmap(arguments, |arg| self.translate_value(*arg));
556
557        let call_stack = self.source_function.dfg.get_instruction_call_stack(call_id);
558        let call_stack_len = call_stack.len();
559        let new_call_stack = self
560            .context
561            .builder
562            .current_function
563            .dfg
564            .call_stack_data
565            .extend_call_stack(self.context.call_stack, &call_stack);
566
567        self.context.call_stack = new_call_stack;
568        let new_results =
569            self.context.inline_function(ssa, function, &arguments, should_inline_call)?;
570        self.context.call_stack = self
571            .context
572            .builder
573            .current_function
574            .dfg
575            .call_stack_data
576            .unwind_call_stack(self.context.call_stack, call_stack_len);
577
578        // If the inlined function has no reachable return (e.g., contains `while true {}`),
579        // new_results will be empty even though old_results expects values. The function
580        // diverges, so any code after this call is unreachable. Create a fresh block with
581        // parameters of the correct types as dummy values.
582        if new_results.is_empty() && !old_results.is_empty() {
583            let unreachable_block = self.context.builder.insert_block();
584            for old_result in old_results {
585                let typ = self.source_function.dfg.type_of_value(*old_result);
586                let param = self.context.builder.add_block_parameter(unreachable_block, typ);
587                self.values.insert(*old_result, param);
588            }
589            self.context.builder.switch_to_block(unreachable_block);
590            return Ok(());
591        }
592
593        let new_results = InsertInstructionResult::Results(call_id, &new_results);
594        Self::insert_new_instruction_results(&mut self.values, old_results, new_results);
595        Ok(())
596    }

598    /// Push the given instruction from the source_function into the current block of the
599    /// function being inlined into.
600    fn push_instruction(&mut self, id: InstructionId) {
601        let instruction = self.source_function.dfg[id].map_values(|id| self.translate_value(id));
602
603        let mut call_stack = self.context.call_stack;
604        let source_call_stack = self.source_function.dfg.get_instruction_call_stack(id);
605        call_stack = self
606            .context
607            .builder
608            .current_function
609            .dfg
610            .call_stack_data
611            .extend_call_stack(call_stack, &source_call_stack);
612        let results = self.source_function.dfg.instruction_results(id).to_vec();
613
614        let ctrl_typevars = instruction
615            .requires_ctrl_typevars()
616            .then(|| vecmap(&results, |result| self.source_function.dfg.type_of_value(*result)));
617
618        self.context.builder.set_call_stack(call_stack);
619
620        let new_results = self.context.builder.insert_instruction(instruction, ctrl_typevars);
621        Self::insert_new_instruction_results(&mut self.values, &results, new_results);
622    }

624    /// Modify the values HashMap to remember the mapping between an instruction result's previous
625    /// ValueId (from the source_function) and its new ValueId in the destination function.
626    fn insert_new_instruction_results(
627        values: &mut HashMap<ValueId, ValueId>,
628        old_results: &[ValueId],
629        new_results: InsertInstructionResult,
630    ) {
631        assert_eq!(old_results.len(), new_results.len());
632        for i in 0..old_results.len() {
633            values.insert(old_results[i], new_results[i]);
634        }
635    }

637    /// Handle the given terminator instruction from the given source function block.
638    /// This will push any new blocks to the destination function as needed, add them
639    /// to the block queue, and set the terminator instruction for the current block.
640    ///
641    /// If the terminator instruction was a Return, this will return the block this instruction
642    /// was in as well as the values that were returned.
643    fn handle_terminator_instruction(
644        &mut self,
645        block_id: BasicBlockId,
646        block_queue: &mut VisitOnceDeque,
647    ) -> Option<(BasicBlockId, Vec<ValueId>)> {
648        match &self.source_function.dfg[block_id].unwrap_terminator() {
649            TerminatorInstruction::Jmp { destination, arguments, call_stack } => {
650                let destination = self.translate_block(*destination, block_queue);
651                let arguments = vecmap(arguments, |arg| self.translate_value(*arg));
652                self.extend_call_stack(*call_stack);
653                self.context.builder.terminate_with_jmp(destination, arguments);
654                None
655            }
656            TerminatorInstruction::JmpIf {
657                condition,
658                then_destination,
659                else_destination,
660                call_stack,
661            } => {
662                let condition = self.translate_value(*condition);
663
664                // See if the value of the condition is known, and if so only inline the reachable
665                // branch. This lets us inline some recursive functions without recurring forever.
666                let dfg = &mut self.context.builder.current_function.dfg;
667                match dfg.get_numeric_constant(condition) {
668                    Some(constant) => {
669                        let next_block =
670                            if constant.is_zero() { *else_destination } else { *then_destination };
671
672                        let next_block = self.translate_block(next_block, block_queue);
673                        self.extend_call_stack(*call_stack);
674                        self.context.builder.terminate_with_jmp(next_block, vec![]);
675                    }
676                    None => {
677                        let then_block = self.translate_block(*then_destination, block_queue);
678                        let else_block = self.translate_block(*else_destination, block_queue);
679                        self.extend_call_stack(*call_stack);
680                        self.context
681                            .builder
682                            .terminate_with_jmpif(condition, then_block, else_block);
683                    }
684                }
685                None
686            }
687            TerminatorInstruction::Return { return_values, call_stack } => {
688                let return_values = vecmap(return_values, |value| self.translate_value(*value));
689
690                // Note that `translate_block` would take us back to the point at which the
691                // inlining of this source block began. Since additional blocks may have been
692                // inlined since, we are interested in the block representing the current program
693                // point, obtained via `current_block`.
694                let block_id = self.context.builder.current_block();
695
696                if self.inlining_entry {
697                    self.extend_call_stack(*call_stack);
698                    self.context.builder.terminate_with_return(return_values.clone());
699                }
700
701                Some((block_id, return_values))
702            }
703            TerminatorInstruction::Unreachable { .. } => {
704                // Note: `unreachable` terminators are only added during the `remove_unreachable_instructions`,
705                // which runs near the end of the optimization pipeline, so never before inlining.
706                // If we ever want to run it before inlining we'll have to handle this case.
707                panic!("Unreachable terminator instruction should not exist during inlining.")
708            }
709        }
710    }
711
712    fn extend_call_stack(&mut self, call_stack: CallStackId) {
713        let call_stack = self.source_function.dfg.get_call_stack(call_stack);
714        let call_stack_data = &mut self.context.builder.current_function.dfg.call_stack_data;
715        let new_call_stack =
716            call_stack_data.extend_call_stack(self.context.call_stack, &call_stack);
717        self.context.builder.set_call_stack(new_call_stack);
718    }
