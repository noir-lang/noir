1//! This is an algorithm for identifying branch starts and ends.
2//!
3//! The algorithm is split into two parts:
4//! 1. The outer part:
5//!    1. An (unrolled) CFG can be though of as a linear sequence of blocks where some nodes split
6//!       off, but eventually rejoin to a new node and continue the linear sequence.
7//!
8//!    2. Follow this sequence in order, and whenever a split is found call
9//!       `find_join_point_of_branches` and then recur from the join point it returns until the
10//!       return instruction is found.
11//!
12//! 2. The inner part defined by `find_join_point_of_branches`:
13//!
14//!    The recursive version of the algorithm is as follows:
15//!    1. For each of the two branches in a jmpif block:
16//!     - Check if either has multiple predecessors. If so, it is a join point.
17//!     - If not, continue to search the linear sequence of successor blocks from that block.
18//!       - If another split point is found, recur in `find_join_point_of_branches`
19//!       - If a block with multiple predecessors is found, return it.
20//!     - After, we should have identified a join point for both branches. This is expected to be
21//!       the same block for both and can be returned from here to continue iteration.
22//!
23//!    The recursive variant can encounter a stack overflow on certain CFG.
24//!
25//!    The iterative variant goes like this:
26//!    1. Pop the next branch from a stack of branches we need to process.
27//!     - Find the next join or branch point on both left and right branches.
28//!     - If the branches rejoin immediately:
29//!       - Repeatedly find the next point following the join point:
30//!         - If it's another join point, it must be for the parent level:
31//!           - If this is the first time we see a join point for this level, mark it as pending.
32//!           - If it's the second time, mark it as the end point, ensuring it matches the pending value.
33//!         - If a new branch is encountered, push it onto the stack, noting the parent level to return to.
34//!     - If either branches are followed by further branching:
35//!       - Push the branching children onto the stack for visiting later, noting to return to the current level.
36//!       - Mark any children that goes to in a join point as a pending end for this branch.
37//!
38//! This algorithm will remember each join point found in `find_join_point_of_branches` and
39//! the resulting map from each split block to each join block is returned.
40
41use std::collections::HashSet;
42
43use crate::ssa::ir::{basic_block::BasicBlockId, cfg::ControlFlowGraph, function::Function};
44use rustc_hash::FxHashMap as HashMap;
45
46/// Returns a `HashMap` mapping blocks that start a branch (i.e. blocks terminated with jmpif) to
47/// their corresponding blocks that end the branch.
48///
49/// This implementation assumes all branches re-converge. It would be little work to change it to
50/// support non-convergence (i.e. for multiple returns), with the caveat that there would be some
51/// inefficiency when processing such CFGs.
52pub(super) fn find_branch_ends(
53    function: &Function,
54    cfg: &ControlFlowGraph,
55) -> HashMap<BasicBlockId, BasicBlockId> {
56    let mut block = function.entry_block();
57    let mut context = Context::new(cfg);
58
59    loop {
60        let mut successors = cfg.successors(block);
61
62        if successors.len() == 2 {
63            block = context.find_join_point_of_branches(block, successors);
64        } else if successors.len() == 1 {
65            block = successors.next().unwrap();
66        } else if successors.len() == 0 {
67            // return encountered. We have nothing to join, so we're done
68            break;
69        } else {
70            unreachable!("A block can only have 0, 1, or 2 successors");
71        }
72    }
73
74    context.branch_ends
75}
76
77/// The next interesting point following some block.
78#[derive(Debug, Clone)]
79enum Point {
80    /// The CFG rejoined.
81    Join(BasicBlockId),
82    /// The CFG branched again at a parent block into a left and right child.
83    Branch(BasicBlockId, BasicBlockId, BasicBlockId),
84}
85
86struct Context<'cfg> {
87    branch_ends: HashMap<BasicBlockId, BasicBlockId>,
88    branch_ends_pending: HashMap<BasicBlockId, BasicBlockId>,
89    branch_parents: HashMap<BasicBlockId, BasicBlockId>,
90    stack: Vec<(BasicBlockId, BasicBlockId, BasicBlockId)>,
91    cfg: &'cfg ControlFlowGraph,
92}
93
94impl<'cfg> Context<'cfg> {
95    fn new(cfg: &'cfg ControlFlowGraph) -> Self {
96        Self {
97            cfg,
98            branch_ends: HashMap::default(),
99            branch_ends_pending: HashMap::default(),
100            branch_parents: HashMap::default(),
101            stack: Vec::new(),
102        }
103    }
104
105    fn find_join_point_of_branches(
106        &mut self,
107        start: BasicBlockId,
108        mut successors: impl Iterator<Item = BasicBlockId>,
109    ) -> BasicBlockId {
110        let left = successors.next().unwrap();
111        let right = successors.next().unwrap();
112        let mut visited = HashSet::new();
113
114        // Kick off the stack from the starting branch. It doesn't have a parent.
115        self.stack.push((start, left, right));
116
117        while let Some((branch, left, right)) = self.stack.pop() {
118            if !visited.insert(branch) {
119                continue;
120            }
121            let left = self.find_next_point(left, false);
122            let right = self.find_next_point(right, false);
123
124            if let Some(join) = self.maybe_join(branch, &left, &right) {
125                // If we managed to join the branches immediately, then we know where this branch ends,
126                // and we can check if we can complete any parent levels.
127                self.complete_parents(branch, join);
128            } else {
129                // At least one of the children further branches off.
130                for child in [left, right] {
131                    match child {
132                        Point::Join(next) => {
133                            self.branch_ends_pending.insert(branch, next);
134                        }
135                        Point::Branch(next, left, right) => {
136                            self.push_branch(branch, next, left, right);
137                        }
138                    }
139                }
140            }
141        }
142
143        self.branch_ends
144            .get(&start)
145            .cloned()
146            .unwrap_or_else(|| panic!("should have found the join point for {start}"))
147    }
148
149    /// Try to complete as many of the parent levels as we can, after finding the end point of a branch.
150    ///
151    /// Say we have this CFG:
152    /// ```text
153    ///                    b11
154    ///                   /   \
155    ///       b5        b9     b13
156    ///      /  \      /  \   /   \
157    ///    b1    b7--b8    b12     b15
158    ///    / \  /      \          /   \
159    ///  b0   b6        b10----b14    b4
160    ///    \                          /
161    ///    b2-----------------------b3
162    /// ```
163    /// At some point during the algorithm:
164    /// * We process `b8` we see that on the `b10` side it's followed by the join point `b15` and on the `b9` side it's another branch.
165    /// * We take note of `b15` as the pending ending for `b8`, and queue `b9` for processing.
166    /// * When we process `b9`, we get the join point `b13` on both `b11` and `b12`, so we "immediately" know that `b9` ends at `b13`.
167    /// * At that point we look for the next point after `b13`, which is the join point `b15`.
168    /// * The parent of `b9` is `b8`, which already has `b15` as a pending end, which checks out, so we can "complete" `b8`.
169    /// * Then we follow up the chain: the parent level of `b8` is `b0`; `b15` is followed by `b4`; `b0` already has `b4` as pending end, so we can "complete" that as well.
170    fn complete_parents(&mut self, mut branch: BasicBlockId, mut join: BasicBlockId) {
171        loop {
172            // If we reached the starting point, we can stop.
173            let Some(parent) = self.branch_parents.get(&branch).cloned() else {
174                break;
175            };
176            // We can skip this join point (we know it completes this level, not the parent), and look for the next one.
177            match self.find_next_point(join, true) {
178                Point::Join(next) => {
179                    // If it's a second join point, then we went back to the parent level. Try to complete it.
180                    if self.maybe_join_pending(parent, next) {
181                        branch = parent;
182                        join = next;
183                    } else {
184                        break;
185                    }
186                }
187                Point::Branch(next, left, right) => {
188                    // We found another branch on the same level as we are currently at.
189                    // We must visit it before we can return the the previous level.
190                    self.push_branch(parent, next, left, right);
191                    break;
192                }
193            }
194        }
195    }
196
197    /// Push a branch to the stack for exploration, remembering what the parent level branch was.
198    fn push_branch(
199        &mut self,
200        parent: BasicBlockId,
201        branch: BasicBlockId,
202        left: BasicBlockId,
203        right: BasicBlockId,
204    ) {
205        self.stack.push((branch, left, right));
206        // Remember where we need to go back to.
207        self.branch_parents.insert(branch, parent);
208    }
209
210    /// Check if the left and right branches joined.
211    /// If so, mark the branch as completed, and return `Some` join point, otherwise return `None`.
212    /// Panics if both points are joins, but have different values.
213    fn maybe_join(
214        &mut self,
215        branch: BasicBlockId,
216        left: &Point,
217        right: &Point,
218    ) -> Option<BasicBlockId> {
219        if let (Point::Join(left), Point::Join(right)) = (left, right) {
220            self.must_join(branch, *left, *right);
221            Some(*left)
222        } else {
223            None
224        }
225    }
226
227    /// Try to join a pending branch, once the next join point is found:
228    /// * if this is the first time we encounter this, mark it as pending, and return `false`,
229    /// * if this is the second time, then we mark it as completed, and return `true`.
230    ///
231    /// Panics if the join point does not match the existing one.
232    fn maybe_join_pending(&mut self, parent: BasicBlockId, join: BasicBlockId) -> bool {
233        let Some(pending) = self.branch_ends_pending.insert(parent, join) else {
234            return false;
235        };
236        self.must_join(parent, pending, join);
237        true
238    }
239
240    /// Check that the left and right join points are the same, and mark this as the join point for the start block.
241    fn must_join(&mut self, start: BasicBlockId, left: BasicBlockId, right: BasicBlockId) {
242        assert_eq!(left, right, "Expected two blocks to join to the same block");
243        self.branch_ends.insert(start, left);
244    }
245
246    /// Starting with the current block, find the next join or branching point,
247    /// skipping over blocks with single successors and predecessors.
248    ///
249    /// If a block is both a join and a branch block, it is returned as a join.
250    ///
251    /// When `skip` is true, we don't consider the current block for an immediate join point,
252    /// although it can be returned as a branch.
253    fn find_next_point(&self, mut block: BasicBlockId, mut skip: bool) -> Point {
254        loop {
255            if !skip {
256                let predecessors = self.cfg.predecessors(block);
257                if predecessors.len() > 1 {
258                    return Point::Join(block);
259                }
260            }
261            let mut successors = self.cfg.successors(block);
262            match successors.len() {
263                2 => {
264                    let left = successors.next().unwrap();
265                    let right = successors.next().unwrap();
266                    return Point::Branch(block, left, right);
267                }
268                1 => {
269                    skip = false;
270                    block = successors.next().unwrap();
271                }
272                0 => {
273                    unreachable!(
274                        "return encountered before a join point was found. This can only happen if early-return was added to the language without implementing it by jumping to a join block first"
275                    );
276                }
277                _ => {
278                    unreachable!("A block can only have 0, 1, or 2 successors");
279                }
280            }
281        }
282    }
283}

621            jmp b4()
622          b8():
623            return v4
624        }
625        "#;
626        let ssa = Ssa::from_str(src).unwrap();
627
628        // If we try to run the branch analysis now, it panics, it doesn't have the expected CFG structure.
629        // Instead, run the pipeline up to just before the flattening pass.
630        let ssa = run_pipeline_up_to_pass(ssa, "Flattening");
631
632        // The resulting SSA has more than 70k blocks.
633        // Both functions in the SSA have blocks such as b1->[b5, b6]->b7 where
634        // a branch immediately rejoins. For whatever combination, it's too much for
635        // a recursive algorithm, and can cause a stack overflow.
636
637        let function = ssa.main();
638        let cfg = ControlFlowGraph::with_function(function);
639        let _ = find_branch_ends(function, &cfg);
640    }
641
642    fn run_pipeline_up_to_pass(mut ssa: Ssa, stop_before_pass: &str) -> Ssa {
643        let options = SsaEvaluatorOptions {
644            ssa_logging: crate::ssa::SsaLogging::None,
645            brillig_options: BrilligOptions::default(),
646            print_codegen_timings: false,
647            emit_ssa: None,
648            skip_underconstrained_check: true,
649            skip_brillig_constraints_check: true,
650            enable_brillig_constraints_check_lookback: false,
651            inliner_aggressiveness: 0,
652            constant_folding_max_iter: constant_folding::DEFAULT_MAX_ITER,
653            small_function_max_instruction: inlining::MAX_INSTRUCTIONS,
654            max_bytecode_increase_percent: None,
655            force_unroll_threshold: FORCE_UNROLL_THRESHOLD,
656            skip_passes: Vec::new(),
657        };
658        let pipeline = primary_passes(&options);
659        for pass in pipeline {
660            if pass.msg() == stop_before_pass {
661                break;
662            }
663            ssa = pass
664                .run(ssa)
665                .unwrap_or_else(|e| panic!("failed to run pass '{}': {e}", pass.msg()));
666        }
667        ssa
668    }
