17/// The maximum number of instructions chosen below is an expert estimation of a "small" function
18/// in our SSA IR. Generally, inlining small functions with no control flow should enable further optimizations
19/// in the compiler while avoiding code size bloat.
20///
21/// For example, a common "simple" function is writing into a mutable reference.
22/// When that function has no control flow, it generally means we can expect all loads and stores within the
23/// function to be resolved upon inlining. Inlining this type of basic function both reduces the number of
24/// loads/stores to be executed and enables the compiler to continue optimizing at the inline site.
25pub(crate) const MAX_INSTRUCTIONS: usize = 10;
26
27/// The maximum Brillig weight for a function to be considered "simple" by the inlining cost model.
28/// This is used in `compute_function_should_be_inlined` where instruction weights are in Brillig
29/// cost units (not raw SSA instruction count). A value of 80 covers the worst case of 10 SSA
30/// instructions at maximum per-instruction cost (checked u32 mul = 8 opcodes each).
31/// The `inline_simple_functions` pass handles the common case using raw SSA count;
32/// this threshold acts as a safety net in the cost model.
33pub const MAX_SIMPLE_FUNCTION_WEIGHT: usize = 80;

35/// Information about a function to aid the decision about whether to inline it or not.
36/// The final decision depends on what we're inlining it into.
37#[derive(Default, Debug)]
38pub(crate) struct InlineInfo {
39    is_brillig_entry_point: bool,
40    is_acir_entry_point: bool,
41    is_recursive: bool,
42    pub(crate) should_inline: bool,
43    pub(super) contains_static_assertion: bool,
44    weight: i64,
45    cost: i64,
46}
47
48impl InlineInfo {
49    /// Functions which are to be retained, not inlined.
50    pub(crate) fn is_inline_target(&self, dfg: &DataFlowGraph) -> bool {
51        self.is_brillig_entry_point
52            || self.is_acir_entry_point
53            // We still want to attempt inlining recursive ACIR functions in case
54            // they have a compile-time completion point.
55            // A recursive function is going to set `should_inline` to false as well,
56            // so we need to determine whether a function is an inline target
57            // with auxiliary runtime information.
58            || ((self.is_recursive || !self.should_inline) && !dfg.runtime().is_acir())
59    }
60
61    pub(crate) fn should_inline(inline_infos: &InlineInfos, called_func_id: FunctionId) -> bool {
62        inline_infos.get(&called_func_id).is_some_and(|info| info.should_inline)
63    }
64}

66pub(crate) type InlineInfos = BTreeMap<FunctionId, InlineInfo>;

68/// Build a map from function ID to body weight for all functions that will be inlined.
69/// This is used by loop unrolling to estimate the true cost of call instructions
70/// to functions that will be inlined (instead of using call overhead as the cost).
71pub(crate) fn inlineable_callee_costs(
72    infos: &InlineInfos,
73) -> rustc_hash::FxHashMap<FunctionId, usize> {
74    infos
75        .iter()
76        .filter(|(_, info)| info.should_inline)
77        .map(|(id, info)| (*id, info.weight.max(0) as usize))
78        .collect()
79}

81/// The functions we should inline into (and that should be left in the final program) are:
82///  - main
83///  - Any Brillig function called from Acir
84///  - Some Brillig functions depending on aggressiveness and some metrics
85///  - Any Acir functions with a [fold inline type][noirc_frontend::monomorphization::ast::InlineType],
86///
87/// The returned `InlineInfos` won't have every function in it, only the ones which the algorithm visited.
88pub(crate) fn compute_inline_infos(
89    ssa: &Ssa,
90    call_graph: &CallGraph,
91    inline_no_predicates_functions: bool,
92    small_function_max_instructions: usize,
93    aggressiveness: i64,
94) -> InlineInfos {
95    let mut inline_infos = InlineInfos::default();
96
97    inline_infos.insert(
98        ssa.main_id,
99        InlineInfo {
100            is_acir_entry_point: ssa.main().runtime().is_acir(),
101            is_brillig_entry_point: ssa.main().runtime().is_brillig(),
102            ..Default::default()
103        },
104    );
105
106    // Handle ACIR functions.
107    for (func_id, function) in &ssa.functions {
108        if function.runtime().is_brillig() {
109            continue;
110        }
111
112        // If we have not already finished the flattening pass, functions marked
113        // to not have predicates should be preserved.
114        let preserve_function = !inline_no_predicates_functions && function.is_no_predicates();
115        if function.runtime().is_entry_point() || preserve_function {
116            inline_infos.entry(*func_id).or_default().is_acir_entry_point = true;
117        }
118
119        // Any Brillig function called from ACIR is an entry into the Brillig VM.
120        for called_func_id in called_functions(function) {
121            if ssa.functions[&called_func_id].runtime().is_brillig() {
122                inline_infos.entry(called_func_id).or_default().is_brillig_entry_point = true;
123            }
124        }
125    }
126
127    let times_called = call_graph.times_called();
128    // Find mutual recursion in our call graph
129    let recursive_functions = call_graph.get_recursive_functions();
130    let small_function_max_instructions = small_function_max_instructions as i64;
131    for recursive_func in &recursive_functions {
132        inline_infos.entry(*recursive_func).or_default().is_recursive = true;
133        compute_function_should_be_inlined(
134            ssa,
135            inline_no_predicates_functions,
136            small_function_max_instructions,
137            aggressiveness,
138            &times_called,
139            &mut inline_infos,
140            call_graph,
141            call_graph.ids_to_indices()[recursive_func],
142        );
143    }
144
145    let acyclic_graph = call_graph.build_acyclic_subgraph(&recursive_functions);
146    let topological_order = petgraph::algo::toposort(acyclic_graph.graph(), None).unwrap();
147
148    // We need to reverse the topological sort as we want to process the weight of the leaves first,
149    // as the weight of all callees will be used to compute a function's total weight.
150    for index in topological_order.into_iter().rev() {
151        compute_function_should_be_inlined(
152            ssa,
153            inline_no_predicates_functions,
154            small_function_max_instructions,
155            aggressiveness,
156            &times_called,
157            &mut inline_infos,
158            &acyclic_graph,
159            index,
160        );
161    }
162
163    inline_infos
164}

166/// Determines whether a function should be inlined.
167///
168/// Inlining is determined by the following:
169/// - the function is not recursive
170/// - the cost of inlining outweighs the cost of not doing so
171///
172/// The total weight of a function and its cost are computed in this method.
173/// The total weight is calculated by taking the function's own weight and adding
174/// the weight of each callee that will be inlined. We then determine the cost of
175/// inlining to be the times a function has been called multiplied by (total weight
176/// minus return cost), since a function's return terminator is eliminated when inlined.
177///
178/// To determine the cost of retaining a function we first need the [function interface cost][crate::ssa::ir::function::Function::call_overhead],
179/// The cost of retaining of a function is then (times a function has been called) * (interface cost) + total weight.
180///
181/// A function's net cost is then (cost of inlining - cost of retaining).
182/// The net cost is then compared against the inliner aggressiveness setting. If the net cost is less than the aggressiveness,
183/// we inline the function (granted there are not other restrictions such as recursion).
184#[allow(clippy::too_many_arguments)]
185fn compute_function_should_be_inlined(
186    ssa: &Ssa,
187    inline_no_predicates_functions: bool,
188    small_function_max_instructions: i64,
189    aggressiveness: i64,
190    times_called: &HashMap<FunctionId, usize>,
191    inline_infos: &mut InlineInfos,
192    call_graph: &CallGraph,
193    index: PetGraphIndex,
194) {
195    let func_id = call_graph.indices_to_ids()[&index];
196    if inline_infos.get(&func_id).is_some_and(|info| {
197        info.should_inline
198            || info.weight != 0
199            || info.is_brillig_entry_point
200            || info.is_acir_entry_point
201    }) {
202        return; // Already processed
203    }
204
205    let function = &ssa.functions[&func_id];
206    let runtime = function.runtime();
207
208    if runtime.is_acir() {
209        let info = inline_infos.entry(func_id).or_default();
210        info.should_inline = true;
211        return;
212    }
213
214    let is_recursive = inline_infos.get(&func_id).is_some_and(|info| info.is_recursive);
215    if runtime.is_brillig() && is_recursive {
216        return;
217    }
218
219    let assert_constant_id = function.dfg.get_intrinsic(Intrinsic::AssertConstant).copied();
220    let static_assert_id = function.dfg.get_intrinsic(Intrinsic::StaticAssert).copied();
221
222    let contains_static_assertion = if assert_constant_id.is_none() && static_assert_id.is_none() {
223        false
224    } else {
225        function.reachable_blocks().iter().any(|block| {
226            function.dfg[*block].instructions().iter().any(|instruction| {
227                match &function.dfg[*instruction] {
228                    Instruction::Call { func, .. } => {
229                        Some(*func) == assert_constant_id || Some(*func) == static_assert_id
230                    }
231                    _ => false,
232                }
233            })
234        })
235    };
236
237    let neighbors = call_graph.graph().neighbors(index);
238    let mut total_weight = function.cost() as i64;
239    let instruction_weight = total_weight;
240    for neighbor_index in neighbors {
241        let callee = call_graph.indices_to_ids()[&neighbor_index];
242        if inline_infos.get(&callee).is_some_and(|info| info.should_inline) {
243            total_weight = total_weight.saturating_add(inline_infos[&callee].weight);
244        }
245    }
246    let times = times_called[&func_id] as i64;
247    let interface_cost = function.call_overhead() as i64;
248    let return_cost = function.return_cost();
249    let inline_cost = times.saturating_mul(total_weight.saturating_sub(return_cost));
250    let retain_cost = times.saturating_mul(interface_cost) + total_weight;
251    let net_cost = inline_cost.saturating_sub(retain_cost);
252    let info = inline_infos.entry(func_id).or_default();
253
254    info.contains_static_assertion = contains_static_assertion;
255    info.weight = total_weight;
256    info.cost = net_cost;
257
258    let entry_block_id = function.entry_block();
259    let entry_block = &function.dfg[entry_block_id];
260    let should_inline_no_pred_function =
261        runtime.is_no_predicates() && inline_no_predicates_functions;
262    let is_simple_function = entry_block.successors().next().is_none()
263        && instruction_weight < small_function_max_instructions;
264
265    // Single-caller inlining has zero code duplication: the function body exists
266    // in exactly one place before and after inlining. Aggressiveness controls
267    // duplication tolerance, so it should not gate this decision.
268    let called_once = times == 1;
269
270    let should_inline = !runtime.is_inline_never()
271        && (is_simple_function
272            || net_cost < aggressiveness
273            || called_once
274            || runtime.is_inline_always()
275            || should_inline_no_pred_function
276            || contains_static_assertion
277            || runtime.is_acir());
278
279    info.should_inline = should_inline;
280}

282/// Compute something like a topological order of the functions, starting with the ones
283/// that do not call any other functions, going towards the entry points. When cycles
284/// are detected, take the one which are called by the most to break the ties.
285///
286/// This can be used to simplify the most often called functions first.
287///
288/// Returns the functions paired with their own as well as transitive weight,
289/// which accumulates the weight of all the functions they call, as well as own.
290pub(crate) fn compute_bottom_up_order(
291    ssa: &Ssa,
292    call_graph: &CallGraph,
293) -> Vec<(FunctionId, (usize, usize))> {
294    let mut order = Vec::new();
295    let mut visited = HashSet::default();
296
297    // Construct a new "call graph" which we'll repeatedly prune to find the "leaves".
298    let mut callees = call_graph.callees();
299    let callers = call_graph.callers();
300
301    // Number of times a function is called, used to break cycles in the call graph by popping the next candidate.
302    let mut times_called = call_graph.times_called().into_iter().collect::<Vec<_>>();
303    times_called.sort_by_key(|(id, cnt)| {
304        // Sort by called the *least* by others, as these are less likely to cut the graph when removed.
305        let called_desc = -(*cnt as i64);
306        // Sort entries first (last to be popped).
307        let is_entry_asc = -called_desc.signum();
308        // Finally break ties by ID.
309        (is_entry_asc, called_desc, *id)
310    });
311
312    // Start with the weight of the functions in isolation, then accumulate as we pop off the ones they call.
313    let own_weights =
314        ssa.functions.iter().map(|(id, f)| (*id, f.cost())).collect::<HashMap<_, _>>();
315    let mut weights = own_weights.clone();
316
317    // Seed the queue with functions that don't call anything.
318    let mut queue = callees
319        .iter()
320        .filter_map(|(id, callees)| callees.is_empty().then_some(*id))
321        .collect::<VecDeque<_>>();
322
323    loop {
324        while let Some(id) = queue.pop_front() {
325            // Pull the current weight of yet-to-be emitted callees (a nod to mutual recursion).
326            for (callee, cnt) in &callees[&id] {
327                if *callee != id {
328                    weights[&id] = weights[&id].saturating_add(cnt.saturating_mul(weights[callee]));
329                }
330            }
331            // Own weight plus the weights accumulated from callees.
332            let weight = weights[&id];
333            let own_weight = own_weights[&id];
334
335            // Emit the function.
336            order.push((id, (own_weight, weight)));
337            visited.insert(id);
338
339            // Update the callers of this function.
340            for (caller, cnt) in &callers[&id] {
341                // Update the weight of the caller with the weight of this function.
342                weights[caller] = weights[caller].saturating_add(cnt.saturating_mul(weight));
343                // Remove this function from the callees of the caller.
344                let callees = callees.get_mut(caller).unwrap();
345                callees.remove(&id);
346                // If the caller doesn't call any other function, enqueue it,
347                // unless it's the entry function, which is never called by anything, so it should be last.
348                if callees.is_empty() && !visited.contains(caller) && !callers[caller].is_empty() {
349                    queue.push_back(*caller);
350                }
351            }
352        }
353        // If we ran out of the queue, maybe there is a cycle; take the next most called function.
354        while let Some((id, _)) = times_called.pop() {
355            if !visited.contains(&id) {
356                queue.push_back(id);
357                break;
358            }
359        }
360        if times_called.is_empty() && queue.is_empty() {
361            assert_eq!(order.len(), callers.len());
362            return order;
363        }
364    }
365}

420    #[test]
421    fn bottom_up_order_and_weights() {
422        let src = "
423          brillig(inline) fn main f0 {
424            b0(v0: u32, v1: u1):
425              v3 = call f2(v0) -> u1
426              v4 = eq v3, v1
427              constrain v3 == v1
428              return
429          }
430          brillig(inline) fn is_even f1 {
431            b0(v0: u32):
432              v3 = eq v0, u32 0
433              jmpif v3 then: b2(), else: b1()
434            b1():
435              v5 = call f3(v0) -> u32
436              v7 = call f2(v5) -> u1
437              jmp b3(v7)
438            b2():
439              jmp b3(u1 1)
440            b3(v1: u1):
441              return v1
442          }
443          brillig(inline) fn is_odd f2 {
444            b0(v0: u32):
445              v3 = eq v0, u32 0
446              jmpif v3 then: b2(), else: b1()
447            b1():
448              v5 = call f3(v0) -> u32
449              v7 = call f1(v5) -> u1
450              jmp b3(v7)
451            b2():
452              jmp b3(u1 0)
453            b3(v1: u1):
454              return v1
455          }
456          brillig(inline) fn decrement f3 {
457            b0(v0: u32):
458              v2 = sub v0, u32 1
459              return v2
460          }
461        ";
462        // main
463        //   |
464        //   V
465        // is_odd <-> is_even
466        //      |     |
467        //      V     V
468        //      decrement
469
470        let ssa = Ssa::from_str(src).unwrap();
471        let call_graph = CallGraph::from_ssa_weighted(&ssa);
472        let order = compute_bottom_up_order(&ssa, &call_graph);
473
474        assert_eq!(order.len(), 4);
475        let (ids, ws): (Vec<_>, Vec<_>) = order.into_iter().map(|(id, w)| (id.to_u32(), w)).unzip();
476        let (ows, tws): (Vec<_>, Vec<_>) = ws.into_iter().unzip();
477
478        // Check order
479        assert_eq!(ids[0], 3, "decrement: first, it doesn't call anything");
480        assert_eq!(ids[1], 1, "is_even: called by is_odd; removing first avoids cutting the graph");
481        assert_eq!(ids[2], 2, "is_odd: called by is_odd and main");
482        assert_eq!(ids[3], 0, "main: last, it's the entry");
483
484        // Check own weights
485        // decrement: sub(3, checked u32) + return(2) = 5
486        // is_even/is_odd: b0(eq=1 + jmpif=2) + b1(call=7 + call=7 + jmp=2) + b2(jmp=2) + b3(return=2) = 23
487        // main: call(7) + eq(1) + constrain(4) + return(1) = 13
488        assert_eq!(ows, [5, 23, 23, 13]);
489
490        // Check transitive weights
491        assert_eq!(tws[0], ows[0], "decrement");
492        assert_eq!(
493            tws[1],
494            ows[1] + // own
495            tws[0] + // pushed from decrement
496            (ows[2] + tws[0]), // pulled from is_odd at the time is_even is emitted
497            "is_even"
498        );
499        assert_eq!(
500            tws[2],
501            ows[2] + // own
502            tws[0] + // pushed from decrement
503            tws[1], // pushed from is_even
504            "is_odd"
505        );
506        assert_eq!(
507            tws[3],
508            ows[3] + // own
509            tws[2], // pushed from is_odd
510            "main"
511        );
512        assert!(tws[3] > std::cmp::max(tws[1], tws[2]), "ideally 'main' has the most weight");
513    }
