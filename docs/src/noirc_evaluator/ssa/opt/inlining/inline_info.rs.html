17/// The maximum number of instructions chosen below is an expert estimation of a "small" function
18/// in our SSA IR. Generally, inlining small functions with no control flow should enable further optimizations
19/// in the compiler while avoiding code size bloat.
20///
21/// For example, a common "simple" function is writing into a mutable reference.
22/// When that function has no control flow, it generally means we can expect all loads and stores within the
23/// function to be resolved upon inlining. Inlining this type of basic function both reduces the number of
24/// loads/stores to be executed and enables the compiler to continue optimizing at the inline site.
25pub(crate) const MAX_INSTRUCTIONS: usize = 10;
26
27/// The maximum Brillig weight for a function to be considered "simple" by the inlining cost model.
28/// This is used in `compute_function_should_be_inlined` where instruction weights are in Brillig
29/// cost units (not raw SSA instruction count). A value of 80 covers the worst case of 10 SSA
30/// instructions at maximum per-instruction cost (checked u32 mul = 8 opcodes each).
31/// The `inline_simple_functions` pass handles the common case using raw SSA count;
32/// this threshold acts as a safety net in the cost model.
33pub const MAX_SIMPLE_FUNCTION_WEIGHT: usize = 80;

35/// Information about a function to aid the decision about whether to inline it or not.
36/// The final decision depends on what we're inlining it into.
37#[derive(Default, Debug)]
38pub(crate) struct InlineInfo {
39    is_brillig_entry_point: bool,
40    is_acir_entry_point: bool,
41    is_recursive: bool,
42    pub(crate) should_inline: bool,
43    pub(super) contains_static_assertion: bool,
44    weight: i64,
45    cost: i64,
46}
47
48impl InlineInfo {
49    /// Functions which are to be retained, not inlined.
50    pub(crate) fn is_inline_target(&self, dfg: &DataFlowGraph) -> bool {
51        self.is_brillig_entry_point
52            || self.is_acir_entry_point
53            // We still want to attempt inlining recursive ACIR functions in case
54            // they have a compile-time completion point.
55            // A recursive function is going to set `should_inline` to false as well,
56            // so we need to determine whether a function is an inline target
57            // with auxiliary runtime information.
58            || ((self.is_recursive || !self.should_inline) && !dfg.runtime().is_acir())
59    }
60
61    pub(crate) fn should_inline(inline_infos: &InlineInfos, called_func_id: FunctionId) -> bool {
62        inline_infos.get(&called_func_id).map(|info| info.should_inline).unwrap_or_default()
63    }
64}
65
66pub(crate) type InlineInfos = BTreeMap<FunctionId, InlineInfo>;

68/// The functions we should inline into (and that should be left in the final program) are:
69///  - main
70///  - Any Brillig function called from Acir
71///  - Some Brillig functions depending on aggressiveness and some metrics
72///  - Any Acir functions with a [fold inline type][noirc_frontend::monomorphization::ast::InlineType],
73///
74/// The returned `InlineInfos` won't have every function in it, only the ones which the algorithm visited.
75pub(crate) fn compute_inline_infos(
76    ssa: &Ssa,
77    call_graph: &CallGraph,
78    inline_no_predicates_functions: bool,
79    small_function_max_instructions: usize,
80    aggressiveness: i64,
81) -> InlineInfos {
82    let mut inline_infos = InlineInfos::default();
83
84    inline_infos.insert(
85        ssa.main_id,
86        InlineInfo {
87            is_acir_entry_point: ssa.main().runtime().is_acir(),
88            is_brillig_entry_point: ssa.main().runtime().is_brillig(),
89            ..Default::default()
90        },
91    );
92
93    // Handle ACIR functions.
94    for (func_id, function) in ssa.functions.iter() {
95        if function.runtime().is_brillig() {
96            continue;
97        }
98
99        // If we have not already finished the flattening pass, functions marked
100        // to not have predicates should be preserved.
101        let preserve_function = !inline_no_predicates_functions && function.is_no_predicates();
102        if function.runtime().is_entry_point() || preserve_function {
103            inline_infos.entry(*func_id).or_default().is_acir_entry_point = true;
104        }
105
106        // Any Brillig function called from ACIR is an entry into the Brillig VM.
107        for called_func_id in called_functions(function) {
108            if ssa.functions[&called_func_id].runtime().is_brillig() {
109                inline_infos.entry(called_func_id).or_default().is_brillig_entry_point = true;
110            }
111        }
112    }
113
114    let times_called = call_graph.times_called();
115    // Find mutual recursion in our call graph
116    let recursive_functions = call_graph.get_recursive_functions();
117    let small_function_max_instructions = small_function_max_instructions as i64;
118    for recursive_func in recursive_functions.iter() {
119        inline_infos.entry(*recursive_func).or_default().is_recursive = true;
120        compute_function_should_be_inlined(
121            ssa,
122            inline_no_predicates_functions,
123            small_function_max_instructions,
124            aggressiveness,
125            &times_called,
126            &mut inline_infos,
127            call_graph,
128            call_graph.ids_to_indices()[recursive_func],
129        );
130    }
131
132    let acyclic_graph = call_graph.build_acyclic_subgraph(&recursive_functions);
133    let topological_order = petgraph::algo::toposort(acyclic_graph.graph(), None).unwrap();
134
135    // We need to reverse the topological sort as we want to process the weight of the leaves first,
136    // as the weight of all callees will be used to compute a function's total weight.
137    for index in topological_order.into_iter().rev() {
138        compute_function_should_be_inlined(
139            ssa,
140            inline_no_predicates_functions,
141            small_function_max_instructions,
142            aggressiveness,
143            &times_called,
144            &mut inline_infos,
145            &acyclic_graph,
146            index,
147        );
148    }
149
150    inline_infos
151}

153/// Determines whether a function should be inlined.
154///
155/// Inlining is determined by the following:
156/// - the function is not recursive
157/// - the cost of inlining outweighs the cost of not doing so
158///
159/// The total weight of a function and its cost are computed in this method.
160/// The total weight is calculated by taking the function's own weight and adding
161/// the weight of each callee that will be inlined. We then determine the cost of
162/// inlining to be the times a function has been called multiplied by (total weight
163/// minus return cost), since a function's return terminator is eliminated when inlined.
164///
165/// To determine the cost of retaining a function we first need the [function interface cost][crate::ssa::ir::function::Function::call_overhead],
166/// The cost of retaining of a function is then (times a function has been called) * (interface cost) + total weight.
167///
168/// A function's net cost is then (cost of inlining - cost of retaining).
169/// The net cost is then compared against the inliner aggressiveness setting. If the net cost is less than the aggressiveness,
170/// we inline the function (granted there are not other restrictions such as recursion).
171#[allow(clippy::too_many_arguments)]
172fn compute_function_should_be_inlined(
173    ssa: &Ssa,
174    inline_no_predicates_functions: bool,
175    small_function_max_instructions: i64,
176    aggressiveness: i64,
177    times_called: &HashMap<FunctionId, usize>,
178    inline_infos: &mut InlineInfos,
179    call_graph: &CallGraph,
180    index: PetGraphIndex,
181) {
182    let func_id = call_graph.indices_to_ids()[&index];
183    if inline_infos.get(&func_id).is_some_and(|info| {
184        info.should_inline
185            || info.weight != 0
186            || info.is_brillig_entry_point
187            || info.is_acir_entry_point
188    }) {
189        return; // Already processed
190    }
191
192    let function = &ssa.functions[&func_id];
193    let runtime = function.runtime();
194
195    if runtime.is_acir() {
196        let info = inline_infos.entry(func_id).or_default();
197        info.should_inline = true;
198        return;
199    }
200
201    let is_recursive = inline_infos.get(&func_id).is_some_and(|info| info.is_recursive);
202    if runtime.is_brillig() && is_recursive {
203        return;
204    }
205
206    let assert_constant_id = function.dfg.get_intrinsic(Intrinsic::AssertConstant).copied();
207    let static_assert_id = function.dfg.get_intrinsic(Intrinsic::StaticAssert).copied();
208
209    let contains_static_assertion = if assert_constant_id.is_none() && static_assert_id.is_none() {
210        false
211    } else {
212        function.reachable_blocks().iter().any(|block| {
213            function.dfg[*block].instructions().iter().any(|instruction| {
214                match &function.dfg[*instruction] {
215                    Instruction::Call { func, .. } => {
216                        Some(*func) == assert_constant_id || Some(*func) == static_assert_id
217                    }
218                    _ => false,
219                }
220            })
221        })
222    };
223
224    let neighbors = call_graph.graph().neighbors(index);
225    let mut total_weight = function.cost() as i64;
226    let instruction_weight = total_weight;
227    for neighbor_index in neighbors {
228        let callee = call_graph.indices_to_ids()[&neighbor_index];
229        if inline_infos.get(&callee).is_some_and(|info| info.should_inline) {
230            total_weight = total_weight.saturating_add(inline_infos[&callee].weight);
231        }
232    }
233    let times = times_called[&func_id] as i64;
234    let interface_cost = function.call_overhead() as i64;
235    let return_cost = function.return_cost();
236    let inline_cost = times.saturating_mul(total_weight.saturating_sub(return_cost));
237    let retain_cost = times.saturating_mul(interface_cost) + total_weight;
238    let net_cost = inline_cost.saturating_sub(retain_cost);
239    let info = inline_infos.entry(func_id).or_default();
240
241    info.contains_static_assertion = contains_static_assertion;
242    info.weight = total_weight;
243    info.cost = net_cost;
244
245    let entry_block_id = function.entry_block();
246    let entry_block = &function.dfg[entry_block_id];
247    let should_inline_no_pred_function =
248        runtime.is_no_predicates() && inline_no_predicates_functions;
249    let is_simple_function = entry_block.successors().next().is_none()
250        && instruction_weight < small_function_max_instructions;
251
252    let should_inline = !runtime.is_inline_never()
253        && (is_simple_function
254            || net_cost < aggressiveness
255            || runtime.is_inline_always()
256            || should_inline_no_pred_function
257            || contains_static_assertion
258            || runtime.is_acir());
259
260    info.should_inline = should_inline;
261}

263/// Compute something like a topological order of the functions, starting with the ones
264/// that do not call any other functions, going towards the entry points. When cycles
265/// are detected, take the one which are called by the most to break the ties.
266///
267/// This can be used to simplify the most often called functions first.
268///
269/// Returns the functions paired with their own as well as transitive weight,
270/// which accumulates the weight of all the functions they call, as well as own.
271pub(crate) fn compute_bottom_up_order(
272    ssa: &Ssa,
273    call_graph: &CallGraph,
274) -> Vec<(FunctionId, (usize, usize))> {
275    let mut order = Vec::new();
276    let mut visited = HashSet::default();
277
278    // Construct a new "call graph" which we'll repeatedly prune to find the "leaves".
279    let mut callees = call_graph.callees();
280    let callers = call_graph.callers();
281
282    // Number of times a function is called, used to break cycles in the call graph by popping the next candidate.
283    let mut times_called = call_graph.times_called().into_iter().collect::<Vec<_>>();
284    times_called.sort_by_key(|(id, cnt)| {
285        // Sort by called the *least* by others, as these are less likely to cut the graph when removed.
286        let called_desc = -(*cnt as i64);
287        // Sort entries first (last to be popped).
288        let is_entry_asc = -called_desc.signum();
289        // Finally break ties by ID.
290        (is_entry_asc, called_desc, *id)
291    });
292
293    // Start with the weight of the functions in isolation, then accumulate as we pop off the ones they call.
294    let own_weights =
295        ssa.functions.iter().map(|(id, f)| (*id, f.cost())).collect::<HashMap<_, _>>();
296    let mut weights = own_weights.clone();
297
298    // Seed the queue with functions that don't call anything.
299    let mut queue = callees
300        .iter()
301        .filter_map(|(id, callees)| callees.is_empty().then_some(*id))
302        .collect::<VecDeque<_>>();
303
304    loop {
305        while let Some(id) = queue.pop_front() {
306            // Pull the current weight of yet-to-be emitted callees (a nod to mutual recursion).
307            for (callee, cnt) in &callees[&id] {
308                if *callee != id {
309                    weights[&id] = weights[&id].saturating_add(cnt.saturating_mul(weights[callee]));
310                }
311            }
312            // Own weight plus the weights accumulated from callees.
313            let weight = weights[&id];
314            let own_weight = own_weights[&id];
315
316            // Emit the function.
317            order.push((id, (own_weight, weight)));
318            visited.insert(id);
319
320            // Update the callers of this function.
321            for (caller, cnt) in &callers[&id] {
322                // Update the weight of the caller with the weight of this function.
323                weights[caller] = weights[caller].saturating_add(cnt.saturating_mul(weight));
324                // Remove this function from the callees of the caller.
325                let callees = callees.get_mut(caller).unwrap();
326                callees.remove(&id);
327                // If the caller doesn't call any other function, enqueue it,
328                // unless it's the entry function, which is never called by anything, so it should be last.
329                if callees.is_empty() && !visited.contains(caller) && !callers[caller].is_empty() {
330                    queue.push_back(*caller);
331                }
332            }
333        }
334        // If we ran out of the queue, maybe there is a cycle; take the next most called function.
335        while let Some((id, _)) = times_called.pop() {
336            if !visited.contains(&id) {
337                queue.push_back(id);
338                break;
339            }
340        }
341        if times_called.is_empty() && queue.is_empty() {
342            assert_eq!(order.len(), callers.len());
343            return order;
344        }
345    }
346}

401    #[test]
402    fn bottom_up_order_and_weights() {
403        let src = "
404          brillig(inline) fn main f0 {
405            b0(v0: u32, v1: u1):
406              v3 = call f2(v0) -> u1
407              v4 = eq v3, v1
408              constrain v3 == v1
409              return
410          }
411          brillig(inline) fn is_even f1 {
412            b0(v0: u32):
413              v3 = eq v0, u32 0
414              jmpif v3 then: b2, else: b1
415            b1():
416              v5 = call f3(v0) -> u32
417              v7 = call f2(v5) -> u1
418              jmp b3(v7)
419            b2():
420              jmp b3(u1 1)
421            b3(v1: u1):
422              return v1
423          }
424          brillig(inline) fn is_odd f2 {
425            b0(v0: u32):
426              v3 = eq v0, u32 0
427              jmpif v3 then: b2, else: b1
428            b1():
429              v5 = call f3(v0) -> u32
430              v7 = call f1(v5) -> u1
431              jmp b3(v7)
432            b2():
433              jmp b3(u1 0)
434            b3(v1: u1):
435              return v1
436          }
437          brillig(inline) fn decrement f3 {
438            b0(v0: u32):
439              v2 = sub v0, u32 1
440              return v2
441          }
442        ";
443        // main
444        //   |
445        //   V
446        // is_odd <-> is_even
447        //      |     |
448        //      V     V
449        //      decrement
450
451        let ssa = Ssa::from_str(src).unwrap();
452        let call_graph = CallGraph::from_ssa_weighted(&ssa);
453        let order = compute_bottom_up_order(&ssa, &call_graph);
454
455        assert_eq!(order.len(), 4);
456        let (ids, ws): (Vec<_>, Vec<_>) = order.into_iter().map(|(id, w)| (id.to_u32(), w)).unzip();
457        let (ows, tws): (Vec<_>, Vec<_>) = ws.into_iter().unzip();
458
459        // Check order
460        assert_eq!(ids[0], 3, "decrement: first, it doesn't call anything");
461        assert_eq!(ids[1], 1, "is_even: called by is_odd; removing first avoids cutting the graph");
462        assert_eq!(ids[2], 2, "is_odd: called by is_odd and main");
463        assert_eq!(ids[3], 0, "main: last, it's the entry");
464
465        // Check own weights
466        // decrement: sub(3, checked u32) + return(2) = 5
467        // is_even/is_odd: b0(eq=1 + jmpif=2) + b1(call=7 + call=7 + jmp=2) + b2(jmp=2) + b3(return=2) = 23
468        // main: call(7) + eq(1) + constrain(4) + return(1) = 13
469        assert_eq!(ows, [5, 23, 23, 13]);
470
471        // Check transitive weights
472        assert_eq!(tws[0], ows[0], "decrement");
473        assert_eq!(
474            tws[1],
475            ows[1] + // own
476            tws[0] + // pushed from decrement
477            (ows[2] + tws[0]), // pulled from is_odd at the time is_even is emitted
478            "is_even"
479        );
480        assert_eq!(
481            tws[2],
482            ows[2] + // own
483            tws[0] + // pushed from decrement
484            tws[1], // pushed from is_even
485            "is_odd"
486        );
487        assert_eq!(
488            tws[3],
489            ows[3] + // own
490            tws[2], // pushed from is_odd
491            "main"
492        );
493        assert!(tws[3] > std::cmp::max(tws[1], tws[2]), "ideally 'main' has the most weight");
494    }
