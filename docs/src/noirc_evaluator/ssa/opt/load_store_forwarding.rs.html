1//! Per-block load/store forwarding pass.
2//!
3//! This pass performs simple, fast, per-block optimizations:
4//! - **Load forwarding**: If a load reads from an address whose value is already known
5//!   (from a prior store in the same block), replace the load with the known value.
6//! - **Dead store elimination**: If two stores write to the same address with no
7//!   intervening load, the first store is dead and can be removed.
8//!
9//! This pass does not track values across block boundaries (that is handled by
10//! `mem2reg_simple` which promotes variables to block parameters). It is designed
11//! to be fast on large, single-block ACIR functions that result from inlining,
12//! unrolling, and flattening.
13//!
14//! ## Alias handling
15//!
16//! This pass does not consume a standalone alias analysis. Instead it uses two
17//! conservative heuristics to stay sound:
18//!
19//! 1. **Clear-on-unknown-store**: When a store targets an address that is neither
20//!    a block-local allocation nor already tracked in `known_values`, it may be an
21//!    alias (e.g. a reference extracted via `array_get`). All known values are
22//!    conservatively cleared. See [`forward_loads_and_stores_in_block`].
23//!
24//! 2. **Call invalidation**: When a reference is passed to a call, its known value
25//!    is invalidated. When a container that holds references (array/tuple) is passed,
26//!    all known values are cleared since the callee could extract and write through
27//!    any contained reference.
28//!
29//! 3. **Loop-carried alias detection**: Before forwarding, `analyze_loop_aliases`
30//!    scans all loops for stores that write a reference value into a reference
31//!    address (`store ref_value at ref_address`). These create cross-iteration
32//!    aliases: the stored reference can alias a "local" variable that is
33//!    re-initialized in a later iteration, bypassing heuristic (1). When a store
34//!    targets an address in the loop-alias set, all known values are conservatively
35//!    cleared. This allows forwarding in loop blocks that lack such patterns while
36//!    staying sound for those that have them.
37//!
38//! Once this pass consumes a standalone alias analysis (see [#12005]), the ad-hoc
39//! loop-alias heuristic can be replaced with precise alias queries.
40//!
41//! [#12005]: https://github.com/noir-lang/noir/issues/12005

57impl Ssa {
58    #[tracing::instrument(level = "trace", skip(self))]
59    pub(crate) fn load_store_forwarding(mut self) -> Ssa {
60        for function in self.functions.values_mut() {
61            function.load_store_forwarding();
62        }
63        self
64    }
65}

67impl Function {
68    pub(crate) fn load_store_forwarding(&mut self) {
69        let loop_aliases = analyze_loop_aliases(self);
70
71        let mut inserter = FunctionInserter::new(self);
72        let blocks = PostOrder::with_function(inserter.function).into_vec_reverse();
73
74        // Single pass in RPO: forward loads/stores and remap instructions.
75        // RPO guarantees predecessors are visited before successors (in acyclic
76        // graphs), so value mappings from forwarded loads are always available
77        // before blocks that use those values.
78        for block in &blocks {
79            let block = *block;
80            let instructions_to_remove =
81                forward_loads_and_stores_in_block(&mut inserter, block, &loop_aliases);
82
83            if !instructions_to_remove.is_empty() {
84                inserter.function.dfg[block]
85                    .instructions_mut()
86                    .retain(|id| !instructions_to_remove.contains(id));
87            }
88
89            // Re-insert instructions through the DFG simplify path. This resolves
90            // value mappings from load forwarding AND triggers simplification
91            // (e.g. `lt v2, u32 3` folds to a constant when v2 was forwarded).
92            // Instructions marked for removal (forwarded loads, dead stores) are skipped.
93            let instructions = inserter.function.dfg[block].take_instructions();
94            for instruction_id in &instructions {
95                if !instructions_to_remove.contains(instruction_id) {
96                    inserter.push_instruction(*instruction_id, block, true);
97                }
98            }
99            inserter.map_terminator_in_place(block);
100        }
101        inserter.map_data_bus_in_place();
102    }
103}

105/// Identify addresses involved in loop-carried alias patterns.
106///
107/// A loop-carried alias occurs in two forms:
108///
109/// 1. **Store of a reference inside a loop** (`store ref_value at ref_address`):
110///    The stored reference can alias a local variable that gets re-initialized
111///    in a later iteration, making the per-block "clear-on-unknown-store"
112///    heuristic insufficient.
113///
114/// 2. **Reference-typed block parameters on loop headers**: If `mem2reg_simple`
115///    has already promoted a `store ref at ref` to a block parameter, the
116///    aliasing is implicit. A reference-typed loop header parameter and the
117///    corresponding jmp arguments from within the loop body create the same
118///    cross-iteration aliasing.
119fn analyze_loop_aliases(function: &Function) -> HashSet<ValueId> {
120    use crate::ssa::ir::instruction::TerminatorInstruction;
121
122    let loops = Loops::find_all(function, LoopOrder::OutsideIn);
123    let mut aliases: HashSet<ValueId> = HashSet::default();
124    for loop_info in &loops.yet_to_unroll {
125        // Form 1: store of a reference inside a loop block.
126        for block_id in &loop_info.blocks {
127            let block = &function.dfg[*block_id];
128            for instruction_id in block.instructions() {
129                if let Instruction::Store { address, value } = &function.dfg[*instruction_id]
130                    && function.dfg.value_is_reference(*value)
131                {
132                    aliases.insert(*address);
133                    aliases.insert(*value);
134                }
135            }
136        }
137
138        // Form 2: reference-typed block parameters on the loop header.
139        // If mem2reg_simple promoted `store ref at ref` to a block parameter,
140        // the corresponding jmp arguments from within the loop carry the alias.
141        let header = loop_info.header;
142        let header_params = function.dfg[header].parameters();
143        let ref_param_indices: Vec<usize> = header_params
144            .iter()
145            .enumerate()
146            .filter(|(_, param)| function.dfg.value_is_reference(**param))
147            .map(|(idx, param)| {
148                aliases.insert(*param);
149                idx
150            })
151            .collect();
152
153        if !ref_param_indices.is_empty() {
154            for block_id in &loop_info.blocks {
155                let block = &function.dfg[*block_id];
156                match block.terminator() {
157                    Some(TerminatorInstruction::Jmp { destination, arguments, .. })
158                        if *destination == header =>
159                    {
160                        for &idx in &ref_param_indices {
161                            if let Some(arg) = arguments.get(idx) {
162                                aliases.insert(*arg);
163                            }
164                        }
165                    }
166                    Some(TerminatorInstruction::JmpIf {
167                        then_destination,
168                        then_arguments,
169                        else_destination,
170                        else_arguments,
171                        ..
172                    }) => {
173                        if *then_destination == header {
174                            for &idx in &ref_param_indices {
175                                if let Some(arg) = then_arguments.get(idx) {
176                                    aliases.insert(*arg);
177                                }
178                            }
179                        }
180                        if *else_destination == header {
181                            for &idx in &ref_param_indices {
182                                if let Some(arg) = else_arguments.get(idx) {
183                                    aliases.insert(*arg);
184                                }
185                            }
186                        }
187                    }
188                    _ => {}
189                }
190            }
191        }
192    }
193    aliases
194}

196/// Perform load/store forwarding within a single block.
197///
198/// Returns the set of instructions to remove from the block.
199///
200/// ## Alias safety: clear-on-unknown-store
201///
202/// Instructions like `MakeArray`, `ArraySet`, `ArrayGet`, `IncrementRc`, etc. move
203/// references around but don't modify pointed-to memory, so we keep known values
204/// through them. Soundness is maintained by the Store handler: when a store writes
205/// to an address that is neither a local allocation nor already in `known_values`,
206/// it could be an alias (e.g. extracted via `array_get`), so we conservatively
207/// clear all known values. Calls are handled separately since the callee could
208/// dereference and write through any reference argument.
209fn forward_loads_and_stores_in_block(
210    inserter: &mut FunctionInserter,
211    block: BasicBlockId,
212    loop_aliases: &HashSet<ValueId>,
213) -> HashSet<InstructionId> {
214    // Maps address -> last stored value (after resolving through the inserter)
215    let mut known_values: HashMap<ValueId, ValueId> = HashMap::default();
216    // Maps address -> last store instruction (candidate for dead store elimination)
217    let mut last_stores: HashMap<ValueId, InstructionId> = HashMap::default();
218    let mut instructions_to_remove: HashSet<InstructionId> = HashSet::default();
219    // Maps address -> last load result (for load-to-load forwarding).
220    // Kept separate from known_values so that load entries don't interfere
221    // with the store handler's clear-on-unknown-store alias heuristic.
222    let mut last_loads: HashMap<ValueId, ValueId> = HashMap::default();
223    // Track addresses from Allocate instructions in this block.
224    // These are definitionally fresh and cannot alias anything else.
225    let mut local_allocations: HashSet<ValueId> = HashSet::default();
226
227    let instructions = inserter.function.dfg[block].instructions().to_vec();
228
229    for instruction_id in instructions {
230        let instruction = &inserter.function.dfg[instruction_id];
231        match instruction {
232            Instruction::Allocate => {
233                let result = inserter.function.dfg.instruction_results(instruction_id)[0];
234                local_allocations.insert(result);
235            }
236            Instruction::Store { address, value } => {
237                let is_loop_aliased = loop_aliases.contains(address);
238                let address = inserter.resolve(*address);
239                let value = inserter.resolve(*value);
240
241                if is_loop_aliased {
242                    // This address participates in a loop-carried alias pattern:
243                    // a reference stored here in one iteration may alias a "local"
244                    // variable in a subsequent iteration. Conservatively clear all
245                    // known values to prevent stale forwarding.
246                    known_values.clear();
247                    last_loads.clear();
248                    last_stores.clear();
249                } else if !known_values.contains_key(&address)
250                    && !local_allocations.contains(&address)
251                {
252                    // This address wasn't allocated locally and wasn't seen in a prior
253                    // store — it could be an alias of an existing known reference
254                    // (e.g. extracted via array_get). Clear known values for addresses
255                    // of the same type, since different-typed references cannot alias.
256                    let store_type = inserter.function.dfg.type_of_value(address);
257                    let can_alias =
258                        |addr: &ValueId| inserter.function.dfg.type_of_value(*addr) == store_type;
259                    known_values.retain(|addr, _| !can_alias(addr));
260                    last_loads.retain(|addr, _| !can_alias(addr));
261                    last_stores.retain(|addr, _| !can_alias(addr));
262                } else if let Some(prev_store) = last_stores.get(&address) {
263                    // Previous store to this known/local address with no intervening
264                    // load is dead.
265                    instructions_to_remove.insert(*prev_store);
266                }
267
268                // A store supersedes any prior load from this address.
269                last_loads.remove(&address);
270                known_values.insert(address, value);
271                last_stores.insert(address, instruction_id);
272            }
273            Instruction::Load { address } => {
274                let address = inserter.resolve(*address);
275
276                if let Some(value) = known_values.get(&address) {
277                    // Store-to-load: we know the value from a prior store.
278                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
279                    inserter.map_value(result, *value);
280                    instructions_to_remove.insert(instruction_id);
281                } else if let Some(prev_result) = last_loads.get(&address) {
282                    // Load-to-load: no store to this address since the last load,
283                    // so we can reuse the previous load's result.
284                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
285                    inserter.map_value(result, *prev_result);
286                    instructions_to_remove.insert(instruction_id);
287                } else {
288                    // No known value — record for future load-to-load forwarding.
289                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
290                    last_loads.insert(address, result);
291                }
292
293                // This address was loaded from, so the last store to it is not dead.
294                last_stores.remove(&address);
295            }
296            Instruction::Call { .. } => {
297                // A call could dereference and modify any reference argument.
298                // For direct references we only invalidate that specific address.
299                // For containers (arrays/tuples) holding references, we can't know
300                // which addresses they hold, so we conservatively clear everything.
301                //
302                // Additionally, if a reference's inner type also contains a reference
303                // (e.g. `&mut &mut Field`), the callee can chase the indirection and
304                // write through the inner reference, so we must clear all known values.
305                //
306                // We also remove passed references from local_allocations: once an
307                // address escapes to a callee, it may be returned as an alias, so the
308                // "local and non-aliasing" invariant no longer holds.
309                instruction.for_each_value(|value| {
310                    let value = inserter.resolve(value);
311                    let typ = inserter.function.dfg.type_of_value(value);
312                    // Simple reference to a non-reference type (e.g. &mut Field):
313                    // only invalidate this specific address. Also remove from
314                    // local_allocations since the address now escapes to the callee
315                    // and may be returned as an alias.
316                    //
317                    // Everything else that contains a reference — double-references
318                    // (&mut &mut Field), containers ([&mut Field; N]), etc. — requires
319                    // clearing all state since the callee can chase indirections or
320                    // extract inner references and write through them.
321                    let is_simple_ref = matches!(typ.reference_element_type(), Some(inner) if !inner.contains_reference());
322                    if is_simple_ref {
323                        known_values.remove(&value);
324                        last_loads.remove(&value);
325                        last_stores.remove(&value);
326                        local_allocations.remove(&value);
327                    } else if typ.contains_reference() {
328                        known_values.clear();
329                        last_loads.clear();
330                        last_stores.clear();
331                        local_allocations.clear();
332                    }
333                });
334            }
335            _ => {
336                // MakeArray, ArraySet, ArrayGet, IfElse, IncrementRc, DecrementRc, etc.
337                // don't modify pointed-to memory — they just move references around.
338                // Safe to keep known values. The Store handler's clear-on-unknown-store
339                // ensures soundness if an alias is later written through.
340            }
341        }
342    }
343
344    // Any remaining entries in last_stores are stores with no subsequent load in
345    // this block. We do NOT remove them here because they may be needed by successor
346    // blocks or by later passes. Only truly dead stores (overwritten before being
347    // read) are removed above.
348
349    instructions_to_remove
350}

823    fn loop_carried_alias_prevents_incorrect_dead_store() {
824        // Minimized from `test_programs/execution_success/loop_carried_aliases`.
825        //
826        // v2 holds a reference-to-reference; `store v3 at v2` inside the loop
827        // creates a loop-carried alias: on the next iteration `load v2` returns
828        // v3, so `load (load v2)` reads from v3 through the alias.
829        //
830        // Without loop-alias analysis, `store Field 0xdeadbeef at v3` looks like
831        // a dead store (overwritten by `store v5 at v3`), and the load through
832        // the alias incorrectly forwards a stale value.
833        let src = "

835          b0(v0: &mut Field, v1: Field):
836            v2 = allocate -> &mut &mut Field
837            store v0 at v2
838            v3 = allocate -> &mut Field
839            store v1 at v3
840            jmp b1()
841          b1():
842            store Field 3735928559 at v3
843            v5 = load v2 -> &mut Field
844            v6 = load v5 -> Field
845            store v6 at v3
846            store v3 at v2
847            jmp b1()
848        }
849        ";
850        // Verify the pass does not miscompile: the store of 0xdeadbeef at v3
851        // must NOT be eliminated as a dead store, because `load v5` (where v5
852        // is loaded from v2, which aliases v3 after `store v3 at v2`) reads
853        // through the alias in the next iteration.

857    #[test]
858    fn remove_redundant_loads_from_ref_params() {
859        // Loads from reference parameters (not local allocations) should still be
860        // forwarded load-to-load when no intervening store invalidates them.
861        // After a store, subsequent loads should pick up the stored value.
862        let src = "
863        brillig(inline) impure fn push f0 {
864          b0(v0: &mut [Field; 4], v1: &mut u32, v2: Field):
865            v3 = load v0 -> [Field; 4]
866            v4 = load v1 -> u32
867            v5 = load v0 -> [Field; 4]
868            v6 = load v1 -> u32
869            v8 = lt v6, u32 4
870            constrain v8 == u1 1
871            v10 = array_set v3, index v6, value v2
872            v12 = unchecked_add v6, u32 1
873            store v10 at v0
874            store v4 at v1
875            v13 = load v0 -> [Field; 4]
876            v14 = add v4, u32 1
877            v15 = load v0 -> [Field; 4]
878            store v15 at v0
879            store v14 at v1
880            return
881        }
882        brillig(inline) impure fn next_counter f1 {
883          b0(v0: &mut [Field; 4], v1: &mut u32, v2: &mut Field):
884            v3 = load v0 -> [Field; 4]
885            v4 = load v1 -> u32
886            v5 = load v2 -> Field
887            v6 = load v0 -> [Field; 4]
888            v7 = load v1 -> u32
889            v8 = load v2 -> Field
890            v10 = add v8, Field 1
891            v11 = load v0 -> [Field; 4]
892            v12 = load v1 -> u32
893            v13 = load v2 -> Field
894            store v11 at v0
895            store v12 at v1
896            store v10 at v2
897            return v5
898        }
899        ";
900        let ssa = Ssa::from_str(src).unwrap();
901        let ssa = ssa.load_store_forwarding();
902
903        // In push: v5/v6 forward to v3/v4 (load-to-load before any stores).
904        // v13 forwards to v10 (store-to-load: v0 kept through store to v1 since types differ),
905        // v15 forwards to v10 (store-to-load), making store v4 at v1 a dead store.
906        // In next_counter: v6/v7/v8 forward to v3/v4/v5 (load-to-load),
907        // v11/v12/v13 forward to v3/v4/v5 (load-to-load, no intervening stores to v0/v1/v2
908        // between the first loads and these).
909        assert_ssa_snapshot!(ssa, @r"
910        brillig(inline) impure fn push f0 {
911          b0(v0: &mut [Field; 4], v1: &mut u32, v2: Field):
912            v3 = load v0 -> [Field; 4]
913            v4 = load v1 -> u32
914            v6 = lt v4, u32 4
915            constrain v6 == u1 1
916            v8 = array_set v3, index v4, value v2
917            v10 = unchecked_add v4, u32 1
918            store v8 at v0
919            v11 = add v4, u32 1
920            store v8 at v0
921            store v11 at v1
922            return
923        }
924        brillig(inline) impure fn next_counter f1 {
925          b0(v0: &mut [Field; 4], v1: &mut u32, v2: &mut Field):
926            v3 = load v0 -> [Field; 4]
927            v4 = load v1 -> u32
928            v5 = load v2 -> Field
929            v7 = add v5, Field 1
930            store v3 at v0
931            store v4 at v1
932            store v7 at v2
933            return v5
934        }
935        ");
936    }
