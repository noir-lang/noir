1//! Single-block load/store forwarding pass.
2//!
3//! This pass performs simple, fast optimizations on single-block functions:
4//! - **Load forwarding**: If a load reads from an address whose value is already known
5//!   (from a prior store), replace the load with the known value.
6//! - **Dead store elimination**: If two stores write to the same address with no
7//!   intervening load, the first store is dead and can be removed.
8//!
9//! This pass only runs on single-block functions. Multi-block functions are
10//! handled by `mem2reg` which promotes variables to block parameters.
11//! After inlining, unrolling, and flattening, ACIR functions are single-block.
12//! Brillig functions with multiple blocks are skipped.
13//!
14//! ## Alias handling
15//!
16//! Alias reasoning uses allocation identity via [`may_alias`]:
17//! - Two different `allocate` results never alias (each is a fresh, unique address).
18//! - Different reference types never alias (`&mut Field` vs `&mut u32`).
19//! - Any other pair of same-typed addresses may alias (conservative).
20//!
21//! This handles references extracted via `array_get` with dynamic indices:
22//! the result is a new ValueId that could alias any same-typed allocation.
23//! Constant-index `array_get` on reference arrays is simplified by the DFG
24//! simplifier during re-insertion, resolving the alias.
25//!
26//! Calls are handled conservatively: simple reference arguments invalidate
27//! that address and all its potential aliases; containers or nested
28//! references clear all state.
29use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};

36        function_inserter::FunctionInserter,
37        instruction::{Instruction, InstructionId},
38        post_order::PostOrder,

45impl Ssa {
46    #[tracing::instrument(level = "trace", skip(self))]
47    pub(crate) fn load_store_forwarding(mut self) -> Ssa {
48        for function in self.functions.values_mut() {
49            function.load_store_forwarding();
50        }
51        self
52    }
53}
54
55impl Function {
56    pub(crate) fn load_store_forwarding(&mut self) {
57        let mut inserter = FunctionInserter::new(self);
58        let blocks = PostOrder::with_function(inserter.function).into_vec_reverse();
59
60        // Only run on single-block functions. Multi-block functions rely on
61        // mem2reg for cross-block promotion; Brillig natively supports loads/stores.
62        if blocks.len() > 1 {
63            return;
64        }
65
66        let block = blocks[0];
67        let instructions_to_remove = forward_loads_and_stores_in_block(&mut inserter, block);
68
69        if !instructions_to_remove.is_empty() {
70            inserter.function.dfg[block]
71                .instructions_mut()
72                .retain(|id| !instructions_to_remove.contains(id));
73        }
74
75        // Re-insert instructions through the DFG simplify path. This resolves
76        // value mappings from load forwarding AND triggers simplification
77        // (e.g. `lt v2, u32 3` folds to a constant when v2 was forwarded).
78        let instructions = inserter.function.dfg[block].take_instructions();
79        for instruction_id in &instructions {
80            inserter.push_instruction(*instruction_id, block, true);
81        }
82        inserter.map_terminator_in_place(block);
83        inserter.map_data_bus_in_place();
84    }
85}

87/// Returns true if two types are compatible for aliasing purposes.
88///
89/// `&mut T` and `&T` with the same inner type must be treated as potentially aliasing.
90///
91/// The comparison is recursive so that nested references (e.g. `&&mut T` vs
92/// `&mut &T`) are also handled correctly.
93fn same_type_for_aliasing(a: &Type, b: &Type) -> bool {
94    match (a, b) {
95        (Type::Reference(inner_a, _), Type::Reference(inner_b, _)) => {
96            same_type_for_aliasing(inner_a, inner_b)
97        }
98        _ => a == b,
99    }
100}
101
102/// Returns true if two addresses might refer to the same memory.
103///
104/// Conservative: returns true when uncertain.
105/// - Same ValueId -> always alias.
106/// - Incompatible reference types -> never alias.
107/// - Both from different `allocate` instructions -> never alias.
108/// - Otherwise -> may alias.
109fn may_alias(a: ValueId, b: ValueId, allocations: &HashSet<ValueId>, dfg: &DataFlowGraph) -> bool {
110    if a == b {
111        return true;
112    }
113    if !same_type_for_aliasing(&dfg.type_of_value(a), &dfg.type_of_value(b)) {
114        return false;
115    }
116    if allocations.contains(&a) && allocations.contains(&b) {
117        return false;
118    }
119    true
120}

122/// Perform load/store forwarding within a single block.
123///
124/// Returns the set of instructions to remove from the block.
125fn forward_loads_and_stores_in_block(
126    inserter: &mut FunctionInserter,
127    block: BasicBlockId,
128) -> HashSet<InstructionId> {
129    let mut allocations: HashSet<ValueId> = HashSet::default();
130    let mut known_values: HashMap<ValueId, ValueId> = HashMap::default();
131    let mut last_stores: HashMap<ValueId, InstructionId> = HashMap::default();
132    let mut instructions_to_remove: HashSet<InstructionId> = HashSet::default();
133
134    let instructions = inserter.function.dfg[block].instructions().to_vec();
135
136    for instruction_id in instructions {
137        let instruction = &inserter.function.dfg[instruction_id];
138        match instruction {
139            Instruction::Allocate => {
140                let result = inserter.function.dfg.instruction_results(instruction_id)[0];
141                allocations.insert(result);
142            }
143            Instruction::Store { address, value } => {
144                let address = inserter.resolve(*address);
145                let value = inserter.resolve(*value);
146                let dfg = &inserter.function.dfg;
147
148                // Dead store elimination: exact address match only.
149                if let Some(prev_store) = last_stores.get(&address) {
150                    instructions_to_remove.insert(*prev_store);
151                }
152
153                // Clear aliased entries (Y != address where may_alias).
154                let aliases =
155                    |k: &ValueId| *k != address && may_alias(address, *k, &allocations, dfg);
156                known_values.retain(|k, _| !aliases(k));
157                last_stores.retain(|k, _| !aliases(k));
158
159                known_values.insert(address, value);
160                last_stores.insert(address, instruction_id);
161            }
162            Instruction::Load { address } => {
163                let address = inserter.resolve(*address);
164
165                let result = inserter.function.dfg.instruction_results(instruction_id)[0];
166                if let Some(value) = known_values.get(&address) {
167                    inserter.map_value(result, *value);
168                    instructions_to_remove.insert(instruction_id);
169                } else {
170                    known_values.insert(address, result);
171                }
172
173                // Mark aliased stores as used (not dead).
174                let dfg = &inserter.function.dfg;
175                last_stores.retain(|k, _| !may_alias(address, *k, &allocations, dfg));
176            }
177            Instruction::Call { .. } => {
178                // Simple reference (`&mut T` where T has no refs): invalidate that
179                // address and all its potential aliases.
180                // Container or nested reference: clear all state.
181                instruction.for_each_value(|value| {
182                    let value = inserter.resolve(value);
183                    let typ = inserter.function.dfg.type_of_value(value);
184                    let is_simple_ref = matches!(typ.reference_element_type(), Some(inner) if !inner.contains_reference());
185                    if is_simple_ref {
186                        let dfg = &inserter.function.dfg;
187                        known_values
188                            .retain(|k, _| !may_alias(value, *k, &allocations, dfg));
189                        last_stores
190                            .retain(|k, _| !may_alias(value, *k, &allocations, dfg));
191                    } else if typ.contains_reference() {
192                        known_values.clear();
193                        last_stores.clear();
194                    }
195                });
196            }
197            _ => {}
198        }
199    }
200
201    instructions_to_remove
202}
