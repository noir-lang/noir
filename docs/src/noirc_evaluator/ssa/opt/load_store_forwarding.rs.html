1//! Single-block load/store forwarding pass.
2//!
3//! This pass performs simple, fast optimizations on single-block functions:
4//! - **Load forwarding**: If a load reads from an address whose value is already known
5//!   (from a prior store), replace the load with the known value.
6//! - **Dead store elimination**: If two stores write to the same address with no
7//!   intervening load, the first store is dead and can be removed.
8//!
9//! This pass only runs on single-block functions. Multi-block functions are
10//! handled by `mem2reg` which promotes variables to block parameters.
11//! After inlining, unrolling, and flattening, ACIR functions are single-block.
12//! Brillig functions with multiple blocks are skipped.
13//!
14//! ## Alias handling
15//!
16//! Alias reasoning uses allocation identity via [`may_alias`]:
17//! - Two different `allocate` results never alias (each is a fresh, unique address).
18//! - Different reference types never alias (`&mut Field` vs `&mut u32`).
19//! - Any other pair of same-typed addresses may alias (conservative).
20//!
21//! This handles references extracted via `array_get` with dynamic indices:
22//! the result is a new ValueId that could alias any same-typed allocation.
23//! Constant-index `array_get` on reference arrays is simplified by the DFG
24//! simplifier during re-insertion, resolving the alias.
25//!
26//! Calls are handled conservatively: simple reference arguments invalidate
27//! that specific address; containers or nested references clear all state.
28use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};

43impl Ssa {
44    #[tracing::instrument(level = "trace", skip(self))]
45    pub(crate) fn load_store_forwarding(mut self) -> Ssa {
46        for function in self.functions.values_mut() {
47            function.load_store_forwarding();
48        }
49        self
50    }
51}

53impl Function {
54    pub(crate) fn load_store_forwarding(&mut self) {
55        let mut inserter = FunctionInserter::new(self);
56        let blocks = PostOrder::with_function(inserter.function).into_vec_reverse();
57
58        // Only run on single-block functions. Multi-block functions rely on
59        // mem2reg for cross-block promotion; Brillig natively supports loads/stores.
60        if blocks.len() > 1 {
61            return;
62        }
63
64        let allocations = collect_allocations(inserter.function, &blocks);
65
66        let block = blocks[0];
67        let instructions_to_remove =
68            forward_loads_and_stores_in_block(&mut inserter, block, &allocations);
69
70        if !instructions_to_remove.is_empty() {
71            inserter.function.dfg[block]
72                .instructions_mut()
73                .retain(|id| !instructions_to_remove.contains(id));
74        }
75
76        // Re-insert instructions through the DFG simplify path. This resolves
77        // value mappings from load forwarding AND triggers simplification
78        // (e.g. `lt v2, u32 3` folds to a constant when v2 was forwarded).
79        let instructions = inserter.function.dfg[block].take_instructions();
80        for instruction_id in &instructions {
81            if !instructions_to_remove.contains(instruction_id) {
82                inserter.push_instruction(*instruction_id, block, true);
83            }
84        }
85        inserter.map_terminator_in_place(block);
86        inserter.map_data_bus_in_place();
87    }
88}

90/// Collect all ValueIds produced by Allocate instructions.
91fn collect_allocations(function: &Function, blocks: &[BasicBlockId]) -> HashSet<ValueId> {
92    let mut allocations = HashSet::default();
93    for block in blocks {
94        for instruction_id in function.dfg[*block].instructions() {
95            if let Instruction::Allocate = &function.dfg[*instruction_id] {
96                let result = function.dfg.instruction_results(*instruction_id)[0];
97                allocations.insert(result);
98            }
99        }
100    }
101    allocations
102}

104/// Returns true if two addresses might refer to the same memory.
105///
106/// Conservative: returns true when uncertain.
107/// - Same ValueId -> always alias.
108/// - Different reference types -> never alias.
109/// - Both from different `allocate` instructions -> never alias.
110/// - Otherwise -> may alias.
111fn may_alias(a: ValueId, b: ValueId, allocations: &HashSet<ValueId>, dfg: &DataFlowGraph) -> bool {
112    if a == b {
113        return true;
114    }
115    if dfg.type_of_value(a) != dfg.type_of_value(b) {
116        return false;
117    }
118    if allocations.contains(&a) && allocations.contains(&b) {
119        return false;
120    }
121    true
122}

124/// Perform load/store forwarding within a single block.
125///
126/// Returns the set of instructions to remove from the block.
127fn forward_loads_and_stores_in_block(
128    inserter: &mut FunctionInserter,
129    block: BasicBlockId,
130    allocations: &HashSet<ValueId>,
131) -> HashSet<InstructionId> {
132    let mut known_values: HashMap<ValueId, ValueId> = HashMap::default();
133    let mut last_stores: HashMap<ValueId, InstructionId> = HashMap::default();
134    let mut last_loads: HashMap<ValueId, ValueId> = HashMap::default();
135    let mut instructions_to_remove: HashSet<InstructionId> = HashSet::default();
136
137    let instructions = inserter.function.dfg[block].instructions().to_vec();
138
139    for instruction_id in instructions {
140        let instruction = &inserter.function.dfg[instruction_id];
141        match instruction {
142            Instruction::Store { address, value } => {
143                let address = inserter.resolve(*address);
144                let value = inserter.resolve(*value);
145                let dfg = &inserter.function.dfg;
146
147                // Dead store elimination: exact address match only.
148                if let Some(prev_store) = last_stores.get(&address) {
149                    instructions_to_remove.insert(*prev_store);
150                }
151
152                // Clear aliased entries (Y != address where may_alias).
153                let aliases =
154                    |k: &ValueId| *k != address && may_alias(address, *k, allocations, dfg);
155                known_values.retain(|k, _| !aliases(k));
156                last_loads.retain(|k, _| !aliases(k));
157                last_stores.retain(|k, _| !aliases(k));
158
159                // A store supersedes any prior load from this address.
160                last_loads.remove(&address);
161                known_values.insert(address, value);
162                last_stores.insert(address, instruction_id);
163            }
164            Instruction::Load { address } => {
165                let address = inserter.resolve(*address);
166
167                if let Some(value) = known_values.get(&address) {
168                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
169                    inserter.map_value(result, *value);
170                    instructions_to_remove.insert(instruction_id);
171                } else if let Some(prev_result) = last_loads.get(&address) {
172                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
173                    inserter.map_value(result, *prev_result);
174                    instructions_to_remove.insert(instruction_id);
175                } else {
176                    let result = inserter.function.dfg.instruction_results(instruction_id)[0];
177                    last_loads.insert(address, result);
178                }
179
180                // Mark aliased stores as used (not dead).
181                let dfg = &inserter.function.dfg;
182                last_stores.retain(|k, _| !may_alias(address, *k, allocations, dfg));
183            }
184            Instruction::Call { .. } => {
185                // Simple reference (`&mut T` where T has no refs): invalidate that address.
186                // Container or nested reference: clear all state.
187                instruction.for_each_value(|value| {
188                    let value = inserter.resolve(value);
189                    let typ = inserter.function.dfg.type_of_value(value);
190                    let is_simple_ref = matches!(typ.reference_element_type(), Some(inner) if !inner.contains_reference());
191                    if is_simple_ref {
192                        known_values.remove(&value);
193                        last_loads.remove(&value);
194                        last_stores.remove(&value);
195                    } else if typ.contains_reference() {
196                        known_values.clear();
197                        last_loads.clear();
198                        last_stores.clear();
199                    }
200                });
201            }
202            _ => {}
203        }
204    }
205
206    instructions_to_remove
207}

679        // Loads from reference parameters (not local allocations) should still be
680        // forwarded load-to-load when no intervening store invalidates them.
681        // After a store, subsequent loads should pick up the stored value.
682        let src = "
683        brillig(inline) impure fn push f0 {
684          b0(v0: &mut [Field; 4], v1: &mut u32, v2: Field):
685            v3 = load v0 -> [Field; 4]
686            v4 = load v1 -> u32
687            v5 = load v0 -> [Field; 4]
688            v6 = load v1 -> u32
689            v8 = lt v6, u32 4
690            constrain v8 == u1 1
691            v10 = array_set v3, index v6, value v2
692            v12 = unchecked_add v6, u32 1
693            store v10 at v0
694            store v4 at v1
695            v13 = load v0 -> [Field; 4]
696            v14 = add v4, u32 1
697            v15 = load v0 -> [Field; 4]
698            store v15 at v0
699            store v14 at v1
700            return
701        }
702        brillig(inline) impure fn next_counter f1 {
703          b0(v0: &mut [Field; 4], v1: &mut u32, v2: &mut Field):
704            v3 = load v0 -> [Field; 4]
705            v4 = load v1 -> u32
706            v5 = load v2 -> Field
707            v6 = load v0 -> [Field; 4]
708            v7 = load v1 -> u32
709            v8 = load v2 -> Field
710            v10 = add v8, Field 1
711            v11 = load v0 -> [Field; 4]
712            v12 = load v1 -> u32
713            v13 = load v2 -> Field
714            store v11 at v0
715            store v12 at v1
716            store v10 at v2
717            return v5
718        }
719        ";
720        let ssa = Ssa::from_str(src).unwrap();
721        let ssa = ssa.load_store_forwarding();
722
723        // In push: v5/v6 forward to v3/v4 (load-to-load before any stores).
724        // v13 forwards to v10 (store-to-load: v0 kept through store to v1 since types differ),
725        // v15 forwards to v10 (store-to-load), making store v4 at v1 a dead store.
726        // In next_counter: v6/v7/v8 forward to v3/v4/v5 (load-to-load),
727        // v11/v12/v13 forward to v3/v4/v5 (load-to-load, no intervening stores to v0/v1/v2
728        // between the first loads and these).
729        assert_ssa_snapshot!(ssa, @r"
730        brillig(inline) impure fn push f0 {
731          b0(v0: &mut [Field; 4], v1: &mut u32, v2: Field):
732            v3 = load v0 -> [Field; 4]
733            v4 = load v1 -> u32
734            v6 = lt v4, u32 4
735            constrain v6 == u1 1
736            v8 = array_set v3, index v4, value v2
737            v10 = unchecked_add v4, u32 1
738            store v8 at v0
739            v11 = add v4, u32 1
740            store v8 at v0
741            store v11 at v1
742            return
743        }
744        brillig(inline) impure fn next_counter f1 {
745          b0(v0: &mut [Field; 4], v1: &mut u32, v2: &mut Field):
746            v3 = load v0 -> [Field; 4]
747            v4 = load v1 -> u32
748            v5 = load v2 -> Field
749            v7 = add v5, Field 1
750            store v3 at v0
751            store v4 at v1
752            store v7 at v2
753            return v5
754        }
755        ");
756    }
