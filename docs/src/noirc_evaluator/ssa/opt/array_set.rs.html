1//! The purpose of the `array_set_optimization` SSA pass is to mark `ArraySet` instructions
2//! as mutable _iff_ the array is not potentially shared with the callers or callees of the
3//! function and won't be used again in the function itself either. In other words, if this
4//! is the last time we use this version of the array, we can mutate it in place, and avoid
5//! having to make a copy of it.
6//!
7//! This optimization only applies to ACIR. In Brillig we use ref-counting to decide when
8//! there are no other references to an array.
9//! This pass assumes that Load and Store instructions have been previously removed.
10//!
11//! The pass is expected to run at most once, and requires these passes to occur before itself:
12//! * unrolling
13//! * flattening
14//! * removal of if-else instructions
15//! * mem2reg and DIE, in order to remove Load/Store instructions
16

32impl Ssa {
33    /// Finds the last instruction that writes to an array and modifies it
34    /// to do an in-place mutation instead of making a copy if there are
35    /// no potential shared references to it.
36    #[tracing::instrument(level = "trace", skip(self))]
37    pub(crate) fn array_set_optimization(mut self) -> Self {
38        for func in self.functions.values_mut() {
39            #[cfg(debug_assertions)]
40            array_set_optimization_pre_check(func);
41
42            func.array_set_optimization();
43
44            #[cfg(debug_assertions)]
45            array_set_optimization_post_check(func);
46        }
47        self
48    }
49}

51/// Pre-check condition for [Function::array_set_optimization].
52///
53/// Panics if:
54///   - An ACIR function contains more than 1 block, i.e. it hasn't been flattened yet.
55///   - There already exists a mutable array set instruction.
56///   - There is an `IfElse` instruction which hasn't been removed yet.
57#[cfg(debug_assertions)]
58fn array_set_optimization_pre_check(func: &Function) {
59    // We only want to run this pass for ACIR.
60    if func.runtime().is_brillig() {
61        return;
62    }
63
64    let reachable_blocks = func.reachable_blocks();
65
66    assert_eq!(
67        reachable_blocks.len(),
68        1,
69        "Expected there to be 1 block remaining in ACIR function for array_set optimization"
70    );
71
72    for block_id in reachable_blocks {
73        let instruction_ids = func.dfg[block_id].instructions();
74        for instruction_id in instruction_ids {
75            match func.dfg[*instruction_id] {
76                // There should be no mutable array sets.
77                Instruction::ArraySet { mutable: true, .. } => {
78                    panic!(
79                        "mutable ArraySet instruction exists before `array_set_optimization` pass"
80                    );
81                }
82                // The pass might mutate an array result of an `IfElse` and thus modify the input even if it's used later,
83                // so we assert that such instructions have already been removed by the `remove_if_else` pass.
84                Instruction::IfElse { .. } => {
85                    panic!("IfElse instruction exists before `array_set_optimization` pass");
86                }
87                Instruction::Load { .. } => {
88                    panic!("Load instruction exists before `array_set_optimization` pass");
89                }
90                Instruction::Store { .. } => {
91                    panic!("Store instruction exists before `array_set_optimization` pass");
92                }
93                _ => {}
94            }
95        }
96    }
97}

99/// Post-check condition for [Function::array_set_optimization].
100///
101/// Panics if:
102///   - Mutable array_set optimization has been applied to Brillig function.
103#[cfg(debug_assertions)]
104fn array_set_optimization_post_check(func: &Function) {
105    // Brillig functions should not have any mutable array sets.
106    if func.runtime().is_brillig() {
107        for block_id in func.reachable_blocks() {
108            let instruction_ids = func.dfg[block_id].instructions();
109            for instruction_id in instruction_ids {
110                if matches!(func.dfg[*instruction_id], Instruction::ArraySet { mutable: true, .. })
111                {
112                    panic!("Mutable array set instruction in Brillig function");
113                }
114            }
115        }
116    }
117}

119impl Function {
120    pub(crate) fn array_set_optimization(&mut self) {
121        if self.runtime().is_brillig() {
122            // Brillig is supposed to use ref-counting to decide whether to mutate an array;
123            // array mutation was only meant for ACIR. We could use it with Brillig as well,
124            // but then some of the optimizations that we can do in ACIR around shared
125            // references have to be skipped, which makes it more cumbersome.
126            return;
127        }
128
129        let mut context = Context::new(&self.dfg);
130        let entry_block = self.entry_block();
131        context.analyze_last_uses(entry_block);
132
133        let instructions_to_update = mem::take(&mut context.instructions_that_can_be_made_mutable);
134        make_mutable(&mut self.dfg, &instructions_to_update);
135    }
136}
137
138struct Context<'f> {
139    dfg: &'f DataFlowGraph,
140    array_to_last_use: HashMap<ValueId, InstructionId>,
141    instructions_that_can_be_made_mutable: HashSet<InstructionId>,
142}
143
144impl<'f> Context<'f> {
145    fn new(dfg: &'f DataFlowGraph) -> Self {
146        Context {
147            dfg,
148            array_to_last_use: HashMap::default(),
149            instructions_that_can_be_made_mutable: HashSet::default(),
150        }
151    }
152
153    /// Remember this instruction as the last time the array has been read or written to.
154    ///
155    /// Any previous instruction marked to be made mutable needs to be cancelled,
156    /// as it turned out not to be the last use.
157    fn set_last_use(&mut self, array: ValueId, instruction_id: InstructionId) {
158        if let Some(existing) = self.array_to_last_use.insert(array, instruction_id) {
159            self.instructions_that_can_be_made_mutable.remove(&existing);
160        }
161    }
162
163    /// Builds the set of `ArraySet` instructions that can be made mutable
164    /// because their input value is unused elsewhere afterward.
165    ///
166    /// Only expected to execute on ACIR functions.
167    fn analyze_last_uses(&mut self, block_id: BasicBlockId) {
168        assert!(self.dfg.runtime().is_acir());
169
170        let block = &self.dfg[block_id];
171
172        let terminator = self.dfg[block_id].unwrap_terminator();
173
174        // If we are in a return block we are not concerned about the array potentially being mutated again.
175        // In ACIR this should be the only kind of block we encounter, unless it's marked unreachable,
176        // in which case we don't need to optimize the array writes since we will end up with a failure anyway.
177        match terminator {
178            TerminatorInstruction::Return { .. } => {}
179            TerminatorInstruction::Unreachable { .. } => {
180                return;
181            }
182            other => {
183                panic!("unexpected terminator in ACIR: {other:?}")
184            }
185        };
186
187        for instruction_id in block.instructions() {
188            match &self.dfg[*instruction_id] {
189                // Reading an array constitutes as use, replacing any previous last use.
190                Instruction::ArrayGet { array, .. } => {
191                    self.set_last_use(*array, *instruction_id);
192                }
193                // Writing to an array is a use; mark it for mutation unless it might be shared.
194                Instruction::ArraySet { array, .. } => {
195                    self.set_last_use(*array, *instruction_id);
196
197                    // We also want to check that the array is not part of the terminator arguments, as this means it is used again.
198                    let mut is_array_in_terminator = false;
199                    let mut is_nested = false;
200                    terminator.for_each_value(|value| {
201                        let is_value_array_in_terminator = value == *array;
202                        if !is_value_array_in_terminator && self.dfg.type_of_value(value).is_array()
203                        {
204                            is_nested = true;
205                            self.set_last_use(value, *instruction_id);
206                        }
207                        is_array_in_terminator |= is_value_array_in_terminator;
208                    });
209
210                    let can_mutate = !is_array_in_terminator && !is_nested;
211
212                    if can_mutate {
213                        self.instructions_that_can_be_made_mutable.insert(*instruction_id);
214                    }
215                }
216                // Array arguments passed in calls constitute a use.
217                Instruction::Call { arguments, .. } => {
218                    for argument in arguments {
219                        if self.dfg.type_of_value(*argument).is_array() {
220                            self.set_last_use(*argument, *instruction_id);
221                        }
222                    }
223                }
224
225                // Arrays nested in other arrays are a use.
226                Instruction::MakeArray { elements, .. } => {
227                    for element in elements {
228                        if self.dfg.type_of_value(*element).is_array() {
229                            self.set_last_use(*element, *instruction_id);
230                        }
231                    }
232                }
233
234                // The pass might mutate an array result of an `IfElse` and thus modify the input even if it's used later,
235                // so we assert that such instructions have already been removed by the `remove_if_else` pass.
236                Instruction::IfElse { .. } => {
237                    unreachable!("IfElse instruction exists before `array_set_optimization` pass");
238                }
239
240                // Arrays loaded from memory might reference an existing array
241                // For instance if the array comes from a load we may potentially be mutating an array
242                // at a reference that is loaded from by other values.
243                Instruction::Load { .. } => {
244                    unreachable!("Load instruction exists before `array_set_optimization` pass");
245                }
246                // We also disallow Store instructions for the same reason.
247                Instruction::Store { .. } => {
248                    unreachable!("Store instruction exists before `array_set_optimization` pass");
249                }
250
251                // These instructions do not interact with arrays, so we do not need to track them.
252                Instruction::Binary(..)
253                | Instruction::Cast(..)
254                | Instruction::Not(..)
255                | Instruction::Truncate { .. }
256                | Instruction::Constrain(..)
257                | Instruction::ConstrainNotEqual(..)
258                | Instruction::RangeCheck { .. }
259                | Instruction::Allocate
260                | Instruction::EnableSideEffectsIf { .. }
261                | Instruction::IncrementRc { .. }
262                | Instruction::DecrementRc { .. }
263                | Instruction::Noop => (),
264            }
265        }
266    }
267}

269/// Make each ArraySet instruction in `instructions_to_update` mutable.
270fn make_mutable(dfg: &mut DataFlowGraph, instructions_to_update: &HashSet<InstructionId>) {
271    for instruction_id in instructions_to_update {
272        let instruction = &mut dfg[*instruction_id];
273        if let Instruction::ArraySet { mutable, .. } = instruction {
274            *mutable = true;
275        } else {
276            unreachable!("Non-ArraySet instruction in instructions_to_update!\n{instruction:?}");
277        }
278    }
279}

289    #[test]
290    fn does_not_mutate_array_used_in_make_array() {
291        // Regression test for https://github.com/noir-lang/noir/issues/8563
292        // Previously `v2` would be marked as mutable in the first array_set, which results in `v5` being invalid.
293        let src = "
294            acir(inline) fn main f0 {
295              b0():
296                v2 = make_array [Field 0] : [Field; 1]
297                v3 = array_set v2, index u32 0, value Field 2
298                v4 = make_array [v2, v2] : [[Field; 1]; 2]
299                v5 = array_set v4, index u32 0, value v2
300                return v5
301            }
302            ";
303        let ssa = Ssa::from_str(src).unwrap();
304
305        let ssa = ssa.array_set_optimization();
306        assert_ssa_snapshot!(ssa, @r"
307        acir(inline) fn main f0 {
308          b0():
309            v1 = make_array [Field 0] : [Field; 1]
310            v4 = array_set v1, index u32 0, value Field 2
311            v5 = make_array [v1, v1] : [[Field; 1]; 2]
312            v6 = array_set v5, index u32 0, value v1
