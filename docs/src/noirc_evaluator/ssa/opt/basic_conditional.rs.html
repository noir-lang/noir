1//! An SSA pass that operates on Brillig functions
2//! This optimization pass identifies simple conditional control flow patterns in unconstrained code
3//! and flattens them to reduce the number of basic blocks and improve performance.
4//!
5//! e.g: `if c {a} else {b}` would be flattened to `c*(a-b)+b`
6//! A simple conditional pattern is defined as a conditional sub-graph of the form `jmpif c: A, else B`, where A and B are basic blocks which join
7//! on the same successor. This exclude the graph from having any nested conditional or loop statements.
8//! Performance improvement is based on a simple execution cost metric
9//!
10//! This pass does not have any pre/post conditions.
11

35struct BasicConditional {
36    block_entry: BasicBlockId,
37    block_then: Option<BasicBlockId>,
38    block_else: Option<BasicBlockId>,
39    block_exit: BasicBlockId,
40}

42impl Ssa {
43    #[tracing::instrument(level = "trace", skip(self))]
44    /// Apply the basic_conditional pass to all functions of the program.
45    /// It first retrieve the `no_predicates` attribute of each function which will be used during the flattening.
46    pub(crate) fn flatten_basic_conditionals(mut self) -> Ssa {
47        // Retrieve the 'no_predicates' attribute of the functions in a map, to avoid problems with borrowing
48        let mut no_predicates = HashMap::default();
49        for function in self.functions.values() {
50            no_predicates.insert(function.id(), function.is_no_predicates());
51        }
52        for function in self.functions.values_mut() {
53            flatten_function(function, &no_predicates);
54        }
55        self
56    }
57}

59/// Returns the blocks of the simple conditional sub-graph of the CFG whose input block is the entry.
60/// Returns None if the input block is not the entry block of a simple conditional.
61/// A simple conditional is an if-then(-else) statement where branches are 'small' basic blocks.
62/// 'Small' basic blocks means that we expect their execution cost to be small.
63///
64/// In case the block is the entry of a 'simple conditional', the function returns a BasicConditional which
65/// consist of the list of the conditional blocks:
66///     block_entry
67///      /       \
68/// block_then   block_else
69///     \       /
70///     block_exit
71/// block_then and block_else are optionals, in order to account for the case when there is no 'then' or no 'else' branch
72/// Only structured CFG with this shape are considered:
73/// - block_entry has exactly 2 successors
74/// - block_then and block_else have exactly 1 successor, which is block_exit, or one of them is block_exit
75/// - block_exit has exactly 2 predecessors (block_then and block_else)
76///
77/// Furthermore, cost of block_then + cost of block_else must be less than their average cost + jump overhead cost
78fn is_conditional(
79    block: BasicBlockId,
80    cfg: &ControlFlowGraph,
81    function: &Function,
82) -> Option<BasicConditional> {
83    // A conditional must end with a JmpIf
84    let Some(TerminatorInstruction::JmpIf {
85        condition: _,
86        then_destination,
87        else_destination,
88        call_stack: _,
89    }) = function.dfg[block].terminator()
90    else {
91        return None;
92    };
93
94    // Cost of the JmpIf terminator (2 opcodes: jump_if + jump)
95    let jmpif_cost = function.dfg[block].unwrap_terminator().cost();
96
97    let mut then_successors = cfg.successors(*then_destination);
98    let mut else_successors = cfg.successors(*else_destination);
99    let then_successors_len = then_successors.len();
100    let else_successors_len = else_successors.len();
101    let next_then = then_successors.next();
102    let next_else = else_successors.next();
103
104    if next_then == Some(block) || next_else == Some(block) {
105        // this is a loop, not a conditional
106        return None;
107    }
108
109    let result = if then_successors_len == 1 && else_successors_len == 1 && next_then == next_else {
110        // The branches join on one block so it is a non-nested conditional with a classical diamond shape:
111        //    block
112        //    /    \
113        // then   else
114        //    \    /
115        //   next_then
116        // We check that the cost of the flattened code is lower than the cost of the branches
117        let cost_left = block_flatten_cost(*then_destination, &function.dfg)?;
118        let cost_right = block_flatten_cost(*else_destination, &function.dfg)?;
119        // Compute the actual branching overhead for this conditional:
120        // Flattening eliminates: JmpIf + then's Jmp + else's Jmp
121        // Flattening adds merge (IfElse) ops only for exit params where branches differ.
122        // SingleAddr merges (numeric/reference) cost 1 opcode (ConditionalMov), array merges cost ~20 (memory copy).
123        // Params where both branches pass the same value simplify to a no-op.
124        let then_term_cost = function.dfg[*then_destination].unwrap_terminator().cost();
125        let else_term_cost = function.dfg[*else_destination].unwrap_terminator().cost();
126        let exit_block = next_then.unwrap();
127        let merge_cost =
128            differing_merge_cost(*then_destination, *else_destination, exit_block, &function.dfg);
129        let jump_overhead =
130            (jmpif_cost + then_term_cost + else_term_cost).saturating_sub(merge_cost) as u32;
131        let cost = cost_right.saturating_add(cost_left);
132        if cost >= cost / 2 + jump_overhead {
133            return None;
134        }
135        BasicConditional {
136            block_entry: block,
137            block_then: Some(*then_destination),
138            block_else: Some(*else_destination),
139            block_exit: next_then.unwrap(),
140        }
141    } else if then_successors_len == 1 && next_then == Some(*else_destination) {
142        // Left branch joins the right branch, e.g if/then statement with no else:
143        //    block
144        //    /    \
145        // then     \
146        //     \    |
147        //      -> else
148        // This case may not happen (i.e not generated), but it is safer to handle it (e.g in case it happens due to some optimizations)
149        let cost = block_flatten_cost(*then_destination, &function.dfg)?;
150        // Flattening eliminates: JmpIf + then's Jmp; adds IfElse per exit param
151        let then_term_cost = function.dfg[*then_destination].unwrap_terminator().cost();
152        let merge_cost = function.dfg.block_parameters(*else_destination).len() * 3;
153        let jump_overhead = (jmpif_cost + then_term_cost).saturating_sub(merge_cost) as u32;
154        if cost >= cost / 2 + jump_overhead {
155            return None;
156        }
157        BasicConditional {
158            block_entry: block,
159            block_then: Some(*then_destination),
160            block_else: None,
161            block_exit: *else_destination,
162        }
163    } else if else_successors_len == 1 && next_else == Some(*then_destination) {
164        // Right branch joins the left branch, e.g if/else statement with no then
165        // This case may not happen (i.e not generated), but it is safer to handle it (e.g in case it happens due to some optimizations)
166        //    block
167        //    /    \
168        //   |     else
169        //   |      |
170        //    \    /
171        //     then
172        let cost = block_flatten_cost(*else_destination, &function.dfg)?;
173        // Flattening eliminates: JmpIf + else's Jmp; adds IfElse per exit param
174        let else_term_cost = function.dfg[*else_destination].unwrap_terminator().cost();
175        let merge_cost = function.dfg.block_parameters(*then_destination).len() * 3;
176        let jump_overhead = (jmpif_cost + else_term_cost).saturating_sub(merge_cost) as u32;
177        if cost >= cost / 2 + jump_overhead {
178            return None;
179        }
180        BasicConditional {
181            block_entry: block,
182            block_then: None,
183            block_else: Some(*else_destination),
184            block_exit: *else_destination,
185        }
186    } else {
187        return None;
188    };
189
190    // A conditional exit would have exactly 2 predecessors
191    (cfg.predecessors(result.block_exit).len() == 2).then_some(result)
192}

194/// Estimate the Brillig opcode cost of the IfElse merge instructions that flattening
195/// would generate. Only parameters where both branches pass different values need
196/// merge instructions; identical values simplify to a no-op.
197///
198/// For SingleAddr types (numerics and references), the merge is a single
199/// `ConditionalMov` = 1 opcode.
200/// For array/slice parameters, the IfElse generates a conditional memory copy
201/// which is much more expensive (~20 opcodes).
202fn differing_merge_cost(
203    then_block: BasicBlockId,
204    else_block: BasicBlockId,
205    exit_block: BasicBlockId,
206    dfg: &DataFlowGraph,
207) -> usize {
208    let then_args = match dfg[then_block].terminator() {
209        Some(TerminatorInstruction::Jmp { arguments, .. }) => arguments.as_slice(),
210        _ => return 0,
211    };
212    let else_args = match dfg[else_block].terminator() {
213        Some(TerminatorInstruction::Jmp { arguments, .. }) => arguments.as_slice(),
214        _ => return 0,
215    };
216    let exit_params = dfg.block_parameters(exit_block);
217    let mut cost = 0;
218    for ((a, b), param) in then_args.iter().zip(else_args.iter()).zip(exit_params.iter()) {
219        if a != b {
220            let typ = dfg.type_of_value(*param);
221            if typ.is_numeric() || matches!(typ, Type::Reference(_)) {
222                cost += 1; // SingleAddr: one ConditionalMov in Brillig
223            } else {
224                // Array/slice: conditional memory copy (~20 opcodes)
225                cost += 20;
226            }
227        }
228    }
229    cost
230}

232/// Computes a cost estimate for flattening a basic block in a conditional.
233///
234/// Returns `None` if the block contains instructions that cannot be safely
235/// flattened (side-effectful instructions like constraints, calls, memory ops,
236/// div/mod, shifts). Otherwise returns the estimated Brillig opcode cost.
237///
238/// Memory management instructions (IncrementRc, DecrementRc, Allocate) are
239/// excluded from the cost — they are safe to flatten but represent overhead
240/// that shouldn't influence the flatten/not-flatten decision.
241fn block_flatten_cost(block: BasicBlockId, dfg: &DataFlowGraph) -> Option<u32> {
242    let mut cost: u32 = 0;
243    for instruction_id in dfg[block].instructions() {
244        let instruction = &dfg[*instruction_id];
245        // Skip memory management instructions — these are always allowed to flatten
246        // but don't represent meaningful compute that should affect the decision.
247        if matches!(
248            instruction,
249            Instruction::Allocate
250                | Instruction::IncrementRc { .. }
251                | Instruction::DecrementRc { .. }
252                | Instruction::Noop
253        ) {
254            continue;
255        }
256
257        if !instruction.can_flatten_in_conditional(dfg) {
258            return None;
259        }
260
261        cost = cost.saturating_add(instruction.cost(*instruction_id, dfg) as u32);
262    }
263    Some(cost)
264}

266/// Identifies all simple conditionals in the function and flattens them
267fn flatten_function(function: &mut Function, no_predicates: &HashMap<FunctionId, bool>) {
268    // This pass is dedicated to brillig functions
269    if !function.runtime().is_brillig() {
270        return;
271    }
272    let cfg = ControlFlowGraph::with_function(function);
273    let mut stack = vec![function.entry_block()];
274    let mut processed = HashSet::new();
275    // List of all the simple conditionals that we will identify in the function
276    let mut conditionals = Vec::new();
277
278    // 1. Process all blocks of the cfg, starting from the root and following the successors
279    while let Some(block) = stack.pop() {
280        // Avoid cycles
281        if processed.contains(&block) {
282            continue;
283        }
284        processed.insert(block);
285
286        // Identify the simple conditionals
287        if let Some(conditional) = is_conditional(block, &cfg, function) {
288            // no need to check the branches, process the join block directly
289            stack.push(conditional.block_exit);
290            conditionals.push(conditional);
291        } else {
292            stack.extend(cfg.successors(block));
293        }
294    }
295
296    // 2. Flatten all simple conditionals
297    // process basic conditionals in reverse order so that
298    // a conditional does not impact the previous ones
299    conditionals.reverse();
300    flatten_multiple(&conditionals, function, no_predicates);
301}

303/// Flattens multiple basic conditionals within a function.
304///
305/// This function processes a collection of basic conditionals identified in the CFG and flattens them
306/// to reduce control flow complexity. Each conditional is processed with its own context, and the
307/// flattening results are then propagated throughout the entire function.
308///
309/// # Parameters
310/// * `conditionals` - The list of basic conditionals to flatten, assumed in reverse order
311/// * `function` - The function being optimized
312/// * `no_predicates` - Map of function IDs to their no_predicates attribute for handling function calls
313///
314/// # Process
315/// 1. Each conditional is flattened independently using a fresh context
316/// 2. Value mappings from all conditionals are collected into a unified mapping
317/// 3. The entire function is remapped only once in post-order to apply all value simplifications
318fn flatten_multiple(
319    conditionals: &Vec<BasicConditional>,
320    function: &mut Function,
321    no_predicates: &HashMap<FunctionId, bool>,
322) {
323    // 1. process each basic conditional, using a new context per conditional
324    let post_order = PostOrder::with_function(function);
325
326    let mut mapping = HashMap::default();
327    for conditional in conditionals {
328        let cfg = ControlFlowGraph::with_function(function);
329        let cfg_root = function.entry_block();
330        let mut branch_ends = HashMap::default();
331        branch_ends.insert(conditional.block_entry, conditional.block_exit);
332        let mut context = Context::new(function, cfg, branch_ends, cfg_root);
333        context.flatten_single_conditional(conditional, no_predicates);
334        // extract the mapping into 'mapping
335        context.inserter.extract_mapping(&mut mapping);
336    }
337    // 2. re-map the full program for values that may been simplified.
338    if !mapping.is_empty() {
339        for block in post_order.as_slice() {
340            Context::map_block_with_mapping(mapping.clone(), function, *block);
341        }
342    }
343}

345impl Context<'_> {
346    /// Flattens a single basic conditional by inlining the 2 branches into the entry block.
347    ///
348    /// This method transforms a conditional control flow pattern (if-then-else) into straight-line code
349    /// by merging the entry, then, else, and exit blocks. The conditional logic is converted into
350    /// predicated operations using cast and multiplication operations to select between branch values.
351    /// The method is adapted from `flatten_cfg`, tailored to do flatten only the input conditional.
352    ///
353    /// # Parameters
354    /// * `conditional` - The basic conditional structure to flatten
355    /// * `no_predicates` - Map of function IDs to their no_predicates attribute
356    ///
357    /// # Implementation Details
358    /// - Sets up context state (target_block, no_predicate) to enable proper inlining
359    /// - Inlines each block's instructions into the entry block
360    /// - Handles terminators to manage control flow during inlining
361    /// - Uses a WorkList to track which blocks need processing
362    /// - Copies the exit block's terminator to the entry block after inlining
363    /// - Restores original context state after completion
364    fn flatten_single_conditional(
365        &mut self,
366        conditional: &BasicConditional,
367        no_predicates: &HashMap<FunctionId, bool>,
368    ) {
369        // Manually inline 'then', 'else' and 'exit' into the entry block
370        //0. initialize the context for flattening a 'single conditional'
371        let old_target = self.target_block;
372        let old_no_predicate = self.no_predicate;
373        self.target_block = conditional.block_entry;
374        self.no_predicate = true;
375        //1. process 'then' branch
376        self.inline_block(conditional.block_entry, no_predicates);
377        let mut work_list = WorkList::new();
378        let to_process = self.handle_terminator(conditional.block_entry, &work_list);
379        work_list.extend(to_process);
380
381        if let Some(then) = conditional.block_then {
382            assert_eq!(work_list.pop(), conditional.block_then);
383            self.inline_block(then, no_predicates);
384            let to_process = self.handle_terminator(then, &work_list);
385            work_list.extend(to_process);
386        }
387
388        //2. process 'else' branch, in case there is no 'then'
389        let next = work_list.pop();
390        if next == conditional.block_else {
391            let next = next.unwrap();
392            self.inline_block(next, no_predicates);
393            let _ = self.handle_terminator(next, &work_list);
394        } else {
395            assert_eq!(next, Some(conditional.block_exit));
396        }
397
398        //3. process 'exit' block
399        self.inline_block(conditional.block_exit, no_predicates);
400        // Manually set the terminator of the entry block to the one of the exit block
401        let terminator =
402            self.inserter.function.dfg[conditional.block_exit].terminator().unwrap().clone();
403        let new_terminator = match terminator {
404            TerminatorInstruction::JmpIf {
405                condition,
406                then_destination,
407                else_destination,
408                call_stack,
409            } => {
410                let condition = self.inserter.resolve(condition);
411                TerminatorInstruction::JmpIf {
412                    condition,
413                    then_destination,
414                    else_destination,
415                    call_stack,
416                }
417            }
418            TerminatorInstruction::Jmp { destination, arguments, call_stack } => {
419                let arguments = vecmap(arguments, |value| self.inserter.resolve(value));
420                TerminatorInstruction::Jmp { destination, arguments, call_stack }
421            }
422            TerminatorInstruction::Return { return_values, call_stack } => {
423                let return_values = vecmap(return_values, |value| self.inserter.resolve(value));
424                TerminatorInstruction::Return { return_values, call_stack }
425            }
426            TerminatorInstruction::Unreachable { call_stack } => {
427                TerminatorInstruction::Unreachable { call_stack }
428            }
429        };
430        self.inserter.function.dfg.set_block_terminator(conditional.block_entry, new_terminator);
431        self.inserter.map_data_bus_in_place();
432        //4. restore the context, in case it is re-used.
433        self.target_block = old_target;
434        self.no_predicate = old_no_predicate;
435    }
436
437    /// Applies value mappings to all instructions and terminators in a block.
438    ///
439    /// This method rewrites a block by replacing old value IDs with their mapped equivalents
440    /// according to the provided mapping. This is used to propagate value simplifications
441    /// from conditional flattening throughout the rest of the function.
442    ///
443    /// # Parameters
444    /// * `mapping` - HashMap mapping old ValueIds to their simplified/replaced ValueIds
445    /// * `func` - The function containing the block to update
446    /// * `block` - The BasicBlockId of the block to remap
447    fn map_block_with_mapping(
448        mapping: HashMap<ValueId, ValueId>,
449        func: &mut Function,
450        block: BasicBlockId,
451    ) {
452        // Map all instructions in the block
453        let mut inserter = FunctionInserter::new(func);
454        inserter.set_mapping(mapping);
455        let instructions = inserter.function.dfg[block].instructions().to_vec();
456        for instruction in instructions {
457            inserter.map_instruction_in_place(instruction);
458        }
459        inserter.map_terminator_in_place(block);
460    }
461}
