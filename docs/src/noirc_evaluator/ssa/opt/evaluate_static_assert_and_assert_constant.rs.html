1//! A simple SSA pass to go through each instruction and evaluate:
2//! - each call to `assert_constant`, issuing an error if any arguments to the function
3//!   are not constants
4//! - each call to `static_assert`, issuing an error if the assertion does not hold,
5//!   the value to test is not a constant, or the message is dynamic.
6//!
7//! Note that this pass must be placed directly before [`loop unrolling`](super::unrolling) to be
8//! useful. Any optimization passes between this and loop unrolling will cause
9//! the constants that this pass sees to be potentially different than the constants
10//! seen by loop unrolling. Furthermore, this pass cannot be a part of loop unrolling
11//! since we must go through every instruction to find all references to `assert_constant`
12//! while loop unrolling only touches blocks with loops in them.
13use acvm::{FieldElement, acir::brillig::ForeignCallParam};
14use iter_extended::vecmap;
15use noirc_printable_type::{PrintableValueDisplay, TryFromParamsError};
16use rustc_hash::FxHashSet as HashSet;

34impl Ssa {
35    /// See [`evaluate_static_assert_and_assert_constant`][self] module for more information.
36    #[tracing::instrument(level = "trace", skip(self))]
37    pub(crate) fn evaluate_static_assert_and_assert_constant(self) -> Result<Ssa, RuntimeError> {
38        let error_on_failure = true;
39        self.evaluate_static_assert_and_assert_constant_helper(error_on_failure)
40    }
41
42    /// See [`evaluate_static_assert_and_assert_constant`][self] module for more information.
43    #[tracing::instrument(level = "trace", skip(self))]
44    pub(crate) fn try_evaluate_static_assert_and_assert_constant(
45        self,
46    ) -> Result<Ssa, RuntimeError> {
47        let error_on_failure = false;
48        self.evaluate_static_assert_and_assert_constant_helper(error_on_failure)
49    }
50
51    /// If `error_on_failure` is set, an error is thrown when a failing `assert_constant`
52    /// or `static_assert` is detected.
53    fn evaluate_static_assert_and_assert_constant_helper(
54        mut self,
55        error_on_failure: bool,
56    ) -> Result<Ssa, RuntimeError> {
57        for function in self.functions.values_mut() {
58            function.evaluate_static_assert_and_assert_constant(error_on_failure)?;
59        }
60        Ok(self)
61    }
62}

64impl Function {
65    fn evaluate_static_assert_and_assert_constant(
66        &mut self,
67        error_on_failure: bool,
68    ) -> Result<(), RuntimeError> {
69        let assert_constant_id = self.dfg.get_intrinsic(Intrinsic::AssertConstant).copied();
70        let static_assert_id = self.dfg.get_intrinsic(Intrinsic::StaticAssert).copied();
71        if assert_constant_id.is_none() && static_assert_id.is_none() {
72            // If there are no calls to either intrinsic there's nothing to evaluate
73            return Ok(());
74        }
75
76        let blocks_within_empty_loop = get_blocks_within_empty_loop(self);
77
78        for block in self.reachable_blocks() {
79            // Unfortunately we can't just use instructions.retain(...) here since
80            // check_instruction can also return an error
81            let instructions = self.dfg[block].take_instructions();
82            let mut filtered_instructions = Vec::with_capacity(instructions.len());
83
84            let inside_empty_loop = blocks_within_empty_loop.contains(&block);
85            for instruction in instructions {
86                if check_instruction(
87                    self,
88                    instruction,
89                    assert_constant_id,
90                    static_assert_id,
91                    inside_empty_loop,
92                    error_on_failure,
93                )? {
94                    filtered_instructions.push(instruction);
95                }
96            }
97
98            *self.dfg[block].instructions_mut() = filtered_instructions;
99        }
100        Ok(())
101    }
102}

104/// Returns all of a function's block that are part of empty loops.
105fn get_blocks_within_empty_loop(function: &Function) -> HashSet<BasicBlockId> {
106    let loops = Loops::find_all(function, LoopOrder::OutsideIn);
107
108    let cfg = ControlFlowGraph::with_function(function);
109    let mut blocks_within_empty_loop = HashSet::default();
110    for loop_ in loops.yet_to_unroll {
111        let Ok(pre_header) = loop_.get_pre_header(function, &cfg) else {
112            // If the loop does not have a preheader we skip checking whether the loop is empty
113            continue;
114        };
115        let const_bounds = loop_.get_const_bounds(&function.dfg, pre_header);
116
117        let does_execute = const_bounds
118            .and_then(|(lower_bound, upper_bound)| {
119                upper_bound.reduce(lower_bound, |u, l| u > l, |u, l| u > l)
120            })
121            // We default to `true` if the bounds are dynamic so that we still
122            // evaluate static assertion in dynamic loops.
123            .unwrap_or(true);
124
125        if !does_execute {
126            blocks_within_empty_loop.extend(loop_.blocks);
127        }
128    }
129
130    blocks_within_empty_loop
131}

133/// During the loop unrolling pass we also evaluate calls to `assert_constant`.
134/// This is done in this pass because loop unrolling is the only pass that will error
135/// if a value (the loop bounds) are not known constants.
136///
137/// This returns Ok(true) if the given instruction should be kept in the block and
138/// Ok(false) if it should be removed.
139///
140/// If `error_on_failure` is set, an error is thrown when a failing `assert_constant`
141/// or `static_assert` is detected.
142fn check_instruction(
143    function: &Function,
144    instruction: InstructionId,
145    assert_constant_id: Option<ValueId>,
146    static_assert_id: Option<ValueId>,
147    inside_empty_loop: bool,
148    error_on_failure: bool,
149) -> Result<bool, RuntimeError> {
150    match &function.dfg[instruction] {
151        Instruction::Call { func, arguments } => {
152            let is_assert_constant = Some(*func) == assert_constant_id;
153            let is_static_assert = Some(*func) == static_assert_id;
154
155            // Skip assertions inside known empty loops
156            if inside_empty_loop && (is_assert_constant || is_static_assert) {
157                return Ok(false);
158            }
159
160            if is_assert_constant {
161                evaluate_assert_constant(function, instruction, arguments, error_on_failure)
162            } else if is_static_assert {
163                evaluate_static_assert(function, instruction, arguments, error_on_failure)
164            } else {
165                Ok(true)
166            }
167        }
168        _ => Ok(true),
169    }
170}

172/// Evaluate a call to `assert_constant`, returning an error if any of the elements are not
173/// constants. If all of the elements are constants, Ok(false) is returned. This signifies a
174/// success but also that the instruction need not be reinserted into the block being unrolled
175/// since it has already been evaluated.
176///
177/// If `error_on_failure` is set, an error is thrown when a failing `static_assert`
178/// is detected.
179fn evaluate_assert_constant(
180    function: &Function,
181    instruction: InstructionId,
182    arguments: &[ValueId],
183    error_on_failure: bool,
184) -> Result<bool, RuntimeError> {
185    if arguments.iter().all(|arg| function.dfg.is_constant(*arg)) {
186        Ok(false)
187    } else if error_on_failure {
188        let call_stack = function.dfg.get_instruction_call_stack(instruction);
189        Err(RuntimeError::AssertConstantFailed { call_stack })
190    } else {
191        Ok(true)
192    }
193}

195/// Evaluate a call to `static_assert`, returning an error if the value is false
196/// or not constant (see assert_constant).
197///
198/// When it passes, Ok(false) is returned. This signifies a
199/// success but also that the instruction need not be reinserted into the block being unrolled
200/// since it has already been evaluated.
201///
202/// If `error_on_failure` is set, an error is thrown when a failing `assert_constant`
203/// is detected.
204fn evaluate_static_assert(
205    function: &Function,
206    instruction: InstructionId,
207    arguments: &[ValueId],
208    error_on_failure: bool,
209) -> Result<bool, RuntimeError> {
210    if arguments.len() < 2 {
211        panic!("ICE: static_assert called with wrong number of arguments")
212    }
213
214    // To turn the arguments into a string we do the same as we'd do if the arguments
215    // were passed to the built-in foreign call "print" functions.
216    //
217    // We do this before checking whether the argument to `static_assert` is true because
218    // the message argument might end up being dynamic, and we want to report that error
219    // even if the assertion would have passed.
220    let mut foreign_call_params = Vec::with_capacity(arguments.len() - 1);
221    for arg in arguments.iter().skip(1) {
222        append_foreign_call_param(*arg, &function.dfg, instruction, &mut foreign_call_params)?;
223    }
224
225    if function.dfg.is_constant_true(arguments[0]) {
226        return Ok(false);
227    }
228
229    let message = match PrintableValueDisplay::<FieldElement>::try_from_params(&foreign_call_params)
230    {
231        Ok(display_values) => display_values.to_string(),
232        Err(err) => match err {
233            TryFromParamsError::MissingForeignCallInputs => {
234                panic!("ICE: missing foreign call inputs")
235            }
236            TryFromParamsError::ParsingError(error) => {
237                panic!("ICE: could not decode printable type {error:?}")
238            }
239        },
240    };
241
242    let call_stack = function.dfg.get_instruction_call_stack(instruction);
243    if !function.dfg.is_constant(arguments[0]) {
244        if error_on_failure {
245            return Err(RuntimeError::StaticAssertDynamicPredicate { message, call_stack });
246        } else {
247            return Ok(true);
248        }
249    }
250
251    if error_on_failure {
252        Err(RuntimeError::StaticAssertFailed { message, call_stack })
253    } else {
254        Ok(true)
255    }
256}
257
258fn append_foreign_call_param(
259    value: ValueId,
260    dfg: &DataFlowGraph,
261    instruction: InstructionId,
262    foreign_call_params: &mut Vec<ForeignCallParam<FieldElement>>,
263) -> Result<(), RuntimeError> {
264    if let Some(field) = dfg.get_numeric_constant(value) {
265        foreign_call_params.push(ForeignCallParam::Single(field));
266        Ok(())
267    } else if let Some((values, _typ)) = dfg.get_array_constant(value) {
268        let values = vecmap(values, |value| {
269            dfg.get_numeric_constant(value).expect("ICE: expected constant value")
270        });
271        foreign_call_params.push(ForeignCallParam::Array(values));
272        Ok(())
273    } else {
274        let call_stack = dfg.get_instruction_call_stack(instruction);
275        Err(RuntimeError::StaticAssertDynamicMessage { call_stack })
276    }
277}
