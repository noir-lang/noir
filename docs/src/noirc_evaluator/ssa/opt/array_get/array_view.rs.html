1//! The [`ArrayView`] cache used by [`array_get_optimization`][super::Function::array_get_optimization].
2//!
3//! An [`ArrayView`] records the known contents of an array value: the values known to live at
4//! specific constant indices (each with the side-effects predicate it was written under) over a
5//! [base][ArrayBase] that says where any other index gets its value from. The pass builds one view
6//! per array value it scans and resolves a constant-index `array_get` with [`ArrayView::resolve`],
7//! a lookup rather than a walk back over previous instructions.
8//!
9//! The fields are private to this module: outside code seeds a view with [`ArrayView::for_value`],
10//! extends it with [`ArrayView::with_element`], and reads it with [`ArrayView::resolve`], so the
11//! predicate rule that makes a cached element safe to use lives in one place.
12use im::OrdMap;
13
14use crate::ssa::ir::{
15    dfg::DataFlowGraph,
16    instruction::Instruction,
17    types::Type,
18    value::{Value, ValueId},
19};
20
21/// The known contents of an array value, built up incrementally by
22/// [`array_get_optimization`][super::Function::array_get_optimization] as it scans a function.
23#[derive(Clone)]
24pub(super) struct ArrayView {
25    /// Values known to live at specific constant indices, overriding `base`.
26    elements: OrdMap<u32, KnownElement>,
27    /// Where an index that isn't present in `elements` gets its value from.
28    base: ArrayBase,
29}
30
31#[derive(Clone, Copy)]
32struct KnownElement {
33    value: ValueId,
34    /// The side-effects predicate under which this element was written by an `array_set`.
35    predicate: ValueId,
36}
37
38#[derive(Clone)]
39enum ArrayBase {
40    /// Indices not in `elements` come from this `make_array`'s elements, which are stored in logical
41    /// order, so a constant index reads that element directly.
42    MakeArray { elements: im::Vector<ValueId> },
43    /// Indices not in `elements` can be read directly from this array (a function parameter), at
44    /// the same index. `length` bounds which indices that is valid for.
45    ReadFrom { array: ValueId, length: u32 },
46    /// Nothing is known about indices not in `elements`.
47    Unknown,
48}
49
50/// How an `array_get` at a known index can be resolved against an [`ArrayView`].
51pub(super) enum Resolution {
52    /// The `array_get` is equal to this value.
53    Value(ValueId),
54    /// The `array_get` can read from this array instead, at the same index.
55    ReadFrom(ValueId),
56}
57
58impl ArrayView {
59    /// Seeds the view of an array that hasn't been written by an `array_set` earlier in the current
60    /// block: a `make_array` (local or global) exposes its elements directly, a parameter can be
61    /// read from at the same index, and anything else (including arrays from other blocks) is
62    /// opaque.
63    pub(super) fn for_value(dfg: &DataFlowGraph, array: ValueId) -> Self {
64        if let Some((Instruction::MakeArray { elements, .. }, _)) =
65            dfg.get_local_or_global_instruction_with_id(array)
66        {
67            return ArrayView::from_make_array(elements.clone());
68        }
69
70        if let Value::Param { typ: Type::Array(_, length), .. } = &dfg[array] {
71            return ArrayView {
72                elements: OrdMap::new(),
73                base: ArrayBase::ReadFrom { array, length: length.0 },
74            };
75        }
76
77        ArrayView::unknown()
78    }
79
80    /// Records that `value` lives at `index`, written under `predicate`, overriding whatever the
81    /// base or an earlier write said about that index.
82    pub(super) fn with_element(mut self, index: u32, value: ValueId, predicate: ValueId) -> Self {
83        self.elements.insert(index, KnownElement { value, predicate });
84        self
85    }
86
87    /// Resolves an `array_get` of `array` at a known `index` under `predicate`, if the view knows
88    /// what it reads. `array` is the value being read, passed so a read is never rewritten to fetch
89    /// from itself.
90    pub(super) fn resolve(
91        &self,
92        array: ValueId,
93        index: u32,
94        predicate: ValueId,
95        dfg: &DataFlowGraph,
96    ) -> Option<Resolution> {
97        if let Some(element) = self.elements.get(&index) {
98            // A known element can only be used if it was written unconditionally or under the same
99            // predicate as the `array_get`; otherwise the write might not have happened.
100            //
101            // Once the index is known to have been written under *any* predicate the write may have
102            // happened, so we must not fall through to `base`: `base` is fixed to the original array
103            // when the view is seeded and never updated as writes accumulate, so it no longer
104            // describes this index. Return here even on a predicate mismatch, leaving the
105            // `array_get` in place, rather than reading a stale value from `base`.
106            return (dfg.is_constant_true(element.predicate) || element.predicate == predicate)
107                .then_some(Resolution::Value(element.value));
108        }
109
110        match self.base {
111            // `make_array` elements are in logical order, so a constant index reads directly.
112            ArrayBase::MakeArray { ref elements } => {
113                elements.get(index as usize).copied().map(Resolution::Value)
114            }
115            // Reading directly from `array` itself wouldn't be an improvement.
116            ArrayBase::ReadFrom { array: source, length } => {
117                (index < length && source != array).then_some(Resolution::ReadFrom(source))
118            }
119            ArrayBase::Unknown => None,
120        }
121    }
122
123    fn from_make_array(elements: im::Vector<ValueId>) -> Self {
124        ArrayView { elements: OrdMap::new(), base: ArrayBase::MakeArray { elements } }
125    }
126
127    fn unknown() -> Self {
128        ArrayView { elements: OrdMap::new(), base: ArrayBase::Unknown }
129    }
130}
