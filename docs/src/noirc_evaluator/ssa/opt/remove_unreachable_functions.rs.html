1//! Removes any unreachable functions from the code. These can result from
2//! optimizations making existing functions unreachable, e.g. `if false { foo() }`,
3//! or even from monomorphizing an unconstrained version of a constrained function
4//! where the original constrained version ends up never being used.
5//!
6//! This pass identifies all unreachable functions and prunes them from the
7//! function set. Reachability is defined as:
8//! - A function is reachable if it is an entry point (e.g., `main`)
9//! - A function is reachable if it is called from another reachable function
10//! - A function is reachable if it is stored in a reference (e.g., in a `Store` instruction) from another reachable function.
11//!   Even if not immediately called, it may later be dynamically loaded and invoked.
12//!   This marking is conservative but ensures correctness. We should instead rely on [mem2reg][crate::ssa::opt::mem2reg]
13//!   for resolving loads/stores.
14//! - A function is reachable if it is used in a block terminator (e.g., returned from a function)
15//!
16//! The pass builds a call graph based upon the definition of reachability above.
17//! It then identifies all entry points and uses the [CallGraph::reachable_from] utility
18//! to mark all transitively reachable functions. It then discards the rest.
19//!
20//! This pass helps shrink the SSA before compilation stages like inlining and dead code elimination.
21
22use std::collections::BTreeSet;
23
24use rustc_hash::FxHashSet as HashSet;
25
26use crate::ssa::{
27    ir::{
28        call_graph::CallGraph,
29        function::{Function, FunctionId},
30        instruction::Instruction,
31        value::Value,
32    },
33    ssa_gen::Ssa,
34};

36impl Ssa {
37    /// See [`remove_unreachable`][self] module for more information.
38    pub(crate) fn remove_unreachable_functions(mut self) -> Self {
39        let reachable_functions = reachable_functions(&self);
40
41        // Discard all functions not marked as reachable
42        self.functions.retain(|id, _| reachable_functions.contains(id));
43
44        #[cfg(debug_assertions)]
45        remove_unreachable_functions_post_check(&self);
46
47        self
48    }
49}
50
51/// Post-check condition for [Ssa::remove_unreachable_functions].
52///
53/// Succeeds if:
54///   - `ssa` contains no unreachable functions, i.e., all functions are reachable from the entry points.
55///
56/// Otherwise panics.
57///
58/// ## Note
59/// We reuse the same logic for checking reachability as in the main pass, so this function will not
60/// catch any bugs in the pass itself, but it will ensure that the pass is idempotent.
61#[cfg(debug_assertions)]
62fn remove_unreachable_functions_post_check(ssa: &Ssa) {
63    let reachable_functions = reachable_functions(ssa);
64
65    let has_unreachable_functions =
66        ssa.functions.keys().any(|id| !reachable_functions.contains(id));
67
68    assert!(!has_unreachable_functions, "SSA contains unreachable functions");
69}

71/// Identifies all reachable function IDs within the provided [Ssa].
72/// This includes:
73/// - Function calls (functions used via `Call` instructions)
74/// - Function references (functions stored via `Store` instructions)
75///
76/// # Arguments
77/// - `ssa`: The [Ssa] to analyze for usage
78///
79/// # Returns
80/// A sorted set of [`FunctionId`]s that are reachable from the entry points of the SSA.
81fn reachable_functions(ssa: &Ssa) -> HashSet<FunctionId> {
82    // Identify entry points
83    let entry_points =
84        ssa.functions.iter().filter_map(|(&id, _)| ssa.is_entry_point(id).then_some(id));
85
86    // Build call graph dependencies using this passes definition of reachability.
87    let dependencies = ssa.functions.iter().map(|(&id, func)| (id, used_functions(func))).collect();
88    let call_graph = CallGraph::from_deps(dependencies);
89
90    // Traverse the call graph from all entry points
91    call_graph.reachable_from(entry_points)
92}

94/// Identifies all reachable function IDs within a given function.
95/// This includes:
96/// - Function calls (functions used via `Call` instructions)
97/// - Function references (functions stored via `Store` instructions)
98///
99/// # Arguments
100/// - `func`: The [Function] to analyze for usage
101///
102/// # Returns
103/// A sorted set of [`FunctionId`]s that are reachable from the function.
104fn used_functions(func: &Function) -> BTreeSet<FunctionId> {
105    let mut used_function_ids = BTreeSet::default();
106
107    let mut find_functions = |value| {
108        if let Value::Function(function) = func.dfg[value] {
109            used_function_ids.insert(function);
110        }
111    };
112
113    for block_id in func.reachable_blocks() {
114        let block = &func.dfg[block_id];
115
116        for instruction_id in block.instructions() {
117            let instruction = &func.dfg[*instruction_id];
118
119            if matches!(
120                instruction,
121                Instruction::Store { .. }
122                    | Instruction::Call { .. }
123                    | Instruction::MakeArray { .. }
124                    | Instruction::ArraySet { .. }
125            ) {
126                instruction.for_each_value(&mut find_functions);
127            }
128        }
129
130        block.unwrap_terminator().for_each_value(&mut find_functions);
131    }
132
133    used_function_ids
134}

142    #[test]
143    fn remove_unused_brillig() {
144        let src = "
145          brillig(inline) fn main f0 {
146            b0(v0: u32):
147              v2 = call f1(v0) -> u32
148              v4 = add v0, u32 1
149              v5 = eq v2, v4
150              constrain v2 == v4
151              return
152          }
153          brillig(inline) fn increment f1 {
154            b0(v0: u32):
155              v2 = add v0, u32 1
156              return v2
157          }
158          brillig(inline) fn increment_acir f2 {
159            b0(v0: u32):
160              v2 = add v0, u32 1
161              return v2
162          }
163        ";
164
165        let ssa = Ssa::from_str(src).unwrap();
166        let ssa = ssa.remove_unreachable_functions();
167
168        assert_ssa_snapshot!(ssa, @r"
169        brillig(inline) fn main f0 {
170          b0(v0: u32):
171            v2 = call f1(v0) -> u32
172            v4 = add v0, u32 1
173            v5 = eq v2, v4
174            constrain v2 == v4
175            return
176        }
177        brillig(inline) fn increment f1 {
178          b0(v0: u32):
179            v2 = add v0, u32 1
180            return v2
181        }
182        ");
183    }
