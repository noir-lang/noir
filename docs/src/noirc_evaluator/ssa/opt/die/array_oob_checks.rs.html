15    /// Replaces unused ArrayGet/ArraySet instructions with out of bounds checks.
16    /// Returns `true` if at least one check was inserted.
17    /// Because some ArrayGet might happen in groups (for composite types), if just
18    /// some of the instructions in a group are used but not all of them, no check
19    /// is inserted, so this method might return `false`.
20    pub(super) fn replace_array_instructions_with_out_of_bounds_checks(
21        &self,
22        function: &mut Function,
23        block_id: BasicBlockId,
24        possible_index_out_of_bounds_indexes: &mut Vec<usize>,
25    ) -> bool {
26        let mut inserted_check = false;
27
28        // Keep track of the current side effects condition
29        let mut side_effects_condition = None;
30
31        // Keep track of the next index we need to handle
32        let mut next_out_of_bounds_index = possible_index_out_of_bounds_indexes.pop();
33
34        let instructions = function.dfg[block_id].take_instructions();
35        for (index, instruction_id) in instructions.iter().enumerate() {
36            let instruction_id = *instruction_id;
37            let instruction = &function.dfg[instruction_id];
38
39            if let Instruction::EnableSideEffectsIf { condition } = instruction {
40                side_effects_condition = Some(*condition);
41
42                // We still need to keep the EnableSideEffects instruction
43                function.dfg[block_id].instructions_mut().push(instruction_id);
44                continue;
45            }
46
47            // If it's an ArrayGet we'll deal with groups of it in case the array type is a composite type,
48            // and adjust `next_out_of_bounds_index` and `possible_index_out_of_bounds_indexes` accordingly
49            if let Instruction::ArrayGet { array, .. } = instruction {
50                handle_array_get_group(
51                    function,
52                    array,
53                    index,
54                    &mut next_out_of_bounds_index,
55                    possible_index_out_of_bounds_indexes,
56                    &instructions,
57                );
58            }
59
60            let Some(out_of_bounds_index) = next_out_of_bounds_index else {
61                // No more out of bounds instructions to insert, just push the current instruction
62                function.dfg[block_id].instructions_mut().push(instruction_id);
63                continue;
64            };
65
66            if index != out_of_bounds_index {
67                // This instruction is not out of bounds: let's just push it
68                function.dfg[block_id].instructions_mut().push(instruction_id);
69                continue;
70            }
71
72            // This is an instruction that might be out of bounds: let's add a constrain.
73            let (array, index) = match instruction {
74                Instruction::ArrayGet { array, index, .. }
75                | Instruction::ArraySet { array, index, .. } => (array, index),
76                _ => panic!("Expected an ArrayGet or ArraySet instruction here"),
77            };
78
79            let call_stack = function.dfg.get_instruction_call_stack_id(instruction_id);
80
81            let (lhs, rhs) = if function.dfg.get_numeric_constant(*index).is_some() {
82                // If we are here it means the index is known but out of bounds. That's always an error!
83                let false_const = function.dfg.make_constant(false.into(), NumericType::bool());
84                let true_const = function.dfg.make_constant(true.into(), NumericType::bool());
85                (false_const, true_const)
86            } else {
87                let array_typ = function.dfg.type_of_value(*array);
88                let element_size = array_typ.element_size();
89                let len = match array_typ {
90                    Type::Array(_, len) => len,
91                    _ => panic!("Expected an array"),
92                };
93                // `index` will be relative to the flattened array length, so we need to take that into account
94                let array_length = element_size * len;
95
96                // If we are here it means the index is dynamic, so let's add a check that it's less than length.
97
98                // Normally the indexes are expected to be u32, however if the array element is a composite type,
99                // the value could have overflown due to the unchecked multiplication with the element size.
100                // In ACIR, we rely on the array operation itself to fail the circuit if it encounters an overflown value,
101                // however we are just removing the array operation and replacing it with a LessThan, which in ACIR gen
102                // lays down an RangeCheck that would fail if the value doesn't fit 32 bits. As a workaround,
103                // instead of finding the index instruction and changing into a checked multiplication,
104                // we cast to a higher bitsize, which we expect should fit any overflown index type.
105                let length_type = NumericType::unsigned(64);
106
107                let index = function.dfg.insert_instruction_and_results(
108                    Instruction::Cast(*index, length_type),
109                    block_id,
110                    None,
111                    call_stack,
112                );
113                let index = index.first();
114                let array_length = function.dfg.make_constant(array_length.0.into(), length_type);
115
116                let is_index_in_bounds = function.dfg.insert_instruction_and_results(
117                    Instruction::binary(BinaryOp::Lt, index, array_length),
118                    block_id,
119                    None,
120                    call_stack,
121                );
122                let is_index_in_bounds = is_index_in_bounds.first();
123                let true_const = function.dfg.make_constant(true.into(), NumericType::bool());
124                (is_index_in_bounds, true_const)
125            };
126
127            let (lhs, rhs) = apply_side_effects(
128                side_effects_condition,
129                lhs,
130                rhs,
131                function,
132                block_id,
133                call_stack,
134            );
135
136            let message = Some("Index out of bounds".to_owned().into());
137            function.dfg.insert_instruction_and_results(
138                Instruction::Constrain(lhs, rhs, message),
139                block_id,
140                None,
141                call_stack,
142            );
143            inserted_check = true;
144
145            next_out_of_bounds_index = possible_index_out_of_bounds_indexes.pop();
146        }
147
148        inserted_check
149    }
150}

152/// Array get/set has explicit out of bounds (OOB) checks laid down in the Brillig runtime.
153/// These checks are not laid down in the ACIR runtime as that runtime maps the SSA
154/// to a memory model where OOB accesses will be prevented. Essentially all array ops
155/// in ACIR will have a side effect where they check for the index being OOB.
156/// However, in order to maintain parity between the Brillig and ACIR runtimes,
157/// if we have an unused array operation we need insert an OOB check so that the
158/// side effects ordering remains correct.
159pub(super) fn should_insert_oob_check(function: &Function, instruction: &Instruction) -> bool {
160    if !function.runtime().is_acir() {
161        return false;
162    }
163
164    use Instruction::*;
165    match instruction {
166        ArrayGet { array, index } | ArraySet { array, index, .. } => {
167            // We only care about arrays here as vectors are expected to have explicit checks laid down in the initial SSA.
168            function.dfg.try_get_array_length(*array).is_some()
169                && !function.dfg.is_safe_index(*index, *array)
170        }
171        _ => false,
172    }
173}

175/// Handle the case when an `ArrayGet` is potentially out-of-bounds and the array contains composite types
176/// by figuring out whether all `ArrayGet` of different parts of the complex item are unused, and if so
177/// then insert a single constraint to replace all of them.
178///
179/// Consumes all items from `possible_index_out_of_bounds_indexes` that belong to the current group and
180/// sets `next_out_of_bounds_index` to the *current* index, expecting that `replace_array_instructions_with_out_of_bounds_checks`
181/// will see that as a signal that the current index is out of bounds and it should insert a constraint.
182/// Then, the next a `ArrayGet`s in the group will be re-inserted, but they won't be treated as potentially
183/// OOB any more, and shall be removed in the next DIE pass as simply unused.
184fn handle_array_get_group(
185    function: &Function,
186    // The array from which we are getting an item.
187    array: &ValueId,
188    // Index of the current instruction. If it's not the same as `Some(next_out_of_bounds_index)`
189    // then this instruction was not unsafe.
190    index: usize,
191    // The last index popped from `possible_index_out_of_bounds_indexes`.
192    next_out_of_bounds_index: &mut Option<usize>,
193    // Remaining out of bounds indexes, all of which are unused.
194    possible_index_out_of_bounds_indexes: &mut Vec<usize>,
195    // All the instructions in this block.
196    instructions: &[InstructionId],
197) {
198    if function.dfg.try_get_array_length(*array).is_none() {
199        // Nothing to do for vectors
200        return;
201    }
202
203    let element_size = function.dfg.type_of_value(*array).element_size().to_usize();
204    if element_size <= 1 {
205        // Not a composite type
206        return;
207    }
208
209    // It's a composite type.
210    // When doing ArrayGet on a composite type, this **always** results in instructions like these
211    // (assuming element_size == 3):
212    //
213    // 1.    v27 = array_get v1, index v26
214    // 2.    v28 = add v26, u32 1
215    // 3.    v29 = array_get v1, index v28
216    // 4.    v30 = add v26, u32 2
217    // 5.    v31 = array_get v1, index v30
218    //
219    // That means that after this instructions, (element_size - 1) instructions will be
220    // part of this composite array get, and they'll be two instructions apart.
221    //
222    // However, that is only true for the initial SSA. After we run DIE, it might remove
223    // some of the instructions that were unused, leaving the ones which had uses, destroying
224    // the group, so in general we cannot assume to see all element_size instruction to be present.
225    //
226    // Assuming we have identified a group, three things can happen:
227    // a) none of the array_get instructions are unused: in this case they won't be in
228    //    `possible_index_out_of_bounds_indexes` and they won't be removed, nothing to do here
229    // b) all of the array_get instructions are unused: in this case we can replace **all**
230    //    of them with just one constrain: no need to do one per array_get
231    // c) some of the array_get instructions are unused, but not all: in this case
232    //    we don't need to insert any constrain, because on a later stage array bound checks
233    //    will be performed anyway. We'll let DIE remove the unused ones, without replacing
234    //    them with bounds checks, and leave the used ones.
235    //
236    // To check in which scenario we are we can get from `possible_index_out_of_bounds_indexes`
237    // (starting from `next_out_of_bounds_index`) while we are in the group ranges
238    // (1..=5 in the example above)
239    let Some(out_of_bounds_index) = *next_out_of_bounds_index else {
240        // No next unused instruction, so this is case a) and nothing needs to be done here
241        return;
242    };
243
244    if index != out_of_bounds_index {
245        // The next index is not the one for the current instructions,
246        // so we are in case a), and nothing needs to be done here
247        return;
248    }
249
250    // Initially we would expect the last index of the group (5 in the example above)
251    // to be `index + 2 * (element_size - 1)`, however, we can't expect this to hold
252    // after previous DIE passes have partially removed the group.
253    let last_possible_index = index + 2 * (element_size - 1);
254    // Instead we need to check how many `ArrayGet` and `Add` we have following this
255    // instruction that read the same array, and how many of these instructions are unused.
256    let max_index = last_possible_index.min(instructions.len() - 1);
257
258    // How many unused instructions are in this group? We know the current instruction is unused.
259    let mut unused_count = 1;
260    let mut group_count = 1;
261
262    for (i, next_id) in instructions.iter().enumerate().take(max_index + 1).skip(index + 1) {
263        let next_instruction = &function.dfg[*next_id];
264        match next_instruction {
265            // Skip `Add`
266            Instruction::Binary(Binary { operator: BinaryOp::Add { .. }, .. }) => {
267                continue;
268            }
269            Instruction::ArrayGet { array: next_array, index: next_index }
270                if next_array == array =>
271            {
272                // Still reading the same array.
273                // There is a chance that *this* instruction is safe, which means the one before it
274                // needs to be replaced with a constraint, even if this does not.
275                if function.dfg.is_safe_index(*next_index, *next_array) {
276                    break;
277                }
278                // This instruction is also OOB, so it belongs to the same group.
279                group_count += 1;
280                // Check if this result is also unused.
281                *next_out_of_bounds_index = possible_index_out_of_bounds_indexes.pop();
282                let Some(out_of_bounds_index) = *next_out_of_bounds_index else {
283                    // This ArrayGet is not recorded as a potential OOB; we know it's OOB, so this means it's not unused.
284                    // That means we can let the built-in OOB check take care of it.
285                    break;
286                };
287                if out_of_bounds_index == i {
288                    unused_count += 1;
289                    continue;
290                } else {
291                    // The next OOB index is for some other array, not this one; the last array get
292                    // reading this array is not OOB or not unused.
293                    break;
294                }
295            }
296            _ => {
297                // Some other instruction that doesn't belong to the group.
298                break;
299            }
300        }
301    }
302
303    if unused_count == group_count {
304        // We are in case b): we need to insert just one constrain.
305        // Since we popped all of the group indexes, and given that we
306        // are analyzing the first instruction in the group, we can
307        // set `next_out_of_bounds_index` to the current index:
308        // then a check will be inserted, and no other check will be
309        // inserted for the rest of the group.
310        *next_out_of_bounds_index = Some(index);
311    } else {
312        // We are in case c): some of the instructions are unused.
313        // We don't need to insert any checks, and given that we already popped
314        // all of the indexes in the group, there's nothing else to do here.
315    }
316}

318/// Given `lhs` and `rhs` values, if there's a side effects condition this will
319/// return (`lhs * condition`, `rhs * condition`), otherwise just (`lhs`, `rhs`)
320fn apply_side_effects(
321    side_effects_condition: Option<ValueId>,
322    lhs: ValueId,
323    rhs: ValueId,
324    function: &mut Function,
325    block_id: BasicBlockId,
326    call_stack: CallStackId,
327) -> (ValueId, ValueId) {
328    // See if there's an active "enable side effects" condition
329    let Some(condition) = side_effects_condition else {
330        return (lhs, rhs);
331    };
332
333    let dfg = &mut function.dfg;
334
335    // Condition needs to be cast to argument type in order to multiply them together.
336    // In our case, lhs is always a boolean.
337    let cast = Instruction::Cast(condition, NumericType::bool());
338    let casted_condition = dfg.insert_instruction_and_results(cast, block_id, None, call_stack);
339    let casted_condition = casted_condition.first();
340
341    // Unchecked mul because the side effects var is always 0 or 1
342    let lhs = dfg.insert_instruction_and_results(
343        Instruction::binary(BinaryOp::Mul { unchecked: true }, lhs, casted_condition),
344        block_id,
345        None,
346        call_stack,
347    );
348    let lhs = lhs.first();
349
350    // Unchecked mul because the side effects var is always 0 or 1
351    let rhs = dfg.insert_instruction_and_results(
352        Instruction::binary(BinaryOp::Mul { unchecked: true }, rhs, casted_condition),
353        block_id,
354        None,
355        call_stack,
356    );
357    let rhs = rhs.first();
358
359    (lhs, rhs)
