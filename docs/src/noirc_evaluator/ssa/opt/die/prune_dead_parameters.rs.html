1//! This module implements the pruning of unused block parameters from functions.
2//!
3//! Blocks can accept parameters that are passed via terminator instructions (e.g., `jmp`).
4//! During the compilation pipeline, it's common for some of these parameters to become unused.
5//! This module eliminates those unused parameters and correspondingly
6//! adjusts the terminators of predecessor blocks. This work is ultimately done to reduce code size.
7//!
8//! ## How this pass works:
9//! - Iterates through all blocks in post-order (to ensure predecessors are visited after successors).
10//! - Detects and removes unused block parameters, except for those on entry blocks of program entry point
11//!   functions (e.g., `main`, foldable functions). Brillig entry points (except `main`) are not counted
12//!   here as their signature does not determine the ABI and is safe to update.
13//! - Clears the list of unused block parameters after removing them from the block.
14//! - Updates the corresponding argument lists in predecessor terminator instructions to keep
15//!   them aligned with the new parameter lists.
16//! - **Entry block parameters** of non-program entry point functions may be pruned if they are unused,
17//!   and the corresponding call instructions are rewritten to remove the dead arguments.
18//! - Entry point functions often correspond to ABI inputs and must remain to preserve the program's external interface.
19//!
20//! ## Preconditions:
21//! - This pass should be run *after* [Dead Instruction Elimination (DIE)][super] so that parameter
22//!   liveness is up-to-date.
23//!
24//! ## Panics
25//! Return blocks are not expected to have successors, so encountering one as a predecessor
26//! is treated as an internal compiler error (ICE).
27//!
28//! ## Example:
29//!
30//! Before pruning:
31//! ```text
32//! b0():
33//!   jmp b1(Field 1, Field 2, Field 3)
34//! b1(v0: Field, v1: Field, v2: Field):
35//!   return v1
36//! ```
37//!
38//! After pruning:
39//! ```text
40//! b0():
41//!   jmp b1(Field 2)
42//! b1(v0: Field):
43//!   return v0
44//! ```
45//!
46//! ## Entry block callsite update example
47//!
48//! Original function `f1`:
49//! ```text
50//! brillig(inline) fn f1:
51//!   b0(v0: Field, v1: Field):
52//!     jmp b1(Field 1)
53//!   b1(v2: Field):
54//!     return v2
55//! ```
56//!
57//! Call to `f1` in another function:
58//! ```text
59//! v0 = call f1(Field 5, Field 10) -> Field
60//! ```
61//!
62//! After pruning::
63//! ```text
64//! brillig(inline) fn f1:
65//!   b0():
66//!     jmp b1(Field 1)
67//!   b1(v2: Field):
68//!     return v2
69//! ```
70//!
71//! Correspondingly, the callsite is rewritten to remove the now unused arguments:
72//! ```text
73//! v0 = call f1() -> Field
74//! ```

90impl Ssa {
91    /// See [`prune_dead_parameters`][self] module for more information.
92    #[tracing::instrument(level = "trace", skip_all)]
93    pub(crate) fn prune_dead_parameters(
94        mut self,
95        unused_parameters: &HashMap<FunctionId, HashMap<BasicBlockId, Vec<ValueId>>>,
96    ) -> Self {
97        let mut entry_block_keep_lists = HashMap::default();
98        for (func_id, unused_parameters) in unused_parameters {
99            let function = self.functions.get_mut(func_id).expect("ICE: Function should exist");
100            if let Some(entry_keep_list) =
101                function.prune_dead_parameters(unused_parameters, self.main_id)
102            {
103                entry_block_keep_lists.insert(*func_id, entry_keep_list);
104            }
105        }
106
107        let call_graph = CallGraph::from_ssa(&self);
108        self.rewrite_calls_to_pruned_entry_blocks(&call_graph, &entry_block_keep_lists);
109
110        self
111    }
112
113    /// Helper to rewrite [Call][Instruction::Call] instructions targeting functions whose entry block parameters were pruned.
114    fn rewrite_calls_to_pruned_entry_blocks(
115        &mut self,
116        call_graph: &CallGraph,
117        entry_block_keep_lists: &HashMap<FunctionId, Vec<bool>>,
118    ) {
119        let callers = call_graph.callers();
120
121        for (callee_id, keep_list) in entry_block_keep_lists {
122            let Some(caller_map) = callers.get(callee_id) else {
123                continue;
124            };
125
126            for caller_id in caller_map.keys() {
127                let caller_func =
128                    self.functions.get_mut(caller_id).expect("ICE: Caller function should exist");
129
130                for block_id in caller_func.reachable_blocks() {
131                    // Collect call sites to rewrite
132                    let mut call_sites_to_update = HashMap::default();
133                    for &instruction_id in caller_func.dfg[block_id].instructions() {
134                        let instruction = &caller_func.dfg[instruction_id];
135                        let Instruction::Call { func, arguments } = instruction else {
136                            continue;
137                        };
138
139                        let Value::Function(target_id) = caller_func.dfg[*func] else {
140                            continue;
141                        };
142
143                        if target_id != *callee_id {
144                            continue;
145                        }
146
147                        // Filter the arguments using keep_list
148                        let new_args: Vec<ValueId> = arguments
149                            .iter()
150                            .zip(keep_list.iter())
151                            .filter_map(|(arg, &keep)| if keep { Some(*arg) } else { None })
152                            .collect();
153
154                        call_sites_to_update.insert(instruction_id, new_args);
155                    }
156
157                    // Apply call site rewrites
158                    for (instruction_id, new_args) in call_sites_to_update {
159                        let Instruction::Call { arguments, .. } =
160                            &mut caller_func.dfg[instruction_id]
161                        else {
162                            unreachable!("expected call site to be call instruction");
163                        };
164                        *arguments = new_args;
165                    }
166                }
167            }
168        }
169    }
170}

172impl Function {
173    /// See [`prune_dead_parameters`][self] module for more information
174    fn prune_dead_parameters(
175        &mut self,
176        unused_params: &HashMap<BasicBlockId, Vec<ValueId>>,
177        main_id: FunctionId,
178    ) -> Option<Vec<bool>> {
179        let cfg = ControlFlowGraph::with_function(self);
180        let post_order = PostOrder::with_cfg(&cfg);
181
182        let is_entry_point = self.runtime().is_entry_point();
183        let is_main = self.id() == main_id;
184        let can_prune_entry_block = !(is_entry_point || is_main);
185
186        let mut entry_block_keep_list = None;
187        for &block in post_order.as_slice() {
188            // We do not support removing the function arguments of entry points. This is because function signatures,
189            // which are used for setting up the program artifact inputs, are set by the frontend.
190            if block == self.entry_block() && !can_prune_entry_block {
191                continue;
192            }
193
194            let empty_params = Vec::new();
195            let unused_params = unused_params.get(&block).unwrap_or(&empty_params);
196            if unused_params.is_empty() {
197                // Nothing to do if the block has no unused params
198                continue;
199            }
200
201            let old_params = self.dfg[block].take_parameters();
202
203            // Create the list of new params for updating the block with unused parameters
204            // as well as an indexed list of the removed parameters to update each predecessor's terminator argument list.
205            let mut keep_list = Vec::with_capacity(old_params.len());
206            let mut new_params = Vec::with_capacity(old_params.len());
207            let unused_set = unused_params.iter().copied().collect::<HashSet<_>>();
208            for param in old_params {
209                let keep = !unused_set.contains(&param);
210                keep_list.push(keep);
211                if keep {
212                    new_params.push(param);
213                }
214            }
215
216            self.dfg[block].set_parameters(new_params);
217
218            // Update the predecessor argument list to match the new parameter list
219            self.update_predecessor_terminators(&cfg, block, &keep_list);
220
221            if block == self.entry_block() {
222                entry_block_keep_list = Some(keep_list);
223            }
224        }
225        entry_block_keep_list
226    }
227
228    /// Update terminator arguments of predecessor blocks after pruning.
229    fn update_predecessor_terminators(
230        &mut self,
231        cfg: &ControlFlowGraph,
232        target_block: BasicBlockId,
233        keep_list: &[bool],
234    ) {
235        let predecessors = cfg.predecessors(target_block);
236        for pred in predecessors {
237            let terminator = self.dfg[pred].unwrap_terminator_mut();
238
239            match terminator {
240                TerminatorInstruction::JmpIf { .. } => {
241                    // No terminator arguments in a JmpIf
242                }
243                TerminatorInstruction::Jmp { destination, arguments, .. } => {
244                    // Predecessor must jump to its successor
245                    assert_eq!(*destination, target_block);
246                    let new_args = arguments
247                        .iter()
248                        .zip(keep_list.iter())
249                        .filter_map(|(arg, &keep)| if keep { Some(*arg) } else { None })
250                        .collect();
251                    *arguments = new_args;
252                }
253                TerminatorInstruction::Return { .. } => {
254                    unreachable!("ICE: A return block should not be a predecessor");
255                }
256                TerminatorInstruction::Unreachable { .. } => {
257                    unreachable!("ICE: An unreachable block should not be a predecessor");
258                }
259            }
260        }
261    }

273    fn prune_unused_block_params() {
274        let src = r#"
275        brillig(inline) fn test f0 {
276          b0():
277            jmp b1(Field 1, Field 2, Field 3)
278          b1(v0: Field, v1: Field, v2: Field):
279            return v1
280        }"#;
281
282        let ssa = Ssa::from_str(src).unwrap();
283        // DIE is necessary to fetch the block parameters liveness information
284        let (ssa, die_result) = ssa.dead_instruction_elimination_inner(false);
285

288            .unused_parameters

290            .expect("Should have unused parameters");
291        let b0_unused = function.get(&Id::test_new(0)).expect("Should have unused parameters");
292        // b0 has no parameters
293        assert!(b0_unused.is_empty());

551    fn prune_parameter_used_in_a_separate_block_terminator() {
552        // The following SSA comes this code:
553        // ```noir
554        // fn main(input: i16) {
555        //     let result = if input > 3 {
556        //         if input > 4 {
557        //             1
558        //         } else {
559        //             2
560        //         }
561        //     } else {
562        //         3
563        //     };
564        //     let result2 = if input > 5 {
565        //         result
566        //     } else {
567        //         result + 1
568        //     };
569        // }
570        // ```
571        let src = r#"
572        brillig(inline) predicate_pure fn main f0 {
573          b0(v0: i16):
574            v5 = lt i16 3, v0
575            jmpif v5 then: b1, else: b2
576          b1():
577            v8 = lt i16 4, v0
578            jmpif v8 then: b3, else: b4
579          b2():
580            jmp b5(Field 3)
581          b3():
582            jmp b6(Field 1)
583          b4():
584            jmp b6(Field 2)
585          b5(v1: Field):
586            v12 = lt i16 5, v0
587            jmpif v12 then: b7, else: b8
588          b6(v2: Field):
589            jmp b5(v2)
590          b7():
591            jmp b9(v1)
592          b8():
593            v13 = add v1, Field 1
594            jmp b9(v13)
595          b9(v3: Field):
596            return
597        }
598        "#;
599        let ssa = Ssa::from_str(src).unwrap();
600
601        // DIE is necessary to fetch the block parameters liveness information
602        let (ssa, die_result) = ssa.dead_instruction_elimination_inner(false);
603
604        assert!(die_result.unused_parameters.len() == 1);
605        let function = die_result
606            .unused_parameters
607            .get(&Id::test_new(0))
608            .expect("Should have unused parameters");
609        for (block_id, unused_params) in function {
610            if block_id.to_u32() == 9 {
611                assert!(unused_params.len() == 1);
612                assert_eq!(unused_params[0].to_u32(), 3);
613            } else if block_id.to_u32() == 5 {
614                assert!(unused_params.len() == 1);
615                assert_eq!(unused_params[0].to_u32(), 1);
616            } else if block_id.to_u32() == 6 {
617                assert!(unused_params.len() == 1);
618                assert_eq!(unused_params[0].to_u32(), 2);
619            } else {
620                assert!(unused_params.is_empty());
621            }
622        }
623
624        let ssa = ssa.prune_dead_parameters(&die_result.unused_parameters);
625
626        let (ssa, die_result) = ssa.dead_instruction_elimination_inner(false);
627
628        assert!(die_result.unused_parameters.len() == 1);
629        let function = die_result
630            .unused_parameters
631            .get(&Id::test_new(0))
632            .expect("Should have unused parameters");
633        for unused_params in function.values() {
634            assert!(unused_params.is_empty());
635        }
636
