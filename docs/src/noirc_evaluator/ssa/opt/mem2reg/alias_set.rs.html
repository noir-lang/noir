1use vec_collections::{AbstractVecSet, VecSet};
2
3use crate::ssa::ir::value::ValueId;
4
5/// A set of possible aliases. Each ValueId in this set represents one possible value the reference
6/// holding this AliasSet may be aliased to. This struct wrapper is provided so that when we take
7/// the union of multiple alias sets, the result should be empty if any individual set is empty.
8///
9/// Note that we distinguish between "definitely has no aliases" - `Some(BTreeSet::new())`, and
10/// "unknown which aliases this may refer to" - `None`.
11#[derive(Debug, Default, Clone)]
12pub(super) struct AliasSet {
13    aliases: Option<VecSet<[ValueId; 1]>>,
14}
15
16impl AliasSet {
17    pub(super) fn unknown() -> AliasSet {
18        Self { aliases: None }
19    }
20
21    pub(super) fn known(value: ValueId) -> AliasSet {
22        Self { aliases: Some(VecSet::single(value)) }
23    }
24
25    pub(super) fn known_multiple(values: VecSet<[ValueId; 1]>) -> AliasSet {
26        Self { aliases: Some(values) }
27    }
28
29    /// In rare cases, such as when creating an empty array of references, the set of aliases for a
30    /// particular value will be known to be zero, which is distinct from being unknown and
31    /// possibly referring to any alias.
32    pub(super) fn known_empty() -> AliasSet {
33        Self { aliases: Some(VecSet::empty()) }
34    }
35
36    pub(super) fn is_unknown(&self) -> bool {
37        self.aliases.is_none()
38    }
39
40    /// Return the single known alias if there is exactly one.
41    /// Otherwise, return None.
42    pub(super) fn single_alias(&self) -> Option<ValueId> {
43        self.aliases
44            .as_ref()
45            .and_then(|aliases| (aliases.len() == 1).then(|| *aliases.iter().next().unwrap()))
46    }
47
48    /// Unify this alias set with another. The result of this set is empty if either set is empty.
49    /// Otherwise, it is the union of both alias sets.
50    pub(super) fn unify(&mut self, other: &Self) {
51        if let (Some(self_aliases), Some(other_aliases)) = (&mut self.aliases, &other.aliases) {
52            self_aliases.extend(other_aliases.iter().cloned());
53        } else {
54            self.aliases = None;
55        }
56    }
57
58    /// Returns true if calling `unify` would change something in this alias set.
59    ///
60    /// This is an optimization to avoid having to look up an entry ready to be modified in the [Block](crate::ssa::opt::mem2reg::block::Block),
61    /// because doing so would involve calling `Arc::make_mut` which clones the entry, ready for modification.
62    pub(super) fn should_unify(&self, other: &Self) -> bool {
63        if let (Some(self_aliases), Some(other_aliases)) = (&self.aliases, &other.aliases) {
64            // `unify` would extend `self_aliases` with `other_aliases`, so if `other_aliases` is a subset, then nothing would happen.
65            !other_aliases.is_subset(self_aliases)
66        } else {
67            // `unify` would set `aliases` to `None`, so if it's not `Some`, then nothing would happen.
68            self.aliases.is_some()
69        }
70    }
71
72    /// Inserts a new alias into this set if it is not unknown
73    pub(super) fn insert(&mut self, new_alias: ValueId) {
74        if let Some(aliases) = &mut self.aliases {
75            aliases.insert(new_alias);
76        }
77    }
78
79    /// Returns `Some(true)` if `f` returns true for any known alias in this set.
80    /// If this alias set is unknown, None is returned.
81    pub(super) fn any(&self, f: impl FnMut(ValueId) -> bool) -> Option<bool> {
82        self.aliases.as_ref().map(|aliases| aliases.iter().copied().any(f))
83    }
84
85    /// Returns an iterator over the values in this alias set.
86    pub(super) fn iter(&self) -> impl Iterator<Item = ValueId> {
87        self.aliases.iter().flat_map(|aliases| aliases.iter().copied())
88    }
89
90    /// Return the first ValueId in the alias set as long as there is at least one.
91    /// The ordering is arbitrary (by lowest ValueId) so this method should only be
92    /// used when you need an arbitrary ValueId from the alias set.
93    pub(super) fn first(&self) -> Option<ValueId> {
94        self.iter().next()
95    }
96}
