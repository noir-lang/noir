11/// A `Block` acts as a per-block context for the mem2reg pass.
12/// Most notably, it contains the current alias set thought to track each
13/// reference value if known, and it contains the expected ReferenceValue
14/// for each ValueId. When a block is finished, the final values of these
15/// are expected to match the values held by each ValueId at the very end
16/// of a block.
17#[derive(Debug, Default, Clone)]
18pub(super) struct Block {
19    /// Maps a ValueId to the Expression it represents.
20    /// Multiple ValueIds can map to the same Expression, e.g.
21    /// dereferences to the same allocation.
22    pub(super) expressions: im::OrdMap<ValueId, Expression>,
23
24    /// Each expression is tracked as to how many aliases it
25    /// may have. If there is only 1, we can attempt to optimize
26    /// out any known loads to that alias. Note that "alias" here
27    /// includes the original reference as well.
28    pub(super) aliases: im::OrdMap<Expression, AliasSet>,
29
30    /// Each allocate instruction result (and some reference block parameters)
31    /// will map to a Reference value which tracks whether the last value stored
32    /// to the reference is known.
33    pub(super) references: im::OrdMap<ValueId, ValueId>,
34
35    /// The last instance of a `Store` instruction to each address in this block
36    pub(super) last_stores: im::OrdMap<ValueId, InstructionId>,
37
38    // The last instance of a `Load` instruction to each address in this block
39    pub(super) last_loads: im::OrdMap<ValueId, InstructionId>,
40}
41
42/// An `Expression` here is used to represent a canonical key
43/// into the aliases map since otherwise two dereferences of the
44/// same address will be given different ValueIds.
45#[derive(Debug, Copy, Clone, PartialOrd, Ord, PartialEq, Eq, Hash)]
46pub(super) enum Expression {
47    Dereference(ValueId),
48    ArrayElement(ValueId),
49    Other(ValueId),
50}
51
52impl Block {
53    /// If the given reference id points to a known value, return the value
54    pub(super) fn get_known_value(&self, address: ValueId) -> Option<ValueId> {
55        self.references.get(&address).copied()
56    }
57
58    /// If the given address is known, set its value to `value`.
59    pub(super) fn set_known_value(&mut self, address: ValueId, value: ValueId) {
60        self.set_value(address, Some(value));
61    }
62
63    pub(super) fn set_unknown(&mut self, address: ValueId) {
64        self.set_value(address, None);
65    }
66
67    fn set_value(&mut self, address: ValueId, value: Option<ValueId>) {
68        let expression = self.expressions.entry(address).or_insert(Expression::Other(address));
69        let aliases = self.aliases.entry(*expression).or_default();
70
71        if aliases.is_unknown() {
72            // uh-oh, we don't know at all what this reference refers to, could be anything.
73            // Now we have to invalidate every reference we know of
74            self.invalidate_all_references();
75        } else if let Some(alias) = aliases.single_alias() {
76            // We always know address points to value
77            self.set_reference_value(alias, value);
78        } else {
79            // More than one alias. We're not sure which it refers to so we have to
80            // conservatively invalidate all references it may refer to.
81            for alias in aliases.iter() {
82                self.references.remove(&alias);
83            }
84        }
85
86        // We always know address points to value
87        self.set_reference_value(address, value);
88    }
89
90    fn set_reference_value(&mut self, address: ValueId, value: Option<ValueId>) {
91        if let Some(value) = value {
92            self.references.insert(address, value);
93        } else {
94            self.references.remove(&address);
95        }
96    }
97
98    fn invalidate_all_references(&mut self) {
99        self.references.clear();
100        self.last_stores.clear();
101    }
102
103    pub(super) fn unify(mut self, other: &Self) -> Self {
104        for (value_id, expression) in &other.expressions {
105            if let Some(existing) = self.expressions.get(value_id) {
106                assert_eq!(existing, expression, "Expected expressions for {value_id} to be equal");
107            } else {
108                self.expressions.insert(*value_id, *expression);
109            }
110        }
111
112        for (expression, new_aliases) in &other.aliases {
113            // If nothing would change, then don't call `.entry(...).and_modify(...)` as it involves creating more `Arc`s.
114            if let Some(aliases) = self.aliases.get(expression)
115                && !aliases.should_unify(new_aliases)
116            {
117                continue;
118            }
119            self.aliases
120                .entry(*expression)
121                .and_modify(|aliases| aliases.unify(new_aliases))
122                .or_insert_with(|| new_aliases.clone());
123        }
124
125        // Keep only the references present in both maps.
126        let mut intersection = im::OrdMap::new();
127        for (value_id, reference) in &other.references {
128            if let Some(existing) = self.references.get(value_id)
129                && reference == existing
130            {
131                intersection.insert(*value_id, *reference);
132            }
133        }
134        self.references = intersection;
135
136        // Keep only the last loads present in both maps, if they map to the same InstructionId
137        let mut intersection = im::OrdMap::new();
138        for (value_id, instruction) in &other.last_loads {
139            if let Some(existing) = self.last_loads.get(value_id)
140                && existing == instruction
141            {
142                intersection.insert(*value_id, *instruction);
143            }
144        }
145        self.last_loads = intersection;
146
147        self
148    }

150    /// Remember that `result` is the result of dereferencing `address`. This is important to
151    /// track aliasing when references are stored within other references.
152    pub(super) fn remember_dereference(
153        &mut self,
154        function: &Function,
155        address: ValueId,
156        result: ValueId,
157    ) {
158        if function.dfg.value_is_reference(result) {
159            if let Some(known_address) = self.get_known_value(address) {
160                self.expressions.insert(result, Expression::Other(known_address));
161            } else {
162                let expression = Expression::Dereference(address);
163                self.expressions.insert(result, expression);
164                // No known aliases to insert for this expression... can we find an alias
165                // even if we don't have a known address? If not we'll have to invalidate all
166                // known references if this reference is ever stored to.
167            }
168        }
169    }
170
171    /// Forget the last store to an address and all of its aliases, to eliminate them
172    /// from the candidates for removal at the end.
173    ///
174    /// Note that this only affects this block: when we merge blocks we clear the
175    /// last stores anyway, we don't inherit them from predecessors, so if one
176    /// block stores to an address and a descendant block loads it, this mechanism
177    /// does not affect the candidacy of the last store in the predecessor block.
178    fn keep_last_stores_for(&mut self, address: ValueId, function: &Function) {
179        self.keep_last_store(address, function);
180
181        for alias in (*self.get_aliases_for_value(address)).clone().iter() {
182            self.keep_last_store(alias, function);
183        }
184    }
185
186    /// Forget the last store to an address, to remove it from the set of instructions
187    /// which are candidates for removal at the end. Also marks the values in the last
188    /// store as used, now that we know we want to keep them.
189    fn keep_last_store(&mut self, address: ValueId, function: &Function) {
190        if let Some(instruction) = self.last_stores.remove(&address) {
191            // Whenever we decide we want to keep a store instruction, we also need
192            // to go through its stored value and mark that used as well.
193            match &function.dfg[instruction] {
194                Instruction::Store { value, .. } => {
195                    self.mark_value_used(*value, function);
196                }
197                other => {
198                    unreachable!("last_store held an id of a non-store instruction: {other:?}")
199                }
200            }
201        }
202    }
203
204    /// Mark a value (for example an address we loaded) as used by forgetting the last store instruction,
205    /// which removes it from the candidates for removal.
206    pub(super) fn mark_value_used(&mut self, value: ValueId, function: &Function) {
207        self.keep_last_stores_for(value, function);
208
209        // We must do a recursive check for arrays since they're the only Values which may contain
210        // other ValueIds.
211        if let Some((array, _)) = function.dfg.get_array_constant(value) {
212            for value in array {
213                self.mark_value_used(value, function);
214            }
215        }
216    }
217
218    /// Collect all aliases used by the given value list
219    pub(super) fn collect_all_aliases(
220        &self,
221        values: impl IntoIterator<Item = ValueId>,
222    ) -> AliasSet {
223        let mut aliases = AliasSet::known_empty();
224        for value in values {
225            aliases.unify(&self.get_aliases_for_value(value));
226        }
227        aliases
228    }
229
230    pub(super) fn get_aliases_for_value(&self, value: ValueId) -> Cow<AliasSet> {
231        if let Some(expression) = self.expressions.get(&value)
232            && let Some(aliases) = self.aliases.get(expression)
233        {
234            return Cow::Borrowed(aliases);
235        }
236
237        Cow::Owned(AliasSet::unknown())
238    }
239
240    pub(super) fn set_last_load(&mut self, address: ValueId, instruction: InstructionId) {
241        let aliases = self.get_aliases_for_value(address);
242        if !aliases.is_unknown() {
243            self.last_loads.insert(address, instruction);
244        }
245    }
246
247    pub(super) fn keep_last_load_for(&mut self, address: ValueId) {
248        self.last_loads.remove(&address);
249
250        if let Some(expr) = self.expressions.get(&address)
251            && let Some(aliases) = self.aliases.get(expr)
252        {
253            for alias in aliases.iter() {
254                self.last_loads.remove(&alias);
255            }
256        }
257    }
