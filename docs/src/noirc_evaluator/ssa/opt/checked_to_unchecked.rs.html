1//! This SSA pass will turn checked unsigned binary additions, subtractions and multiplications
2//! into unchecked ones if it's guaranteed that the operations cannot overflow.
3//!
4//! Signed checked binary operations should have already been converted to unchecked ones with
5//! an explicit overflow check during [`super::expand_signed_checks`].
6use acvm::AcirField as _;
7use rustc_hash::FxHashMap as HashMap;
8
9use crate::ssa::{
10    ir::{
11        dfg::DataFlowGraph,
12        function::Function,
13        instruction::{Binary, BinaryOp, Instruction},
14        types::NumericType,
15        value::{Value, ValueId},
16    },
17    ssa_gen::Ssa,
18};
19
20impl Ssa {
21    /// See [`checked_to_unchecked`][self] module for more information.
22    #[tracing::instrument(level = "trace", skip(self))]
23    pub(crate) fn checked_to_unchecked(mut self) -> Ssa {
24        for function in self.functions.values_mut() {
25            function.checked_to_unchecked();
26        }
27        self
28    }
29}
30
31impl Function {
32    fn checked_to_unchecked(&mut self) {
33        let mut value_max_num_bits = HashMap::<ValueId, u32>::default();
34
35        self.simple_optimization(|context| {
36            let instruction = context.instruction();
37            let Instruction::Binary(binary) = instruction else {
38                return;
39            };
40            let lhs = binary.lhs;
41            let rhs = binary.rhs;
42
43            let lhs_type = context.dfg.type_of_value(lhs).unwrap_numeric();
44            let NumericType::Unsigned { .. } = lhs_type else {
45                return;
46            };
47
48            let dfg = &context.dfg;
49
50            let unchecked = match binary.operator {
51                BinaryOp::Add { unchecked: false } => {
52                    let bit_size = dfg.type_of_value(lhs).bit_size();
53                    let max_lhs_bits = get_max_num_bits(dfg, lhs, &mut value_max_num_bits);
54                    let max_rhs_bits = get_max_num_bits(dfg, rhs, &mut value_max_num_bits);
55
56                    // 1. If both lhs and rhs have less max bits than the result it means their
57                    //    value is at most `2^(n-1) - 1`, assuming `n = bit_size`. Adding those
58                    //    we get `2^(n-1) - 1 + 2^(n-1) - 1`, so `2*(2^(n-1)) - 2`,
59                    //    so `2^n - 2` which fits in `0..2^n`.
60                    // In that case, `lhs` and `rhs` have both been casted up from smaller types and so cannot overflow.
61                    max_lhs_bits < bit_size && max_rhs_bits < bit_size
62                }
63                BinaryOp::Sub { unchecked: false } => {
64                    let Some(lhs_const) = dfg.get_numeric_constant(lhs) else {
65                        return;
66                    };
67
68                    let max_rhs_bits = get_max_num_bits(dfg, rhs, &mut value_max_num_bits);
69                    let max_rhs =
70                        if max_rhs_bits == 128 { u128::MAX } else { (1 << max_rhs_bits) - 1 };
71
72                    // 1. `lhs` is a fixed constant and `rhs` is restricted such that `lhs - rhs > 0`
73                    //    Note strict inequality as `rhs > lhs` while `lhs_bits == max_rhs_bits` is possible.
74                    // 2. `lhs` is the maximum value for the maximum bitsize of `rhs`.
75                    //    For example: `lhs` is 1 and `rhs` max bitsize is 1, so at most it's `1 - 1` which cannot overflow.
76                    //    Another example: `lhs` is 255 and `rhs` max bitsize is 8, so at most it's `255 - 255` which cannot overflow, etc.
77                    lhs_const >= max_rhs.into()
78                }
79                BinaryOp::Mul { unchecked: false } => {
80                    let bit_size = dfg.type_of_value(lhs).bit_size();
81                    let max_lhs_bits = get_max_num_bits(dfg, lhs, &mut value_max_num_bits);
82                    let max_rhs_bits = get_max_num_bits(dfg, rhs, &mut value_max_num_bits);
83
84                    // 1. Bool multiplication cannot overflow
85                    // 2. `2^max_lhs_bits * 2^max_rhs_bits` is `2^(max_lhs_bits + max_rhs_bits)` so if that sum is
86                    //    less than or equal to the bit size of the result then it cannot overflow.
87                    // 3. lhs was upcasted from a boolean
88                    // 4. rhs was upcasted from a boolean
89                    // So either performing boolean multiplication (which cannot overflow),
90                    // or `lhs` and `rhs` have both been casted up from smaller types and cannot overflow.
91                    max_lhs_bits + max_rhs_bits <= bit_size
92                        || max_lhs_bits == 1
93                        || max_rhs_bits == 1
94                }
95                _ => false,
96            };
97            if unchecked {
98                let operator = binary.operator.into_unchecked();
99                context.replace_current_instruction_with(Instruction::Binary(Binary {
100                    lhs: binary.lhs,
101                    rhs: binary.rhs,
102                    operator,
103                }));
104            }
105        });
106    }
107}

109/// The logic here is almost the same as [`DataFlowGraph::get_value_max_num_bits`] except that
110/// - it takes into account that the bitsize of multiplying two bools is 1
111/// - it recurses by memoizing the results in `value_max_num_bits`
112fn get_max_num_bits(
113    dfg: &DataFlowGraph,
114    value: ValueId,
115    value_max_num_bits: &mut HashMap<ValueId, u32>,
116) -> u32 {
117    if let Some(bits) = value_max_num_bits.get(&value) {
118        return *bits;
119    }
120
121    let value_bit_size = dfg.type_of_value(value).bit_size();
122
123    let bits = match dfg[value] {
124        Value::Instruction { instruction, .. } => {
125            match dfg[instruction] {
126                Instruction::Cast(original_value, _) => {
127                    let original_bit_size =
128                        get_max_num_bits(dfg, original_value, value_max_num_bits);
129                    // We might have cast e.g. `u1` to `u8` to be able to do arithmetic,
130                    // in which case we want to recover the original smaller bit size;
131                    // OTOH if we cast down, then we don't need the higher original size.
132                    value_bit_size.min(original_bit_size)
133                }
134                Instruction::Binary(Binary { lhs, operator: BinaryOp::Mul { .. }, rhs })
135                    if get_max_num_bits(dfg, lhs, value_max_num_bits) == 1
136                        && get_max_num_bits(dfg, rhs, value_max_num_bits) == 1 =>
137                {
138                    // When multiplying two values, if their bitsize is 1 then the result's bitsize will be 1 too
139                    1
140                }
141                Instruction::Truncate { value, bit_size, .. } => {
142                    let value_bit_size =
143                        value_bit_size.min(get_max_num_bits(dfg, value, value_max_num_bits));
144                    value_bit_size.min(bit_size)
145                }
146                _ => value_bit_size,
147            }
148        }
149        Value::NumericConstant { constant, .. } => constant.num_bits(),
150        _ => value_bit_size,
151    };
152
153    assert!(bits <= value_bit_size);
154    value_max_num_bits.insert(value, bits);
155
156    bits
157}

166    #[test]
167    fn checked_to_unchecked_when_casting_two_u16_to_u32_then_adding() {
168        let src = "
169        acir(inline) fn main f0 {
170          b0(v0: u16, v1: u16):
171            v2 = cast v0 as u32
172            v3 = cast v1 as u32
173            v4 = add v2, v3
174            return v4
175        }
176        ";
177        let ssa = Ssa::from_str(src).unwrap();
178        let ssa = ssa.checked_to_unchecked();
179        assert_ssa_snapshot!(ssa, @r"
180        acir(inline) fn main f0 {
181          b0(v0: u16, v1: u16):
182            v2 = cast v0 as u32
183            v3 = cast v1 as u32
184            v4 = unchecked_add v2, v3
185            return v4
186        }
187        ");
188    }
