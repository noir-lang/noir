1//! A simple SSA pass to find any calls to `Intrinsic::AsVector` and replacing any references to the length of the
2//! resulting vector with the length of the array from which it was generated.
3//!
4//! This allows the length of a vector generated from an array to be used in locations where a constant value is
5//! necessary when the value of the array is unknown.
6//!
7//! Note that this pass must be placed before loop unrolling to be useful.
8
9use crate::ssa::{
10    ir::{
11        function::Function,
12        instruction::{Instruction, Intrinsic},
13        types::{NumericType, Type},
14    },
15    ssa_gen::Ssa,
16};
17
18impl Ssa {
19    /// Finds any calls to `Intrinsic::AsVector` and replaces any references to the length of the
20    /// resulting vector with the length of the array from which it was generated.
21    #[expect(clippy::wrong_self_convention)]
22    #[tracing::instrument(level = "trace", skip(self))]
23    pub(crate) fn as_vector_optimization(mut self) -> Self {
24        for func in self.functions.values_mut() {
25            func.as_vector_optimization();
26        }
27        self
28    }
29}
30
31impl Function {
32    /// Finds any calls to `Intrinsic::AsVector` and replaces any references to the length of the
33    /// resulting vector with the length of the array from which it was generated.
34    pub(crate) fn as_vector_optimization(&mut self) {
35        // If `as_vector` isn't called in this function there's nothing to do
36        let Some(as_vector) = self.dfg.get_intrinsic(Intrinsic::AsVector).copied() else {
37            return;
38        };
39
40        self.simple_optimization(|context| {
41            let instruction_id = context.instruction_id;
42            let instruction = context.instruction();
43
44            let (target_func, arguments) = match &instruction {
45                Instruction::Call { func, arguments } => (func, arguments),
46                _ => return,
47            };
48
49            if *target_func != as_vector {
50                return;
51            }
52
53            let first_argument =
54                arguments.first().expect("AsVector should always have one argument");
55            let array_typ = context.dfg.type_of_value(*first_argument);
56            let Type::Array(_, length) = array_typ else {
57                unreachable!("AsVector called with non-array {}", array_typ);
58            };
59
60            let [original_vector_length, _] = context.dfg.instruction_result(instruction_id);
61            let known_length =
62                context.dfg.make_constant(length.0.into(), NumericType::length_type());
63            context.replace_value(original_vector_length, known_length);
64        });
65    }
66}

68#[cfg(test)]
69mod tests {
70    use crate::assert_ssa_snapshot;
71
72    use super::Ssa;
73
74    #[test]
75    fn as_vector_length_optimization() {
76        // In this code we expect `return v2` to be replaced with `return u32 3` because
77        // that's the length of the v0 array.
78        let src = "
79        acir(inline) fn main f0 {
80          b0(v0: [Field; 3]):
81            v2, v3 = call as_vector(v0) -> (u32, [Field])
82            return v2
83        }
84        ";
85        let ssa = Ssa::from_str(src).unwrap();
86
87        let ssa = ssa.as_vector_optimization();
88        assert_ssa_snapshot!(ssa, @r"
89        acir(inline) fn main f0 {
90          b0(v0: [Field; 3]):
91            v2, v3 = call as_vector(v0) -> (u32, [Field])
92            return u32 3
93        }
94        ");
95    }
96
97    #[test]
98    fn as_vector_length_multiple_different_arrays() {
99        let src = "
100        acir(inline) fn main f0 {
101          b0(v0: [Field; 3], v1: [Field; 5]):
102            v3, v4 = call as_vector(v0) -> (u32, [Field])
103            v5, v6 = call as_vector(v1) -> (u32, [Field])
104            return v3, v5
105        }
106        ";
107        let ssa = Ssa::from_str(src).unwrap();
108        let ssa = ssa.as_vector_optimization();
109        assert_ssa_snapshot!(ssa, @r"
110        acir(inline) fn main f0 {
111          b0(v0: [Field; 3], v1: [Field; 5]):
112            v3, v4 = call as_vector(v0) -> (u32, [Field])
113            v5, v6 = call as_vector(v1) -> (u32, [Field])
114            return u32 3, u32 5
115        }
116        ");
117    }
