1//! Redundant block parameter elimination.
2//!
3//! After `mem2reg` promotes memory operations to block parameters, many of those
4//! parameters become redundant: every predecessor passes the same value, or the
5//! parameter is only fed by itself (loop back-edge). This pass detects and
6//! removes such parameters using a fixed-point dataflow analysis, closely
7//! following Cranelift's [`remove_constant_phis`] algorithm.
8//!
9//! [`remove_constant_phis`]: https://docs.wasmtime.dev/api/src/cranelift_codegen/remove_constant_phis.rs.html
10//!
11//! ## Algorithm (three phases)
12//!
13//! **Phase 1 — Summarize.**
14//! For each non-entry block, record its parameters and the arguments each
15//! predecessor passes via `Jmp` / `JmpIf` terminators.
16//!
17//! **Phase 2 — Dataflow.**
18//! Iterate in reverse post-order until a fixed point. Each block parameter
19//! carries an [`AbstractValue`]:
20//!
21//! | Variant    | Meaning                                  |
22//! |------------|------------------------------------------|
23//! | `None`     | No value observed yet (lattice bottom).   |
24//! | `One(v)`   | Exactly one distinct value flows here.    |
25//! | `Many`     | Multiple distinct values (lattice top).   |
26//!
27//! The join rule is: `None ⊔ x = x`, `One(v) ⊔ One(v) = One(v)`,
28//! `One(v) ⊔ One(w) = Many`, `Many ⊔ _ = Many`.
29//!
30//! Self-references (loop back-edges passing a param to itself) and transitive
31//! chains are handled implicitly by the lattice: a block parameter argument is
32//! resolved to its current abstract value before joining, so self-references
33//! contribute `x ⊔ x = x` (a no-op) and chains converge over iterations.
34//!
35//! **Phase 3 — Transform.**
36//! For every parameter that resolved to `One(v)`, replace all uses of the
37//! parameter with `v`, remove the parameter from the block, and strip the
38//! corresponding argument from all predecessor terminators.
39//!
40//! The entry block is never modified (its parameters are the function ABI).
41//!
42//! ## Pipeline placement
43//!
44//! Chained after every `mem2reg` call in the pipeline. `mem2reg` is likely to generate
45//! redundant parameters when promoting memory variables.

73/// Lattice element for the abstract value flowing to a block parameter.
74#[derive(Debug, Clone, Copy, PartialEq, Eq)]
75enum AbstractValue {
76    /// No value observed yet (bottom of the lattice).
77    None,
78    /// Exactly one distinct value flows to this parameter.
79    One(ValueId),
80    /// Two or more distinct values flow here (top — not redundant).
81    Many,
82}
83
84impl AbstractValue {
85    /// Lattice join: combine two abstract values.
86    fn join(self, other: AbstractValue) -> Self {
87        match (self, other) {
88            (AbstractValue::None, other) | (other, AbstractValue::None) => other,
89            (AbstractValue::Many, _) | (_, AbstractValue::Many) => AbstractValue::Many,
90            (AbstractValue::One(a), AbstractValue::One(b)) => {
91                if a == b {
92                    AbstractValue::One(a)
93                } else {
94                    AbstractValue::Many
95                }
96            }
97        }
98    }
99}

101/// Summary of a single non-entry block: its parameters and the arguments
102/// each predecessor passes (one `Vec<ValueId>` per incoming edge).
103struct BlockSummary {
104    params: Vec<ValueId>,
105    edges: Vec<Vec<ValueId>>,
106}
107
108impl Function {
109    pub(crate) fn remove_redundant_params(&mut self) {
110        let cfg = ControlFlowGraph::with_function(self);
111        let rpo = PostOrder::with_cfg(&cfg).into_vec_reverse();
112
113        let summaries = self.summarize_block_params(&cfg, &rpo);
114        if summaries.is_empty() {
115            return;
116        }
117
118        let lattice = Self::solve_dataflow(&summaries, &rpo);
119
120        self.apply_transforms(&cfg, &rpo, &summaries, &lattice);
121    }
122
123    /// Phase 1: Summarize — collect block params and predecessor arguments.
124    fn summarize_block_params(
125        &self,
126        cfg: &ControlFlowGraph,
127        rpo: &[BasicBlockId],
128    ) -> HashMap<BasicBlockId, BlockSummary> {
129        let mut summaries: HashMap<BasicBlockId, BlockSummary> = HashMap::default();
130
131        for &block in rpo {
132            if block == self.entry_block() {
133                continue;
134            }
135            let params = self.dfg.block_parameters(block).to_vec();
136            if params.is_empty() {
137                continue;
138            }
139
140            let mut edges = Vec::new();
141            for pred in cfg.predecessors(block) {
142                let terminator = self.dfg[pred].unwrap_terminator();
143                collect_edges_for_target(terminator, block, &mut edges);
144            }
145
146            summaries.insert(block, BlockSummary { params, edges });
147        }
148
149        summaries
150    }
151
152    /// Phase 2: Dataflow — iterate in RPO until fixed point.
153    fn solve_dataflow(
154        summaries: &HashMap<BasicBlockId, BlockSummary>,
155        rpo: &[BasicBlockId],
156    ) -> HashMap<ValueId, AbstractValue> {
157        let mut lattice: HashMap<ValueId, AbstractValue> = HashMap::default();
158
159        for summary in summaries.values() {
160            for &param in &summary.params {
161                lattice.insert(param, AbstractValue::None);
162            }
163        }
164
165        let mut changed = true;
166        while changed {
167            changed = false;
168
169            for &block in rpo {
170                let Some(summary) = summaries.get(&block) else {
171                    continue;
172                };
173
174                for edge in &summary.edges {
175                    for (&param, &arg) in summary.params.iter().zip_eq(edge) {
176                        let current = lattice[&param];
177                        if current == AbstractValue::Many {
178                            continue;
179                        }
180
181                        // Resolve the argument: if it's a block parameter in the
182                        // lattice, use its abstract value; otherwise it's a "Group B"
183                        // value (instruction result / entry param / constant) → One(arg).
184                        let arg_val = lattice.get(&arg).copied().unwrap_or(AbstractValue::One(arg));
185                        let new_val = current.join(arg_val);
186
187                        if new_val != current {
188                            lattice.insert(param, new_val);
189                            changed = true;
190                        }
191                    }
192                }
193            }
194        }
195
196        lattice
197    }
198
199    /// Phase 3: Transform — replace redundant params and prune them from blocks
200    /// and predecessor terminators.
201    fn apply_transforms(
202        &mut self,
203        cfg: &ControlFlowGraph,
204        rpo: &[BasicBlockId],
205        summaries: &HashMap<BasicBlockId, BlockSummary>,
206        lattice: &HashMap<ValueId, AbstractValue>,
207    ) {
208        let mut mapping = ValueMapping::default();
209
210        for summary in summaries.values() {
211            for &param in &summary.params {
212                if let AbstractValue::One(value) = lattice[&param] {
213                    mapping.insert(param, value);
214                }
215            }
216        }
217
218        if mapping.is_empty() {
219            return;
220        }
221
222        // Remove redundant parameters from blocks and predecessor terminators.
223        for (&block, summary) in summaries {
224            let keep_list: Vec<bool> = summary
225                .params
226                .iter()
227                .map(|p| !matches!(lattice[p], AbstractValue::One(_)))
228                .collect();
229
230            // Skip blocks where nothing is removed.
231            if keep_list.iter().all(|&k| k) {
232                continue;
233            }
234
235            // Filter the block's parameter list.
236            let old_params = self.dfg[block].take_parameters();
237            let new_params: Vec<ValueId> = old_params
238                .into_iter()
239                .zip(&keep_list)
240                .filter_map(|(p, &k)| k.then_some(p))
241                .collect();
242            self.dfg[block].set_parameters(new_params);
243
244            // Strip corresponding arguments from predecessor terminators.
245            for pred in cfg.predecessors(block) {
246                remove_terminator_args(&mut self.dfg[pred], block, &keep_list);
247            }
248        }
249
250        // Apply value replacements across all reachable blocks.
251        for &block in rpo {
252            self.dfg.replace_values_in_block(block, &mapping);
253        }
254    }
255}

257/// Collect all edges from a terminator to `target`, appending to `edges`.
258///
259/// A `JmpIf` where both branches target the same block produces two edges.
260fn collect_edges_for_target(
261    terminator: &TerminatorInstruction,
262    target: BasicBlockId,
263    edges: &mut Vec<Vec<ValueId>>,
264) {
265    match terminator {
266        TerminatorInstruction::Jmp { destination, arguments, .. } => {
267            debug_assert_eq!(*destination, target);
268            edges.push(arguments.clone());
269        }
270        TerminatorInstruction::JmpIf {
271            then_destination,
272            then_arguments,
273            else_destination,
274            else_arguments,
275            ..
276        } => {
277            if *then_destination == target {
278                edges.push(then_arguments.clone());
279            }
280            if *else_destination == target {
281                edges.push(else_arguments.clone());
282            }
283        }
284        _ => unreachable!("ICE: Return/Unreachable cannot be a predecessor"),
285    }
286}

288/// Remove arguments at positions where `keep_list[i]` is false from the
289/// terminator of `pred_block` that targets `target`.
290fn remove_terminator_args(
291    pred_block: &mut crate::ssa::ir::basic_block::BasicBlock,
292    target: BasicBlockId,
293    keep_list: &[bool],
294) {
295    let remove_args = |args: &mut Vec<ValueId>| {
296        let mut i = 0;
297        args.retain(|_| {
298            let keep = keep_list[i];
299            i += 1;
300            keep
301        });
302    };
303
304    let terminator = pred_block.unwrap_terminator_mut();
305    match terminator {
306        TerminatorInstruction::Jmp { arguments, .. } => {
307            remove_args(arguments);
308        }
309        TerminatorInstruction::JmpIf {
310            then_destination,
311            then_arguments,
312            else_destination,
313            else_arguments,
314            ..
315        } => {
316            if *then_destination == target {
317                remove_args(then_arguments);
318            }
319            if *else_destination == target {
320                remove_args(else_arguments);
321            }
322        }
323        _ => unreachable!("ICE: Return/Unreachable cannot be a predecessor"),
324    }
