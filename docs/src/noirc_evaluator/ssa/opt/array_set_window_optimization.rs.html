1//! Replaces `array_set` instructions with `make_array` instructions inside "conditional
2//! windows" when the result of the `array_set` is only used within that same "conditional window"
3//! and eventually in an `IfElse` instruction under the predicate associated with the conditional
4//! window.
5//!
6//! A "conditional window" is the sequence of instructions between
7//! `enable_side_effects v_cond` and the matching `enable_side_effects u1 1`.
8//!
9//! The optimization is correct only within the window because `array_set` under a conditional
10//! predicate is a conditional operation: when the predicate is false the result equals the
11//! original array, not the modified one. If the result were used outside the window (where
12//! the predicate might be false) a plain `make_array` would produce the wrong value.
13//!
14//! When the result *is* exclusive to the window the replacement is safe, and it is
15//! beneficial: it exposes the constant set-value directly as a `make_array` element so
16//! that [`super::remove_if_else`]'s `ValueMerger` can short-circuit the conditional multiply for
17//! that element rather than emitting a conditional `array_get`.
18//!
19//! Example – this SSA:
20//!
21//! ```ssa
22//! enable_side_effects v_cond
23//! v_set = array_set v_arr, index u32 2, value Field 99
24//! enable_side_effects u1 1
25//! v_out = if v_cond then v_set else (if v_not_cond) v_arr
26//! ```
27//!
28//! becomes:
29//!
30//! ```ssa
31//! enable_side_effects v_cond
32//! v0 = array_get v_arr, index u32 0 -> Field
33//! v1 = array_get v_arr, index u32 1 -> Field
34//! v_set = make_array [v0, v1, Field 99] : [Field; 3]
35//! enable_side_effects u1 1
36//! v_out = if v_cond then v_set else (if v_not_cond) v_arr
37//! ```
38//!
39//! Because `v_set` is only used within the "conditional window", and then just as the input
40//! of an `if-then-else` instruction, the `array_set` can be executed unconditionally because
41//! the [`super::remove_if_else`] pass that comes after this pass will merge `v_set` with `v_arr` make sure
42//! to only use the values from `v_set` when `v_cond` is true.
43//!
44//! Note 1: this optimization only applies to `array_set` instructions operating on arrays, not
45//! vectors, because for arrays we can statically determine the number of elements, while for
46//! vectors we can only do that if the `array_set` value operates on a `make_array`. This could
47//! be done, but it's left as a follow-up optimization: <https://github.com/noir-lang/noir/issues/11810>
48//!
49//! Note 2: because the optimization expands to multiple `array_get` and a `make_array` instruction,
50//! for large arrays this might result in too many `array_get` instructions that slow down SSA optimization.
51//! For this reason, only small arrays (up to `MAX_ARRAY_SEMI_FLATTENED_LENGTH` elements) are optimized by this pass.
52use std::collections::{HashMap, HashSet};

73const MAX_ARRAY_SEMI_FLATTENED_LENGTH: u32 = 64;

75impl Ssa {
76    /// Replaces qualifying `array_set` instructions with `make_array` instructions.
77    /// See the [`array_set_to_make_array`][self] module for more information.
78    #[tracing::instrument(level = "trace", skip(self))]
79    pub(crate) fn array_set_window_optimization(mut self) -> Self {
80        for func in self.functions.values_mut() {
81            func.array_set_window_optimization();
82        }
83        self
84    }
85}

87impl Function {
88    fn array_set_window_optimization(&mut self) {
89        // This optimization only applies to ACIR functions (same precondition as
90        // `remove_if_else` which this pass is designed to feed into).
91        if self.runtime().is_brillig() {
92            return;
93        }
94
95        #[cfg(debug_assertions)]
96        array_set_to_make_array_pre_check(self);
97
98        let block_id = self.entry_block();
99
100        // Pass 1: find which array_set instructions are eligible.
101        let candidates = find_candidates(&self.dfg, block_id);
102        if candidates.is_empty() {
103            return;
104        }
105
106        // Pass 2: replace each candidate array_set with array_gets + make_array.
107        self.simple_optimization(|context| {
108            let inst_id = context.instruction_id;
109            if !candidates.contains(&inst_id) {
110                return;
111            }
112
113            let Instruction::ArraySet { array, index, value, .. } = *context.instruction() else {
114                unreachable!("candidate must be an ArraySet instruction");
115            };
116
117            let Some(const_index) =
118                context.dfg.get_numeric_constant(index).and_then(|v| v.try_to_u32())
119            else {
120                unreachable!("candidate ArraySet index must be a constant u32");
121            };
122
123            let Type::Array(ref element_types, len) = context.dfg.type_of_value(array) else {
124                // In theory this optimization could also run on vectors with a known length, but currently
125                // it doesn't. See https://github.com/noir-lang/noir/issues/11810
126                unreachable!("candidate ArraySet array must be of array type");
127            };
128
129            let array_constant = context.dfg.get_array_constant(array);
130
131            let element_types = element_types.clone();
132            let element_count = ElementTypesLength(element_types.len() as u32);
133            let total_elements = len * element_count;
134
135            // Remove the array_set; we will emit replacement instructions instead.
136            context.remove_current_instruction();
137            let [old_result] = context.dfg.instruction_result(inst_id);
138
139            // Build the element list for the make_array.
140            let mut elements = Vector::new();
141            for semi_flattened_index in 0..total_elements.0 {
142                if semi_flattened_index == const_index {
143                    elements.push_back(value);
144                } else if let Some((array_element, _)) = &array_constant {
145                    elements.push_back(array_element[semi_flattened_index as usize]);
146                } else {
147                    let element_index = (semi_flattened_index % element_count.0) as usize;
148                    let element_type = element_types[element_index].clone();
149                    let index = context.dfg.make_constant(
150                        FieldElement::from(u128::from(semi_flattened_index)),
151                        NumericType::length_type(),
152                    );
153                    let get = Instruction::ArrayGet { array, index };
154                    let get_result =
155                        context.insert_instruction(get, Some(vec![element_type])).first();
156                    elements.push_back(get_result);
157                }
158            }
159
160            let typ = Type::Array(element_types, len);
161            let make_array = Instruction::MakeArray { elements, typ: typ.clone() };
162            let new_result = context.insert_instruction(make_array, Some(vec![typ])).first();
163            context.replace_value(old_result, new_result);
164        });
165    }
166}

174#[derive(Debug, Copy, Clone, PartialEq, Eq)]
175struct ConditionalWindowId(usize);
176
177#[derive(Debug, Copy, Clone)]
178struct ConditionalWindow {
179    id: ConditionalWindowId,
180    predicate: ValueId,
181}
182
183#[derive(Debug)]
184struct TrackedValue {
185    window: ConditionalWindow,
186    // The set of original array_set candidate results that this value transitively depends on.
187    // If any of these candidates is disqualified, this value must be disqualified too.
188    dependencies: im::HashSet<ValueId>,
189}
190
191/// Returns the set of `InstructionId`s for `array_set` instructions that are safe
192/// to replace with `make_array + array_get` — i.e. those whose result (and any value
193/// transitively derived from it within the same window) is used only within that window
194/// or in an `IfElse` whose `then_condition` matches the window's predicate.
195///
196/// A "derived value" is the result of any instruction inside the window that takes a
197/// candidate (or another derived value) as an input — e.g. a `call` or another
198/// `array_set`. If a derived value escapes the window, the candidates it was derived
199/// from are disqualified: replacing them with unconditional `make_array`s would lead to
200/// incorrect results.
201fn find_candidates(dfg: &DataFlowGraph, block_id: BasicBlockId) -> HashSet<InstructionId> {
202    // Maps array_set candidate results → instruction id.
203    let mut candidates: HashMap<ValueId, InstructionId> = HashMap::new();
204    // Maps every "tracked" value (a candidate or a value derived from candidates within the same window).
205    let mut tracked: HashMap<ValueId, TrackedValue> = HashMap::new();
206    // Candidates that were found in an `IfElse` instruction under their associated predicate.
207    let mut if_else_candidates = HashSet::<ValueId>::new();
208
209    let mut current_window: Option<ConditionalWindow> = None;
210    let mut window_counter: ConditionalWindowId = ConditionalWindowId(0);
211
212    // Remember all windows by their predicate. In case a new window shows up with a predicate we've
213    // already seen in a previous window, we can consider it to be the same window.
214    let mut windows_by_predicate = HashMap::<ValueId, ConditionalWindow>::new();
215
216    let instructions = dfg[block_id].instructions();
217
218    for &instruction_id in instructions {
219        let instruction = &dfg[instruction_id];
220
221        // A window gets closed or opened. This check must be done before checking for escaped
222        // values because a tracked value used in an `enable_side_effects` instruction should be
223        // considered as escaping the previous window.
224        if let Instruction::EnableSideEffectsIf { condition } = instruction {
225            let is_unconditional = dfg.get_numeric_constant(*condition).is_some_and(|v| v.is_one());
226            if is_unconditional {
227                current_window = None;
228            } else if let Some(existing_window) = windows_by_predicate.get(condition) {
229                current_window = Some(*existing_window);
230            } else {
231                window_counter = ConditionalWindowId(window_counter.0 + 1);
232                let new_window = ConditionalWindow { id: window_counter, predicate: *condition };
233                current_window = Some(new_window);
234                windows_by_predicate.insert(*condition, new_window);
235            }
236        }
237
238        // Check whether any input to this instruction is a tracked value being used in
239        // a context that escapes the conditional window.
240        instruction.for_each_value(|value| {
241            if let Some(tracked_value) = tracked.get(&value) {
242                // A `store` instruction writes the value into memory from which it can be
243                // loaded after the window closes — treat this as an escape even when the
244                // store is inside the same window.
245                let escapes_via_store = matches!(
246                    instruction,
247                    Instruction::Store { value: stored, .. } if *stored == value
248                );
249                // A call with reference arguments might store the value in one of those references
250                let escapes_via_call_arguments = matches!(
251                    instruction,
252                    Instruction::Call { arguments, .. } if arguments.iter().any(|&arg| dfg.type_of_value(arg).contains_reference()),
253                );
254
255                // Determine whether this value stays within the window it was defined in.
256                let stays_within_window = if escapes_via_store || escapes_via_call_arguments {
257                    false
258                } else {
259                    // Otherwise, if the pair of if "condition/value" matches the tracked value and its window,
260                    // then the value might escape the window but through an `if-else` that's going to
261                    // be merged, in which case we consider this as not escaping the window.
262                    match instruction {
263                        // Check the "then-condition/then-value" case.
264                        Instruction::IfElse { then_condition, then_value, .. }
265                            if *then_value == value && *then_condition == tracked_value.window.predicate =>
266                        {
267                            if_else_candidates.insert(value);
268                            if_else_candidates.extend(tracked_value.dependencies.clone());
269                            true
270                        }
271                        // Check the "else-condition/else-value" pair
272                        Instruction::IfElse { else_condition, else_value, .. }
273                            if *else_value == value && *else_condition == tracked_value.window.predicate =>
274                        {
275                            if_else_candidates.insert(value);
276                            if_else_candidates.extend(tracked_value.dependencies.clone());
277                            true
278                        }
279                        _ => {
280                            // Otherwise: safe as long as we are inside the tracked window.
281                            current_window
282                                .is_some_and(|window| window.id == tracked_value.window.id)
283                        }
284                    }
285                };
286                if !stays_within_window {
287                    let tracked_value_dependencies = tracked_value.dependencies.clone();
288
289                    candidates.remove(&value);
290                    tracked.remove(&value);
291                    for dependency in tracked_value_dependencies {
292                        candidates.remove(&dependency);
293                        tracked.remove(&dependency);
294                    }
295                }
296            }
297        });
298
299        match instruction {
300            Instruction::EnableSideEffectsIf { .. } => {
301                // This was already handled above
302            }
303            Instruction::ArraySet { array, index, value, mutable: false } => {
304                if let Some(window) = current_window {
305                    let [result] = dfg.instruction_result(instruction_id);
306
307                    // array_set with a constant in-bound index on a small array
308                    if let (Type::Array(elements, len), Some(index)) =
309                        (dfg.type_of_value(*array), dfg.get_numeric_constant(*index))
310                    {
311                        let elements_length = ElementTypesLength(assert_u32(elements.len()));
312                        let semi_flattened_length = len * elements_length;
313                        if index.to_u128() < u128::from(semi_flattened_length.0)
314                            && semi_flattened_length.0 <= MAX_ARRAY_SEMI_FLATTENED_LENGTH
315                        {
316                            candidates.insert(result, instruction_id);
317                        }
318                    }
319
320                    let mut dependencies = im::HashSet::new();
321                    for value in [array, value] {
322                        if let Some(tracked_value) = tracked.get(value) {
323                            dependencies.insert(*value);
324                            dependencies = dependencies.union(tracked_value.dependencies.clone());
325                        }
326                    }
327                    tracked.insert(result, TrackedValue { window, dependencies });
328                }
329            }
330            Instruction::IfElse { .. } => {
331                // We already disqualified candidates used in `IfElse` that escape the current
332                // window, so there's no need to keep track of `IfElse` dependencies.
333            }
334            _ => {
335                // For any other instruction inside a window: if it consumes tracked values,
336                // its results are "derived" and must be tracked with the union of their deps.
337                if let Some(current_window) = current_window {
338                    let results = dfg.instruction_results(instruction_id);
339                    if !results.is_empty() {
340                        let mut dependencies = im::HashSet::new();
341                        instruction.for_each_value(|value| {
342                            if let Some(tracked_value) = tracked.get(&value) {
343                                let tracked_value_dependencies = tracked_value.dependencies.clone();
344                                dependencies.insert(value);
345                                dependencies =
346                                    dependencies.clone().union(tracked_value_dependencies);
347                            }
348                        });
349                        if !dependencies.is_empty() {
350                            for &result in results {
351                                tracked.insert(
352                                    result,
353                                    TrackedValue {
354                                        window: current_window,
355                                        dependencies: dependencies.clone(),
356                                    },
357                                );
358                            }
359                        }
360                    }
361                }
362            }
363        }
364    }
365
366    // Also check the block terminator: if a tracked value is used there it escapes.
367    if let Some(terminator) = dfg[block_id].terminator() {
368        terminator.for_each_value(|value| {
369            if let Some(tracked_value) = tracked.get(&value) {
370                let tracked_value_dependencies = tracked_value.dependencies.clone();
371                candidates.remove(&value);
372                tracked.remove(&value);
373                for dependency in tracked_value_dependencies {
374                    candidates.remove(&dependency);
375                    tracked.remove(&dependency);
376                }
377            }
378        });
379    }
380
381    // Only keep candidates that were eventually used in an `IfElse` instruction
382    candidates
383        .into_iter()
384        .filter_map(|(value_id, instruction_id)| {
385            if_else_candidates.contains(&value_id).then_some(instruction_id)
386        })
387        .collect()
388}
