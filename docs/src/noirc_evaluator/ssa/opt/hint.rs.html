17    fn run_all_passes(ssa: Ssa) -> Result<Ssa, RuntimeError> {
18        let options = &SsaEvaluatorOptions {
19            ssa_logging: SsaLogging::None,
20            brillig_options: BrilligOptions::default(),
21            print_codegen_timings: false,
22            emit_ssa: None,
23            skip_underconstrained_check: true,
24            enable_brillig_constraints_check_lookback: false,
25            skip_brillig_constraints_check: true,
26            inliner_aggressiveness: 0,
27            constant_folding_max_iter: constant_folding::DEFAULT_MAX_ITER,
28            small_function_max_instruction: inlining::MAX_SIMPLE_FUNCTION_WEIGHT,
29            max_bytecode_increase_percent: None,
30            force_unroll_threshold: FORCE_UNROLL_THRESHOLD,
31            skip_passes: Default::default(),
32            ssa_logging_hide_unchanged: false,
33        };
34
35        let builder = SsaBuilder::from_ssa(
36            ssa,
37            options.ssa_logging.clone(),
38            options.ssa_logging_hide_unchanged,
39            false,
40            None,
41        );
42        Ok(builder.run_passes(&primary_passes(options))?.finish())
43    }

45    /// Test that the `std::hint::black_box` function prevents some of the optimizations.
46    #[test]
47    fn test_black_box_hint() {
48        // fn main(sum: u32) {
49        //     // This version simplifies into a single `constraint 50 == sum`
50        //     assert_eq(loop(5, 10), sum);
51        //     // This should preserve additions because `k` is opaque, as if it came from an input.
52        //     assert_eq(loop(5, std::hint::black_box(10)), sum);
53        // }
54        // fn loop(n: u32, k: u32) -> u32 {
55        //     let mut sum = 0;
56        //     for _ in 0..n {
57        //         sum = sum + k;
58        //     }
59        //     sum
60        // }
61
62        // Initial SSA:
63        let src = "
64          acir(inline) fn main f0 {
65            b0(v0: u32):
66              v4 = call f1(u32 5, u32 10) -> u32
67              v5 = eq v4, v0
68              constrain v4 == v0
69              v7 = call black_box(u32 10) -> u32
70              v9 = call f1(u32 5, v7) -> u32
71              v10 = eq v9, v0
72              constrain v9 == v0
73              return
74          }
75          acir(inline) fn loop f1 {
76            b0(v0: u32, v1: u32):
77              v3 = allocate -> &mut u32
78              store u32 0 at v3
79              jmp b1(u32 0)
80            b1(v2: u32):
81              v5 = lt v2, v0
82              jmpif v5 then: b3, else: b2
83            b3():
84              v7 = load v3 -> u32
85              v8 = add v7, v1
86              store v8 at v3
87              v10 = add v2, u32 1
88              jmp b1(v10)
89            b2():
90              v6 = load v3 -> u32
91              return v6
92          }
93        ";
94
95        let ssa = Ssa::from_str(src).unwrap();
96        let ssa = run_all_passes(ssa).unwrap();
97        assert_ssa_snapshot!(ssa, @r"
98        acir(inline) impure fn main f0 {
99          b0(v0: u32):
100            constrain u32 50 == v0
101            v4 = call black_box(u32 10) -> u32
102            v5 = add v4, v4
103            v6 = add v5, v4
104            v7 = add v6, v4
105            v8 = add v7, v4
106            constrain v8 == u32 50
107            return
108        }
109        ");
110    }
