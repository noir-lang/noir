23impl Function {
24    /// Performs a simple optimization according to the given callback.
25    ///
26    /// The function's reverse [post order][PostOrder] are traversed in turn, and instructions in those blocks
27    /// are then traversed in turn. For each one, `f` will be called with a context.
28    ///
29    /// The current instruction will be inserted at the end of the callback given to `mutate` unless
30    /// `remove_current_instruction` or `insert_current_instruction` are called.
31    ///
32    /// `insert_current_instruction` is useful if you need to insert new instructions after the current
33    /// one, so this can be done before the callback ends.
34    ///
35    /// `replace_value` can be used to replace a value with another one. This substitution will be
36    /// performed in all subsequent instructions.
37    pub(crate) fn simple_optimization<F>(&mut self, mut f: F)
38    where
39        F: FnMut(&mut SimpleOptimizationContext<'_, '_>),
40    {
41        self.simple_optimization_result(move |context| {
42            f(context);
43            Ok(())
44        })
45        .expect("`f` cannot error internally so this should be unreachable");
46    }
47
48    /// Performs a simple optimization according to the given callback, returning early if
49    /// an error occurred.
50    ///
51    /// The function's reverse [post order][PostOrder] are traversed in turn, and instructions in those blocks
52    /// are then traversed in turn. For each one, `f` will be called with a context.
53    ///
54    /// The current instruction will be inserted at the end of the callback given to `mutate` unless
55    /// `remove_current_instruction` or `insert_current_instruction` are called.
56    ///
57    /// `insert_current_instruction` is useful if you need to insert new instructions after the current
58    /// one, so this can be done before the callback ends.
59    ///
60    /// `replace_value` can be used to replace a value with another one. This substitution will be
61    /// performed in all subsequent instructions.
62    pub(crate) fn simple_optimization_result<F>(&mut self, mut f: F) -> RtResult<()>
63    where
64        F: FnMut(&mut SimpleOptimizationContext<'_, '_>) -> RtResult<()>,
65    {
66        let mut values_to_replace = ValueMapping::default();
67        let mut dirty_values = HashSet::<ValueId>::new();
68        let one = self.dfg.make_constant(FieldElement::from(1_u128), NumericType::bool());
69        let reverse_post_order = PostOrder::with_function(self).into_vec_reverse();
70        for block_id in reverse_post_order {
71            let mut enable_side_effects = one;
72
73            let instruction_ids = self.dfg[block_id].take_instructions();
74            self.dfg[block_id].instructions_mut().reserve(instruction_ids.len());
75            for instruction_id in &instruction_ids {
76                let instruction_id = *instruction_id;
77                let instruction = &mut self.dfg[instruction_id];
78                let orig_instruction_hash = rustc_hash::FxBuildHasher.hash_one(&instruction);
79                if !values_to_replace.is_empty() {
80                    instruction.replace_values(&values_to_replace);
81                }
82                if let Instruction::EnableSideEffectsIf { condition } = instruction {
83                    enable_side_effects = *condition;
84                }
85                let call_stack_id = self.dfg.get_instruction_call_stack_id(instruction_id);
86                let mut context = SimpleOptimizationContext {
87                    block_id,
88                    instruction_id,
89                    call_stack_id,
90                    dfg: &mut self.dfg,
91                    values_to_replace: &mut values_to_replace,
92                    insert_current_instruction_at_callback_end: true,
93                    enable_side_effects,
94                    orig_instruction_hash,
95                    dirty_values: &mut dirty_values,
96                };
97                f(&mut context)?;
98
99                if context.insert_current_instruction_at_callback_end {
100                    context.insert_current_instruction();
101                }
102            }
103
104            self.dfg.replace_values_in_block_terminator(block_id, &values_to_replace);
105        }
106
107        self.dfg.data_bus.replace_values(&values_to_replace);
108        Ok(())
109    }
110}

112pub(crate) struct SimpleOptimizationContext<'dfg, 'mapping> {
113    #[allow(unused)]
114    pub(crate) block_id: BasicBlockId,
115    pub(crate) instruction_id: InstructionId,
116    pub(crate) call_stack_id: CallStackId,
117    pub(crate) dfg: &'dfg mut DataFlowGraph,
118    pub(crate) enable_side_effects: ValueId,
119    values_to_replace: &'mapping mut ValueMapping,
120    insert_current_instruction_at_callback_end: bool,
121    orig_instruction_hash: u64,
122    dirty_values: &'mapping mut HashSet<ValueId>,
123}
124
125impl SimpleOptimizationContext<'_, '_> {
126    /// Returns the current instruction being visited.
127    ///
128    /// The instruction has already had its values updated with any replacements to be done.
129    pub(crate) fn instruction(&self) -> &Instruction {
130        &self.dfg[self.instruction_id]
131    }
132
133    /// Instructs this context to replace a value with another value. The value will be replaced
134    /// in all subsequent instructions.
135    pub(crate) fn replace_value(&mut self, from: ValueId, to: ValueId) {
136        self.values_to_replace.insert(from, to);
137    }
138
139    /// Check if the instruction has changed relative to its original contents,
140    /// e.g. because any of its values have been replaced.
141    fn has_instruction_changed(&self) -> bool {
142        // If the instruction changed, then there is a chance that we can (or have to)
143        // simplify it before we insert it back into the block.
144        let instruction_hash = rustc_hash::FxBuildHasher.hash_one(self.instruction());
145        self.orig_instruction_hash != instruction_hash
146    }
147
148    /// Instructs this context to insert the current instruction right away, as opposed
149    /// to doing this at the end of `mutate`'s block (unless `remove_current_instruction is called`).
150    ///
151    /// If the instruction or its values has changed relative to their original content,
152    /// we attempt to simplify the instruction before re-inserting it into the block.
153    pub(crate) fn insert_current_instruction(&mut self) {
154        // If the instruction changed, or if any of its values have changed, then there is a chance
155        // that we can (or have to) simplify it before we insert it back into the block.
156        let instruction_changed = self.has_instruction_changed();
157        let simplify = instruction_changed
158            || self.instruction().any_value(|value| self.dirty_values.contains(&value));
159
160        if simplify {
161            // Based on FunctionInserter::push_instruction_value.
162            let instruction = self.instruction().clone();
163            let results = self.dfg.instruction_results(self.instruction_id).to_vec();
164
165            let ctrl_typevars = instruction
166                .requires_ctrl_typevars()
167                .then(|| vecmap(&results, |result| self.dfg.type_of_value(*result).into_owned()));
168
169            let new_results = self.dfg.insert_instruction_and_results_if_simplified(
170                instruction,
171                self.block_id,
172                ctrl_typevars,
173                self.call_stack_id,
174                Some(self.instruction_id),
175            );
176            assert_eq!(results.len(), new_results.len());
177            for i in 0..results.len() {
178                if results[i] == new_results[i] && instruction_changed {
179                    // If the result didn't change, but the instruction itself did, we'd still like
180                    // to simplify instructions that depend on the unchanged result.
181                    // This for example can happen with a `v2 = make_array [v1]` that got turned
182                    // into `v2 = make_array [Field 0]`: `v2` didn't get a new result (it's not `v3`),
183                    // but an instruction that uses `v2` could get simplified now when it wasn't before
184                    // (an example is a call to `posiedon2_permutation(v2)`)
185                    self.dirty_values.insert(results[i]);
186                }
187
188                self.values_to_replace.insert(results[i], new_results[i]);
189            }
190        } else {
191            self.dfg[self.block_id].insert_instruction(self.instruction_id);
192        }
193
194        self.insert_current_instruction_at_callback_end = false;
195    }
196
197    /// Instructs this context to remove the current instruction from its block.
198    pub(crate) fn remove_current_instruction(&mut self) {
199        self.insert_current_instruction_at_callback_end = false;
200    }
201
202    /// Inserts an instruction in the current block right away.
203    pub(crate) fn insert_instruction(
204        &mut self,
205        instruction: Instruction,
206        ctrl_typevars: Option<Vec<Type>>,
207    ) -> InsertInstructionResult {
208        self.dfg.insert_instruction_and_results(
209            instruction,
210            self.block_id,
211            ctrl_typevars,
212            self.call_stack_id,
213        )
214    }
215
216    /// Inserts an instruction by id in the current block right away.
217    pub(crate) fn insert_instruction_by_id(&mut self, instruction_id: InstructionId) {
218        self.dfg[self.block_id].insert_instruction(instruction_id);
219    }
220
221    /// Replaces the current instruction with another one.
222    ///
223    /// This assumes no change in the number and type of results,
224    /// it simply reassigns the existing ID with a modified instruction.
225    pub(crate) fn replace_current_instruction_with(&mut self, instruction: Instruction) {
226        self.dfg[self.instruction_id] = instruction;
227    }
228}

235    fn optimizes_instruction_dependent_on_changed_make_array() {
236        // Here `v2` will be optimized to Field 4, to `v3` will be an `make_array` with all

246            v9 = make_array [v5, Field 10, Field 11, Field 12] : [Field; 4]

249        }
250        ";
251
252        // Replace the single array_get above with `Field 1`
