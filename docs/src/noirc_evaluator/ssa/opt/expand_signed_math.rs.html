1//! An SSA pass for ACIR functions that transforms "less than", "div" and "mod" operation on
2//! signed integers into equivalent sequences of operations that rely on unsigned integers.
3//!
4//! The purpose of this pass is to avoid ACIR having to handle signed integers "less than",
5//! "div" and "mod" operations (for simplicity), while also allowing further optimizations to
6//! be done during subsequent SSA passes on the expanded instructions.
7use acvm::FieldElement;
8
9use crate::ssa::{
10    ir::{
11        function::Function,
12        instruction::{Binary, BinaryOp, ConstrainError, Instruction},
13        types::NumericType,
14        value::ValueId,
15    },
16    ssa_gen::Ssa,
17};
18
19use super::simple_optimization::SimpleOptimizationContext;
20
21impl Ssa {
22    /// Expands signed "less than", "div" and "mod" operations in ACIR to be done using
23    /// unsigned operations.
24    ///
25    /// See [`expand_signed_math`][self] module for more information.
26    #[tracing::instrument(level = "trace", skip(self))]
27    pub(crate) fn expand_signed_math(mut self) -> Ssa {
28        for function in self.functions.values_mut() {
29            function.expand_signed_math();
30        }
31        self
32    }
33}
34
35impl Function {
36    /// The structure of this pass is simple:
37    /// Go through each block and re-insert all instructions, decomposing any signed
38    /// "less than", "div" and "mod" operations to be done using unsigned types, but only if this
39    /// is an ACIR function.
40    fn expand_signed_math(&mut self) {
41        if !self.dfg.runtime().is_acir() {
42            return;
43        }
44
45        self.simple_optimization(|context| {
46            let instruction_id = context.instruction_id;
47            let instruction = context.instruction();
48
49            // We only care about "less than"
50            let Instruction::Binary(Binary {
51                lhs,
52                rhs,
53                operator: operator @ (BinaryOp::Lt | BinaryOp::Div | BinaryOp::Mod),
54            }) = instruction
55            else {
56                return;
57            };
58
59            // ... and it must be a signed integer operation.
60            if !context.dfg.type_of_value(*lhs).is_signed() {
61                return;
62            }
63
64            let lhs = *lhs;
65            let rhs = *rhs;
66            let operator = *operator;
67
68            // We remove the current instruction, as we will need to replace it with multiple new instructions.
69            context.remove_current_instruction();
70
71            let [old_result] = context.dfg.instruction_result(instruction_id);
72
73            let mut expansion_context = Context { context };
74            let new_result = match operator {
75                BinaryOp::Lt => expansion_context.insert_lt(lhs, rhs),
76                BinaryOp::Div => expansion_context.insert_div(lhs, rhs),
77                BinaryOp::Mod => expansion_context.insert_mod(lhs, rhs),
78                _ => unreachable!("ICE: expand_signed_math called on non-lt/div/mod"),
79            };
80
81            context.replace_value(old_result, new_result);
82        });
83
84        #[cfg(debug_assertions)]
85        expand_signed_math_post_check(self);
86    }
87}

89struct Context<'m, 'dfg, 'mapping> {
90    context: &'m mut SimpleOptimizationContext<'dfg, 'mapping>,
91}
92
93impl Context<'_, '_, '_> {
94    fn insert_lt(&mut self, lhs: ValueId, rhs: ValueId) -> ValueId {
95        // First cast lhs and rhs to their unsigned equivalents
96        let bit_size = self.context.dfg.type_of_value(lhs).bit_size();
97        let unsigned_typ = NumericType::unsigned(bit_size);
98        let lhs_unsigned = self.insert_cast(lhs, unsigned_typ);
99        let rhs_unsigned = self.insert_cast(rhs, unsigned_typ);
100
101        // Check if lhs and rhs are positive or negative, respectively.
102        // Values greater than or equal to 2^(bit_size-1) are negative so dividing by that would
103        // give 0 (positive) or 1 (negative).
104        let first_negative_value = self.numeric_constant(1_u128 << (bit_size - 1), unsigned_typ);
105        let lhs_is_negative = self.insert_binary(lhs_unsigned, BinaryOp::Div, first_negative_value);
106        let lhs_is_negative = self.insert_cast(lhs_is_negative, NumericType::bool());
107        let rhs_is_negative = self.insert_binary(rhs_unsigned, BinaryOp::Div, first_negative_value);
108        let rhs_is_negative = self.insert_cast(rhs_is_negative, NumericType::bool());
109
110        // Do rhs and lhs have a different sign?
111        let different_sign = self.insert_binary(lhs_is_negative, BinaryOp::Xor, rhs_is_negative);
112
113        // Check lhs < rhs using their unsigned equivalents
114        let unsigned_lt = self.insert_binary(lhs_unsigned, BinaryOp::Lt, rhs_unsigned);
115
116        // It can be shown that the result is given by xor'ing the two results above:
117        // - if lhs and rhs have the same sign (different_sign is 0):
118        //   - if both are positive then the unsigned comparison is correct, xor'ing it with 0 gives
119        //     the same result
120        //   - if both are negative then the unsigned comparison is also correct, as, for example,
121        //     for i8, -128 i8 is Field 128 and -1 i8 is Field 255 and `-128 < -1` and `128 < 255`
122        // - if lhs and rhs have different signs (different_sign is 1):
123        //   - if lhs is positive and rhs is negative then, as fields, rhs will be greater, but
124        //     the result is the opposite (so xor'ing with 1 gives the correct result)
125        //   - if lhs is negative and rhs is positive then, as fields, lhs will be greater, but
126        //     the result is the opposite (so xor'ing with 1 gives the correct result)
127        self.insert_binary(different_sign, BinaryOp::Xor, unsigned_lt)
128    }
129
130    fn insert_div(&mut self, lhs: ValueId, rhs: ValueId) -> ValueId {
131        let is_division = true;
132        self.insert_div_or_mod(lhs, rhs, is_division)
133    }
134
135    fn insert_mod(&mut self, lhs: ValueId, rhs: ValueId) -> ValueId {
136        let is_division = false;
137        self.insert_div_or_mod(lhs, rhs, is_division)
138    }
139
140    fn insert_div_or_mod(&mut self, lhs: ValueId, rhs: ValueId, is_division: bool) -> ValueId {
141        // First cast lhs and rhs to their unsigned equivalents
142        let bit_size = self.context.dfg.type_of_value(lhs).bit_size();
143        let unsigned_typ = NumericType::unsigned(bit_size);
144        let lhs_unsigned = self.insert_cast(lhs, unsigned_typ);
145        let rhs_unsigned = self.insert_cast(rhs, unsigned_typ);
146
147        // There's one condition that could generate an overflow: dividing the minimum
148        // negative value by -1. For example dividing -128 i8 by -1 would give 128, but that
149        // does not fit i8. So the first thing we do is check for this case.
150        let min_negative_value = self.numeric_constant(1_u128 << (bit_size - 1), unsigned_typ);
151        let minus_one = self.numeric_constant((1_u128 << bit_size) - 1, unsigned_typ);
152        let lhs_is_min_negative_value =
153            self.insert_binary(lhs_unsigned, BinaryOp::Eq, min_negative_value);
154        let rhs_is_minus_one = self.insert_binary(rhs_unsigned, BinaryOp::Eq, minus_one);
155        let min_overflow = self.insert_binary(
156            lhs_is_min_negative_value,
157            BinaryOp::Mul { unchecked: true },
158            rhs_is_minus_one,
159        );
160
161        let zero = self.numeric_constant(0_u128, NumericType::bool());
162        let message = if is_division {
163            "Attempt to divide with overflow".to_string()
164        } else {
165            "Attempt to calculate the remainder with overflow".to_string()
166        };
167        self.insert_constrain(min_overflow, zero, Some(message.into()));
168
169        // What about checking that the divisor is not zero? We don't need to explicitly check
170        // this here because it'll be checked when doing the unsigned div/mod.
171
172        // Check if lhs and rhs are positive or negative, respectively.
173        // Values greater than or equal to 2^(bit_size-1) are negative so dividing by that would
174        // give 0 (positive) or 1 (negative).
175        let lhs_is_negative = self.insert_binary(lhs_unsigned, BinaryOp::Div, min_negative_value);
176        let rhs_is_negative = self.insert_binary(rhs_unsigned, BinaryOp::Div, min_negative_value);
177
178        // Here we compute the absolute values of lhs and rhs using their 2-complement
179        let lhs_absolute =
180            self.two_complement(lhs_unsigned, lhs_is_negative, unsigned_typ, bit_size);
181        let rhs_absolute =
182            self.two_complement(rhs_unsigned, rhs_is_negative, unsigned_typ, bit_size);
183
184        // We then perform the division (or modulo) using the absolute values
185        let operator = if is_division { BinaryOp::Div } else { BinaryOp::Mod };
186        let absolute_result = self.insert_binary(lhs_absolute, operator, rhs_absolute);
187
188        let lhs_is_negative = self.insert_cast(lhs_is_negative, NumericType::bool());
189
190        // The result changes slightly depending on whether we are doing division or modulo.
191        let result_is_negative = if is_division {
192            // For division, the result is negative if lhs and rhs have different signs.
193            let rhs_is_negative = self.insert_cast(rhs_is_negative, NumericType::bool());
194            self.insert_binary(lhs_is_negative, BinaryOp::Xor, rhs_is_negative)
195        } else {
196            // For modulo, the result has the same sign as lhs
197            lhs_is_negative
198        };
199        let result_is_negative = self.insert_cast(result_is_negative, unsigned_typ);
200
201        // We return the 2-complement again if lhs and rhs have different signs, with the
202        // intention of making the result be negative.
203        let result_unsigned =
204            self.two_complement(absolute_result, result_is_negative, unsigned_typ, bit_size);
205
206        // If we divide, for example 4 i8 by -5, the absolute division will give 0.
207        // Because the signs are different, if we do the two complement of 0 we'll get 256, which
208        // is out of range. Here we take this case into account: if absolute_div is zero the result
209        // should be zero, otherwise it should be that result.
210        // Then, we need to multiply result_unsigned by `absolute_div != 0`.
211        //
212        // The same is true for modulo: -4 i8 mod 4 is 0, but taking its two-complement would give 256.
213        let zero = self.numeric_constant(0_u128, unsigned_typ);
214        let absolute_result_is_zero = self.insert_binary(absolute_result, BinaryOp::Eq, zero);
215        let absolute_result_is_not_zero = self.insert_not(absolute_result_is_zero);
216        let absolute_result_is_not_zero =
217            self.insert_cast(absolute_result_is_not_zero, unsigned_typ);
218
219        let result_unsigned = self.insert_binary(
220            result_unsigned,
221            BinaryOp::Mul { unchecked: true },
222            absolute_result_is_not_zero,
223        );
224
225        // Make sure we return the signed type
226        self.insert_cast(result_unsigned, NumericType::signed(bit_size))
227    }
228
229    /// Returns the 2-complement of `value`, given `value_is_negative` is 1 if the value is negative,
230    /// and 0 if it's positive.
231    ///
232    /// The math here is:
233    ///
234    /// result = value + 2*((2^(bit_size - 1) - value)*value_is_negative)
235    ///
236    /// For example, for i8 we have bit_size = 8 so:
237    ///
238    /// result = value + 2*(128 - value)*value_is_negative
239    ///
240    /// If the value is positive, so value_is_negative = 0:
241    ///
242    /// result = value
243    ///
244    /// That is, the value stays the same.
245    ///
246    /// If value_is_negative = 1 we get:
247    ///
248    /// result = value + 2*(128 - value) = value + 256 - 2*value = 256 - value
249    ///
250    /// which effectively negates the value in 2-complement representation.
251    fn two_complement(
252        &mut self,
253        value: ValueId,
254        value_is_negative: ValueId,
255        unsigned_type: NumericType,
256        bit_size: u32,
257    ) -> ValueId {
258        let max_power_of_two = self.numeric_constant(1_u128 << (bit_size - 1), unsigned_type);
259
260        let intermediate =
261            self.insert_binary(max_power_of_two, BinaryOp::Sub { unchecked: true }, value);
262        let intermediate =
263            self.insert_binary(intermediate, BinaryOp::Mul { unchecked: true }, value_is_negative);
264        let two = self.numeric_constant(2_u128, unsigned_type);
265        let intermediate = self.insert_binary(intermediate, BinaryOp::Mul { unchecked: true }, two);
266        self.insert_binary(value, BinaryOp::Add { unchecked: true }, intermediate)
267    }

269    /// Insert a numeric constant into the current function
270    fn numeric_constant(&mut self, value: impl Into<FieldElement>, typ: NumericType) -> ValueId {
271        self.context.dfg.make_constant(value.into(), typ)
272    }
273
274    /// Insert a not instruction at the end of the current block.
275    /// Returns the result of the instruction.
276    fn insert_not(&mut self, rhs: ValueId) -> ValueId {
277        self.context.insert_instruction(Instruction::Not(rhs), None).first()
278    }
279
280    /// Insert a binary instruction at the end of the current block.
281    /// Returns the result of the binary instruction.
282    fn insert_binary(&mut self, lhs: ValueId, operator: BinaryOp, rhs: ValueId) -> ValueId {
283        let instruction = Instruction::Binary(Binary { lhs, rhs, operator });
284        self.context.insert_instruction(instruction, None).first()
285    }
286
287    /// Insert a cast instruction at the end of the current block.
288    /// Returns the result of the cast instruction.
289    fn insert_cast(&mut self, value: ValueId, typ: NumericType) -> ValueId {
290        self.context.insert_instruction(Instruction::Cast(value, typ), None).first()
291    }
292
293    /// Insert a constrain instruction at the end of the current block.
294    fn insert_constrain(
295        &mut self,
296        lhs: ValueId,
297        rhs: ValueId,
298        assert_message: Option<ConstrainError>,
299    ) {
300        self.context.insert_instruction(Instruction::Constrain(lhs, rhs, assert_message), None);
301    }
302}
303
304/// Post-check condition for [Function::expand_signed_math].
305///
306/// Succeeds if:
307///   - `func` does not contain any signed "less than" ops
308///
309/// Otherwise panics.
310#[cfg(debug_assertions)]
311fn expand_signed_math_post_check(func: &Function) {
312    for block_id in func.reachable_blocks() {
313        let instruction_ids = func.dfg[block_id].instructions();
314        for instruction_id in instruction_ids {
315            if let Instruction::Binary(binary) = &func.dfg[*instruction_id] {
316                if func.dfg.type_of_value(binary.lhs).is_signed() && binary.operator == BinaryOp::Lt
317                {
318                    panic!("Checked signed 'less than' has not been removed")
319                }
320            }
321        }
322    }
323}

332    #[test]
333    fn expands_signed_lt_in_acir() {
334        let src = "
335        acir(inline) fn main f0 {
336          b0(v0: i8, v1: i8):
337            v2 = lt v0, v1
338            return v2
339        }
340        ";
341        let ssa = Ssa::from_str(src).unwrap();
342        let ssa = ssa.expand_signed_math();
343
344        // Check that the expanded code works as expected
345        let test_cases = [
346            (10, 20, true),
347            (20, 10, false),
348            (-20, -10, true),
349            (-10, -20, false),
350            (-20, 10, true),
351            (-10, 20, true),
352            (10, -20, false),
353            (20, -10, false),
354        ];
355        for (lhs, rhs, expected) in test_cases {
356            let result = ssa.interpret(vec![Value::i8(lhs), Value::i8(rhs)]);
357            assert!(result.is_ok());
358            let result = result.unwrap();
359            assert_eq!(result.len(), 1);
360            assert_eq!(result[0], Value::bool(expected));
361        }
362
363        assert_ssa_snapshot!(ssa, @r"
364        acir(inline) fn main f0 {
365          b0(v0: i8, v1: i8):
366            v2 = cast v0 as u8
367            v3 = cast v1 as u8
368            v5 = div v2, u8 128
369            v6 = cast v5 as u1
370            v7 = div v3, u8 128
371            v8 = cast v7 as u1
372            v9 = xor v6, v8
373            v10 = lt v2, v3
374            v11 = xor v9, v10
375            return v11
376        }
377        ");
378    }
