1//! Analyzes the purity of each function and tag each function call with that function's purity.
2//! This is purely an analysis pass on its own but can help future optimizations.
3//!
4//! There is no constraint on when this pass needs to be run, but it is generally more
5//! beneficial to perform this pass before inlining or loop unrolling so that it can:
6//! 1. Run faster by processing fewer instructions.
7//! 2. Be run earlier in the pass list so that more passes afterward can use the results of
8//!    this pass.
9//!
10//! Performing this pass after defunctionalization may also help more function calls be
11//! identified as calling known pure functions.
12

27impl Ssa {
28    /// Analyzes the purity of each function and tag each function call with that function's purity.
29    /// This is purely an analysis pass on its own but can help future optimizations.
30    #[tracing::instrument(level = "trace", skip(self))]
31    pub(crate) fn purity_analysis(mut self) -> Ssa {
32        let call_graph = CallGraph::from_ssa(&self);
33
34        let (sccs, recursive_functions) = call_graph.sccs();
35
36        // First look through each function to get a baseline on its purity and collect
37        // the functions it calls to build a call graph.
38        let purities: HashMap<_, _> =
39            self.functions.values().map(|function| (function.id(), function.is_pure())).collect();
40
41        // Then transitively 'infect' any functions which call impure functions as also
42        // impure.
43        let purities = analyze_call_graph(call_graph, purities, &sccs, &recursive_functions);
44        let purities = Arc::new(purities);
45
46        // We're done, now store purities somewhere every dfg can find it.
47        for function in self.functions.values_mut() {
48            function.dfg.set_function_purities(purities.clone());
49        }
50
51        #[cfg(debug_assertions)]
52        purity_analysis_post_check(&self);
53
54        self
55    }
56}
57
58/// Post-check condition for [Ssa::purity_analysis].
59///
60/// Succeeds if:
61///   - all functions have a purity status attached to it.
62///
63/// Otherwise panics.
64#[cfg(debug_assertions)]
65fn purity_analysis_post_check(ssa: &Ssa) {
66    if let Some((id, _)) =
67        ssa.functions.iter().find(|(id, function)| function.dfg.purity_of(**id).is_none())
68    {
69        panic!("Function {id} does not have a purity status")
70    }
71}

73pub(crate) type FunctionPurities = HashMap<FunctionId, Purity>;
74
75#[derive(Debug, Copy, Clone, PartialEq, Eq)]
76pub enum Purity {
77    /// Function is completely pure and doesn't rely on a predicate at all.
78    /// Pure functions can be freely deduplicated or even removed from the program.
79    Pure,
80
81    /// Function is mostly pure. As long as the predicate is the same.
82    /// This applies to functions with `constrain` in them. So long as their
83    /// parameters are the same, the `constrain` should be to the same values
84    /// so the function is conceptually pure from a deduplication perspective
85    /// even though it can still interact with the `enable_side_effects`/predicate variable.
86    ///
87    /// PureWithPredicate functions can only be deduplicated with identical predicates
88    /// or a predicate that is a subset of the original.
89    PureWithPredicate,
90
91    /// This function is impure and cannot be deduplicated even with identical inputs.
92    /// This is most commonly the case for any function taking or returning a
93    /// reference value.
94    Impure,
95}
96
97impl Purity {
98    /// Unifies two purity values, returning the lower common denominator of the two
99    pub(crate) fn unify(self, other: Purity) -> Purity {
100        match (self, other) {
101            (Purity::Pure, Purity::Pure) => Purity::Pure,
102            (Purity::Impure, _) | (_, Purity::Impure) => Purity::Impure,
103            _ => Purity::PureWithPredicate,
104        }
105    }
106}

108impl std::fmt::Display for Purity {
109    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
110        match self {
111            Purity::Pure => write!(f, "pure"),
112            Purity::PureWithPredicate => write!(f, "predicate_pure"),
113            Purity::Impure => write!(f, "impure"),
114        }
115    }
116}

118impl Function {
119    fn is_pure(&self) -> Purity {
120        let contains_reference = |value_id: &ValueId| {
121            let typ = self.dfg.type_of_value(*value_id);
122            typ.contains_reference()
123        };
124
125        if self.parameters().iter().any(&contains_reference) {
126            return Purity::Impure;
127        }
128
129        let mut result = if self.runtime().is_acir() {
130            Purity::Pure
131        } else {
132            // Because we return bogus values when a brillig function is called from acir
133            // in a disabled predicate, brillig functions can never be truly pure unfortunately.
134            Purity::PureWithPredicate
135        };
136
137        for block in self.reachable_blocks() {
138            for instruction in self.dfg[block].instructions() {
139                // We don't defer to Instruction::can_be_deduplicated, Instruction::requires_acir_gen_predicate,
140                // etc. since we don't consider local mutations to be impure. Local mutations should
141                // be invisible to calling functions so as long as no references are taken as
142                // parameters or returned, we can ignore them.
143                // We even ignore Constrain instructions. As long as the external parameters are
144                // identical, we should be constraining the same values anyway.
145                match &self.dfg[*instruction] {
146                    Instruction::Constrain(..)
147                    | Instruction::ConstrainNotEqual(..)
148                    | Instruction::RangeCheck { .. } => result = Purity::PureWithPredicate,
149
150                    // These instructions may be pure unless:
151                    // - We may divide by zero
152                    // - The array index is out of bounds.
153                    // For both cases we can still treat them as pure if the arguments are known
154                    // constants.
155                    ins @ (Instruction::Binary(_)
156                    | Instruction::ArrayGet { .. }) => {
157                        if ins.requires_acir_gen_predicate(&self.dfg) {
158                            result = Purity::PureWithPredicate;
159                        }
160                    }
161                    ins @ Instruction::ArraySet { array, .. } => {
162                      if self.runtime().is_brillig() && (self.parameters().contains(array) || self.dfg.is_global(*array)) {
163                          return Purity::Impure;
164                      } else if ins.requires_acir_gen_predicate(&self.dfg) {
165                            result = Purity::PureWithPredicate;
166                      }
167                    }
168                    Instruction::Call { func, .. } => {
169                        match &self.dfg[*func] {
170                            Value::Function(_) => {
171                                // We don't know if this function is pure or not yet,
172                                //
173                                // `is_pure` is intended to be called on each function, building
174                                // up a call graph of sorts to check afterwards to propagate impurity
175                                // from called functions to their callers. Therefore, an initial "Pure"
176                                // result here could be overridden by one of these dependencies being impure.
177                            }
178                            Value::Intrinsic(intrinsic) => match intrinsic.purity() {
179                                Purity::Pure => (),
180                                Purity::PureWithPredicate => result = Purity::PureWithPredicate,
181                                Purity::Impure => return Purity::Impure,
182                            },
183                            Value::ForeignFunction(_) => return Purity::Impure,
184                            // The function we're calling is unknown in the remaining cases,
185                            // so just assume the worst.
186                            Value::Global(_)
187                            | Value::Instruction { .. }
188                            | Value::Param { .. }
189                            | Value::NumericConstant { .. } => return Purity::Impure,
190                        }
191                    }
192
193                    // The rest are always pure (including allocate, load, & store)
194                    Instruction::Cast(_, _)
195                    | Instruction::Not(_)
196                    | Instruction::Truncate { .. }
197                    | Instruction::Allocate
198                    // Load and store are considered pure since there is a separate check ensuring
199                    // no parameters or return values are references. With this check, we can be
200                    // sure any load/store is purely local.
201                    | Instruction::Load { .. }
202                    | Instruction::Store { .. }
203                    | Instruction::EnableSideEffectsIf { .. }
204                    | Instruction::IfElse { .. }
205                    | Instruction::MakeArray { .. }
206                    | Instruction::Noop => (),
207
208                    Instruction::IncrementRc { value }
209                    | Instruction::DecrementRc { value } => {
210                      if self.parameters().contains(value) || self.dfg.is_global(*value) {
211                        return Purity::Impure
212                      }
213                    }
214                };
215            }
216
217            // If the function returns a reference it is impure
218            let terminator = self.dfg[block].terminator();
219            if let Some(TerminatorInstruction::Return { return_values, .. }) = terminator
220                && return_values.iter().any(&contains_reference)
221            {
222                return Purity::Impure;
223            }
224        }
225
226        result
227    }
228}

230fn analyze_call_graph(
231    call_graph: CallGraph,
232    starting_purities: FunctionPurities,
233    sccs: &[Vec<FunctionId>],
234    recursive_functions: &HashSet<FunctionId>,
235) -> FunctionPurities {
236    let mut finished = HashMap::default();
237
238    // Map FunctionId -> SCC index for quick lookup
239    let mut func_to_scc = HashMap::default();
240    for (i, scc) in sccs.iter().enumerate() {
241        for &func in scc {
242            // Each function belongs to exactly one SCC by definition of SCCs.
243            // Therefore inserting into func_to_scc here is safe, and there will
244            // be no overwrites.
245            let inserted = func_to_scc.insert(func, i);
246            assert!(inserted.is_none(), "Function appears in multiple SCCs");
247        }
248    }
249
250    // Track SCC purity
251    let mut scc_purities: Vec<Purity> = sccs
252        .iter()
253        .map(|scc| scc.iter().map(|f| starting_purities[f]).fold(Purity::Pure, |a, b| a.unify(b)))
254        .collect();
255
256    // Iteratively propagate purity between SCCs until convergence
257    let mut changed = true;
258    while changed {
259        changed = false;
260
261        for (i, scc) in sccs.iter().enumerate() {
262            let mut combined_purity = scc_purities[i];
263
264            // Look at neighbors outside the SCC
265            for &func in scc {
266                let idx = call_graph.ids_to_indices()[&func];
267                for neighbor_idx in call_graph.graph().neighbors(idx) {
268                    let neighbor = call_graph.indices_to_ids()[&neighbor_idx];
269                    let neighbor_scc = func_to_scc[&neighbor];
270                    if neighbor_scc != i {
271                        combined_purity = combined_purity.unify(scc_purities[neighbor_scc]);
272                    }
273                }
274
275                // Recursive functions cannot be fully pure (may recurse indefinitely),
276                // but we still treat them as PureWithPredicate for deduplication purposes.
277                // If we were to mark recursive functions pure we may entirely eliminate an infinite loop.
278                if recursive_functions.contains(&func) {
279                    combined_purity = combined_purity.unify(Purity::PureWithPredicate);
280                }
281            }
282
283            if combined_purity != scc_purities[i] {
284                scc_purities[i] = combined_purity;
285                changed = true;
286            }
287        }
288    }
289
290    // Assign SCC purity to all functions in the SCC
291    for (i, scc) in sccs.iter().enumerate() {
292        for &func in scc {
293            finished.insert(func, scc_purities[i]);
294        }
295    }
296
297    finished
298}

618    fn mutual_recursion_marks_functions_pure() {
619        // We want to test that two pure mutually recursive functions do in fact mark each other as PureWithPredicate.
620        // If we have indefinite recursion and we may accidentally eliminate an infinite loop before inlining can catch it.
621        let src = r#"
622        acir(inline) fn main f0 {
623          b0():
624            v0 = call f1(u32 4) -> bool
625            return
626        }
627        acir(inline) fn is_even f1 {
628          b0(v0: u32):
629            v1 = eq v0, u32 0
630            jmpif v1 then: b1, else: b2
631          b1():
632            jmp b3(u1 1)
633          b2():
634            v2 = unchecked_sub v0, u32 1
635            v3 = call f2(v2) -> bool
636            jmp b3(v3)
637          b3(v4: bool):
638            return v4
639        }
640        acir(inline) fn is_odd f2 {
641          b0(v0: u32):
642            v1 = eq v0, u32 0
643            jmpif v1 then: b1, else: b2
644          b1():

710    #[test]
711    fn mutual_recursion_marks_functions_impure() {
712        // f1 -> f2 -> f3 -> f1 (a cycle of three functions)
713        // Only f3 is locally impure (it returns a reference).
714        // All three must be marked Impure.
715        //
716        // We call f2 in main as we want the DFS to not look at f3 first (which is "Impure").
717        // If f3 is found first the cycle will get correctly marked as impure.
718        // We want to make sure that even when the first function in the recursive cycle
719        // is not marked as impure that we still accurately mark the entire cycle impure.
720        // Calling f2 first, means the cycle will look at f1 first, which still
721        // has a starting purity of "Pure".
722        let src = r#"
723        acir(inline) fn main f0 {
724          b0():
725            v0 = call f2() -> Field
726            return
727        }
728        acir(inline) fn f1 f1 {
729          b0():
730            v0 = call f2() -> Field
731            return v0
732        }
733        acir(inline) fn f2 f2 {
734          b0():
735            v0 = call f3() -> &mut Field
736            v1 = load v0 -> Field
737            return v1
738        }
739        acir(inline) fn f3 f3 {
740          b0():
741            v0 = call f1() -> Field
742            v1 = allocate -> &mut Field
743            return v1
744        }
745        "#;
746
747        let ssa = Ssa::from_str(src).unwrap();
748        let ssa = ssa.purity_analysis();
749
750        let purities = &ssa.main().dfg.function_purities;
751        // All must be impure due to the cycle involved f3 when returns a reference.
752        assert_eq!(purities[&FunctionId::test_new(1)], Purity::Impure);
753        assert_eq!(purities[&FunctionId::test_new(2)], Purity::Impure);
754        assert_eq!(purities[&FunctionId::test_new(3)], Purity::Impure);
755    }
