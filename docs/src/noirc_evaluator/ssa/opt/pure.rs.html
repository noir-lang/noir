1//! Analyzes the purity of each function and tag each function call with that function's purity.
2//! This is purely an analysis pass on its own but can help future optimizations.
3//!
4//! There is no constraint on when this pass needs to be run, but it is generally more
5//! beneficial to perform this pass before inlining or loop unrolling so that it can:
6//! 1. Run faster by processing fewer instructions.
7//! 2. Be run earlier in the pass list so that more passes afterward can use the results of
8//!    this pass.
9//!
10//! Performing this pass after defunctionalization may also help more function calls be
11//! identified as calling known pure functions.
12

28impl Ssa {
29    /// Analyzes the purity of each function and tag each function call with that function's purity.
30    /// This is purely an analysis pass on its own but can help future optimizations.
31    #[tracing::instrument(level = "trace", skip(self))]
32    pub(crate) fn purity_analysis(mut self) -> Ssa {
33        let purities = Arc::new(compute_function_purities(&self));
34
35        for function in self.functions.values_mut() {
36            function.dfg.set_function_purities(purities.clone());
37        }
38
39        #[cfg(debug_assertions)]
40        purity_analysis_post_check(&self);
41
42        self
43    }
44}

46/// Compute the purity of every function in the SSA, including call-graph propagation,
47/// without mutating the SSA. Shared by [Ssa::purity_analysis] and by the SSA parser,
48/// which uses it to validate hand-written purity annotations against the actual
49/// instruction-level behavior.
50pub(crate) fn compute_function_purities(ssa: &Ssa) -> FunctionPurities {
51    // Purity falls back to `Impure` for any call whose callee cannot be statically
52    // resolved, so an incomplete call graph is fine — use the partial constructor
53    // to allow running on pre-defunctionalize SSA in unit tests.
54    let call_graph = CallGraph::from_ssa_partial(ssa);
55    let (sccs, recursive_functions) = call_graph.sccs();
56    let purities: HashMap<_, _> =
57        ssa.functions.values().map(|function| (function.id(), function.is_pure())).collect();
58    analyze_call_graph(call_graph, purities, &sccs, &recursive_functions)
59}
60
61/// Post-check condition for [Ssa::purity_analysis].
62///
63/// Succeeds if:
64///   - all functions have a purity status attached to it.
65///
66/// Otherwise panics.
67#[cfg(debug_assertions)]
68fn purity_analysis_post_check(ssa: &Ssa) {
69    if let Some((id, _)) =
70        ssa.functions.iter().find(|(id, function)| function.dfg.purity_of(**id).is_none())
71    {
72        panic!("Function {id} does not have a purity status")
73    }
74}

76pub(crate) type FunctionPurities = HashMap<FunctionId, Purity>;
77
78#[derive(Debug, Copy, Clone, PartialEq, Eq)]
79pub enum Purity {
80    /// Function is completely pure and doesn't rely on a predicate at all.
81    /// Pure functions can be freely deduplicated or even removed from the program.
82    Pure,
83
84    /// Function is mostly pure. As long as the predicate is the same.
85    /// This applies to functions with `constrain` in them. So long as their
86    /// parameters are the same, the `constrain` should be to the same values
87    /// so the function is conceptually pure from a deduplication perspective
88    /// even though it can still interact with the `enable_side_effects`/predicate variable.
89    ///
90    /// PureWithPredicate functions can only be deduplicated with identical predicates
91    /// or a predicate that is a subset of the original.
92    PureWithPredicate,
93
94    /// This function is impure and cannot be deduplicated even with identical inputs.
95    /// This is most commonly the case for any function taking or returning a
96    /// reference value.
97    Impure,
98}
99
100impl Purity {
101    /// Unifies two purity values, returning the lower common denominator of the two
102    pub(crate) fn unify(self, other: Purity) -> Purity {
103        match (self, other) {
104            (Purity::Pure, Purity::Pure) => Purity::Pure,
105            (Purity::Impure, _) | (_, Purity::Impure) => Purity::Impure,
106            _ => Purity::PureWithPredicate,
107        }
108    }
109}

111impl std::fmt::Display for Purity {
112    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
113        match self {
114            Purity::Pure => write!(f, "pure"),
115            Purity::PureWithPredicate => write!(f, "predicate_pure"),
116            Purity::Impure => write!(f, "impure"),
117        }
118    }
119}

121impl Function {
122    pub(crate) fn is_pure(&self) -> Purity {
123        let contains_reference = |value_id: &ValueId| {
124            let typ = self.dfg.type_of_value(*value_id);
125            typ.contains_reference()
126        };
127
128        if self.parameters().iter().any(&contains_reference) {
129            return Purity::Impure;
130        }
131
132        // Collect all parameters that are arrays or contain arrays, but only for Brillig.
133        // If we detect an array_set potentially operating on a brillig array input, the entire
134        // function becomes impure.
135        let brillig_array_inputs = if self.runtime().is_brillig() {
136            self.parameters()
137                .iter()
138                .filter(|param| {
139                    let typ = self.dfg.type_of_value(**param);
140                    typ.contains_an_array()
141                })
142                .collect::<HashSet<_>>()
143        } else {
144            HashSet::default()
145        };
146        let has_brillig_array_input = !brillig_array_inputs.is_empty();
147
148        // Records whether there's an `array_set`, `inc_rc` or `dec_rc` in this function.
149        let mut has_array_set_or_rc = false;
150
151        // Records whether a brillig array input was used in an instruction that could have moved
152        // it to another value. Examples include `store`, `array_set`, and even `array_get` for parameters
153        // that have nested arrays.
154        let mut brillig_array_input_was_moved = false;
155
156        let mut result = if self.runtime().is_acir() {
157            Purity::Pure
158        } else {
159            // Because we return bogus values when a brillig function is called from acir
160            // in a disabled predicate, brillig functions can never be truly pure unfortunately.
161            Purity::PureWithPredicate
162        };
163
164        for block in self.reachable_blocks() {
165            for instruction in self.dfg[block].instructions() {
166                // We don't defer to Instruction::can_be_deduplicated, Instruction::requires_acir_gen_predicate,
167                // etc. since we don't consider local mutations to be impure. Local mutations should
168                // be invisible to calling functions so as long as no references are taken as
169                // parameters or returned, we can ignore them.
170                // We even ignore Constrain instructions. As long as the external parameters are
171                // identical, we should be constraining the same values anyway.
172                let ins = &self.dfg[*instruction];
173                match ins {
174                    Instruction::Constrain(..)
175                    | Instruction::ConstrainNotEqual(..)
176                    | Instruction::RangeCheck { .. } => result = Purity::PureWithPredicate,
177
178                    // These instructions may be pure unless:
179                    // - We may divide by zero
180                    // - The array index is out of bounds.
181                    // For both cases we can still treat them as pure if the arguments are known
182                    // constants.
183                    Instruction::Binary(_) | Instruction::ArrayGet { .. } => {
184                        if ins.requires_acir_gen_predicate(&self.dfg) {
185                            result = Purity::PureWithPredicate;
186                        }
187                    }
188                    Instruction::ArraySet { .. } => {
189                      has_array_set_or_rc = true;
190                      result = Purity::PureWithPredicate;
191                    }
192                    Instruction::Call { func, .. } => {
193                        match &self.dfg[*func] {
194                            Value::Function(_) => {
195                                // We don't know if this function is pure or not yet,
196                                //
197                                // `is_pure` is intended to be called on each function, building
198                                // up a call graph of sorts to check afterwards to propagate impurity
199                                // from called functions to their callers. Therefore, an initial "Pure"
200                                // result here could be overridden by one of these dependencies being impure.
201                            }
202                            Value::Intrinsic(intrinsic) => match intrinsic.purity() {
203                                Purity::Pure => (),
204                                Purity::PureWithPredicate => result = Purity::PureWithPredicate,
205                                Purity::Impure => return Purity::Impure,
206                            },
207                            Value::ForeignFunction { pure: true, .. } => {
208                                // A `#[pure]` oracle is treated as `PureWithPredicate`, because
209                                // they are unconstrained functions.
210                                result = Purity::PureWithPredicate;
211                            }
212                            Value::ForeignFunction { pure: false, .. } => return Purity::Impure,
213                            // The function we're calling is unknown in the remaining cases,
214                            // so just assume the worst.
215                            Value::Global(_)
216                            | Value::Instruction { .. }
217                            | Value::Param { .. }
218                            | Value::NumericConstant { .. } => return Purity::Impure,
219                        }
220                    }
221
222                    // The rest are always pure (including allocate, load, & store)
223                    Instruction::Cast(_, _)
224                    | Instruction::Not(_)
225                    | Instruction::Truncate { .. }
226                    | Instruction::Allocate
227                    // Load and store are considered pure since there is a separate check ensuring
228                    // no parameters or return values are references. With this check, we can be
229                    // sure any load/store is purely local.
230                    | Instruction::Load { .. }
231                    | Instruction::Store { .. }
232                    | Instruction::EnableSideEffectsIf { .. }
233                    | Instruction::IfElse { .. }
234                    | Instruction::MakeArray { .. }
235                    | Instruction::Noop => (),
236
237                    Instruction::IncrementRc { .. } | Instruction::DecrementRc { .. } => {
238                        has_array_set_or_rc = true;
239                    }
240                }
241
242                // Separately, check if any instruction could be moving a Brillig array input.
243                if has_brillig_array_input {
244                    match ins {
245                        Instruction::Binary(_)
246                        | Instruction::Cast(..)
247                        | Instruction::Not(_)
248                        | Instruction::Truncate { .. }
249                        | Instruction::Constrain(..)
250                        | Instruction::ConstrainNotEqual(..)
251                        | Instruction::RangeCheck { .. }
252                        | Instruction::Call { .. }
253                        | Instruction::Allocate
254                        | Instruction::EnableSideEffectsIf { .. }
255                        | Instruction::Noop => {
256                            // This can't possibly move a Brillig array input.
257                            // A `call` could mutate a Brillig array input, but if that is the case
258                            // the the call itself will be marked as impure, and so then this function will
259                            // be impure... but that is a check that is done later on.
260                        }
261
262                        Instruction::Load { .. }
263                        | Instruction::Store { .. }
264                        | Instruction::ArraySet { .. }
265                        | Instruction::IncrementRc { .. }
266                        | Instruction::DecrementRc { .. }
267                        | Instruction::IfElse { .. }
268                        | Instruction::MakeArray { .. } => {
269                            // Check if any of these instructions is operating on a Brillig array input
270                            brillig_array_input_was_moved |= has_brillig_array_input
271                                && ins.any_value(|value| brillig_array_inputs.contains(&value));
272                        }
273                        Instruction::ArrayGet { array, index: _ } => {
274                            // For ArrayGet we do something slightly different: if it operates on a Brillig array input
275                            // array, an array could be moved if it's nested inside `array` (for example if the type
276                            // is `[[Field; 2]; 3]`. However, if the `array` is an array without nested arrays, no
277                            // array will be moved here. We consider this case specifically because fetching from a
278                            // non-nested Brillig array input is a common pattern.
279                            if brillig_array_inputs.contains(array) {
280                                let typ = self.dfg.type_of_value(*array);
281                                let typ = typ.as_ref();
282                                match typ {
283                                    Type::Array(items, _) | Type::Vector(items) => {
284                                        if items.iter().any(|item| item.contains_an_array()) {
285                                            brillig_array_input_was_moved = true;
286                                        }
287                                    }
288                                    Type::Numeric(_) | Type::Reference(_, _) | Type::Function => (),
289                                }
290                            }
291                        }
292                    }
293                }
294            }
295
296            // If the function returns a reference it is impure
297            let terminator = self.dfg[block].terminator();
298            if let Some(terminator) = terminator {
299                if let TerminatorInstruction::Return { return_values, .. } = terminator
300                    && return_values.iter().any(&contains_reference)
301                {
302                    return Purity::Impure;
303                }
304
305                // Also check if any Brillig array input is moved in a terminator
306                if has_brillig_array_input
307                    && terminator.any_value(|value| brillig_array_inputs.contains(&value))
308                {
309                    brillig_array_input_was_moved = true;
310                }
311            }
312        }
313
314        // If a Brillig array input was moved, and we found any instruction that could mutate it
315        // (`array_set`, `inc_rc` or `dec_rc`) then we consider the function impure.
316        if has_array_set_or_rc && brillig_array_input_was_moved {
317            return Purity::Impure;
318        }
319
320        result
321    }
322}

324fn analyze_call_graph(
325    call_graph: CallGraph,
326    starting_purities: FunctionPurities,
327    sccs: &[Vec<FunctionId>],
328    recursive_functions: &HashSet<FunctionId>,
329) -> FunctionPurities {
330    let mut finished = HashMap::default();
331
332    // Map FunctionId -> SCC index for quick lookup
333    let mut func_to_scc = HashMap::default();
334    for (i, scc) in sccs.iter().enumerate() {
335        for &func in scc {
336            // Each function belongs to exactly one SCC by definition of SCCs.
337            // Therefore inserting into func_to_scc here is safe, and there will
338            // be no overwrites.
339            let inserted = func_to_scc.insert(func, i);
340            assert!(inserted.is_none(), "Function appears in multiple SCCs");
341        }
342    }
343
344    // Track SCC purity
345    let mut scc_purities: Vec<Purity> = sccs
346        .iter()
347        .map(|scc| scc.iter().map(|f| starting_purities[f]).fold(Purity::Pure, |a, b| a.unify(b)))
348        .collect();
349
350    // Iteratively propagate purity between SCCs until convergence
351    let mut changed = true;
352    while changed {
353        changed = false;
354
355        for (i, scc) in sccs.iter().enumerate() {
356            let mut combined_purity = scc_purities[i];
357
358            // Look at neighbors outside the SCC
359            for &func in scc {
360                let idx = call_graph.ids_to_indices()[&func];
361                for neighbor_idx in call_graph.graph().neighbors(idx) {
362                    let neighbor = call_graph.indices_to_ids()[&neighbor_idx];
363                    let neighbor_scc = func_to_scc[&neighbor];
364                    if neighbor_scc != i {
365                        combined_purity = combined_purity.unify(scc_purities[neighbor_scc]);
366                    }
367                }
368
369                // Recursive functions cannot be fully pure (may recurse indefinitely),
370                // but we still treat them as PureWithPredicate for deduplication purposes.
371                // If we were to mark recursive functions pure we may entirely eliminate an infinite loop.
372                if recursive_functions.contains(&func) {
373                    combined_purity = combined_purity.unify(Purity::PureWithPredicate);
374                }
375            }
376
377            if combined_purity != scc_purities[i] {
378                scc_purities[i] = combined_purity;
379                changed = true;
380            }
381        }
382    }
383
384    // Assign SCC purity to all functions in the SCC
385    for (i, scc) in sccs.iter().enumerate() {
386        for &func in scc {
387            finished.insert(func, scc_purities[i]);
388        }
389    }
390
391    finished
392}
