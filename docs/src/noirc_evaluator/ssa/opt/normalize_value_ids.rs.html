1//! This is a debugging pass which re-inserts each instruction
2//! and block in a fresh DFG context for each function so that ValueIds,
3//! BasicBlockIds, and FunctionIds are always identical for the same SSA code.
4//!
5//! During normal compilation this is often not the case since prior passes
6//! may increase the ID counter so that later passes start at different offsets,
7//! even if they contain the same SSA code.
8
9use std::{collections::BTreeMap, sync::Arc};
10
11use crate::ssa::{
12    ir::{
13        basic_block::BasicBlockId,
14        function::{Function, FunctionId},
15        map::SparseMap,
16        post_order::PostOrder,
17        value::{Value, ValueId},
18    },
19    ssa_gen::Ssa,
20};
21use iter_extended::vecmap;
22use rustc_hash::FxHashMap as HashMap;
23
24impl Ssa {
25    /// Re-inserts each instruction and block in a fresh DFG context for each function so that
26    /// ValueIds, BasicBlockIds, and FunctionIds are always identical for the same SSA code.
27    pub fn normalize_ids(&mut self) {
28        let mut context = Context::default();
29        context.populate_functions(&self.functions);
30        for function in self.functions.values_mut() {
31            context.normalize_ids(function);
32        }
33        self.functions = context.functions.into_btree();
34    }
35}
36
37#[derive(Default)]
38struct Context {
39    functions: SparseMap<Function>,
40
41    new_ids: IdMaps,
42}
43
44/// Maps from old ids to new ones.
45/// Separate from the rest of Context so we can call mutable methods on it
46/// while Context gives out mutable references to functions within.
47#[derive(Default)]
48struct IdMaps {
49    // Maps old function id -> new function id
50    function_ids: HashMap<FunctionId, FunctionId>,
51
52    // Maps old block id -> new block id
53    // Cleared in between each function.
54    blocks: HashMap<BasicBlockId, BasicBlockId>,
55
56    // Maps old value id -> new value id
57    // Cleared in between each function.
58    values: HashMap<ValueId, ValueId>,
59}
60
61impl Context {
62    fn populate_functions(&mut self, functions: &BTreeMap<FunctionId, Function>) {
63        let Some(old_purities) = &functions.iter().next().map(|f| &f.1.dfg.function_purities)
64        else {
65            return;
66        };
67        let mut new_purities = HashMap::default();
68
69        for (id, function) in functions {
70            self.functions.insert_with_id(|new_id| {
71                self.new_ids.function_ids.insert(*id, new_id);
72
73                if let Some(purity) = old_purities.get(id) {
74                    new_purities.insert(new_id, *purity);
75                }
76
77                Function::clone_signature(new_id, function)
78            });
79        }
80
81        let new_purities = Arc::new(new_purities);
82        for new_id in self.new_ids.function_ids.values() {
83            self.functions[*new_id].dfg.set_function_purities(new_purities.clone());
84        }
85    }
86
87    fn normalize_ids(&mut self, old_function: &mut Function) {
88        self.new_ids.blocks.clear();
89        self.new_ids.values.clear();
90
91        let new_function_id = self.new_ids.function_ids[&old_function.id()];
92        let new_function = &mut self.functions[new_function_id];
93
94        for (_, value) in old_function.dfg.globals.values_iter() {
95            new_function.dfg.make_global(value.get_type().into_owned());
96        }
97
98        let reachable_blocks = old_function.reachable_blocks();
99        self.new_ids.populate_blocks(reachable_blocks, old_function, new_function);
100
101        let reverse_post_order = PostOrder::with_function(old_function).into_vec_reverse();
102
103        // Map each parameter, instruction, and terminator
104        for old_block_id in reverse_post_order {
105            let new_block_id = self.new_ids.blocks[&old_block_id];
106
107            let old_block = &mut old_function.dfg[old_block_id];
108            for old_instruction_id in old_block.take_instructions() {
109                let instruction = old_function.dfg[old_instruction_id]
110                    .map_values(|value| self.new_ids.map_value(new_function, old_function, value));
111
112                let call_stack = old_function.dfg.get_instruction_call_stack_id(old_instruction_id);
113                let locations = old_function.dfg.get_call_stack(call_stack);
114                let new_call_stack =
115                    new_function.dfg.call_stack_data.get_or_insert_locations(&locations);
116                let old_results = old_function.dfg.instruction_results(old_instruction_id);
117
118                let ctrl_typevars = instruction
119                    .requires_ctrl_typevars()
120                    .then(|| vecmap(old_results, |result| old_function.dfg.type_of_value(*result)));
121
122                let new_results =
123                    new_function.dfg.insert_instruction_and_results_without_simplification(
124                        instruction,
125                        new_block_id,
126                        ctrl_typevars,
127                        new_call_stack,
128                    );
129
130                assert_eq!(old_results.len(), new_results.len());
131                for (old_result, new_result) in old_results.iter().zip(new_results.results().iter())
132                {
133                    let old_result = *old_result;
134                    self.new_ids.values.insert(old_result, *new_result);
135                }
136            }
137
138            let old_block = &mut old_function.dfg[old_block_id];
139            let mut terminator = old_block.take_terminator();
140            terminator
141                .map_values_mut(|value| self.new_ids.map_value(new_function, old_function, value));
142
143            terminator.mutate_blocks(|old_block| self.new_ids.blocks[&old_block]);
144            let locations = old_function.dfg.get_call_stack(terminator.call_stack());
145            let new_call_stack =
146                new_function.dfg.call_stack_data.get_or_insert_locations(&locations);
147            terminator.set_call_stack(new_call_stack);
148            new_function.dfg.set_block_terminator(new_block_id, terminator);
149        }
150
151        // Also map the values in the databus
152        let old_databus = &old_function.dfg.data_bus;
153        new_function.dfg.data_bus = old_databus
154            .map_values(|old_value| self.new_ids.map_value(new_function, old_function, old_value));
155    }
156}
157
158impl IdMaps {
159    fn populate_blocks(
160        &mut self,
161        reachable_blocks: impl IntoIterator<Item = BasicBlockId>,
162        old_function: &mut Function,
163        new_function: &mut Function,
164    ) {
165        let old_entry = old_function.entry_block();
166        self.blocks.insert(old_entry, new_function.entry_block());
167
168        for old_id in reachable_blocks {
169            if old_id != old_entry {
170                let new_id = new_function.dfg.make_block();
171                self.blocks.insert(old_id, new_id);
172            }
173
174            let new_id = self.blocks[&old_id];
175            let old_block = &mut old_function.dfg[old_id];
176            for old_parameter in old_block.take_parameters() {
177                let typ = old_function.dfg.type_of_value(old_parameter);
178                let new_parameter = new_function.dfg.add_block_parameter(new_id, typ);
179                self.values.insert(old_parameter, new_parameter);
180            }
181        }
182    }
183
184    fn map_value(
185        &self,
186        new_function: &mut Function,
187        old_function: &Function,
188        old_value: ValueId,
189    ) -> ValueId {
190        if old_function.dfg.is_global(old_value) {
191            // Globals are computed at compile-time and thus are expected to be remain normalized
192            // between SSA passes
193            return old_value;
194        }
195        match &old_function.dfg[old_value] {
196            value @ Value::Instruction { instruction, .. } => {
197                *self.values.get(&old_value).unwrap_or_else(|| {
198                    let instruction = &old_function.dfg[*instruction];
199                    unreachable!("Unmapped value with id {old_value}: {value:?}\n  from instruction: {instruction:?}, from function: {}", old_function.id())
200                })
201            }
202
203            value @ Value::Param { .. } => {
204                *self.values.get(&old_value).unwrap_or_else(|| {
205                    unreachable!("Unmapped value with id {old_value}: {value:?}")
206                })
207            }
208
209            Value::Function(id) => {
210                let new_id = *self.function_ids.get(id).unwrap_or_else(|| {
211                    unreachable!("Unmapped function with id {id}")
212                });
213                new_function.dfg.import_function(new_id)
214            }
215
216            Value::NumericConstant { constant, typ } => {
217                new_function.dfg.make_constant(*constant, *typ)
218            }
219            Value::Intrinsic(intrinsic) => new_function.dfg.import_intrinsic(*intrinsic),
220            Value::ForeignFunction(name) => new_function.dfg.import_foreign_function(name),
221            Value::Global(_) => {
222                unreachable!("Should have handled the global case already");
223            },
224        }
225    }
