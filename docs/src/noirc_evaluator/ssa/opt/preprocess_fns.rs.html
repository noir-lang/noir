1//! Pre-process functions before inlining them into others.
2
3use crate::ssa::{
4    RuntimeError, Ssa,
5    ir::{call_graph::CallGraph, function::Function},
6};
7
8use super::FORCE_UNROLL_THRESHOLD;
9use super::inlining::{self, InlineInfo};
10
11impl Ssa {
12    /// Run pre-processing steps on functions in isolation.
13    pub(crate) fn preprocess_functions(
14        mut self,
15        aggressiveness: i64,
16        small_function_max_instructions: usize,
17    ) -> Result<Ssa, RuntimeError> {
18        let call_graph = CallGraph::from_ssa_weighted(&self);
19        // Bottom-up order, starting with the "leaf" functions, so we inline already optimized code into the ones that call them.
20        let bottom_up = inlining::inline_info::compute_bottom_up_order(&self, &call_graph);
21
22        // Preliminary inlining decisions.
23        let inline_infos = inlining::inline_info::compute_inline_infos(
24            &self,
25            &call_graph,
26            false,
27            small_function_max_instructions,
28            aggressiveness,
29        );
30
31        let should_inline_call =
32            |callee: &Function| -> bool { InlineInfo::should_inline(&inline_infos, callee.id()) };
33
34        for (id, (own_weight, transitive_weight)) in bottom_up {
35            let function = &self.functions[&id];
36
37            // Skip preprocessing heavy functions that gained most of their weight from transitive accumulation, which tend to be near the entry.
38            // These can be processed later by the regular SSA passes.
39            let is_heavy = transitive_weight > own_weight * 10;
40
41            // Functions which are inline targets will be processed in later passes.
42            // Here we want to treat the functions which will be inlined into them.
43            let is_target = inline_infos
44                .get(&id)
45                .map(|info| info.is_inline_target(&function.dfg))
46                .unwrap_or_default();
47
48            if is_heavy || is_target {
49                continue;
50            }
51
52            // Start with an inline pass.
53            let mut function = function.inlined(&self, &should_inline_call)?;
54            // Help unrolling determine bounds.
55            function.as_vector_optimization();
56            // Prepare for unrolling
57            function.loop_invariant_code_motion();
58            // Clear out constant jmpifs to ensure that loops are properly unrolled.
59            function.simplify_function();
60            // We might not be able to unroll all loops without fully inlining them, so ignore errors.
61            // Use default threshold for force-unrolling.
62            let _ = function.unroll_loops_iteratively(FORCE_UNROLL_THRESHOLD);
63            // Reduce the number of redundant stores/loads after unrolling
64            function.mem2reg();
65
66            // Try to reduce the number of blocks.
67            function.simplify_function();
68
69            // Put it back into the SSA, so the next functions can pick it up.
70            self.functions.insert(id, function);
71        }
72
73        // Remove any functions that have been inlined into others already.
74        let ssa = self.remove_unreachable_functions();
75        // Remove leftover instructions.
76        Ok(ssa.dead_instruction_elimination_pre_flattening())
77    }
78}

80#[cfg(test)]
81mod tests {
82    use crate::{
83        assert_ssa_snapshot,
84        ssa::{opt::inlining::inline_info::MAX_INSTRUCTIONS, ssa_gen::Ssa},
85    };
86
87    #[test]
88    fn dead_block_params() {
89        // This test makes sure that we are appropriately triggering DIE+parameter pruning.
90        //
91        // DIE must be run over the full SSA to correctly identify unused block parameters.
92        // If it's run in isolation on a single function (e.g., during preprocessing),
93        // it may leave dangling block parameters.
94        //
95        // We need to call f0 from an entry point as inline targets are not preprocessed.
96        let src = r#"
97        acir(inline) fn main f0 {
98          b0():
99            call f0(u32 1, Field 2)
100            return
101        }
102        acir(inline) fn foo f0 {
103          b0(v0: u32, v1: Field):
104            v2 = eq v0, u32 1
105            jmpif v2 then: b1, else: b2
106          b1():
107            v6 = add v0, u32 1
108            jmp b3(v6, v1)
109          b2():
110            v5 = sub v0, u32 1
111            jmp b3(v5, v1)
112          b3(v3: u32, v4: Field):
113            return
114        }
115        "#;
116
117        let ssa = Ssa::from_str(src).unwrap();
118        let ssa = ssa.preprocess_functions(i64::MAX, MAX_INSTRUCTIONS).unwrap();
119
120        assert_ssa_snapshot!(ssa, @r"
121        acir(inline) fn main f0 {
122          b0():
123            call f1(u32 1)
124            return
125        }
126        acir(inline) fn foo f1 {
127          b0(v0: u32):
128            v2 = eq v0, u32 1
129            jmpif v2 then: b1, else: b2
130          b1():
131            v4 = add v0, u32 1
132            jmp b3()
133          b2():
134            v3 = sub v0, u32 1
135            jmp b3()
136          b3():
137            return
138        }
139        ");
140    }
