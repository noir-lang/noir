17impl LoopInvariantContext<'_> {
18    /// Checks whether a binary operation can be evaluated using the bounds of a given loop induction variables.
19    ///
20    /// If it cannot be evaluated, it means that we either have a dynamic loop bound or
21    /// that the operation can potentially overflow during a given loop iteration.
22    pub(super) fn can_evaluate_binary_op(
23        &self,
24        loop_context: &LoopContext,
25        binary: &Binary,
26    ) -> bool {
27        match binary.operator {
28            BinaryOp::Div | BinaryOp::Mod => {
29                // Division can be evaluated if we ensure that the divisor cannot be zero
30                let Some((left, value, lower, upper)) =
31                    self.match_induction_and_constant(loop_context, &binary.lhs, &binary.rhs, true)
32                else {
33                    // Not a constant vs non-constant case, we cannot evaluate it.
34                    return false;
35                };
36
37                if left {
38                    // If the induction variable is on the LHS, we're dividing with a constant.
39                    if !value.is_zero() && !value.is_minus_one() {
40                        return true;
41                    }
42                } else {
43                    // Otherwise we are dividing a constant with the induction variable, and we have to check whether
44                    // at any point in the loop the induction variable can be zero.
45                    let can_be_zero = lower.is_negative() && !upper.is_negative()
46                        || lower.is_zero()
47                        || upper.is_zero();
48
49                    if !can_be_zero {
50                        return true;
51                    }
52                }
53
54                false
55            }
56            // An unchecked operation cannot overflow, so it can be safely evaluated.
57            // Some checked operations can be safely evaluated, depending on the loop bounds, but in that case,
58            // they would have been already converted to unchecked operation in `simplify_induction_variable_in_binary()`.
59            // These are all handled by `requires_acir_gen_predicate`, and are redundant with `can_be_hoisted`.
60            _ => !binary.requires_acir_gen_predicate(&self.inserter.function.dfg),
61        }
62    }

64    /// Some instructions can take advantage of that our induction variable has a fixed minimum/maximum,
65    /// For instance operations can be transformed from a checked operation to an unchecked operation.
66    ///
67    /// Checked operations require more bytecode and thus we aim to minimize their usage wherever possible.
68    ///
69    /// For example, if one side of an add/mul operation is a constant and the other is an induction variable
70    /// with a known upper bound, we know whether that binary operation will ever overflow.
71    /// If we determine that an overflow is not possible we can convert the checked operation to unchecked.
72    ///
73    /// The function returns `false` if the instruction must be added to the block, which involves further simplification.
74    pub(super) fn simplify_from_loop_bounds(
75        &mut self,
76        loop_context: &LoopContext,
77        block_context: &BlockContext,
78        instruction_id: InstructionId,
79    ) -> bool {
80        // Simplify the instruction and update it in the DFG.
81        match self.simplify_induction_variable(loop_context, block_context, instruction_id) {
82            SimplifyResult::SimplifiedTo(id) => {
83                let [result] = self.inserter.function.dfg.instruction_result(instruction_id);
84                self.inserter.map_value(result, id);
85                true
86            }
87            SimplifyResult::SimplifiedToInstruction(instruction) => {
88                self.inserter.function.dfg[instruction_id] = instruction;
89                false
90            }
91            SimplifyResult::None => false,
92            _ => unreachable!(
93                "ICE - loop bounds simplification should only simplify to a value or an instruction"
94            ),
95        }
96    }

98    /// Replace 'assert(invariant != induction)' with assert((invariant < min(induction) || (invariant > max(induction)))
99    /// For this simplification to be valid, we need to ensure that the induction variable takes all the values from min(induction) up to max(induction)
100    /// This means that the assert must be executed at each loop iteration, and that the loop processes all the iteration space.
101    /// This is ensured via control dependence and the check for break patterns, before calling this function.
102    fn simplify_not_equal_constraint(
103        &mut self,
104        loop_context: &LoopContext,
105        lhs: &ValueId,
106        rhs: &ValueId,
107        err: &Option<ConstrainError>,
108        call_stack: CallStackId,
109    ) -> SimplifyResult {
110        let (invariant, min, max) = match self.match_induction_and_invariant(loop_context, lhs, rhs)
111        {
112            Some((true, min, max)) => (rhs, min, max),
113            Some((false, min, max)) => (lhs, min, max),
114            _ => return SimplifyResult::None,
115        };
116
117        let mut insert_binary_to_preheader = |lhs, rhs, operator| {
118            let binary = Instruction::Binary(Binary { lhs, rhs, operator });
119            let results = self
120                .inserter
121                .function
122                .dfg
123                .insert_instruction_and_results(binary, loop_context.pre_header(), None, call_stack)
124                .results();
125            assert!(results.len() == 1);
126            results[0]
127        };
128        // The comparisons can be safely hoisted to the pre-header because they are loop invariant and control independent
129        let check_min = insert_binary_to_preheader(*invariant, min, BinaryOp::Lt);
130        let check_max = insert_binary_to_preheader(max, *invariant, BinaryOp::Lt);
131        let check_bounds = insert_binary_to_preheader(check_min, check_max, BinaryOp::Or);
132
133        SimplifyResult::SimplifiedToInstruction(Instruction::Constrain(
134            check_bounds,
135            self.true_value,
136            err.clone(),
137        ))
138    }

140    /// If the inputs are an induction and loop invariant variables, it returns
141    /// the maximum and minimum values of the induction variable, based on the loop bounds,
142    /// and a boolean indicating if the induction variable is on the lhs or rhs (true for lhs)
143    fn match_induction_and_invariant(
144        &mut self,
145        loop_context: &LoopContext,
146        lhs: &ValueId,
147        rhs: &ValueId,
148    ) -> Option<(bool, ValueId, ValueId)> {
149        let (is_left, lower, upper) = match (
150            loop_context.get_current_induction_variable_bounds(*lhs),
151            loop_context.get_current_induction_variable_bounds(*rhs),
152        ) {
153            (_, Some((lower, upper))) => Some((false, lower, upper)),
154            (Some((lower, upper)), _) => Some((true, lower, upper)),
155            _ => None,
156        }?;
157
158        let (upper_field, upper_type) = upper.dec().into_numeric_constant();
159        let (lower_field, lower_type) = lower.into_numeric_constant();
160
161        let min_iter = self.inserter.function.dfg.make_constant(lower_field, lower_type);
162        let max_iter = self.inserter.function.dfg.make_constant(upper_field, upper_type);
163        if (is_left && loop_context.is_loop_invariant(rhs))
164            || (!is_left && loop_context.is_loop_invariant(lhs))
165        {
166            return Some((is_left, min_iter, max_iter));
167        }
168        None
169    }

171    /// Simplify certain instructions using the lower/upper bounds of induction variables
172    fn simplify_induction_variable(
173        &mut self,
174        loop_context: &LoopContext,
175        block_context: &BlockContext,
176        instruction_id: InstructionId,
177    ) -> SimplifyResult {
178        let (instruction, call_stack) = self.inserter.map_instruction(instruction_id);
179        match &instruction {
180            Instruction::Binary(binary) => self.simplify_induction_variable_in_binary(
181                loop_context,
182                binary,
183                block_context.is_header,
184            ),
185            Instruction::ConstrainNotEqual(x, y, err) => {
186                // Ensure the loop is fully executed
187                if loop_context.no_break
188                    && block_context.can_simplify_control_dependent_instruction()
189                {
190                    assert!(block_context.does_execute, "executing a non executable loop");
191                    self.simplify_not_equal_constraint(loop_context, x, y, err, call_stack)
192                } else {
193                    SimplifyResult::None
194                }
195            }
196            _ => SimplifyResult::None,
197        }
198    }

200    /// If the inputs are an induction variable and a constant, it returns
201    /// the constant value, the maximum and minimum values of the induction variable, based on the loop bounds,
202    /// and a boolean indicating if the induction variable is on the lhs or rhs (true for lhs)
203    /// if `only_outer_induction`, we only consider outer induction variables, else we also consider the induction variables from the current loop.
204    fn match_induction_and_constant(
205        &self,
206        loop_context: &LoopContext,
207        lhs: &ValueId,
208        rhs: &ValueId,
209        only_outer_induction: bool,
210    ) -> Option<(bool, IntegerConstant, IntegerConstant, IntegerConstant)> {
211        let lhs_const = self.inserter.function.dfg.get_integer_constant(*lhs);
212        let rhs_const = self.inserter.function.dfg.get_integer_constant(*rhs);
213        match (
214            lhs_const,
215            rhs_const,
216            loop_context
217                .get_current_induction_variable_bounds(*lhs)
218                .filter(|_| !only_outer_induction)
219                .or(self.outer_induction_variables.get(lhs).copied()),
220            loop_context
221                .get_current_induction_variable_bounds(*rhs)
222                .filter(|_| !only_outer_induction)
223                .or(self.outer_induction_variables.get(rhs).copied()),
224        ) {
225            // LHS is a constant, RHS is the induction variable with a known lower and upper bound.
226            (Some(lhs), None, None, Some((lower_bound, upper_bound))) => {
227                Some((false, lhs, lower_bound, upper_bound))
228            }
229            // RHS is a constant, LHS is the induction variable with a known lower an upper bound
230            (None, Some(rhs), Some((lower_bound, upper_bound)), None) => {
231                Some((true, rhs, lower_bound, upper_bound))
232            }
233            _ => None,
234        }
235    }

237    /// Given a constant `c` and an induction variable `i`:
238    /// - Replace comparisons `i < c` by true if `max(i) < c`, and false if `min(i) >= c`
239    /// - Replace comparisons `c < i` by true if `min(i) > c`, and false if `max(i) <= c`
240    /// - Replace equalities `i == c` by false if `min(i) > c or max(i) < c`
241    /// - Replace checked operations with unchecked version if the induction variable bounds prove that the operation will not overflow
242    ///
243    /// `header` indicates if we are in the loop header where loop bounds do not apply yet
244    fn simplify_induction_variable_in_binary(
245        &self,
246        loop_context: &LoopContext,
247        binary: &Binary,
248        is_header: bool,
249    ) -> SimplifyResult {
250        // Checks the operands are an induction variable and a constant
251        // Note that here we allow all_induction_variables
252        let operand_type = self.inserter.function.dfg.type_of_value(binary.lhs).unwrap_numeric();
253
254        let Some((is_induction_var_lhs, constant, lower_bound, upper_bound)) =
255            self.match_induction_and_constant(loop_context, &binary.lhs, &binary.rhs, is_header)
256        else {
257            return SimplifyResult::None;
258        };
259
260        // Handle arithmetic operations:
261        // Check if we can simplify either `lower op const` or `const op upper` into an unchecked version of the operation.
262        if let Some((lhs, rhs)) = match binary.operator {
263            BinaryOp::Add { unchecked }
264            | BinaryOp::Sub { unchecked }
265            | BinaryOp::Mul { unchecked }
266                if unchecked =>
267            {
268                // Already unchecked, no need to simplify.
269                return SimplifyResult::None;
270            }
271            BinaryOp::Sub { .. } => {
272                if is_induction_var_lhs {
273                    // `i - const` won't overflow if the lowest `i` doesn't.
274                    Some((lower_bound, constant))
275                } else {
276                    // `const - i` won't overflow if the highest `i` doesn't.
277                    Some((constant, upper_bound))
278                }
279            }
280            BinaryOp::Add { .. } | BinaryOp::Mul { .. } => {
281                // `i + const` won't overflow if the highest `i` value doesn't.
282                Some((constant, upper_bound))
283            }
284            BinaryOp::Div | BinaryOp::Mod => return SimplifyResult::None,
285            _ => None,
286        } {
287            // We evaluate this expression using the upper bounds (or lower in the case of sub)
288            // of its inputs to check whether it will ever overflow.
289            // If `eval_constant_binary_op` won't overflow we can simplify the instruction to an unchecked version.
290            let lhs = lhs.into_numeric_constant().0;
291            let rhs = rhs.into_numeric_constant().0;
292            match eval_constant_binary_op(lhs, rhs, binary.operator, operand_type) {
293                BinaryEvaluationResult::Success(..) => {
294                    // Unchecked version of the binary operation
295                    let unchecked = Instruction::Binary(Binary {
296                        operator: binary.operator.into_unchecked(),
297                        lhs: binary.lhs,
298                        rhs: binary.rhs,
299                    });
300                    return SimplifyResult::SimplifiedToInstruction(unchecked);
301                }
302                BinaryEvaluationResult::CouldNotEvaluate | BinaryEvaluationResult::Failure(..) => {
303                    return SimplifyResult::None;
304                }
305            }
306        }
307
308        // Handle comparisons. (The upper_bound is exclusive).
309        match binary.operator {
310            BinaryOp::Eq => {
311                if constant >= upper_bound || constant < lower_bound {
312                    // `i == const` cannot be true if the constant is out of range.
313                    SimplifyResult::SimplifiedTo(self.false_value)
314                } else {
315                    SimplifyResult::None
316                }
317            }
318            BinaryOp::Lt => match is_induction_var_lhs {
319                // `i < const`
320                true if upper_bound <= constant => SimplifyResult::SimplifiedTo(self.true_value),
321                true if lower_bound >= constant => SimplifyResult::SimplifiedTo(self.false_value),
322                // `const < i`
323                false if lower_bound > constant => SimplifyResult::SimplifiedTo(self.true_value),
324                false if upper_bound <= constant.inc() => {
325                    // If `const >= upper_bound - 1` then it will never be less than `i`.
326                    SimplifyResult::SimplifiedTo(self.false_value)
327                }
328                _ => SimplifyResult::None,
329            },
330            _ => SimplifyResult::None,
331        }
332    }
