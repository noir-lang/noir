1//! This file contains the simplify cfg pass of the SSA IR.
2//!
3//! This is a rather simple pass that is expected to be cheap to perform. It:
4//! 1. Removes blocks with no predecessors
5//! 2. Inlines a block into its sole predecessor if that predecessor only has one successor.
6//! 3. Removes any block arguments for blocks with only a single predecessor.
7//! 4. Removes any blocks which have no instructions other than a single terminating jmp.
8//! 5. Replaces any jmpifs with constant conditions with jmps. If this causes the block to have
9//!    only 1 successor then (2) also will be applied.
10//! 6. Replacing any jmpifs with a negated condition with a jmpif with a un-negated condition and reversed branches.
11//! 7. Replaces any jmpif whose branches converge. Convergence is defined as two paths that ultimately jump to the same block through empty jump chains.
12//!    If the branches converge, the jmpif is unnecessary and can be replaced with a simple jmp.
13//!
14//! Currently only 1 is unimplemented.

30impl Ssa {
31    /// See [`simplify_cfg`][self] module for more information
32    #[tracing::instrument(level = "trace", skip(self))]
33    pub(crate) fn simplify_cfg(mut self) -> Self {
34        for function in self.functions.values_mut() {
35            function.simplify_function();
36        }
37        self
38    }
39}

41impl Function {
42    /// Simplify a function's cfg by going through each block to check for any simple blocks that can
43    /// be inlined into their predecessor.
44    pub(crate) fn simplify_function(&mut self) {
45        let mut cfg = ControlFlowGraph::with_function(self);
46        let mut values_to_replace = ValueMapping::default();
47        let mut stack = vec![self.entry_block()];
48        let mut visited = HashSet::new();
49
50        while let Some(block) = stack.pop() {
51            if cfg.predecessors(block).len() == 0 && block != self.entry_block() {
52                // If the block has no predecessors, it's no longer reachable and can be ignored.
53                cfg.invalidate_block_successors(block);
54                continue;
55            }
56
57            if !values_to_replace.is_empty() {
58                self.dfg.replace_values_in_block_instructions(block, &values_to_replace);
59            }
60
61            // First perform any simplifications on the current block, this ensures that we add the proper successors
62            // to the stack.
63            simplify_current_block(self, block, &mut cfg, &mut values_to_replace);
64
65            if visited.insert(block) {
66                stack.extend(self.dfg[block].successors().filter(|block| !visited.contains(block)));
67            }
68
69            let mut predecessors = cfg.predecessors(block);
70            if predecessors.len() == 1 {
71                let predecessor =
72                    predecessors.next().expect("Already checked length of predecessors");
73                drop(predecessors);
74
75                try_inline_successor(self, &mut cfg, predecessor, &mut values_to_replace);
76            } else {
77                drop(predecessors);
78
79                check_for_double_jmp(self, block, &mut cfg);
80            }
81
82            if !values_to_replace.is_empty() {
83                self.dfg.replace_values_in_block_terminator(block, &values_to_replace);
84            }
85        }
86
87        if !values_to_replace.is_empty() {
88            // Values from previous blocks might need to be replaced
89            for block in self.reachable_blocks() {
90                self.dfg.replace_values_in_block(block, &values_to_replace);
91            }
92            self.dfg.data_bus.replace_values(&values_to_replace);
93        }
94    }
95}

97/// A helper function to simplify the current block based on information on its successors.
98///
99/// This function will recursively simplify the current block until no further simplifications can be made.
100fn simplify_current_block(
101    function: &mut Function,
102    block: BasicBlockId,
103    cfg: &mut ControlFlowGraph,
104    values_to_replace: &mut ValueMapping,
105) {
106    // These functions return `true` if they successfully simplified the CFG for the current block.
107    let mut simplified = true;
108
109    while simplified {
110        simplified = check_for_negated_jmpif_condition(function, block, cfg)
111            | check_for_constant_jmpif(function, block, cfg)
112            | check_for_converging_jmpif(function, block, cfg)
113            | try_inline_successor(function, cfg, block, values_to_replace);
114    }
115}

117/// Optimize a jmpif into a jmp if the condition is known
118fn check_for_constant_jmpif(
119    function: &mut Function,
120    block: BasicBlockId,
121    cfg: &mut ControlFlowGraph,
122) -> bool {
123    if let Some(TerminatorInstruction::JmpIf {
124        condition,
125        then_destination,
126        else_destination,
127        call_stack,
128    }) = function.dfg[block].terminator()
129    {
130        if let Some(constant) = function.dfg.get_numeric_constant(*condition) {
131            let (destination, unchosen_destination) = if constant.is_zero() {
132                (*else_destination, *then_destination)
133            } else {
134                (*then_destination, *else_destination)
135            };
136
137            let arguments = Vec::new();
138            let call_stack = *call_stack;
139            let jmp = TerminatorInstruction::Jmp { destination, arguments, call_stack };
140            function.dfg[block].set_terminator(jmp);
141            cfg.recompute_block(function, block);
142
143            // If `block` was the only predecessor to `unchosen_destination` then it's no long reachable through the CFG,
144            // we can then invalidate it successors as it's an invalid predecessor.
145            if cfg.predecessors(unchosen_destination).len() == 0 {
146                cfg.invalidate_block_successors(unchosen_destination);
147            }
148
149            return true;
150        }
151    }
152    false
153}

155/// Optimize a jmp to a block which immediately jmps elsewhere to just jmp to the second block.
156fn check_for_double_jmp(function: &mut Function, block: BasicBlockId, cfg: &mut ControlFlowGraph) {
157    if matches!(function.runtime(), RuntimeType::Acir(_)) {
158        // We can't remove double jumps in ACIR functions as this interferes with the `flatten_cfg` pass.
159        return;
160    }
161
162    // We only want to remove double jumps if the block has no instructions or parameters.
163    if !function.dfg[block].instructions().is_empty()
164        || !function.dfg[block].parameters().is_empty()
165    {
166        return;
167    }
168
169    // We expect the block to have a simple jmp terminator with no arguments.
170    let Some(TerminatorInstruction::Jmp { destination: final_destination, arguments, .. }) =
171        function.dfg[block].terminator()
172    else {
173        return;
174    };
175
176    if !arguments.is_empty() {
177        return;
178    }
179
180    let final_destination = *final_destination;
181
182    // At this point we know that `block` is a simple jmp block with no instructions or parameters.
183    // We can then update all of its predecessors to jump directly to the final destination.
184    let predecessors: Vec<_> = cfg.predecessors(block).collect();
185    for predecessor_block in predecessors {
186        let terminator_instruction = function.dfg[predecessor_block].take_terminator();
187        let redirected_terminator_instruction = match terminator_instruction {
188            TerminatorInstruction::JmpIf {
189                condition,
190                then_destination,
191                else_destination,
192                call_stack,
193            } => {
194                let then_destination =
195                    if then_destination == block { final_destination } else { then_destination };
196                let else_destination =
197                    if else_destination == block { final_destination } else { else_destination };
198                TerminatorInstruction::JmpIf {
199                    condition,
200                    then_destination,
201                    else_destination,
202                    call_stack,
203                }
204            }
205            TerminatorInstruction::Jmp { destination, arguments, call_stack } => {
206                assert_eq!(
207                    destination, block,
208                    "ICE: predecessor block doesn't jump to current block"
209                );
210                assert!(arguments.is_empty(), "ICE: predecessor jmp has arguments");
211                TerminatorInstruction::Jmp { destination: final_destination, arguments, call_stack }
212            }
213            TerminatorInstruction::Return { .. } => {
214                unreachable!("ICE: predecessor block should not have return terminator instruction")
215            }
216            TerminatorInstruction::Unreachable { .. } => {
217                unreachable!(
218                    "ICE: predecessor block should not have unreachable terminator instruction"
219                )
220            }
221        };
222
223        function.dfg[predecessor_block].set_terminator(redirected_terminator_instruction);
224        cfg.recompute_block(function, predecessor_block);
225    }
226    cfg.recompute_block(function, block);
227}

229/// Optimize a jmpif on a negated condition by swapping the branches.
230fn check_for_negated_jmpif_condition(
231    function: &mut Function,
232    block: BasicBlockId,
233    cfg: &mut ControlFlowGraph,
234) -> bool {
235    if matches!(function.runtime(), RuntimeType::Acir(_)) {
236        // Swapping the `then` and `else` branches of a `JmpIf` within an ACIR function
237        // can result in the situation where the branches merge together again in the `then` block, e.g.
238        //
239        // acir(inline) fn main f0 {
240        //   b0(v0: u1):
241        //     jmpif v0 then: b2, else: b1
242        //   b2():
243        //     return
244        //   b1():
245        //     jmp b2()
246        // }
247        //
248        // This breaks the `flatten_cfg` pass as it assumes that merges only happen in
249        // the `else` block or a 3rd block.
250        //
251        // See: https://github.com/noir-lang/noir/pull/5891#issuecomment-2500219428
252        return false;
253    }
254
255    if let Some(TerminatorInstruction::JmpIf {
256        condition,
257        then_destination,
258        else_destination,
259        call_stack,
260    }) = function.dfg[block].terminator()
261    {
262        if let Value::Instruction { instruction, .. } = function.dfg[*condition] {
263            if let Instruction::Not(negated_condition) = function.dfg[instruction] {
264                let call_stack = *call_stack;
265                let jmpif = TerminatorInstruction::JmpIf {
266                    condition: negated_condition,
267                    then_destination: *else_destination,
268                    else_destination: *then_destination,
269                    call_stack,
270                };
271                function.dfg[block].set_terminator(jmpif);
272                cfg.recompute_block(function, block);
273                return true;
274            }
275        }
276    }
277    false
278}

280/// Attempts to simplify a `jmpif` terminator if both branches converge.
281///
282/// We define convergence as when two branches of a `jmpif` ultimately lead to the same
283/// destination block, after following chains of empty blocks. If they do, the conditional
284/// jump is unnecessary and can be replaced with a simple `jmp`.
285fn check_for_converging_jmpif(
286    function: &mut Function,
287    block: BasicBlockId,
288    cfg: &mut ControlFlowGraph,
289) -> bool {
290    if matches!(function.runtime(), RuntimeType::Acir(_)) {
291        // The `flatten_cfg` pass expects two blocks to join to the same block.
292        // If we have a nested if the inner if statement could potentially be a converging jmpif.
293        // This may change the final block we converge into.
294        return false;
295    }
296
297    let Some(TerminatorInstruction::JmpIf {
298        then_destination, else_destination, call_stack, ..
299    }) = function.dfg[block].terminator()
300    else {
301        return false;
302    };
303
304    let then_final = resolve_jmp_chain(function, *then_destination);
305    let else_final = resolve_jmp_chain(function, *else_destination);
306
307    // If both branches end at the same target, we can replace the jmpif with a jmp
308    if then_final == else_final {
309        let jmp = TerminatorInstruction::Jmp {
310            destination: then_final,
311            // The blocks in a jmp chain are checked to have empty arguments by resolve_jmp_chain
312            arguments: Vec::new(),
313            call_stack: *call_stack,
314        };
315        function.dfg[block].set_terminator(jmp);
316        cfg.recompute_block(function, block);
317        true
318    } else {
319        false
320    }
321}

323/// Follow a chain of empty blocks to find the real destination.
324///
325/// This function assumes only [unconditional jumps][TerminatorInstruction::Jmp] are allowed
326/// in a block for chaining. It returns the final destination reached through empty blocks.
327fn resolve_jmp_chain(function: &Function, mut current: BasicBlockId) -> BasicBlockId {
328    // Need to maintain a visited set to prevent infinite cycles
329    let mut visited = HashSet::new();
330
331    while visited.insert(current) {
332        let block = &function.dfg[current];
333        // Exit early if block has instructions or parameters
334        if !block.instructions().is_empty() || !block.parameters().is_empty() {
335            return current;
336        }
337
338        match block.terminator() {
339            Some(TerminatorInstruction::Jmp { destination, arguments, .. })
340                if arguments.is_empty() =>
341            {
342                // Continue following the current chain
343                current = *destination;
344            }
345            _ => return current,
346        }
347    }
348
349    current
350}

352/// If the given block has block parameters, replace them with the jump arguments from the predecessor.
353///
354/// Currently, if this function is needed, `try_inline_into_predecessor` will also always apply,
355/// although in the future it is possible for only this function to apply if jmpif instructions
356/// with block arguments are ever added.
357fn remove_block_parameters(
358    function: &mut Function,
359    block: BasicBlockId,
360    predecessor: BasicBlockId,
361    values_to_replace: &mut ValueMapping,
362) {
363    let block = &mut function.dfg[block];
364
365    if !block.parameters().is_empty() {
366        let block_params = block.take_parameters();
367
368        let jump_args = match function.dfg[predecessor].unwrap_terminator_mut() {
369            TerminatorInstruction::Jmp { arguments, .. } => std::mem::take(arguments),
370            TerminatorInstruction::JmpIf { .. } => unreachable!(
371                "If jmpif instructions are modified to support block arguments in the future, this match will need to be updated"
372            ),
373            _ => unreachable!(
374                "Predecessor was already validated to have only a single jmp destination"
375            ),
376        };
377
378        assert_eq!(block_params.len(), jump_args.len());
379        for (param, arg) in block_params.iter().zip(jump_args) {
380            values_to_replace.insert(*param, arg);
381        }
382    }
383}

385/// Try to inline a block into its predecessor, returning true if successful.
386///
387/// This will only occur if the predecessor's only successor is the given block.
388/// It is also expected that the given block's only predecessor is the given one.
389fn try_inline_successor(
390    function: &mut Function,
391    cfg: &mut ControlFlowGraph,
392    block: BasicBlockId,
393    values_to_replace: &mut ValueMapping,
394) -> bool {
395    if let Some(TerminatorInstruction::Jmp { destination, .. }) = function.dfg[block].terminator() {
396        let destination = *destination;
397        let predecessors = cfg.predecessors(destination);
398        if predecessors.len() == 1 {
399            drop(predecessors);
400
401            // If the block has only 1 predecessor, we can safely remove its block parameters
402            remove_block_parameters(function, destination, block, values_to_replace);
403
404            // Note: this function relies on `remove_block_parameters` being called first.
405            // Otherwise the inlined block will refer to parameters that no longer exist.
406            //
407            // If successful, `block` will be empty and unreachable after this call, so any
408            // optimizations performed after this point on the same block should check if
409            // the inlining here was successful before continuing.
410            try_inline_into_predecessor(function, cfg, destination, block)
411        } else {
412            false
413        }
414    } else {
415        false
416    }
417}

419/// Try to inline a block into its predecessor, returning true if successful.
420///
421/// This will only occur if the predecessor's only successor is the given block.
422/// It is also expected that the given block's only predecessor is the given one.
423fn try_inline_into_predecessor(
424    function: &mut Function,
425    cfg: &mut ControlFlowGraph,
426    block: BasicBlockId,
427    predecessor: BasicBlockId,
428) -> bool {
429    let mut successors = cfg.successors(predecessor);
430    if successors.len() == 1 && successors.next() == Some(block) {
431        drop(successors);
432        function.dfg.inline_block(block, predecessor);
433
434        cfg.recompute_block(function, block);
435        cfg.recompute_block(function, predecessor);
436        true
437    } else {
438        false
439    }
440}

859        // Simplifying the CFG from a block can result the block being updated to a form which can be simplified further.
860        // We want to ensure that we handle any followup simplifications correctly.
861        //
862        // In this case we have a jmpif which is simplified to a jmp, which then can be inlined into its predecessor.
863        // The new terminator instruction of the block is then a jmpif which can be simplified to a jmp.

879            return
880        }
881        "#;
882
883        let ssa = Ssa::from_str(src).unwrap();
884        let ssa = ssa.simplify_cfg();
885
886        assert_ssa_snapshot!(ssa, @r"
887        brillig(inline) impure fn main f0 {
