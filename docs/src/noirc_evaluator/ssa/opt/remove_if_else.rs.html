1//! This file contains the SSA `remove_if_else` pass - a required pass for ACIR to remove any
2//! remaining `Instruction::IfElse` in the singular program-function, and replace them with
3//! arithmetic operations using the `then_condition`.
4//!
5//! ACIR/Brillig differences within this pass:
6//!   - This pass is strictly ACIR-only and never mutates brillig functions.
7//!
8//!
9//! Conditions:
10//!   - Precondition: Flatten CFG has been performed which should result in the function having only
11//!     one basic block.
12//!   - Precondition: `then_value` and `else_value` of `Instruction::IfElse` return arrays or vectors.
13//!     Numeric values should be handled previously by the flattening pass.
14//!     Reference or function values are not handled by remove if-else and will cause an error.
15//!   - Postcondition: A program without any `IfElse` instructions.
16//!
17//! Relevance to other passes:
18//!   - Flattening inserts `Instruction::IfElse` to merge array or vector values from an
19//!     if-expression's "then" and "else" branches. `Instruction::IfElse` with numeric values are
20//!     directly handled during flattening, [via instruction simplification][crate::ssa::ir::dfg::simplify::simplify],
21//!     and will cause a panic in the `remove_if_else` pass.
22//!   - Defunctionalize removes first-class function values from the program which eliminates the need
23//!     for remove-if-else to handle `Instruction::IfElse` returning function values.
24//!
25//! Implementation details & examples:
26//! `IfElse` instructions choose between its two operand values,
27//! `then_value` and `else_value`, based on the `then_condition`:
28//! ```ssa
29//!  if then_condition {
30//!      then_value
31//!  } else {
32//!      else_value
33//!  }
34//! ```
35//!
36//! These instructions are inserted during the flatten cfg pass, which convert conditional control flow
37//! at the basic block level into simple ternary operations returning a value, using these IfElse instructions,
38//! and leaving only one basic block. The flatten cfg pass directly handles numeric values and issues
39//! `Instruction::IfElse` only for arrays and vectors. The remove-if-else pass is used for array and vectors
40//! in order to track their lengths, depending on existing vector intrinsics which modify vectors,
41//! or the array set instructions.
42//! The `Instruction::IfElse` is removed using a `ValueMerger` which operates recursively for nested arrays/vectors.
43//!
44//! For example, this code:
45//! ```noir
46//! fn main(x: bool, mut y: [u32; 2]) {
47//!     if x {
48//!          y[0] = 1;
49//!     } else {
50//!          y[0] = 2;
51//!     }
52//!
53//!     assert(y[0] == 3);
54//!  }
55//!  ```
56//!
57//! will be translated into this code, where the `IfElse` instruction: `v9 = if v0 then v5 else (if v6) v8`
58//! is using array v5 from then branch, and array v8 from the else branch:
59//! ```ssa
60//! acir(inline) predicate_pure fn main f0 {
61//!   b0(v0: u1, v1: [u32; 2]):
62//!     v2 = allocate -> &mut [u32; 2]
63//!     enable_side_effects v0
64//!     v5 = array_set v1, index u32 0, value u32 1
65//!     v6 = not v0
66//!     enable_side_effects v6
67//!     v8 = array_set v1, index u32 0, value u32 2
68//!     v9 = if v0 then v5 else (if v6) v8
69//!     enable_side_effects u1 1
70//!     v11 = array_get v9, index u32 0 -> u32
71//!     constrain v11 == u32 3
72//!     return
73//! }
74//! ```
75//!
76//! The IfElse instruction is then replaced by these instruction during the remove if-else pass:
77//! ```ssa
78//! v13 = cast v0 as u32
79//! v14 = cast v6 as u32
80//! v15 = unchecked_mul v14, u32 2
81//! v16 = unchecked_add v13, v15
82//! v17 = array_get v5, index u32 1 -> u32
83//! v18 = array_get v8, index u32 1 -> u32
84//! v19 = cast v0 as u32
85//! v20 = cast v6 as u32
86//! v21 = unchecked_mul v19, v17
87//! v22 = unchecked_mul v20, v18
88//! v23 = unchecked_add v21, v22
89//! v24 = make_array [v16, v23] : [u32; 2]
90//! ```
91//!
92//! The result of the removed `IfElse` instruction, array `v24`, is a merge of each of the elements of `v5` and `v8`.
93//! The elements at index 0 are replaced by their known value, instead of doing an additional array get.
94//! Operations with the conditions are unchecked operations, because the conditions are 0 or 1, so it cannot overflow.
95//!
96//! For vectors the logic is similar except that vector lengths need to be tracked in order to know
97//! the length of the merged vector resulting in a `make_array` instruction. This length will be the
98//! maximum length of the two input vectors. Note that the actual length of the merged vector should
99//! have been merged during flattening.

124impl Ssa {
125    /// Replaces all `Instruction::IfElse` instructions with the result of a
126    /// value merger of the then and else values. The specifics of the value merger
127    /// depends on the type but is expected to be an equivalent value to the IfElse.
128    /// For example, on integers, the merger will be:
129    /// `then_condition * then_value + !then_condition * else_value`
130    /// which should zero out the branch that was not taken.
131    ///
132    /// In general this is not possible for all types - notably references - which is
133    /// why the Noir frontend does not allow references to be returned from if expressions.
134    ///
135    /// Also note that `Instruction::IfElse` are first inserted after the flattening pass,
136    /// so before then this pass will have no effect.
137    #[tracing::instrument(level = "trace", skip(self))]
138    pub(crate) fn remove_if_else(mut self) -> RtResult<Ssa> {
139        for function in self.functions.values_mut() {
140            function.remove_if_else()?;
141        }
142        Ok(self)
143    }
144}
145
146impl Function {
147    pub(crate) fn remove_if_else(&mut self) -> RtResult<()> {
148        if self.runtime().is_brillig() {
149            return Ok(());
150        }
151
152        #[cfg(debug_assertions)]
153        remove_if_else_pre_check(self);
154
155        Context::default().remove_if_else(self)?;
156
157        #[cfg(debug_assertions)]
158        remove_if_else_post_check(self);
159        Ok(())
160    }
161}

163#[derive(Default)]
164struct Context {
165    /// Keeps track of each size a vector is known to be.
166    ///
167    /// This is passed to the `ValueMerger` because when merging two vectors
168    /// we need to know their sizes to create the merged vector.
169    ///
170    /// Note: as this pass operates on a single block, which is an entry block,
171    /// and because vectors are disallowed in entry blocks, all vector lengths
172    /// should be known at this point.
173    vector_sizes: HashMap<ValueId, SemanticLength>,
174}
175
176impl Context {
177    /// Process each instruction in the entry block of the (fully flattened) function.
178    /// Merge any `IfElse` instruction using a `ValueMerger` and track vector sizes
179    /// through intrinsic calls and array set instructions.
180    fn remove_if_else(&mut self, function: &mut Function) -> RtResult<()> {
181        let block = function.entry_block();
182
183        // Early return if there is no IfElse instruction.
184        if !function.dfg[block]
185            .instructions()
186            .iter()
187            .any(|inst| matches!(function.dfg[*inst], Instruction::IfElse { .. }))
188        {
189            return Ok(());
190        }
191
192        // Keeps track of side effect vars associated to each `array_set` instruction.
193        let mut array_set_predicates = std::collections::HashMap::new();
194
195        function.simple_optimization_result(|context| {
196            let instruction_id = context.instruction_id;
197            let instruction = context.instruction();
198
199            match instruction {
200                Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
201                    let then_condition = *then_condition;
202                    let else_condition = *else_condition;
203                    let then_value = *then_value;
204                    let else_value = *else_value;
205
206                    // Register values for the merger to use.
207                    self.ensure_capacity(context.dfg, then_value);
208                    self.ensure_capacity(context.dfg, else_value);
209
210                    // Because the ValueMerger might produce some `array_get` instructions, we
211                    // need those to always execute as otherwise they'll produce incorrect
212                    // merged arrays. For this, we set the side effects var to `true` for the merge.
213                    let old_side_effects = context.enable_side_effects;
214                    let old_side_effects_is_not_one = context
215                        .dfg
216                        .get_numeric_constant(old_side_effects)
217                        .is_none_or(|value| !value.is_one());
218
219                    if old_side_effects_is_not_one {
220                        let one =
221                            context.dfg.make_constant(FieldElement::one(), NumericType::bool());
222                        let _ = context.insert_instruction(
223                            Instruction::EnableSideEffectsIf { condition: one },
224                            None,
225                        );
226                    }
227
228                    let call_stack = context.dfg.get_instruction_call_stack_id(instruction_id);
229                    let array_get_optimization_data = Some(ArrayGetOptimizationSideEffects {
230                        side_effects_var: context.enable_side_effects,
231                        array_set_predicates: &array_set_predicates,
232                    });
233                    let mut value_merger = ValueMerger::new(
234                        context.dfg,
235                        block,
236                        &self.vector_sizes,
237                        call_stack,
238                        array_get_optimization_data,
239                    );
240
241                    let value = value_merger.merge_values(
242                        then_condition,
243                        else_condition,
244                        then_value,
245                        else_value,
246                    )?;
247
248                    if old_side_effects_is_not_one {
249                        let _ = context.insert_instruction(
250                            Instruction::EnableSideEffectsIf { condition: old_side_effects },
251                            None,
252                        );
253                    }
254
255                    let [result] = context.dfg.instruction_result(instruction_id);
256
257                    context.remove_current_instruction();
258                    // The `IfElse` instruction is replaced by the merge done with the `ValueMerger`
259                    context.replace_value(result, value);
260                }
261                Instruction::Call { func, arguments } => {
262                    // Track vector sizes through intrinsic calls
263                    if let Value::Intrinsic(intrinsic) = context.dfg[*func] {
264                        let results = context.dfg.instruction_results(instruction_id);
265
266                        self.vector_constant_size_override(context.dfg, intrinsic, arguments);
267
268                        let size_change =
269                            self.vector_capacity_change(context.dfg, intrinsic, arguments, results);
270
271                        self.change_size(size_change, context);
272                    }
273                }
274                // Track vector sizes through array set instructions
275                Instruction::ArraySet { array, .. } => {
276                    array_set_predicates.insert(instruction_id, context.enable_side_effects);
277
278                    let [result] = context.dfg.instruction_result(instruction_id);
279                    self.set_capacity(context.dfg, *array, result, |c| c);
280                }
281                _ => (),
282            }
283            Ok(())
284        })
285    }
286
287    fn change_size(&mut self, size_change: SizeChange, context: &mut SimpleOptimizationContext) {
288        match size_change {
289            SizeChange::None => (),
290            SizeChange::SetTo { old, new } => {
291                self.set_capacity(context.dfg, old, new, |c| c);
292            }
293            SizeChange::Inc { old, new } => {
294                self.set_capacity(context.dfg, old, new, |c| {
295                    // Checked addition because increasing the capacity must increase it (cannot wrap around or saturate).
296                    SemanticLength(c.0.checked_add(1).expect("Vector capacity overflow"))
297                });
298            }
299            SizeChange::Dec { old, new } => {
300                // We use a saturating sub here as calling `pop_front` or `pop_back` on a zero-length vector
301                // would otherwise underflow.
302                self.set_capacity(context.dfg, old, new, |c| SemanticLength(c.0.saturating_sub(1)));
303            }
304            SizeChange::Many(changes) => {
305                for change in changes {
306                    self.change_size(change, context);
307                }
308            }
309        }
310    }

312    /// Set the capacity of the new vector based on the capacity of the old array/vector.
313    fn set_capacity(
314        &mut self,
315        dfg: &DataFlowGraph,
316        old: ValueId,
317        new: ValueId,
318        f: impl Fn(SemanticLength) -> SemanticLength,
319    ) {
320        // No need to store the capacity of arrays, only vectors.
321        if !matches!(dfg.type_of_value(new), Type::Vector(_)) {
322            return;
323        }
324        let capacity = self.get_or_find_capacity(dfg, old);
325        self.vector_sizes.insert(new, f(capacity));
326    }
327
328    /// Make sure the vector capacity is recorded.
329    fn ensure_capacity(&mut self, dfg: &DataFlowGraph, vector: ValueId) {
330        self.set_capacity(dfg, vector, vector, |c| c);
331    }
332
333    /// Get the tracked size of array/vectors, or retrieve (and track) it for arrays.
334    fn get_or_find_capacity(&mut self, dfg: &DataFlowGraph, value: ValueId) -> SemanticLength {
335        match self.vector_sizes.entry(value) {
336            Entry::Occupied(entry) => *entry.get(),
337            Entry::Vacant(entry) => {
338                if let Some(length) = dfg.try_get_vector_capacity(value) {
339                    return *entry.insert(length);
340                }
341                // For non-constant vectors we can't tell the size, which would mean we can't merge it.
342                let dbg_value = &dfg[value];
343                unreachable!("ICE: No size for vector {value} = {dbg_value:?}")
344            }
345        }
346    }

348    /// If we have already determined a constant for the vector length, we can override the backing capacity
349    /// of the vector contents. There is no need to use the backing capacity if we have already determined the actual length of the vector.
350    /// In these situations, using the capacity over the vector length would require laying down more instructions to handle the extra padding
351    /// while preventing downstream passes or runtimes from implementing optimizations using the vector length.
352    fn vector_constant_size_override(
353        &mut self,
354        dfg: &DataFlowGraph,
355        intrinsic: Intrinsic,
356        arguments: &[ValueId],
357    ) {
358        match intrinsic {
359            Intrinsic::VectorPushBack
360            | Intrinsic::VectorPushFront
361            | Intrinsic::VectorInsert
362            | Intrinsic::VectorPopBack
363            | Intrinsic::VectorRemove
364            | Intrinsic::VectorPopFront => {
365                if let Some(const_len) = dfg.get_numeric_constant(arguments[0]) {
366                    self.vector_sizes.insert(
367                        arguments[1],
368                        SemanticLength(const_len.try_to_u32().expect("Type should be u32")),
369                    );
370                }
371            }
372            Intrinsic::Hint(Hint::BlackBox) => {
373                // Try to set the length of any vector argument to be that of the preceding constant.
374                let arguments_types =
375                    arguments.iter().map(|x| dfg.type_of_value(*x)).collect::<Vec<_>>();
376
377                for (i, argument) in arguments.iter().enumerate().skip(1) {
378                    if !matches!(arguments_types[i], Type::Vector(_)) {
379                        continue;
380                    }
381                    assert!(matches!(arguments_types[i - 1], Type::Numeric(_)));
382                    if let Some(const_len) = dfg.get_numeric_constant(arguments[i - 1]) {
383                        self.vector_sizes.insert(
384                            *argument,
385                            SemanticLength(const_len.try_to_u32().expect("Type should be u32")),
386                        );
387                    }
388                }
389            }
390            _ => {}
391        }
392    }

394    /// Find the change to a vector's capacity an instruction would have
395    fn vector_capacity_change(
396        &self,
397        dfg: &DataFlowGraph,
398        intrinsic: Intrinsic,
399        arguments: &[ValueId],
400        results: &[ValueId],
401    ) -> SizeChange {
402        match intrinsic {
403            Intrinsic::VectorPushBack | Intrinsic::VectorPushFront | Intrinsic::VectorInsert => {
404                // All of these return `Self` (the vector), we are expecting: len, vector = ...
405                assert_eq!(results.len(), 2);
406                let old = arguments[1];
407                let new = results[1];
408                assert!(matches!(dfg.type_of_value(old), Type::Vector(_)));
409                assert!(matches!(dfg.type_of_value(new), Type::Vector(_)));
410                SizeChange::Inc { old, new }
411            }
412
413            Intrinsic::VectorPopBack | Intrinsic::VectorRemove => {
414                // fn pop_back(self) -> (Self, T)
415                // fn remove(self, index: u32) -> (Self, T)
416                //
417                // These functions return the vector as the result `(len, vector, ...item)`,
418                // so the vector is the second result.
419                let old = arguments[1];
420                let new = results[1];
421                assert!(matches!(dfg.type_of_value(old), Type::Vector(_)));
422                assert!(matches!(dfg.type_of_value(new), Type::Vector(_)));
423                SizeChange::Dec { old, new }
424            }
425
426            Intrinsic::VectorPopFront => {
427                // fn pop_front(self) -> (T, Self)
428                //
429                // These functions return the vector as the result `(...item, len, vector)`,
430                // so the vector is the last result.
431                let old = arguments[1];
432                let new = results[results.len() - 1];
433                assert!(matches!(dfg.type_of_value(old), Type::Vector(_)));
434                assert!(matches!(dfg.type_of_value(new), Type::Vector(_)));
435                SizeChange::Dec { old, new }
436            }
437
438            Intrinsic::AsVector => {
439                assert_eq!(arguments.len(), 1);
440                assert_eq!(results.len(), 2);
441                let old = arguments[0];
442                let new = results[1];
443                assert!(matches!(dfg.type_of_value(old), Type::Array(_, _)));
444                assert!(matches!(dfg.type_of_value(new), Type::Vector(_)));
445                SizeChange::SetTo { old, new }
446            }
447
448            Intrinsic::Hint(Hint::BlackBox) => {
449                assert_eq!(arguments.len(), results.len());
450                let arguments_types =
451                    arguments.iter().map(|x| dfg.type_of_value(*x)).collect::<Vec<_>>();
452                let results_types =
453                    results.iter().map(|x| dfg.type_of_value(*x)).collect::<Vec<_>>();
454
455                assert_eq!(arguments_types, results_types);
456
457                let mut changes = Vec::new();
458                for (i, argument) in arguments.iter().enumerate() {
459                    if self.vector_sizes.contains_key(argument)
460                        && matches!(arguments_types[i], Type::Vector(_))
461                    {
462                        assert!(matches!(arguments_types[i - 1], Type::Numeric(_)));
463                        let new = results[i];
464                        changes.push(SizeChange::SetTo { old: *argument, new });
465                    }
466                }
467
468                SizeChange::Many(changes)
469            }
470
471            // These cases don't affect vector capacities
472            Intrinsic::AssertConstant
473            | Intrinsic::StaticAssert
474            | Intrinsic::ApplyRangeConstraint
475            | Intrinsic::ArrayLen
476            | Intrinsic::ArrayAsStrUnchecked
477            | Intrinsic::StrAsBytes
478            | Intrinsic::BlackBox(_)
479            | Intrinsic::AsWitness
480            | Intrinsic::IsUnconstrained
481            | Intrinsic::DerivePedersenGenerators
482            | Intrinsic::ToBits(_)
483            | Intrinsic::ToRadix(_)
484            | Intrinsic::ArrayRefCount
485            | Intrinsic::VectorRefCount
486            | Intrinsic::FieldLessThan => SizeChange::None,
487        }
488    }
489}
490
491enum SizeChange {
492    None,
493    /// Make the size of the new vector equal to the old array.
494    SetTo {
495        old: ValueId,
496        new: ValueId,
497    },
498    /// Make the size of the new vector equal to old+1.
499    Inc {
500        old: ValueId,
501        new: ValueId,
502    },
503    /// Make the size of the new vector equal to old-1.
504    Dec {
505        old: ValueId,
506        new: ValueId,
507    },
508    Many(Vec<SizeChange>),
509}

511#[cfg(debug_assertions)]
512fn remove_if_else_pre_check(func: &Function) {
513    // flatten_cfg must have run
514    super::checks::assert_cfg_is_flattened(func);
515    // IfElse should only be on arrays/vectors, not numeric types
516    super::checks::for_each_instruction(func, |instruction, dfg| {
517        super::checks::assert_not_if_else_on_numeric(instruction, dfg);
518    });
519}
520
521/// Post-check condition for [Function::remove_if_else].
522///
523/// Succeeds if:
524///   - `func` is a Brillig function, OR
525///   - `func` does not contain any if-else instructions.
526///
527/// Otherwise panics.
528#[cfg(debug_assertions)]
529fn remove_if_else_post_check(func: &Function) {
530    // All IfElse instructions should be removed
531    super::checks::for_each_instruction(func, |instruction, _dfg| {
532        super::checks::assert_not_if_else(instruction);
533    });
534}

548        // This is the flattened SSA for the following Noir logic:
549        // ```
550        // fn main(x: bool, mut y: [u32; 2]) {
551        //     if x {
552        //         y[0] = 2;
553        //         y[1] = 3;
554        //     }
555        //
556        //     let z = y[0] + y[1];
557        //     assert(z == 5);
558        // }
559        // ```
560        let src = "
561        acir(inline) predicate_pure fn main f0 {
562          b0(v0: u1, v1: [u32; 2]):
563            v2 = allocate -> &mut [u32; 2]
564            enable_side_effects v0
565            v5 = array_set v1, index u32 0, value u32 2
566            v7 = array_set v5, index u32 1, value u32 3
567            v8 = not v0
568            v9 = if v0 then v7 else (if v8) v1
569            enable_side_effects u1 1
570            v11 = array_get v9, index u32 0 -> u32
571            v12 = array_get v9, index u32 1 -> u32
572            v13 = add v11, v12
573            v15 = eq v13, u32 5
574            constrain v13 == u32 5
575            return
576        }
577        ";
578
579        let mut ssa = Ssa::from_str(src).unwrap();
580        ssa = ssa.remove_if_else().unwrap();
581
582        // In case our if block is never activated, we need to fetch each value from the original array.
583        // We then should create a new array where each value can be mapped to `(then_condition * then_value) + (!then_condition * else_value)`.

588          b0(v0: u1, v1: [u32; 2]):
589            v2 = allocate -> &mut [u32; 2]
590            enable_side_effects v0
591            v5 = array_set v1, index u32 0, value u32 2
592            v8 = array_set v5, index u32 1, value u32 3
593            v9 = not v0
594            enable_side_effects u1 1
595            v11 = array_get v1, index u32 0 -> u32
596            v12 = cast v0 as u32
597            v13 = cast v9 as u32
598            v14 = unchecked_mul v12, u32 2
599            v15 = unchecked_mul v13, v11
600            v16 = unchecked_add v14, v15
601            v17 = array_get v1, index u32 1 -> u32
602            v18 = cast v0 as u32
603            v19 = cast v9 as u32
604            v20 = unchecked_mul v18, u32 3
605            v21 = unchecked_mul v19, v17
606            v22 = unchecked_add v20, v21
607            v23 = make_array [v16, v22] : [u32; 2]
608            enable_side_effects v0
609            enable_side_effects u1 1
610            v24 = add v16, v22
611            v26 = eq v24, u32 5
612            constrain v24 == u32 5
613            return
614        }
615        ");
616    }
