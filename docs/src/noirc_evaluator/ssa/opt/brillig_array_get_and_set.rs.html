1//! This SSA pass adjusts constant indexes of array operations inside Brillig functions
2//! to avoid performing an extra binary operation.
3//!
4//! For example, if we have this SSA:
5//!
6//! ```ssa
7//! b0(v0: [Field; 10]):
8//!   v1 = array_get v0, index u32 3 -> Field
9//! ```
10//!
11//! Brillig would have to fetch the element at index 3 of the array. However,
12//! in the Brillig runtime arrays are represented as [RC, ...items],
13//! where `RC` holds the reference count of the array. That means that the final
14//! index that needs to be retrieved is 4, not 3. With the above operation
15//! the final Brillig code would have to add 1 to 3 to get the desired element.
16//!
17//! So, this pass will transform the above SSA into this:
18//!
19//! ```ssa
20//! b0(v0: [Field; 10]):
21//!   v1 = array_get v0, index u32 4 minus 1 -> Field
22//! ```
23//!
24//! The "minus 1" part is just there so that readers can understand that the index
25//! was offset and that the actual element index is 3. On the Brillig side,
26//! array operations with constant indexes are always assumed to have already been
27//! shifted.
28//!
29//! Now the index to retrieve is 4 and there's no need to offset it in Brillig,
30//! avoiding one addition.
31//!
32//! In the case of vectors, they are represented as Brillig vectors as [RC, Size, Capacity, ...items],
33//! thus the items pointer instead starts at three rather than one. So for a vector
34//! this pass will transform this:
35//!
36//! ```ssa
37//! b0(v0: [Field]):
38//!   v1 = array_get v0, index u32 3 -> Field
39//! ```
40//!
41//! to this:
42//!
43//! ```ssa
44//! b0(v0: [Field]):
45//!   v1 = array_get v0, index u32 6 minus 3 -> Field
46//! ```
47//!
48//! This pass must be the very last, just before generating ACIR and Brillig opcodes from the SSA.
49//! Shifting indexes can break some of the assumptions of earlier compiler operations, so this is
50//! done outside the normal SSA pipeline, as an internal step, and must be run only once.
51//!
52//! The main motivation behind this pass is to make it possible to reuse the same constants across
53//! Brillig opcodes without having to re-allocate another register for them every time they appear.
54//! For this to happen the constants need to be part of the DFG, where they can be hoisted and
55//! deduplicated across instructions. Doing this during Brillig codegen would be too late, as at
56//! that time we have a read-only DFG and we would be forced to generate more Brillig opcodes.
57
58use crate::{
59    brillig::brillig_ir::BRILLIG_MEMORY_ADDRESSING_BIT_SIZE,
60    ssa::{
61        Ssa,
62        ir::{function::Function, instruction::Instruction, types::NumericType, value::ValueId},
63    },
64};
65
66use super::simple_optimization::SimpleOptimizationContext;
67
68impl Ssa {
69    /// See [`brillig_array_get_and_set`][self] module for more information.
70    #[tracing::instrument(level = "trace", skip(self))]
71    pub(crate) fn brillig_array_get_and_set(mut self) -> Ssa {
72        for function in self.functions.values_mut() {
73            function.brillig_array_get_and_set();
74        }
75
76        self
77    }
78}
79
80impl Function {
81    fn brillig_array_get_and_set(&mut self) {
82        if !self.runtime().is_brillig() {
83            return;
84        }
85
86        assert!(!self.dfg.brillig_arrays_offset, "Brillig arrays can be offset at most once!");
87        self.dfg.brillig_arrays_offset = true;
88
89        self.simple_optimization(|context| {
90            let instruction = context.instruction();
91            match instruction {
92                Instruction::ArrayGet { array, index } => {
93                    let array = *array;
94                    let index = *index;
95                    let Some(index) = compute_offset_index(context, array, index) else {
96                        return;
97                    };
98                    let new_instruction = Instruction::ArrayGet { array, index };
99                    context.replace_current_instruction_with(new_instruction);
100                }
101                Instruction::ArraySet { array, index, value, mutable } => {
102                    let array = *array;
103                    let index = *index;
104                    let value = *value;
105                    let mutable = *mutable;
106                    let Some(index) = compute_offset_index(context, array, index) else {
107                        return;
108                    };
109                    let new_instruction = Instruction::ArraySet { array, index, value, mutable };
110                    context.replace_current_instruction_with(new_instruction);
111                }
112                _ => (),
113            }
114        });
115    }
116}
117
118/// Given an array or vector value and a constant index, returns an offset (shifted) index.
119fn compute_offset_index(
120    context: &mut SimpleOptimizationContext,
121    array_or_vector: ValueId,
122    index: ValueId,
123) -> Option<ValueId> {
124    let constant_index = context.dfg.get_numeric_constant(index)?;
125    let offset = context.dfg.array_offset(array_or_vector, index);
126    let index = context.dfg.make_constant(
127        constant_index + offset.to_u32().into(),
128        NumericType::unsigned(BRILLIG_MEMORY_ADDRESSING_BIT_SIZE),
129    );
130    Some(index)
131}

275    // This test is here to demonstrate how trying to use the common machinery
276    // after this pass would lead to unexpected results.
277    #[test]
278    fn is_safe_index_unexpected_after_pass() {
279        let src = "
280        brillig(inline) fn main f0 {
281          b0(v0: [Field; 1]):
282            v1 = array_get v0, index u32 0 -> Field
283            return v1
284        }
285        ";
286
287        fn has_side_effects(ssa: &Ssa) -> bool {
288            let func = ssa.main();
289            let b0 = &func.dfg[func.entry_block()];
290            let instruction = &func.dfg[b0.instructions()[0]];
291            instruction.has_side_effects(&func.dfg)
292        }
293
294        let ssa = Ssa::from_str(src).unwrap();
295
296        assert!(
297            !has_side_effects(&ssa),
298            "Indexing 1-element array with index 0 should be safe and have no side effects."
299        );
300
301        let ssa = ssa.brillig_array_get_and_set();
302
303        assert!(
304            has_side_effects(&ssa),
305            "It should have no side effects, but the index is now considered unsafe."
306        );
307    }
