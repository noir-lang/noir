16/// An SSA pass reified as a construct we can put into a list,
17/// which facilitates equivalence testing between different
18/// stages of the processing pipeline.
19pub struct SsaPass<'a> {
20    msg: &'static str,
21    run: Box<dyn Fn(Ssa) -> SsaPassResult + 'a>,
22}
23
24impl<'a> SsaPass<'a> {
25    pub fn new<F>(f: F, msg: &'static str) -> Self
26    where
27        F: Fn(Ssa) -> Ssa + 'a,
28    {
29        Self::new_try(move |ssa| Ok(f(ssa)), msg)
30    }
31
32    pub fn new_try<F>(f: F, msg: &'static str) -> Self
33    where
34        F: Fn(Ssa) -> SsaPassResult + 'a,
35    {
36        Self { msg, run: Box::new(f) }
37    }
38
39    pub fn msg(&self) -> &str {
40        self.msg
41    }
42
43    pub fn run(&self, ssa: Ssa) -> SsaPassResult {
44        (self.run)(ssa)
45    }
46
47    /// Follow up the pass with another one, without adding a separate message for it.
48    ///
49    /// This is useful for attaching cleanup passes that we don't want to appear on their
50    /// own in the pipeline, because it would just increase the noise.
51    pub fn and_then<F>(self, f: F) -> Self
52    where
53        F: Fn(Ssa) -> Ssa + 'a,
54    {
55        self.and_then_try(move |ssa| Ok(f(ssa)))
56    }
57
58    /// Same as `and_then` but for passes that can fail.
59    pub fn and_then_try<F>(self, f: F) -> Self
60    where
61        F: Fn(Ssa) -> SsaPassResult + 'a,
62    {
63        Self {
64            msg: self.msg,
65            run: Box::new(move |ssa| {
66                let ssa = self.run(ssa)?;
67                let ssa = f(ssa)?;
68                Ok(ssa)
69            }),
70        }
71    }
72}

74// This is just a convenience object to bundle the ssa with `print_ssa_passes` for debug printing.
75pub struct SsaBuilder<'local> {
76    /// The SSA being built; it is the input and the output of every pass ran by the builder.
77    ssa: Ssa,
78    /// Options to control which SSA passes to print.
79    ssa_logging: SsaLogging,
80    /// Whether to print the amount of time it took to run individual SSA passes.
81    print_codegen_timings: bool,
82    /// Counters indexed by the message in the SSA pass, so we can distinguish between multiple
83    /// runs of the same pass in the printed messages.
84    passed: HashMap<String, usize>,
85    /// List of SSA pass message fragments that we want to skip, for testing purposes.
86    skip_passes: Vec<String>,
87
88    /// Providing a file manager is optional - if provided it can be used to print source
89    /// locations along with each ssa instructions when debugging.
90    files: Option<&'local fm::FileManager>,
91}
92
93impl<'local> SsaBuilder<'local> {
94    pub fn from_program(
95        program: Program,
96        ssa_logging: SsaLogging,
97        print_codegen_timings: bool,
98        emit_ssa: &Option<PathBuf>,
99        files: Option<&'local fm::FileManager>,
100    ) -> Result<Self, RuntimeError> {
101        let ssa = generate_ssa(program)?;
102        if let Some(emit_ssa) = emit_ssa {
103            let mut emit_ssa_dir = emit_ssa.clone();
104            // We expect the full package artifact path to be passed in here,
105            // and attempt to create the target directory if it does not exist.
106            emit_ssa_dir.pop();
107            create_named_dir(emit_ssa_dir.as_ref(), "target");
108            let ssa_path = emit_ssa.with_extension("ssa.json");
109            write_to_file(&serde_json::to_vec(&ssa).unwrap(), &ssa_path);
110        }
111        Ok(Self::from_ssa(ssa, ssa_logging, print_codegen_timings, files).print("Initial SSA"))
112    }
113
114    pub fn from_ssa(
115        ssa: Ssa,
116        ssa_logging: SsaLogging,
117        print_codegen_timings: bool,
118        files: Option<&'local fm::FileManager>,
119    ) -> Self {
120        Self {
121            ssa_logging,
122            print_codegen_timings,
123            ssa,
124            files,
125            passed: Default::default(),
126            skip_passes: Default::default(),
127        }
128    }
129
130    pub fn ssa(&self) -> &Ssa {
131        &self.ssa
132    }
133
134    pub fn with_passed(mut self, passed: HashMap<String, usize>) -> Self {
135        self.passed = passed;
136        self
137    }
138
139    pub fn with_skip_passes(mut self, skip_passes: Vec<String>) -> Self {
140        self.skip_passes = skip_passes;
141        self
142    }
143
144    pub fn finish(self) -> Ssa {
145        self.ssa.generate_entry_point_index()
146    }
147
148    /// Run a list of SSA passes.
149    pub fn run_passes(mut self, passes: &[SsaPass]) -> Result<Self, RuntimeError> {
150        for pass in passes {
151            self = self.try_run_pass(|ssa| pass.run(ssa), pass.msg)?;
152        }
153        Ok(self)
154    }
155
156    /// Runs the given SSA pass and prints the SSA afterward if `print_ssa_passes` is true.
157    #[allow(dead_code)]
158    fn run_pass<F>(mut self, pass: F, msg: &str) -> Self
159    where
160        F: FnOnce(Ssa) -> Ssa,
161    {
162        self.ssa = time(msg, self.print_codegen_timings, || pass(self.ssa));
163        self.print(msg)
164    }
165
166    /// The same as `run_pass` but for passes that may fail
167    fn try_run_pass<F>(mut self, pass: F, msg: &str) -> Result<Self, RuntimeError>
168    where
169        F: FnOnce(Ssa) -> Result<Ssa, RuntimeError>,
170    {
171        // Count the number of times we have seen this message.
172        let cnt = *self.passed.entry(msg.to_string()).and_modify(|cnt| *cnt += 1).or_insert(1);
173        let step = self.passed.values().sum::<usize>();
174        let msg = format!("{msg} ({cnt}) (step {step})");
175
176        // See if we should skip this pass, including the count, so we can skip the n-th occurrence of a step.
177        let skip = self.skip_passes.iter().any(|s| msg.contains(s));
178
179        if !skip {
180            self.ssa = time(&msg, self.print_codegen_timings, || pass(self.ssa))?;
181            Ok(self.print(&msg))
182        } else {
183            Ok(self)
184        }
185    }
186
187    fn print(mut self, msg: &str) -> Self {
188        let print_ssa_pass = self.ssa_logging.matches(msg);
189
190        // Always normalize if we are going to print at least one of the passes
191        if !matches!(self.ssa_logging, SsaLogging::None) {
192            self.ssa.normalize_ids();
193        }
194
195        if print_ssa_pass {
196            println_to_stdout!("After {msg}:\n{}", self.ssa.print_with(self.files));
197        }
198        self
199    }
200}

202// Helper to time SSA passes
203pub(super) fn time<T>(name: &str, print_timings: bool, f: impl FnOnce() -> T) -> T {
204    let start_time = chrono::Utc::now().time();
205    let result = f();
206
207    if print_timings {
208        let end_time = chrono::Utc::now().time();
209        println_to_stdout!("{name}: {} ms", (end_time - start_time).num_milliseconds());
210    }
211
212    result
213}

215fn create_named_dir(named_dir: &Path, name: &str) -> PathBuf {
216    std::fs::create_dir_all(named_dir)
217        .unwrap_or_else(|_| panic!("could not create the `{name}` directory"));
218
219    PathBuf::from(named_dir)
220}

222fn write_to_file(bytes: &[u8], path: &Path) {
223    let display = path.display();
224
225    let mut file = match File::create(path) {
226        Err(why) => panic!("couldn't create {display}: {why}"),
227        Ok(file) => file,
228    };
229
230    if let Err(why) = file.write_all(bytes) {
231        panic!("couldn't write to {display}: {why}");
232    }
