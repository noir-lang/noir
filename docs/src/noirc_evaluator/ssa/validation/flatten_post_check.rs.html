1//! Post-flattening validation pass that ensures predicated values escaping
2//! their enable-side-effect definition are properly predicated.
3//!
4//! In ACIR, the result of a
5//! [`requires_acir_gen_predicate`][crate::ssa::ir::instruction::Instruction::requires_acir_gen_predicate]
6//! instruction depends on the active `enable_side_effects` predicate. For
7//! side-effecting *arithmetic* (the overflow-checked `Binary` operations) the
8//! result is multiplied by the predicate, so it is `0` whenever the predicate is
9//! false. The other predicated instructions consume the predicate differently
10//! (for example `ArrayGet` clamps its index and a `Call` forwards the predicate
11//! to the callee), so their result is not necessarily `0` when the predicate is
12//! false; this pass conservatively tracks all of them and still requires them to
13//! be guarded before they can escape.
14//!
15//! Some optimization passes (such as `checked_to_unchecked`) can remove
16//! *side-effect* checks and rewrite instructions into ones having
17//! `requires_acir_gen_predicate` set to `false`. The rewritten instruction's
18//! result is then no longer `0` when the predicate is false. This stays sound
19//! only because flattening adds a predicate multiplication, so the escaping
20//! value is still `0` when the predicate is false.
21//!
22//! This pass makes that an invariant and validates that no `requires_acir_gen_predicate`
23//! value can escape its enclosing `enable_side_effects` without a predicate
24//! multiplication (or an equivalent `IfElse` merge, which gates each branch by
25//! its own condition).
26//!
27//! As a result, any optimization on `requires_acir_gen_predicate` done after
28//! flattening is ensured to be sound.
29//!
30//! This is a post-flattening check, so each ACIR function is a single block. The
31//! pass asserts this rather than assuming it, which makes the validation a simple
32//! iteration over the instructions:
33//! - mark `requires_acir_gen_predicate` instructions,
34//! - mark also their uses,
35//! - report a violation if used outside `enable_side_effects`,
36//!   unless multiplied with the predicate.
37

52pub(crate) fn verify_side_effect_predicates(ssa: &Ssa) -> RtResult<()> {
53    for function in ssa.functions.values() {
54        verify_function(function)?;
55    }
56    Ok(())
57}
58
59fn verify_function(function: &Function) -> RtResult<()> {
60    // Brillig functions do not have `enable_side_effects` instructions
61    if function.runtime().is_brillig() {
62        return Ok(());
63    }
64
65    // The single-instruction-iteration below only inspects the entry block, which
66    // is only complete because flattening reduces every ACIR function to one block.
67    // Assert the invariant instead of trusting it: it is a cheap check when it holds
68    // and turns a future regression into a clear failure rather than silently
69    // skipping the instructions of any other block.
70    assert_eq!(
71        function.reachable_blocks().len(),
72        1,
73        "ACIR function {} should be a single block after flattening",
74        function.name()
75    );
76
77    let dfg = &function.dfg;
78    let block = function.entry_block();
79
80    // Values resulting from `requires_acir_gen_predicate` instructions, or
81    // using such predicated values. They are mapped to their predicate.
82    // These are the 'unsafe' values that we want to track for a non-predicated
83    // use outside their enable-side-effect definition.
84    let mut predicated_values: HashMap<ValueId, ValueId> = HashMap::default();
85
86    // The active predicate: `None` if no predicate (or `1`).
87    let mut current: Option<ValueId> = None;
88
89    for instruction_id in dfg[block].instructions() {
90        let instruction = &dfg[*instruction_id];
91
92        if let Instruction::EnableSideEffectsIf { condition } = instruction {
93            current = (!is_one(function, *condition)).then_some(*condition);
94            continue;
95        }
96
97        // Match instructions for
98        // - using predicated operands
99        // - using predicate operands outside enable-side-effect context
100        // - predicate guards: predicate * value
101        let mut guard_a_predicated_value = false;
102        let mut use_a_predicated_value = None;
103        let mut violation = None;
104        instruction.for_each_value(|operand| {
105            // If the instruction uses a `predicated_value`
106            let Some(&p) = predicated_values.get(&operand) else {
107                return;
108            };
109            if is_predicate_guard(dfg, instruction, operand, p) {
110                guard_a_predicated_value = true;
111            } else {
112                // Propagate the predicate to the current instruction
113                use_a_predicated_value.get_or_insert(p);
114                if current.is_none() {
115                    // The `predicated_value` operand is not used under a predicate,
116                    // flag it as an error.
117                    violation.get_or_insert(operand);
118                }
119            }
120        });
121        if let Some(operand) = violation {
122            let call_stack = function.dfg.get_instruction_call_stack(*instruction_id);
123            return Err(escape_error(function, operand, call_stack));
124        }
125
126        // `result_predicate` is the predicate of the current requires_acir_gen_predicate instruction,
127        // or the predicate of the instruction's operand, if any.
128        // When the instruction does not depend on a predicate, it is `None`
129        let result_predicate = if guard_a_predicated_value {
130            // A guard is safe
131            None
132        } else if use_a_predicated_value.is_some() {
133            use_a_predicated_value
134        } else if instruction.requires_acir_gen_predicate(dfg) {
135            current
136        } else {
137            None
138        };
139        if let Some(p) = result_predicate {
140            for result in dfg.instruction_results(*instruction_id) {
141                predicated_values.insert(*result, p);
142            }
143        }
144    }
145
146    // Error on returning an unsafe value
147    if let Some(TerminatorInstruction::Return { return_values, call_stack }) =
148        dfg[block].terminator()
149    {
150        for value in return_values {
151            if predicated_values.contains_key(value) {
152                let call_stack = function.dfg.get_call_stack(*call_stack);
153                return Err(escape_error(function, *value, call_stack));
154            }
155        }
156    }
157
158    Ok(())
159}

161/// True if `instruction` re-applies predicate `p` to `operand`, zeroing it
162/// whenever `p` is false: `mul p, operand` (either order), or a branch of an
163/// `IfElse` whose condition is `p` and whose value is `operand`. A merge gates
164/// each branch by its own condition (`then_condition * then_value +
165/// else_condition * else_value`), so both branches are checked.
166fn is_predicate_guard(
167    dfg: &crate::ssa::ir::dfg::DataFlowGraph,
168    instruction: &Instruction,
169    operand: ValueId,
170    p: ValueId,
171) -> bool {
172    match instruction {
173        Instruction::Binary(Binary { lhs, rhs, operator: BinaryOp::Mul { .. } }) => {
174            (*lhs == operand && is_predicate(dfg, *rhs, p))
175                || (*rhs == operand && is_predicate(dfg, *lhs, p))
176        }
177        Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
178            (*then_value == operand && is_predicate(dfg, *then_condition, p))
179                || (*else_value == operand && is_predicate(dfg, *else_condition, p))
180        }
181        _ => false,
182    }
183}
184
185/// Whether `value` is the predicate `p`, possibly through `cast`s.
186fn is_predicate(dfg: &crate::ssa::ir::dfg::DataFlowGraph, mut value: ValueId, p: ValueId) -> bool {
187    loop {
188        if value == p {
189            return true;
190        }
191        match &dfg[value] {
192            crate::ssa::ir::value::Value::Instruction { instruction, .. } => {
193                if let Instruction::Cast(src, _) = &dfg[*instruction] {
194                    value = *src;
195                    continue;
196                }
197                return false;
198            }
199            _ => return false,
200        }
201    }
202}
203
204fn is_one(function: &Function, value: ValueId) -> bool {
205    function.dfg.get_numeric_constant(value).is_some_and(|c| c.is_one())
206}
207
208fn escape_error(function: &Function, operand: ValueId, call_stack: CallStack) -> RuntimeError {
209    let message =
210        format!("Value {operand} escapes `enable_side_effects` in function {}", function.name());
211
212    RuntimeError::SsaValidationError { message, call_stack }
213}

216mod tests {
217    use super::verify_side_effect_predicates;

220    #[test]
221    fn rejects_ungated_escape_to_return() {
222        // A checked add under `enable_side_effects v0`
223        // is returned directly, so its disabled-branch zeroing is observable.
224        let src = "
225        acir(inline) fn main f0 {
226          b0(v0: u1, v1: u16):
227            enable_side_effects v0
228            v2 = cast v1 as u32
229            v3 = add v2, u32 1
230            enable_side_effects u1 1
231            return v3
232        }
233        ";
234        let ssa = Ssa::from_str_no_validation(src).unwrap();
235        assert!(verify_side_effect_predicates(&ssa).is_err());
236    }
237
238    #[test]
239    fn accepts_gated_escape() {
240        // The flattened-frontend shape: the value leaves the region only through
241        // `mul (cast v0), v3`, so it is `0` whenever `v0` is false.
242        let src = "
243        acir(inline) fn main f0 {
244          b0(v0: u1, v1: u16):
245            enable_side_effects v0
246            v2 = cast v1 as u32
247            v3 = add v2, u32 1
248            enable_side_effects u1 1
249            v4 = cast v0 as u32
250            v5 = mul v4, v3
251            return v5
252        }
253        ";
254        let ssa = Ssa::from_str_no_validation(src).unwrap();
255        assert!(verify_side_effect_predicates(&ssa).is_ok());
256    }

258    #[test]
259    fn accepts_non_side_effecting_op_under_predicate() {
260        // `not`/`cast` are not zeroed at ACIR-gen (`requires_acir_gen_predicate`
261        // is false), so a `not v0` defined under `enable_side_effects v0` is
262        // unconditional and may escape ungated. This is the `regression_11268`
263        // shape that an over-eager "anything under a predicate is tainted" check
264        // would wrongly reject.
265        let src = "
266        acir(inline) fn main f0 {
267          b0(v0: u1):
268            enable_side_effects v0
269            v1 = not v0
270            enable_side_effects u1 1
271            v2 = cast v1 as Field
272            return v2
273        }
274        ";
275        let ssa = Ssa::from_str_no_validation(src).unwrap();
276        assert!(verify_side_effect_predicates(&ssa).is_ok());
277    }
