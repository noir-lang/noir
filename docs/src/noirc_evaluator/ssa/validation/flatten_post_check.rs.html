1//! Post-flattening validation pass that ensures predicated values escaping
2//! their enable-side-effect definition are properly predicated.
3//!
4//! In ACIR, the result of a
5//! [`requires_acir_gen_predicate`][crate::ssa::ir::instruction::Instruction::requires_acir_gen_predicate]
6//! instruction depends on the active `enable_side_effects` predicate. For
7//! side-effecting *arithmetic* (the overflow-checked `Binary` operations) the
8//! result is multiplied by the predicate, so it is `0` whenever the predicate is
9//! false. The other predicated instructions consume the predicate differently
10//! (for example `ArrayGet` clamps its index and a `Call` forwards the predicate
11//! to the callee), so their result is not necessarily `0` when the predicate is
12//! false; this pass conservatively tracks all of them and still requires them to
13//! be guarded before they can escape.
14//!
15//! Some optimization passes (such as `checked_to_unchecked`) can remove
16//! *side-effect* checks and rewrite instructions into ones having
17//! `requires_acir_gen_predicate` set to `false`. The rewritten instruction's
18//! result is then no longer `0` when the predicate is false. This stays sound
19//! only because flattening adds a predicate multiplication, so the escaping
20//! value is still `0` when the predicate is false.
21//!
22//! This pass makes that an invariant and validates that no `requires_acir_gen_predicate`
23//! value can escape its enclosing `enable_side_effects` without a predicate
24//! multiplication (or an equivalent `IfElse` merge, which gates each branch by
25//! its own condition).
26//!
27//! As a result, any optimization on `requires_acir_gen_predicate` done after
28//! flattening is ensured to be sound.
29//!
30//! This is a post-flattening check, so each ACIR function is a single block. The
31//! pass asserts this rather than assuming it, which makes the validation a simple
32//! iteration over the instructions:
33//! - mark `requires_acir_gen_predicate` instructions,
34//! - mark also their uses,
35//! - report a violation if used outside `enable_side_effects`,
36//!   unless multiplied with the predicate.

52pub(crate) fn verify_side_effect_predicates(ssa: &Ssa) -> RtResult<()> {
53    for function in ssa.functions.values() {
54        verify_function(function)?;
55    }
56    Ok(())
57}

59fn verify_function(function: &Function) -> RtResult<()> {
60    // Brillig functions do not have `enable_side_effects` instructions
61    if function.runtime().is_brillig() {
62        return Ok(());
63    }
64
65    // The single-instruction-iteration below only inspects the entry block, which
66    // is only complete because flattening reduces every ACIR function to one block.
67    // Assert the invariant instead of trusting it: it is a cheap check when it holds
68    // and turns a future regression into a clear failure rather than silently
69    // skipping the instructions of any other block.
70    assert_eq!(
71        function.reachable_blocks().len(),
72        1,
73        "ACIR function {} should be a single block after flattening",
74        function.name()
75    );
76
77    let dfg = &function.dfg;
78    let block = function.entry_block();
79
80    // Values resulting from `requires_acir_gen_predicate` instructions, or
81    // using such predicated values. They are mapped to their predicate.
82    // These are the 'unsafe' values that we want to track for a non-predicated
83    // use outside their enable-side-effect definition.
84    let mut predicated_values: HashMap<ValueId, ValueId> = HashMap::default();
85
86    // The active predicate: `None` if no predicate (or `1`).
87    let mut current: Option<ValueId> = None;
88
89    for instruction_id in dfg[block].instructions() {
90        let instruction = &dfg[*instruction_id];
91
92        if let Instruction::EnableSideEffectsIf { condition } = instruction {
93            current = (!is_one(function, *condition)).then_some(*condition);
94            continue;
95        }
96
97        // Match instructions for
98        // - using predicated operands
99        // - using predicate operands outside enable-side-effect context
100        // - predicate guards: predicate * value
101        let mut guard_a_predicated_value = false;
102        let mut use_a_predicated_value = None;
103        let mut violation = None;
104        instruction.for_each_value(|operand| {
105            // If the instruction uses a `predicated_value`
106            let Some(&p) = predicated_values.get(&operand) else {
107                return;
108            };
109            if is_predicate_guard(dfg, instruction, operand, p) {
110                guard_a_predicated_value = true;
111            } else {
112                // Propagate the predicate to the current instruction
113                use_a_predicated_value.get_or_insert(p);
114                if current.is_none() {
115                    // The `predicated_value` operand is not used under a predicate,
116                    // flag it as an error.
117                    violation.get_or_insert(operand);
118                }
119            }
120        });
121        if let Some(operand) = violation {
122            let call_stack = function.dfg.get_instruction_call_stack(*instruction_id);
123            return Err(escape_error(function, operand, call_stack));
124        }
125
126        // `result_predicate` is the predicate of the current requires_acir_gen_predicate instruction,
127        // or the predicate of the instruction's operand, if any.
128        // When the instruction does not depend on a predicate, it is `None`
129        let result_predicate = if guard_a_predicated_value {
130            // A guard is safe
131            None
132        } else if use_a_predicated_value.is_some() {
133            use_a_predicated_value
134        } else if instruction.requires_acir_gen_predicate(dfg)
135            && !is_div_or_mod_by_nonzero_constant(dfg, instruction)
136        {
137            current
138        } else {
139            None
140        };
141        if let Some(p) = result_predicate {
142            for result in dfg.instruction_results(*instruction_id) {
143                predicated_values.insert(*result, p);
144            }
145        }
146    }
147
148    // Error on returning an unsafe value
149    if let Some(TerminatorInstruction::Return { return_values, call_stack }) =
150        dfg[block].terminator()
151    {
152        for value in return_values {
153            if predicated_values.contains_key(value) {
154                let call_stack = function.dfg.get_call_stack(*call_stack);
155                return Err(escape_error(function, *value, call_stack));
156            }
157        }
158    }
159
160    Ok(())
161}

163/// True if `instruction` re-applies predicate `p` to `operand`, zeroing it
164/// whenever `p` is false: `mul p, operand` (either order), or a branch of an
165/// `IfElse` whose condition is `p` and whose value is `operand`. A merge gates
166/// each branch by its own condition (`then_condition * then_value +
167/// else_condition * else_value`), so both branches are checked.
168fn is_predicate_guard(
169    dfg: &crate::ssa::ir::dfg::DataFlowGraph,
170    instruction: &Instruction,
171    operand: ValueId,
172    p: ValueId,
173) -> bool {
174    match instruction {
175        Instruction::Binary(Binary { lhs, rhs, operator: BinaryOp::Mul { .. } }) => {
176            (*lhs == operand && is_predicate(dfg, *rhs, p))
177                || (*rhs == operand && is_predicate(dfg, *lhs, p))
178        }
179        Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
180            (*then_value == operand && is_predicate(dfg, *then_condition, p))
181                || (*else_value == operand && is_predicate(dfg, *else_condition, p))
182        }
183        _ => false,
184    }
185}
186
187/// Whether multiplying by `value` re-applies predicate `p`, i.e. `value` is `0`
188/// whenever `p` is `0`.
189///
190/// Predicates are `u1` conditions; casting one to a wider type preserves its
191/// zero/non-zero-ness, so casts on either side are ignored when comparing. This
192/// holds when `value` is `p` itself, but also when:
193/// - `value` is a product `p * r` (in any nesting or operand order): such a
194///   product is `0` whenever `p` is `0`, so `value * operand` is still `0` on the
195///   disabled branch. Flattening gates a nested branch by the *conjunction* of all
196///   enclosing conditions, so a value tainted by an outer predicate `p` is
197///   re-guarded by an inner predicate of the form `p * inner_condition`.
198/// - `value` and `p` are different casts of the same source: the signed-division
199///   expansion, for example, derives both the branch predicate and the merge
200///   multiplier as separate casts of one sign value.
201/// - `value` is an `IfElse` merge `then_condition * then_value + else_condition *
202///   else_value`: the sum is `0` whenever `p` is `0` exactly when each product term
203///   is, and a product term is `0` whenever either of its factors is.
204fn is_predicate(dfg: &crate::ssa::ir::dfg::DataFlowGraph, value: ValueId, p: ValueId) -> bool {
205    let value = strip_casts(dfg, value);
206    let p = strip_casts(dfg, p);
207    if value == p {
208        return true;
209    }
210    if let crate::ssa::ir::value::Value::Instruction { instruction, .. } = &dfg[value] {
211        match &dfg[*instruction] {
212            Instruction::Binary(Binary { lhs, rhs, operator: BinaryOp::Mul { .. } }) => {
213                return is_predicate(dfg, *lhs, p) || is_predicate(dfg, *rhs, p);
214            }
215            Instruction::IfElse { then_condition, then_value, else_condition, else_value } => {
216                return (is_predicate(dfg, *then_condition, p)
217                    || is_predicate(dfg, *then_value, p))
218                    && (is_predicate(dfg, *else_condition, p)
219                        || is_predicate(dfg, *else_value, p));
220            }
221            _ => {}
222        }
223    }
224    false
225}

227/// Follow a chain of `cast` instructions to the underlying source value.
228fn strip_casts(dfg: &crate::ssa::ir::dfg::DataFlowGraph, mut value: ValueId) -> ValueId {
229    while let crate::ssa::ir::value::Value::Instruction { instruction, .. } = &dfg[value] {
230        if let Instruction::Cast(src, _) = &dfg[*instruction] {
231            value = *src;
232        } else {
233            break;
234        }
235    }
236    value
237}

239fn is_one(function: &Function, value: ValueId) -> bool {
240    function.dfg.get_numeric_constant(value).is_some_and(|c| c.is_one())
241}
242
243fn is_div_or_mod_by_nonzero_constant(
244    dfg: &crate::ssa::ir::dfg::DataFlowGraph,
245    instruction: &Instruction,
246) -> bool {
247    let Instruction::Binary(Binary { rhs, operator: BinaryOp::Div | BinaryOp::Mod, .. }) =
248        instruction
249    else {
250        return false;
251    };
252    dfg.get_numeric_constant(*rhs).is_some_and(|c| !c.is_zero())
253}

255fn escape_error(function: &Function, operand: ValueId, call_stack: CallStack) -> RuntimeError {
256    let message =
257        format!("Value {operand} escapes `enable_side_effects` in function {}", function.name());
258
259    RuntimeError::SsaValidationError { message, call_stack }
260}

268    #[test]
269    fn ifelse_on_the_multiplier_side_is_recognized() {
270        // Here the `IfElse` sits on the *predicate side*: it is the multiplier of a
271        // `mul` guard, not the merge of the tainted value. `is_predicate`'s `IfElse`
272        // arm sees that each branch of `if v1 then v0 else v0` is `0` whenever `v0`
273        // is, so the merge is `0` whenever `v0` is, and the guard is recognized.
274        let src = "
275        acir(inline) fn main f0 {
276          b0(v0: u1, v1: u1, v2: u16):
277            enable_side_effects v0
278            v3 = cast v2 as u32
279            v4 = add v3, u32 1
280            enable_side_effects u1 1
281            v5 = not v1
282            v6 = if v1 then v0 else (if v5) v0
283            v7 = cast v6 as u32
284            v8 = mul v7, v4
285            return v8
286        }
287        ";
288        let ssa = Ssa::from_str_no_validation(src).unwrap();
289        assert!(verify_side_effect_predicates(&ssa).is_ok());
290
291        // `v6 = v0`, so with `v0 = 0` the escaping value really is `0`.
292        let disabled =
293            ssa.interpret(vec![Value::bool(false), Value::bool(true), Value::u16(7)]).unwrap();
294        assert_eq!(disabled, vec![Value::u32(0)]);
295        let enabled =
296            ssa.interpret(vec![Value::bool(true), Value::bool(true), Value::u16(7)]).unwrap();
297        assert_eq!(enabled, vec![Value::u32(8)]);
298    }
