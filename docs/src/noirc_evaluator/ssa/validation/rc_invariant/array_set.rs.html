1//! The `array_set` reachable-aliased-use check.
2//!
3//! Iterates every `array_set` in each Brillig function and rejects the SSA
4//! when the in-place mutation could be observed through an aliased read on a
5//! forward path. The aliasing/coverage/forward-walk machinery it drives lives
6//! in the parent [`super`] module.
7

9    errors::{RtResult, RuntimeError},
10    ssa::{
11        ir::{function::Function, instruction::Instruction},
12        ssa_gen::Ssa,
13    },
14};

18/// Verifies the `array_set` / `inc_rc` aliasing invariant on every Brillig
19/// function in `ssa`. See the [`super`] module docs for details.
20///
21/// The entire module containing this function is gated behind
22/// `#[cfg(debug_assertions)]`, so it is a no-op (and absent at the linker
23/// level) in release builds — see the pipeline wiring in
24/// [`crate::ssa::primary_passes`].
25pub(crate) fn verify(ssa: &Ssa) -> RtResult<()> {
26    for function in ssa.functions.values() {
27        verify_function(function)?;
28    }
29    Ok(())
30}
31
32/// Per-function entry point. Skips ACIR functions (the invariant only
33/// applies to Brillig, where `array_set` may mutate in place) and runs the
34/// alias / dominance / reachable-use checks for every `array_set` in the
35/// function.
36fn verify_function(function: &Function) -> RtResult<()> {
37    if !function.runtime().is_brillig() {
38        return Ok(());
39    }
40
41    let ctx = Context::new(function);
42
43    for block_id in function.reachable_blocks() {
44        for (idx, instruction_id) in function.dfg[block_id].instructions().iter().enumerate() {
45            let Instruction::ArraySet { array, index: write_index, .. } =
46                function.dfg[*instruction_id]
47            else {
48                continue;
49            };
50
51            let alias_set = ctx.alias_set_for(array, block_id, idx);
52
53            // Narrow the alias-set to the members that are *not* provably
54            // protected on every path to this `array_set`. An empty result
55            // means every member is RC-bumped or freshly allocated before the
56            // array_set on all paths, so the in-place mutation is never
57            // observable — accept without the forward walk. A covered member
58            // (e.g. an `inc_rc`'d sibling) is dropped so the forward walk does
59            // not falsely flag a read of storage the array_set never mutates.
60            let use_set = ctx.unprotected_aliases(&alias_set, array, block_id, idx);
61            if use_set.is_empty() {
62                continue;
63            }
64
65            // The array_set's index, if constant — lets the walk skip
66            // `array_get`s on disjoint constant indices (the pedersen-style
67            // pattern). A dynamic write index means we must conservatively
68            // flag every aliased read.
69            let write_index_const = function.dfg.get_numeric_constant(write_index);
70
71            // Expensive: forward CFG walk looking for an aliased read.
72            // A hit means the `array_set` may mutate storage in place
73            // (RC=1) and a downstream instruction will observe the
74            // pre-mutation contents through an aliased name.
75            if let Some(hit) = ctx.find_reachable_aliased_use(
76                &use_set,
77                array,
78                *instruction_id,
79                block_id,
80                idx,
81                write_index_const,
82            ) {
83                let call_stack = function.dfg.get_instruction_call_stack(*instruction_id);
84                let aliased_use_call_stack =
85                    function.dfg.get_instruction_call_stack(hit.instruction);
86                let message = format!(
87                    "array_set in function {} on array {array} has an aliased read of {} on a \
88                     forward path with no preceding `inc_rc`; the in-place mutation would be \
89                     observable through that alias",
90                    function.name(),
91                    hit.value,
92                );
93                return Err(RuntimeError::ArraySetAliasViolation {
94                    message,
95                    call_stack,
96                    aliased_use_call_stack,
97                });
98            }
99        }
100    }
101    Ok(())
102}

133    /// A read value that carries its own `inc_rc` **and** is a loop
134    /// back-edge participant is **still a hazard** when it merely
135    /// *receives* the source's storage on a forward edge. Here `array_set s`
136    /// (b1) mutates `s` in place, then `p := s` (b1 → b2) and `p` is read by
137    /// the `call` in b2. `p` has an `inc_rc` and is a back-edge arg (b3 → b2),
138    /// but the `inc_rc` runs *after* the mutation and `p` flows *out of* the
139    /// source (`p ← s`), not into it — so the bump can't protect the
140    /// `array_set`. The read observes the in-place mutation; the verifier must
141    /// reject.
142    ///
143    /// This pins the soundness boundary of the protected-participant filter:
144    /// it excludes a value only when that value is in the *source's*
145    /// alias-set (flows *into* the source, so its `inc_rc` is loop-carried
146    /// before the mutation).

148    fn end_to_end_forward_threaded_read_with_inc_rc_participant_is_rejected() {

168    /// Inclusive-range (`..=`) peel, accepted by the index-aware
169    /// `array_set`-use rule. This is the SSA the frontend emits for
170    ///
171    /// ```ignore
172    /// for _ in 254_u8..=255_u8 { c4[0] = 9; c4 = c3; c3 = [b[0]]; }
173    /// ```
174    ///
175    /// An inclusive range lowers to an exclusive loop plus a duplicated
176    /// **peel** of the final iteration (see `codegen_for` in `ssa_gen`).
177    /// Here `b1`/`b2` are the loop and `b4` is the peel; the loop variable
178    /// `v29` (`c4`) is `array_set` in *both*, at the **same** constant
179    /// index `0`. The forward walk from the loop's `array_set v29` (`b2`)
180    /// reaches the peel's `array_set v29` (`b4`) — but that write
181    /// *overwrites* index `0`, the only tainted index, so it can't observe
182    /// the loop's in-place mutation. Index-aware handling of `array_set`
183    /// uses (the same `tainted`-index test the `array_get` rule uses)
184    /// therefore accepts it, matching the runtime: the program executes
185    /// identically under Brillig and comptime.
186    ///
187    /// This is *not* peel detection — there's no reliable post-`mem2reg`
188    /// marker for a peel block. It's a precise consequence of `array_set`
189    /// semantics (a write observes only the indices it doesn't overwrite),
190    /// so it also covers the corresponding `..` case and any other shape
191    /// where the aliased write hits the same index. A peel whose duplicated
192    /// body *reads* a tainted index some other way (e.g. an `array_get` or
193    /// a `call`) is still flagged.
194    #[test]
195    fn end_to_end_inclusive_range_peel_array_set_same_index_is_accepted() {
196        let src = r#"
197            brillig(inline) fn main f0 {
198              b0():
199                v1 = make_array [u8 1] : [u8; 1]
200                v4 = make_array [u8 2] : [u8; 1]
201                v7 = make_array [u8 3] : [u8; 1]
202                jmp b1(u8 254, v1, v4)
203              b1(v10: u8, v28: [u8; 1], v29: [u8; 1]):
204                v13 = lt v10, u8 255
205                jmpif v13 then: b2(), else: b3()
206              b2():
207                v18 = array_set v29, index u32 0, value u8 9
208                v20 = make_array [u8 3] : [u8; 1]
209                v21 = unchecked_add v10, u8 1
210                jmp b1(v21, v20, v28)
211              b3():
212                jmpif u1 1 then: b4(), else: b5(v28, v29)
213              b4():
214                v25 = array_set v29, index u32 0, value u8 9
215                v27 = make_array [u8 3] : [u8; 1]
216                jmp b5(v27, v28)
217              b5(v30: [u8; 1], v31: [u8; 1]):
218                return
219            }"#;
220        assert_verifier_accepts_because(
221            src,
222            "the peel's array_set v29 (b4) overwrites index 0 — the only index the loop's \
223             array_set v29 (b2) tainted — so it observes no in-place mutation",
224        );

228    /// SSA for
229    ///
230    /// ```ignore

267                jmpif u1 1 then: b4(), else: b5(v37, v38)

338    /// **Swap-exclusion soundness canary — freshening guard.** The swap
339    /// `v3 ← v2` (`P ← Q`) is present on the back-edge, but `v2`'s own
340    /// back-edge arg is `v2` itself (loop-**invariant** `c3`, not a fresh
341    /// `make_array`). Then `P_k = Q_{k-1} = Q_0 = Q_k`, so `P` and `Q`
342    /// genuinely share storage from the first swap on: the in-loop
343    /// `array_set v3` mutates it and `array_get v2` observes the mutation.
344    /// The exclusion must **not** fire (its freshening precondition fails),
345    /// and the verifier must reject. Guards against dropping `v2` purely
346    /// because it's swapped into the source.

2546    /// A located `array_set` plus everything the per-array_set verifier
2547    /// checks would need. Returned by [`find_array_sets`] / [`first_array_set`]
2548    /// / [`last_array_set`] so each test reads one struct rather than a
2549    /// six-element tuple.
2550    struct ArraySetSite {
2551        block: BasicBlockId,
2552        idx: usize,
2553        instruction_id: InstructionId,
2554        source: ValueId,
2555        alias_set: im::HashSet<ValueId>,
2556        /// The `array_set`'s index when it is a numeric constant. `None`
2557        /// indicates a dynamic index, in which case the verifier
2558        /// conservatively flags every aliased use.
2559        write_index_const: Option<FieldElement>,
2560    }
2561
2562    fn find_array_sets(function: &Function, ctx: &Context<'_>) -> Vec<ArraySetSite> {
2563        let mut out = Vec::new();
2564        for block_id in function.reachable_blocks() {
2565            for (idx, instruction_id) in function.dfg[block_id].instructions().iter().enumerate() {
2566                if let Instruction::ArraySet { array, index, .. } = function.dfg[*instruction_id] {
2567                    out.push(ArraySetSite {
2568                        block: block_id,
2569                        idx,
2570                        instruction_id: *instruction_id,
2571                        source: array,
2572                        alias_set: ctx.alias_set_for(array, block_id, idx),
2573                        write_index_const: function.dfg.get_numeric_constant(index),
2574                    });
2575                }
2576            }
2577        }
2578        out
2579    }
2580
2581    fn first_array_set(function: &Function, ctx: &Context<'_>) -> Option<ArraySetSite> {
2582        find_array_sets(function, ctx).into_iter().next()
2583    }
2584
2585    fn last_array_set(function: &Function, ctx: &Context<'_>) -> Option<ArraySetSite> {
2586        find_array_sets(function, ctx).into_iter().last()
2587    }
