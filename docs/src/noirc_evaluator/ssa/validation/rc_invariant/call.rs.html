1//! The call-argument reachable-aliased-use check.
2//!
3//! A callee may mutate an array argument in place at runtime — directly, or by
4//! returning an alias of it that the caller then mutates ([`super::Context`]'s
5//! alias analysis treats `Call` results as fresh, so that escape is invisible
6//! to the `array_set` scan). When the caller reuses such an argument without a
7//! protecting `inc_rc`, the mutation becomes observable through the alias —
8//! exactly the precondition `purity_analysis` relies on being absent.
9//!
10//! This verifier drives the same aliasing/coverage/forward-walk machinery as
11//! [`super::array_set`], seeded from call arguments instead of `array_set`
12//! sources, and gated on whether the callee can modify its arguments (mirroring
13//! `can_modify_args` in `ssa_gen`).
14

32/// Verify the call-argument aliasing invariant on every Brillig function in
33/// `ssa`. See the module docs for the invariant.
34pub(crate) fn verify(ssa: &Ssa) -> RtResult<()> {
35    let may_mutate = compute_may_mutate_args(ssa);
36    for function in ssa.functions.values() {
37        verify_function(function, &may_mutate)?;
38    }
39    Ok(())
40}
41
42/// Per-function check. Skips ACIR functions (the invariant only applies to
43/// Brillig, where a callee may mutate an argument in place). For every `call`
44/// whose callee may mutate its arguments, treats each array-typed argument as
45/// an all-index in-place mutation and runs the shared coverage + forward walk:
46/// a forward-reachable aliased read with no protecting `inc_rc` is a hazard.
47fn verify_function(function: &Function, may_mutate: &HashMap<FunctionId, bool>) -> RtResult<()> {
48    if !function.runtime().is_brillig() {
49        return Ok(());
50    }
51
52    let ctx = Context::new(function);
53
54    for block_id in function.reachable_blocks() {
55        for (idx, instruction_id) in function.dfg[block_id].instructions().iter().enumerate() {
56            let instruction_id = *instruction_id;
57            let Instruction::Call { func, arguments } = &function.dfg[instruction_id] else {
58                continue;
59            };
60
61            // Mirror `ssa_gen`'s `can_modify_args`: a callee that provably
62            // cannot mutate its arguments (a foreign call, a pure builtin, or a
63            // non-mutating function) has its argument clones elided by the
64            // ownership pass, so a reused argument legitimately carries no
65            // `inc_rc`. Skipping such calls is what keeps this check from
66            // flagging well-formed SSA.
67            if !callee_may_mutate_args(function, *func, may_mutate) {
68                continue;
69            }
70
71            // Treat the call as an all-index mutation of each array-*value*
72            // argument: `derived` is empty (the callee has no in-place result
73            // to chain from here) and the write index is unknown (it may touch
74            // any position). `is_array` matches a top-level array/vector value
75            // and so excludes a reference argument (`&mut [T; N]`): mutation
76            // through a reference is the explicit, caller-visible pattern the
77            // frontend passes a `&mut` for, not the value-array copy-on-write
78            // hazard this check is about (and a reference param already makes
79            // the callee impure). A top-level array of references (`[&mut T;
80            // N]`) is still a value and is checked.
81            let array_args: Vec<ValueId> = arguments
82                .iter()
83                .copied()
84                .filter(|&arg| function.dfg.type_of_value(arg).is_array())
85                .collect();
86            for arg in array_args {
87                let Some(hit) = ctx.aliased_use_for_source(
88                    arg,
89                    block_id,
90                    idx,
91                    instruction_id,
92                    None,
93                    im::HashSet::new(),
94                ) else {
95                    continue;
96                };
97
98                let message = format!(
99                    "call in function {} passes array {arg} that is read again as {} on a \
100                     forward path with no preceding `inc_rc`; if the callee mutates the argument \
101                     in place the mutation would be observable through that alias",
102                    function.name(),
103                    hit.value,
104                );
105                return Err(RuntimeError::CallArgAliasViolation {
106                    message,
107                    call_stack: function.dfg.get_instruction_call_stack(instruction_id),
108                    aliased_use_call_stack: function
109                        .dfg
110                        .get_instruction_call_stack(hit.instruction),
111                });
112            }
113        }
114    }
115    Ok(())
116}

118/// Whether the callee referenced by `func` may mutate an array argument in a
119/// way observable to the caller. Mirrors `ssa_gen`'s `can_modify_args`:
120/// foreign calls only read their inputs; pure builtins that are safe for clone
121/// elision cannot mutate; an unresolved/dynamic callee is assumed to be able
122/// to; and a known function is looked up in the call-graph summary.
123fn callee_may_mutate_args(
124    function: &Function,
125    func: ValueId,
126    may_mutate: &HashMap<FunctionId, bool>,
127) -> bool {
128    match &function.dfg[func] {
129        Value::Function(callee) => may_mutate.get(callee).copied().unwrap_or(true),
130        Value::Intrinsic(intrinsic) => intrinsic_may_mutate_args(*intrinsic),
131        Value::ForeignFunction { .. } => false,
132        _ => true,
133    }
134}
135
136/// Whether a call to `intrinsic` may mutate an array argument in place,
137/// mirroring `is_pure_builtin_func` in `ssa_gen`: a pure intrinsic that is safe
138/// for clone elision in Brillig cannot, everything else conservatively can.
139fn intrinsic_may_mutate_args(intrinsic: Intrinsic) -> bool {
140    intrinsic.unsafe_for_clone_elision_in_brillig()
141        || !matches!(intrinsic.purity(), Purity::Pure | Purity::PureWithPredicate)
142}
143
144/// Compute, for every function, whether a call to it may mutate the storage of
145/// one of its array arguments observably to the caller.
146///
147/// A function may-mutate if it contains an in-place mutation (`array_set` or
148/// `store`), calls a may-mutate function, calls a mutating intrinsic, or calls
149/// an unresolved/dynamic target (assume the worst). Foreign calls contribute
150/// nothing — oracles only read their inputs. This is an over-approximation:
151/// the only callees marked *not* may-mutate are exactly those whose argument
152/// clones the ownership pass elides, so a reused argument with no `inc_rc`
153/// never trips the check on well-formed SSA. Propagated to a fixed point over
154/// the call graph.
155fn compute_may_mutate_args(ssa: &Ssa) -> HashMap<FunctionId, bool> {
156    let mut may_mutate: HashMap<FunctionId, bool> = HashMap::default();
157    let mut callees: HashMap<FunctionId, Vec<FunctionId>> = HashMap::default();
158
159    for function in ssa.functions.values() {
160        let mut base = false;
161        let mut calls = Vec::new();
162        for block_id in function.reachable_blocks() {
163            for instruction_id in function.dfg[block_id].instructions() {
164                match &function.dfg[*instruction_id] {
165                    Instruction::ArraySet { .. } | Instruction::Store { .. } => base = true,
166                    Instruction::Call { func, .. } => match &function.dfg[*func] {
167                        Value::Function(callee) => calls.push(*callee),
168                        Value::Intrinsic(intrinsic) => {
169                            base |= intrinsic_may_mutate_args(*intrinsic);
170                        }
171                        // Foreign calls only read their inputs.
172                        Value::ForeignFunction { .. } => {}
173                        // An unresolved or dynamic callee: assume the worst.
174                        _ => base = true,
175                    },
176                    _ => {}
177                }
178            }
179        }
180        may_mutate.insert(function.id(), base);
181        callees.insert(function.id(), calls);
182    }
183
184    let mut changed = true;
185    while changed {
186        changed = false;
187        for (&id, callee_ids) in &callees {
188            if may_mutate[&id] {
189                continue;
190            }
191            if callee_ids.iter().any(|c| may_mutate.get(c).copied().unwrap_or(true)) {
192                may_mutate.insert(id, true);
193                changed = true;
194            }
195        }
196    }
197
198    may_mutate
199}

206    /// Parse `src`, run the `call` verifier, and require it to reject the SSA
207    /// with a [`crate::errors::RuntimeError::CallArgAliasViolation`]. Panics on
208    /// any other outcome. Runs `call::verify` directly (not the combined check)
209    /// so the assertion proves the *call* verifier is the one that caught the
210    /// hazard.
211    fn assert_verifier_rejects(src: &str) {
212        let ssa = Ssa::from_str(src).expect("SSA parses");
213        let err = super::verify(&ssa).expect_err("expected the verifier to reject");
214        assert!(
215            matches!(err, crate::errors::RuntimeError::CallArgAliasViolation { .. }),
216            "expected CallArgAliasViolation, got {err:?}",
217        );
218    }

220    /// Regression for noir-lang/noir-claude#1426. The ownership pass clones
221    /// (`inc_rc`s) every non-last use of an array, so a well-formed program
222    /// that reuses an array across a call always RC-protects it. This
223    /// hand-written SSA omits those bumps: a pure identity callee (`f2`)
224    /// returns the array input unchanged, `f1` then `array_set`s the returned
225    /// alias — mutating its caller's array in place at RC 1 — and `main`
226    /// reuses the same array across two calls to `f1`. The in-place mutation
227    /// is therefore observable to `main` (the first call's mutation is seen by
228    /// the second), which is exactly the precondition `purity_analysis`
229    /// relies on being absent. The verifier must reject: both the reused arg
230    /// in `main` and the reused-then-read arg in `f1` lack a preceding
231    /// `inc_rc`.
232    #[test]
233    fn end_to_end_array_reused_across_call_without_inc_rc_is_rejected() {
234        let src = r#"
235            brillig(inline) fn main f0 {
236              b0():
237                v0 = make_array [Field 1, Field 2] : [Field; 2]
238                v1 = call f1(v0) -> Field
239                v2 = call f1(v0) -> Field
240                return v1, v2
241            }
242            brillig(inline) fn bump_via_identity f1 {
243              b0(v0: [Field; 2]):
244                v1 = call f2(v0) -> [Field; 2]
245                v2 = array_get v0, index u32 0 -> Field
246                v3 = add v2, Field 1
247                v4 = array_set v1, index u32 0, value v3
248                return v3
249            }
250            brillig(inline) fn identity f2 {
251              b0(v0: [Field; 2]):
252                return v0
253            }"#;
254        assert_verifier_rejects(src);
255    }

257    /// The well-formed counterpart of
258    /// [`end_to_end_array_reused_across_call_without_inc_rc_is_rejected`]: the
259    /// `inc_rc`s the ownership pass emits are present — in `main` before the
260    /// reused call arg, and in `f1` before the array escapes to `identity` and
261    /// is read again. Every reused array call-arg is now RC-protected, so the
262    /// in-place mutation cannot be observed through an alias and the verifier
263    /// accepts. This pins down that the call-arg check credits a preceding
264    /// `inc_rc` rather than flagging every reused call arg unconditionally.
265    #[test]
266    fn end_to_end_array_reused_across_call_with_inc_rc_is_accepted() {
267        let src = r#"
268            brillig(inline) fn main f0 {
269              b0():
270                v0 = make_array [Field 1, Field 2] : [Field; 2]
271                inc_rc v0
272                v1 = call f1(v0) -> Field
273                v2 = call f1(v0) -> Field
274                return v1, v2
275            }
276            brillig(inline) fn bump_via_identity f1 {
277              b0(v0: [Field; 2]):
278                inc_rc v0
279                v1 = call f2(v0) -> [Field; 2]
280                v2 = array_get v0, index u32 0 -> Field
281                v3 = add v2, Field 1
282                v4 = array_set v1, index u32 0, value v3
283                return v3
284            }
285            brillig(inline) fn identity f2 {
286              b0(v0: [Field; 2]):
287                return v0
288            }"#;
289        assert_verifier_accepts_because(
290            src,
291            "every reused array call-arg is protected by a preceding inc_rc",
292        );
293    }

295    /// Reduced from the `array_sort` execution test (`quicksort`): a `&mut`
296    /// **reference** to an array is passed to a callee that sorts it in place
297    /// and then loaded back. The argument is a reference, not an array value,
298    /// so it is *not* a copy-on-write hazard — mutation through a `&mut` is the
299    /// explicit, caller-visible pattern the frontend passes a reference for
300    /// (and a reference parameter already makes the callee impure). The call
301    /// verifier must skip reference arguments and accept; flagging this was a
302    /// false positive fixed by the `contains_reference` filter.
303    #[test]
304    fn end_to_end_reference_argument_read_back_after_call_is_accepted() {
305        let src = r#"
306            brillig(inline) fn main f0 {
307              b0(v0: [u8; 3]):
308                v1 = allocate -> &mut [u8; 3]
309                store v0 at v1
310                call f1(v1)
311                v2 = load v1 -> [u8; 3]
312                return v2
313            }
314            brillig(inline) fn sort_in_place f1 {
315              b0(v0: &mut [u8; 3]):
316                v1 = load v0 -> [u8; 3]
317                v3 = array_set v1, index u32 0, value u8 9
318                store v3 at v0
319                return
320            }"#;
321        assert_verifier_accepts_because(
322            src,
323            "the call argument is a &mut reference, not an array value, so it is not a COW hazard",
324        );
325    }
