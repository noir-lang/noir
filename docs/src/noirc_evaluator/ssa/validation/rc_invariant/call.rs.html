1//! The call-argument reachable-aliased-use check.
2//!
3//! A callee may mutate an array argument in place at runtime — directly, or by
4//! returning an alias of it that the caller then mutates ([`super::Context`]'s
5//! alias analysis treats `Call` results as fresh, so that escape is invisible
6//! to the `array_set` scan). When the caller reuses such an argument without a
7//! protecting `inc_rc`, the mutation becomes observable through the alias —
8//! exactly the precondition `purity_analysis` relies on being absent.
9//!
10//! This verifier drives the same aliasing/coverage/forward-walk machinery as
11//! [`super::array_set`], seeded from call arguments instead of `array_set`
12//! sources, and gated on whether the callee can modify its arguments (mirroring
13//! `can_modify_args` in `ssa_gen`).
14

33/// Verify the call-argument aliasing invariant on every Brillig function in
34/// `ssa`. See the module docs for the invariant.
35pub(crate) fn verify(ssa: &Ssa) -> RtResult<()> {
36    let may_mutate = compute_may_mutate_args(ssa);
37    let returns_arg_alias = compute_returns_arg_alias(ssa);
38
39    // A user-function call must have its array arguments checked if the callee
40    // may mutate one in place, *or* may hand back an alias of one that the
41    // caller then mutates (e.g. an identity function — see [#1443]). Either way,
42    // reusing the argument without a protecting `inc_rc` is a hazard. Both
43    // summaries cover every function, so indexing them is safe (and asserts we
44    // computed a value for every callee).
45    //
46    // [#1443]: https://github.com/noir-lang/noir-claude/issues/1443
47    let needs_check = |callee: FunctionId| may_mutate[&callee] || returns_arg_alias[&callee];
48
49    for function in ssa.functions.values() {
50        verify_function(function, &needs_check)?;
51    }
52    Ok(())
53}

55/// Per-function check. Skips ACIR functions (the invariant only applies to
56/// Brillig, where a callee may mutate an argument in place). For every `call`
57/// whose callee may mutate an argument or return an alias of one (`needs_check`),
58/// treats each array-typed argument as an all-index in-place mutation and runs
59/// the shared coverage + forward walk: a forward-reachable aliased read with no
60/// protecting `inc_rc` is a hazard.
61fn verify_function(function: &Function, needs_check: &impl Fn(FunctionId) -> bool) -> RtResult<()> {
62    if !function.runtime().is_brillig() {
63        return Ok(());
64    }
65
66    let ctx = Context::new(function);
67
68    for block_id in function.reachable_blocks() {
69        for (idx, instruction_id) in function.dfg[block_id].instructions().iter().enumerate() {
70            let instruction_id = *instruction_id;
71            let Instruction::Call { func, arguments } = &function.dfg[instruction_id] else {
72                continue;
73            };
74
75            // Mirror `ssa_gen`'s `can_modify_args`: a callee that provably
76            // cannot mutate an argument *or* hand back an alias of one (a
77            // foreign call, a pure builtin, or a function that neither mutates
78            // an argument nor returns an alias of one) has its argument clones
79            // elided by the ownership pass, so a reused argument legitimately
80            // carries no `inc_rc`. Skipping such calls is what keeps this check
81            // from flagging well-formed SSA.
82            if !callee_needs_arg_check(function, *func, needs_check) {
83                continue;
84            }
85
86            // Treat the call as an all-index mutation of each array-*value*
87            // argument: `derived` is empty (the callee has no in-place result
88            // to chain from here) and the write index is unknown (it may touch
89            // any position). `is_array` matches a top-level array/vector value
90            // and so excludes a reference argument (`&mut [T; N]`): mutation
91            // through a reference is the explicit, caller-visible pattern the
92            // frontend passes a `&mut` for, not the value-array copy-on-write
93            // hazard this check is about (and a reference param already makes
94            // the callee impure). A top-level array of references (`[&mut T;
95            // N]`) is still a value and is checked.
96            let array_args: Vec<ValueId> = arguments
97                .iter()
98                .copied()
99                .filter(|&arg| function.dfg.type_of_value(arg).is_array())
100                .collect();
101            for arg in array_args {
102                let Some(hit) = ctx.aliased_use_for_source(
103                    arg,
104                    block_id,
105                    idx,
106                    instruction_id,
107                    None,
108                    im::HashSet::new(),
109                ) else {
110                    continue;
111                };
112
113                let message = format!(
114                    "call in function {} passes array {arg} that is read again as {} on a \
115                     forward path with no preceding `inc_rc`; if the callee mutates the argument \
116                     in place, or returns an alias of it that is then mutated, the mutation would \
117                     be observable through that alias",
118                    function.name(),
119                    hit.value,
120                );
121                return Err(RuntimeError::CallArgAliasViolation {
122                    message,
123                    call_stack: function.dfg.get_instruction_call_stack(instruction_id),
124                    aliased_use_call_stack: function
125                        .dfg
126                        .get_instruction_call_stack(hit.instruction),
127                });
128            }
129        }
130    }
131    Ok(())
132}

134/// Whether a call to the callee referenced by `func` needs its array arguments
135/// checked — i.e. the callee may mutate an argument in place or return an alias
136/// of one. Mirrors `ssa_gen`'s `can_modify_args`: foreign calls only read their
137/// inputs and return fresh results; pure builtins that are safe for clone
138/// elision do neither; an unresolved/dynamic callee is assumed to need
139/// checking; and a known function is decided by `needs_check`, which combines
140/// the two call-graph summaries (`may_mutate || returns_arg_alias`).
141fn callee_needs_arg_check(
142    function: &Function,
143    func: ValueId,
144    needs_check: &impl Fn(FunctionId) -> bool,
145) -> bool {
146    match &function.dfg[func] {
147        Value::Function(callee) => needs_check(*callee),
148        Value::Intrinsic(intrinsic) => intrinsic_may_mutate_args(*intrinsic),
149        Value::ForeignFunction { .. } => false,
150        _ => true,
151    }
152}
153
154/// Whether a call to `intrinsic` may mutate an array argument in place,
155/// mirroring `is_pure_builtin_func` in `ssa_gen`: a pure intrinsic that is safe
156/// for clone elision in Brillig cannot, everything else conservatively can.
157fn intrinsic_may_mutate_args(intrinsic: Intrinsic) -> bool {
158    intrinsic.unsafe_for_clone_elision_in_brillig()
159        || !matches!(intrinsic.purity(), Purity::Pure | Purity::PureWithPredicate)
160}

162/// Populate per-function state with `init`, then run `update` over every
163/// function until a full round makes no change, and return the final state.
164///
165/// `init` receives each function and a mutable reference to the
166/// default-initialized state to fill in that function's entry. `update`
167/// likewise receives each function and the state, mutates it in place, and
168/// returns whether it changed anything; returning `false` short-circuits a
169/// function whose contribution is already settled. The caller keeps whichever
170/// part of the final state it needs.
171fn fixpoint<S: Default>(
172    ssa: &Ssa,
173    mut init: impl FnMut(&Function, &mut S),
174    mut update: impl FnMut(&Function, &mut S) -> bool,
175) -> S {
176    let mut state = S::default();
177    for function in ssa.functions.values() {
178        init(function, &mut state);
179    }
180    let mut changed = true;
181    while changed {
182        changed = false;
183        for function in ssa.functions.values() {
184            changed |= update(function, &mut state);
185        }
186    }
187    state
188}

190/// Compute, for every function, whether a call to it may mutate the storage of
191/// one of its array arguments observably to the caller.
192///
193/// A function may-mutate if it contains an in-place mutation (`array_set` or
194/// `store`), calls a may-mutate function, calls a mutating intrinsic, or calls
195/// an unresolved/dynamic target (assume the worst). Foreign calls contribute
196/// nothing — oracles only read their inputs. This is an over-approximation:
197/// the only callees marked *not* may-mutate are exactly those whose argument
198/// clones the ownership pass elides, so a reused argument with no `inc_rc`
199/// never trips the check on well-formed SSA. Propagated to a fixed point over
200/// the call graph.
201fn compute_may_mutate_args(ssa: &Ssa) -> HashMap<FunctionId, bool> {
202    let (may_mutate, _callees) = fixpoint(
203        ssa,
204        // Initialize the state to each function's own (non-propagated) may-mutate flag,
205        // plus its callee list. The callee list, built once by `init`, lets the propagation
206        // rounds avoid re-scanning instruction bodies.
207        |function,
208         (may_mutate, callees): &mut (
209            HashMap<FunctionId, bool>,
210            HashMap<FunctionId, Vec<FunctionId>>,
211        )| {
212            let mut base = false;
213            let mut calls = Vec::new();
214            for block_id in function.reachable_blocks() {
215                for instruction_id in function.dfg[block_id].instructions() {
216                    match &function.dfg[*instruction_id] {
217                        Instruction::ArraySet { .. } | Instruction::Store { .. } => base = true,
218                        Instruction::Call { func, .. } => match &function.dfg[*func] {
219                            Value::Function(callee) => calls.push(*callee),
220                            Value::Intrinsic(intrinsic) => {
221                                base |= intrinsic_may_mutate_args(*intrinsic);
222                            }
223                            // Foreign calls only read their inputs.
224                            Value::ForeignFunction { .. } => {}
225                            // An unresolved or dynamic callee: assume the worst.
226                            _ => base = true,
227                        },
228                        _ => {}
229                    }
230                }
231            }
232            may_mutate.insert(function.id(), base);
233            callees.insert(function.id(), calls);
234        },
235        // Propagate: a function may-mutate if any callee may-mutate.
236        |function, (may_mutate, callees)| {
237            let id = function.id();
238            if may_mutate[&id] {
239                return false;
240            }
241            let now = callees[&id].iter().any(|c| may_mutate[c]);
242            if now {
243                may_mutate.insert(id, true);
244            }
245            now
246        },
247    );
248    may_mutate
249}

251/// Compute, for every function, whether it may return an array value that
252/// aliases one of its array parameters — e.g. an identity function that returns
253/// its input unchanged. Such a call hands the caller an alias of the argument,
254/// so an in-place mutation of the *result* mutates the *argument's* storage; the
255/// call must be checked even when the callee does not itself mutate.
256///
257/// Distinct from "returns any array": a callee that returns a *fresh* array (a
258/// `make_array`, or a foreign-call result — the shape of an oracle wrapper that
259/// returns an array) is not flagged, so its caller's clone-elided arguments stay
260/// accepted. Propagated to a fixed point over the call graph because the alias
261/// property flows through `Value::Function` call results. (Alias-returning
262/// intrinsics such as the vector mutators are left to `may_mutate` — see
263/// [`function_returns_arg_alias`].)
264fn compute_returns_arg_alias(ssa: &Ssa) -> HashMap<FunctionId, bool> {
265    // Monotonic fixed point: every function starts `false`; one only ever flips
266    // to true, and `function_returns_arg_alias` reads the current map to resolve
267    // callee results.
268    fixpoint(
269        ssa,
270        |function, returns_arg_alias: &mut HashMap<FunctionId, bool>| {
271            returns_arg_alias.insert(function.id(), false);
272        },
273        |function, returns_arg_alias| {
274            let id = function.id();
275            if returns_arg_alias[&id] {
276                return false;
277            }
278            let now = function_returns_arg_alias(function, returns_arg_alias);
279            if now {
280                returns_arg_alias.insert(id, true);
281            }
282            now
283        },
284    )
285}

287/// Whether `function` returns an array value that may alias one of its array
288/// parameters, given the current `returns_arg_alias` summary for resolving
289/// callee results.
290///
291/// Computes the set of *parameter-derived* values to a fixed point: an array
292/// parameter, a block parameter threaded from one, an `array_set` or nested
293/// `array_get` of one, or a `Value::Function` call result whose callee
294/// `returns_arg_alias` and is fed a parameter-derived argument. `make_array` and
295/// foreign-call results stop the trace (genuinely fresh). Intrinsic results also
296/// stop the trace here — the alias-returning ones
297/// (`unsafe_for_clone_elision_in_brillig`) are instead covered by `may_mutate`,
298/// since calling one sets that flag. The function returns an arg alias iff any
299/// returned value is parameter-derived.
300fn function_returns_arg_alias(
301    function: &Function,
302    returns_arg_alias: &HashMap<FunctionId, bool>,
303) -> bool {
304    let dfg = &function.dfg;
305
306    // Incoming block-parameter arguments per destination block, to thread
307    // parameter-derived-ness across edges.
308    let mut incoming: HashMap<BasicBlockId, Vec<Vec<ValueId>>> = HashMap::default();
309    for block_id in function.reachable_blocks() {
310        match dfg[block_id].terminator() {
311            Some(TerminatorInstruction::Jmp { destination, arguments, .. }) => {
312                incoming.entry(*destination).or_default().push(arguments.clone());
313            }
314            Some(TerminatorInstruction::JmpIf {
315                then_destination,
316                then_arguments,
317                else_destination,
318                else_arguments,
319                ..
320            }) => {
321                incoming.entry(*then_destination).or_default().push(then_arguments.clone());
322                incoming.entry(*else_destination).or_default().push(else_arguments.clone());
323            }
324            _ => {}
325        }
326    }
327
328    // Seed with the function's array-value parameters (entry block parameters).
329    let entry = function.entry_block();
330    let mut param_derived: HashSet<ValueId> = dfg
331        .block_parameters(entry)
332        .iter()
333        .copied()
334        .filter(|&p| dfg.type_of_value(p).is_array())
335        .collect();
336
337    let mut changed = true;
338    while changed {
339        changed = false;
340
341        for block_id in function.reachable_blocks() {
342            // Block parameters fed a parameter-derived argument on some edge.
343            if let Some(edges) = incoming.get(&block_id) {
344                let params = dfg.block_parameters(block_id);
345                for (i, &param) in params.iter().enumerate() {
346                    if param_derived.contains(&param) {
347                        continue;
348                    }
349                    // Check any of the incoming edges for arguments which are derived from function inputs.
350                    let fed = edges
351                        .iter()
352                        .any(|args| args.get(i).is_some_and(|a| param_derived.contains(a)));
353                    if fed {
354                        param_derived.insert(param);
355                        changed = true;
356                    }
357                }
358            }
359
360            for instruction_id in dfg[block_id].instructions() {
361                let propagate = match &dfg[*instruction_id] {
362                    // An array_set result shares the operand's storage.
363                    Instruction::ArraySet { array, .. } => param_derived.contains(array),
364                    // A *nested* array_get returns a sub-array that shares the
365                    // source's storage (a brillig array_get on a nested array
366                    // aliases rather than copies). The `is_array` gate on the
367                    // result below restricts this to the nested case — a
368                    // non-nested get yields a scalar and propagates nothing.
369                    Instruction::ArrayGet { array, .. } => param_derived.contains(array),
370                    // A user-function call result aliases an argument only if
371                    // the callee returns an arg alias and is fed a
372                    // parameter-derived argument.
373                    //
374                    // Foreign-call results are genuinely fresh (oracles copy
375                    // across the boundary). Intrinsic results are *not* always
376                    // fresh — the `unsafe_for_clone_elision_in_brillig`
377                    // intrinsics (vector mutators, `str_as_bytes`,
378                    // `array_as_str_unchecked`) return an alias of their input —
379                    // but we deliberately don't trace through them here: calling
380                    // one already makes the function `may_mutate`
381                    // (`intrinsic_may_mutate_args`), so `needs_check` flags it
382                    // via that summary. This pass only has to cover the gap
383                    // `may_mutate` misses: a non-mutating function that passes an
384                    // input straight back.
385                    Instruction::Call { func, arguments } => match &dfg[*func] {
386                        Value::Function(callee) => {
387                            returns_arg_alias[callee]
388                                && arguments.iter().any(|a| param_derived.contains(a))
389                        }
390                        _ => false,
391                    },
392                    _ => false,
393                };
394                if propagate {
395                    for &result in dfg.instruction_results(*instruction_id) {
396                        if dfg.type_of_value(result).is_array() && param_derived.insert(result) {
397                            changed = true;
398                        }
399                    }
400                }
401            }
402        }
403    }
404
405    // Check if the return block contains a value that was derived from the inputs.
406    function.reachable_blocks().iter().any(|&block_id| {
407        matches!(
408            dfg[block_id].terminator(),
409            Some(TerminatorInstruction::Return { return_values, .. })
410                if return_values.iter().any(|v| param_derived.contains(v))
411        )
412    })
413}

434    /// Regression for noir-lang/noir-claude#1426. The ownership pass clones
435    /// (`inc_rc`s) every non-last use of an array, so a well-formed program
436    /// that reuses an array across a call always RC-protects it. This
437    /// hand-written SSA omits those bumps: a pure identity callee (`f2`)
438    /// returns the array input unchanged, `f1` then `array_set`s the returned
439    /// alias — mutating its caller's array in place at RC 1 — and `main`
440    /// reuses the same array across two calls to `f1`. The in-place mutation
441    /// is therefore observable to `main` (the first call's mutation is seen by
442    /// the second), which is exactly the precondition `purity_analysis`
443    /// relies on being absent. The verifier must reject: both the reused arg
444    /// in `main` and the reused-then-read arg in `f1` lack a preceding
445    /// `inc_rc`.

448        let src = r#"
449            brillig(inline) fn main f0 {
450              b0():
451                v0 = make_array [Field 1, Field 2] : [Field; 2]
452                v1 = call f1(v0) -> Field
453                v2 = call f1(v0) -> Field
454                return v1, v2
455            }
456            brillig(inline) fn bump_via_identity f1 {
457              b0(v0: [Field; 2]):
458                v1 = call f2(v0) -> [Field; 2]
459                v2 = array_get v0, index u32 0 -> Field
460                v3 = add v2, Field 1
461                v4 = array_set v1, index u32 0, value v3
462                return v3
463            }
464            brillig(inline) fn identity f2 {
465              b0(v0: [Field; 2]):
466                return v0
467            }"#;
468        assert_verifier_rejects(src);
469    }

568    /// The well-formed counterpart of
569    /// [`end_to_end_callee_returns_input_alias_mutated_by_caller_is_rejected`]:
570    /// the `inc_rc v0` the ownership pass emits before the reused call argument
571    /// is present, so the later `array_set` copies rather than mutating `v0` in
572    /// place and the read of `v0` is sound. Accepted.
573    #[test]
574    fn end_to_end_callee_returns_input_alias_with_inc_rc_is_accepted() {
575        let src = r#"
576            brillig(inline) fn main f0 {
577              b0():
578                v0 = make_array [Field 1, Field 2] : [Field; 2]
579                inc_rc v0
580                v1 = call f1(v0) -> [Field; 2]
581                v2 = array_set v1, index u32 0, value Field 9
582                v3 = array_get v0, index u32 0 -> Field
583                return v3
584            }
585            brillig(inline) fn identity f1 {
586              b0(v0: [Field; 2]):
587                return v0
588            }"#;
589        assert_verifier_accepts_because(
590            src,
591            "the reused argument is protected by a preceding inc_rc",
592        );
593    }
594
595    /// Soundness guard for `returns_arg_alias`: a callee that returns a *fresh*
596    /// array (here a foreign-call result, the shape of an oracle wrapper that
597    /// returns an array) does **not** alias its input. Even though the caller
598    /// reuses the argument with no `inc_rc` — which the frontend legitimately
599    /// elides for oracle wrappers — there is no aliasing hazard, so the call
600    /// verifier must skip it and accept. A coarser "returns any array" rule
601    /// would have falsely flagged this.
602    #[test]
603    fn end_to_end_callee_returns_fresh_array_reused_arg_is_accepted() {
604        let src = r#"
605            brillig(inline) fn main f0 {
606              b0():
607                v0 = make_array [Field 1, Field 2] : [Field; 2]
608                v1 = call f1(v0) -> [Field; 2]
609                v2 = array_set v1, index u32 0, value Field 9
610                v3 = array_get v0, index u32 0 -> Field
611                return v3
612            }
613            brillig(inline) fn wrapper f1 {
614              b0(v0: [Field; 2]):
615                v1 = call my_oracle(v0) -> [Field; 2]
616                return v1
617            }"#;
618        assert_verifier_accepts_because(
619            src,
620            "the callee returns a fresh foreign-call result, not an alias of its input",
621        );
622    }

624    /// `returns_arg_alias` must trace through a **nested** `array_get`: a brillig
625    /// `array_get` on a nested array returns a sub-array that *aliases* the
626    /// source's storage (the same "the input was moved" case `pure.rs` models).
627    /// Here `f1` returns `v0[0]`, an alias of its nested-array input; the caller
628    /// `array_set`s that result (mutating the input's storage in place) and then
629    /// reads the input. With no `inc_rc` on the reused argument the verifier must
630    /// reject.
631    #[test]
632    fn end_to_end_callee_returns_nested_array_get_alias_is_rejected() {
633        let src = r#"
634            brillig(inline) fn main f0 {
635              b0(v0: [[u8; 3]; 2]):
636                v1 = call f1(v0) -> [u8; 3]
637                v2 = array_set v1, index u32 0, value u8 9
638                v3 = array_get v0, index u32 0 -> [u8; 3]
639                return v3
640            }
641            brillig(inline) fn nested_identity f1 {
642              b0(v0: [[u8; 3]; 2]):
643                v1 = array_get v0, index u32 0 -> [u8; 3]
644                return v1
645            }"#;
646        assert_verifier_rejects(src);
647    }
