57#[derive(Debug, Clone, Copy, Default)]
58pub struct InterpreterOptions {

60    pub trace: bool,

67struct CallContext {
68    /// The function that was called. This is `None` only for the top-level global
69    /// scope where global instructions are evaluated.
70    called_function: Option<FunctionId>,
71
72    /// Contains each value currently defined and visible to the current function.
73    scope: HashMap<ValueId, Value>,
74
75    /// This variable can be modified by `enable_side_effects_if` instructions and is
76    /// expected to have no effect if there are no such instructions or if the code
77    /// being executed is an unconstrained function.
78    side_effects_enabled: bool,
79}

144    /// Increment the step counter, or return [`InterpreterError::OutOfBudget`].
145    ///
146    /// If there is no step limit, then it doesn't increment the counter.
147    fn inc_step_counter(&mut self) -> IResult<()> {
148        if let Some(limit) = self.options.step_limit {
149            if self.step_counter >= limit {
150                return Err(InterpreterError::OutOfBudget { steps: self.step_counter });
151            }
152            // With a limit we shouldn't wrap around, but just in case we wanted move this outside,
153            // use a safe wrap-around increment.
154            self.step_counter = self.step_counter.wrapping_add(1);
155        }
156        Ok(())
157    }

190    /// Define or redefine a value.
191    /// Redefinitions are expected in the case of loops.
192    fn define(&mut self, id: ValueId, value: Value) -> IResult<()> {
193        if self.options.trace {
194            println!("{id} = {value}");
195        }
196
197        if let Some(func) = self.try_current_function() {
198            let expected_type = func.dfg.type_of_value(id);
199            let actual_type = value.get_type();
200
201            if *expected_type != actual_type {
202                // Special case for ZST (Zero-Sized Type) arrays: Allow length mismatches.
203                // In early SSA passes, ZST arrays like [(); 3] are represented with empty element lists.
204                // Later optimization passes will fix the representation.
205                let types_compatible = match (&*expected_type, &actual_type) {
206                    (Type::Array(expected_elem, _), Type::Array(actual_elem, actual_len)) => {
207                        expected_elem == actual_elem
208                            && expected_elem.is_empty()
209                            && actual_len.to_usize() == 0
210                    }
211                    // Reference mutability doesn't affect runtime behavior —
212                    // the interpreter treats &T and &mut T identically.
213                    (Type::Reference(expected_elem, _), Type::Reference(actual_elem, _)) => {
214                        expected_elem == actual_elem
215                    }
216                    _ => false,
217                };
218
219                if !types_compatible {
220                    return Err(internal(InternalError::ValueTypeDoesNotMatchReturnType {
221                        value_id: id,
222                        expected_type: expected_type.to_string(),
223                        actual_type: actual_type.to_string(),
224                    }));
225                }
226            }
227        }
228
229        self.call_context_mut().scope.insert(id, value);
230
231        Ok(())
232    }

234    /// Interpret the global instructions.
235    ///
236    /// Once this is complete, the interpreter can be reused for multiple
237    /// function calls within the same SSA.
238    pub(crate) fn interpret_globals(&mut self) -> IResult<()> {
239        assert_eq!(self.call_stack.len(), 1, "should be in the global context");
240        let Some((_, function)) = self.functions.first_key_value() else {
241            return Ok(());
242        };
243
244        let globals = &function.dfg.globals;
245        for (global_id, global) in globals.values_iter() {
246            let value = match global {
247                super::ir::value::Value::Instruction { instruction, .. } => {
248                    let instruction = &globals[*instruction];
249                    self.interpret_instruction(instruction, &[global_id])?;
250                    continue;
251                }
252                super::ir::value::Value::NumericConstant { constant, typ } => {
253                    Value::from_constant(*constant, *typ)?
254                }
255                super::ir::value::Value::Function(id) => Value::Function(*id),
256                super::ir::value::Value::Intrinsic(intrinsic) => Value::Intrinsic(*intrinsic),
257                super::ir::value::Value::ForeignFunction { name, .. } => {
258                    Value::ForeignFunction(name.clone())
259                }
260                super::ir::value::Value::Global(_) | super::ir::value::Value::Param { .. } => {
261                    unreachable!()
262                }
263            };
264            self.define(global_id, value)?;
265        }
266        Ok(())
267    }

269    /// Interpret an entry point, assuming the globals have already been interpreted.
270    ///
271    /// This resets any previous call stack and step counter.
272    pub(crate) fn interpret_function(
273        &mut self,
274        function_id: FunctionId,
275        arguments: Vec<Value>,
276    ) -> IResults {
277        self.step_counter = 0;
278        self.call_stack.truncate(1);
279        self.call_function(function_id, arguments)
280    }

282    /// Interpret a function call.
283    ///
284    /// Unlike `interpret_function` this does not reset the state;
285    /// it is meant to be used for internal calls.
286    fn call_function(&mut self, function_id: FunctionId, mut arguments: Vec<Value>) -> IResults {
287        self.call_stack.push(CallContext::new(function_id));
288
289        if self.call_stack.len() >= MAX_INTERPRETER_CALL_STACK_SIZE {
290            let call_stack = self
291                .call_stack
292                .iter()
293                .skip(1)
294                .map(|ctx| {
295                    let id = ctx
296                        .called_function
297                        .expect("all but the first global context has a called function");
298                    let name = self.functions[&id].name().to_string();
299                    (id, name)
300                })
301                .collect();
302            return Err(InterpreterError::StackOverflow { call_stack });
303        }
304
305        let function = &self.functions[&function_id];
306        if self.options.trace {
307            println!();
308            println!("enter function {} ({})", function_id, function.name());
309        }
310
311        let mut block_id = function.entry_block();
312        let dfg = self.dfg();
313
314        // Loop over blocks & instructions inline here to avoid pushing more
315        // call frames (in rust). We only push call frames for function calls which
316        // should prevent stack overflows for all but excessively large call stacks
317        // that may overflow in the brillig vm as well.
318        let return_values = loop {
319            let block = &dfg[block_id];
320
321            if arguments.len() != block.parameters().len() {
322                return Err(internal(InternalError::BlockArgumentCountMismatch {
323                    block: block_id,
324                    arguments: arguments.len(),
325                    parameters: block.parameters().len(),
326                }));
327            }
328
329            for (parameter, argument) in block.parameters().iter().zip_eq(arguments) {
330                self.define(*parameter, argument)?;
331            }
332
333            for instruction_id in block.instructions() {
334                let results = dfg.instruction_results(*instruction_id);
335                self.interpret_instruction(&dfg[*instruction_id], results)?;
336            }
337
338            // Account for the terminator; a function might not have actual instructions, other than jumping around.
339            self.inc_step_counter()?;
340
341            match block.terminator() {
342                None => {
343                    return Err(internal(InternalError::BlockMissingTerminator {
344                        block: block_id,
345                    }));
346                }
347                Some(TerminatorInstruction::Jmp { destination, arguments: jump_args, .. }) => {
348                    block_id = *destination;
349                    if self.options.trace {
350                        println!("jump to {block_id}");
351                    }
352                    arguments = self.lookup_all(jump_args)?;
353                }
354                Some(TerminatorInstruction::JmpIf {
355                    condition,
356                    then_destination,
357                    then_arguments,
358                    else_destination,
359                    else_arguments,
360                    call_stack: _,
361                }) => {
362                    (block_id, arguments) = if self.lookup_bool(*condition, "jmpif condition")? {
363                        (*then_destination, self.lookup_all(then_arguments)?)
364                    } else {
365                        (*else_destination, self.lookup_all(else_arguments)?)
366                    };
367                    if self.options.trace {
368                        println!("jump to {block_id}");
369                    }
370                }
371                Some(TerminatorInstruction::Return { return_values, call_stack: _ }) => {
372                    let return_values = self.lookup_all(return_values)?;
373                    if self.options.trace && !return_values.is_empty() {
374                        let return_values =
375                            return_values.iter().map(ToString::to_string).collect::<Vec<_>>();
376                        println!("return {}", return_values.join(", "));
377                    }
378
379                    break return_values;
380                }
381                Some(TerminatorInstruction::Unreachable { .. }) => {
382                    return Err(InterpreterError::ReachedTheUnreachable);
383                }
384            }
385        };
386
387        if self.options.trace {
388            println!("exit function {} ({})", function_id, function.name());
389            println!();
390        }
391
392        self.call_stack.pop();
393
394        if self.options.trace
395            && let Some(context) = self.call_stack.last()
396            && let Some(function_id) = context.called_function
397        {
398            let function = &self.functions[&function_id];
399            println!("back in function {} ({})", function_id, function.name());
400        }
401
402        Ok(return_values)
403    }

405    fn lookup(&self, id: ValueId) -> IResult<Value> {
406        if let Some(value) = self.call_context().scope.get(&id) {
407            return Ok(value.clone());
408        }
409
410        if let Some(value) = self.global_scope().get(&id) {
411            return Ok(value.clone());
412        }
413
414        Ok(match &self.dfg()[id] {
415            super::ir::value::Value::NumericConstant { constant, typ } => {
416                Value::from_constant(*constant, *typ)?
417            }
418            super::ir::value::Value::Function(id) => Value::Function(*id),
419            super::ir::value::Value::Intrinsic(intrinsic) => Value::Intrinsic(*intrinsic),
420            super::ir::value::Value::ForeignFunction { name, .. } => {
421                Value::ForeignFunction(name.clone())
422            }
423            super::ir::value::Value::Instruction { .. }
424            | super::ir::value::Value::Param { .. }
425            | super::ir::value::Value::Global(_) => {
426                unreachable!("`{id}` should already be in scope")
427            }
428        })
429    }

431    fn lookup_helper<T>(
432        &self,
433        value_id: ValueId,
434        instruction: &'static str,
435        expected_type: &'static str,
436        convert: impl FnOnce(&Value) -> Option<T>,
437    ) -> IResult<T> {
438        let value = self.lookup(value_id)?;
439        match convert(&value) {
440            Some(value) => Ok(value),
441            None => {
442                let value = value.to_string();
443                Err(internal(InternalError::TypeError {
444                    value_id,
445                    value,
446                    expected_type,
447                    instruction,
448                }))
449            }
450        }
451    }

481    /// Look up an array index.
482    ///
483    /// If the value exists but is an out-of-range `u32`, returns `IndexOutOfBounds`.
484    fn lookup_array_index(
485        &self,
486        value_id: ValueId,
487        instruction: &'static str,
488        length: u32,
489    ) -> IResult<u32> {
490        self.lookup_helper(value_id, instruction, "u32", Value::as_u32).map_err(|e| {
491            if matches!(e, InterpreterError::Internal(InternalError::TypeError { .. }))
492                && let Ok(Value::Numeric(value)) = self.lookup(value_id)
493                && value.get_type() == NumericType::unsigned(32)
494                && !value.is_in_range()
495            {
496                return InterpreterError::IndexOutOfBounds { index: value.to_field(), length };
497            }
498            e
499        })

594    fn side_effects_enabled(&self, instruction: &Instruction) -> bool {
595        let Some(current_function) = self.try_current_function() else {
596            // If there's no current function it means we are evaluating global instructions
597            return true;
598        };
599
600        match current_function.runtime() {
601            RuntimeType::Acir(_) => {
602                self.call_context().side_effects_enabled
603                    || !instruction.requires_acir_gen_predicate(&current_function.dfg)
604            }
605            RuntimeType::Brillig(_) => true,
606        }
607    }

609    #[allow(unused)]
610    fn interpret_instruction(
611        &mut self,
612        instruction: &Instruction,
613        results: &[ValueId],
614    ) -> IResult<()> {
615        self.inc_step_counter()?;
616
617        let side_effects_enabled = self.side_effects_enabled(instruction);
618
619        match instruction {
620            Instruction::Binary(binary) => {
621                let result = self.interpret_binary(binary, side_effects_enabled)?;
622                self.define(results[0], result)?;
623                Ok(())
624            }
625            // Cast in SSA relabels the value's type, keeping its bit pattern. Like ACIR (where a
626            // cast is a no-op on the underlying field) it does not range-check: a value that
627            // overflowed its source type via unchecked field arithmetic keeps its extended bits.
628            // Narrowing casts are preceded by an explicit `truncate`, so no truncation is needed here.
629            Instruction::Cast(value, numeric_type) => {
630                let value = self.lookup_numeric(*value, "cast")?;
631                let field = value.to_field();
632                let result = Value::int_from_field(field, *numeric_type)?;
633                self.define(results[0], result)?;
634                Ok(())
635            }
636            Instruction::Not(id) => self.interpret_not(*id, results[0]),
637            Instruction::Truncate { value, bit_size, max_bit_size } => {
638                self.interpret_truncate(*value, *bit_size, *max_bit_size, results[0])
639            }
640            Instruction::Constrain(lhs_id, rhs_id, constrain_error) => {
641                let lhs = self.lookup(*lhs_id)?;
642                let rhs = self.lookup(*rhs_id)?;
643                if side_effects_enabled && lhs != rhs {
644                    let lhs = lhs.to_string();
645                    let rhs = rhs.to_string();
646                    let lhs_id = *lhs_id;
647                    let rhs_id = *rhs_id;
648                    let msg = if let Some(ConstrainError::StaticString(msg)) = constrain_error {
649                        Some(msg.clone())
650                    } else {
651                        None
652                    };
653                    return Err(InterpreterError::ConstrainEqFailed {
654                        lhs,
655                        lhs_id,
656                        rhs,
657                        rhs_id,
658                        msg,
659                    });
660                }
661                Ok(())
662            }
663            Instruction::ConstrainNotEqual(lhs_id, rhs_id, constrain_error) => {
664                let lhs = self.lookup(*lhs_id)?;
665                let rhs = self.lookup(*rhs_id)?;
666                if side_effects_enabled && lhs == rhs {
667                    let lhs = lhs.to_string();
668                    let rhs = rhs.to_string();
669                    let lhs_id = *lhs_id;
670                    let rhs_id = *rhs_id;
671                    let msg = if let Some(ConstrainError::StaticString(msg)) = constrain_error {
672                        Some(msg.clone())
673                    } else {
674                        None
675                    };
676                    return Err(InterpreterError::ConstrainNeFailed {
677                        lhs,
678                        lhs_id,
679                        rhs,
680                        rhs_id,
681                        msg,
682                    });
683                }
684                Ok(())
685            }
686            Instruction::RangeCheck { value, max_bit_size, assert_message } => self
687                .interpret_range_check(
688                    *value,
689                    *max_bit_size,
690                    assert_message.as_ref(),
691                    side_effects_enabled,
692                ),
693            Instruction::Call { func, arguments } => {
694                self.interpret_call(*func, arguments, results, side_effects_enabled)
695            }
696            Instruction::Allocate => {
697                self.interpret_allocate(results[0]);
698                Ok(())
699            }
700            Instruction::Load { address } => self.interpret_load(*address, results[0]),
701            Instruction::Store { address, value } => self.interpret_store(*address, *value),
702            Instruction::EnableSideEffectsIf { condition } => {
703                self.call_context_mut().side_effects_enabled =
704                    self.lookup_bool(*condition, "enable_side_effects")?;
705                Ok(())
706            }
707            Instruction::ArrayGet { array, index } => {
708                self.interpret_array_get(*array, *index, results[0], side_effects_enabled)
709            }
710            Instruction::ArraySet { array, index, value, mutable } => self.interpret_array_set(
711                *array,
712                *index,
713                *value,
714                *mutable,
715                results[0],
716                side_effects_enabled,
717            ),
718            Instruction::IncrementRc { value } => self.interpret_inc_rc(*value),
719            Instruction::DecrementRc { value } => self.interpret_dec_rc(*value),
720            Instruction::IfElse { then_condition, then_value, else_condition, else_value } => self
721                .interpret_if_else(
722                    *then_condition,
723                    *then_value,
724                    *else_condition,
725                    *else_value,
726                    results[0],
727                ),
728            Instruction::MakeArray { elements, typ } => {
729                self.interpret_make_array(elements, results[0], typ)
730            }
731            Instruction::Noop => Ok(()),
732        }
733    }

735    fn interpret_not(&mut self, id: ValueId, result: ValueId) -> IResult<()> {
736        let num_value = self.lookup_numeric(id, "not instruction")?;
737
738        if num_value.as_field().is_some() {
739            return Err(internal(InternalError::UnsupportedOperatorForType {
740                operator: "!",
741                typ: "Field",
742            }));
743        }
744        if let Some(value) = num_value.as_bool() {
745            return self.define(result, Value::Numeric(NumericValue::bool(!value)));
746        }
747
748        // Based on AcirContext::not_var: `!x == max - x` where `max == 2^bit_size - 1`, computed on
749        // the bit pattern reduced into range. The result is always in range.
750        let typ = num_value.get_type();
751        let bit_size = num_value.bit_size();
752        let reduced = truncate_field(num_value.to_field(), bit_size);
753        let max =
754            FieldElement::from(2u128).pow(&FieldElement::from(bit_size)) - FieldElement::one();
755        let new_result = NumericValue::int_from_field(max - reduced, typ)?;
756        self.define(result, Value::Numeric(new_result))
757    }

759    fn interpret_truncate(
760        &mut self,
761        value_id: ValueId,
762        bit_size: u32,
763        max_bit_size: u32,
764        result: ValueId,
765    ) -> IResult<()> {
766        let value = self.lookup_numeric(value_id, "truncate")?;
767        let typ = value.get_type();
768        if bit_size == 0 {
769            return Err(internal(InternalError::TruncateToZeroBits { value_id, max_bit_size }));
770        }
771
772        if value.as_bool().is_some() {
773            return self.define(result, Value::Numeric(value));
774        }
775
776        let field = value.to_field();
777
778        let truncated = NumericValue::int_from_field(truncate_field(field, bit_size), typ)?;
779        self.define(result, Value::Numeric(truncated))
780    }

782    fn interpret_range_check(
783        &self,
784        value_id: ValueId,
785        max_bit_size: u32,
786        error_message: Option<&String>,
787        side_effects_enabled: bool,
788    ) -> IResult<()> {
789        if !side_effects_enabled {
790            return Ok(());
791        }
792
793        if max_bit_size == 0 {
794            return Err(internal(InternalError::RangeCheckToZeroBits { value_id }));
795        }
796
797        let value = self.lookup_numeric(value_id, "range check")?;
798
799        // max_bit_size > 0 so u1 always passes; every other variant stores its bit pattern as a
800        // field, so the number of bits is read directly off that field (matching ACIR, which
801        // range-checks the field representation).
802        if value.as_bool().is_some() {
803            return Ok(());
804        }
805        let bit_count = value.to_field().num_bits();
806
807        if bit_count > max_bit_size {
808            let value = value.to_string();
809            let actual_bits = bit_count;
810            let max_bits = max_bit_size;
811
812            Err(InterpreterError::RangeCheckFailed {
813                value,
814                value_id,
815                actual_bits,
816                max_bits,
817                msg: error_message.cloned(),
818            })
819        } else {
820            Ok(())
821        }
822    }

824    fn interpret_call(
825        &mut self,
826        function_id: ValueId,
827        argument_ids: &[ValueId],
828        results: &[ValueId],
829        side_effects_enabled: bool,
830    ) -> IResult<()> {
831        let function = self.lookup(function_id)?;
832        let mut arguments = try_vecmap(argument_ids, |argument| self.lookup(*argument))?;
833
834        let new_results = if side_effects_enabled {
835            match function {
836                Value::Function(id) => {
837                    // If we're crossing a constrained -> unconstrained boundary we have to wipe
838                    // any shared mutable fields in our arguments since brillig should conceptually
839                    // receive fresh array on each invocation.
840                    if !self.in_unconstrained_context()
841                        && self.functions[&id].runtime().is_brillig()
842                    {
843                        for argument in &mut arguments {
844                            Self::reset_array_state(argument)?;
845                        }
846                    }
847                    self.call_function(id, arguments)?
848                }
849                Value::Intrinsic(intrinsic) => {
850                    self.call_intrinsic(intrinsic, argument_ids, results)?
851                }
852                Value::ForeignFunction(name) if self.options.no_foreign_calls => {
853                    return Err(InterpreterError::UnknownForeignFunctionCall { name });
854                }
855                Value::ForeignFunction(name) if name == "print" => self.call_print(arguments)?,
856                Value::ForeignFunction(name) => {
857                    return Err(InterpreterError::UnknownForeignFunctionCall { name });
858                }
859                other => {
860                    return Err(internal(InternalError::CalledNonFunction {
861                        value: other.to_string(),
862                        value_id: function_id,
863                    }));
864                }
865            }
866        } else {
867            self.uninitialized_call_results(&function, argument_ids, results)?
868        };
869
870        if new_results.len() != results.len() {
871            let function_name = self.try_get_function_name(function_id);
872            return Err(internal(InternalError::FunctionReturnedIncorrectArgCount {
873                function: function_id,
874                function_name,
875                expected: results.len(),
876                actual: new_results.len(),
877            }));
878        }
879
880        for (result, new_result) in results.iter().zip_eq(new_results) {
881            self.define(*result, new_result)?;
882        }
883        Ok(())
884    }

886    /// Create uninitialized results for a call that was skipped due to disabled side effects.
887    ///
888    /// For vector intrinsics, we create properly-sized zeroed vectors rather than empty ones,
889    /// to avoid out-of-bounds error after Remove `IfElse` that need to do `array_get` to
890    /// merge the vector from a 'side effect disabled' branch.
891    fn uninitialized_call_results(
892        &self,
893        function: &Value,
894        argument_ids: &[ValueId],
895        results: &[ValueId],
896    ) -> IResult<Vec<Value>> {
897        use crate::ssa::ir::instruction::Intrinsic;
898        // Get the length of the vector
899        if let Value::Intrinsic(intrinsic) = function {
900            let input_vector_info = match intrinsic {
901                Intrinsic::VectorPushBack
902                | Intrinsic::VectorPushFront
903                | Intrinsic::VectorInsert
904                | Intrinsic::VectorPopBack
905                | Intrinsic::VectorPopFront
906                | Intrinsic::VectorRemove => {
907                    let vec = self.lookup_array_or_vector(
908                        argument_ids[1],
909                        "uninitialized vector intrinsic",
910                    )?;
911                    Some((vec.elements.borrow().len(), vec.element_types.clone()))
912                }
913                _ => None,
914            };
915
916            if let Some((input_len, element_types)) = input_vector_info {
917                let element_count = element_types.len();
918                let output_len = match intrinsic {
919                    Intrinsic::VectorPushBack
920                    | Intrinsic::VectorPushFront
921                    | Intrinsic::VectorInsert => input_len + element_count,
922                    Intrinsic::VectorPopBack
923                    | Intrinsic::VectorPopFront
924                    | Intrinsic::VectorRemove => input_len.saturating_sub(element_count),
925                    _ => unreachable!(),
926                };
927
928                return Ok(vecmap(results, |result| {
929                    let typ = self.dfg().type_of_value(*result);
930                    if matches!(*typ, Type::Vector(_)) {
931                        Value::uninitialized_vector(&element_types, output_len, *result)
932                    } else {
933                        Value::uninitialized(&typ, *result)
934                    }
935                }));
936            }
937        }
938
939        Ok(vecmap(results, |result| {
940            let typ = self.dfg().type_of_value(*result);
941            Value::uninitialized(&typ, *result)
942        }))
943    }

958    /// Reset the value's `Shared` states in each array within. This is used to mimic each
959    /// invocation of the brillig vm receiving fresh values. No matter the history of this value
960    /// (e.g. even if they were previously returned from another brillig function) the reference
961    /// count should always be 1 and it shouldn't alias any other arrays.
962    fn reset_array_state(value: &mut Value) -> IResult<()> {
963        match value {
964            Value::Numeric(_)
965            | Value::Function(_)
966            | Value::Intrinsic(_)
967            | Value::ForeignFunction(_) => Ok(()),
968
969            // Immutable references are allowed to cross the constrained->unconstrained
970            // boundary. Mutable references are rejected earlier by the frontend type check.
971            Value::Reference(_) => Ok(()),
972
973            Value::ArrayOrVector(array_value) => {
974                let mut elements = array_value.elements.borrow().to_vec();
975                for element in &mut elements {
976                    Self::reset_array_state(element)?;
977                }
978                array_value.elements = Shared::new(elements);
979                array_value.rc = Shared::new(1);
980                Ok(())
981            }
982        }
983    }

1024    fn interpret_array_get(
1025        &mut self,
1026        array: ValueId,
1027        index: ValueId,
1028        result: ValueId,
1029        side_effects_enabled: bool,
1030    ) -> IResult<()> {
1031        // When there is a problem indexing the array, but side effects are disabled,
1032        // define the value as uninitialized.
1033        let uninitialized = |this: &mut Self| {
1034            let typ = this.dfg().type_of_value(result);
1035            let value = Value::uninitialized(&typ, result);
1036            this.define(result, value)
1037        };
1038
1039        let offset = self.dfg().array_offset(array, index);
1040        let array = self.lookup_array_or_vector(array, "array get")?;
1041        let length = array.elements.borrow().len() as u32;
1042
1043        // Per `Instruction::requires_acir_gen_predicate`, in Brillig an
1044        // `array_get` is pure-in-isolation: the OOB check is inserted as a
1045        // separate constraint, not part of the access itself. Match that here
1046        // so the interpreter agrees with the Brillig VM on dead/unused gets.
1047        let oob_is_pure = self.current_function().runtime().is_brillig();
1048
1049        let index = match self.lookup_array_index(index, "array get index", length) {
1050            Err(InterpreterError::IndexOutOfBounds { .. })
1051                if !side_effects_enabled || oob_is_pure =>
1052            {
1053                return uninitialized(self);
1054            }
1055            other => other?,
1056        };
1057        let mut index = index - offset.to_u32();
1058
1059        if length == 0 {
1060            // Accessing an array of 0-len is replaced by asserting
1061            // the branch is not-taken during acir-gen and
1062            // a zeroed type is used in case of array get
1063            // So we can simply replace it with uninitialized value
1064            if side_effects_enabled && !oob_is_pure {
1065                return Err(InterpreterError::IndexOutOfBounds { index: index.into(), length });
1066            } else {
1067                return uninitialized(self);
1068            }
1069        }
1070
1071        if oob_is_pure && index >= length {
1072            return uninitialized(self);
1073        }
1074
1075        let element = {
1076            // An array_get with false side_effects_enabled is replaced
1077            // by a load at a valid index during acir-gen.
1078            if !side_effects_enabled {
1079                // Find a valid index
1080                let typ = self.dfg().type_of_value(result);
1081                for (i, element) in array.elements.borrow().iter().enumerate() {
1082                    if element.get_type() == *typ {
1083                        index = i as u32;
1084                        break;
1085                    }
1086                }
1087            }
1088            let elements = array.elements.borrow();
1089            let element = elements
1090                .get(index as usize)
1091                .ok_or(InterpreterError::IndexOutOfBounds { index: index.into(), length })?;
1092
1093            // Either return a fresh nested array (in constrained context) or just clone the element.
1094            if !self.in_unconstrained_context() {
1095                if let Some(array) = element.as_array_or_vector() {
1096                    // In the ACIR runtime we expect fresh arrays when accessing a nested array.
1097                    // If we do not clone the elements here a mutable array set afterwards could mutate
1098                    // not just this returned array but the array we are fetching from in this array get.
1099                    Value::ArrayOrVector(ArrayValue {
1100                        elements: Shared::new(array.elements.borrow().to_vec()),
1101                        rc: array.rc,
1102                        element_types: array.element_types,
1103                        length: array.length,
1104                    })
1105                } else {
1106                    element.clone()
1107                }
1108            } else {
1109                element.clone()
1110            }
1111        };
1112        self.define(result, element)?;
1113        Ok(())
1114    }

1116    #[allow(clippy::too_many_arguments)]
1117    fn interpret_array_set(
1118        &mut self,
1119        array: ValueId,
1120        index: ValueId,
1121        value: ValueId,
1122        mutable: bool,
1123        result: ValueId,
1124        side_effects_enabled: bool,
1125    ) -> IResult<()> {
1126        let offset = self.dfg().array_offset(array, index);
1127        let array = self.lookup_array_or_vector(array, "array set")?;
1128
1129        let result_array = if side_effects_enabled {
1130            let length = array.elements.borrow().len() as u32;
1131            let index = self.lookup_array_index(index, "array set index", length)?;
1132            let index = index - offset.to_u32();
1133            let value = self.lookup(value)?;
1134
1135            let is_rc_one = *array.rc.borrow() == 1;
1136            let should_mutate = if self.in_unconstrained_context() { is_rc_one } else { mutable };
1137
1138            if index >= length {
1139                return Err(InterpreterError::IndexOutOfBounds { index: index.into(), length });
1140            }
1141
1142            if should_mutate {
1143                array.elements.borrow_mut()[index as usize] = value;
1144                Value::ArrayOrVector(array)
1145            } else {
1146                if !is_rc_one {
1147                    Self::decrement_rc(&array);
1148                }
1149                let mut elements = array.elements.borrow().to_vec();
1150                elements[index as usize] = value;
1151                let elements = Shared::new(elements);
1152                let rc = Shared::new(1);
1153                let element_types = array.element_types.clone();
1154                let length = array.length;
1155                Value::ArrayOrVector(ArrayValue { elements, rc, element_types, length })
1156            }
1157        } else {
1158            // Side effects are disabled, return the original array
1159            Value::ArrayOrVector(array)
1160        };
1161        self.define(result, result_array)?;
1162        Ok(())
1163    }

1198    fn interpret_if_else(
1199        &mut self,
1200        then_condition_id: ValueId,
1201        then_value: ValueId,
1202        else_condition_id: ValueId,
1203        else_value: ValueId,
1204        result: ValueId,
1205    ) -> IResult<()> {
1206        let then_condition = self.lookup_bool(then_condition_id, "then condition")?;
1207        let else_condition = self.lookup_bool(else_condition_id, "else condition")?;
1208        let then_value = self.lookup(then_value)?;
1209        let else_value = self.lookup(else_value)?;
1210
1211        // Note that `then_condition = !else_condition` doesn't always hold!
1212        // Notably if this is a nested if expression we could have something like:
1213        //   then_condition = outer_condition & a
1214        //   else_condition = outer_condition & !a
1215        // If `outer_condition` is false, both will be false.
1216        if then_condition && else_condition {
1217            return Err(InterpreterError::DoubleTrueIfElse {
1218                then_condition_id,
1219                else_condition_id,
1220            });
1221        }
1222
1223        let new_result = if !then_condition && !else_condition {
1224            // Returning uninitialized/zero if both conditions are false to match
1225            // the decomposition of `cond * then_value + !cond * else_value` for numeric values.
1226            let typ = self.dfg().type_of_value(result);
1227            Value::uninitialized(&typ, result)
1228        } else if then_condition {
1229            then_value
1230        } else {
1231            else_value
1232        };
1233
1234        self.define(result, new_result)
1235    }

1237    fn interpret_make_array(
1238        &mut self,
1239        elements: &im::Vector<ValueId>,
1240        result: ValueId,
1241        result_type: &Type,
1242    ) -> IResult<()> {
1243        let elements = try_vecmap(elements, |element| self.lookup(*element))?;
1244        let length = if let Type::Array(_, length) = result_type { Some(*length) } else { None };
1245
1246        // The number of elements in the array must be a multiple of the number of element types
1247        let element_types = result_type.element_types();
1248        if element_types.is_empty() {
1249            if !elements.is_empty() {
1250                return Err(internal(InternalError::MakeArrayElementCountMismatch {
1251                    result,
1252                    elements_count: elements.len(),
1253                    types_count: element_types.len(),
1254                }));
1255            }
1256        } else if elements.len() % element_types.len() != 0 {
1257            return Err(internal(InternalError::MakeArrayElementCountMismatch {
1258                result,
1259                elements_count: elements.len(),
1260                types_count: element_types.len(),
1261            }));
1262        }
1263
1264        // Make sure each element's type matches the one in element_types
1265        for (index, (element, expected_type)) in
1266            elements.iter().zip(element_types.iter().cycle()).enumerate()
1267        {
1268            let actual_type = element.get_type();
1269            if !actual_type.canonical_eq(expected_type) {
1270                return Err(internal(InternalError::MakeArrayElementTypeMismatch {
1271                    result,
1272                    index,
1273                    actual_type: actual_type.to_string(),
1274                    expected_type: expected_type.to_string(),
1275                }));
1276            }
1277        }
1278
1279        let array = Value::ArrayOrVector(ArrayValue {
1280            elements: Shared::new(elements),
1281            rc: Shared::new(1),
1282            element_types,
1283            length,
1284        });
1285        self.define(result, array)
1286    }

1301/// Evaluate an integer (non-`Field`, non-`u1`) binary operation in the field+type model.
1302///
1303/// `lhs`/`rhs` are integer values of the same type whose stored field is the two's-complement bit
1304/// pattern (which may be out of range in ACIR mode). The runtime selects overflow behaviour:
1305/// Brillig wraps in fixed-width registers, while ACIR carries the extended field and range-checks at
1306/// checked operations. Comparisons, bitwise ops, shifts, div/mod and *signed* checked arithmetic
1307/// reduce their operands and reuse [`eval_constant_binary_op`]; only unchecked arithmetic and
1308/// *unsigned* checked arithmetic depend on the runtime.
1309fn evaluate_integer_binary(
1310    binary: &Binary,
1311    lhs: NumericValue,
1312    rhs: NumericValue,
1313    is_brillig: bool,
1314    display_binary: impl Fn(&Binary) -> String,
1315) -> IResult<NumericValue> {
1316    use BinaryOp::{Add, Mul, Sub};
1317
1318    let operator = binary.operator;
1319    let typ = lhs.get_type();
1320    let bit_size = lhs.bit_size();
1321    let lhs_field = lhs.to_field();
1322    let rhs_field = rhs.to_field();
1323
1324    let overflow = || overflow_error(binary, &display_binary, lhs, rhs);
1325
1326    // Field arithmetic for add/sub/mul (no wrapping, no reduction).
1327    let field_arith = || match operator {
1328        Add { .. } => lhs_field + rhs_field,
1329        Sub { .. } => lhs_field - rhs_field,
1330        Mul { .. } => lhs_field * rhs_field,
1331        _ => unreachable!("field_arith called on a non-arithmetic operator"),
1332    };
1333
1334    match operator {
1335        // Unchecked arithmetic: Brillig wraps to the bit size; ACIR extends in the field.
1336        Add { unchecked: true } | Sub { unchecked: true } | Mul { unchecked: true } => {
1337            let result = if is_brillig {
1338                let a = bits_u128(lhs_field, bit_size);
1339                let b = bits_u128(rhs_field, bit_size);
1340                let wrapped = match operator {
1341                    Add { .. } => a.wrapping_add(b),
1342                    Sub { .. } => a.wrapping_sub(b),
1343                    Mul { .. } => a.wrapping_mul(b),
1344                    _ => unreachable!(),
1345                };
1346                FieldElement::from(truncate(wrapped, bit_size))
1347            } else {
1348                field_arith()
1349            };
1350            NumericValue::int_from_field(result, typ)
1351        }
1352
1353        // Unsigned checked arithmetic. ACIR computes in the field and range-checks the result, so an
1354        // out-of-range operand or an overflowing result is rejected; Brillig does true fixed-width
1355        // checked arithmetic. (They only diverge once values can exceed the field modulus, i.e.
1356        // `u128`, but keeping them separate is faithful to both.)
1357        Add { unchecked: false } | Sub { unchecked: false } | Mul { unchecked: false }
1358            if !lhs.is_signed() =>
1359        {
1360            if is_brillig {
1361                eval_via_constant_binary_op(lhs_field, rhs_field, operator, typ, binary, &overflow)
1362            } else {
1363                let value = NumericValue::int_from_field(field_arith(), typ)?;
1364                if value.is_in_range() { Ok(value) } else { Err(overflow()) }
1365            }
1366        }
1367
1368        // Shifts wrap the value to the bit size (a shift amount >= the bit width is an overflow,
1369        // matching `checked_shl`/`checked_shr`); they are not folded as checked overflows.
1370        BinaryOp::Shl | BinaryOp::Shr => {
1371            let value_bits = bits_u128(lhs_field, bit_size);
1372            let shift = bits_u128(rhs_field, bit_size);
1373            if shift >= u128::from(bit_size) {
1374                return Err(overflow());
1375            }
1376            let shift = shift as u32;
1377            let result = match operator {
1378                BinaryOp::Shl => {
1379                    FieldElement::from(truncate(value_bits.wrapping_shl(shift), bit_size))
1380                }
1381                // Arithmetic shift for signed values, logical shift for unsigned.
1382                BinaryOp::Shr if lhs.is_signed() => {
1383                    let signed = try_convert_field_element_to_signed_integer(
1384                        FieldElement::from(value_bits),
1385                        bit_size,
1386                    )
1387                    .expect("a reduced signed value converts");
1388                    convert_signed_integer_to_field_element(signed >> shift, bit_size)
1389                }
1390                BinaryOp::Shr => FieldElement::from(value_bits >> shift),
1391                _ => unreachable!(),
1392            };
1393            NumericValue::int_from_field(result, typ)
1394        }
1395
1396        // Signed checked arithmetic, div/mod, comparisons and bitwise ops all reduce their
1397        // operands; reuse the constant-folder's semantics.
1398        _ => eval_via_constant_binary_op(lhs_field, rhs_field, operator, typ, binary, &overflow),
1399    }
1400}

1402/// Apply [`eval_constant_binary_op`] to two operand fields and map the result into the interpreter's
1403/// value/error types. `eval_constant_binary_op` truncates its operands to the type's bit size, so
1404/// this is the "reduce then compute" path shared by both runtimes.
1405fn eval_via_constant_binary_op(
1406    lhs_field: FieldElement,
1407    rhs_field: FieldElement,
1408    operator: BinaryOp,
1409    typ: NumericType,
1410    binary: &Binary,
1411    overflow: &impl Fn() -> InterpreterError,
1412) -> IResult<NumericValue> {
1413    // Reduce the operands to their bit-pattern representatives first. `eval_constant_binary_op`
1414    // also truncates internally, but its signed conversion requires the field to fit in `u128`,
1415    // which an extended (out-of-range) ACIR operand such as `p - delta` does not.
1416    let bit_size = typ.bit_size::<FieldElement>();
1417    let lhs_field = truncate_field(lhs_field, bit_size);
1418    let rhs_field = truncate_field(rhs_field, bit_size);
1419
1420    match eval_constant_binary_op(lhs_field, rhs_field, operator, typ) {
1421        BinaryEvaluationResult::Success(field, result_type) => {
1422            NumericValue::int_from_field(field, result_type)
1423        }
1424        BinaryEvaluationResult::Failure(_) => {
1425            let divisor_is_zero = matches!(operator, BinaryOp::Div | BinaryOp::Mod)
1426                && truncate_field(rhs_field, typ.bit_size::<FieldElement>()).is_zero();
1427            if divisor_is_zero {
1428                Err(division_by_zero(binary.lhs, lhs_field, binary.rhs, rhs_field))
1429            } else {
1430                Err(overflow())
1431            }
1432        }
1433        // Only an overflowing `shl` reports this; treat it as an overflow.
1434        BinaryEvaluationResult::CouldNotEvaluate => Err(overflow()),
1435    }
1436}

1438fn evaluate_binary(
1439    binary: &Binary,
1440    lhs: NumericValue,
1441    rhs: NumericValue,
1442    side_effects_enabled: bool,
1443    is_brillig: bool,
1444    display_binary: impl Fn(&Binary) -> String,
1445) -> IResult<NumericValue> {
1446    let lhs_id = binary.lhs;
1447    let rhs_id = binary.rhs;
1448
1449    if lhs.get_type() != rhs.get_type() {
1450        return Err(internal(InternalError::MismatchedTypesInBinaryOperator {
1451            lhs_id,
1452            lhs: lhs.to_string(),
1453            operator: binary.operator,
1454            rhs_id,
1455            rhs: rhs.to_string(),
1456        }));
1457    }
1458
1459    // Disable this instruction if it is side-effectual and side effects are disabled.
1460    if !side_effects_enabled {
1461        return Ok(NumericValue::zero(lhs.get_type()));
1462    }
1463
1464    if let (Some(lhs), Some(rhs)) = (lhs.as_field(), rhs.as_field()) {
1465        return interpret_field_binary_op(lhs, binary.operator, rhs, lhs_id, rhs_id);
1466    }
1467
1468    if let (Some(lhs), Some(rhs)) = (lhs.as_bool(), rhs.as_bool()) {
1469        return interpret_u1_binary_op(lhs, rhs, binary, &display_binary);
1470    }
1471
1472    evaluate_integer_binary(binary, lhs, rhs, is_brillig, display_binary)
1473}
1474
1475fn interpret_field_binary_op(
1476    lhs: FieldElement,
1477    operator: BinaryOp,
1478    rhs: FieldElement,
1479    lhs_id: ValueId,
1480    rhs_id: ValueId,
1481) -> IResult<NumericValue> {
1482    let unsupported_operator = |operator| -> IResult<NumericValue> {
1483        let typ = "Field";
1484        Err(internal(InternalError::UnsupportedOperatorForType { operator, typ }))
1485    };
1486
1487    let result = match operator {
1488        BinaryOp::Add { unchecked: _ } => NumericValue::field(lhs + rhs),
1489        BinaryOp::Sub { unchecked: _ } => NumericValue::field(lhs - rhs),
1490        BinaryOp::Mul { unchecked: _ } => NumericValue::field(lhs * rhs),
1491        BinaryOp::Div => {
1492            if rhs.is_zero() {
1493                return Err(division_by_zero(lhs_id, lhs, rhs_id, rhs));
1494            }
1495            NumericValue::field(lhs / rhs)
1496        }
1497        BinaryOp::Mod => return unsupported_operator("%"),
1498        BinaryOp::Eq => NumericValue::bool(lhs == rhs),
1499        BinaryOp::Lt => NumericValue::bool(lhs < rhs),
1500        BinaryOp::And => return unsupported_operator("&"),
1501        BinaryOp::Or => return unsupported_operator("|"),
1502        BinaryOp::Xor => return unsupported_operator("^"),
1503        BinaryOp::Shl => return unsupported_operator("<<"),
1504        BinaryOp::Shr => return unsupported_operator(">>"),
1505    };
1506    Ok(result)
1507}

1509fn interpret_u1_binary_op(
1510    lhs: bool,
1511    rhs: bool,
1512    binary: &Binary,
1513    display_binary: &impl Fn(&Binary) -> String,
1514) -> IResult<NumericValue> {
1515    let overflow = || overflow_error(binary, display_binary, lhs, rhs);
1516
1517    let lhs_id = binary.lhs;
1518    let rhs_id = binary.rhs;
1519
1520    let result = match binary.operator {
1521        BinaryOp::Add { unchecked: true } => lhs ^ rhs,
1522        BinaryOp::Add { unchecked: false } => {
1523            if lhs && rhs {
1524                return Err(overflow());
1525            } else {
1526                lhs ^ rhs
1527            }
1528        }
1529        BinaryOp::Sub { unchecked: true } => {
1530            if !lhs && rhs {
1531                return Err(overflow());
1532            } else {
1533                lhs ^ rhs
1534            }
1535        }
1536        BinaryOp::Sub { unchecked: false } => {
1537            if !lhs && rhs {
1538                return Err(overflow());
1539            } else {
1540                lhs ^ rhs
1541            }
1542        }
1543        BinaryOp::Mul { unchecked: _ } => lhs & rhs, // (*) = (&) for u1
1544        BinaryOp::Div => {
1545            // (0, 0) -> (division by 0)
1546            // (0, 1) -> 0
1547            // (1, 0) -> (division by 0)
1548            // (1, 1) -> 1
1549            if !rhs {
1550                return Err(division_by_zero(lhs_id, u8::from(lhs), rhs_id, u8::from(rhs)));
1551            }
1552            lhs
1553        }
1554        BinaryOp::Mod => {
1555            // (0, 0) -> (division by 0)
1556            // (0, 1) -> 0
1557            // (1, 0) -> (division by 0)
1558            // (1, 1) -> 0
1559            if !rhs {
1560                let lhs = format!("u1 {}", u8::from(lhs));
1561                let rhs = format!("u1 {}", u8::from(rhs));
1562                return Err(division_by_zero(lhs_id, lhs, rhs_id, rhs));
1563            }
1564            false
1565        }
1566        BinaryOp::Eq => lhs == rhs,
1567        // clippy complains when you do `lhs < rhs` and recommends this instead
1568        BinaryOp::Lt => !lhs & rhs,
1569        BinaryOp::And => lhs & rhs,
1570        BinaryOp::Or => lhs | rhs,
1571        BinaryOp::Xor => lhs ^ rhs,
1572        BinaryOp::Shl => {
1573            if rhs {
1574                return Err(overflow());
1575            } else {
1576                lhs
1577            }
1578        }
1579        BinaryOp::Shr => {
1580            if rhs {
1581                return Err(overflow());
1582            } else {
1583                lhs
1584            }
1585        }
1586    };
1587    Ok(NumericValue::bool(result))
1588}

1590#[cfg_attr(not(test), expect(dead_code))]
1591fn truncate_unsigned<T>(value: T, bit_size: u32) -> IResult<T>
1592where
1593    u128: From<T>,
1594    T: TryFrom<u128>,
1595    <T as TryFrom<u128>>::Error: std::fmt::Debug,
1596{
1597    let value_u128 = u128::from(value);
1598    let bit_mask = match bit_size.cmp(&MAX_UNSIGNED_BIT_SIZE) {
1599        Ordering::Less => (1u128 << bit_size) - 1,
1600        Ordering::Equal => u128::MAX,
1601        Ordering::Greater => {
1602            return Err(internal(InternalError::InvalidUnsignedTruncateBitSize { bit_size }));
1603        }
1604    };
1605
1606    let result = value_u128 & bit_mask;
1607    Ok(T::try_from(result).expect(
1608        "The truncated result should always be smaller than or equal to the original `value`",
1609    ))
1610}

1612/// Reduce `field` to its `bit_size`-bit pattern as a `u128`.
1613///
1614/// ACIR operands may carry an extended (out-of-range) field; truncating to the bit size yields the
1615/// canonical bit pattern, which always fits in `u128` for the integer widths the interpreter models.
1616fn bits_u128(field: FieldElement, bit_size: u32) -> u128 {
1617    truncate_field(field, bit_size).try_into_u128().expect("a reduced value fits in u128")
1618}

1620/// Build an `Overflow` error for `binary`, rendering the instruction as
1621/// `` `<ssa instruction>` (<operator> <lhs>, <rhs>) ``.
