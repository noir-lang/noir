52/// Generates brillig bytecode which computes the inverse of its input if not null, else zero.
53pub(crate) fn directive_invert<F: AcirField>() -> GeneratedBrillig<F> {
54    // We generate the following code:
55    // fn invert(x : Field) -> Field {
56    //    1 / x
57    // }
58
59    // The input argument, ie the value that will be inverted.
60    // We store the result in this register too.
61    let input = MemoryAddress::direct(0);
62    let one_const = MemoryAddress::direct(1);
63    let zero_const = MemoryAddress::direct(2);
64    let input_is_zero = MemoryAddress::direct(3);
65    let zero_usize = MemoryAddress::direct(20);
66    let one_usize = MemoryAddress::direct(21);
67    // Location of the stop opcode
68    let stop_location = 8;
69
70    GeneratedBrillig {
71        byte_code: vec![
72            BrilligOpcode::Const {
73                destination: one_usize,
74                bit_size: BitSize::Integer(IntegerBitSize::U32),
75                value: F::from(1_usize),
76            },
77            BrilligOpcode::Const {
78                destination: zero_usize,
79                bit_size: BitSize::Integer(IntegerBitSize::U32),
80                value: F::from(0_usize),
81            },
82            BrilligOpcode::CalldataCopy {
83                destination_address: input,
84                size_address: one_usize,
85                offset_address: zero_usize,
86            },
87            // Put value zero in register (2)
88            BrilligOpcode::Const {
89                destination: zero_const,
90                value: F::from(0_usize),
91                bit_size: BitSize::Field,
92            },
93            BrilligOpcode::BinaryFieldOp {
94                op: BinaryFieldOp::Equals,
95                lhs: input,
96                rhs: zero_const,
97                destination: input_is_zero,
98            },
99            // If the input is zero, then we jump to the stop opcode
100            BrilligOpcode::JumpIf { condition: input_is_zero, location: stop_location },
101            // Put value one in register (1)
102            BrilligOpcode::Const {
103                destination: one_const,
104                value: F::one(),
105                bit_size: BitSize::Field,
106            },
107            // Divide 1 by the input, and set the result of the division into register (0)
108            BrilligOpcode::BinaryFieldOp {
109                op: BinaryFieldOp::Div,
110                lhs: one_const,
111                rhs: input,
112                destination: input,
113            },
114            BrilligOpcode::Stop {
115                return_data: HeapVector { pointer: zero_usize, size: one_usize },
116            },
117        ],
118        name: "directive_invert".to_string(),
119        ..Default::default()
120    }
121}

123/// Generates brillig bytecode which computes `a / b` and returns the quotient and remainder.
124///
125/// This is equivalent to the Noir (pseudo)code
126///
127/// ```text
128/// fn quotient<T>(a: T, b: T) -> (T,T) {
129///    (a/b, a-a/b*b)
130/// }
131/// ```
132pub(crate) fn directive_quotient<F: AcirField>() -> GeneratedBrillig<F> {
133    // `a` is (0) (i.e register index 0)
134    // `b` is (1)
135
136    GeneratedBrillig {
137        byte_code: vec![
138            BrilligOpcode::Const {
139                destination: MemoryAddress::direct(10),
140                bit_size: BitSize::Integer(IntegerBitSize::U32),
141                value: F::from(2_usize),
142            },
143            BrilligOpcode::Const {
144                destination: MemoryAddress::direct(11),
145                bit_size: BitSize::Integer(IntegerBitSize::U32),
146                value: F::from(0_usize),
147            },
148            BrilligOpcode::CalldataCopy {
149                destination_address: MemoryAddress::direct(0),
150                size_address: MemoryAddress::direct(10),
151                offset_address: MemoryAddress::direct(11),
152            },
153            // No cast, since calldata is typed as field by default
154            // q = a/b is set into register (2)
155            BrilligOpcode::BinaryFieldOp {
156                op: BinaryFieldOp::IntegerDiv, // We want integer division, not field division!
157                lhs: MemoryAddress::direct(0),
158                rhs: MemoryAddress::direct(1),
159                destination: MemoryAddress::direct(2),
160            },
161            // (1)= q*b
162            BrilligOpcode::BinaryFieldOp {
163                op: BinaryFieldOp::Mul,
164                lhs: MemoryAddress::direct(2),
165                rhs: MemoryAddress::direct(1),
166                destination: MemoryAddress::direct(1),
167            },
168            // (1) = a-q*b
169            BrilligOpcode::BinaryFieldOp {
170                op: BinaryFieldOp::Sub,
171                lhs: MemoryAddress::direct(0),
172                rhs: MemoryAddress::direct(1),
173                destination: MemoryAddress::direct(1),
174            },
175            // (0) = q
176            BrilligOpcode::Mov {
177                destination: MemoryAddress::direct(0),
178                source: MemoryAddress::direct(2),
179            },
180            BrilligOpcode::Stop {
181                return_data: HeapVector {
182                    pointer: MemoryAddress::direct(11),
183                    size: MemoryAddress::direct(10),
184                },
185            },
186        ],
187        name: "directive_integer_quotient".to_string(),
188        ..Default::default()
189    }
190}

192/// Generates brillig bytecode which performs a radix-base decomposition of `a`
193/// The brillig inputs are 'a', the numbers of limbs and the radix
194pub(crate) fn directive_to_radix<F: AcirField>() -> GeneratedBrillig<F> {
195    let memory_adr_int_size = IntegerBitSize::U32;
196    let memory_adr_size = BitSize::Integer(memory_adr_int_size);
197
198    // (0) is the input field `a` to decompose
199    // (1) contains the number of limbs (second input)
200    let limbs_nb = MemoryAddress::direct(1);
201    // (2) contains the radix (third input)
202    let radix = MemoryAddress::direct(2);
203    // (3) and (4) are intermediate registers
204    // (5,6,7) are constants: 0,1,3
205    let zero = MemoryAddress::direct(5);
206    let one = MemoryAddress::direct(6);
207    let three = MemoryAddress::direct(7);
208    // (7) is the iteration bound, it is the same register as three because the latter is only used at the start
209    let bound = MemoryAddress::direct(7);
210    // (8) is the register for storing the loop condition
211    let cond = MemoryAddress::direct(8);
212    // (9) is the pointer to the result array
213    let result_pointer = MemoryAddress::direct(9);
214    // address of the result array
215    let result_base_adr = 10_usize;
216
217    let result_vector = HeapVector { pointer: result_pointer, size: limbs_nb };
218
219    let byte_code = vec![
220        // Initialize registers
221        // Constants
222        // Zero
223        BrilligOpcode::Const { destination: zero, bit_size: memory_adr_size, value: F::zero() },
224        // One
225        BrilligOpcode::Const {
226            destination: one,
227            bit_size: memory_adr_size,
228            value: F::from(1_usize),
229        },
230        // Three
231        BrilligOpcode::Const {
232            destination: three,
233            bit_size: memory_adr_size,
234            value: F::from(3_usize),
235        },
236        // Brillig Inputs
237        BrilligOpcode::CalldataCopy {
238            destination_address: MemoryAddress::direct(0),
239            size_address: three,
240            offset_address: zero,
241        },
242        // The number of limbs needs to be an integer
243        BrilligOpcode::Cast { destination: limbs_nb, source: limbs_nb, bit_size: memory_adr_size },
244        // Result_pointer starts at the base address
245        BrilligOpcode::Const {
246            destination: result_pointer,
247            bit_size: memory_adr_size,
248            value: F::from(result_base_adr),
249        },
250        // Loop bound
251        BrilligOpcode::BinaryIntOp {
252            destination: bound,
253            op: BinaryIntOp::Add,
254            bit_size: memory_adr_int_size,
255            lhs: result_pointer,
256            rhs: limbs_nb,
257        },
258        // loop label: (3) = a / radix (integer division)
259        BrilligOpcode::BinaryFieldOp {
260            op: BinaryFieldOp::IntegerDiv,
261            lhs: MemoryAddress::direct(0),
262            rhs: radix,
263            destination: MemoryAddress::direct(3),
264        },
265        // (4) = (3)*256
266        BrilligOpcode::BinaryFieldOp {
267            op: BinaryFieldOp::Mul,
268            lhs: MemoryAddress::direct(3),
269            rhs: radix,
270            destination: MemoryAddress::direct(4),
271        },
272        // (4) = a-(3)*256 (remainder)
273        BrilligOpcode::BinaryFieldOp {
274            op: BinaryFieldOp::Sub,
275            lhs: MemoryAddress::direct(0),
276            rhs: MemoryAddress::direct(4),
277            destination: MemoryAddress::direct(4),
278        },
279        // Store the remainder in the result array
280        BrilligOpcode::Store {
281            destination_pointer: result_pointer,
282            source: MemoryAddress::direct(4),
283        },
284        // Increment the result pointer
285        BrilligOpcode::BinaryIntOp {
286            op: BinaryIntOp::Add,
287            lhs: result_pointer,
288            rhs: one,
289            destination: result_pointer,
290            bit_size: memory_adr_int_size,
291        },
292        // a := quotient
293        BrilligOpcode::Mov {
294            destination: MemoryAddress::direct(0),
295            source: MemoryAddress::direct(3),
296        },
297        // loop condition
298        BrilligOpcode::BinaryIntOp {
299            op: BinaryIntOp::LessThan,
300            lhs: result_pointer,
301            rhs: bound,
302            destination: cond,
303            bit_size: memory_adr_int_size,
304        },
305        // loop back
306        BrilligOpcode::JumpIf { condition: cond, location: 7 },
307        // reset result pointer to the start of the array
308        BrilligOpcode::Const {
309            destination: result_pointer,
310            bit_size: memory_adr_size,
311            value: F::from(result_base_adr),
312        },
313        BrilligOpcode::Stop { return_data: result_vector },
314    ];
315
316    GeneratedBrillig { byte_code, name: "directive_to_radix".to_string(), ..Default::default() }
317}
