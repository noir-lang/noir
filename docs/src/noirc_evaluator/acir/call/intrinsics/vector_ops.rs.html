14    /// Pushes one or more elements to the back of a non-nested vector.
15    ///
16    /// # Arguments
17    ///
18    /// * `arguments[0]` - Current vector length
19    /// * `arguments[1]` - Vector contents
20    /// * `arguments[2..]` - Elements to push back
21    /// * `result_ids[0]` - Updated vector length
22    /// * `result_ids[1]` - Update vector contents
23    ///
24    /// # Returns
25    ///
26    /// A vector of [AcirValue]s containing:
27    /// 1. Updated vector length (incremented by one)
28    /// 2. New vector with elements appended
29    pub(super) fn convert_vector_push_back(
30        &mut self,
31        arguments: &[ValueId],
32        dfg: &DataFlowGraph,
33        result_ids: &[ValueId],
34    ) -> Result<Vec<AcirValue>, RuntimeError> {
35        let vector_length = self.convert_value(arguments[0], dfg).into_var()?;
36        let vector_contents = arguments[1];
37        let elements_to_push = &arguments[2..];
38
39        // Increase the vector length by one to enable accessing more elements in the vector.
40        let one = self.acir_context.add_constant(FieldElement::one());
41
42        let vector = self.convert_value(vector_contents, dfg);
43        let vector_typ = dfg.type_of_value(vector_contents);
44
45        let new_vector_val = if let Some(len_const) = dfg.get_numeric_constant(arguments[0]) {
46            // Length is known at compile time - we can precisely determine where to write
47            let mut new_vector = self.read_array_with_type(vector, &vector_typ)?;
48            // length of Acir Values vector
49            let len = len_const.to_u128() as usize * elements_to_push.len();
50            for (i, elem) in elements_to_push.iter().enumerate() {
51                let element = self.convert_value(*elem, dfg);
52                let write_index = len + i;
53
54                // If the array is already large enough, replace the element at the write position.
55                // Otherwise, append to the end.
56                if write_index < new_vector.len() {
57                    new_vector[write_index] = element;
58                } else {
59                    new_vector.push_back(element);
60                }
61            }
62            AcirValue::Array(new_vector)
63        } else {
64            // Length is not known, we are going to:
65            // 1. Push dummy data to the vector, so that it's capacity covers for the push_back
66            // 2. Generate a DynamicArray corresponding to the new vector flattened content
67            // 3. Write the elements to push to this array at the correct length
68            let value_types = flat_element_types(&vector_typ);
69            let Type::Vector(vector_types) = &vector_typ else {
70                unreachable!("ICE: vector operation on a non vector type");
71            };
72
73            let mut elements_var = Vec::new();
74            let mut element_size = 0;
75            let mut new_vector = self.read_array_with_type(vector.clone(), &vector_typ)?;
76            let zero = self.acir_context.add_constant(FieldElement::zero());
77
78            // 1. Convert the elements-to-push into flattened acir_var and at the same time
79            // push_back corresponding dummy zero values to the AcirValues vector.
80            for (elem, ssa_typ) in elements_to_push.iter().zip(vector_types.to_vec()) {
81                let element = self.convert_value(*elem, dfg);
82                element_size += super::arrays::flattened_value_size(&element).to_usize();
83                match element {
84                    AcirValue::Var(acir_var, acir_type) => {
85                        new_vector.push_back(AcirValue::Var(zero, acir_type));
86                        elements_var.push(acir_var);
87                    }
88                    AcirValue::Array(vector) => {
89                        let zero_value = self.array_zero_value(&ssa_typ)?;
90                        new_vector.push_back(zero_value);
91                        for acir_value in vector {
92                            let acir_vars = self.flatten(&acir_value)?;
93                            elements_var.extend(acir_vars);
94                        }
95                    }
96                    AcirValue::DynamicArray(_) => {
97                        unimplemented!(
98                            "pushing a dynamic array into a vector is not yet supported"
99                        );
100                    }
101                }
102            }
103
104            // The actual flattened size of new_vector after the dummy push_back
105            let new_vector_array = AcirValue::Array(new_vector);
106            let len = super::arrays::flattened_value_size(&new_vector_array);
107
108            // 2. Copy the vector into an AcirDynamicArray
109            // Generates the element_type_sizes array
110            let element_type_sizes =
111                if super::arrays::array_has_constant_element_size(&vector_typ).is_none() {
112                    Some(self.init_element_type_sizes_array(
113                        &vector_typ,
114                        result_ids[1],
115                        Some(new_vector_array.clone()),
116                        dfg,
117                        // We do not need extra capacity here as `new_vector_array` has already pushed back new elements
118                        ElementTypeSizesArrayShift::None,
119                    )?)
120                } else {
121                    None
122                };
123
124            // The block ID for the new vector is the one for the resulting vector
125            let block_id = self.block_id(result_ids[1]);
126            self.initialize_array(block_id, len, Some(new_vector_array))?;
127            let flattened_dynamic_array =
128                AcirDynamicArray { block_id, len, value_types, element_type_sizes };
129
130            // 3. Write to the dynamic array
131
132            // 3.1 Computes the flatten_idx where to write into the dynamic array:
133            // Use element_type_size if it exists; convert the user index (vector_length) into the AcirValues index,
134            // and then flatten it with element_type_size
135            let mut flatten_idx = if let Some(element_type_sizes) = element_type_sizes {
136                let predicate_index = self
137                    .acir_context
138                    .mul_var(vector_length, self.current_side_effects_enabled_var)?;
139                let acir_element_size = self.acir_context.add_constant(elements_to_push.len());
140                let acir_value_index =
141                    self.acir_context.mul_var(predicate_index, acir_element_size)?;
142                self.acir_context
143                    .read_from_memory(element_type_sizes, &acir_value_index)
144                    .map_err(RuntimeError::from)?
145            } else {
146                // If it does not exist; the array is homogenous and we can simply multiply by size of the array elements
147                let element_size_var = self.acir_context.add_constant(element_size);
148                self.acir_context.mul_var(vector_length, element_size_var)?
149            };
150            // Write the elements to the dynamic array
151            for element in &elements_var {
152                self.acir_context.write_to_memory(block_id, &flatten_idx, element)?;
153                flatten_idx = self.acir_context.add_var(flatten_idx, one)?;
154            }
155            AcirValue::DynamicArray(flattened_dynamic_array)
156        };
157
158        let new_vector_length = self.acir_context.add_var(vector_length, one)?;
159        Ok(vec![AcirValue::Var(new_vector_length, NumericType::length_type()), new_vector_val])
160    }

162    /// Pushes one or more elements to the front of a non-nested vector.
163    ///
164    /// # Arguments
165    ///
166    /// * `arguments[0]` - Current vector length
167    /// * `arguments[1]` - Vector contents
168    /// * `arguments[2..]` - Elements to push to the front
169    ///
170    /// # Returns
171    ///
172    /// A vector of [AcirValue]s containing:
173    /// 1. Updated vector length (incremented by one)
174    /// 2. New vector with elements prepended
175    pub(super) fn convert_vector_push_front(
176        &mut self,
177        arguments: &[ValueId],
178        dfg: &DataFlowGraph,
179    ) -> Result<Vec<AcirValue>, RuntimeError> {
180        let vector_length = self.convert_value(arguments[0], dfg).into_var()?;
181        let vector_contents = arguments[1];
182        let elements_to_push = &arguments[2..];
183
184        // Increase the vector length by one to enable accessing more elements in the vector.
185        let one = self.acir_context.add_constant(FieldElement::one());
186        let new_vector_length = self.acir_context.add_var(vector_length, one)?;
187
188        let vector = self.convert_value(vector_contents, dfg);
189        let vector_type = dfg.type_of_value(vector_contents);
190        let mut new_vector = self.read_array_with_type(vector, &vector_type)?;
191
192        // We must directly push front elements for non-nested vectors
193        for elem in elements_to_push.iter().rev() {
194            let element = self.convert_value(*elem, dfg);
195            new_vector.push_front(element);
196        }
197
198        let new_vector_val = AcirValue::Array(new_vector);
199
200        Ok(vec![AcirValue::Var(new_vector_length, NumericType::length_type()), new_vector_val])
201    }

203    /// Removes and returns one or more elements from the back of a non-nested vector.
204    ///
205    /// # Arguments
206    ///
207    /// * `arguments[0]` - Current vector length
208    /// * `arguments[1]` - Vector contents
209    /// * `result_ids[0]` - Updated vector length
210    /// * `result_ids[1]` - Update vector contents
211    /// * `result_ids[2..]` - Locations where popped elements will be stored
212    ///
213    /// # Returns
214    ///
215    /// A vector of [AcirValue]s containing:
216    /// 1. Updated vector length (decremented by one)
217    /// 2. Updated vector contents with the back elements removed
218    /// 3. Popped elements in order
219    ///
220    /// # Design
221    ///
222    /// The vector is represented in **flattened form** in memory. Popping the back
223    /// involves:
224    ///
225    /// 1. Decrementing the vector length by one.
226    /// 2. Using the decremented length as an offset for the elements to remove.
227    /// 3. Read out the elements located at that offset.
228    ///
229    /// The `result_ids` provided by the SSA to fetch the appropriate type information to be popped.
230    /// The `result_ids` encode the type/shape of the removed element.
231    ///
232    /// # Empty Vector Handling
233    ///
234    /// If the vector has zero length, this function skips the memory read and returns zero values.
235    /// It asserts that the current side effects must be disabled (predicate = 0), otherwise fails
236    /// with "cannot pop from a vector with length 0". This prevents reading from empty memory blocks
237    /// which would cause "Index out of bounds" errors.
238    pub(super) fn convert_vector_pop_back(
239        &mut self,
240        arguments: &[ValueId],
241        dfg: &DataFlowGraph,
242        result_ids: &[ValueId],
243    ) -> Result<Vec<AcirValue>, RuntimeError> {
244        let vector_length_var = arguments[0];
245        let vector_contents = arguments[1];
246
247        let vector_value = self.convert_value(vector_length_var, dfg);
248        let vector_length = vector_value.clone().into_var()?;
249        let block_id = self.ensure_array_is_initialized(vector_contents, dfg)?;
250        let vector = self.convert_value(vector_contents, dfg);
251
252        if self.has_zero_length(vector_contents, dfg) {
253            // Make sure this code is disabled, or fail with "Index out of bounds".
254            let msg = "cannot pop from a vector with length 0".to_string();
255            self.acir_context.assert_zero_var(self.current_side_effects_enabled_var, msg)?;
256
257            // Fill the result with default values.
258            let mut results = Vec::with_capacity(result_ids.len());
259
260            // The results shall be: [new len, new vector, ...popped]
261            results.push(vector_value);
262            results.push(vector);
263
264            for result_id in &result_ids[2..] {
265                let result_type = dfg.type_of_value(*result_id);
266                let result_zero = self.array_zero_value(&result_type)?;
267                results.push(result_zero);
268            }
269
270            return Ok(results);
271        }
272
273        // For unknown length under a side effect variable, we want to multiply with the side effect variable
274        // to ensure we don't end up trying to look up an item at index -1, when the semantic length is 0.
275        let is_unknown_length = dfg.get_numeric_constant(vector_length_var).is_none();
276
277        let one = self.acir_context.add_constant(FieldElement::one());
278        let mut new_vector_length = self.acir_context.sub_var(vector_length, one)?;
279
280        if is_unknown_length {
281            new_vector_length = self
282                .acir_context
283                .mul_var(new_vector_length, self.current_side_effects_enabled_var)?;
284        }
285
286        // For a pop back operation we want to fetch from the `length - 1` as this is the
287        // last valid index that can be accessed in a vector. After the pop back operation
288        // the elements stored at that index will no longer be able to be accessed.
289        let mut var_index = new_vector_length;
290
291        let vector_type = dfg.type_of_value(vector_contents);
292        let item_size = vector_type.element_types();
293        // Must read from the flattened last index of the vector in case the vector contains nested arrays.
294        let flat_item_size: FlattenedLength =
295            item_size.iter().map(|typ| typ.flattened_size()).sum();
296        let item_size = self.acir_context.add_constant(flat_item_size.0);
297        var_index = self.acir_context.mul_var(var_index, item_size)?;
298
299        let mut popped_elements = Vec::new();
300        for res in &result_ids[2..] {
301            let elem = self.array_get_value(&dfg.type_of_value(*res), block_id, &mut var_index)?;
302            popped_elements.push(elem);
303        }
304
305        let mut new_vector = self.read_array_with_type(vector, &vector_type)?;
306        for _ in 0..popped_elements.len() {
307            new_vector.pop_back();
308        }
309
310        let mut results = vec![
311            AcirValue::Var(new_vector_length, NumericType::length_type()),
312            AcirValue::Array(new_vector),
313        ];
314        results.append(&mut popped_elements);
315
316        Ok(results)
317    }

319    /// Removes and returns one or more elements from the front of a non-nested vector.
320    ///
321    /// # Arguments
322    ///
323    /// * `arguments[0]` - Current vector length
324    /// * `arguments[1]` - Vector contents
325    /// * `result_ids[..element_size]` - Locations for the popped elements
326    /// * `result_ids[element_size]` - Updated vector length
327    /// * `result_ids[element_size + 1]` - Updated vector contents
328    ///
329    /// `element_size` refers to the result of [crate::ssa::ir::types::Type::element_size].
330    ///
331    /// # Returns
332    ///
333    /// A vector of [AcirValue]s containing:
334    /// 1. Popped elements in order
335    /// 2. Updated vector length (decremented by one)
336    /// 3. Updated vector contents with the front elements removed
337    ///
338    /// # Design
339    ///
340    /// Vectors are stored in **flattened form** in memory. To pop from the front:
341    ///
342    /// 1. Decrement the vector length by the size of one element.
343    /// 2. Read out the first `element_size` values at index `0`.
344    /// 3. Shift the update vector's memory forward by `element_size` slots to represent the updated vector.
345    ///
346    /// Unlike in [Self::convert_vector_pop_back], the returned vector contents differ from the input:
347    /// the underlying array is logically truncated at the *front* rather than
348    /// the back. The `result_ids` ensure that this logical shift is applied
349    /// consistently with the element's type.
350    ///
351    /// # Empty Vector Handling
352    ///
353    /// If the vector has zero length, this function skips the memory read and returns zero values.
354    /// It asserts that the current side effects must be disabled (predicate = 0), otherwise fails
355    /// with "cannot pop from a vector with length 0". This prevents reading from empty memory blocks
356    /// which would cause "Index out of bounds" errors.
357    pub(super) fn convert_vector_pop_front(
358        &mut self,
359        arguments: &[ValueId],
360        dfg: &DataFlowGraph,
361        result_ids: &[ValueId],
362    ) -> Result<Vec<AcirValue>, RuntimeError> {
363        let vector_length = self.convert_value(arguments[0], dfg).into_var()?;
364        let vector_contents = arguments[1];
365
366        let vector_typ = dfg.type_of_value(vector_contents);
367        let block_id = self.ensure_array_is_initialized(vector_contents, dfg)?;
368
369        // Check if we're trying to pop from a known empty vector.
370        if self.has_zero_length(vector_contents, dfg) {
371            // Make sure this code is disabled, or fail with "Index out of bounds".
372            let msg = "cannot pop from a vector with length 0".to_string();
373            self.acir_context.assert_zero_var(self.current_side_effects_enabled_var, msg)?;
374
375            // Fill the result with default values.
376            let mut results = Vec::with_capacity(result_ids.len());
377
378            let element_size = vector_typ.element_size();
379            // For pop_front, results order is: [popped_elements..., new_len, new_vector]
380            for result_id in &result_ids[..element_size.to_usize()] {
381                let result_type = dfg.type_of_value(*result_id);
382                let result_zero = self.array_zero_value(&result_type)?;
383                results.push(result_zero);
384            }
385
386            let vector_value = self.convert_value(arguments[0], dfg);
387            results.push(vector_value);
388
389            let vector = self.convert_value(vector_contents, dfg);
390            results.push(vector);
391
392            return Ok(results);
393        }
394
395        // Check that the vector length is not zero.
396        // This is different from the previous check as this is a runtime check.
397        let zero = self.acir_context.add_constant(FieldElement::zero());
398        let assert_message = self.acir_context.generate_assertion_message_payload(
399            "Attempt to pop_front from an empty vector".to_string(),
400        );
401        self.acir_context.assert_neq_var(
402            vector_length,
403            zero,
404            self.current_side_effects_enabled_var,
405            Some(assert_message),
406        )?;
407
408        let one = self.acir_context.add_constant(FieldElement::one());
409        let new_vector_length = self.acir_context.sub_var(vector_length, one)?;
410
411        let vector = self.convert_value(vector_contents, dfg);
412
413        let mut new_vector = self.read_array_with_type(vector, &vector_typ)?;
414        let element_size = vector_typ.element_size();
415
416        let mut popped_elements: Vec<AcirValue> = Vec::new();
417        let mut var_index = self.acir_context.add_constant(FieldElement::zero());
418        // Fetch the values we are popping off of the vector.
419        // In the case of non-nested vector the logic is simple as we do not
420        // need to account for the internal vector sizes or flattening the index.
421        for res in &result_ids[..element_size.to_usize()] {
422            let element =
423                self.array_get_value(&dfg.type_of_value(*res), block_id, &mut var_index)?;
424            popped_elements.push(element);
425        }
426
427        let popped_elements_size = popped_elements.len();
428
429        new_vector = new_vector.slice(popped_elements_size..);
430        popped_elements.push(AcirValue::Var(new_vector_length, NumericType::length_type()));
431        popped_elements.push(AcirValue::Array(new_vector));
432
433        Ok(popped_elements)
434    }

436    /// Inserts one or more elements into a vector at a given index.
437    ///
438    /// # Arguments
439    ///
440    /// * `arguments[0]` - Current vector length
441    /// * `arguments[1]` - Vector contents
442    /// * `arguments[2]` - Insert index (logical element index, not flattened)
443    /// * `arguments[3..]` - Elements to insert
444    /// * `result_ids[0]` - Updated vector length
445    /// * `result_ids[1]` - Updated vector contents
446    ///
447    /// # Returns
448    ///
449    /// A vector of [AcirValue]s containing:
450    /// 1. Updated vector length (incremented by one)
451    /// 2. Updated vector contents with the new elements inserted at the given index
452    ///
453    /// # Design
454    ///
455    /// Vectors are represented in **flattened form** in memory. Inserting requires
456    /// shifting a contiguous region of elements upward to make room for the new ones:
457    ///
458    /// 1. Compute the flattened insert index:
459    ///    - Multiply the logical insert index by the element size.
460    ///    - Adjust for non-homogenous structures via [Self::get_flattened_index].
461    /// 2. Flatten the new elements (`flattened_elements`)
462    /// 3. For each position in the result vector:
463    ///    - If below the insert index, copy from the original vector.
464    ///    - If within the insertion window, write values from `flattened_elements`.
465    ///    - If above the window, shift elements upward by the size of the inserted data.
466    /// 4. Initialize a new memory block for the resulting vector, ensuring its type information is preserved.
467    ///
468    /// # Empty Vector Handling
469    ///
470    /// If the vector has zero length, this function skips the memory read and returns zero values.
471    /// It asserts that the current side effects must be disabled (predicate = 0), otherwise fails
472    /// with "Index out of bounds, vector has size 0". This prevents reading from empty memory blocks
473    /// which would cause "Index out of bounds" errors.
474    pub(super) fn convert_vector_insert(
475        &mut self,
476        arguments: &[ValueId],
477        dfg: &DataFlowGraph,
478        result_ids: &[ValueId],
479    ) -> Result<Vec<AcirValue>, RuntimeError> {
480        let vector_length = self.convert_value(arguments[0], dfg).into_var()?;
481        let vector_contents = arguments[1];
482
483        let vector_typ = dfg.type_of_value(vector_contents);
484        let block_id = self.ensure_array_is_initialized(vector_contents, dfg)?;
485
486        // Check if we're trying to insert into an empty vector
487        if self.has_zero_length(vector_contents, dfg) {
488            // Make sure this code is disabled, or fail with "Index out of bounds".
489            let msg = "Index out of bounds, vector has size 0".to_string();
490            self.acir_context.assert_zero_var(self.current_side_effects_enabled_var, msg)?;
491
492            // Fill the result with default values.
493            let mut results = Vec::with_capacity(result_ids.len());
494
495            // For insert, results are: [new_len, new_vector]
496            let vector_length_value = self.convert_value(arguments[0], dfg);
497            results.push(vector_length_value);
498
499            let vector = self.convert_value(vector_contents, dfg);
500            results.push(vector);
501
502            return Ok(results);
503        }
504
505        let vector = self.convert_value(vector_contents, dfg);
506        let insert_index = self.convert_value(arguments[2], dfg).into_var()?;
507
508        let one = self.acir_context.add_constant(FieldElement::one());
509        let new_vector_length = self.acir_context.add_var(vector_length, one)?;
510
511        let mut vector_size: FlattenedLength = super::arrays::flattened_value_size(&vector);
512
513        let elements_to_insert = &arguments[3..];
514
515        // Fetch the flattened index from the user provided index argument.
516        let item_size = self.acir_context.add_constant(elements_to_insert.len());
517        let is_safe_index = Self::is_index_safe(arguments[2], dfg, &vector_typ, vector_size);
518        let insert_index = self.acir_context.mul_var(insert_index, item_size)?;
519
520        // Because the insert index might be at the end of the slice, the element type sizes we
521        // index here need to have room for this extra element.
522        let shift = ElementTypeSizesArrayShift::Increase;
523        let flat_user_index = self.get_flattened_index(
524            &vector_typ,
525            vector_contents,
526            insert_index,
527            dfg,
528            is_safe_index,
529            shift,
530        )?;
531
532        // Determine the elements we need to write into our resulting dynamic array.
533        // We need to a fully flat list of AcirVar's as a dynamic array is represented with flat memory.
534        let mut inner_elem_size: FlattenedLength = FlattenedLength(0);
535        let mut flattened_elements = Vec::new();
536        for elem in elements_to_insert {
537            let element = self.convert_value(*elem, dfg);
538            // Flatten into (AcirVar, NumericType) pairs
539            let flat_element = self.flatten(&element)?;
540            let elem_size = FlattenedLength(assert_u32(flat_element.len()));
541            inner_elem_size += elem_size;
542            vector_size += elem_size;
543            flattened_elements.extend(flat_element);
544        }
545
546        let inner_elem_size_var = self.acir_context.add_constant(inner_elem_size.0);
547        // Set the maximum flattened index at which a new element should be inserted.
548        let max_flat_user_index =
549            self.acir_context.add_var(flat_user_index, inner_elem_size_var)?;
550
551        // Go through the entire vector argument and determine what value should be written to the new vector.
552        // 1. If we are below the starting insertion index we should insert the value that was already
553        //    in the original vector.
554        // 2. If we are above the starting insertion index but below the max insertion index we should insert
555        //    the flattened element arguments.
556        // 3. If we are above the max insertion index we should insert the previous value from the original vector,
557        //    as during an insertion we want to shift all elements after the insertion up an index.
558        let result_block_id = self.block_id(result_ids[1]);
559        self.initialize_array(result_block_id, vector_size, None)?;
560        let mut current_insert_index = 0;
561
562        let vector_size_usize = vector_size.to_usize();
563        let inner_elem_size_usize = inner_elem_size.to_usize();
564
565        // This caches each `is_after_insert` var for each index for an optimization that is
566        // explained below, above `is_after_insert`.
567        let mut cached_is_after_inserts = Vec::with_capacity(vector_size_usize);
568
569        for i in 0..vector_size_usize {
570            let current_index = self.acir_context.add_constant(i);
571
572            // Check that we are above the lower bound of the insertion index
573            let is_after_insert =
574                self.acir_context.more_than_eq_var(current_index, flat_user_index, 64)?;
575            cached_is_after_inserts.push(is_after_insert);
576
577            // Check that we are below the upper bound of the insertion index
578            let is_before_insert = if i >= inner_elem_size_usize {
579                // Optimization: we first note that `max_flat_user_index = flat_user_index + inner_elem_size`.
580                // Then we note that at each index we do these comparisons:
581                // - is_after_insert: `i >= flat_user_index`
582                // - is_before_insert: `i < (flat_user_index + inner_elem_size)`
583                //
584                // As `i` is incremented, for example to `i + n`, we get:
585                // - is_before_insert: `i + n < (flat_user_index + inner_elem_size)`
586                // If `n == inner_elem_size` then we have:
587                // - is_before_insert: `i + n < (flat_user_index + n)` which is equivalent to:
588                // - is_before_insert: `i < flat_user_index`
589                // Then we note that this is the opposite of `i >= flat_user_index`.
590                // So once `i >= inner_elem_size` we can use the previously made comparisons, negated,
591                // instead of performing them again (for dynamic indexes they incur a brillig call).
592                let cached_is_after_insert = cached_is_after_inserts[i - inner_elem_size_usize];
593                self.acir_context.sub_var(one, cached_is_after_insert)?
594            } else {
595                self.acir_context.less_than_var(current_index, max_flat_user_index, 64)?
596            };
597
598            // Read from the original vector the value we want to insert into our new vector.
599            // We need to make sure that we read the previous element when our current index is greater than insertion index.
600            // If the index for the previous element is out of the array bounds we can avoid the check for whether
601            // the current index is over the insertion index.
602            let shifted_index = if i < inner_elem_size_usize {
603                current_index
604            } else {
605                let index_minus_elem_size =
606                    self.acir_context.add_constant(i - inner_elem_size_usize);
607
608                let use_shifted_index_pred =
609                    self.acir_context.mul_var(index_minus_elem_size, is_after_insert)?;
610
611                let not_pred = self.acir_context.sub_var(one, is_after_insert)?;
612                let use_current_index_pred = self.acir_context.mul_var(not_pred, current_index)?;
613
614                self.acir_context.add_var(use_shifted_index_pred, use_current_index_pred)?
615            };
616
617            let value_shifted_index =
618                self.acir_context.read_from_memory(block_id, &shifted_index)?;
619
620            // Final predicate to determine whether we are within the insertion bounds
621            let should_insert_value_pred =
622                self.acir_context.mul_var(is_after_insert, is_before_insert)?;
623            let insert_value_pred = self
624                .acir_context
625                .mul_var(flattened_elements[current_insert_index], should_insert_value_pred)?;
626
627            let not_pred = self.acir_context.sub_var(one, should_insert_value_pred)?;
628            let shifted_value_pred = self.acir_context.mul_var(not_pred, value_shifted_index)?;
629
630            let new_value = self.acir_context.add_var(insert_value_pred, shifted_value_pred)?;
631
632            self.acir_context.write_to_memory(result_block_id, &current_index, &new_value)?;
633
634            current_insert_index += 1;
635            if inner_elem_size_usize == current_insert_index {
636                current_insert_index = 0;
637            }
638        }
639
640        let element_type_sizes =
641            if super::arrays::array_has_constant_element_size(&vector_typ).is_none() {
642                // Note that here we pass `Some(vector)` as the supplied acir value. This is
643                // the input vector before insertion, so we still need an increase shift here.
644                Some(self.init_element_type_sizes_array(
645                    &vector_typ,
646                    result_ids[1],
647                    Some(vector),
648                    dfg,
649                    shift,
650                )?)
651            } else {
652                None
653            };
654
655        let value_types = flat_numeric_types(&vector_typ);
656
657        let result = AcirValue::DynamicArray(AcirDynamicArray {
658            block_id: result_block_id,
659            len: vector_size,
660            value_types,
661            element_type_sizes,
662        });
663
664        Ok(vec![AcirValue::Var(new_vector_length, NumericType::length_type()), result])
665    }

667    /// Removes one or more elements from a vector at a given index.
668    ///
669    /// # Arguments
670    ///
671    /// * `arguments[0]` - Current vector length
672    /// * `arguments[1]` - Vector contents
673    /// * `arguments[2]` - Removal index (logical element index, not flattened)
674    /// * `result_ids[0]` - Updated vector length
675    /// * `result_ids[1]` - Updated vector contents
676    /// * `result_ids[2..]` - Locations for the removed elements
677    ///
678    /// # Returns
679    ///
680    /// A vector of [AcirValue]s containing:
681    /// 1. Updated vector length (decremented by one)
682    /// 2. Updated vector contents with the target elements removed
683    /// 3. The removed elements, in order
684    ///
685    /// # Design
686    ///
687    /// Vectors are stored in **flattened form** in memory. Removing requires
688    /// shifting a contiguous region of elements downward to overwrite the removed values:
689    ///
690    /// 1. Compute the flattened remove index:
691    ///    - Multiply the logical remove index by the element size.
692    ///    - Adjust for non-homogenous structures via [Self::get_flattened_index].
693    /// 2. Read out the element(s) to be removed:
694    ///    - Iterate over `result_ids[2..(2 + element_size)]`
695    ///    - `element_size` refers to the result of [crate::ssa::ir::types::Type::element_size].
696    ///    - Use these IDs to fetch the appropriate type information for the values to remove and drive `array_get_value`.
697    ///      While extracting the values to remove we compute the total `popped_elements_size` (the flattened width of the removed data).
698    /// 3. For each index in the result vector:
699    ///   - If the index is below the remove index, copy directly.
700    ///   - If the index is at or beyond the removed element, fetch the value from `index + popped_elements_size`
701    ///     in the original vector and write it to the current index.
702    ///   - If `index + popped_elements_size` would exceed the vector length we do nothing. This ensures safe access at the tail of the array
703    ///     and is safe to do as we are decreasing the vector length which gates vector accesses.
704    /// 4. Initialize a new memory block for the resulting vector, ensuring its type information is preserved.
705    ///
706    /// # Empty Vector Handling
707    ///
708    /// If the vector has zero length, this function skips the memory read and returns zero values.
709    /// It asserts that the current side effects must be disabled (predicate = 0), otherwise fails
710    /// with "Index out of bounds, vector has size 0". This prevents reading from empty memory blocks
711    /// which would cause "Index out of bounds" errors.
712    pub(super) fn convert_vector_remove(
713        &mut self,
714        arguments: &[ValueId],
715        dfg: &DataFlowGraph,
716        result_ids: &[ValueId],
717    ) -> Result<Vec<AcirValue>, RuntimeError> {
718        // arguments = [vector_length, vector_contents, remove_index]
719        let vector_length = self.convert_value(arguments[0], dfg).into_var()?;
720        let vector_contents = arguments[1];
721
722        let vector_typ = dfg.type_of_value(vector_contents);
723        let block_id = self.ensure_array_is_initialized(vector_contents, dfg)?;
724
725        // Check if we're trying to remove from an empty vector
726        if self.has_zero_length(vector_contents, dfg) {
727            // Make sure this code is disabled, or fail with "Index out of bounds".
728            let msg = "Index out of bounds, vector has size 0".to_string();
729            self.acir_context.assert_zero_var(self.current_side_effects_enabled_var, msg)?;
730
731            // Fill the result with default values.
732            let mut results = Vec::with_capacity(result_ids.len());
733
734            // For remove, results are: [new_len, new_vector, ...removed_elements]
735            let vector_length_value = self.convert_value(arguments[0], dfg);
736            results.push(vector_length_value);
737
738            let vector = self.convert_value(vector_contents, dfg);
739            results.push(vector);
740
741            // Add zero values for removed elements
742            for result_id in &result_ids[2..] {
743                let result_type = dfg.type_of_value(*result_id);
744                let result_zero = self.array_zero_value(&result_type)?;
745                results.push(result_zero);
746            }
747
748            return Ok(results);
749        }
750
751        let vector = self.convert_value(vector_contents, dfg);
752        let remove_index = self.convert_value(arguments[2], dfg).into_var()?;
753
754        let one = self.acir_context.add_constant(FieldElement::one());
755        let new_vector_length = self.acir_context.sub_var(vector_length, one)?;
756
757        let vector_size = super::arrays::flattened_value_size(&vector);
758
759        let flat_vector = self.flatten(&vector)?;
760        // Compiler sanity check
761        assert_eq!(
762            FlattenedLength(assert_u32(flat_vector.len())),
763            vector_size,
764            "ICE: The read flattened vector should match the computed size"
765        );
766
767        let item_size = vector_typ.element_size().to_usize();
768        let item_size_var = self.acir_context.add_constant(item_size);
769        let remove_index = self.acir_context.mul_var(remove_index, item_size_var)?;
770        let is_safe_index = Self::is_index_safe(arguments[2], dfg, &vector_typ, vector_size);
771
772        // Fetch the flattened index from the user provided index argument.
773        let flat_user_index = self.get_flattened_index(
774            &vector_typ,
775            vector_contents,
776            remove_index,
777            dfg,
778            is_safe_index,
779            ElementTypeSizesArrayShift::None,
780        )?;
781
782        // Fetch the values we are remove from the vector.
783        // As we fetch the values we can determine the size of the removed values
784        // which we will later use for writing the correct resulting vector.
785        let mut popped_elements = Vec::new();
786        let mut popped_elements_size: FlattenedLength = FlattenedLength(0);
787        // Set a temp index just for fetching from the original vector as `array_get_value` mutates
788        // the index internally.
789        let mut temp_index = flat_user_index;
790        let element_size = vector_typ.element_size().to_usize();
791        for res in &result_ids[2..(2 + element_size)] {
792            let element =
793                self.array_get_value(&dfg.type_of_value(*res), block_id, &mut temp_index)?;
794            let elem_size = super::arrays::flattened_value_size(&element);
795            popped_elements_size += elem_size;
796            popped_elements.push(element);
797        }
798
799        let popped_elements_size = popped_elements_size;
800
801        // Go through the entire vector argument and determine what value should be written to the new vector.
802        // 1. If the current index is greater than the removal index we must write the next value
803        //    from the original vector to the current index
804        // 2. For indices beyond the range of the removed elements (i + popped_elements_size >= vector_size),
805        //    we skip shifting because there is no element to move.
806        //    This prevents out-of-bounds reads from the original vector.
807        let result_block_id = self.block_id(result_ids[1]);
808        // We expect a preceding check to have been laid down that the remove index is within bounds.
809        // In practice `popped_elements_size` should never exceed the `vector_size` but we do a saturating sub to be safe.
810        let result_size = FlattenedLength(vector_size.0.saturating_sub(popped_elements_size.0));
811        self.initialize_array(result_block_id, result_size, None)?;
812        for (i, current_value) in flat_vector.iter().enumerate().take(result_size.to_usize()) {
813            let current_index = self.acir_context.add_constant(i);
814
815            let shifted_index = self.acir_context.add_constant(i + popped_elements_size.to_usize());
816
817            // Fetch the value from the initial vector
818            let value_shifted_index =
819                self.acir_context.read_from_memory(block_id, &shifted_index)?;
820
821            let use_shifted_value =
822                self.acir_context.more_than_eq_var(current_index, flat_user_index, 64)?;
823
824            let shifted_value_pred =
825                self.acir_context.mul_var(value_shifted_index, use_shifted_value)?;
826            let not_pred = self.acir_context.sub_var(one, use_shifted_value)?;
827            let current_value_pred = self.acir_context.mul_var(not_pred, *current_value)?;
828
829            let new_value = self.acir_context.add_var(shifted_value_pred, current_value_pred)?;
830
831            self.acir_context.write_to_memory(result_block_id, &current_index, &new_value)?;
832        }
833
834        let element_type_sizes =
835            if super::arrays::array_has_constant_element_size(&vector_typ).is_none() {
836                // The resulting vector has one less element than before
837                let shift = ElementTypeSizesArrayShift::Decrease;
838                Some(self.init_element_type_sizes_array(
839                    &vector_typ,
840                    result_ids[1],
841                    Some(vector),
842                    dfg,
843                    shift,
844                )?)
845            } else {
846                None
847            };
848
849        let value_types = flat_numeric_types(&vector_typ);
850
851        let result = AcirValue::DynamicArray(AcirDynamicArray {
852            block_id: result_block_id,
853            len: result_size,
854            value_types,
855            element_type_sizes,
856        });
857
858        let mut result =
859            vec![AcirValue::Var(new_vector_length, NumericType::length_type()), result];
860        result.append(&mut popped_elements);
861
862        Ok(result)
863    }

865    /// Returns true if the user-facing index is less than the vector capacity
866    fn is_index_safe(
867        index: ValueId,
868        dfg: &DataFlowGraph,
869        vector_typ: &Type,
870        vector_size: FlattenedLength,
871    ) -> bool {
872        // Compute the number of logical elements in the vector by dividing the fully flattened size (in ACIR)
873        // by the vector's elements size (in ACIR), and checking that value against the SSA index.
874        let acir_fields_per_element: usize = match vector_typ {
875            Type::Array(elements, _) | Type::Vector(elements) => {
876                elements.iter().map(|typ| typ.flattened_size().to_usize()).sum()
877            }
878            _ => unreachable!("element_flattened_size: Expected array or vector"),
879        };
880
881        if let Some(index_const) = dfg.get_numeric_constant(index) {
882            let index_const = index_const.to_u128() as usize;
883            // A vector with zero-sized elements has a flattened_size of 0 because of the multiplication with element size.
884            // As a result, every index will be out-of-bound.
885            if acir_fields_per_element == 0 {
886                false
887            } else {
888                let num_logical_elements = vector_size.to_usize() / acir_fields_per_element;
889                index_const < num_logical_elements
890            }
891        } else {
892            false
893        }
894    }
