20impl Context<'_> {
21    /// Returns a vector of `AcirVar`s constrained to be result of the function call.
22    ///
23    /// The function being called is required to be intrinsic.
24    pub(super) fn convert_ssa_intrinsic_call(
25        &mut self,
26        intrinsic: Intrinsic,
27        arguments: &[ValueId],
28        dfg: &DataFlowGraph,
29        result_ids: &[ValueId],
30    ) -> Result<Vec<AcirValue>, RuntimeError> {
31        match intrinsic {
32            Intrinsic::Hint(Hint::BlackBox) => {
33                // Identity function; at the ACIR level this is a no-op, it only affects the SSA.
34                assert_eq!(
35                    arguments.len(),
36                    result_ids.len(),
37                    "ICE: BlackBox input and output lengths should match."
38                );
39                Ok(arguments.iter().map(|v| self.convert_value(*v, dfg)).collect())
40            }
41            Intrinsic::BlackBox(black_box) => {
42                // Vector arguments to blackbox functions would break the following logic (due to being split over two `ValueIds`)
43                // No blackbox functions currently take vector arguments so we have an assertion here to catch if this changes in the future.
44                assert!(
45                    !arguments.iter().any(|arg| matches!(dfg.type_of_value(*arg), Type::Vector(_))),
46                    "ICE: Vector arguments passed to blackbox function"
47                );
48
49                let inputs = vecmap(arguments, |arg| self.convert_value(*arg, dfg));
50
51                let output_count: FlattenedLength = result_ids
52                    .iter()
53                    .map(|result_id| dfg.type_of_value(*result_id).flattened_size())
54                    .sum();
55
56                let vars = self.acir_context.black_box_function(
57                    black_box,
58                    inputs,
59                    None,
60                    output_count,
61                    Some(self.current_side_effects_enabled_var),
62                )?;
63
64                Ok(self.convert_vars_to_values(vars, dfg, result_ids))
65            }
66            Intrinsic::ToRadix(endian) => {
67                let field = self.convert_value(arguments[0], dfg).into_var()?;
68                let radix = self.convert_value(arguments[1], dfg).into_var()?;
69
70                let Type::Array(result_type, array_length) = dfg.type_of_value(result_ids[0])
71                else {
72                    unreachable!("ICE: ToRadix result must be an array");
73                };
74                assert!(
75                    result_type.len() == 1,
76                    "ICE: ToRadix result type must have a single element type"
77                );
78                let Type::Numeric(numeric_type) = result_type[0] else {
79                    unreachable!("ICE: ToRadix result element type must be numeric");
80                };
81
82                self.acir_context
83                    .radix_decompose(endian, field, radix, array_length, numeric_type)
84                    .map(|array| vec![array])
85            }
86            Intrinsic::ToBits(endian) => {
87                let field = self.convert_value(arguments[0], dfg).into_var()?;
88
89                let Type::Array(result_type, array_length) = dfg.type_of_value(result_ids[0])
90                else {
91                    unreachable!("ICE: ToBits result must be an array");
92                };
93                assert!(
94                    result_type.len() == 1,
95                    "ICE: ToBits result type must have a single element type"
96                );
97                let Type::Numeric(numeric_type) = result_type[0] else {
98                    unreachable!("ICE: ToBits result element type must be numeric");
99                };
100
101                self.acir_context
102                    .bit_decompose(endian, field, array_length, numeric_type)
103                    .map(|array| vec![array])
104            }
105            Intrinsic::AsVector => {
106                let array_contents = arguments[0];
107                let array_type = dfg.type_of_value(array_contents);
108                assert!(
109                    !array_type.is_nested_vector(),
110                    "ICE: Nested vector used in ACIR generation"
111                );
112                let Type::Array(_, vector_length) = array_type else {
113                    unreachable!("Expected Array input for `as_vector` intrinsic");
114                };
115                let vector_length = self.acir_context.add_constant(vector_length.0);
116                let acir_value = self.convert_value(array_contents, dfg);
117                let result = self.read_array_with_type(acir_value, &array_type)?;
118                Ok(vec![
119                    AcirValue::Var(vector_length, NumericType::length_type()),
120                    AcirValue::Array(result),
121                ])
122            }
123
124            Intrinsic::VectorPushBack => self.convert_vector_push_back(arguments, dfg, result_ids),
125            Intrinsic::VectorPushFront => self.convert_vector_push_front(arguments, dfg),
126            Intrinsic::VectorPopBack => self.convert_vector_pop_back(arguments, dfg, result_ids),
127            Intrinsic::VectorPopFront => self.convert_vector_pop_front(arguments, dfg, result_ids),
128            Intrinsic::VectorInsert => self.convert_vector_insert(arguments, dfg, result_ids),
129            Intrinsic::VectorRemove => self.convert_vector_remove(arguments, dfg, result_ids),
130
131            Intrinsic::AsWitness => {
132                let arg = arguments[0];
133                let input = self.convert_value(arg, dfg).into_var()?;
134                Ok(self
135                    .acir_context
136                    .get_or_create_witness_var(input)
137                    .map(|val| self.convert_vars_to_values(vec![val], dfg, result_ids))?)
138            }
139            Intrinsic::DerivePedersenGenerators => Err(RuntimeError::AssertConstantFailed {
140                call_stack: self.acir_context.get_call_stack(),
141            }),
142            Intrinsic::ApplyRangeConstraint => {
143                unreachable!(
144                    "ICE: `Intrinsic::ApplyRangeConstraint` calls should be transformed into an `Instruction::RangeCheck`"
145                );
146            }
147            Intrinsic::FieldLessThan => {
148                unreachable!("FieldLessThan can only be called in unconstrained")
149            }
150            Intrinsic::IsUnconstrained
151            | Intrinsic::ArrayLen
152            | Intrinsic::ArrayAsStrUnchecked
153            | Intrinsic::StrAsBytes
154            | Intrinsic::StaticAssert
155            | Intrinsic::AssertConstant
156            | Intrinsic::ArrayRefCount
157            | Intrinsic::VectorRefCount => {
158                unreachable!("Expected {intrinsic} to have been removing during SSA optimizations")
159            }
160        }
161    }
162}
