34pub(super) fn codegen_acir(
35    ssa: Ssa,
36    brillig: &Brillig,
37    brillig_stdlib: BrilligStdLib<FieldElement>,
38    brillig_options: &BrilligOptions,
39) -> Result<Artifacts, RuntimeError> {
40    let mut acirs = Vec::new();
41
42    let used_globals = ssa.used_globals_in_functions();
43
44    // TODO(https://github.com/noir-lang/noir/issues/10269): can we parallelize this?
45    let mut shared_context = SharedContext::new(brillig_stdlib.clone(), used_globals);
46
47    for function in ssa.functions.values() {
48        let context =
49            Context::new(&mut shared_context, brillig, brillig_stdlib.clone(), brillig_options);
50
51        if let Some(mut generated_acir) = context.convert_ssa_function(&ssa, function)? {
52            // We want to be able to insert Brillig stdlib functions anywhere during the ACIR generation process (e.g. such as on the `GeneratedAcir`).
53            // As we don't want a reference to the `SharedContext` on the generated ACIR itself,
54            // we instead store the opcode location at which a Brillig call to a std lib function occurred.
55            // We then defer resolving the function IDs of those Brillig functions to when we have generated Brillig
56            // for all normal Brillig calls.
57            for (opcode_location, brillig_stdlib_func) in
58                &generated_acir.brillig_stdlib_func_locations
59            {
60                shared_context.generate_brillig_calls_to_resolve(
61                    brillig_stdlib_func,
62                    function.id(),
63                    *opcode_location,
64                );
65            }
66
67            // Fetch the Brillig stdlib calls to resolve for this function
68            if let Some(calls_to_resolve) = shared_context.remove_call_to_resolve(function.id()) {
69                // Resolve the Brillig stdlib calls
70                // We have to do a separate loop as the generated ACIR cannot be borrowed as mutable after an immutable borrow
71                for (opcode_location, brillig_function_pointer) in calls_to_resolve {
72                    generated_acir
73                        .resolve_brillig_stdlib_call(opcode_location, brillig_function_pointer);
74                }
75            }
76
77            generated_acir.name = function.name().to_owned();
78            acirs.push(generated_acir);
79        }
80    }
81
82    let generated_brillig = shared_context.finish();
83    let brillig_bytecode = generated_brillig
84        .into_iter()
85        .map(|brillig| BrilligBytecode { function_name: brillig.name, bytecode: brillig.byte_code })
86        .collect();
87
88    Ok((acirs, brillig_bytecode, ssa.error_selector_to_type))
89}
