56/// Context struct for the acir generation pass.
57/// May be similar to the Evaluator struct in the current SSA IR.
58struct Context<'a> {
59    /// Maps SSA values to `AcirVar`'s.
60    ///
61    /// This is needed so that we only create a single
62    /// AcirVar per SSA value. Before creating an `AcirVar`
63    /// for an SSA value, we check this map. If an `AcirVar`
64    /// already exists for this Value, we return the `AcirVar`.
65    ssa_values: HashMap<Id<Value>, AcirValue>,
66
67    /// The `AcirVar` that describes the condition belonging to the most recently invoked
68    /// `SideEffectsEnabled` instruction.
69    current_side_effects_enabled_var: AcirVar,
70
71    /// Manages and builds the `AcirVar`s to which the converted SSA values refer.
72    acir_context: AcirContext<FieldElement>,
73
74    /// Track initialized acir dynamic arrays
75    ///
76    /// An acir array must start with a MemoryInit ACIR opcodes
77    /// and then have MemoryOp opcodes
78    /// This set is used to ensure that a MemoryOp opcode is only pushed to the circuit
79    /// if there is already a MemoryInit opcode.
80    initialized_arrays: HashSet<BlockId>,
81
82    /// Maps SSA values to BlockId's
83    /// A BlockId is an ACIR structure which identifies a memory block
84    /// Each acir memory block corresponds to a different SSA array.
85    memory_blocks: HashMap<Id<Value>, BlockId>,
86
87    /// The BlockId dedicated to return_data
88    /// It is not managed by memory_blocks to ensure getting always a fresh block for return_data, even if
89    /// the SSA array has already been initialized to a block.
90    return_data_block_id: Option<BlockId>,
91
92    /// Maps SSA values to BlockId's used internally for computing the accurate flattened
93    /// index of non-homogenous arrays.
94    /// See [arrays] for more information about the purpose of the type sizes array.
95    ///
96    /// A BlockId is an ACIR structure which identifies a memory block
97    /// Each memory blocks corresponds to a different SSA value
98    /// which utilizes this internal memory for ACIR generation.
99    element_type_sizes_blocks: HashMap<Id<Value>, BlockId>,
100
101    /// Maps type sizes to BlockId. This is used to reuse the same BlockId if different
102    /// non-homogenous arrays end up having the same type sizes layout.
103    type_sizes_to_blocks: HashMap<Vec<u32>, BlockId>,
104
105    /// Number of the next BlockId, it is used to construct
106    /// a new BlockId
107    max_block_id: u32,
108
109    data_bus: DataBus,
110
111    /// Contains state that is generated and also used across ACIR functions
112    shared_context: &'a mut SharedContext<FieldElement>,
113
114    brillig: &'a Brillig,
115
116    /// Options affecting Brillig code generation.
117    brillig_options: &'a BrilligOptions,
118}
119
120impl<'a> Context<'a> {
121    fn new(
122        shared_context: &'a mut SharedContext<FieldElement>,
123        brillig: &'a Brillig,
124        brillig_stdlib: BrilligStdLib<FieldElement>,
125        brillig_options: &'a BrilligOptions,
126    ) -> Context<'a> {
127        let mut acir_context = AcirContext::new(brillig_stdlib);
128        let current_side_effects_enabled_var = acir_context.add_constant(FieldElement::one());
129
130        Context {
131            ssa_values: HashMap::default(),
132            current_side_effects_enabled_var,
133            acir_context,
134            initialized_arrays: HashSet::default(),
135            memory_blocks: HashMap::default(),
136            return_data_block_id: None,
137            element_type_sizes_blocks: HashMap::default(),
138            type_sizes_to_blocks: HashMap::default(),
139            max_block_id: 0,
140            data_bus: DataBus::default(),
141            shared_context,
142            brillig,
143            brillig_options,
144        }
145    }

147    fn convert_ssa_function(
148        mut self,
149        ssa: &Ssa,
150        function: &Function,
151    ) -> Result<Option<GeneratedAcir<FieldElement>>, RuntimeError> {
152        self.acir_context.set_call_stack_helper(self.brillig.call_stacks().clone());
153        match function.runtime() {
154            RuntimeType::Acir(inline_type) => {
155                match inline_type {
156                    InlineType::Inline | InlineType::InlineAlways => {
157                        if function.id() != ssa.main_id {
158                            panic!(
159                                "ACIR function should have been inlined earlier if not marked otherwise"
160                            );
161                        }
162                    }
163                    InlineType::NoPredicates => {
164                        panic!(
165                            "All ACIR functions marked with #[no_predicates] should be inlined before ACIR gen. This is an SSA exclusive codegen attribute"
166                        );
167                    }
168                    InlineType::InlineNever => {
169                        panic!(
170                            "ACIR function marked with #[inline_never]. This attribute is only allowed on unconstrained functions"
171                        );
172                    }
173                    InlineType::Fold => {}
174                }
175                // We only want to convert entry point functions. This being `main` and those marked with `InlineType::Fold`
176                Ok(Some(self.convert_acir_main(function, ssa)?))
177            }
178            RuntimeType::Brillig(_) => {
179                if function.id() == ssa.main_id {
180                    Ok(Some(self.convert_brillig_main(function)?))
181                } else {
182                    Ok(None)
183                }
184            }
185        }
186    }

188    fn convert_acir_main(
189        mut self,
190        main_func: &Function,
191        ssa: &Ssa,
192    ) -> Result<GeneratedAcir<FieldElement>, RuntimeError> {
193        let dfg = &main_func.dfg;
194        let entry_block = &dfg[main_func.entry_block()];
195        self.acir_context.acir_ir.input_witnesses =
196            self.convert_ssa_block_params(entry_block.parameters(), dfg)?;
197
198        let num_return_witnesses = dfg.get_num_return_witnesses(main_func)?;
199
200        // Create a witness for each return witness we have to guarantee that the return witnesses match the standard
201        // layout for serializing those types as if they were being passed as inputs.
202        //
203        // This is required for recursion as otherwise in situations where we cannot make use of the program's ABI
204        // (e.g. for `std::verify_proof` or the solidity verifier), we need extra knowledge about the program we're
205        // working with rather than following the standard ABI encoding rules.
206        //
207        // We allocate these witnesses now before performing ACIR gen for the rest of the program as the location of
208        // the function's return values can then be determined through knowledge of its ABI alone.
209        let return_witness_vars =
210            vecmap(0..num_return_witnesses, |_| self.acir_context.add_variable());
211
212        let return_witnesses = vecmap(&return_witness_vars, |return_var| {
213            let expr = self.acir_context.var_to_expression(*return_var).unwrap();
214            expr.to_witness().expect("return vars should be witnesses")
215        });
216
217        let mut warnings = Vec::new();
218
219        let used_globals = self.shared_context.get_and_remove_used_globals_set(main_func.id());
220
221        let globals_dfg = (*main_func.dfg.globals).clone();
222        let globals_dfg = DataFlowGraph::from(globals_dfg);
223        for (id, value) in globals_dfg.values_iter() {
224            if !used_globals.contains(&id) {
225                continue;
226            }
227            match value {
228                Value::NumericConstant { .. } => {
229                    self.convert_value(id, dfg);
230                }
231                Value::Instruction { instruction, .. } => {
232                    warnings.extend(self.convert_ssa_instruction(
233                        *instruction,
234                        &globals_dfg,
235                        ssa,
236                    )?);
237                }
238                _ => {
239                    panic!(
240                        "Expected either an instruction or a numeric constant for a global value"
241                    )
242                }
243            }
244        }
245
246        self.data_bus = dfg.data_bus.to_owned();
247        for instruction_id in entry_block.instructions() {
248            warnings.extend(self.convert_ssa_instruction(*instruction_id, dfg, ssa)?);
249        }
250        let (return_vars, return_warnings) =
251            self.convert_ssa_return(entry_block.unwrap_terminator(), dfg)?;
252
253        // This is a naive method of assigning the return values to their witnesses as
254        // we're likely to get a number of constraints which are asserting one witness to be equal to another.
255        //
256        // But an attempt at searching through the program and relabeling these witnesses so we could remove
257        // this constraint was [closed](https://github.com/noir-lang/noir/pull/10112#event-20171150226)
258        // but "the opcode count doesn't even change in real circuits."
259        for (witness_var, return_var) in return_witness_vars.iter().zip(return_vars) {
260            self.acir_context.assert_eq_var(*witness_var, return_var, None)?;
261        }
262
263        self.initialize_databus(&return_witnesses, dfg)?;
264        warnings.extend(return_warnings);
265        warnings.extend(self.acir_context.warnings.clone());
266
267        #[cfg(debug_assertions)]
268        acir_post_check(&self, &self.acir_context.acir_ir);
269
270        // Add the warnings from the alter Ssa passes
271        Ok(self.acir_context.finish(
272            // Don't embed databus return witnesses into the circuit.
273            if self.data_bus.return_data.is_some() { Vec::new() } else { return_witnesses },
274            warnings,
275        ))
276    }

278    fn convert_brillig_main(
279        mut self,
280        main_func: &Function,
281    ) -> Result<GeneratedAcir<FieldElement>, RuntimeError> {
282        let dfg = &main_func.dfg;
283
284        let inputs = try_vecmap(dfg[main_func.entry_block()].parameters(), |param_id| {
285            let typ = dfg.type_of_value(*param_id);
286            self.create_value_from_type(&typ, &mut |this, _| Ok(this.acir_context.add_variable()))
287        })?;
288        let arguments = self.gen_brillig_parameters(dfg[main_func.entry_block()].parameters(), dfg);
289
290        self.acir_context.acir_ir.input_witnesses = self.acir_context.extract_witnesses(&inputs);
291        let returns = main_func.returns().unwrap_or_default();
292
293        // Check the flattened size of return values to avoid OOM during Brillig entry point generation
294        let num_return_values: usize = returns
295            .iter()
296            .map(|result_id| dfg.type_of_value(*result_id).flattened_size().to_usize())
297            .sum();
298        if num_return_values > MAX_ELEMENTS {
299            let entry_block = &dfg[main_func.entry_block()];
300            let call_stack_id = match entry_block.unwrap_terminator() {
301                TerminatorInstruction::Return { call_stack, .. } => *call_stack,
302                _ => unreachable!("ICE: expected return terminator"),
303            };
304            let call_stack = dfg.call_stack_data.get_call_stack(call_stack_id);
305            return Err(RuntimeError::ReturnLimitExceeded {
306                num_witnesses: num_return_values,
307                max_witnesses: MAX_ELEMENTS,
308                call_stack,
309            });
310        }
311
312        let outputs: Vec<AcirType> =
313            vecmap(returns, |result_id| dfg.type_of_value(*result_id).into());
314
315        let code =
316            gen_brillig_for(main_func, arguments.clone(), self.brillig, self.brillig_options)?;
317
318        // We specifically do not attempt execution of the brillig code being generated as this can result in it being
319        // replaced with constraints on witnesses to the program outputs.
320        let unsafe_return_values = true;
321        let output_values = self.acir_context.brillig_call(
322            self.current_side_effects_enabled_var,
323            &code,
324            inputs,
325            outputs,
326            unsafe_return_values,
327            // We are guaranteed to have a Brillig function pointer of `0` as main itself is marked as unconstrained
328            BrilligFunctionId(0),
329            None,
330        )?;
331        self.shared_context.insert_generated_brillig(
332            main_func.id(),
333            arguments,
334            BrilligFunctionId(0),
335            code,
336        );
337
338        let return_witnesses: Vec<Witness> = output_values
339            .iter()
340            .flat_map(|value| value.clone().flatten())
341            .map(|(value, _)| self.acir_context.var_to_witness(value))
342            .collect::<Result<_, _>>()?;
343
344        let generated_acir = self.acir_context.finish(return_witnesses, Vec::new());
345
346        assert_eq!(
347            generated_acir.opcodes().len(),
348            1,
349            "Unconstrained programs should only generate a single opcode but multiple were emitted"
350        );
351
352        Ok(generated_acir)
353    }

355    /// Adds and binds `AcirVar`s for each numeric block parameter or block parameter array element.
356    fn convert_ssa_block_params(
357        &mut self,
358        params: &[ValueId],
359        dfg: &DataFlowGraph,
360    ) -> Result<Vec<Witness>, RuntimeError> {
361        // The first witness (if any) is the next one
362        let start_witness = self.acir_context.current_witness_index().0;
363        for &param_id in params {
364            let typ = dfg.type_of_value(param_id);
365            let value = self.convert_ssa_block_param(&typ)?;
366            match &value {
367                AcirValue::Var(_, _) => (),
368                AcirValue::Array(_) => {
369                    let block_id = self.block_id(param_id);
370                    let len = if matches!(typ, Type::Array(_, _)) {
371                        typ.flattened_size()
372                    } else {
373                        return Err(InternalError::Unexpected {
374                            expected: "Block params should be an array".to_owned(),
375                            found: format!("Instead got {typ:?}"),
376                            call_stack: self.acir_context.get_call_stack(),
377                        }
378                        .into());
379                    };
380                    self.initialize_array(block_id, len, Some(value.clone()))?;
381                }
382                AcirValue::DynamicArray(_) => unreachable!(
383                    "The dynamic array type is created in Acir gen and therefore cannot be a block parameter"
384                ),
385            }
386            self.ssa_values.insert(param_id, value);
387        }
388        let end_witness = self.acir_context.current_witness_index().0;
389        let witnesses = (start_witness..=end_witness).map(Witness::from).collect();
390        Ok(witnesses)
391    }

393    fn convert_ssa_block_param(&mut self, param_type: &Type) -> Result<AcirValue, RuntimeError> {
394        self.create_value_from_type(param_type, &mut |this, typ| this.add_numeric_input_var(&typ))
395    }

397    fn create_value_from_type(
398        &mut self,
399        param_type: &Type,
400        make_var: &mut impl FnMut(&mut Self, NumericType) -> Result<AcirVar, RuntimeError>,
401    ) -> Result<AcirValue, RuntimeError> {
402        match param_type {
403            Type::Numeric(numeric_type) => {
404                Ok(AcirValue::Var(make_var(self, *numeric_type)?, *numeric_type))
405            }
406            Type::Array(element_types, length) => {
407                let mut elements = im::Vector::new();
408
409                for _ in 0..length.0 {
410                    for element in element_types.iter() {
411                        elements.push_back(self.create_value_from_type(element, make_var)?);
412                    }
413                }
414
415                Ok(AcirValue::Array(elements))
416            }
417            _ => unreachable!("ICE: Params to the program should only contains numbers and arrays"),
418        }
419    }

421    /// Creates an `AcirVar` corresponding to a parameter witness to appears in the abi. A range
422    /// constraint is added if the numeric type requires it.
423    ///
424    /// This function is used not only for adding numeric block parameters, but also for adding
425    /// any array elements that belong to reference type block parameters.
426    fn add_numeric_input_var(
427        &mut self,
428        numeric_type: &NumericType,
429    ) -> Result<AcirVar, RuntimeError> {
430        let acir_var = self.acir_context.add_variable();
431
432        if !numeric_type.is_field() {
433            let one = self.acir_context.add_constant(FieldElement::one());
434            // The predicate is one so that this constraint is is always applied to Signed/Unsigned NumericTypes
435
436            self.acir_context.range_constrain_var(
437                acir_var,
438                numeric_type.bit_size::<FieldElement>(),
439                None,
440                one,
441            )?;
442        }
443        Ok(acir_var)
444    }

446    /// Converts an SSA instruction into its ACIR representation
447    fn convert_ssa_instruction(
448        &mut self,
449        instruction_id: InstructionId,
450        dfg: &DataFlowGraph,
451        ssa: &Ssa,
452    ) -> Result<Vec<SsaReport>, RuntimeError> {
453        let instruction = &dfg[instruction_id];
454        self.acir_context.set_call_stack(dfg.get_instruction_call_stack(instruction_id));
455        let mut warnings = Vec::new();
456
457        match instruction {
458            Instruction::Binary(binary) => {
459                // Disable the side effects if the binary instruction does not require them
460                let predicate = if instruction.requires_acir_gen_predicate(dfg) {
461                    self.current_side_effects_enabled_var
462                } else {
463                    self.acir_context.add_constant(FieldElement::one())
464                };
465                let result_acir_var = self.convert_ssa_binary(binary, dfg, predicate)?;
466                self.define_result_var(dfg, instruction_id, result_acir_var);
467            }
468            Instruction::Constrain(lhs, rhs, assert_message) => {
469                let lhs = self.convert_numeric_value(*lhs, dfg)?;
470                let rhs = self.convert_numeric_value(*rhs, dfg)?;
471                let assert_payload = self.convert_constrain_error(dfg, assert_message)?;
472                self.acir_context.assert_eq_var(lhs, rhs, assert_payload)?;
473            }
474            Instruction::ConstrainNotEqual(lhs, rhs, assert_message) => {
475                let lhs = self.convert_numeric_value(*lhs, dfg)?;
476                let rhs = self.convert_numeric_value(*rhs, dfg)?;
477                let assert_payload = self.convert_constrain_error(dfg, assert_message)?;
478                let predicate = self.current_side_effects_enabled_var;
479                self.acir_context.assert_neq_var(lhs, rhs, predicate, assert_payload)?;
480            }
481            Instruction::Cast(value_id, _) => {
482                let acir_var = self.convert_numeric_value(*value_id, dfg)?;
483                self.define_result_var(dfg, instruction_id, acir_var);
484            }
485            Instruction::Call { .. } => {
486                let result_ids = dfg.instruction_results(instruction_id);
487                warnings.extend(self.convert_ssa_call(instruction, dfg, ssa, result_ids)?);
488            }
489            Instruction::Not(value_id) => {
490                let (acir_var, typ) = match self.convert_value(*value_id, dfg) {
491                    AcirValue::Var(acir_var, typ) => (acir_var, typ),
492                    _ => unreachable!("NOT is only applied to numerics"),
493                };
494                let result_acir_var = self.acir_context.not_var(acir_var, typ)?;
495                self.define_result_var(dfg, instruction_id, result_acir_var);
496            }
497            Instruction::Truncate { value, bit_size, max_bit_size } => {
498                let result_acir_var =
499                    self.convert_ssa_truncate(*value, *bit_size, *max_bit_size, dfg)?;
500                self.define_result_var(dfg, instruction_id, result_acir_var);
501            }
502            Instruction::EnableSideEffectsIf { condition } => {
503                let acir_var = self.convert_numeric_value(*condition, dfg)?;
504                self.current_side_effects_enabled_var = acir_var;
505            }
506            Instruction::ArrayGet { .. } | Instruction::ArraySet { .. } => {
507                self.handle_array_operation(instruction_id, dfg)?;
508            }
509            Instruction::Allocate => {
510                return Err(RuntimeError::UnknownReference {
511                    call_stack: self.acir_context.get_call_stack().clone(),
512                });
513            }
514            Instruction::Store { .. } => {
515                unreachable!("Expected all store instructions to be removed before acir_gen")
516            }
517            Instruction::Load { .. } => {
518                unreachable!("Expected all load instructions to be removed before acir_gen")
519            }
520            Instruction::IncrementRc { .. } | Instruction::DecrementRc { .. } => {
521                // Only Brillig needs to worry about reference counted arrays
522                unreachable!("Expected all Rc instructions to be removed before acir_gen")
523            }
524            Instruction::RangeCheck { value, max_bit_size, assert_message } => {
525                let acir_var = self.convert_numeric_value(*value, dfg)?;
526                let one = self.acir_context.add_constant(FieldElement::one());
527                // Predicate is one because the predicate has already been
528                // handled in the RangeCheck instruction during the flattening pass.
529                self.acir_context.range_constrain_var(
530                    acir_var,
531                    *max_bit_size,
532                    assert_message.clone(),
533                    one,
534                )?;
535            }
536            Instruction::IfElse { .. } => {
537                unreachable!("IfElse instruction remaining in acir-gen")
538            }
539            Instruction::MakeArray { elements, typ: _ } => {
540                let elements = elements.iter().map(|element| self.convert_value(*element, dfg));
541                let value = AcirValue::Array(elements.collect());
542                let [result] = dfg.instruction_result(instruction_id);
543                self.ssa_values.insert(result, value);
544            }
545            Instruction::Noop => (),
546        }
547
548        self.acir_context.set_call_stack(CallStack::new());
549        Ok(warnings)
550    }

552    /// Converts an optional constrain error message into an ACIR assertion payload
553    fn convert_constrain_error(
554        &mut self,
555        dfg: &DataFlowGraph,
556        assert_message: &Option<ConstrainError>,
557    ) -> Result<Option<AssertionPayload<FieldElement>>, RuntimeError> {
558        let Some(error) = assert_message else {
559            return Ok(None);
560        };
561
562        let assert_payload = match error {
563            ConstrainError::StaticString(string) => {
564                self.acir_context.generate_assertion_message_payload(string.clone())
565            }
566            ConstrainError::Dynamic(error_selector, is_string_type, values) => {
567                if let Some(constant_string) =
568                    try_to_extract_string_from_error_payload(*is_string_type, values, dfg)
569                {
570                    self.acir_context.generate_assertion_message_payload(constant_string)
571                } else {
572                    let acir_values: Vec<_> =
573                        vecmap(values, |value| self.convert_value(*value, dfg));
574
575                    let expressions_or_memory =
576                        self.acir_context.values_to_expressions_or_memory(&acir_values)?;
577
578                    let error_selector = error_selector.as_u64();
579                    AssertionPayload { error_selector, payload: expressions_or_memory }
580                }
581            }
582        };
583        Ok(Some(assert_payload))
584    }

586    /// Remember the result of an instruction returning a single value
587    fn define_result(
588        &mut self,
589        dfg: &DataFlowGraph,
590        instruction: InstructionId,
591        result: AcirValue,
592    ) {
593        let [result_id] = dfg.instruction_result(instruction);
594        self.ssa_values.insert(result_id, result);
595    }

609    /// Converts an SSA terminator's return values into their ACIR representations
610    fn convert_ssa_return(
611        &mut self,
612        terminator: &TerminatorInstruction,
613        dfg: &DataFlowGraph,
614    ) -> Result<(Vec<AcirVar>, Vec<SsaReport>), RuntimeError> {
615        let (return_values, call_stack) = match terminator {
616            TerminatorInstruction::Return { return_values, call_stack } => {
617                (return_values, *call_stack)
618            }
619            // TODO(https://github.com/noir-lang/noir/issues/4616): Enable recursion on foldable/non-inlined ACIR functions
620            TerminatorInstruction::JmpIf { .. } | TerminatorInstruction::Jmp { .. } => {
621                unreachable!("ICE: Program must have a singular return")
622            }
623            TerminatorInstruction::Unreachable { .. } => return Ok((vec![], vec![])),
624        };
625
626        let mut has_constant_return = false;
627        let mut return_vars: Vec<AcirVar> = Vec::new();
628        for value_id in return_values {
629            let value = self.convert_value(*value_id, dfg);
630
631            // `value` may or may not be an array reference. Calling `flatten` will expand the array if there is one.
632            let acir_vars = self.acir_context.flatten(value)?;
633            for (acir_var, _) in acir_vars {
634                has_constant_return |= self.acir_context.is_constant(&acir_var);
635                return_vars.push(acir_var);
636            }
637        }
638
639        let call_stack = dfg.call_stack_data.get_call_stack(call_stack);
640        let warnings = if has_constant_return {
641            vec![SsaReport::Warning(InternalWarning::ReturnConstant { call_stack })]
642        } else {
643            Vec::new()
644        };
645
646        Ok((return_vars, warnings))
647    }

649    /// Gets the cached `AcirVar` that was converted from the corresponding `ValueId`. If it does
650    /// not already exist in the cache, a conversion is attempted and cached for simple values
651    /// that require no further context such as numeric types - values requiring more context
652    /// should have already been cached elsewhere.
653    ///
654    /// Conversion is assumed to have already been performed for instruction results and block
655    /// parameters. This is because block parameters are converted before anything else, and
656    /// because instructions results are converted when the corresponding instruction is
657    /// encountered. (An instruction result cannot be referenced before the instruction occurs.)
658    ///
659    /// It is not safe to call this function on value ids that represent pointers. Instructions
660    /// involving such values are evaluated via a separate path and stored in
661    /// `ssa_value_to_array_address` instead.
662    fn convert_value(&mut self, value_id: ValueId, dfg: &DataFlowGraph) -> AcirValue {
663        assert!(
664            !matches!(dfg.type_of_value(value_id), Type::Reference(_)),
665            "convert_value: did not expect a Reference type"
666        );
667
668        let value = &dfg[value_id];
669        if let Some(acir_value) = self.ssa_values.get(&value_id) {
670            return acir_value.clone();
671        }
672
673        let acir_value = match value {
674            Value::NumericConstant { constant, typ } => {
675                AcirValue::Var(self.acir_context.add_constant(*constant), *typ)
676            }
677            Value::Intrinsic(..) => {
678                unreachable!("ICE: Intrinsics should be resolved via separate logic")
679            }
680            Value::Function(function_id) => {
681                // This conversion is for debugging support only, to allow the
682                // debugging instrumentation code to work. Taking the reference
683                // of a function in ACIR is useless.
684                let id = self.acir_context.add_constant(function_id.to_u32());
685                AcirValue::Var(id, NumericType::NativeField)
686            }
687            Value::ForeignFunction(_) => unimplemented!(
688                "Oracle calls directly in constrained functions are not yet available."
689            ),
690            Value::Instruction { .. } | Value::Param { .. } => {
691                unreachable!("ICE: Should have been in cache {value_id} {value:?}")
692            }
693            Value::Global(_) => {
694                unreachable!("ICE: All globals should have been inlined");
695            }
696        };
697        self.ssa_values.insert(value_id, acir_value.clone());
698        acir_value
699    }

701    fn convert_numeric_value(
702        &mut self,
703        value_id: ValueId,
704        dfg: &DataFlowGraph,
705    ) -> Result<AcirVar, InternalError> {
706        match self.convert_value(value_id, dfg) {
707            AcirValue::Var(acir_var, _) => Ok(acir_var),
708            AcirValue::Array(array) => Err(InternalError::Unexpected {
709                expected: "a numeric value".to_string(),
710                found: format!("{array:?}"),
711                call_stack: self.acir_context.get_call_stack(),
712            }),
713            AcirValue::DynamicArray(_) => Err(InternalError::Unexpected {
714                expected: "a numeric value".to_string(),
715                found: "an array".to_string(),
716                call_stack: self.acir_context.get_call_stack(),
717            }),
718        }
719    }

721    /// Processes a binary operation and converts the result into an `AcirVar`
722    fn convert_ssa_binary(
723        &mut self,
724        binary: &Binary,
725        dfg: &DataFlowGraph,
726        predicate: AcirVar,
727    ) -> Result<AcirVar, RuntimeError> {
728        let lhs = self.convert_numeric_value(binary.lhs, dfg)?;
729        let rhs = self.convert_numeric_value(binary.rhs, dfg)?;
730        let num_type = self.type_of_binary_operation(binary, dfg).unwrap_numeric();
731
732        if num_type.is_signed()
733            && matches!(
734                binary.operator,
735                BinaryOp::Add { unchecked: false }
736                    | BinaryOp::Sub { unchecked: false }
737                    | BinaryOp::Mul { unchecked: false }
738            )
739        {
740            panic!("Checked signed operations should all be removed before ACIRgen")
741        }
742
743        let result = match binary.operator {
744            BinaryOp::Add { .. } => self.acir_context.add_var(lhs, rhs),
745            BinaryOp::Sub { .. } => self.acir_context.sub_var(lhs, rhs),
746            BinaryOp::Mul { .. } => self.acir_context.mul_var(lhs, rhs),
747            BinaryOp::Div => self.acir_context.div_var(lhs, rhs, num_type, predicate),
748            // Note: that this produces unnecessary constraints when
749            // this Eq instruction is being used for a constrain statement
750            BinaryOp::Eq => self.acir_context.eq_var(lhs, rhs),
751            BinaryOp::Lt => match num_type {
752                NumericType::Unsigned { bit_size } => {
753                    self.acir_context.less_than_var(lhs, rhs, bit_size)
754                }
755                _ => {
756                    panic!("ICE: unexpected binary type for Lt operation: {num_type:?}")
757                }
758            },
759            BinaryOp::Xor => self.acir_context.xor_var(lhs, rhs, num_type),
760            BinaryOp::And => self.acir_context.and_var(lhs, rhs, num_type),
761            BinaryOp::Or => self.acir_context.or_var(lhs, rhs, num_type),
762            BinaryOp::Mod => match num_type {
763                NumericType::Unsigned { bit_size } => {
764                    self.acir_context.modulo_var(lhs, rhs, bit_size, predicate)
765                }
766                _ => {
767                    panic!("ICE: unexpected binary type for Mod operation: {num_type:?}")
768                }
769            },
770            BinaryOp::Shl | BinaryOp::Shr => unreachable!(
771                "ICE - bit shift operators do not exist in ACIR and should have been replaced"
772            ),
773        }?;
774
775        if let NumericType::Unsigned { bit_size } = &num_type {
776            // Check for integer overflow
777            self.check_unsigned_overflow(result, *bit_size, binary, predicate)
778        } else {
779            Ok(result)
780        }
781    }
782
783    /// Adds a range check against the bit size of the result of addition, subtraction or multiplication
784    fn check_unsigned_overflow(
785        &mut self,
786        result: AcirVar,
787        bit_size: u32,
788        binary: &Binary,
789        predicate: AcirVar,
790    ) -> Result<AcirVar, RuntimeError> {
791        let msg = match binary.operator {
792            BinaryOp::Add { unchecked: false } => "attempt to add with overflow",
793            BinaryOp::Sub { unchecked: false } => "attempt to subtract with overflow",
794            BinaryOp::Mul { unchecked: false } => "attempt to multiply with overflow",
795            _ => return Ok(result),
796        };
797
798        self.acir_context.range_constrain_var(result, bit_size, Some(msg.to_string()), predicate)
799    }

801    /// Operands in a binary operation are checked to have the same type.
802    ///
803    /// In Noir, binary operands should have the same type due to the language
804    /// semantics.
805    fn type_of_binary_operation(&self, binary: &Binary, dfg: &DataFlowGraph) -> Type {
806        let lhs_type = dfg.type_of_value(binary.lhs);
807        let rhs_type = dfg.type_of_value(binary.rhs);
808
809        match (lhs_type, rhs_type) {
810            // Function type should not be possible, since all functions
811            // have been inlined.
812            (_, Type::Function) | (Type::Function, _) => {
813                unreachable!("all functions should be inlined")
814            }
815            (_, Type::Reference(_)) | (Type::Reference(_), _) => {
816                unreachable!("References are invalid in binary operations")
817            }
818            (_, Type::Array(..)) | (Type::Array(..), _) => {
819                unreachable!("Arrays are invalid in binary operations")
820            }
821            (_, Type::Vector(..)) | (Type::Vector(..), _) => {
822                unreachable!("Arrays are invalid in binary operations")
823            }
824            // If either side is a numeric type, then we expect their types to be
825            // the same.
826            (Type::Numeric(lhs_type), Type::Numeric(rhs_type)) => {
827                assert_eq!(lhs_type, rhs_type, "lhs and rhs types in {binary:?} are not the same");
828                Type::Numeric(lhs_type)
829            }
830        }
831    }

833    /// Returns an `AcirVar`that is constrained to be result of the truncation.
834    fn convert_ssa_truncate(
835        &mut self,
836        value_id: ValueId,
837        bit_size: u32,
838        mut max_bit_size: u32,
839        dfg: &DataFlowGraph,
840    ) -> Result<AcirVar, RuntimeError> {
841        assert_ne!(bit_size, max_bit_size, "Attempted to generate a noop truncation");
842        assert!(
843            bit_size < max_bit_size,
844            "Attempted to generate a truncation into size larger than max input"
845        );
846
847        let mut var = self.convert_numeric_value(value_id, dfg)?;
848        match &dfg[value_id] {
849            Value::Instruction { instruction, .. } => {
850                if matches!(
851                    &dfg[*instruction],
852                    Instruction::Binary(Binary { operator: BinaryOp::Sub { .. }, .. })
853                ) {
854                    // Subtractions must first have the integer modulus added before truncation can be
855                    // applied. This is done in order to prevent underflow.
856                    //
857                    // FieldElements have max bit size equals to max_num_bits so
858                    // we filter out this bit size because there is no underflow
859                    // for FieldElements. Furthermore, adding a power of two
860                    // would be incorrect for a FieldElement (cf. #8519).
861                    if max_bit_size < FieldElement::max_num_bits() {
862                        // When max_bit_size is max_num_bits() - 1, adding
863                        // 2**max_bit_size to an element of max_bit_size bits
864                        // gives an element of max_num_bits() bits which may overflow
865                        assert!(
866                            max_bit_size != FieldElement::max_num_bits() - 1,
867                            "potential underflow in subtraction when max_bit_size is {max_bit_size}"
868                        );
869                        let integer_modulus = power_of_two::<FieldElement>(max_bit_size);
870                        let integer_modulus = self.acir_context.add_constant(integer_modulus);
871                        var = self.acir_context.add_var(var, integer_modulus)?;
872                        max_bit_size += 1;
873                    }
874                }
875            }
876            Value::Param { .. } => {
877                // Binary operations on params may have been entirely simplified if the operation
878                // results in the identity of the parameter
879            }
880            _ => unreachable!(
881                "ICE: Truncates are only ever applied to the result of a binary op or a param"
882            ),
883        }
884
885        self.acir_context.truncate_var(var, bit_size, max_bit_size)
886    }

888    /// Fetch a flat list of [AcirVar].
889    ///
890    /// Flattens an [AcirValue] into a vector of `AcirVar`.
891    ///
892    /// This is an extension of [AcirValue::flatten] that also supports [AcirValue::DynamicArray].
893    fn flatten(&mut self, value: &AcirValue) -> Result<Vec<AcirVar>, RuntimeError> {
894        Ok(match value {
895            AcirValue::Var(var, _) => vec![*var],
896            AcirValue::Array(array) => {
897                let mut result = Vec::new();
898                for elem in array {
899                    result.extend(self.flatten(elem)?);
900                }
901                result
902            }
903            AcirValue::DynamicArray(AcirDynamicArray { block_id, len, value_types, .. }) => {
904                let elements = self.read_dynamic_array(*block_id, *len, value_types);
905                let mut result = Vec::new();
906
907                for value in elements {
908                    match value? {
909                        AcirValue::Var(var, _typ) => result.push(var),
910                        _ => unreachable!("ICE: Dynamic memory should already be flat"),
911                    }
912                }
913                result
914            }
915        })
916    }

919/// Check post ACIR generation properties
920/// * No memory opcodes should be laid down that write to the internal type sizes array.
921///   See [arrays] for more information on the type sizes array.
922#[cfg(debug_assertions)]
923fn acir_post_check(context: &Context<'_>, acir: &GeneratedAcir<FieldElement>) {
924    use acvm::acir::circuit::Opcode;
925    for opcode in acir.opcodes() {
926        let Opcode::MemoryOp { block_id, op } = opcode else {
927            continue;
928        };
929        if op.operation.is_one() {
930            // Check that we have no writes to the type size arrays
931            let is_type_sizes_array =
932                context.element_type_sizes_blocks.values().any(|id| id == block_id);
933            assert!(
934                !is_type_sizes_array,
935                "ICE: Writes to the internal type sizes array are forbidden"
936            );
937        }
938    }
