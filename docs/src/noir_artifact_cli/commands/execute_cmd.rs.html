18/// Execute a binary program or a circuit artifact.
19#[derive(Debug, Clone, Args)]
20pub struct ExecuteCommand {
21    /// Path to the JSON build artifact (either a program or a contract).
22    #[clap(long, short, value_parser = parse_and_normalize_path)]
23    pub artifact_path: PathBuf,
24
25    /// Path to the Prover.toml file which contains the inputs and the
26    /// optional return value in ABI format.
27    #[clap(long, short, value_parser = parse_and_normalize_path)]
28    pub prover_file: PathBuf,
29
30    /// Path to the directory where the output witness should be saved.
31    /// If empty then the results are discarded.
32    #[clap(long, short, value_parser = parse_and_normalize_path)]
33    pub output_dir: Option<PathBuf>,
34
35    /// Write the execution witness to named file
36    ///
37    /// Defaults to the name of the circuit being executed.
38    #[clap(long, short)]
39    pub witness_name: Option<String>,
40
41    /// Name of the function to execute, if the artifact is a contract.
42    #[clap(long)]
43    pub contract_fn: Option<String>,
44
45    /// Path to the oracle transcript that is to be replayed during the
46    /// execution in response to foreign calls. The format is expected
47    /// to be JSON Lines, with each request/response on a separate line.
48    ///
49    /// Note that a transcript might be invalid if the inputs change and
50    /// the circuit takes a different path during execution.
51    #[clap(long, conflicts_with = "oracle_resolver")]
52    pub oracle_file: Option<PathBuf>,
53
54    /// JSON RPC url to solve oracle calls.
55    #[clap(long, conflicts_with = "oracle_file")]
56    pub oracle_resolver: Option<String>,
57
58    /// Root directory for the RPC oracle resolver.
59    #[clap(long, value_parser = parse_and_normalize_path)]
60    pub oracle_root_dir: Option<PathBuf>,
61
62    /// Package name for the RPC oracle resolver
63    #[clap(long)]
64    pub oracle_package_name: Option<String>,
65}

67pub fn run(args: ExecuteCommand) -> Result<(), CliError> {
68    let artifact = Artifact::read_from_file(&args.artifact_path)?;
69    let artifact_name = args.artifact_path.file_stem().and_then(|s| s.to_str()).unwrap_or_default();
70
71    let (circuit, circuit_name): (CompiledProgram, String) = match artifact {
72        Artifact::Program(program) => (program.into(), artifact_name.to_string()),
73        Artifact::Contract(contract) => {
74            let names = || contract.functions.iter().map(|f| f.name.clone()).collect::<Vec<_>>();
75
76            let Some(ref name) = args.contract_fn else {
77                return Err(CliError::MissingContractFn { names: names() });
78            };
79            let Some(program) = contract.function_as_compiled_program(name) else {
80                return Err(CliError::UnknownContractFn { name: name.clone(), names: names() });
81            };
82
83            (program, format!("{artifact_name}::{name}"))
84        }
85    };
86
87    match execute(&circuit, &args) {
88        Ok(results) => {
89            execution::save_and_check_witness(
90                &circuit,
91                results,
92                &circuit_name,
93                args.output_dir.as_deref(),
94                args.witness_name.as_deref(),
95            )?;
96        }
97        Err(e) => {
98            if let CliError::CircuitExecutionError(ref err) = e {
99                execution::show_diagnostic(&circuit, err);
100            }
101            // Still returning the error to facilitate command forwarding, to indicate that the command failed.
102            return Err(e);
103        }
104    }
105    Ok(())
106}

108/// Execute a circuit and return the output witnesses.
109fn execute(circuit: &CompiledProgram, args: &ExecuteCommand) -> Result<ExecutionResults, CliError> {
110    // Build a custom foreign call executor that replays the Oracle transcript,
111    // and use it as a base for the default executor. Using it as the innermost rather
112    // than top layer so that any extra `print` added for debugging is handled by the
113    // default, rather than trying to match it to the transcript.
114    let transcript_executor = match args.oracle_file {
115        Some(ref path) => layers::Either::Left(ReplayForeignCallExecutor::from_file(path)?),
116        None => layers::Either::Right(layers::Empty),
117    };
118
119    let mut foreign_call_executor = DefaultForeignCallBuilder {
120        output: std::io::stdout(),
121        enable_mocks: false,
122        resolver_url: args.oracle_resolver.clone(),
123        root_path: None,
124        package_name: None,
125    }
126    .build_with_base(transcript_executor);
127
128    let blackbox_solver = Bn254BlackBoxSolver;
129
130    execution::execute(circuit, &blackbox_solver, &mut foreign_call_executor, &args.prover_file)
131}
