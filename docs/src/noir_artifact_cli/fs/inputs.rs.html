9/// Returns the circuit's parameters and its return value, if one exists.
10///
11/// The file is is expected to contain ABI encoded inputs in TOML or JSON format.
12pub fn read_inputs_from_file(
13    file_path: &Path,
14    abi: &Abi,
15) -> Result<(InputMap, Option<InputValue>), CliError> {
16    use crate::errors::FilesystemError::{InvalidInputFile, MissingInputFile};
17    use CliError::FilesystemError;
18
19    let has_params = !abi.parameters.is_empty();
20    let has_return = abi.return_type.is_some();
21    let has_file = file_path.exists();
22
23    if !has_params && !has_return {
24        return Ok((BTreeMap::new(), None));
25    }
26    if !has_params && !has_file {
27        // Reading a return value from the `Prover.toml` is optional,
28        // so if the ABI has no parameters we can skip reading the file if it doesn't exist.
29        return Ok((BTreeMap::new(), None));
30    }
31    if has_params && !has_file {
32        return Err(FilesystemError(MissingInputFile(file_path.to_path_buf())));
33    }
34
35    let Some(ext) = file_path.extension().and_then(|e| e.to_str()) else {
36        return Err(FilesystemError(InvalidInputFile(
37            file_path.to_path_buf(),
38            "cannot determine input format".to_string(),
39        )));
40    };
41
42    let Some(format) = Format::from_ext(ext) else {
43        return Err(FilesystemError(InvalidInputFile(
44            file_path.to_path_buf(),
45            format!("unknown input format: {ext}"),
46        )));
47    };
48
49    let inputs = std::fs::read_to_string(file_path)
50        .map_err(|e| FilesystemError(InvalidInputFile(file_path.to_path_buf(), e.to_string())))?;
51
52    let mut inputs = format.parse(&inputs, abi)?;
53    let return_value = inputs.remove(MAIN_RETURN_NAME);
54
55    Ok((inputs, return_value))
56}

58/// Writes input map to a file
59pub fn write_inputs_to_file<P: AsRef<Path>>(
60    path: P,
61    file_name: &str,
62    format: Format,
63    abi: &Abi,
64    input_map: &InputMap,
65) -> Result<(), CliError> {
66    use crate::errors::FilesystemError::OutputFileCreationFailed;
67    if abi.is_empty() {
68        return Ok(());
69    }
70    use std::fs::File;
71    let file_path = path.as_ref().join(file_name).with_extension(format.ext());
72    let mut file =
73        File::create(&file_path).map_err(|e| OutputFileCreationFailed(file_path, e.to_string()))?;
74
75    let input_string = format.serialize(input_map, abi)?;
76    file.write_all(input_string.as_bytes()).expect("Failed to write to file");
77
78    Ok(())
79}
