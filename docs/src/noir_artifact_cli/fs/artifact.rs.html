12impl Artifact {
13    /// Try to parse an artifact as a binary program or a contract
14    pub fn read_from_file(path: &Path) -> Result<Self, CliError> {
15        let file = path.with_extension("json");
16        let json = std::fs::read(&file)
17            .map_err(|err| FilesystemError::FailedToReadFile(file.to_path_buf(), err))?;
18
19        let as_program = || serde_json::from_slice::<ProgramArtifact>(&json).map(Artifact::Program);
20        let as_contract =
21            || serde_json::from_slice::<ContractArtifact>(&json).map(Artifact::Contract);
22
23        as_program()
24            .or_else(|e| as_contract().map_err(|_| e))
25            .map_err(CliError::ArtifactDeserializationError)
26    }
27}
28
29/// Returns the circuit's bytecode read from the file at the given location
30pub fn read_bytecode_from_file(
31    work_dir: &Path,
32    file_name: &str,
33) -> Result<Vec<u8>, FilesystemError> {
34    let file_path = work_dir.join(file_name);
35    if !file_path.exists() {
36        return Err(FilesystemError::MissingBytecodeFile(file_path.clone()));
37    }
38    let bytecode: Vec<u8> = std::fs::read(&file_path)
39        .map_err(|e| FilesystemError::InvalidBytecodeFile(file_path, e.to_string()))?;
40    Ok(bytecode)
41}

43/// Read a `ProgramArtifact`. Returns error if it turns out to be a `ContractArtifact`.
44pub fn read_program_from_file(path: &Path) -> Result<ProgramArtifact, CliError> {
45    match Artifact::read_from_file(path)? {
46        Artifact::Program(program) => Ok(program),
47        Artifact::Contract(contract) => {
48            let msg = format!(
49                "expected a program artifact but found a contract in {}: {}",
50                path.display(),
51                contract.name
52            );
53            Err(CliError::ArtifactDeserializationError(serde_json::Error::custom(msg)))
54        }
55    }
56}

58pub fn save_program_to_file(
59    program_artifact: &ProgramArtifact,
60    crate_name: &CrateName,
61    output_dir: &Path,
62) -> Result<PathBuf, CliError> {
63    let circuit_name: String = crate_name.into();
64    save_build_artifact_to_file(program_artifact, &circuit_name, output_dir)
65        .map_err(|err| CliError::FailedToSaveProgram(circuit_name, Box::new(err)))
66}

77fn save_build_artifact_to_file<T: ?Sized + serde::Serialize>(
78    build_artifact: &T,
79    artifact_name: &str,
80    output_dir: &Path,
81) -> Result<PathBuf, CliError> {
82    let artifact_path = output_dir.join(artifact_name).with_extension("json");
83    let bytes = serde_json::to_vec(build_artifact)?;
84    write_to_file(&bytes, &artifact_path)?;
85    Ok(artifact_path)
86}

88/// Create the parent directory if needed and write the bytes to a file.
89pub fn write_to_file(bytes: &[u8], path: &Path) -> Result<(), FilesystemError> {
90    if let Some(dir) = path.parent() {
91        std::fs::create_dir_all(dir)
92            .map_err(|err| FilesystemError::FailedToCreateDirectory(dir.to_path_buf(), err))?;
93    }
94    std::fs::write(path, bytes)
95        .map_err(|err| FilesystemError::FailedToWriteFile(path.to_path_buf(), err))?;
96    Ok(())
97}
