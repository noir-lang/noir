22fn unsatisfied_constraint<F>(opcode_index: usize, message: String) -> OpcodeResolutionError<F> {
23    OpcodeResolutionError::UnsatisfiedConstrain {
24        opcode_location: ErrorLocation::Resolved(OpcodeLocation::Acir(opcode_index)),
25        payload: Some(ResolvedAssertionPayload::String(message)),
26    }
27}

29fn witness_value<F: AcirField>(
30    w: &Witness,
31    witness_map: &WitnessMap<F>,
32) -> Result<F, OpcodeResolutionError<F>> {
33    Ok(*witness_map.get(w).ok_or(OpcodeNotSolvable::MissingAssignment(w.witness_index()))?)
34}

36pub fn validate_witness<F: AcirField>(
37    backend: &impl BlackBoxFunctionSolver<F>,
38    witness_map: WitnessMap<F>,
39    circuit: &Circuit<F>,
40) -> Result<(), OpcodeResolutionError<F>> {
41    let mut block_solvers: HashMap<BlockId, MemoryOpSolver<F>> = HashMap::new();
42
43    for (opcode_index, opcode) in circuit.opcodes.iter().enumerate() {
44        match opcode {
45            Opcode::AssertZero(expression) => {
46                let result = &ExpressionSolver::evaluate(expression, &witness_map);
47                if !result.is_zero() {
48                    return Err(unsatisfied_constraint(
49                        opcode_index,
50                        format!("Invalid witness assignment: {expression}"),
51                    ));
52                }
53            }
54            Opcode::BlackBoxFuncCall(black_box_func_call) => {
55                match black_box_func_call {
56                    BlackBoxFuncCall::AES128Encrypt { inputs, iv, key, outputs } => {
57                        let ciphertext = blackbox::aes128::execute_aes128_encryption_opcode(
58                            &witness_map,
59                            inputs,
60                            iv,
61                            key,
62                        )?;
63                        assert_eq!(outputs.len(), ciphertext.len());
64                        for (output_witness, value) in outputs.iter().zip(ciphertext.into_iter()) {
65                            let witness_value = witness_value(output_witness, &witness_map)?;
66                            let output_value = F::from(u128::from(value));
67                            if witness_value != output_value {
68                                return Err(unsatisfied_constraint(
69                                    opcode_index,
70                                    format!(
71                                        "AES128 opcode violation: expected {output_value} but found {witness_value} for output witness {output_witness}",
72                                    ),
73                                ));
74                            }
75                        }
76                    }
77                    BlackBoxFuncCall::AND { lhs, rhs, num_bits, output } => {
78                        let lhs_value = input_to_value(&witness_map, *lhs)?;
79                        let rhs_value = input_to_value(&witness_map, *rhs)?;
80                        let and_result = bit_and(lhs_value, rhs_value, *num_bits);
81                        let output_value = witness_map
82                            .get(output)
83                            .ok_or(OpcodeNotSolvable::MissingAssignment(output.0))?;
84                        if and_result != *output_value {
85                            return Err(unsatisfied_constraint(
86                                opcode_index,
87                                format!(
88                                    "AND opcode violation: {lhs_value} AND {rhs_value} != {output_value} for {num_bits} bits"
89                                ),
90                            ));
91                        }
92                    }
93                    BlackBoxFuncCall::XOR { lhs, rhs, num_bits, output } => {
94                        let lhs_value = input_to_value(&witness_map, *lhs)?;
95                        let rhs_value = input_to_value(&witness_map, *rhs)?;
96                        let xor_result = bit_xor(lhs_value, rhs_value, *num_bits);
97                        let output_value = witness_map
98                            .get(output)
99                            .ok_or(OpcodeNotSolvable::MissingAssignment(output.0))?;
100                        if xor_result != *output_value {
101                            return Err(unsatisfied_constraint(
102                                opcode_index,
103                                format!(
104                                    "XOR opcode violation: {lhs_value} XOR {rhs_value} != {output_value} for {num_bits} bits"
105                                ),
106                            ));
107                        }
108                    }
109                    BlackBoxFuncCall::RANGE { input, num_bits } => {
110                        let value = input_to_value(&witness_map, *input)?;
111                        if value.num_bits() > *num_bits {
112                            return Err(unsatisfied_constraint(
113                                opcode_index,
114                                format!(
115                                    "RANGE opcode violation: value {value} does not fit in {num_bits} bits"
116                                ),
117                            ));
118                        }
119                    }
120                    BlackBoxFuncCall::Blake2s { inputs, outputs } => {
121                        let message_input = get_hash_input(&witness_map, inputs, None, 8)?;
122                        let digest: [u8; 32] = blake2s(&message_input)?;
123                        for i in 0..32 {
124                            let output_witness = &outputs[i];
125                            let witness_value = witness_map
126                                .get(output_witness)
127                                .ok_or(OpcodeNotSolvable::MissingAssignment(output_witness.0))?;
128                            if *witness_value != F::from_be_bytes_reduce(&[digest[i]]) {
129                                return Err(unsatisfied_constraint(
130                                    opcode_index,
131                                    format!(
132                                        "BLAKE2s opcode violation: expected {:?} but found {:?} for output witness {:?}",
133                                        F::from_be_bytes_reduce(&[digest[i]]),
134                                        witness_value,
135                                        output_witness
136                                    ),
137                                ));
138                            }
139                        }
140                    }
141                    BlackBoxFuncCall::Blake3 { inputs, outputs } => {
142                        let message_input = get_hash_input(&witness_map, inputs, None, 8)?;
143                        let digest: [u8; 32] = blake3(&message_input)?;
144                        for i in 0..32 {
145                            let output_witness = &outputs[i];
146                            let witness_value = witness_value(output_witness, &witness_map)?;
147                            if witness_value != F::from_be_bytes_reduce(&[digest[i]]) {
148                                return Err(unsatisfied_constraint(
149                                    opcode_index,
150                                    format!(
151                                        "BLAKE3 opcode violation: expected {:?} but found {:?} for output witness {:?}",
152                                        F::from_be_bytes_reduce(&[digest[i]]),
153                                        witness_value,
154                                        output_witness
155                                    ),
156                                ));
157                            }
158                        }
159                    }
160                    BlackBoxFuncCall::EcdsaSecp256k1 {
161                        public_key_x,
162                        public_key_y,
163                        signature,
164                        hashed_message,
165                        predicate,
166                        output,
167                    } => {
168                        let predicate_value = input_to_value(&witness_map, *predicate)?.is_one();
169                        if predicate_value {
170                            let is_valid = blackbox::signature::ecdsa::execute_ecdsa(
171                                &witness_map,
172                                public_key_x,
173                                public_key_y,
174                                signature,
175                                hashed_message,
176                                predicate,
177                                true,
178                            )?;
179                            let output_value = witness_value(output, &witness_map)?;
180                            if output_value != F::from(is_valid) {
181                                return Err(unsatisfied_constraint(
182                                    opcode_index,
183                                    format!(
184                                        "EcdsaSecp256k1 opcode violation: expected {:?} but found {:?} for output witness {:?}",
185                                        F::from(is_valid),
186                                        output_value,
187                                        output
188                                    ),
189                                ));
190                            }
191                        }
192                    }
193                    BlackBoxFuncCall::EcdsaSecp256r1 {
194                        public_key_x,
195                        public_key_y,
196                        signature,
197                        hashed_message,
198                        predicate,
199                        output,
200                    } => {
201                        let predicate_value = input_to_value(&witness_map, *predicate)?.is_one();
202                        if predicate_value {
203                            let is_valid = blackbox::signature::ecdsa::execute_ecdsa(
204                                &witness_map,
205                                public_key_x,
206                                public_key_y,
207                                signature,
208                                hashed_message,
209                                predicate,
210                                false,
211                            )?;
212                            let output_value = witness_value(output, &witness_map)?;
213                            if output_value != F::from(is_valid) {
214                                return Err(unsatisfied_constraint(
215                                    opcode_index,
216                                    format!(
217                                        "EcdsaSecp256r1 opcode violation: expected {:?} but found {:?} for output witness {:?}",
218                                        F::from(is_valid),
219                                        output_value,
220                                        output
221                                    ),
222                                ));
223                            }
224                        }
225                    }
226                    BlackBoxFuncCall::MultiScalarMul { points, scalars, predicate, outputs } => {
227                        let predicate_value = input_to_value(&witness_map, *predicate)?.is_one();
228                        if predicate_value {
229                            let (res_x, res_y, res_infinite) = execute_multi_scalar_mul(
230                                backend,
231                                &witness_map,
232                                points,
233                                scalars,
234                                *predicate,
235                            )?;
236                            let output_x_value = witness_value(&outputs.0, &witness_map)?;
237                            let output_y_value = witness_value(&outputs.1, &witness_map)?;
238                            let output_infinite_value = witness_value(&outputs.2, &witness_map)?;
239                            if res_x != output_x_value
240                                || res_y != output_y_value
241                                || res_infinite != output_infinite_value
242                            {
243                                //TODO: on check pas les x,y si infinite est true
244                                return Err(unsatisfied_constraint(
245                                    opcode_index,
246                                    format!(
247                                        "MultiScalarMul opcode violation: expected ({res_x}, {res_y}, {res_infinite}) but found ({output_x_value}, {output_y_value}, {output_infinite_value})"
248                                    ),
249                                ));
250                            }
251                        }
252                    }
253                    BlackBoxFuncCall::EmbeddedCurveAdd { input1, input2, predicate, outputs } => {
254                        let predicate_value = input_to_value(&witness_map, *predicate)?.is_one();
255                        if predicate_value {
256                            let (res_x, res_y, res_infinite) = execute_embedded_curve_add(
257                                backend,
258                                &witness_map,
259                                **input1,
260                                **input2,
261                                *predicate,
262                            )?;
263                            let output_x_value = witness_value(&outputs.0, &witness_map)?;
264                            let output_y_value = witness_value(&outputs.1, &witness_map)?;
265                            let output_infinite_value = witness_value(&outputs.2, &witness_map)?;
266                            if res_x != output_x_value
267                                || res_y != output_y_value
268                                || res_infinite != output_infinite_value
269                            {
270                                //TODO: on check pas les x,y si infinite est true
271                                return Err(unsatisfied_constraint(
272                                    opcode_index,
273                                    format!(
274                                        "EmbeddedCurveAdd opcode violation: expected ({res_x}, {res_y}, {res_infinite}) but found ({output_x_value}, {output_y_value}, {output_infinite_value})"
275                                    ),
276                                ));
277                            }
278                        }
279                    }
280                    BlackBoxFuncCall::Keccakf1600 { inputs, outputs } => {
281                        let mut state = [0; 25];
282                        for (it, input) in state.iter_mut().zip(inputs.as_ref()) {
283                            let witness_assignment = input_to_value(&witness_map, *input)?;
284                            let lane = witness_assignment.try_to_u64();
285                            *it = lane.unwrap();
286                        }
287                        let output_state = keccakf1600(state)?;
288                        for (output_witness, value) in outputs.iter().zip(output_state.into_iter())
289                        {
290                            let witness_value = witness_value(output_witness, &witness_map)?;
291                            if witness_value != F::from(u128::from(value)) {
292                                return Err(unsatisfied_constraint(
293                                    opcode_index,
294                                    format!(
295                                        "Keccakf1600 opcode violation: expected {value} but found {witness_value} for output witness {output_witness}",
296                                    ),
297                                ));
298                            }
299                        }
300                    }
301                    // Recursive aggregation is checked outside of ACVM
302                    BlackBoxFuncCall::RecursiveAggregation { .. } => (),
303                    BlackBoxFuncCall::Poseidon2Permutation { inputs, outputs } => {
304                        let state = blackbox::hash::execute_poseidon2_permutation_opcode(
305                            backend,
306                            &witness_map,
307                            inputs,
308                        )?;
309
310                        for (output_witness, value) in outputs.iter().zip(state.into_iter()) {
311                            let witness_value = witness_map
312                                .get(output_witness)
313                                .ok_or(OpcodeNotSolvable::MissingAssignment(output_witness.0))?;
314                            if *witness_value != value {
315                                return Err(unsatisfied_constraint(
316                                    opcode_index,
317                                    format!(
318                                        "Poseidon2 opcode violation: expected {value} but found {witness_value} for output witness {output_witness}",
319                                    ),
320                                ));
321                            }
322                        }
323                    }
324                    BlackBoxFuncCall::Sha256Compression { inputs, hash_values, outputs } => {
325                        let state = blackbox::hash::execute_sha_256_permutation_opcode(
326                            &witness_map,
327                            inputs,
328                            hash_values,
329                        )?;
330
331                        for (output_witness, value) in outputs.iter().zip(state.into_iter()) {
332                            let witness_value = witness_map
333                                .get(output_witness)
334                                .ok_or(OpcodeNotSolvable::MissingAssignment(output_witness.0))?;
335                            if *witness_value != F::from(u128::from(value)) {
336                                return Err(unsatisfied_constraint(
337                                    opcode_index,
338                                    format!(
339                                        "SHA256 Compression opcode violation: expected {:?} but found {:?} for output witness {:?}",
340                                        F::from(u128::from(value)),
341                                        witness_value,
342                                        output_witness
343                                    ),
344                                ));
345                            }
346                        }
347                    }
348                }
349            }
350            Opcode::MemoryOp { block_id, op } => {
351                let solver = block_solvers
352                    .get_mut(block_id)
353                    .expect("Memory block should have been initialized");
354                solver.check_memory_op(op, &witness_map, opcode_index)?;
355            }
356            Opcode::MemoryInit { block_id, init, .. } => {
357                MemoryOpSolver::new(init, &witness_map).map(|solver| {
358                    let existing_block_id = block_solvers.insert(*block_id, solver);
359                    assert!(existing_block_id.is_none(), "Memory block already initialized");
360                })?;
361            }
362            // BrilligCall is unconstrained
363            Opcode::BrilligCall { .. } => (),
364            Opcode::Call { id: _, inputs, outputs, predicate } => {
365                // Skip validation when predicate is false
366                let pred_value = get_value(predicate, &witness_map)?;
367                if pred_value.is_zero() {
368                    continue;
369                }
370
371                // Verify input witnesses exist
372                for input in inputs {
373                    if witness_map.get(input).is_none() {
374                        return Err(OpcodeNotSolvable::MissingAssignment(input.0).into());
375                    }
376                }
377
378                // Verify output witnesses exist (value should have been validated by the called function)
379                for output in outputs {
380                    if witness_map.get(output).is_none() {
381                        return Err(OpcodeNotSolvable::MissingAssignment(output.0).into());
382                    }
383                }
384            }
385        }
386    }
387
388    Ok(())
389}

391impl<F: AcirField> MemoryOpSolver<F> {
392    pub(crate) fn check_memory_op(
393        &mut self,
394        op: &MemOp<F>,
395        witness_map: &WitnessMap<F>,
396        opcode_index: usize,
397    ) -> Result<(), OpcodeResolutionError<F>> {
398        let operation = get_value(&op.operation, witness_map)?;
399
400        // Find the memory index associated with this memory operation.
401        let index = get_value(&op.index, witness_map)?;
402        let memory_index = self.index_from_field(index)?;
403
404        // Calculate the value associated with this memory operation.
405        let value = get_value(&op.value, witness_map)?;
406
407        // `operation == 0` for read operation, `operation == 1` for write operation.
408        let is_read_operation = operation.is_zero();
409
410        if is_read_operation {
411            // `value = arr[memory_index]`
412            let value_in_array = self.read_memory_index(memory_index)?;
413            if value != value_in_array {
414                return Err(unsatisfied_constraint(
415                    opcode_index,
416                    format!(
417                        "Memory read opcode violation at index {memory_index}: expected {value_in_array} but found {value}",
418                    ),
419                ));
420            }
421            Ok(())
422        } else {
423            // `arr[memory_index] = value`
424            self.write_memory_index(memory_index, value)
425        }
426    }
427}
