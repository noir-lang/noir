12pub enum CompareResults {
13    Agree(WitnessStack<FieldElement>, WitnessStack<FieldElement>),
14    Disagree(WitnessStack<FieldElement>, WitnessStack<FieldElement>),
15    BothFailed(String, String),
16    AcirFailed(String, WitnessStack<FieldElement>),
17    BrilligFailed(String, WitnessStack<FieldElement>),
18}

27/// High level function to execute the given ACIR and Brillig programs with the given initial witness
28/// It returns a tuple with a boolean indicating if the programs succeeded,
29/// and the results of the ACIR and Brillig programs
30pub fn run_and_compare(
31    acir_program: &Program<FieldElement>,
32    brillig_program: &Program<FieldElement>,
33    initial_witness: WitnessMap<FieldElement>,
34) -> CompareResults {
35    let acir_result = execute_single(acir_program, initial_witness.clone());
36    let brillig_result = execute_single(brillig_program, initial_witness);
37
38    let return_witnesses_acir = &acir_program.functions[0].return_values;
39    let return_witnesses_brillig = &brillig_program.functions[0].return_values;
40    let return_witness_acir: BTreeSet<Witness> = return_witnesses_acir.0.clone();
41    let return_witness_brillig: BTreeSet<Witness> = return_witnesses_brillig.0.clone();
42
43    // we found bug in case of
44    // 1) acir_result != brillig_result
45    // 2) acir execution failed, brillig execution succeeded
46    // 3) acir execution succeeded, brillig execution failed
47    match (acir_result, brillig_result) {
48        (Ok(acir_witness), Ok(brillig_witness)) => {
49            // we assume that if execution for both modes succeeds both programs returned something
50            let acir_witness_map = acir_witness.peek().unwrap().witness.clone();
51            let brillig_witness_map = brillig_witness.peek().unwrap().witness.clone();
52            let acir_results = return_witness_acir.iter().map(|w| acir_witness_map[w]);
53            let brillig_results = return_witness_brillig.iter().map(|w| brillig_witness_map[w]);
54            let results_equal = acir_results.eq(brillig_results);
55            if results_equal {
56                CompareResults::Agree(acir_witness, brillig_witness)
57            } else {
58                CompareResults::Disagree(acir_witness, brillig_witness)
59            }
60        }
61        (Err(acir_error), Ok(brillig_witness)) => {
62            CompareResults::AcirFailed(acir_error.to_string(), brillig_witness)
63        }
64        (Ok(acir_witness), Err(brillig_error)) => {
65            CompareResults::BrilligFailed(brillig_error.to_string(), acir_witness)
66        }
67        (Err(acir_error), Err(brillig_error)) => {
68            CompareResults::BothFailed(acir_error.to_string(), brillig_error.to_string())
69        }
70    }
71}
