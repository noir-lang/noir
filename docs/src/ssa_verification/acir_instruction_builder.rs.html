28/// Represents artifacts generated from compiling an instruction
29#[derive(Serialize, Deserialize)]
30pub(crate) struct InstructionArtifacts {
31    /// Name of the instruction
32    pub(crate) instruction_name: String,
33
34    /// SSA representation formatted as "acir(inline) {...}"
35    pub(crate) formatted_ssa: String,
36
37    /// JSON serialized SSA
38    pub(crate) serialized_ssa: String,
39
40    /// Gzipped bytes of ACIR program
41    pub(crate) serialized_acir: Vec<u8>,
42}

44/// Represents the type of a variable in the instruction
45#[derive(Debug)]
46pub(crate) enum VariableType {
47    /// Field element type
48    Field,
49    /// Unsigned integer type
50    Unsigned,
51    /// Signed integer type
52    Signed,
53}
54
55/// Represents a variable with its type and size information
56pub(crate) struct Variable {
57    /// Type of the variable (Field, Unsigned, or Signed)
58    pub(crate) variable_type: VariableType,
59    /// Bit size of the variable (ignored for Field type)
60    pub(crate) variable_size: u32,
61}
62
63impl Variable {
64    /// Gets a string representation of the variable's type and size
65    pub(crate) fn get_name(&self) -> String {
66        format!("{:?}_{}", self.variable_type, self.variable_size)
67    }
68}

70impl InstructionArtifacts {
71    /// Converts a Variable into its corresponding SSA Type
72    fn get_type(variable: &Variable) -> Type {
73        match variable.variable_type {
74            VariableType::Field => Type::field(),
75            VariableType::Signed => Type::signed(variable.variable_size),
76            VariableType::Unsigned => Type::unsigned(variable.variable_size),
77        }
78    }
79
80    /// Creates a new binary operation instruction artifact
81    fn new_binary(
82        op: BinaryOp,
83        instruction_name: String,
84        first_variable: &Variable,
85        second_variable: &Variable,
86    ) -> Self {
87        let first_variable_type = Self::get_type(first_variable);
88        let second_variable_type = Self::get_type(second_variable);
89        let ssa = binary_function(op, first_variable_type, second_variable_type);
90        let serialized_ssa = &serde_json::to_string(&ssa).unwrap();
91        let formatted_ssa = format!("{}", ssa.print_without_locations());
92
93        let program = ssa_to_acir_program(ssa);
94        let serialized_program = AcirProgram::serialize_program(&program);
95        let name = format!(
96            "{}_{}_{}",
97            instruction_name,
98            first_variable.get_name(),
99            second_variable.get_name()
100        );
101
102        Self {
103            instruction_name: name,
104            formatted_ssa,
105            serialized_ssa: serialized_ssa.to_string(),
106            serialized_acir: serialized_program,
107        }
108    }
109
110    /// Creates a new instruction artifact using a provided SSA generation function

235/// Converts SSA to ACIR program
236fn ssa_to_acir_program(ssa: Ssa) -> AcirProgram<FieldElement> {
237    // third brillig names, fourth errors
238    let builder = SsaBuilder::from_ssa(ssa, SsaLogging::None, false, None);
239    let ssa_evaluator_options = SsaEvaluatorOptions {
240        ssa_logging: SsaLogging::None,
241        print_codegen_timings: false,
242        emit_ssa: { None },
243        skip_underconstrained_check: true,
244        skip_brillig_constraints_check: true,
245        inliner_aggressiveness: 0,
246        constant_folding_max_iter: CONSTANT_FOLDING_MAX_ITER,
247        small_function_max_instruction: INLINING_MAX_INSTRUCTIONS,
248        max_bytecode_increase_percent: None,
249        force_unroll_threshold: FORCE_UNROLL_THRESHOLD,
250        brillig_options: BrilligOptions::default(),
251        enable_brillig_constraints_check_lookback: false,
252        skip_passes: vec![],
253    };
254    let (acir_functions, brillig, _) = match optimize_ssa_builder_into_acir(
255        builder,
256        &ssa_evaluator_options,
257        &primary_passes(&ssa_evaluator_options),
258    ) {
259        Ok(artifacts_and_warnings) => artifacts_and_warnings.0,
260        Err(_) => panic!("Should compile manually generated SSA into acir"),
261    };
262
263    let mut functions: Vec<Circuit<FieldElement>> = Vec::new();
264
265    for acir_func in acir_functions.iter() {
266        let mut private_params: BTreeSet<Witness> =
267            acir_func.input_witnesses.clone().into_iter().collect();
268        let ret_values: BTreeSet<Witness> =
269            acir_func.return_witnesses.clone().into_iter().collect();
270
271        private_params.extend(ret_values.iter().cloned());
272        let circuit: Circuit<FieldElement> = Circuit {
273            current_witness_index: acir_func.current_witness_index().witness_index(),
274            opcodes: acir_func.opcodes.clone(),
275            private_parameters: private_params.clone(),
276            ..Circuit::<FieldElement>::default()
277        };
278        functions.push(circuit);
279    }
280    AcirProgram { functions, unconstrained_functions: brillig }
281}

283/// Creates an SSA function for binary operations
284fn binary_function(op: BinaryOp, first_variable_type: Type, second_variable_type: Type) -> Ssa {
285    // returns v0 op v1
286    let main_id: Id<Function> = Id::new(0);
287    let mut builder = FunctionBuilder::new("main".into(), main_id);
288    let v0 = builder.add_parameter(first_variable_type);
289    let v1 = builder.add_parameter(second_variable_type);
290    let v2 = builder.insert_binary(v0, op, v1);
291    builder.terminate_with_return(vec![v2]);
292
293    builder.finish()
294}

296/// Creates an SSA function for constraint operations
297fn constrain_function(variable_type: Type) -> Ssa {
298    // constrains v0 == v1, returns v1
299    let main_id: Id<Function> = Id::new(0);
300    let mut builder = FunctionBuilder::new("main".into(), main_id);
301
302    let v0 = builder.add_parameter(variable_type.clone());
303    let v1 = builder.add_parameter(variable_type);
304    builder.insert_constrain(v0, v1, None);
305    builder.terminate_with_return(vec![v1]);
306
307    builder.finish()
308}

310/// Creates an SSA function for range check operations
311fn range_check_function(variable_type: Type, bit_size: u32) -> Ssa {
312    let main_id: Id<Function> = Id::new(0);
313    let mut builder = FunctionBuilder::new("main".into(), main_id);
314
315    let v0 = builder.add_parameter(variable_type);
316    builder.insert_range_check(v0, bit_size, Some("Range Check failed".to_string()));
317    builder.terminate_with_return(vec![v0]);
318
319    builder.finish()
320}
