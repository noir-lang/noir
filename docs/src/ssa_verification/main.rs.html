34/// Decompresses gzipped data into a byte vector
35fn ungzip(compressed_data: &[u8]) -> Vec<u8> {
36    let mut decompressed_data: Vec<u8> = Vec::new();
37    let mut decoder = GzDecoder::new(compressed_data);
38    decoder.read_to_end(&mut decompressed_data).unwrap();
39    decompressed_data
40}

42/// Saves byte data to a file at the specified path
43fn save_to_file(data: &[u8], filename: &str) -> Result<(), std::io::Error> {
44    let path = Path::new(filename);
45    let mut file = File::create(path)?;
46    file.write_all(data)?;
47    Ok(())
48}

50/// Saves instruction artifacts to files in the artifacts directory
51/// Prints the formatted SSA for each artifact and saves the decompressed ACIR
52fn save_artifacts(all_artifacts: Vec<InstructionArtifacts>, dir: &str) {
53    for artifacts in all_artifacts.iter() {
54        println!("{}\n{}", artifacts.instruction_name, artifacts.formatted_ssa);
55        let filename = format!("{}{}{}", dir, artifacts.instruction_name, ".acir");
56        let acir = &artifacts.serialized_acir;
57        match save_to_file(&ungzip(acir), &filename) {
58            Ok(_) => (),
59            Err(error) => {
60                eprintln!("Error saving data: {error}");
61                std::process::exit(1);
62            }
63        }
64    }
65}
