12/// Adds the file from the file system at `Path` to the crate graph as a root file
13///
14/// Note: If the stdlib dependency has not been added yet, it's added. Otherwise
15/// this method assumes the root crate is the stdlib (useful for running tests
16/// in the stdlib, getting LSP stuff for the stdlib, etc.).
17pub fn prepare_crate(context: &mut Context, file_name: &Path) -> CrateId {
18    let path_to_std_lib_file = Path::new(STD_CRATE_NAME).join("lib.nr");
19    let std_file_id = context.file_manager.name_to_id(path_to_std_lib_file);
20    let std_crate_id = std_file_id.map(|std_file_id| context.crate_graph.add_stdlib(std_file_id));
21
22    let root_file_id = context.file_manager.name_to_id(file_name.to_path_buf()).unwrap_or_else(|| panic!("files are expected to be added to the FileManager before reaching the compiler file_path: {}", file_name.display()));
23
24    if let Some(std_crate_id) = std_crate_id {
25        let root_crate_id = context.crate_graph.add_crate_root(root_file_id);
26
27        add_dep(context, root_crate_id, std_crate_id, STD_CRATE_NAME.parse().unwrap());
28
29        root_crate_id
30    } else {
31        context.crate_graph.add_crate_root_and_stdlib(root_file_id)
32    }
33}

35pub fn link_to_debug_crate(context: &mut Context, root_crate_id: CrateId) {
36    let path_to_debug_lib_file = Path::new(DEBUG_CRATE_NAME).join("lib.nr");
37    let debug_crate_id = prepare_dependency(context, &path_to_debug_lib_file);
38    add_dep(context, root_crate_id, debug_crate_id, DEBUG_CRATE_NAME.parse().unwrap());
39    context.debug_crate_id = Some(debug_crate_id);
40}

42// Adds the file from the file system at `Path` to the crate graph
43pub fn prepare_dependency(context: &mut Context, file_name: &Path) -> CrateId {
44    let root_file_id = context
45        .file_manager
46        .name_to_id(file_name.to_path_buf())
47        .unwrap_or_else(|| panic!("files are expected to be added to the FileManager before reaching the compiler file_path: {}", file_name.display()));
48
49    let crate_id = context.crate_graph.add_crate(root_file_id);
50
51    // Every dependency has access to stdlib
52    let std_crate_id = context.stdlib_crate_id();
53    add_dep(context, crate_id, *std_crate_id, STD_CRATE_NAME.parse().unwrap());
54
55    crate_id
56}

58/// Adds a edge in the crate graph for two crates
59pub fn add_dep(
60    context: &mut Context,
61    this_crate: CrateId,
62    depends_on: CrateId,
63    crate_name: CrateName,
64) {
65    context
66        .crate_graph
67        .add_dep(this_crate, crate_name, depends_on)
68        .expect("cyclic dependency triggered");
