3#[derive(RustEmbed)]

6#[cfg_attr(target_os = "windows", prefix = r"std\")] // Note reversed slash direction
7struct StdLibAssets;
8
9// Returns a vector of tuples containing the path to a stdlib file in the std lib crate
10// along with the source code of that file.
11//
12// This is needed because when we preload the file manager, it needs to know where
13// the source code for the stdlib is. The stdlib is treated special because it comes with
14// the compiler and is never included as a dependency like other user defined crates.
15pub fn stdlib_paths_with_source() -> Vec<(String, String)> {
16    StdLibAssets::iter()
17        .map(|path| {
18            let source = std::str::from_utf8(StdLibAssets::get(&path).unwrap().data.as_ref())
19                .unwrap()
20                .to_string();
21            (path.to_string(), source)
22        })
23        .collect()
24}
25
26/// Returns the contents of the Nargo.toml file for the standard library as a string.
27pub fn stdlib_nargo_toml_source() -> String {
28    include_str!("../../../noir_stdlib/Nargo.toml").to_string()
