11// Returns a vector of tuples containing the path to a stdlib file in the std lib crate
12// along with the source code of that file.
13//
14// This is needed because when we preload the file manager, it needs to know where
15// the source code for the stdlib is. The stdlib is treated special because it comes with
16// the compiler and is never included as a dependency like other user defined crates.
17pub fn stdlib_paths_with_source() -> Vec<(String, String)> {
18    StdLibAssets::iter()
19        .map(|path| {
20            let source = std::str::from_utf8(StdLibAssets::get(&path).unwrap().data.as_ref())
21                .unwrap()
22                .to_string();
23            (path.to_string(), source)
24        })
25        .collect()
26}

28/// Returns the contents of the Nargo.toml file for the standard library as a string.

30    include_str!("../../../noir_stdlib/Nargo.toml").to_string()
31}

33/// Returns the absolute path to the `noir_stdlib/src` directory on disk, when the
34/// running binary was built from the monorepo (debug build only) and the directory
35/// is still reachable from where the binary was compiled. Returns `None` for
36/// release builds or when the source tree is no longer present (e.g. a distributed
37/// `nargo`), so callers must fall back to the embedded stdlib copy.
38pub fn stdlib_disk_path() -> Option<PathBuf> {
39    if !cfg!(debug_assertions) {
40        return None;
41    }
42    PathBuf::from(env!("CARGO_MANIFEST_DIR")).join("../../noir_stdlib/src").canonicalize().ok()
43}
