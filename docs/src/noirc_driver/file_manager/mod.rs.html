26/// Adds the source code for the stdlib into the file manager
27fn add_stdlib_source_to_file_manager(file_manager: &mut FileManager) {
28    // Add the stdlib contents to the file manager, since every package automatically has a dependency
29    // on the stdlib. For other dependencies, we read the package.Dependencies file to add their file
30    // contents to the file manager. However since the dependency on the stdlib is implicit, we need
31    // to manually add it here.
32    let stdlib_paths_with_source = stdlib_paths_with_source();
33    for (path, source) in stdlib_paths_with_source {
34        file_manager.add_file_with_source_canonical_path(Path::new(&path), source);
35    }
36}

38/// Adds the source code of the debug crate needed to support instrumentation to
39/// track variables values
40fn add_debug_source_to_file_manager(file_manager: &mut FileManager) {
41    // Adds the synthetic debug module for instrumentation into the file manager
42    let path_to_debug_lib_file = Path::new(DEBUG_CRATE_NAME).join("lib.nr");
43    file_manager
44        .add_file_with_source_canonical_path(&path_to_debug_lib_file, build_debug_crate_file());
45}
