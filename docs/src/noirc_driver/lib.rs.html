364/// Adds the source code for the stdlib into the file manager
365fn add_stdlib_source_to_file_manager(file_manager: &mut FileManager) {
366    // Add the stdlib contents to the file manager, since every package automatically has a dependency
367    // on the stdlib. For other dependencies, we read the package.Dependencies file to add their file
368    // contents to the file manager. However since the dependency on the stdlib is implicit, we need
369    // to manually add it here.
370    let stdlib_paths_with_source = stdlib_paths_with_source();
371    for (path, source) in stdlib_paths_with_source {
372        file_manager.add_file_with_source_canonical_path(Path::new(&path), source);
373    }
374}

385/// Adds the file from the file system at `Path` to the crate graph as a root file
386///
387/// Note: If the stdlib dependency has not been added yet, it's added. Otherwise
388/// this method assumes the root crate is the stdlib (useful for running tests
389/// in the stdlib, getting LSP stuff for the stdlib, etc.).
390pub fn prepare_crate(context: &mut Context, file_name: &Path) -> CrateId {
391    let path_to_std_lib_file = Path::new(STD_CRATE_NAME).join("lib.nr");
392    let std_file_id = context.file_manager.name_to_id(path_to_std_lib_file);
393    let std_crate_id = std_file_id.map(|std_file_id| context.crate_graph.add_stdlib(std_file_id));
394
395    let root_file_id = context.file_manager.name_to_id(file_name.to_path_buf()).unwrap_or_else(|| panic!("files are expected to be added to the FileManager before reaching the compiler file_path: {}", file_name.display()));
396
397    if let Some(std_crate_id) = std_crate_id {
398        let root_crate_id = context.crate_graph.add_crate_root(root_file_id);
399
400        add_dep(context, root_crate_id, std_crate_id, STD_CRATE_NAME.parse().unwrap());
401
402        root_crate_id
403    } else {
404        context.crate_graph.add_crate_root_and_stdlib(root_file_id)
405    }
406}

414// Adds the file from the file system at `Path` to the crate graph
415pub fn prepare_dependency(context: &mut Context, file_name: &Path) -> CrateId {
416    let root_file_id = context
417        .file_manager
418        .name_to_id(file_name.to_path_buf())
419        .unwrap_or_else(|| panic!("files are expected to be added to the FileManager before reaching the compiler file_path: {}", file_name.display()));
420
421    let crate_id = context.crate_graph.add_crate(root_file_id);
422
423    // Every dependency has access to stdlib
424    let std_crate_id = context.stdlib_crate_id();
425    add_dep(context, crate_id, *std_crate_id, STD_CRATE_NAME.parse().unwrap());
426
427    crate_id
428}

443/// Run the def collection, elaboration and type checking passes.
444///
445/// On success, this returns () alongside any warnings found during checking.
446///
447/// On error, this returns a non-empty vector of warnings and error messages, with at least one error.
448#[tracing::instrument(level = "trace", skip_all)]
449pub fn check_crate(
450    context: &mut Context,
451    crate_id: CrateId,
452    options: &CompileOptions,
453) -> CompilationResult<()> {
454    if options.disable_comptime_printing {
455        context.disable_comptime_printing();
456    }
457
458    let diagnostics = CrateDefMap::collect_defs(crate_id, context, options.frontend_options());
459    let crate_files = context.crate_files(&crate_id);
460    let warnings_and_errors: Vec<CustomDiagnostic> = diagnostics
461        .iter()
462        .map(CustomDiagnostic::from)
463        .filter(|diagnostic| {
464            // We filter out any warnings if they're going to be ignored later on to free up memory.
465            !options.silence_warnings || diagnostic.kind != DiagnosticKind::Warning
466        })
467        .filter(|error| {
468            // Only keep warnings from the crate we are checking
469            if error.is_warning() { crate_files.contains(&error.file) } else { true }
470        })
471        .collect();
472
473    if has_errors(&warnings_and_errors, options.deny_warnings) {
474        Err(warnings_and_errors)
475    } else {
476        Ok(((), warnings_and_errors))
477    }
478}

489/// Run the frontend to check the crate for errors then compile the main function if there were none
490///
491/// On success this returns the compiled program alongside any warnings that were found.
492/// On error this returns the non-empty list of warnings and errors.
493///
494/// See [compile_no_check] for further information about the use of `cached_program`.
495pub fn compile_main(
496    context: &mut Context,
497    crate_id: CrateId,
498    options: &CompileOptions,
499    cached_program: Option<CompiledProgram>,
500) -> CompilationResult<CompiledProgram> {
501    let (_, mut warnings) = check_crate(context, crate_id, options)?;
502
503    let main = context.get_main_function(&crate_id).ok_or_else(|| {
504        // TODO(#2155): This error might be a better to exist in Nargo
505        let err = CustomDiagnostic::from_message(
506            "cannot compile crate into a program as it does not contain a `main` function",
507            FileId::default(),
508        );
509        vec![err]
510    })?;
511
512    let compiled_program =
513        compile_no_check(context, options, main, cached_program, options.force_compile)
514            .map_err(|error| vec![CustomDiagnostic::from(error)])?;
515
516    let compilation_warnings =
517        vecmap(compiled_program.warnings.clone(), ssa_report_to_custom_diagnostic);
518    if options.deny_warnings && !compilation_warnings.is_empty() {
519        return Err(compilation_warnings);
520    }
521    if !options.silence_warnings {
522        warnings.extend(compilation_warnings);
523    }
524
525    if options.print_acir {
526        noirc_errors::println_to_stdout!("Compiled ACIR for main:");
527        noirc_errors::println_to_stdout!("{}", compiled_program.program);
528    }
529
530    Ok((compiled_program, warnings))
531}

533/// Run the frontend to check the crate for errors then compile all contracts if there were none
534pub fn compile_contract(
535    context: &mut Context,
536    crate_id: CrateId,
537    options: &CompileOptions,
538) -> CompilationResult<CompiledContract> {
539    let (_, warnings) = check_crate(context, crate_id, options)?;
540
541    let def_map = context.def_map(&crate_id).expect("The local crate should be analyzed already");
542    let mut contracts = def_map.get_all_contracts();
543
544    let Some((module_id, name)) = contracts.next() else {
545        let err = CustomDiagnostic::from_message(
546            "cannot compile crate into a contract as it does not contain any contracts",
547            FileId::default(),
548        );
549        return Err(vec![err]);
550    };
551
552    if contracts.next().is_some() {
553        let err = CustomDiagnostic::from_message(
554            "Packages are limited to a single contract",
555            FileId::default(),
556        );
557        return Err(vec![err]);
558    }
559    drop(contracts);
560
561    let module_id = ModuleId { krate: crate_id, local_id: module_id };
562    let contract = read_contract(context, module_id, name);
563
564    let mut errors = warnings;
565
566    let compiled_contract = match compile_contract_inner(context, contract, options) {
567        Ok(contract) => contract,
568        Err(mut more_errors) => {
569            errors.append(&mut more_errors);
570            return Err(errors);
571        }
572    };
573
574    if has_errors(&errors, options.deny_warnings) {
575        Err(errors)
576    } else {
577        if options.print_acir {
578            for contract_function in &compiled_contract.functions {
579                if let Some(ref name) = options.show_contract_fn
580                    && name != &contract_function.name
581                {
582                    continue;
583                }
584                println!(
585                    "Compiled ACIR for {}::{} (non-transformed):",
586                    compiled_contract.name, contract_function.name
587                );
588                println!("{}", contract_function.bytecode);
589            }
590        }
591        // errors here is either empty or contains only warnings
592        Ok((compiled_contract, errors))
593    }
594}

596/// Return a Vec of all `contract` declarations in the source code and the functions they contain
597fn read_contract(context: &Context, module_id: ModuleId, name: String) -> Contract {
598    let module = context.module(module_id);
599
600    let functions: Vec<ContractFunctionMeta> = module
601        .value_definitions()
602        .filter_map(|id| {
603            id.as_function().map(|function_id| {
604                let attrs = context.def_interner.function_attributes(&function_id);
605                let is_entry_point = attrs.is_contract_entry_point();
606                ContractFunctionMeta { function_id, is_entry_point }
607            })
608        })
609        .collect();
610
611    let mut outputs = ContractOutputs { structs: HashMap::new(), globals: HashMap::new() };
612
613    context.def_interner.get_all_globals().iter().for_each(|global_info| {
614        context.def_interner.global_attributes(&global_info.id).iter().for_each(|attr| {
615            if let SecondaryAttributeKind::Abi(tag) = &attr.kind {
616                if let Some(tagged) = outputs.globals.get_mut(tag) {
617                    tagged.push(global_info.id);
618                } else {
619                    outputs.globals.insert(tag.to_string(), vec![global_info.id]);
620                }
621            }
622        });
623    });
624
625    module.type_definitions().for_each(|id| {
626        if let ModuleDefId::TypeId(struct_id) = id {
627            context.def_interner.type_attributes(&struct_id).iter().for_each(|attr| {
628                if let SecondaryAttributeKind::Abi(tag) = &attr.kind {
629                    if let Some(tagged) = outputs.structs.get_mut(tag) {
630                        tagged.push(struct_id);
631                    } else {
632                        outputs.structs.insert(tag.to_string(), vec![struct_id]);
633                    }
634                }
635            });
636        }
637    });
638
639    Contract { name, functions, outputs }
640}

642/// True if there are (non-warning) errors present and we should halt compilation
643fn has_errors(errors: &[CustomDiagnostic], deny_warnings: bool) -> bool {
644    if deny_warnings { !errors.is_empty() } else { errors.iter().any(|error| error.is_error()) }
645}

647/// Compile all of the functions associated with a Noir contract.
648fn compile_contract_inner(
649    context: &mut Context,
650    contract: Contract,
651    options: &CompileOptions,
652) -> Result<CompiledContract, ErrorsAndWarnings> {
653    let mut functions = Vec::new();
654    let mut errors = Vec::new();
655    let mut warnings = Vec::new();
656    for contract_function in &contract.functions {
657        let function_id = contract_function.function_id;
658        let is_entry_point = contract_function.is_entry_point;
659
660        let name = context.function_name(&function_id).to_owned();
661
662        // We assume that functions have already been type checked.
663        // This is the exact same assumption that compile_no_check makes.
664        // If it is not an entry-point point, we can then just skip the
665        // compilation step. It will also not be added to the ABI.
666        if !is_entry_point {
667            continue;
668        }
669
670        let mut options = options.clone();
671        if name == "public_dispatch" {
672            options.inliner_aggressiveness = 0;
673        }
674
675        if let Some(ref name_filter) = options.show_contract_fn {
676            let show = name == *name_filter;
677            options.show_ssa &= show;
678            if !show {
679                options.show_ssa_pass.clear();
680            }
681        }
682
683        let function = match compile_no_check(context, &options, function_id, None, true) {
684            Ok(function) => function,
685            Err(new_error) => {
686                errors.push(new_error.into());
687                continue;
688            }
689        };
690        warnings.extend(function.warnings);
691        let modifiers = context.def_interner.function_modifiers(&function_id);
692        let is_unconstrained = context.def_interner.function_meta(&function_id).is_unconstrained();
693
694        let custom_attributes = modifiers
695            .attributes
696            .secondary
697            .iter()
698            .filter_map(|attr| match &attr.kind {
699                SecondaryAttributeKind::Tag(contents) => Some(contents.clone()),
700                SecondaryAttributeKind::Meta(meta_attribute) => {
701                    context.def_interner.get_meta_attribute_name(meta_attribute)
702                }
703                _ => None,
704            })
705            .collect();
706
707        functions.push(ContractFunction {
708            name,
709            hash: function.hash,
710            custom_attributes,
711            abi: function.abi,
712            bytecode: function.program,
713            debug: function.debug,
714            is_unconstrained,
715        });
716    }
717
718    if errors.is_empty() {
719        let debug_infos: Vec<_> =
720            functions.iter().flat_map(|function| function.debug.clone()).collect();
721        let file_map = filter_relevant_files(&debug_infos, &context.file_manager);
722
723        let out_structs = contract
724            .outputs
725            .structs
726            .into_iter()
727            .map(|(tag, structs)| {
728                let structs = structs
729                    .into_iter()
730                    .map(|struct_id| {
731                        let typ = context.def_interner.get_type(struct_id);
732                        let typ = typ.borrow();
733                        let fields =
734                            vecmap(typ.get_fields(&[]).unwrap_or_default(), |(name, typ, _)| {
735                                (name, abi_type_from_hir_type(context, &typ))
736                            });
737                        let path =
738                            context.fully_qualified_struct_path(context.root_crate_id(), typ.id);
739                        AbiType::Struct { path, fields }
740                    })
741                    .collect();
742                (tag.to_string(), structs)
743            })
744            .collect();
745
746        let out_globals = contract
747            .outputs
748            .globals
749            .iter()
750            .map(|(tag, globals)| {
751                let globals: Vec<AbiValue> = globals
752                    .iter()
753                    .map(|global_id| {
754                        let let_statement =
755                            context.def_interner.get_global_let_statement(*global_id).unwrap();
756                        let hir_expression =
757                            context.def_interner.expression(&let_statement.expression);
758                        value_from_hir_expression(context, hir_expression)
759                    })
760                    .collect();
761                (tag.to_string(), globals)
762            })
763            .collect();
764
765        Ok(CompiledContract {
766            name: contract.name,
767            functions,
768            outputs: CompiledContractOutputs { structs: out_structs, globals: out_globals },
769            file_map,
770            noir_version: NOIR_ARTIFACT_VERSION_STRING.to_string(),
771            warnings,
772        })
773    } else {
774        Err(errors)
775    }
776}

778pub fn filter_relevant_files(
779    debug_symbols: &[DebugInfo],
780    file_manager: &FileManager,
781) -> BTreeMap<FileId, DebugFile> {
782    let mut files_with_debug_symbols: BTreeSet<FileId> = debug_symbols
783        .iter()
784        .flat_map(|function_symbols| {
785            function_symbols.acir_locations.values().flat_map(|call_stack_id| {
786                function_symbols
787                    .location_tree
788                    .get_call_stack(*call_stack_id)
789                    .into_iter()
790                    .map(|location| location.file)
791            })
792        })
793        .collect();
794
795    let files_with_brillig_debug_symbols: BTreeSet<FileId> = debug_symbols
796        .iter()
797        .flat_map(|function_symbols| {
798            function_symbols.brillig_locations.values().flat_map(|brillig_location_map| {
799                brillig_location_map.values().flat_map(|call_stack_id| {
800                    function_symbols
801                        .location_tree
802                        .get_call_stack(*call_stack_id)
803                        .into_iter()
804                        .map(|location| location.file)
805                })
806            })
807        })
808        .collect();
809
810    files_with_debug_symbols.extend(files_with_brillig_debug_symbols);
811
812    let mut file_map = BTreeMap::new();
813
814    for file_id in files_with_debug_symbols {
815        let file_path = file_manager.path(file_id).expect("file should exist");
816        let file_source = file_manager.fetch_file(file_id).expect("file should exist");
817
818        file_map.insert(
819            file_id,
820            DebugFile { source: file_source.to_string(), path: file_path.to_path_buf() },
821        );
822    }
823    file_map
824}

826/// Compile the current crate using `main_function` as the entrypoint.
827///
828/// This function assumes [`check_crate`] is called beforehand.
829///
830/// If the program is not returned from cache, it is backend-agnostic and must go through a transformation
831/// pass before usage in proof generation; if it's returned from cache these transformations might have
832/// already been applied.
833///
834/// The transformations are _not_ covered by the check that decides whether we can use the cached artifact.
835/// That comparison is based on on [CompiledProgram::hash] which is a persisted version of the hash of the input
836/// [`ast::Program`][noirc_frontend::monomorphization::ast::Program], whereas the output [`circuit::Program`][acvm::acir::circuit::Program]
837/// contains the final optimized ACIR opcodes, including the transformation done after this compilation.
838#[tracing::instrument(level = "trace", skip_all, fields(function_name = context.function_name(&main_function)))]
839#[allow(clippy::result_large_err)]
840pub fn compile_no_check(
841    context: &mut Context,
842    options: &CompileOptions,
843    main_function: FuncId,
844    cached_program: Option<CompiledProgram>,
845    force_compile: bool,
846) -> Result<CompiledProgram, CompileError> {
847    let force_unconstrained = options.force_brillig || options.minimal_ssa;
848
849    let program = if options.instrument_debug {
850        monomorphize_debug(
851            main_function,
852            &mut context.def_interner,
853            &context.debug_instrumenter,
854            force_unconstrained,
855        )?
856    } else {
857        monomorphize(main_function, &mut context.def_interner, force_unconstrained)?
858    };
859
860    if options.show_monomorphized {
861        println!("{program}");
862    }
863
864    // If user has specified that they want to see intermediate steps printed then we should
865    // force compilation even if the program hasn't changed.
866    let force_compile = force_compile
867        || options.print_acir
868        || options.show_brillig
869        || options.force_brillig
870        || options.count_array_copies
871        || options.show_ssa
872        || !options.show_ssa_pass.is_empty()
873        || options.emit_ssa
874        || options.minimal_ssa;
875
876    // Hash the AST program, which is going to be used to fingerprint the compilation artifact.
877    let hash = rustc_hash::FxBuildHasher.hash_one(&program);
878
879    if let Some(cached_program) = cached_program
880        && !force_compile
881        && cached_program.hash == hash
882    {
883        info!("Program matches existing artifact, returning early");
884        return Ok(cached_program);
885    }
886
887    let return_visibility = program.return_visibility();
888    let ssa_evaluator_options = options.as_ssa_options(context.package_build_path.clone());
889
890    let SsaProgramArtifact { program, debug, warnings, error_types, .. } = if options.minimal_ssa {
891        create_program_with_minimal_passes(program, &ssa_evaluator_options, &context.file_manager)?
892    } else {
893        create_program(
894            program,
895            &ssa_evaluator_options,
896            if options.with_ssa_locations { Some(&context.file_manager) } else { None },
897        )?
898    };
899
900    let abi = gen_abi(context, &main_function, return_visibility, error_types);
901    let file_map = filter_relevant_files(&debug, &context.file_manager);
902
903    Ok(CompiledProgram {
904        hash,
905        program,
906        debug,
907        abi,
908        file_map,
909        noir_version: NOIR_ARTIFACT_VERSION_STRING.to_string(),
910        warnings,
911    })
912}

940fn ssa_report_to_custom_diagnostic(error: SsaReport) -> CustomDiagnostic {
941    match error {
942        SsaReport::Warning(warning) => {
943            let message = warning.to_string();
944            let (secondary_message, call_stack) = match warning {
945                    InternalWarning::ReturnConstant { call_stack } => {
946                        ("This variable contains a value which is constrained to be a constant. Consider removing this value as additional return values increase proving/verification time".to_string(), call_stack)
947                    },
948                };
949            let call_stack = vecmap(call_stack, |location| location);
950            let location = call_stack.last().expect("Expected RuntimeError to have a location");
951            let diagnostic =
952                CustomDiagnostic::simple_warning(message, secondary_message, *location);
953            diagnostic.with_call_stack(call_stack)
954        }
955        SsaReport::Bug(bug) => {
956            let mut message = bug.to_string();
957            let (secondary_message, call_stack) = match bug {
958                    InternalBug::IndependentSubgraph { call_stack } => {
959                        ("There is no path from the output of this Brillig call to either return values or inputs of the circuit, which creates an independent subgraph. This is quite likely a soundness vulnerability".to_string(), call_stack)
960                    }
961                    InternalBug::UncheckedBrilligCall { call_stack } => {
962                        ("This Brillig call's inputs and its return values haven't been sufficiently constrained. This should be done to prevent potential soundness vulnerabilities".to_string(), call_stack)
963                    }
964                    InternalBug::AssertFailed { call_stack, message: assertion_failure_message } => {
965                        if let Some(assertion_failure_message) = assertion_failure_message {
966                            message.push_str(&format!(": {assertion_failure_message}"));
967                        }
968                        ("As a result, the compiled circuit is ensured to fail. Other assertions may also fail during execution".to_string(), call_stack)
969                    }
970                };
971            let call_stack = vecmap(call_stack, |location| location);
972            let location = call_stack.last().expect("Expected RuntimeError to have a location");
973            let diagnostic = CustomDiagnostic::simple_bug(message, secondary_message, *location);
974            diagnostic.with_call_stack(call_stack)
975        }
976    }
977}
