10/// Returns the expanded code for the given crate.
11/// This method calls out to the frontend's HIR display functionality and either formats it or adds a warning if there are syntax errors.
12pub fn get_expanded_crate(
13    crate_id: CrateId,
14    crate_graph: &CrateGraph,
15    def_maps: &DefMaps,
16    interner: &NodeInterner,
17) -> String {
18    let mut expanded_source = display_crate(crate_id, crate_graph, def_maps, interner);
19    let (parsed_module, errors) = parse_program_with_dummy_file(&expanded_source);
20    if errors.is_empty() {
21        let config = nargo_fmt::Config {
22            reorder_imports: true,
23            imports_granularity: ImportsGranularity::Crate,
24            ..Default::default()
25        };
26        nargo_fmt::format(&expanded_source, parsed_module, &config)
27    } else {
28        expanded_source.push_str("\n\n// Warning: the generated code has syntax errors");
29        expanded_source
30    }
31}
