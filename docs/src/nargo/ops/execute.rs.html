18struct ProgramExecutor<'a, F: AcirField, B: BlackBoxFunctionSolver<F>, E: ForeignCallExecutor<F>> {
19    functions: &'a [Circuit<F>],
20
21    unconstrained_functions: &'a [BrilligBytecode<F>],
22
23    // This gets built as we run through the program looking at each function call
24    witness_stack: WitnessStack<F>,
25
26    blackbox_solver: &'a B,
27
28    foreign_call_executor: &'a mut E,
29
30    // The Noir compiler codegens per function and call stacks are not shared across ACIR function calls.
31    // We must rebuild a call stack when executing a program of many circuits.
32    call_stack: Vec<ResolvedOpcodeLocation>,
33
34    // Tracks the index of the current function we are executing.
35    // This is used to fetch the function we want to execute
36    // and to resolve call stack locations across many function calls.
37    current_function_index: usize,
38
39    // Flag that states whether we want to profile the VM. Profiling can add extra
40    // execution costs so we want to make sure we only trigger it explicitly.
41    profiling_active: bool,
42
43    // Flag that states whether we want to trace brillig VM execution
44    brillig_fuzzing_active: bool,
45
46    // Brillig branch to feature map
47    brillig_branch_to_feature_map: Option<&'a BranchToFeatureMap>,
48
49    // Last recorded fuzzing trace
50    last_fuzzing_trace: Option<Vec<u32>>,
51
52    // Flag that states whether we want to return the witness on failure (useful for fuzzing)
53    return_witness_on_failure: bool,
54
55    // Partial witness on failure
56    failing_partial_witness: Option<WitnessMap<F>>,
57}

59impl<'a, F: AcirField, B: BlackBoxFunctionSolver<F>, E: ForeignCallExecutor<F>>
60    ProgramExecutor<'a, F, B, E>
61{
62    fn new(
63        functions: &'a [Circuit<F>],
64        unconstrained_functions: &'a [BrilligBytecode<F>],
65        blackbox_solver: &'a B,
66        foreign_call_executor: &'a mut E,
67        profiling_active: bool,
68    ) -> Self {
69        ProgramExecutor {
70            functions,
71            unconstrained_functions,
72            witness_stack: WitnessStack::default(),
73            blackbox_solver,
74            foreign_call_executor,
75            call_stack: Vec::default(),
76            current_function_index: 0,
77            profiling_active,
78            brillig_fuzzing_active: false,
79            brillig_branch_to_feature_map: None,
80            last_fuzzing_trace: None,
81            return_witness_on_failure: false,
82            failing_partial_witness: None,
83        }
84    }
85
86    fn with_brillig_fuzzing(
87        &mut self,
88        brillig_branch_to_feature_map: Option<&'a BranchToFeatureMap>,
89    ) {
90        self.brillig_fuzzing_active = brillig_branch_to_feature_map.is_some();
91        self.brillig_branch_to_feature_map = brillig_branch_to_feature_map;
92    }
93
94    fn with_partial_witness_on_failure(&mut self, return_witness_on_failure: bool) {
95        self.return_witness_on_failure = return_witness_on_failure;
96    }
97
98    fn finalize(self) -> WitnessStack<F> {
99        self.witness_stack
100    }

102    #[tracing::instrument(level = "trace", skip_all)]
103    fn execute_circuit(
104        &mut self,
105        initial_witness: WitnessMap<F>,
106    ) -> Result<(WitnessMap<F>, ProfilingSamples), NargoError<F>> {
107        let circuit = &self.functions[self.current_function_index];
108        let mut acvm = ACVM::new(
109            self.blackbox_solver,
110            &circuit.opcodes,
111            initial_witness,
112            self.unconstrained_functions,
113            &circuit.assert_messages,
114        );
115        acvm.with_profiler(self.profiling_active);
116        acvm.with_brillig_fuzzing(self.brillig_branch_to_feature_map);
117
118        loop {
119            let solver_status = acvm.solve();
120
121            match solver_status {
122                ACVMStatus::Solved => break,
123                ACVMStatus::InProgress => {
124                    unreachable!("Execution should not stop while in `InProgress` state.")
125                }
126                ACVMStatus::Failure(error) => {
127                    self.last_fuzzing_trace = acvm.get_brillig_fuzzing_trace();
128                    if self.return_witness_on_failure {
129                        self.failing_partial_witness = Some(acvm.witness_map().clone());
130                    }
131                    match &error {
132                        OpcodeResolutionError::UnsatisfiedConstrain {
133                            opcode_location: ErrorLocation::Resolved(opcode_location),
134                            ..
135                        }
136                        | OpcodeResolutionError::IndexOutOfBounds {
137                            opcode_location: ErrorLocation::Resolved(opcode_location),
138                            ..
139                        }
140                        | OpcodeResolutionError::InvalidInputBitSize {
141                            opcode_location: ErrorLocation::Resolved(opcode_location),
142                            ..
143                        } => {
144                            let resolved_location = ResolvedOpcodeLocation {
145                                acir_function_index: self.current_function_index,
146                                opcode_location: *opcode_location,
147                            };
148                            self.call_stack.push(resolved_location);
149                        }
150                        OpcodeResolutionError::BrilligFunctionFailed { call_stack, .. } => {
151                            let brillig_call_stack =
152                                call_stack.iter().map(|location| ResolvedOpcodeLocation {
153                                    acir_function_index: self.current_function_index,
154                                    opcode_location: *location,
155                                });
156                            self.call_stack.extend(brillig_call_stack);
157                        }
158                        _ => (),
159                    };
160
161                    return Err(NargoError::ExecutionError(execution_error_from(
162                        error,
163                        &self.call_stack,
164                    )));
165                }
166                ACVMStatus::RequiresForeignCall(foreign_call) => {
167                    match self.foreign_call_executor.execute(&foreign_call) {
168                        Ok(foreign_call_result) => {
169                            acvm.resolve_pending_foreign_call(foreign_call_result);
170                        }
171                        Err(error) => {
172                            if self.return_witness_on_failure {
173                                self.failing_partial_witness = Some(acvm.witness_map().clone());
174                            }
175                            return Err(NargoError::ForeignCallError(error));
176                        }
177                    }
178                }
179                ACVMStatus::RequiresAcirCall(call_info) => {
180                    // Store the parent function index whose context we are currently executing
181                    let acir_function_caller = self.current_function_index;
182                    // Add call opcode to the call stack with a reference to the parent function index
183                    self.call_stack.push(ResolvedOpcodeLocation {
184                        acir_function_index: acir_function_caller,
185                        opcode_location: OpcodeLocation::Acir(acvm.instruction_pointer()),
186                    });
187
188                    // Set current function to the circuit we are about to execute
189                    self.current_function_index = call_info.id.as_usize();
190                    // Execute the ACIR call
191                    let acir_to_call = &self.functions[call_info.id.as_usize()];
192                    let initial_witness = call_info.initial_witness;
193                    // TODO: Profiling among multiple circuits is not supported
194                    let (call_solved_witness, _) = self.execute_circuit(initial_witness)?;
195
196                    // Set tracking index back to the parent function after ACIR call execution
197                    self.current_function_index = acir_function_caller;
198
199                    let mut call_resolved_outputs = Vec::new();
200                    for return_witness_index in acir_to_call.return_values.indices() {
201                        if let Some(return_value) =
202                            call_solved_witness.get_index(return_witness_index)
203                        {
204                            call_resolved_outputs.push(*return_value);
205                        } else {
206                            return Err(ExecutionError::SolvingError(
207                                OpcodeNotSolvable::MissingAssignment(return_witness_index).into(),
208                                None, // Missing assignment errors do not supply user-facing diagnostics so we do not need to attach a call stack
209                            )
210                            .into());
211                        }
212                    }
213                    acvm.resolve_pending_acir_call(call_resolved_outputs);
214                    self.witness_stack.push(call_info.id.0, call_solved_witness);
215                }
216            }
217        }
218        // Clear the call stack if we have succeeded in executing the circuit.
219        // This needs to be done or else all successful ACIR call stacks will also be
220        // included in a failure case.
221        self.call_stack.clear();
222
223        let profiling_samples = acvm.take_profiling_samples();
224        self.last_fuzzing_trace = acvm.get_brillig_fuzzing_trace();
225        Ok((acvm.finalize(), profiling_samples))
226    }
227}

229/// Execute a [Program], returning the [WitnessStack].
230pub fn execute_program<F: AcirField, B: BlackBoxFunctionSolver<F>, E: ForeignCallExecutor<F>>(
231    program: &Program<F>,
232    initial_witness: WitnessMap<F>,
233    blackbox_solver: &B,
234    foreign_call_executor: &mut E,
235) -> Result<WitnessStack<F>, NargoError<F>> {
236    let profiling_active = false;
237    let (witness_stack, profiling_samples) = execute_program_inner(
238        program,
239        initial_witness,
240        blackbox_solver,
241        foreign_call_executor,
242        profiling_active,
243    )?;
244    assert!(profiling_samples.is_empty(), "Expected no profiling samples");
245
246    Ok(witness_stack)
247}

269pub(crate) fn execute_program_with_brillig_fuzzing<
270    F: AcirField,
271    B: BlackBoxFunctionSolver<F>,
272    E: ForeignCallExecutor<F>,
273>(
274    program: &Program<F>,
275    initial_witness: WitnessMap<F>,
276    blackbox_solver: &B,
277    foreign_call_executor: &mut E,
278    brillig_branch_to_feature_map: Option<&BranchToFeatureMap>,
279) -> Result<WitnessAndCoverage<F>, NargoErrorAndCoverage<F>> {
280    let mut executor = ProgramExecutor::new(
281        &program.functions,
282        &program.unconstrained_functions,
283        blackbox_solver,
284        foreign_call_executor,
285        false,
286    );
287    assert!(brillig_branch_to_feature_map.is_some());
288    executor.with_brillig_fuzzing(brillig_branch_to_feature_map);
289    match executor.execute_circuit(initial_witness) {
290        Ok((main_witness, _)) => {
291            executor.witness_stack.push(0, main_witness);
292            let trace = executor.last_fuzzing_trace.clone();
293            Ok((executor.finalize(), trace))
294        }
295        Err(err) => Err((err, executor.last_fuzzing_trace)),
296    }
297}

299/// Execute the program with ACIR fuzzing enabled (returns the witness on failure, so that the fuzzer can detect boolean states)
300pub(crate) fn execute_program_with_acir_fuzzing<
301    F: AcirField,
302    B: BlackBoxFunctionSolver<F>,
303    E: ForeignCallExecutor<F>,
304>(
305    program: &Program<F>,
306    initial_witness: WitnessMap<F>,
307    blackbox_solver: &B,
308    foreign_call_executor: &mut E,
309) -> Result<WitnessStack<F>, NargoErrorAndWitnessStack<F>> {
310    let mut executor = ProgramExecutor::new(
311        &program.functions,
312        &program.unconstrained_functions,
313        blackbox_solver,
314        foreign_call_executor,
315        false,
316    );
317    executor.with_partial_witness_on_failure(true);
318    match executor.execute_circuit(initial_witness) {
319        Ok((main_witness, _)) => {
320            executor.witness_stack.push(0, main_witness);
321            Ok(executor.finalize())
322        }
323        Err(err) => {
324            executor.witness_stack.push(
325                0,
326                executor
327                    .failing_partial_witness
328                    .as_ref()
329                    .expect("No partial witness on failure")
330                    .clone(),
331            );
332            Err((err, executor.finalize()))
333        }
334    }
335}

338fn execute_program_inner<F: AcirField, B: BlackBoxFunctionSolver<F>, E: ForeignCallExecutor<F>>(
339    program: &Program<F>,
340    initial_witness: WitnessMap<F>,
341    blackbox_solver: &B,
342    foreign_call_executor: &mut E,
343    profiling_active: bool,
344) -> Result<(WitnessStack<F>, ProfilingSamples), NargoError<F>> {
345    let mut executor = ProgramExecutor::new(
346        &program.functions,
347        &program.unconstrained_functions,
348        blackbox_solver,
349        foreign_call_executor,
350        profiling_active,
351    );
352    let (main_witness, profiling_samples) = executor.execute_circuit(initial_witness)?;
353    executor.witness_stack.push(0, main_witness);
354
355    Ok((executor.finalize(), profiling_samples))
356}
