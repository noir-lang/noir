20pub struct TestDefinition {
21    pub name: String,
22    pub function: TestFunction,
23}
24pub fn get_test_function_for_debug(
25    crate_id: CrateId,
26    context: &Context,
27    test_name: &str,
28) -> Result<TestDefinition, String> {
29    let test_pattern = FunctionNameMatch::Contains(vec![test_name.into()]);
30
31    let test_functions = context.get_all_test_functions_in_crate_matching(&crate_id, &test_pattern);
32
33    let (test_name, test_function) = match test_functions {
34        matchings if matchings.is_empty() => {
35            return Err(format!("`{test_name}` does not match with any test function"));
36        }
37        matchings if matchings.len() == 1 => matchings.into_iter().next().unwrap(),
38        matchings => {
39            let exact_match_op = matchings
40                .into_iter()
41                .filter(|(name, _)| name.split("::").last() == Some(test_name))
42                .collect::<Vec<(String, TestFunction)>>();
43            // There can be multiple matches but only one that matches exactly
44            // this would be the case of tests names that englobe others
45            // i.e.:
46            //  - test_something
47            //  - unconstrained_test_something
48            // in this case, looking up "test_something" throws two matchings
49            // but only one matches exact
50            if exact_match_op.len() == 1 {
51                exact_match_op.into_iter().next().unwrap()
52            } else {
53                return Err(format!("`{test_name}` matches with more than one test function"));
54            }
55        }
56    };
57
58    let test_function_has_arguments =
59        !context.def_interner.function_meta(&test_function.id).parameters.is_empty();
60
61    if test_function_has_arguments {
62        return Err(String::from("Cannot debug tests with arguments"));
63    }
64    Ok(TestDefinition { name: test_name, function: test_function })
65}

67pub fn compile_test_fn_for_debugging(
68    test_def: &TestDefinition,
69    context: &mut Context,
70    compile_options: CompileOptions,
71) -> Result<CompiledProgram, noirc_driver::CompileError> {
72    let compiled_program =
73        compile_no_check(context, &compile_options, test_def.function.id, None, false)?;
74    let compiled_program = optimize_program(compiled_program);
75    Ok(compiled_program)
76}

78pub fn compile_bin_package_for_debugging(
79    workspace: &Workspace,
80    package: &Package,
81    compile_options: &CompileOptions,
82) -> Result<CompiledProgram, CompileError> {
83    let (workspace_file_manager, mut parsed_files) = load_workspace_files(workspace);
84
85    let compilation_result = if compile_options.instrument_debug {
86        let debug_state =
87            instrument_package_files(&mut parsed_files, &workspace_file_manager, package);
88
89        compile_program_with_debug_instrumenter(
90            &workspace_file_manager,
91            &parsed_files,
92            workspace,
93            package,
94            compile_options,
95            None,
96            debug_state,
97        )
98    } else {
99        compile_program(
100            &workspace_file_manager,
101            &parsed_files,
102            workspace,
103            package,
104            compile_options,
105            None,
106        )
107    };
108
109    report_errors(
110        compilation_result,
111        &workspace_file_manager,
112        compile_options.deny_warnings,
113        compile_options.silence_warnings,
114    )
115    .map(optimize_program)
116}

118pub fn compile_options_for_debugging(
119    acir_mode: bool,
120    skip_instrumentation: bool,
121    compile_options: CompileOptions,
122) -> CompileOptions {
123    CompileOptions {
124        // Compilation warnings are disabled when
125        // compiling for debugging
126        //
127        // For instrumenting the program the debugger
128        // will import functions that may not be used,
129        // which would generate compilation warnings
130        silence_warnings: true,
131        deny_warnings: false,
132        instrument_debug: !skip_instrumentation,
133        force_brillig: !acir_mode,
134        ..compile_options
135    }
136}

138pub fn prepare_package_for_debug<'a>(
139    file_manager: &'a FileManager,
140    parsed_files: &'a mut ParsedFiles,
141    package: &'a Package,
142    workspace: &Workspace,
143) -> (Context<'a, 'a>, CrateId) {
144    let debug_instrumenter = instrument_package_files(parsed_files, file_manager, package);
145
146    // -- This :down: is from nargo::ops(compile).compile_program_with_debug_instrumenter
147    let (mut context, crate_id) = prepare_package(file_manager, parsed_files, package);
148    link_to_debug_crate(&mut context, crate_id);
149    context.debug_instrumenter = debug_instrumenter;
150    context.package_build_path = workspace.package_build_path(package);
151    (context, crate_id)
152}

154pub fn load_workspace_files(workspace: &Workspace) -> (FileManager, ParsedFiles) {
155    let mut file_manager = file_manager_with_stdlib(Path::new(""));
156    insert_all_files_for_workspace_into_file_manager(workspace, &mut file_manager);
157
158    let parsed_files = parse_all(&file_manager);
159    (file_manager, parsed_files)
160}

162/// Add debugging instrumentation to all parsed files belonging to the package
163/// being compiled
164fn instrument_package_files(
165    parsed_files: &mut ParsedFiles,
166    file_manager: &FileManager,
167    package: &Package,
168) -> DebugInstrumenter {
169    // Start off at the entry path and read all files in the parent directory.
170    let entry_path_parent = package
171        .entry_path
172        .parent()
173        .unwrap_or_else(|| panic!("The entry path is expected to be a single file within a directory and so should have a parent {}", package.entry_path.display()));
174
175    let mut debug_instrumenter = DebugInstrumenter::default();
176
177    for (file_id, parsed_file) in parsed_files.iter_mut() {
178        let file_path =
179            file_manager.path(*file_id).expect("Parsed file ID not found in file manager");
180        for ancestor in file_path.ancestors() {
181            if ancestor == entry_path_parent {
182                // file is in package
183                debug_instrumenter.instrument_module(&mut parsed_file.0, *file_id);
184            }
185        }
186    }
187
188    debug_instrumenter
189}
