35#[derive(Debug)]
36pub enum TestStatus {
37    Pass,
38    Fail { message: String, error_diagnostic: Option<CustomDiagnostic> },
39    Skipped,
40    CompileError(CustomDiagnostic),
41}
42
43impl TestStatus {
44    pub fn failed(&self) -> bool {
45        !matches!(self, TestStatus::Pass | TestStatus::Skipped)
46    }
47}

54/// Runs a test function. This will either run the test or fuzz it, depending on whether the function has arguments.
55#[allow(clippy::too_many_arguments)]
56pub fn run_or_fuzz_test<'a, W, B, F, E>(
57    blackbox_solver: &B,
58    context: &mut Context,
59    test_function: &TestFunction,
60    output: W,
61    package_name: String,
62    config: &CompileOptions,
63    fuzz_config: FuzzConfig,
64    build_foreign_call_executor: F,
65) -> TestStatus
66where
67    W: std::io::Write + 'a,
68    B: BlackBoxFunctionSolver<FieldElement> + Default,
69    F: Fn(Box<dyn std::io::Write + 'a>, layers::Unhandled) -> E + Sync,
70    E: ForeignCallExecutor<FieldElement>,
71{
72    if test_function.has_arguments {
73        fuzz_test::<B, F, E>(
74            context,
75            test_function,
76            package_name,
77            config,
78            fuzz_config,
79            build_foreign_call_executor,
80        )
81    } else {
82        run_test::<W, B, F, E>(
83            blackbox_solver,
84            context,
85            test_function,
86            output,
87            config,
88            build_foreign_call_executor,
89        )
90    }
91}

93/// Runs a test function. This assumes the function has no arguments.
94pub fn run_test<'a, W, B, F, E>(
95    blackbox_solver: &B,
96    context: &mut Context,
97    test_function: &TestFunction,
98    output: W,
99    config: &CompileOptions,
100    build_foreign_call_executor: F,
101) -> TestStatus
102where
103    W: std::io::Write + 'a,
104    B: BlackBoxFunctionSolver<FieldElement>,
105    F: Fn(Box<dyn std::io::Write + 'a>, layers::Unhandled) -> E,
106    E: ForeignCallExecutor<FieldElement>,
107{
108    match compile_no_check(context, config, test_function.id, None, false) {
109        Ok(compiled_program) => run_test_impl(
110            blackbox_solver,
111            compiled_program,
112            test_function,
113            output,
114            build_foreign_call_executor,
115        ),
116        Err(err) => test_status_program_compile_fail(err, test_function),
117    }
118}

120fn run_test_impl<'a, W, B, F, E>(
121    blackbox_solver: &B,
122    compiled_program: CompiledProgram,
123    test_function: &TestFunction,
124    output: W,
125    build_foreign_call_executor: F,
126) -> TestStatus
127where
128    W: std::io::Write + 'a,
129    B: BlackBoxFunctionSolver<FieldElement>,
130    F: Fn(Box<dyn std::io::Write + 'a>, layers::Unhandled) -> E,
131    E: ForeignCallExecutor<FieldElement>,
132{
133    // Do the same optimizations as `compile_cmd`.
134    let compiled_program = crate::ops::optimize_program(compiled_program);
135
136    let ignore_foreign_call_failures =
137        std::env::var("NARGO_IGNORE_TEST_FAILURES_FROM_FOREIGN_CALLS")
138            .is_ok_and(|var| &var == "true");
139
140    let writer: Box<dyn std::io::Write> = match std::env::var("NARGO_TEST_FOREIGN_CALL_LOG") {
141        Err(_) => Box::new(std::io::empty()),
142        Ok(s) if s == "stdout" => Box::new(std::io::stdout()),
143        Ok(s) => Box::new(
144            OpenOptions::new()
145                .create(true)
146                .truncate(true)
147                .write(true)
148                .open(PathBuf::from(s))
149                .unwrap(),
150        ),
151    };
152
153    // Run the backend to ensure the PWG evaluates functions like std::hash::pedersen,
154    // otherwise constraints involving these expressions will not error.
155    // Use a base layer that doesn't handle anything, which we handle in the `execute` below.
156    let foreign_call_executor = build_foreign_call_executor(Box::new(output), layers::Unhandled);
157    let foreign_call_executor = TestForeignCallExecutor::new(foreign_call_executor);
158    let mut foreign_call_executor = LoggingForeignCallExecutor::new(foreign_call_executor, writer);
159
160    let circuit_execution = execute_program(
161        &compiled_program.program,
162        WitnessMap::new(),
163        blackbox_solver,
164        &mut foreign_call_executor,
165    );
166
167    let status = test_status_program_compile_pass(
168        test_function,
169        &compiled_program.abi,
170        &compiled_program.debug,
171        &circuit_execution,
172    );
173
174    let foreign_call_executor = foreign_call_executor.executor;
175
176    if let TestStatus::Fail { .. } = status {
177        if ignore_foreign_call_failures && foreign_call_executor.encountered_unknown_foreign_call {
178            TestStatus::Skipped
179        } else {
180            status
181        }
182    } else {
183        status
184    }
185}

187/// Runs the fuzzer on a test function. This assumes the function has arguments.
188pub fn fuzz_test<'a, B, F, E>(
189    context: &mut Context,
190    test_function: &TestFunction,
191    package_name: String,
192    config: &CompileOptions,
193    fuzz_config: FuzzConfig,
194    build_foreign_call_executor: F,
195) -> TestStatus
196where
197    B: BlackBoxFunctionSolver<FieldElement> + Default,
198    F: Fn(Box<dyn std::io::Write + 'a>, layers::Unhandled) -> E + Sync,
199    E: ForeignCallExecutor<FieldElement>,
200{
201    match compile_no_check(context, config, test_function.id, None, false) {
202        Ok(_) => fuzz_test_impl::<B, F, E>(
203            context,
204            test_function,
205            package_name,
206            config,
207            fuzz_config,
208            build_foreign_call_executor,
209        ),
210        Err(err) => test_status_program_compile_fail(err, test_function),
211    }

214fn fuzz_test_impl<'a, B, F, E>(
215    context: &mut Context,
216    test_function: &TestFunction,
217    package_name: String,
218    config: &CompileOptions,
219    fuzz_config: FuzzConfig,
220    build_foreign_call_executor: F,
221) -> TestStatus
222where
223    B: BlackBoxFunctionSolver<FieldElement> + Default,
224    F: Fn(Box<dyn std::io::Write + 'a>, layers::Unhandled) -> E + Sync,
225    E: ForeignCallExecutor<FieldElement>,
226{
227    let id = test_function.id;
228    let scope = match &test_function.scope {
229        TestScope::ShouldFailWith { reason } => {
230            FuzzingScope::ShouldFailWith { reason: reason.clone() }
231        }
232        TestScope::OnlyFailWith { reason } => FuzzingScope::OnlyFailWith { reason: reason.clone() },
233        TestScope::None => FuzzingScope::None,
234    };
235    let location = test_function.location;
236    let fuzzing_harness = FuzzingHarness { id, scope, location };
237
238    let mut temporary_dirs_to_delete = Vec::new();
239
240    let mut config_or_temporary_dir = |dir: Option<String>| match dir {
241        Some(ref dir) => PathBuf::from(dir),
242        None => {
243            let corpus_dir = tempfile::tempdir().expect("Couldn't create temporary directory");
244            let corpus_dir = corpus_dir.keep();
245            temporary_dirs_to_delete.push(corpus_dir.clone());
246            corpus_dir
247        }
248    };
249
250    let corpus_dir = config_or_temporary_dir(fuzz_config.folder_config.corpus_dir);
251    let fuzzing_failure_dir =
252        config_or_temporary_dir(fuzz_config.folder_config.fuzzing_failure_dir);
253
254    let fuzz_folder_config = FuzzFolderConfig {
255        corpus_dir: Some(corpus_dir.to_string_lossy().to_string()),
256        fuzzing_failure_dir: Some(fuzzing_failure_dir.to_string_lossy().to_string()),
257        minimized_corpus_dir: fuzz_config.folder_config.minimized_corpus_dir,
258    };
259    let fuzz_execution_config = fuzz_config.execution_config;
260
261    // TODO: show output?
262    let show_output = false;
263    let fuzz_result = run_fuzzing_harness::<B, _, _>(
264        context,
265        &fuzzing_harness,
266        show_output,
267        package_name,
268        config,
269        &fuzz_folder_config,
270        &fuzz_execution_config,
271        build_foreign_call_executor,
272    );
273
274    for temporary_dir_to_delete in temporary_dirs_to_delete {
275        // Not a big deal if we can't delete a temporary directory
276        let _ = std::fs::remove_dir_all(temporary_dir_to_delete);
277    }
278
279    match fuzz_result {
280        FuzzingRunStatus::ExecutionPass | FuzzingRunStatus::MinimizationPass => TestStatus::Pass,
281        FuzzingRunStatus::CorpusFailure { message } => {
282            let message = format!("Corpus failure: {message}");
283            TestStatus::Fail { message, error_diagnostic: None }
284        }
285        FuzzingRunStatus::ExecutionFailure { message, counterexample, error_diagnostic } => {
286            let message = format!("Execution failed: {message}");
287            if let Some((input_map, abi)) = &counterexample {
288                let input =
289                    serialize_to_json(input_map, abi).expect("Couldn't serialize input to JSON");
290                let message = format!("{message}\nFailing input: {input}");
291                TestStatus::Fail { message, error_diagnostic }
292            } else {
293                TestStatus::Fail { message, error_diagnostic }
294            }
295        }
296        FuzzingRunStatus::MinimizationFailure { message } => {
297            let message = format!("Minimization failed: {message}");
298            TestStatus::Fail { message, error_diagnostic: None }
299        }
300        FuzzingRunStatus::ForeignCallFailure { message } => {
301            let message = format!("Foreign call failed: {message}");
302            TestStatus::Fail { message, error_diagnostic: None }
303        }
304        FuzzingRunStatus::CompileError(custom_diagnostic) => {
305            TestStatus::CompileError(custom_diagnostic)
306        }
307    }
308}
309/// Test function failed to compile
310///
311/// Note: This could be because the compiler was able to deduce
312/// that a constraint was never satisfiable.
313/// An example of this is the program `assert(false)`
314/// In that case, we check if the test function should fail, and if so, we return `TestStatus::Pass`.
315pub fn test_status_program_compile_fail(
316    err: CompileError,
317    test_function: &TestFunction,
318) -> TestStatus {
319    // The test has failed compilation, but it should never fail. Report error.
320    if !test_function.should_fail() {
321        return TestStatus::CompileError(err.into());
322    }
323
324    check_expected_failure_message(test_function, None, Some(err.into()))
325}

327/// The test function compiled successfully.
328///
329/// We now check whether execution passed/failed and whether it should have
330/// passed/failed to determine the test status.
331pub fn test_status_program_compile_pass(
332    test_function: &TestFunction,
333    abi: &Abi,
334    debug: &[DebugInfo],
335    circuit_execution: &Result<WitnessStack<FieldElement>, NargoError<FieldElement>>,
336) -> TestStatus {
337    let circuit_execution_err = match circuit_execution {
338        // Circuit execution was successful; ie no errors or unsatisfied constraints
339        // were encountered.
340        Ok(_) => {
341            if test_function.should_fail() {
342                return TestStatus::Fail {
343                    message: "error: Test passed when it should have failed".to_string(),
344                    error_diagnostic: None,
345                };
346            }
347            return TestStatus::Pass;
348        }
349        Err(err) => err,
350    };
351
352    // If we reach here, then the circuit execution failed.
353    //
354    // Check if the function should have passed
355    let diagnostic = try_to_diagnose_runtime_error(circuit_execution_err, abi, debug);
356    let test_should_have_passed = !test_function.should_fail();
357    if test_should_have_passed {
358        return TestStatus::Fail {
359            message: circuit_execution_err.to_string(),
360            error_diagnostic: diagnostic,
361        };
362    }
363
364    check_expected_failure_message(
365        test_function,
366        circuit_execution_err.user_defined_failure_message(&abi.error_types),
367        diagnostic,
368    )
369}

371pub fn check_expected_failure_message(
372    test_function: &TestFunction,
373    failed_assertion: Option<String>,
374    error_diagnostic: Option<CustomDiagnostic>,
375) -> TestStatus {
376    // Extract the expected failure message, if there was one
377    //
378    // #[test(should_fail)] will not produce any message
379    // #[test(should_fail_with = "reason")] will produce a message
380    //
381    let expected_failure_message = match test_function.failure_reason() {
382        Some(reason) => reason,
383        None => return TestStatus::Pass,
384    };
385
386    // Match the failure message that the user will see, i.e. the failed_assertion
387    // if present or else the error_diagnostic's message, against the
388    // expected_failure_message
389    let expected_failure_message_matches = failed_assertion
390        .as_ref()
391        .or_else(|| error_diagnostic.as_ref().map(|file_diagnostic| &file_diagnostic.message))
392        .map(|message| message.contains(expected_failure_message))
393        .unwrap_or(false);
394    if expected_failure_message_matches {
395        return TestStatus::Pass;
396    }
397
398    // The expected failure message does not match the actual failure message
399    TestStatus::Fail {
400        message: format!(
401            "\nerror: Test failed with the wrong message. \nExpected: {} \nGot: {}",
402            test_function.failure_reason().unwrap_or_default(),
403            failed_assertion.unwrap_or_default().trim_matches('\'')
404        ),
405        error_diagnostic,
406    }
407}

409/// A specialized foreign call executor which tracks whether it has encountered any unknown foreign calls
410pub(crate) struct TestForeignCallExecutor<E> {
411    executor: E,
412    encountered_unknown_foreign_call: bool,
413}
414
415impl<E> TestForeignCallExecutor<E> {
416    pub(crate) fn new(executor: E) -> Self {
417        Self { executor, encountered_unknown_foreign_call: false }
418    }
419}
420
421impl<E, F> ForeignCallExecutor<F> for TestForeignCallExecutor<E>
422where
423    F: AcirField,
424    E: ForeignCallExecutor<F>,
425{
426    fn execute(
427        &mut self,
428        foreign_call: &ForeignCallWaitInfo<F>,
429    ) -> Result<ForeignCallResult<F>, ForeignCallError> {
430        // If the circuit has reached a new foreign call opcode then it can't have failed from any previous unknown foreign calls.
431        self.encountered_unknown_foreign_call = false;
432        match self.executor.execute(foreign_call) {
433            Err(ForeignCallError::NoHandler(_)) => {
434                self.encountered_unknown_foreign_call = true;
435                // If the inner executor cannot handle this foreign call, then it's very likely that this is a custom
436                // foreign call. We then return an empty response in case the foreign call doesn't need return values.
437                layers::Empty.execute(foreign_call)
438            }
439            other => other,
440        }
441    }
