10/// This struct represents an oracle mock. It can be used for testing programs that use oracles.
11#[derive(Debug, PartialEq, Eq, Clone)]
12struct MockedCall<F> {
13    /// The id of the mock, used to update or remove it
14    id: usize,
15    /// The oracle it's mocking
16    name: String,
17    /// Optionally match the parameters
18    params: Option<Vec<ForeignCallParam<F>>>,
19    /// The parameters with which the mock was last called
20    last_called_params: Option<Vec<ForeignCallParam<F>>>,
21    /// The result to return when this mock is called
22    result: ForeignCallResult<F>,
23    /// How many times should this mock be called before it is removed
24    times_left: Option<u64>,
25    /// How many times this mock was actually called
26    times_called: u32,
27}
28
29impl<F> MockedCall<F> {
30    fn new(id: usize, name: String) -> Self {
31        Self {
32            id,
33            name,
34            params: None,
35            last_called_params: None,
36            result: ForeignCallResult { values: vec![] },
37            times_left: None,
38            times_called: 0,
39        }
40    }
41}
42
43impl<F: PartialEq> MockedCall<F> {
44    fn matches(&self, name: &str, params: &[ForeignCallParam<F>]) -> bool {
45        self.name == name && (self.params.is_none() || self.params.as_deref() == Some(params))
46    }
47}

49#[derive(Debug, Default)]
50pub struct MockForeignCallExecutor<F> {
51    /// Mocks have unique ids used to identify them in Noir, allowing to update or remove them.
52    last_mock_id: usize,
53    /// The registered mocks
54    mocked_responses: Vec<MockedCall<F>>,
55}
56
57impl<F: AcirField> MockForeignCallExecutor<F> {
58    fn extract_mock_id(
59        foreign_call_inputs: &[ForeignCallParam<F>],
60    ) -> Result<(usize, &[ForeignCallParam<F>]), ForeignCallError> {
61        let (id, params) =
62            foreign_call_inputs.split_first().ok_or(ForeignCallError::MissingForeignCallInputs)?;
63        let id =
64            usize::try_from(id.unwrap_field().try_to_u64().expect("value does not fit into u64"))
65                .expect("value does not fit into usize");
66        Ok((id, params))
67    }
68
69    fn find_mock_by_id(&self, id: usize) -> Option<&MockedCall<F>> {
70        self.mocked_responses.iter().find(|response| response.id == id)
71    }
72
73    fn find_mock_by_id_mut(&mut self, id: usize) -> Option<&mut MockedCall<F>> {
74        self.mocked_responses.iter_mut().find(|response| response.id == id)
75    }
76
77    fn parse_string(param: &ForeignCallParam<F>) -> String {
78        let fields: Vec<_> = param.fields().to_vec();
79        decode_string_value(&fields)
80    }
81}

83impl<F> ForeignCallExecutor<F> for MockForeignCallExecutor<F>
84where
85    F: AcirField,
86{
87    fn execute(
88        &mut self,
89        foreign_call: &ForeignCallWaitInfo<F>,
90    ) -> Result<ForeignCallResult<F>, ForeignCallError> {
91        let foreign_call_name = foreign_call.function.as_str();
92        match ForeignCall::lookup(foreign_call_name) {
93            Some(ForeignCall::CreateMock) => {
94                let mock_oracle_name = Self::parse_string(&foreign_call.inputs[0]);
95                assert!(ForeignCall::lookup(&mock_oracle_name).is_none());
96                let id = self.last_mock_id;
97                self.mocked_responses.push(MockedCall::new(id, mock_oracle_name));
98                self.last_mock_id += 1;
99
100                Ok(F::from(id).into())
101            }
102            Some(ForeignCall::SetMockParams) => {
103                let (id, params) = Self::extract_mock_id(&foreign_call.inputs)?;
104                self.find_mock_by_id_mut(id)
105                    .unwrap_or_else(|| panic!("Unknown mock id {id}"))
106                    .params = Some(params.to_vec());
107
108                Ok(ForeignCallResult::default())
109            }
110            Some(ForeignCall::GetMockLastParams) => {
111                let (id, _) = Self::extract_mock_id(&foreign_call.inputs)?;
112                let mock =
113                    self.find_mock_by_id(id).unwrap_or_else(|| panic!("Unknown mock id {id}"));
114
115                let last_called_params = mock
116                    .last_called_params
117                    .clone()
118                    .unwrap_or_else(|| panic!("Mock {} was never called", mock.name));
119
120                Ok(last_called_params.into())
121            }
122            Some(ForeignCall::SetMockReturns) => {
123                let (id, params) = Self::extract_mock_id(&foreign_call.inputs)?;
124                self.find_mock_by_id_mut(id)
125                    .unwrap_or_else(|| panic!("Unknown mock id {id}"))
126                    .result = ForeignCallResult { values: params.to_vec() };
127
128                Ok(ForeignCallResult::default())
129            }
130            Some(ForeignCall::SetMockTimes) => {
131                let (id, params) = Self::extract_mock_id(&foreign_call.inputs)?;
132                let times =
133                    params[0].unwrap_field().try_to_u64().expect("Invalid bit size of times");
134
135                self.find_mock_by_id_mut(id)
136                    .unwrap_or_else(|| panic!("Unknown mock id {id}"))
137                    .times_left = Some(times);
138
139                Ok(ForeignCallResult::default())
140            }
141            Some(ForeignCall::ClearMock) => {
142                let (id, _) = Self::extract_mock_id(&foreign_call.inputs)?;
143                self.mocked_responses.retain(|response| response.id != id);
144                Ok(ForeignCallResult::default())
145            }
146            Some(ForeignCall::GetTimesCalled) => {
147                let (id, _) = Self::extract_mock_id(&foreign_call.inputs)?;
148                let mock =
149                    self.find_mock_by_id(id).unwrap_or_else(|| panic!("Unknown mock id {id}"));
150                let times_called = mock.times_called;
151                Ok(ForeignCallResult::from(F::from(times_called)))
152            }
153            _ => {
154                let mock_response_position = self
155                    .mocked_responses
156                    .iter()
157                    .position(|response| response.matches(foreign_call_name, &foreign_call.inputs));
158
159                if let Some(response_position) = mock_response_position {
160                    // If the program has registered a mocked response to this oracle call then we prefer responding
161                    // with that.
162
163                    let mock = self
164                        .mocked_responses
165                        .get_mut(response_position)
166                        .expect("Invalid position of mocked response");
167
168                    mock.last_called_params = Some(foreign_call.inputs.clone());
169
170                    let result = mock.result.values.clone();
171
172                    // Update the total number of calls to the mock
173                    mock.times_called += 1;
174
175                    if let Some(times_left) = &mut mock.times_left {
176                        *times_left -= 1;
177                        if *times_left == 0 {
178                            self.mocked_responses.remove(response_position);
179                        }
180                    }
181
182                    Ok(result.into())
183                } else {
184                    Err(ForeignCallError::NoHandler(foreign_call_name.to_string()))
185                }
186            }
187        }
188    }
189}

191/// Handler that panics if any of the mock functions are called.
192pub struct DisabledMockForeignCallExecutor;
193
194impl<F> ForeignCallExecutor<F> for DisabledMockForeignCallExecutor {
195    fn execute(
196        &mut self,
197        foreign_call: &ForeignCallWaitInfo<F>,
198    ) -> Result<ForeignCallResult<F>, ForeignCallError> {
199        let foreign_call_name = foreign_call.function.as_str();
200        if let Some(
201            ForeignCall::CreateMock
202            | ForeignCall::SetMockParams
203            | ForeignCall::GetMockLastParams
204            | ForeignCall::SetMockReturns
205            | ForeignCall::SetMockTimes
206            | ForeignCall::ClearMock,
207        ) = ForeignCall::lookup(foreign_call_name)
208        {
209            // Returning an error instead of panicking so this can be tested.
210            return Err(ForeignCallError::Disabled(foreign_call.function.to_string()));
211        }
212        Err(ForeignCallError::NoHandler(foreign_call.function.clone()))
213    }
