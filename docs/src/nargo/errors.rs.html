22/// Errors covering situations where a package cannot be compiled.
23#[derive(Debug, Error)]
24pub enum CompileError {
25    #[error("Package `{0}` has type `lib` but only `bin` types can be compiled")]
26    LibraryCrate(CrateName),
27
28    #[error("Package `{0}` is expected to have a `main` function but it does not")]
29    MissingMainFunction(CrateName),
30
31    /// Errors encountered while compiling the Noir program.
32    /// These errors are already written to stderr.
33    #[error("Aborting due to {} previous error{}", .0.error_count, if .0.error_count == 1 { "" } else { "s" })]
34    ReportedErrors(ReportedErrors),
35}
36impl From<ReportedErrors> for CompileError {
37    fn from(errors: ReportedErrors) -> Self {
38        Self::ReportedErrors(errors)
39    }
40}

57impl<F: AcirField> NargoError<F> {
58    /// Extracts the user defined failure message from the ExecutionError
59    /// If one exists.
60    ///
61    /// We want to extract the user defined error so that we can compare it
62    /// in tests to expected failure messages
63    pub fn user_defined_failure_message(
64        &self,
65        error_types: &BTreeMap<ErrorSelector, AbiErrorType>,
66    ) -> Option<String> {
67        match self {
68            NargoError::ExecutionError(error) => match error {
69                ExecutionError::AssertionFailed(payload, _, _) => match payload {
70                    ResolvedAssertionPayload::String(message) => Some(message.to_string()),
71                    ResolvedAssertionPayload::Raw(raw) => {
72                        let abi_type = error_types.get(&raw.selector)?;
73                        let decoded = display_abi_error(&raw.data, abi_type.clone());
74                        Some(decoded.to_string())
75                    }
76                },
77                ExecutionError::SolvingError(error, _) => match error {
78                    OpcodeResolutionError::BlackBoxFunctionFailed(_, reason) => {
79                        Some(reason.to_string())
80                    }
81                    _ => None,
82                },
83            },
84            NargoError::ForeignCallError(error) => Some(error.to_string()),
85            _ => None,
86        }
87    }
88}

92/// and the index in the callers ACIR to the specific call opcode.

113/// Extracts the opcode locations from a nargo error.
114fn extract_locations_from_error<F: AcirField>(
115    error: &ExecutionError<F>,
116    debug: &[DebugInfo],
117) -> Option<Vec<Location>> {
118    let mut opcode_locations = match error {
119        ExecutionError::SolvingError(
120            OpcodeResolutionError::BrilligFunctionFailed { .. },
121            acir_call_stack,
122        ) => acir_call_stack.clone(),
123        ExecutionError::AssertionFailed(_, call_stack, _) => Some(call_stack.clone()),
124        ExecutionError::SolvingError(
125            OpcodeResolutionError::IndexOutOfBounds { opcode_location: error_location, .. },
126            acir_call_stack,
127        )
128        | ExecutionError::SolvingError(
129            OpcodeResolutionError::InvalidInputBitSize { opcode_location: error_location, .. },
130            acir_call_stack,
131        )
132        | ExecutionError::SolvingError(
133            OpcodeResolutionError::UnsatisfiedConstrain { opcode_location: error_location, .. },
134            acir_call_stack,
135        ) => match error_location {
136            ErrorLocation::Unresolved => {
137                unreachable!("Cannot resolve index for unsatisfied constraint")
138            }
139            ErrorLocation::Resolved(_) => acir_call_stack.clone(),
140        },
141        _ => None,
142    }?;
143
144    // Insert the top-level Acir location where the Brillig function failed
145    for (i, resolved_location) in opcode_locations.iter().enumerate() {
146        if let ResolvedOpcodeLocation {
147            acir_function_index,
148            opcode_location: OpcodeLocation::Brillig { acir_index, .. },
149        } = resolved_location
150        {
151            let acir_location = ResolvedOpcodeLocation {
152                acir_function_index: *acir_function_index,
153                opcode_location: OpcodeLocation::Acir(*acir_index),
154            };
155
156            opcode_locations.insert(i, acir_location);
157            // Go until the first brillig opcode as that means we have the start of a Brillig call stack.
158            // We have to loop through the opcode locations in case we had ACIR calls
159            // before the brillig function failure.
160            break;
161        }
162    }
163
164    let brillig_function_id = match error {
165        ExecutionError::SolvingError(
166            OpcodeResolutionError::BrilligFunctionFailed { function_id, .. },
167            _,
168        ) => Some(*function_id),
169        ExecutionError::AssertionFailed(_, _, function_id) => *function_id,
170        _ => None,
171    };
172
173    Some(
174        opcode_locations
175            .iter()
176            .flat_map(|resolved_location| {
177                let call_stack_id = match resolved_location.opcode_location {
178                    OpcodeLocation::Acir(idx) => *debug[resolved_location.acir_function_index]
179                        .acir_locations
180                        .get(&AcirOpcodeLocation::new(idx))
181                        .unwrap_or(&CallStackId::root()),
182                    OpcodeLocation::Brillig { brillig_index, .. } => *debug
183                        [resolved_location.acir_function_index]
184                        .brillig_locations[&brillig_function_id.unwrap()]
185                        .get(&BrilligOpcodeLocation(brillig_index))
186                        .unwrap_or(&CallStackId::root()),
187                };
188                debug[resolved_location.acir_function_index]
189                    .location_tree
190                    .get_call_stack(call_stack_id)
191            })
192            .collect(),
193    )
194}

196fn extract_message_from_error(
197    error_types: &BTreeMap<ErrorSelector, AbiErrorType>,
198    nargo_err: &NargoError<FieldElement>,
199) -> String {
200    match nargo_err {
201        NargoError::ExecutionError(ExecutionError::AssertionFailed(
202            ResolvedAssertionPayload::String(message),
203            _,
204            _,
205        )) => {
206            format!("Assertion failed: '{message}'")
207        }
208        NargoError::ExecutionError(ExecutionError::AssertionFailed(
209            ResolvedAssertionPayload::Raw(RawAssertionPayload { selector, data }),
210            ..,
211        )) => {
212            if let Some(error_type) = error_types.get(selector) {
213                format!("Assertion failed: {}", display_abi_error(data, error_type.clone()))
214            } else {
215                "Assertion failed".to_string()
216            }
217        }
218        NargoError::ExecutionError(ExecutionError::SolvingError(
219            OpcodeResolutionError::IndexOutOfBounds { index, array_size, .. },
220            _,
221        )) => {
222            format!("Index out of bounds, array has size {array_size:?}, but index was {index:?}")
223        }
224        NargoError::ExecutionError(ExecutionError::SolvingError(
225            OpcodeResolutionError::UnsatisfiedConstrain { .. },
226            _,
227        )) => "Failed constraint".into(),
228        _ => nargo_err.to_string(),
229    }
230}

232/// Tries to generate a runtime diagnostic from a nargo error. It will successfully do so if it's a runtime error with a call stack.
233pub fn try_to_diagnose_runtime_error(
234    nargo_err: &NargoError<FieldElement>,
235    abi: &Abi,
236    debug: &[DebugInfo],
237) -> Option<CustomDiagnostic> {
238    let source_locations = match nargo_err {
239        NargoError::ExecutionError(execution_error) => {
240            extract_locations_from_error(execution_error, debug)?
241        }
242        _ => return None,
243    };
244    // The location of the error itself will be the location at the top
245    // of the call stack (the last item in the Vec).
246    let location = *source_locations.last()?;
247    let message = extract_message_from_error(&abi.error_types, nargo_err);
248    let error = CustomDiagnostic::simple_error(message, String::new(), location);
249    Some(error.with_call_stack(source_locations))
250}

252/// Map the given OpcodeResolutionError to the corresponding ExecutionError
253/// In case of resulting in an ExecutionError::AssertionFailedThis it propagates the payload
254pub fn execution_error_from<F: AcirField>(
255    error: OpcodeResolutionError<F>,
256    call_stack: &[ResolvedOpcodeLocation],
257) -> ExecutionError<F> {
258    let (assertion_payload, brillig_function_id) = match &error {
259        OpcodeResolutionError::BrilligFunctionFailed { payload, function_id, .. } => {
260            (payload.clone(), Some(*function_id))
261        }
262        OpcodeResolutionError::UnsatisfiedConstrain { payload, .. } => (payload.clone(), None),
263        _ => (None, None),
264    };
265
266    match assertion_payload {
267        Some(payload) => {
268            ExecutionError::AssertionFailed(payload, call_stack.to_owned(), brillig_function_id)
269        }
270        None => {
271            let call_stack = match &error {
272                OpcodeResolutionError::UnsatisfiedConstrain { .. }
273                | OpcodeResolutionError::IndexOutOfBounds { .. }
274                | OpcodeResolutionError::InvalidInputBitSize { .. }
275                | OpcodeResolutionError::BrilligFunctionFailed { .. } => {
276                    Some(call_stack.to_owned())
277                }
278                _ => None,
279            };
280            ExecutionError::SolvingError(error, call_stack)
281        }
282    }
