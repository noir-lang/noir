41impl From<WitnessStack<FieldElement>> for JsWitnessStack {
42    fn from(mut witness_stack: WitnessStack<FieldElement>) -> Self {
43        let js_witness_stack = JsWitnessStack::new();
44        while let Some(stack_item) = witness_stack.pop() {
45            let js_map = JsWitnessMap::from(stack_item.witness);
46            let js_index = JsValue::from_f64(stack_item.index.into());
47
48            let entry_map = Map::new();
49            entry_map.set(&JsValue::from_str("index"), &js_index);
50            entry_map.set(&JsValue::from_str("witness"), &js_map);
51            let stack_item = Object::from_entries(&entry_map).unwrap();
52
53            js_witness_stack.push(&stack_item);
54        }
55        // `reverse()` returns an `Array` so we have to wrap it
56        JsWitnessStack { obj: js_witness_stack.reverse() }
57    }
58}

60impl From<JsWitnessStack> for WitnessStack<FieldElement> {
61    fn from(js_witness_stack: JsWitnessStack) -> Self {
62        let mut witness_stack = WitnessStack::default();
63        js_witness_stack.for_each(&mut |stack_item, _, _| {
64            let values_array = Object::values(&Object::from(stack_item));
65            let index = values_array.get(0).as_f64().unwrap() as u32;
66            let js_witness_map: JsWitnessMap = values_array.get(1).into();
67            witness_stack.push(index, js_witness_map.into());
68        });
69        witness_stack
70    }
