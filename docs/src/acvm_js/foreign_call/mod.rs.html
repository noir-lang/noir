9#[wasm_bindgen(typescript_custom_section)]
10const FOREIGN_CALL_HANDLER: &'static str = r#"
11export type ForeignCallInput = string[]
12export type ForeignCallOutput = string | string[]
13
14/**
15* A callback which performs an foreign call and returns the response.
16* @callback ForeignCallHandler
17* @param {string} name - The identifier for the type of foreign call being performed.
18* @param {string[][]} inputs - An array of hex encoded inputs to the foreign call.
19* @returns {Promise<string[]>} outputs - An array of hex encoded outputs containing the results of the foreign call.
20*/
21export type ForeignCallHandler = (name: string, inputs: ForeignCallInput[]) => Promise<ForeignCallOutput[]>;
22"#;
23
24#[wasm_bindgen]
25extern "C" {
26    #[wasm_bindgen(extends = js_sys::Function, typescript_type = "ForeignCallHandler")]
27    pub type ForeignCallHandler;
28}

30pub(super) async fn resolve_brillig(
31    foreign_call_callback: &ForeignCallHandler,
32    foreign_call_wait_info: &ForeignCallWaitInfo<FieldElement>,
33) -> Result<ForeignCallResult<FieldElement>, Error> {
34    // Prepare to call
35    let name = JsString::from(foreign_call_wait_info.function.clone());
36    let inputs = inputs::encode_foreign_call_inputs(&foreign_call_wait_info.inputs);
37
38    // Perform foreign call
39    let outputs = perform_foreign_call(foreign_call_callback, name, inputs).await?;
40
41    // The Brillig VM checks that the number of return values from
42    // the foreign call is valid so we don't need to do it here.
43    outputs::decode_foreign_call_result(outputs).map_err(|message| Error::new(&message))
44}
45
46async fn perform_foreign_call(
47    foreign_call_handler: &ForeignCallHandler,
48    name: JsString,
49    inputs: js_sys::Array,
50) -> Result<js_sys::Array, Error> {
51    // Call and await
52    let this = JsValue::null();
53    let ret_js_val = foreign_call_handler
54        .call2(&this, &name, &inputs)
55        .map_err(|err| wrap_js_error("Error calling `foreign_call_callback`", &err))?;
56    let ret_js_prom: js_sys::Promise = ret_js_val.into();
57    let ret_future: wasm_bindgen_futures::JsFuture = ret_js_prom.into();
58    let js_resolution = ret_future
59        .await
60        .map_err(|err| wrap_js_error("Error awaiting `foreign_call_handler`", &err))?;
61
62    // Check that result conforms to expected shape.
63    if !js_resolution.is_array() {
64        return Err(Error::new("Expected `foreign_call_handler` to return an array"));
65    }
66
67    Ok(js_sys::Array::from(&js_resolution))
68}
69
70fn wrap_js_error(message: &str, err: &JsValue) -> Error {
71    let new_error = Error::new(message);
72    new_error.set_cause(err);
73    new_error
