22/// Executes an ACIR circuit to generate the solved witness from the initial witness.
23///
24/// @param {Uint8Array} circuit - A serialized representation of an ACIR circuit
25/// @param {WitnessMap} initial_witness - The initial witness map defining all of the inputs to `circuit`..
26/// @param {ForeignCallHandler} foreign_call_handler - A callback to process any foreign calls from the circuit.
27/// @returns {WitnessMap} The solved witness calculated by executing the circuit on the provided inputs.
28#[wasm_bindgen(js_name = executeCircuit, skip_jsdoc)]
29pub async fn execute_circuit(
30    program: Vec<u8>,
31    initial_witness: JsWitnessMap,
32    foreign_call_handler: ForeignCallHandler,
33) -> Result<JsWitnessMap, Error> {
34    execute_circuit_pedantic(program, initial_witness, foreign_call_handler).await
35}

37/// `execute_circuit` with pedantic ACVM solving
38async fn execute_circuit_pedantic(
39    program: Vec<u8>,
40    initial_witness: JsWitnessMap,
41    foreign_call_handler: ForeignCallHandler,
42) -> Result<JsWitnessMap, Error> {
43    console_error_panic_hook::set_once();
44
45    let mut witness_stack =
46        execute_program_with_native_type_return(program, initial_witness, &foreign_call_handler)
47            .await?;
48    let witness_map =
49        witness_stack.pop().expect("Should have at least one witness on the stack").witness;
50    Ok(witness_map.into())
51}

53/// Executes an ACIR circuit to generate the solved witness from the initial witness.

63    initial_witness: JsWitnessMap,
64    foreign_call_handler: ForeignCallHandler,
65) -> Result<JsSolvedAndReturnWitness, Error> {
66    execute_circuit_with_return_witness_pedantic(program, initial_witness, foreign_call_handler)
67        .await

70/// `executeCircuitWithReturnWitness` with pedantic ACVM execution
71async fn execute_circuit_with_return_witness_pedantic(
72    program: Vec<u8>,
73    initial_witness: JsWitnessMap,
74    foreign_call_handler: ForeignCallHandler,
75) -> Result<JsSolvedAndReturnWitness, Error> {
76    console_error_panic_hook::set_once();
77
78    let program: Program<FieldElement> = Program::deserialize_program(&program)
79    .map_err(|_| JsExecutionError::new("Failed to deserialize circuit. This is likely due to differing serialization formats between ACVM_JS and your compiler".to_string(), None, None, None, None))?;
80
81    let mut witness_stack = execute_program_with_native_program_and_return(
82        &program,
83        initial_witness,
84        &foreign_call_handler,
85    )
86    .await?;
87    let solved_witness =
88        witness_stack.pop().expect("Should have at least one witness on the stack").witness;
89
90    let main_circuit = &program.functions[0];
91    let return_witness =
92        extract_indices(&solved_witness, main_circuit.return_values.0.iter().copied().collect())
93            .map_err(|err| JsExecutionError::new(err, None, None, None, None))?;
94
95    Ok((solved_witness, return_witness).into())
96}

144async fn execute_program_with_native_program_and_return(
145    program: &Program<FieldElement>,
146    initial_witness: JsWitnessMap,
147    foreign_call_executor: &ForeignCallHandler,
148) -> Result<WitnessStack<FieldElement>, Error> {
149    let blackbox_solver = Bn254BlackBoxSolver;
150    let executor = ProgramExecutor::new(
151        &program.functions,
152        &program.unconstrained_functions,
153        &blackbox_solver,
154        foreign_call_executor,
155    );
156    let witness_stack = executor.execute(initial_witness.into()).await?;
157
158    Ok(witness_stack)
159}

161struct ProgramExecutor<'a, B: BlackBoxFunctionSolver<FieldElement>> {
162    functions: &'a [Circuit<FieldElement>],
163
164    unconstrained_functions: &'a [BrilligBytecode<FieldElement>],
165
166    blackbox_solver: &'a B,
167
168    foreign_call_handler: &'a ForeignCallHandler,
169}
170
171impl<'a, B: BlackBoxFunctionSolver<FieldElement>> ProgramExecutor<'a, B> {
172    fn new(
173        functions: &'a [Circuit<FieldElement>],
174        unconstrained_functions: &'a [BrilligBytecode<FieldElement>],
175        blackbox_solver: &'a B,
176        foreign_call_handler: &'a ForeignCallHandler,
177    ) -> Self {
178        ProgramExecutor {
179            functions,
180            unconstrained_functions,
181            blackbox_solver,
182            foreign_call_handler,
183        }
184    }
185
186    async fn execute(
187        &self,
188        initial_witness: WitnessMap<FieldElement>,
189    ) -> Result<WitnessStack<FieldElement>, Error> {
190        let main = &self.functions[0];
191
192        let mut witness_stack = WitnessStack::default();
193        let main_witness = self
194            .execute_circuit(main, AcirFunctionId(0), initial_witness, &mut witness_stack)
195            .await?;
196        witness_stack.push(0, main_witness);
197        Ok(witness_stack)
198    }
199
200    fn execute_circuit(
201        &'a self,
202        circuit: &'a Circuit<FieldElement>,
203        acir_function_id: AcirFunctionId,
204        initial_witness: WitnessMap<FieldElement>,
205        witness_stack: &'a mut WitnessStack<FieldElement>,
206    ) -> Pin<Box<dyn Future<Output = Result<WitnessMap<FieldElement>, Error>> + 'a>> {
207        Box::pin(async move {
208            let mut acvm = ACVM::new(
209                self.blackbox_solver,
210                &circuit.opcodes,
211                initial_witness,
212                self.unconstrained_functions,
213                &circuit.assert_messages,
214            );
215
216            loop {
217                let solver_status = acvm.solve();
218
219                match solver_status {
220                    ACVMStatus::Solved => break,
221                    ACVMStatus::InProgress => {
222                        unreachable!("Execution should not stop while in `InProgress` state.")
223                    }
224                    ACVMStatus::Failure(error) => {
225                        // Fetch call stack
226                        let call_stack = match &error {
227                            OpcodeResolutionError::UnsatisfiedConstrain {
228                                opcode_location: ErrorLocation::Resolved(opcode_location),
229                                ..
230                            }
231                            | OpcodeResolutionError::IndexOutOfBounds {
232                                opcode_location: ErrorLocation::Resolved(opcode_location),
233                                ..
234                            } => Some(vec![*opcode_location]),
235                            OpcodeResolutionError::InvalidInputBitSize {
236                                opcode_location: ErrorLocation::Resolved(opcode_location),
237                                ..
238                            } => Some(vec![*opcode_location]),
239                            OpcodeResolutionError::BrilligFunctionFailed { call_stack, .. } => {
240                                Some(call_stack.clone())
241                            }
242                            _ => None,
243                        };
244
245                        let brillig_function_id = match &error {
246                            OpcodeResolutionError::BrilligFunctionFailed {
247                                function_id, ..
248                            } => Some(*function_id),
249                            _ => None,
250                        };
251
252                        // If the failed opcode has an assertion message, integrate it into the error message for backwards compatibility.
253                        // Otherwise, pass the raw assertion payload as is.
254                        let (message, raw_assertion_payload) = match error {
255                            OpcodeResolutionError::UnsatisfiedConstrain {
256                                payload: Some(payload),
257                                ..
258                            }
259                            | OpcodeResolutionError::BrilligFunctionFailed {
260                                payload: Some(payload),
261                                ..
262                            } => match payload {
263                                ResolvedAssertionPayload::Raw(raw_payload) => {
264                                    ("Assertion failed".to_string(), Some(raw_payload))
265                                }
266                                ResolvedAssertionPayload::String(message) => {
267                                    (format!("Assertion failed: {message}"), None)
268                                }
269                            },
270                            _ => (error.to_string(), None),
271                        };
272
273                        return Err(JsExecutionError::new(
274                            message,
275                            call_stack,
276                            raw_assertion_payload,
277                            Some(acir_function_id),
278                            brillig_function_id,
279                        )
280                        .into());
281                    }
282                    ACVMStatus::RequiresForeignCall(foreign_call) => {
283                        let result =
284                            resolve_brillig(self.foreign_call_handler, &foreign_call).await?;
285
286                        acvm.resolve_pending_foreign_call(result);
287                    }
288                    ACVMStatus::RequiresAcirCall(call_info) => {
289                        let acir_to_call = &self.functions[call_info.id.as_usize()];
290                        let initial_witness = call_info.initial_witness;
291                        let call_solved_witness = self
292                            .execute_circuit(
293                                acir_to_call,
294                                call_info.id,
295                                initial_witness,
296                                witness_stack,
297                            )
298                            .await?;
299                        let mut call_resolved_outputs = Vec::new();
300                        for return_witness_index in acir_to_call.return_values.indices() {
301                            if let Some(return_value) =
302                                call_solved_witness.get_index(return_witness_index)
303                            {
304                                call_resolved_outputs.push(*return_value);
305                            } else {
306                                // TODO: look at changing this call stack from None
307                                return Err(JsExecutionError::new(format!("Failed to read from solved witness of ACIR call at witness {return_witness_index}"), None, None, None, None).into());
308                            }
309                        }
310                        acvm.resolve_pending_acir_call(call_resolved_outputs);
311                        witness_stack.push(call_info.id.0, call_solved_witness.clone());
312                    }
313                }
314            }
315
316            Ok(acvm.finalize())
317        })
318    }
