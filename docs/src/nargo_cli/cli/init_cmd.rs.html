10#[allow(rustdoc::broken_intra_doc_links)]
11/// Create a Noir project in the current directory.
12#[derive(Debug, Clone, Args)]
13pub(crate) struct InitCommand {
14    /// Name of the package [default: current directory name]
15    #[clap(long)]
16    name: Option<CrateName>,
17
18    /// Use a library template
19    #[arg(long, conflicts_with = "bin", conflicts_with = "contract")]
20    pub(crate) lib: bool,
21
22    /// Use a binary template [default]
23    #[arg(long, conflicts_with = "lib", conflicts_with = "contract")]
24    pub(crate) bin: bool,
25
26    /// Use a contract template
27    #[arg(long, conflicts_with = "lib", conflicts_with = "bin")]
28    pub(crate) contract: bool,
29}
30
31const BIN_EXAMPLE: &str = include_str!("./noir_template_files/binary.nr");
32const CONTRACT_EXAMPLE: &str = include_str!("./noir_template_files/contract.nr");
33const LIB_EXAMPLE: &str = include_str!("./noir_template_files/library.nr");
34
35pub(crate) fn run(args: InitCommand, config: NargoConfig) -> Result<(), CliError> {
36    let package_name = match args.name {
37        Some(name) => name,
38        None => {
39            let name = config.program_dir.file_name().unwrap().to_str().unwrap();
40            name.parse().map_err(CliError::InvalidPackageName)?
41        }
42    };
43
44    let package_type = if args.lib {
45        PackageType::Library
46    } else if args.contract {
47        PackageType::Contract
48    } else {
49        PackageType::Binary
50    };
51    initialize_project(config.program_dir, package_name, package_type)
52}
53
54/// Initializes a new Noir project in `package_dir`.
55pub(crate) fn initialize_project(
56    package_dir: PathBuf,
57    package_name: CrateName,
58    package_type: PackageType,
59) -> Result<(), CliError> {
60    let src_dir = package_dir.join(SRC_DIR);
61    let toml_path = package_dir.join(PKG_FILE);
62
63    // We don't want to accidentally overwrite an existing Nargo.toml file
64    if toml_path.exists() {
65        return Err(CliError::NargoInitCannotBeRunOnExistingPackages);
66    }
67
68    let toml_contents = format!(
69        r#"[package]
70name = "{package_name}"
71type = "{package_type}"
72authors = [""]
73
74[dependencies]"#
75    );
76
77    write_to_file(toml_contents.as_bytes(), &toml_path).unwrap();
78
79    let (example, entry_name) = match package_type {
80        PackageType::Binary => (BIN_EXAMPLE, "main.nr"),
81        PackageType::Contract => (CONTRACT_EXAMPLE, "main.nr"),
82        PackageType::Library => (LIB_EXAMPLE, "lib.nr"),
83    };
84    let entry_path = src_dir.join(entry_name);
85
86    // It's fine to run `nargo init` if a source directory exists:
87    // it might be that the source is already there and the user just wants to create the Nargo.toml file.
88    if !entry_path.exists() {
89        write_to_file(example.as_bytes(), &entry_path).unwrap();
90    }
91
92    println!("Project successfully created! It is located at {}", package_dir.display());
93
94    Ok(())
95}
