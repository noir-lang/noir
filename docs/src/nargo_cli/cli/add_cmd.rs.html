17/// URL of a git repository for a dependency, validated as it is parsed from the command line.
18///
19/// Only `https` URLs are accepted. A repository's web page offers an HTTPS and an SSH URL to copy;
20/// the SSH one is the scp-style `git@github.com:owner/repo.git`, which is not a URL and cannot be
21/// cached by host and path, so it is rejected with guidance to use the HTTPS form. `http` is
22/// rejected so credentials are never sent unencrypted.
23#[derive(Debug, Clone)]
24pub(crate) struct GitUrl(String);
25
26impl GitUrl {
27    fn as_str(&self) -> &str {
28        &self.0
29    }
30}
31
32#[derive(Debug, Error)]
33pub(crate) enum GitUrlParseError {
34    #[error(
35        "`{input}` is not a valid git URL: {source}.\n\
36         Use the HTTPS URL from the repository's web page, e.g. `https://github.com/owner/repo`.\n\
37         SSH URLs like `git@github.com:owner/repo.git` are not supported."
38    )]
39    InvalidUrl { input: String, source: url::ParseError },
40    #[error(
41        "git dependencies must use an `https` URL, but `{input}` uses `{scheme}`.\n\
42         Use the HTTPS URL from the repository's web page, e.g. `https://github.com/owner/repo`."
43    )]
44    UnsupportedScheme { input: String, scheme: String },
45}
46
47impl FromStr for GitUrl {
48    type Err = GitUrlParseError;
49
50    fn from_str(s: &str) -> Result<Self, Self::Err> {
51        let url = Url::parse(s)
52            .map_err(|source| GitUrlParseError::InvalidUrl { input: s.to_string(), source })?;
53        // `https` is a special scheme, so a successfully parsed URL is guaranteed to have a host
54        // (a domain or an IP for a self-hosted server), which is all the cache path needs.
55        if url.scheme() != "https" {
56            return Err(GitUrlParseError::UnsupportedScheme {
57                input: s.to_string(),
58                scheme: url.scheme().to_string(),
59            });
60        }
61        Ok(GitUrl(s.to_string()))
62    }
63}

65/// Add a dependency to the package's `Nargo.toml`.
66///
67/// The dependency is resolved (and, for git dependencies, downloaded) before the manifest is
68/// written, so a dependency that cannot be found never ends up in `Nargo.toml`.
69#[derive(Debug, Clone, Args)]
70#[clap(group(ArgGroup::new("source").required(true).args(["path", "git"])))]
71pub(crate) struct AddCommand {
72    /// Name to use for the dependency (the key under `[dependencies]`).
73    /// Defaults to the package name declared in the dependency's own `Nargo.toml`.
74    name: Option<CrateName>,
75
76    /// Path to a local dependency, relative to this package's `Nargo.toml`.
77    #[clap(long, conflicts_with = "git")]
78    path: Option<String>,
79
80    /// URL of a git repository to depend on.
81    #[clap(long, conflicts_with = "path", requires = "tag")]
82    git: Option<GitUrl>,
83
84    /// Git branch or tag to use (required with `--git`).
85    #[clap(long, requires = "git")]
86    tag: Option<String>,
87
88    /// Subdirectory within the git repository that contains the package.
89    #[clap(long, requires = "git")]
90    directory: Option<String>,
91
92    /// Replace the dependency if one with the same name already exists.
93    #[clap(long = "override")]
94    overwrite: bool,
95
96    #[clap(flatten)]
97    package_options: PackageOptions,
98}
99
100impl WorkspaceCommand for AddCommand {
101    fn package_selection(&self) -> PackageSelection {
102        self.package_options.package_selection()
103    }
104
105    fn lock_type(&self) -> LockType {
106        // We edit a workspace member's `Nargo.toml`. `lock_workspace` takes its lock on the
107        // manifest file itself, so an exclusive lock serializes concurrent `add` runs.
108        LockType::Exclusive
109    }
110}

112pub(crate) fn run(args: AddCommand, workspace: Workspace) -> Result<(), CliError> {
113    // Adding to several manifests at once is ambiguous, so we require a single selected package.
114    // A workspace root without a `default-member` (or `--workspace`) leaves no package selected.
115    let index = workspace.selected_package_index.ok_or(CliError::AddRequiresPackageSelection)?;
116    let package = &workspace.members[index];
117    let manifest_path = package.root_dir.join(PKG_FILE);
118
119    let dependency = match (&args.path, &args.git) {
120        (Some(path), _) => DependencyConfig::Path { path: path.clone() },
121        (None, Some(git)) => DependencyConfig::Git {
122            git: git.as_str().to_string(),
123            // `clap` guarantees `--tag` is present whenever `--git` is.
124            tag: args.tag.clone().expect("--git requires --tag"),
125            directory: args.directory.clone(),
126        },
127        // `clap`'s required `source` group guarantees exactly one of `--path`/`--git`.
128        (None, None) => unreachable!("clap requires either --path or --git"),
129    };
130
131    // Resolve the dependency before touching the manifest: this validates that it exists and is
132    // not a binary, downloads git dependencies into the cache, and tells us where it landed.
133    let resolved = resolve_dependency(&package.root_dir, &dependency)?;
134
135    let name = match args.name {
136        Some(name) => name,
137        None => resolved.package_name().clone(),
138    };
139
140    add_dependency_to_manifest(&manifest_path, &name.to_string(), &dependency, args.overwrite)?;
141
142    let location = resolved.package().root_dir.display();
143    noirc_errors::println_to_stdout!("Added dependency `{name}`");
144    match &dependency {
145        DependencyConfig::Git { git, tag, .. } => {
146            noirc_errors::println_to_stdout!("    source: {git} (tag {tag})");
147        }
148        DependencyConfig::Path { path } => {
149            noirc_errors::println_to_stdout!("    source: {path}");
150        }
151    }
152    noirc_errors::println_to_stdout!("  location: {location}");
153
154    Ok(())
155}

165    /// Writes a package `Nargo.toml` and its entry file under `root/dir`.
166    fn write_pkg(root: &Path, dir: &str, name: &str, package_type: &str) {
167        let pkg_dir = root.join(dir);
168        std::fs::create_dir_all(pkg_dir.join("src")).unwrap();
169        std::fs::write(
170            pkg_dir.join(PKG_FILE),
171            format!("[package]\nname = \"{name}\"\ntype = \"{package_type}\"\nauthors = [\"\"]\n\n[dependencies]\n"),
172        )
173        .unwrap();
174        let entry = if package_type == "lib" { "lib.nr" } else { "main.nr" };
175        std::fs::write(pkg_dir.join("src").join(entry), "").unwrap();
176    }
177
178    /// Resolves the package at `root/dir` into a single-package workspace.
179    fn workspace_for(root: &Path, dir: &str) -> Workspace {
180        resolve_workspace_from_toml(
181            &root.join(dir).join(PKG_FILE),
182            PackageSelection::DefaultOrAll,
183            None,
184        )
185        .unwrap()
186    }
187
188    fn add_command(path: &str, overwrite: bool) -> AddCommand {
189        AddCommand {
190            name: None,
191            path: Some(path.to_string()),
192            git: None,
193            tag: None,
194            directory: None,
195            overwrite,
196            package_options: PackageOptions::default(),
197        }
198    }

200    #[test]
201    fn add_path_dependency_infers_name_and_guards_existing() {
202        let tmp = tempfile::tempdir().unwrap();
203        let root = tmp.path();
204        write_pkg(root, "the_bin", "demo", "bin");
205        write_pkg(root, "the_lib", "cool_lib", "lib");
206
207        let manifest = root.join("the_bin").join(PKG_FILE);
208
209        // First add: the name is inferred from the dependency's own `Nargo.toml`.
210        run(add_command("../the_lib", false), workspace_for(root, "the_bin")).unwrap();
211        let contents = std::fs::read_to_string(&manifest).unwrap();
212        assert!(
213            contents.contains(r#"cool_lib = { path = "../the_lib" }"#),
214            "expected the inferred dependency entry, got:\n{contents}"
215        );
216
217        // Re-adding the same dependency without `--override` is rejected.
218        let err = run(add_command("../the_lib", false), workspace_for(root, "the_bin"))
219            .expect_err("re-adding an existing dependency should fail");
220        assert!(matches!(
221            err,
222            CliError::ManifestError(nargo_toml::ManifestError::DependencyAlreadyExists(_))
223        ));
224
225        // With `--override` it succeeds.
226        run(add_command("../the_lib", true), workspace_for(root, "the_bin")).unwrap();
227    }
