20/// Provides detailed information on each of a program's function (represented by a single circuit)
21///
22/// Current information provided per circuit:
23/// 1. The number of ACIR opcodes
24/// 2. Counts the final number gates in the circuit used by a backend
25#[derive(Debug, Clone, Args)]
26#[clap(visible_alias = "i")]
27pub(crate) struct InfoCommand {
28    #[clap(flatten)]
29    pub(super) package_options: PackageOptions,
30
31    /// Output a JSON formatted report. Changes to this format are not currently considered breaking.
32    #[clap(long, hide = true)]
33    json: bool,
34
35    #[clap(long)]
36    profile_execution: bool,
37
38    /// The name of the toml file which contains the inputs for the prover
39    #[clap(long, short, default_value = PROVER_INPUT_FILE)]
40    prover_name: String,
41
42    #[clap(flatten)]
43    compile_options: CompileOptions,
44}
45

56pub(crate) fn run(mut args: InfoCommand, workspace: Workspace) -> Result<(), CliError> {
57    if args.json {
58        // In order to have a machine readable output, we cannot allow the program to print arbitrary data to stdout.
59        args.compile_options.disable_comptime_printing = true;
60    }
61
62    if args.profile_execution {
63        // Execution profiling is only relevant with the Brillig VM
64        // as a constrained circuit should have totally flattened control flow (e.g. loops and if statements).
65        args.compile_options.force_brillig = true;
66    }
67    // Compile the full workspace in order to generate any build artifacts.
68    let debug_compile_stdin = None;
69    compile_workspace_full(&workspace, &args.compile_options, debug_compile_stdin)?;
70
71    let binary_packages: Vec<(Package, ProgramArtifact)> = workspace
72        .into_iter()
73        .filter(|package| package.is_binary())
74        .map(|package| -> Result<(Package, ProgramArtifact), CliError> {
75            let program_artifact_path = workspace.package_build_path(package);
76            let program = read_program_from_file(&program_artifact_path)?;
77            Ok((package.clone(), program))
78        })
79        .collect::<Result<_, _>>()?;
80
81    let program_info = if args.profile_execution {
82        assert!(
83            args.compile_options.force_brillig,
84            "Internal CLI Error: --force-brillig must be active when --profile-execution is active"
85        );
86        profile_brillig_execution(binary_packages, &args.prover_name)?
87    } else {
88        binary_packages
89            .into_iter()
90            .par_bridge()
91            .map(|(package, program)| {
92                let package_name = package.name.to_string();
93                count_opcodes_and_gates_in_program(program, package_name)
94            })
95            .collect()
96    };
97
98    let info_report = InfoReport { programs: program_info };
99    show_info_report(info_report, args.json);
100
101    Ok(())
102}

104fn profile_brillig_execution(
105    binary_packages: Vec<(Package, ProgramArtifact)>,
106    prover_name: &str,
107) -> Result<Vec<ProgramInfo>, CliError> {
108    let mut program_info = Vec::new();
109    for (package, program_artifact) in binary_packages.iter() {
110        // Parse the initial witness values from Prover.toml or Prover.json
111        let (inputs_map, _) = read_inputs_from_file(
112            &package.root_dir.join(prover_name).with_extension("toml"),
113            &program_artifact.abi,
114        )?;
115        let initial_witness = program_artifact.abi.encode(&inputs_map, None)?;
116
117        let (_, profiling_samples) = nargo::ops::execute_program_with_profiling(
118            &program_artifact.bytecode,
119            initial_witness,
120            &Bn254BlackBoxSolver,
121            &mut DefaultForeignCallBuilder::default().build(),
122        )
123        .map_err(|e| {
124            CliError::Generic(format!(
125                "failed to execute '{}': {}",
126                package.root_dir.to_string_lossy(),
127                e
128            ))
129        })?;
130
131        program_info.push(ProgramInfo {
132            package_name: package.name.to_string(),
133            functions: vec![FunctionInfo { name: "main".to_string(), opcodes: 0 }],
134            unconstrained_functions_opcodes: profiling_samples.len(),
135            unconstrained_functions: vec![FunctionInfo {
136                name: "main".to_string(),
137                opcodes: profiling_samples.len(),
138            }],
139        });
140    }
141    Ok(program_info)
142}
