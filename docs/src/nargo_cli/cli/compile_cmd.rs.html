53pub(crate) fn run(args: CompileCommand, workspace: Workspace) -> Result<(), CliError> {
54    if args.watch {
55        if args.compile_options.debug_compile_stdin {
56            return Err(CliError::CantWatchStdin);
57        }
58        watch_workspace(&workspace, &args.compile_options)
59            .map_err(|err| CliError::Generic(err.to_string()))?;
60    } else {
61        let debug_compile_stdin = None;
62        compile_workspace_full(&workspace, &args.compile_options, debug_compile_stdin)?;
63    }
64    Ok(())
65}

67/// Continuously recompile the workspace on any Noir file change event.
68fn watch_workspace(workspace: &Workspace, compile_options: &CompileOptions) -> notify::Result<()> {
69    let (tx, rx) = std::sync::mpsc::channel();
70
71    // No specific tickrate, max debounce time 1 seconds
72    let mut debouncer = new_debouncer(Duration::from_secs(1), None, tx)?;
73
74    // Add a path to be watched. All files and directories at that path and
75    // below will be monitored for changes.
76    debouncer.watcher().watch(&workspace.root_dir, RecursiveMode::Recursive)?;
77
78    let mut screen = std::io::stdout();
79    write!(screen, "{}", termion::cursor::Save).unwrap();
80    screen.flush().unwrap();
81    let debug_compile_stdin = None;
82    let _ = compile_workspace_full(workspace, compile_options, debug_compile_stdin);
83    for res in rx {
84        let debounced_events = res.map_err(|mut err| err.remove(0))?;
85
86        // We only want to trigger a rebuild if a noir source file has been modified.
87        let noir_files_modified = debounced_events.iter().any(|event| {
88            let mut event_paths = event.event.paths.iter();
89            let event_affects_noir_file =
90                event_paths.any(|path| path.extension().is_some_and(|ext| ext == "nr"));
91
92            let is_relevant_event_kind = matches!(
93                event.kind,
94                EventKind::Create(_) | EventKind::Modify(_) | EventKind::Remove(_)
95            );
96
97            is_relevant_event_kind && event_affects_noir_file
98        });
99
100        if noir_files_modified {
101            write!(screen, "{}{}", termion::cursor::Restore, termion::clear::AfterCursor).unwrap();
102            screen.flush().unwrap();
103            let debug_compile_stdin = None;
104            let _ = compile_workspace_full(workspace, compile_options, debug_compile_stdin);
105        }
106    }
107
108    screen.flush().unwrap();
109
110    Ok(())
111}

113/// Parse all files in the workspace.
114pub fn parse_workspace(
115    workspace: &Workspace,
116    debug_compile_stdin: Option<String>,
117) -> (FileManager, ParsedFiles) {
118    let mut file_manager = workspace.new_file_manager();
119
120    if let Some(main_nr) = debug_compile_stdin {
121        file_manager.add_file_with_source(Path::new("src/main.nr"), main_nr);
122    } else {
123        insert_all_files_for_workspace_into_file_manager(workspace, &mut file_manager);
124    }
125
126    let parsed_files = parse_all(&file_manager);
127    (file_manager, parsed_files)
128}

130/// Parse and compile the entire workspace, then report errors.
131/// This is the main entry point used by all other commands that need compilation.
132pub fn compile_workspace_full(
133    workspace: &Workspace,
134    compile_options: &CompileOptions,
135    debug_compile_stdin: Option<String>, // use this String as STDIN if present
136) -> Result<(), CliError> {
137    let mut debug_compile_stdin = debug_compile_stdin;
138    if compile_options.debug_compile_stdin && debug_compile_stdin.is_none() {
139        let mut main_nr = String::new();
140        let stdin = std::io::stdin();
141        let mut stdin_handle = stdin.lock();
142        stdin_handle.read_to_string(&mut main_nr).expect("reading from stdin to succeed");
143        debug_compile_stdin = Some(main_nr);
144    }
145    let (workspace_file_manager, parsed_files) = parse_workspace(workspace, debug_compile_stdin);
146
147    let compiled_workspace =
148        compile_workspace(&workspace_file_manager, &parsed_files, workspace, compile_options)?;
149
150    report_errors(
151        compiled_workspace,
152        &workspace_file_manager,
153        compile_options.deny_warnings,
154        compile_options.silence_warnings,
155    )?;
156
157    Ok(())
158}

160/// Compile binary and contract packages.
161/// Returns the merged warnings or errors.
162fn compile_workspace(
163    file_manager: &FileManager,
164    parsed_files: &ParsedFiles,
165    workspace: &Workspace,
166    compile_options: &CompileOptions,
167) -> Result<CompilationResult<()>, CliError> {
168    let (binary_packages, contract_packages): (Vec<_>, Vec<_>) = workspace
169        .into_iter()
170        .filter(|package| !package.is_library())
171        .cloned()
172        .partition(|package| package.is_binary());
173
174    // Compile all of the packages in parallel.
175    let program_warnings_or_errors: CompilationResult<()> =
176        compile_programs(file_manager, parsed_files, workspace, &binary_packages, compile_options)?;
177
178    let contract_warnings_or_errors: CompilationResult<()> = compile_contracts(
179        file_manager,
180        parsed_files,
181        &contract_packages,
182        compile_options,
183        &workspace.target_directory_path(),
184    )?;
185
186    let result = match (program_warnings_or_errors, contract_warnings_or_errors) {
187        (Ok((_, program_warnings)), Ok((_, contract_warnings))) => {
188            let warnings = [program_warnings, contract_warnings].concat();
189            Ok(((), warnings))
190        }
191        (Err(program_errors), Err(contract_errors)) => {
192            Err([program_errors, contract_errors].concat())
193        }
194        (Err(errors), _) | (_, Err(errors)) => Err(errors),
195    };
196    Ok(result)
197}

199/// Compile the given binary packages in the workspace.
200fn compile_programs(
201    file_manager: &FileManager,
202    parsed_files: &ParsedFiles,
203    workspace: &Workspace,
204    binary_packages: &[Package],
205    compile_options: &CompileOptions,
206) -> Result<CompilationResult<()>, CliError> {
207    // Load any existing artifact for a given package, _iff_ it was compiled with the same nargo version.
208    // The loaded circuit includes backend specific transformations, which might be different from the current target.
209    let load_cached_program = |package| {
210        let program_artifact_path = workspace.package_build_path(package);
211        read_program_from_file(&program_artifact_path)
212            .ok()
213            .filter(|p| p.noir_version == NOIR_ARTIFACT_VERSION_STRING)
214            .map(|p| p.into())
215    };
216
217    let compile_package = |package| -> Result<CompilationResult<()>, CliError> {
218        let cached_program = load_cached_program(package);
219
220        // Hash over the entire compiled program, including any post-compile transformations.
221        // This is used to detect whether `cached_program` is returned by `compile_program`.
222        let cached_hash =
223            cached_program.as_ref().map(|prog| rustc_hash::FxBuildHasher.hash_one(prog));
224
225        // Compile the program, or use the cached artifacts if it matches.
226        match compile_program(
227            file_manager,
228            parsed_files,
229            workspace,
230            package,
231            compile_options,
232            cached_program,
233        ) {
234            Ok((program, warnings)) => {
235                // If the compiled program is the same as the cached one, we don't apply transformations again, unless the target width has changed.
236                // The transformations might not be idempotent, which would risk creating witnesses that don't work with earlier versions,
237                // based on which we might have generated a verifier already.
238                if cached_hash == Some(rustc_hash::FxBuildHasher.hash_one(&program)) {
239                    return Ok(Ok(((), warnings)));
240                }
241                // Run ACVM optimizations.
242                let program = nargo::ops::optimize_program(program);
243                // Check solvability.
244                match nargo::ops::check_program(&program) {
245                    Ok(()) => {
246                        // Overwrite the build artifacts with the final circuit, which includes the backend specific transformations.
247                        let _ = save_program_to_file(
248                            &program.into(),
249                            &package.name,
250                            &workspace.target_directory_path(),
251                        )?;
252                        Ok(Ok(((), warnings)))
253                    }
254                    Err(errors_and_warnings) => Ok(Err(errors_and_warnings)),
255                }
256            }
257            Err(errors_and_warnings) => Ok(Err(errors_and_warnings)),
258        }
259    };
260
261    // Configure a thread pool with a larger stack size to prevent overflowing stack in large programs.
262    // Default is 2MB. Limit threads to the number of packages we actually need to compile.
263    let num_threads = rayon::current_num_threads().min(binary_packages.len()).max(1);
264    let pool = rayon::ThreadPoolBuilder::new()
265        .num_threads(num_threads)
266        .stack_size(4 * 1024 * 1024)
267        .build()
268        .unwrap();
269    let program_results = pool.install(|| {
270        binary_packages.par_iter().map(compile_package).collect::<Result<Vec<_>, _>>()
271    })?;
272
273    // Collate any warnings/errors which were encountered during compilation.
274    Ok(collect_errors(program_results).map(|(_, warnings)| ((), warnings)))
275}

277/// Compile the given contracts in the workspace.
278fn compile_contracts(
279    file_manager: &FileManager,
280    parsed_files: &ParsedFiles,
281    contract_packages: &[Package],
282    compile_options: &CompileOptions,
283    target_dir: &Path,
284) -> Result<CompilationResult<()>, CliError> {
285    let contract_results = contract_packages
286        .par_iter()
287        .map(|package| -> Result<CompilationResult<()>, CliError> {
288            match compile_contract(file_manager, parsed_files, package, compile_options) {
289                Ok((contract, warnings)) => {
290                    let contract = nargo::ops::optimize_contract(contract);
291                    save_contract(
292                        contract,
293                        package,
294                        target_dir,
295                        compile_options.show_artifact_paths,
296                    )?;
297                    Ok(Ok(((), warnings)))
298                }
299                Err(errors_and_warnings) => Ok(Err(errors_and_warnings)),
300            }
301        })
302        .collect::<Result<Vec<_>, CliError>>()?;
303
304    // Collate any warnings/errors which were encountered during compilation.
305    let errors = collect_errors(contract_results).map(|(_, warnings)| ((), warnings));
306    Ok(errors)
307}

309fn save_contract(
310    contract: CompiledContract,
311    package: &Package,
312    target_dir: &Path,
313    show_artifact_paths: bool,
314) -> Result<(), CliError> {
315    let contract_name = contract.name.clone();
316    let artifact_path = save_contract_to_file(
317        &contract.into(),
318        &format!("{}-{}", package.name, contract_name),
319        target_dir,
320    )?;
321    if show_artifact_paths {
322        println!("Saved contract artifact to: {}", artifact_path.display());
323    }
324    Ok(())
