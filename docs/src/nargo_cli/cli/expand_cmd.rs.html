17/// Show the result of macro expansion
18#[derive(Debug, Clone, Args, Default)]
19pub(crate) struct ExpandCommand {
20    #[clap(flatten)]
21    pub(super) package_options: PackageOptions,
22
23    #[clap(flatten)]
24    compile_options: CompileOptions,
25}
26
27impl WorkspaceCommand for ExpandCommand {
28    fn package_selection(&self) -> PackageSelection {
29        self.package_options.package_selection()
30    }
31    fn lock_type(&self) -> LockType {
32        // Creates a `Prover.toml` template if it doesn't exist, otherwise only writes if `allow_overwrite` is true,
33        // so it shouldn't lead to accidental conflicts. Doesn't produce compilation artifacts.
34        LockType::None
35    }
36}

38pub(crate) fn run(args: ExpandCommand, workspace: Workspace) -> Result<(), CliError> {
39    let mut workspace_file_manager = workspace.new_file_manager();
40    insert_all_files_for_workspace_into_file_manager(&workspace, &mut workspace_file_manager);
41    let parsed_files = parse_all(&workspace_file_manager);
42
43    for package in &workspace {
44        expand_package(&workspace_file_manager, &parsed_files, package, &args.compile_options)?;
45    }
46
47    Ok(())
48}

61fn get_expanded_package_or_error(
62    file_manager: &FileManager,
63    parsed_files: &ParsedFiles,
64    package: &Package,
65    compile_options: &CompileOptions,
66) -> Result<String, CompileError> {
67    let (mut context, crate_id) = prepare_package(file_manager, parsed_files, package);
68
69    check_crate_and_report_errors(&mut context, crate_id, compile_options)?;
70
71    Ok(get_expanded_crate(crate_id, &context.crate_graph, &context.def_maps, &context.def_interner))
