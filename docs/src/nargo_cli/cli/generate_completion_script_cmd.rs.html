1use clap::{Args, CommandFactory};
2use clap_complete::Shell;
3
4use crate::errors::CliError;
5
6use super::NargoCli;
7
8/// Generates a shell completion script for your favorite shell
9#[derive(Debug, Clone, Args)]
10pub(crate) struct GenerateCompletionScriptCommand {
11    /// The shell to generate completions for. One of: bash, elvish, fish, powershell, zsh
12    pub(crate) shell: String,
13}
14
15pub(crate) fn run(command: GenerateCompletionScriptCommand) -> Result<(), CliError> {
16    let shell = match command.shell.to_lowercase().as_str() {
17        "bash" => Shell::Bash,
18        "elvish" => Shell::Elvish,
19        "fish" => Shell::Fish,
20        "powershell" => Shell::PowerShell,
21        "zsh" => Shell::Zsh,
22        _ => {
23            return Err(CliError::Generic(
24                "Invalid shell. Supported shells are: bash, elvish, fish, powershell, zsh"
25                    .to_string(),
26            ));
27        }
28    };
29
30    clap_complete::generate(shell, &mut NargoCli::command(), "nargo", &mut std::io::stdout());
31
32    Ok(())
33}
