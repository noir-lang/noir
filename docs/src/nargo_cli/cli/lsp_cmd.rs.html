12/// Starts the Noir LSP server
13///
14/// Starts an LSP server which allows IDEs such as VS Code to display diagnostics in Noir source.
15///
16/// VS Code Noir Language Support: <https://marketplace.visualstudio.com/items?itemName=noir-lang.vscode-noir>
17#[derive(Debug, Clone, Args)]
18pub(crate) struct LspCommand;
19
20pub(crate) fn run() -> Result<(), CliError> {
21    use tokio::runtime::Builder;
22
23    let runtime = Builder::new_current_thread().enable_all().build().unwrap();
24
25    runtime.block_on(async {
26        let (server, _) = async_lsp::MainLoop::new_server(|client| {
27            let router = NargoLspService::new(&client, Bn254BlackBoxSolver);
28
29            ServiceBuilder::new()
30                .layer(TracingLayer::default())
31                .layer(LifecycleLayer::default())
32                .layer(CatchUnwindLayer::default())
33                .layer(ConcurrencyLayer::default())
34                .service(router)
35        });
36
37        // Prefer truly asynchronous piped stdin/stdout without blocking tasks.
38        #[cfg(unix)]
39        let (stdin, stdout) = (
40            async_lsp::stdio::PipeStdin::lock_tokio().unwrap(),
41            async_lsp::stdio::PipeStdout::lock_tokio().unwrap(),
42        );
43        // Fallback to spawn blocking read/write otherwise.
44        #[cfg(not(unix))]
45        let (stdin, stdout) = (
46            tokio_util::compat::TokioAsyncReadCompatExt::compat(tokio::io::stdin()),
47            tokio_util::compat::TokioAsyncWriteCompatExt::compat_write(tokio::io::stdout()),
48        );
49
50        server.run_buffered(stdin, stdout).await.map_err(CliError::LspError)
51    })
52}
