1use std::collections::BTreeSet;
2use std::path::PathBuf;
3
4use crate::errors::CliError;
5
6use clap::Args;
7use nargo_toml::{PackageSelection, list_cached_git_dependencies};
8
9use super::{LockType, PackageOptions, WorkspaceCommand};
10
11/// Fetch the dependencies of a package from the network
12#[derive(Debug, Clone, Args)]
13pub(crate) struct FetchCommand {
14    #[clap(flatten)]
15    pub(super) package_options: PackageOptions,
16}
17
18impl WorkspaceCommand for FetchCommand {
19    fn package_selection(&self) -> PackageSelection {
20        self.package_options.package_selection()
21    }
22
23    fn lock_type(&self) -> LockType {
24        // The `LockType` lock guards a workspace member's `target/` artifacts, which this command
25        // never writes. Git dependencies are downloaded into the global cache during workspace
26        // resolution, which serializes concurrent runs with its own `.package-cache` lock before
27        // any `LockType` lock would be taken, so `None` is correct here.
28        LockType::None
29    }
30}
31
32pub(crate) fn run(fetched_before: BTreeSet<PathBuf>) -> Result<(), CliError> {
33    // Resolving the workspace (which happens before this runs) downloads any missing git
34    // dependencies into the global cache. Comparing the cache against the snapshot taken before
35    // resolution tells us exactly which dependencies were fetched during this run; ones that were
36    // already cached are not reported.
37    let fetched_after = list_cached_git_dependencies();
38    let newly_fetched: Vec<_> = fetched_after.difference(&fetched_before).collect();
39
40    if newly_fetched.is_empty() {
41        return Ok(());
42    }
43
44    let count = newly_fetched.len();
45    let noun = if count == 1 { "dependency" } else { "dependencies" };
46    noirc_errors::println_to_stdout!("Fetched {count} {noun}:");
47    for dependency in newly_fetched {
48        noirc_errors::println_to_stdout!("  {}", dependency.display());
49    }
50
51    Ok(())
52}
