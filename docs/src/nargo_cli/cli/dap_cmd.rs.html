62fn find_workspace(project_folder: &str, package: Option<&str>) -> Option<Workspace> {
63    let Ok(toml_path) = get_package_manifest(Path::new(project_folder)) else {
64        eprintln!("ERROR: Failed to get package manifest");
65        return None;
66    };
67    let package = package.and_then(|p| serde_json::from_str::<CrateName>(p).ok());
68    let selection = package.map_or(PackageSelection::DefaultOrAll, PackageSelection::Selected);
69    match resolve_workspace_from_toml(
70        &toml_path,
71        selection,
72        Some(NOIR_ARTIFACT_VERSION_STRING.to_string()),
73    ) {
74        Ok(workspace) => Some(workspace),
75        Err(err) => {
76            eprintln!("ERROR: Failed to resolve workspace: {err}");
77            None
78        }
79    }
80}

100fn compile_test(
101    workspace: &Workspace,
102    package: &Package,
103    compile_options: CompileOptions,
104    test_name: String,
105) -> Result<(CompiledProgram, TestDefinition), LoadError> {
106    let (file_manager, mut parsed_files) = load_workspace_files(workspace);
107
108    let (mut context, crate_id) =
109        prepare_package_for_debug(&file_manager, &mut parsed_files, package, workspace);
110
111    check_crate_and_report_errors(&mut context, crate_id, &compile_options)
112        .map_err(|_| LoadError::Generic("Failed to compile project".into()))?;
113
114    let test = get_test_function_for_debug(crate_id, &context, &test_name)
115        .map_err(|_| LoadError::Generic("Failed to compile project".into()))?;
116
117    let program = compile_test_fn_for_debugging(&test, &mut context, compile_options)
118        .map_err(|_| LoadError::Generic("Failed to compile project".into()))?;
119    Ok((program, test))
120}

122fn load_and_compile_project(
123    project_folder: &str,
124    package: Option<&str>,
125    prover_name: &str,
126    compile_options: CompileOptions,
127    test_name: Option<String>,
128) -> Result<(DebugProject, Option<TestDefinition>), LoadError> {
129    let workspace = find_workspace(project_folder, package)
130        .ok_or(LoadError::Generic(workspace_not_found_error_msg(project_folder, package)))?;
131    let package = workspace
132        .into_iter()
133        .find(|p| p.is_binary() || p.is_contract())
134        .ok_or(LoadError::Generic("No matching binary or contract packages found in workspace. Only these packages can be debugged.".into()))?;
135
136    let (compiled_program, test_def) = match test_name {
137        None => {
138            let program = compile_main(&workspace, package, &compile_options)?;
139            Ok((program, None))
140        }
141        Some(test_name) => {
142            let (program, test_def) =
143                compile_test(&workspace, package, compile_options, test_name)?;
144            Ok((program, Some(test_def)))
145        }
146    }?;
147
148    let (inputs_map, _) = read_inputs_from_file(
149        &package.root_dir.join(prover_name).with_extension("toml"),
150        &compiled_program.abi,
151    )
152    .map_err(|e| {
153        LoadError::Generic(format!("Failed to read program inputs from {prover_name}: {e}"))
154    })?;
155    let initial_witness = compiled_program
156        .abi
157        .encode(&inputs_map, None)
158        .map_err(|_| LoadError::Generic("Failed to encode inputs".into()))?;
159
160    let project = DebugProject {
161        compiled_program,
162        initial_witness,
163        root_dir: workspace.root_dir.clone(),
164        package_name: package.name.to_string(),
165    };
166    Ok((project, test_def))
167}

169fn loop_uninitialized_dap<R: Read, W: Write>(mut server: Server<R, W>) -> Result<(), DapError> {
170    while let Some(req) = server.poll_request()? {
171        match req.command {
172            Command::Initialize(_) => {
173                let rsp = req.success(ResponseBody::Initialize(Capabilities {
174                    supports_disassemble_request: Some(true),
175                    supports_instruction_breakpoints: Some(true),
176                    supports_stepping_granularity: Some(true),
177                    ..Default::default()
178                }));
179                server.respond(rsp)?;
180            }
181
182            Command::Launch(ref arguments) => {
183                let Some(Value::Object(ref additional_data)) = arguments.additional_data else {
184                    server.respond(req.error("Missing launch arguments"))?;
185                    continue;
186                };
187                let Some(Value::String(project_folder)) = additional_data.get("projectFolder")
188                else {
189                    server.respond(req.error("Missing project folder argument"))?;
190                    continue;
191                };
192
193                let project_folder = project_folder.as_str();
194                let package = additional_data.get("package").and_then(|v| v.as_str());
195                let prover_name = additional_data
196                    .get("proverName")
197                    .and_then(|v| v.as_str())
198                    .unwrap_or(PROVER_INPUT_FILE);
199
200                let generate_acir =
201                    additional_data.get("generateAcir").and_then(|v| v.as_bool()).unwrap_or(false);
202                let skip_instrumentation = additional_data
203                    .get("skipInstrumentation")
204                    .and_then(|v| v.as_bool())
205                    .unwrap_or(generate_acir);
206                let test_name =
207                    additional_data.get("testName").and_then(|v| v.as_str()).map(String::from);
208                let oracle_resolver_url = additional_data
209                    .get("oracleResolver")
210                    .and_then(|v| v.as_str())
211                    .map(String::from);
212
213                eprintln!("Project folder: {project_folder}");
214                eprintln!("Package: {}", package.unwrap_or("(default)"));
215                eprintln!("Prover name: {prover_name}");
216
217                let compile_options = compile_options_for_debugging(
218                    generate_acir,
219                    skip_instrumentation,
220                    CompileOptions::default(),
221                );
222
223                match load_and_compile_project(
224                    project_folder,
225                    package,
226                    prover_name,
227                    compile_options,
228                    test_name,
229                ) {
230                    Ok((project, test)) => {
231                        server.respond(req.ack()?)?;
232                        let abi = project.compiled_program.abi.clone();
233                        let debug = project.compiled_program.debug.clone();
234
235                        let result = noir_debugger::run_dap_loop(
236                            &mut server,
237                            project,
238                            RunParams { oracle_resolver_url, raw_source_printing: None },
239                        )?;
240
241                        if let Some(test) = test {
242                            analyze_test_result(&mut server, result, test, abi, debug)?;
243                        }
244                        break;
245                    }
246                    Err(LoadError::Generic(message)) => {
247                        server.respond(req.error(message.as_str()))?;
248                    }
249                }
250            }
251
252            Command::Disconnect(_) => {
253                server.respond(req.ack()?)?;
254                break;
255            }
256
257            _ => {
258                let command = req.command;
259                eprintln!("ERROR: unhandled command: {command:?}");
260            }
261        }
262    }
263    Ok(())
264}

266fn analyze_test_result<R: Read, W: Write>(
267    server: &mut Server<R, W>,
268    result: DebugExecutionResult,
269    test: TestDefinition,
270    abi: Abi,
271    debug: Vec<DebugInfo>,
272) -> Result<(), ServerError> {
273    let test_status = match result {
274        DebugExecutionResult::Solved(result) => {
275            test_status_program_compile_pass(&test.function, &abi, &debug, &Ok(result))
276        }
277        // Test execution failed
278        DebugExecutionResult::Error(error) => {
279            test_status_program_compile_pass(&test.function, &abi, &debug, &Err(error))
280        }
281        // Execution didn't complete
282        DebugExecutionResult::Incomplete => {
283            TestStatus::Fail { message: "Execution halted".into(), error_diagnostic: None }
284        }
285    };
286
287    let test_result_message = match test_status {
288        TestStatus::Pass => "✓ Test passed".into(),
289        TestStatus::Fail { message, error_diagnostic } => {
290            let basic_message = format!("x Test failed: {message}");
291            match error_diagnostic {
292                Some(diagnostic) => format!("{basic_message}.\n{diagnostic:#?}"),
293                None => basic_message,
294            }
295        }
296        TestStatus::CompileError(diagnostic) => format!("x Test failed.\n{diagnostic:#?}"),
297        TestStatus::Skipped => "* Test skipped".into(),
298    };
299
300    server.send_event(dap::events::Event::Output(OutputEventBody {
301        category: Some(OutputEventCategory::Console),
302        output: test_result_message,
303        ..OutputEventBody::default()
304    }))
305}

307fn run_preflight_check(args: DapCommand) -> Result<(), DapError> {
308    let Some(project_folder) = args.preflight_project_folder else {
309        return Err(DapError::PreFlightGenericError("Noir Debugger could not initialize because the IDE (for example, VS Code) did not specify a project folder to debug.".into()));
310    };
311
312    let package = args.preflight_package.as_deref();
313    let test_name = args.preflight_test_name;
314    let prover_name = args.preflight_prover_name.as_deref().unwrap_or(PROVER_INPUT_FILE);
315
316    let compile_options: CompileOptions = compile_options_for_debugging(
317        args.preflight_generate_acir,
318        args.preflight_skip_instrumentation,
319        CompileOptions::default(),
320    );
321
322    let _ = load_and_compile_project(
323        project_folder.as_str(),
324        package,
325        prover_name,
326        compile_options,
327        test_name,
328    )?;
329
330    Ok(())
331}

333pub(crate) fn run(args: DapCommand) -> Result<(), CliError> {
334    // When the --preflight-check flag is present, we run Noir's DAP server in "pre-flight mode", which test runs
335    // the DAP initialization code without actually starting the DAP server.
336    //
337    // This lets the client IDE present any initialization issues (compiler version mismatches, missing prover files, etc)
338    // in its own interface.
339    //
340    // This was necessary due to the VS Code project being reluctant to let extension authors capture
341    // stderr output generated by a DAP server wrapped in DebugAdapterExecutable.
342    //
343    // Exposing this preflight mode lets us gracefully handle errors that happen *before*
344    // the DAP loop is established, which otherwise are considered "out of band" by the maintainers of the DAP spec.
345    // More details here: https://github.com/microsoft/vscode/issues/108138
346    if args.preflight_check {
347        return run_preflight_check(args).map_err(CliError::DapError);
348    }
349
350    let output = BufWriter::new(std::io::stdout());
351    let input = BufReader::new(DapFixingReader::new(BufReader::new(std::io::stdin())));
352    let server = Server::new(input, output);
353
354    loop_uninitialized_dap(server).map_err(CliError::DapError)
355}

357/// Wraps a buffered reader over the DAP input stream and transparently fixes malformed messages
358/// before they reach the [`Server`].
359///
360/// Specifically, some clients send `"arguments": {}` for unit-variant commands like
361/// `configurationDone`. The `dap` crate's serde deserialization rejects non-null `arguments`
362/// for unit variants. This reader strips empty `arguments` objects from those commands.
363///
364/// On any parsing failure the original bytes are passed through unchanged so that [`Server`]
365/// can produce its own error.
366struct DapFixingReader<R: BufRead> {
367    inner: R,
368    /// Pre-processed bytes of the next DAP message ready to be returned by `read`.
369    pending: Vec<u8>,
370    /// The number of bytes we have already copied to the destination in `read`.
371    pos: usize,
372}
373
374impl<R: BufRead> DapFixingReader<R> {
375    fn new(inner: R) -> Self {
376        Self { inner, pending: Vec::new(), pos: 0 }
377    }
378
379    /// Read one DAP message from `inner`, fix it, and store it in `pending`.
380    /// Returns `false` on EOF, `true` if a message was buffered.
381    fn fill_pending(&mut self) -> std::io::Result<bool> {
382        // Read the Content-Length header line.
383        let mut line = String::new();
384        if self.inner.read_line(&mut line)? == 0 {
385            return Ok(false); // EOF
386        }
387
388        // Try to parse the content length.
389        let content_length = line
390            .trim_end()
391            .strip_prefix("Content-Length:")
392            .and_then(|rest| rest.trim().parse::<usize>().ok());
393
394        let Some(content_length) = content_length else {
395            // Not a valid Content-Length header; pass through and let the Server error.
396            self.pending = line.into_bytes();
397            self.pos = 0;
398            return Ok(true);
399        };
400
401        // Read the blank separator line.
402        // (In `Server::poll_request` this happens as part of the `loop`).
403        let mut sep = String::new();
404        if self.inner.read_line(&mut sep)? == 0 {
405            return Ok(false); // EOF
406        }
407
408        // Read exactly content_length bytes.
409        let mut content = vec![0u8; content_length];
410        self.inner.read_exact(&mut content)?;
411
412        // Fix the content, falling back to the original on any error.
413        let fixed = fix_dap_content(&content);
414
415        // Reconstruct the DAP framing with the (possibly updated) length.
416        let header = format!("Content-Length: {}\r\n\r\n", fixed.len());
417        self.pending = header.into_bytes();
418        self.pending.extend_from_slice(&fixed);
419        self.pos = 0;
420        Ok(true)
421    }
422}
423
424impl<R: BufRead> Read for DapFixingReader<R> {
425    /// Read data from `pending` into `buf`
426    fn read(&mut self, buf: &mut [u8]) -> std::io::Result<usize> {
427        // Refill if we've consumed everything in `pending`.
428        while self.pos >= self.pending.len() {
429            if !self.fill_pending()? {
430                return Ok(0); // EOF
431            }
432        }
433        let available = &self.pending[self.pos..];
434        // How much data can we read depends on the size of `buf` and `available`.
435        let n = buf.len().min(available.len());
436        buf[..n].copy_from_slice(&available[..n]);
437        self.pos += n;
438        Ok(n)
439    }
440}

442/// Remove an empty `arguments` object from JSON for unit-variant DAP commands.
443///
444/// Some clients send `{"command": "configurationDone", "arguments": {}}`, but the `dap` crate
445/// expects no `arguments` key at all for unit variants.  Returns the original bytes unchanged if
446/// parsing fails or no fix is needed.
447fn fix_dap_content(content: &[u8]) -> Vec<u8> {
448    let Ok(mut value) = serde_json::from_slice::<Value>(content) else {
449        return content.to_vec();
450    };
451
452    let Some(obj) = value.as_object_mut() else {
453        return content.to_vec();
454    };
455
456    let is_unit_command =
457        obj.get("command").and_then(|v| v.as_str()).is_some_and(|cmd| UNIT_COMMANDS.contains(&cmd));
458
459    if is_unit_command
460        && let Some(Value::Object(args)) = obj.get("arguments")
461        && args.is_empty()
462    {
463        obj.remove("arguments");
464    }
465
466    serde_json::to_vec(&value).unwrap_or_else(|_| content.to_vec())
467}

478    fn read_all<R: BufRead>(reader: &mut DapFixingReader<R>) -> Vec<u8> {
479        let mut out = Vec::new();

486        let body_start = output.find("\r\n\r\n").unwrap() + 4;
487        &output[body_start..]
