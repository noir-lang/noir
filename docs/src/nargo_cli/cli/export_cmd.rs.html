24#[allow(rustdoc::broken_intra_doc_links)]
25/// Exports functions marked with #[export] attribute
26#[derive(Debug, Clone, Args)]
27pub(crate) struct ExportCommand {
28    #[clap(flatten)]
29    pub(super) package_options: PackageOptions,
30
31    #[clap(flatten)]
32    compile_options: CompileOptions,
33}
34
35impl WorkspaceCommand for ExportCommand {
36    fn package_selection(&self) -> PackageSelection {
37        self.package_options.package_selection()
38    }
39
40    fn lock_type(&self) -> LockType {
41        // Writes the exported functions.
42        LockType::Exclusive
43    }
44}

46pub(crate) fn run(args: ExportCommand, workspace: Workspace) -> Result<(), CliError> {
47    let mut workspace_file_manager = workspace.new_file_manager();
48    insert_all_files_for_workspace_into_file_manager(&workspace, &mut workspace_file_manager);
49    let parsed_files = parse_all(&workspace_file_manager);
50
51    let library_packages: Vec<_> =
52        workspace.into_iter().filter(|package| package.is_library()).collect();
53
54    library_packages
55        .par_iter()
56        .map(|package| {
57            compile_exported_functions(
58                &workspace_file_manager,
59                &parsed_files,
60                &workspace,
61                package,
62                &args.compile_options,
63            )
64        })
65        .collect()
66}
67
68fn compile_exported_functions(
69    file_manager: &FileManager,
70    parsed_files: &ParsedFiles,
71    workspace: &Workspace,
72    package: &Package,
73    compile_options: &CompileOptions,
74) -> Result<(), CliError> {
75    let (mut context, crate_id) = prepare_package(file_manager, parsed_files, package);
76    check_crate_and_report_errors(&mut context, crate_id, compile_options)?;
77
78    let exported_functions = context.get_all_exported_functions_in_crate(&crate_id);
79
80    let exported_programs = try_vecmap(
81        exported_functions,
82        |(function_name, function_id)| -> Result<(String, CompiledProgram), CompileError> {
83            // TODO: We should to refactor how to deal with compilation errors to avoid this.
84            let program = compile_no_check(&mut context, compile_options, function_id, None, false)
85                .map_err(|error| vec![CustomDiagnostic::from(error)]);
86
87            let program = report_errors(
88                program.map(|program| (program, Vec::new())),
89                file_manager,
90                compile_options.deny_warnings,
91                compile_options.silence_warnings,
92            )?;
93
94            Ok((function_name, program))
95        },
96    )?;
97
98    let export_dir = workspace.export_directory_path();
99    for (function_name, program) in exported_programs {
100        save_program_to_file(&program.into(), &function_name.parse().unwrap(), &export_dir)?;
101    }
102    Ok(())
103}
