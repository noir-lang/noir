30pub(super) fn run_comptime(args: ExecuteCommand, workspace: Workspace) -> Result<(), CliError> {
31    let (file_manager, parsed_files) = parse_workspace(&workspace, None);
32    let binary_packages = workspace.into_iter().filter(|package| package.is_binary());
33
34    for package in binary_packages {
35        run_package_comptime(package, &args, &workspace, &file_manager, &parsed_files)?;
36    }
37
38    Ok(())
39}

41fn run_package_comptime(
42    package: &Package,
43    args: &ExecuteCommand,
44    workspace: &Workspace,
45    file_manager: &FileManager,
46    parsed_files: &ParsedFiles,
47) -> Result<(), CliError> {
48    let (mut context, crate_id) = nargo::prepare_package(file_manager, parsed_files, package);
49    context.package_build_path = workspace.package_build_path(package);
50    noirc_driver::link_to_debug_crate(&mut context, crate_id);
51    let result = noirc_driver::check_crate(&mut context, crate_id, &args.compile_options);
52    match report_errors(
53        result,
54        &context.file_manager,
55        args.compile_options.deny_warnings,
56        args.compile_options.silence_warnings,
57    ) {
58        Ok(()) => {}
59        Err(err) => {
60            return Err(CliError::CompileError(err));
61        }
62    }
63
64    let Some(main_id) = context.get_main_function(&crate_id) else {
65        return Err(CliError::Generic("Could not find main function".to_string()));
66    };
67
68    let func_meta = context.def_interner.function_meta(&main_id);
69
70    let error_types = BTreeMap::default();
71    let abi = gen_abi(&context, &main_id, func_meta.return_visibility, error_types);
72
73    // Parse the inputs and convert them to what the interpreter expects.
74    let prover_file = package.root_dir.join(&args.prover_name).with_extension("toml");
75    let (prover_input, return_value) =
76        noir_artifact_cli::fs::inputs::read_inputs_from_file(&prover_file, &abi)?;
77
78    let location = func_meta.location;
79    let func_args =
80        input_values_to_comptime_values(&prover_input, func_meta, &context.def_interner);
81    let return_value = return_value.map(|return_value| {
82        input_value_to_comptime_value(&return_value, func_meta.return_type(), location)
83    });
84
85    match context.interpret_function(main_id, func_args) {
86        Ok(result) => {
87            let result_as_string = output_value_to_string(&result, &context);
88
89            println!("[{}] Circuit witness successfully solved", package.name);
90            if !matches!(result, Value::Unit) {
91                println!("[{}] Circuit output: {result_as_string}", package.name);
92            }
93
94            if let Some(return_value) = return_value {
95                if result != return_value {
96                    let return_value_as_string =
97                        return_value.display(&context.def_interner).to_string();
98                    return Err(CliError::Generic(format!(
99                        "Unexpected return value.\nExpected: {return_value_as_string}\nGot:      {result_as_string}"
100                    )));
101                }
102            }
103
104            Ok(())
105        }
106        Err(err) => {
107            let err = CompilationError::InterpreterError(err);
108            let diagnostic = CustomDiagnostic::from(&err);
109            let errors = vec![diagnostic];
110            report_errors(
111                Ok(((), errors)),
112                file_manager,
113                args.compile_options.deny_warnings,
114                args.compile_options.silence_warnings,
115            )?;
116            Err(CliError::Generic("Error interpreting main function".to_string()))
117        }
118    }
119}

121fn input_values_to_comptime_values(
122    prover_input: &InputMap,
123    func_meta: &FuncMeta,
124    interner: &NodeInterner,
125) -> Vec<(Value, Location)> {
126    vecmap(func_meta.parameters.iter(), |(pattern, typ, _visibility)| {
127        let location = pattern.location();
128        let (pattern, typ) = if let HirPattern::Mutable(pattern, _) = pattern {
129            (*pattern.clone(), typ.clone())
130        } else {
131            (pattern.clone(), typ.clone())
132        };
133        let HirPattern::Identifier(ident) = pattern else {
134            panic!("only identifier patterns are supported in main");
135        };
136        let name = interner.definition_name(ident.id);
137        let input = prover_input
138            .get(name)
139            .unwrap_or_else(|| panic!("Expected to find {name} in prover inputs"));
140        let value = input_value_to_comptime_value(input, &typ, location);
141        (value, location)
142    })
143}

145fn input_value_to_comptime_value(input: &InputValue, typ: &Type, location: Location) -> Value {
146    match typ {
147        Type::Unit => Value::Unit,
148        Type::Bool => {
149            let InputValue::Field(value) = input else {
150                panic!("expected field input for bool type");
151            };
152            Value::Bool(!value.is_zero())
153        }
154        Type::Integer(signedness, bit_size) => {
155            let InputValue::Field(value) = input else {
156                panic!("expected field input for bool type");
157            };
158
159            // Here we reuse the logic of converting an input into an SSA numeric value,
160            // which is not ideal but avoids duplicating the conversion logic.
161            let bit_size = u32::from(bit_size.bit_size());
162            let numeric_type = match signedness {
163                Signedness::Unsigned => NumericType::Unsigned { bit_size },
164                Signedness::Signed => NumericType::Signed { bit_size },
165            };
166            let numeric_value = NumericValue::from_constant(*value, numeric_type)
167                .expect("Could not convert field value to integer");
168            match numeric_value {
169                NumericValue::Field(_) => panic!("Field should not happen here"),
170                NumericValue::U1(value) => Value::U1(value),
171                NumericValue::U8(fitted) => match fitted {
172                    Fitted::Fit(value) => Value::U8(value),
173                    Fitted::Unfit(..) => panic!("input value does not fit in u8"),
174                },
175                NumericValue::U16(fitted) => match fitted {
176                    Fitted::Fit(value) => Value::U16(value),
177                    Fitted::Unfit(..) => panic!("input value does not fit in u16"),
178                },
179                NumericValue::U32(fitted) => match fitted {
180                    Fitted::Fit(value) => Value::U32(value),
181                    Fitted::Unfit(..) => panic!("input value does not fit in u32"),
182                },
183                NumericValue::U64(fitted) => match fitted {
184                    Fitted::Fit(value) => Value::U64(value),
185                    Fitted::Unfit(..) => panic!("input value does not fit in u64"),
186                },
187                NumericValue::U128(fitted) => match fitted {
188                    Fitted::Fit(value) => Value::U128(value),
189                    Fitted::Unfit(..) => panic!("input value does not fit in u128"),
190                },
191                NumericValue::I8(fitted) => match fitted {
192                    Fitted::Fit(value) => Value::I8(value),
193                    Fitted::Unfit(..) => panic!("input value does not fit in i8"),
194                },
195                NumericValue::I16(fitted) => match fitted {
196                    Fitted::Fit(value) => Value::I16(value),
197                    Fitted::Unfit(..) => panic!("input value does not fit in i16"),
198                },
199                NumericValue::I32(fitted) => match fitted {
200                    Fitted::Fit(value) => Value::I32(value),
201                    Fitted::Unfit(..) => panic!("input value does not fit in i32"),
202                },
203                NumericValue::I64(fitted) => match fitted {
204                    Fitted::Fit(value) => Value::I64(value),
205                    Fitted::Unfit(..) => panic!("input value does not fit in i64"),
206                },
207            }
208        }
209        Type::FieldElement => {
210            let InputValue::Field(value) = input else {
211                panic!("expected field input for field element type");
212            };
213            Value::Field(SignedField::positive(*value))
214        }
215        Type::Array(length, element_typ) => {
216            let length =
217                length.evaluate_to_u32(location).expect("Could not evaluate array length to u32");
218            let InputValue::Vec(inputs) = input else {
219                panic!("expected vec input for array type");
220            };
221            assert_eq!(inputs.len(), length as usize, "Array length does not match input length");
222            let array = inputs
223                .iter()
224                .map(|input| input_value_to_comptime_value(input, element_typ, location))
225                .collect();
226            Value::Array(array, typ.clone())
227        }
228        Type::String(length) => {
229            let InputValue::String(string) = input else {
230                panic!("expected string input for string type");
231            };
232            let length =
233                length.evaluate_to_u32(location).expect("Could not evaluate string length to u32");
234            assert_eq!(string.len(), length as usize, "String length does not match input length");
235            Value::String(Rc::new(string.clone()))
236        }
237        Type::Tuple(types) => {
238            let InputValue::Vec(inputs) = input else {
239                panic!("expected vec input for tuple type");
240            };
241            assert_eq!(inputs.len(), types.len(), "Tuple length does not match input length");
242            let tuple = vecmap(inputs.iter().zip(types.iter()), |(input, typ)| {
243                let value = input_value_to_comptime_value(input, typ, location);
244                Shared::new(value)
245            });
246            Value::Tuple(tuple)
247        }
248        Type::DataType(data_type, generics) => {
249            let fields = data_type
250                .borrow()
251                .get_fields(generics)
252                .expect("Enums as inputs are not yet supported");
253            let InputValue::Struct(inputs) = input else {
254                panic!("expected struct input for data type");
255            };
256            let fields = fields
257                .into_iter()
258                .map(|(name, typ, _)| {
259                    let input = inputs
260                        .get(&name)
261                        .unwrap_or_else(|| panic!("Expected to find field {name} in input"));
262                    let value = input_value_to_comptime_value(input, &typ, location);
263                    (Rc::new(name), Shared::new(value))
264                })
265                .collect();
266            Value::Struct(fields, typ.clone())
267        }
268        Type::Alias(alias, generics) => {
269            let typ = alias.borrow().get_type(generics);
270            input_value_to_comptime_value(input, &typ, location)
271        }
272        Type::Vector(_)
273        | Type::FmtString(_, _)
274        | Type::TypeVariable(..)
275        | Type::TraitAsType(..)
276        | Type::NamedGeneric(..)
277        | Type::CheckedCast { .. }
278        | Type::Function(..)
279        | Type::Reference(_, _)
280        | Type::Forall(..)
281        | Type::Constant(..)
282        | Type::Quoted(..)
283        | Type::InfixExpr(..)
284        | Type::Error => panic!("Unexpected type in comptime input value conversion"),
285    }
286}

288/// Converts a [Value] into a `String`.
289///
290/// This is similar to `Value::display(..).to_string()` except that:
291/// - only values that can be a circuit output are supported
292/// - strings are quoted
293/// - struct paths are fully-qualified
294///
295/// This is so the output matches the format produced by `nargo execute`.
296fn output_value_to_string(value: &Value, context: &Context) -> String {
297    match value {
298        Value::Unit => "()".to_string(),
299        Value::Bool(value) => value.to_string(),
300        Value::Field(signed_field) => signed_field.to_field_element().to_short_hex(),
301        Value::I8(value) => value.to_string(),
302        Value::I16(value) => value.to_string(),
303        Value::I32(value) => value.to_string(),
304        Value::I64(value) => value.to_string(),
305        Value::U1(false) => "0".to_string(),
306        Value::U1(true) => "1".to_string(),
307        Value::U8(value) => value.to_string(),
308        Value::U16(value) => value.to_string(),
309        Value::U32(value) => value.to_string(),
310        Value::U64(value) => value.to_string(),
311        Value::U128(value) => value.to_string(),
312        Value::String(string) => {
313            format!("{string:?}")
314        }
315        Value::Tuple(values) => {
316            let values = vecmap(values, |value| output_value_to_string(&value.borrow(), context));
317            if values.len() == 1 {
318                format!("({},)", values[0])
319            } else {
320                format!("({})", values.join(", "))
321            }
322        }
323        Value::Array(values, _) => {
324            let values = vecmap(values, |value| output_value_to_string(value, context));
325            format!("[{}]", values.join(", "))
326        }
327        Value::Struct(fields, typ) => {
328            let data_type = match typ.follow_bindings() {
329                Type::DataType(def, _) => def,
330                other => unreachable!("Expected data type, found {other}"),
331            };
332            let data_type = data_type.borrow();
333            let typename =
334                context.fully_qualified_struct_path(context.root_crate_id(), data_type.id);
335
336            // Display fields in the order they are defined in the struct.
337            // Some fields might not be there if they were missing in the constructor.
338            let fields = data_type
339                .fields_raw()
340                .unwrap()
341                .iter()
342                .filter_map(|field| {
343                    let name = field.name.as_string();
344                    fields.get(name).map(|value| {
345                        format!("{}: {}", name, output_value_to_string(&value.borrow(), context))
346                    })
347                })
348                .collect::<Vec<_>>();
349            format!("{typename} {{ {} }}", fields.join(", "))
350        }
351        Value::Enum(tag, args, typ) => {
352            let args = vecmap(args, |arg| output_value_to_string(arg, context)).join(", ");
353
354            match typ.follow_bindings_shallow().as_ref() {
355                Type::DataType(def, _) => {
356                    let def = def.borrow();
357                    let variant = def.variant_at(*tag);
358                    if variant.is_function {
359                        format!("{}::{}({args})", def.name, variant.name)
360                    } else {
361                        format!("{}::{}", def.name, variant.name)
362                    }
363                }
364                other => panic!("Expected a data type, got {other}"),
365            }
366        }
367        Value::Function(..)
368        | Value::Closure(..)
369        | Value::Pointer(..)
370        | Value::Vector(..)
371        | Value::Quoted(..)
372        | Value::TypeDefinition(..)
373        | Value::TraitConstraint(..)
374        | Value::TraitDefinition(..)
375        | Value::TraitImpl(..)
376        | Value::FunctionDefinition(..)
377        | Value::ModuleDefinition(..)
378        | Value::Type(_)
379        | Value::Zeroed(_)
380        | Value::Expr(..)
381        | Value::TypedExpr(..)
382        | Value::UnresolvedType(..)
383        | Value::FormatString(..)
384        | Value::CtString(..) => {
385            panic!("Unexpected output value: {}", value.display(&context.def_interner))
386        }
387    }
