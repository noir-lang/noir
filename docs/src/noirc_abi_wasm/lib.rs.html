56#[wasm_bindgen(js_name = abiEncode)]
57pub fn abi_encode(
58    abi: JsAbi,
59    inputs: JsInputMap,
60    return_value: Option<JsInputValue>,
61) -> Result<JsWitnessMap, JsAbiError> {
62    console_error_panic_hook::set_once();
63    let abi: Abi =
64        JsValueSerdeExt::into_serde(&JsValue::from(abi)).map_err(|err| err.to_string())?;
65    let inputs: BTreeMap<String, JsonTypes> =
66        JsValueSerdeExt::into_serde(&JsValue::from(inputs)).map_err(|err| err.to_string())?;
67    let return_value: Option<InputValue> = return_value
68        .map(|return_value| {
69            let toml_return_value = JsValueSerdeExt::into_serde(&JsValue::from(return_value))
70                .expect("could not decode return value");
71            InputValue::try_from_json(
72                toml_return_value,
73                &abi.return_type.as_ref().unwrap().abi_type,
74                MAIN_RETURN_NAME,
75            )
76        })
77        .transpose()?;
78
79    let abi_map = abi.to_btree_map();
80    let parsed_inputs: BTreeMap<String, InputValue> =
81        try_btree_map(abi_map, |(arg_name, abi_type)| {
82            // Check that toml contains a value for each argument in the ABI.
83            let value = inputs
84                .get(&arg_name)
85                .ok_or_else(|| InputParserError::MissingArgument(arg_name.clone()))?;
86            InputValue::try_from_json(value.clone(), &abi_type, &arg_name)
87                .map(|input_value| (arg_name, input_value))
88        })?;
89
90    let witness_map = abi.encode(&parsed_inputs, return_value)?;
91
92    Ok(witness_map.into())
93}

95#[wasm_bindgen(js_name = abiDecode)]
96pub fn abi_decode(abi: JsAbi, witness_map: JsWitnessMap) -> Result<JsValue, JsAbiError> {
97    console_error_panic_hook::set_once();
98    let abi: Abi =
99        JsValueSerdeExt::into_serde(&JsValue::from(abi)).map_err(|err| err.to_string())?;
100
101    let witness_map = WitnessMap::from(witness_map);
102
103    let (inputs, return_value) = abi.decode(&witness_map)?;
104
105    let abi_types = abi.to_btree_map();
106    let inputs_map: BTreeMap<String, JsonTypes> = try_btree_map(inputs, |(key, value)| {
107        JsonTypes::try_from_input_value(&value, &abi_types[&key]).map(|value| (key, value))
108    })?;
109
110    let return_value = return_value
111        .map(|value| JsonTypes::try_from_input_value(&value, &abi.return_type.unwrap().abi_type))
112        .transpose()?;
113
114    #[derive(Serialize)]
115    struct InputsAndReturn {
116        inputs: BTreeMap<String, JsonTypes>,
117        return_value: Option<JsonTypes>,
118    }
119
120    let return_struct = InputsAndReturn { inputs: inputs_map, return_value };
121    <JsValue as JsValueSerdeExt>::from_serde(&return_struct).map_err(|err| err.to_string().into())
122}

124#[wasm_bindgen(js_name = serializeWitness)]
125pub fn serialize_witness(witness_map: JsWitnessMap) -> Result<Vec<u8>, JsAbiError> {
126    console_error_panic_hook::set_once();
127    let converted_witness: WitnessMap<FieldElement> = witness_map.into();
128    let witness_stack: WitnessStack<FieldElement> = converted_witness.into();
129    let output = witness_stack.serialize();
130    output.map_err(|_| JsAbiError::new("Failed to serialize witness stack".to_string()))
131}

133#[wasm_bindgen(js_name = abiDecodeError)]
134pub fn abi_decode_error(
135    abi: JsAbi,
136    raw_error: JsRawAssertionPayload,
137) -> Result<JsValue, JsAbiError> {
138    console_error_panic_hook::set_once();
139    let mut abi: Abi =
140        JsValueSerdeExt::into_serde(&JsValue::from(abi)).map_err(|err| err.to_string())?;
141
142    let raw_error: RawAssertionPayload<String> =
143        JsValueSerdeExt::into_serde(&JsValue::from(raw_error)).map_err(|err| err.to_string())?;
144    // `FieldElement` is represented as a string in JS so we must convert these into `FieldElements` manually.
145    let error_data = raw_error
146        .data
147        .iter()
148        .map(|field| FieldElement::from_hex(field))
149        .collect::<Option<Vec<_>>>()
150        .unwrap();
151
152    let error_type = abi.error_types.remove(&raw_error.selector).expect("Missing error type");
153    match error_type {
154        AbiErrorType::FmtString { .. } => {
155            let string = display_abi_error(&error_data, error_type).to_string();
156            Ok(JsValue::from_str(&string))
157        }
158        AbiErrorType::Custom(typ) => {
159            let input_value = decode_value(&mut error_data.into_iter(), &typ)?;
160            let json_types = JsonTypes::try_from_input_value(&input_value, &typ)?;
161            <JsValue as JsValueSerdeExt>::from_serde(&json_types)
162                .map_err(|err| err.to_string().into())
163        }
164        AbiErrorType::String { string } => Ok(JsValue::from_str(&string)),
165    }
