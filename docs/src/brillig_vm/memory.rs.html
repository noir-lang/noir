1//! Implementation of the VM's memory.
2//!
3//! # Memory Addressing Limits
4//!
5//! The VM uses u32 addresses, theoretically allowing up to 2^32 memory slots. However,
6//! practical limits apply:
7//!
8//! - **Rust allocator limit**: All allocations are capped at `isize::MAX` bytes. On 32-bit
9//!   systems, this limits addressable memory to approximately `i32::MAX / sizeof(MemoryValue)`
10//!   elements (~44 million with typical element sizes).
11//!
12//! - **RAM limit**: On 64-bit systems, the allocator limit is not a concern, but allocating
13//!   the full u32 address space would require ~200 GB of RAM.
14use acir::{
15    AcirField,
16    brillig::{BitSize, IntegerBitSize, MemoryAddress},
17};
18
19use crate::assert_usize;
20
21/// The bit size used for addressing memory within the Brillig VM.
22///
23/// All memory pointers are interpreted as `u32` values, meaning the VM can directly address up to 2^32 memory slots.
24pub const MEMORY_ADDRESSING_BIT_SIZE: IntegerBitSize = IntegerBitSize::U32;
25
26/// The current stack pointer is always in slot 0.
27///
28/// It gets manipulated by opcodes laid down for calls by codegen.
29pub const STACK_POINTER_ADDRESS: MemoryAddress = MemoryAddress::Direct(0);
30
31/// The _free memory pointer_ is always in slot 1.
32///
33/// We added it here to be able to implement a workaround for wrapping around
34/// the free memory, ie. to detect "out of memory" events, but the AVM is not,
35/// and does not want to be aware of the _free memory pointer_, so we cannot,
36/// in general, build much functionality in the VM around it.
37pub const FREE_MEMORY_POINTER_ADDRESS: MemoryAddress = MemoryAddress::Direct(1);

39/// Offset constants for arrays and vectors:
40/// * Arrays are `[ref-count, ...items]`

45    pub const ARRAY_ITEMS: u32 = 1;

51    pub const VECTOR_ITEMS: u32 = 3;

73/// A single typed value in the Brillig VM's memory.
74///
75/// Memory in the VM is strongly typed and can represent either a native field element
76/// or an integer of a specific bit width. This enum encapsulates all supported
77/// in-memory types and allows conversion between representations.
78#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
79pub enum MemoryValue<F> {
80    Field(F),
81    U1(bool),
82    U8(u8),
83    U16(u16),
84    U32(u32),
85    U64(u64),
86    U128(u128),
87}
88
89/// Represents errors that can occur when interpreting or converting typed memory values.
90#[derive(Debug, thiserror::Error)]
91pub enum MemoryTypeError {
92    /// The value's bit size does not match the expected bit size for the operation.
93    #[error(
94        "Bit size for value {value_bit_size} does not match the expected bit size {expected_bit_size}"
95    )]
96    MismatchedBitSize { value_bit_size: u32, expected_bit_size: u32 },
97    /// The memory value is not an integer and cannot be interpreted as one.
98    /// For example, this can be triggered when attempting to convert a field element to an integer such as in [MemoryValue::to_u128].
99    #[error("Value is not an integer")]
100    NotAnInteger,
101}
102
103impl<F: std::fmt::Display> MemoryValue<F> {
104    /// Builds a field-typed memory value.
105    pub fn new_field(value: F) -> Self {
106        MemoryValue::Field(value)
107    }
108
109    /// Builds an integer-typed memory value.
110    pub fn new_integer(value: u128, bit_size: IntegerBitSize) -> Self {
111        match bit_size {
112            IntegerBitSize::U1 => MemoryValue::U1(value != 0),
113            IntegerBitSize::U8 => {
114                MemoryValue::U8(value.try_into().expect("{value} is out of 8 bits range"))
115            }
116            IntegerBitSize::U16 => {
117                MemoryValue::U16(value.try_into().expect("{value} is out of 16 bits range"))
118            }
119            IntegerBitSize::U32 => {
120                MemoryValue::U32(value.try_into().expect("{value} is out of 32 bits range"))
121            }
122            IntegerBitSize::U64 => {
123                MemoryValue::U64(value.try_into().expect("{value} is out of 64 bits range"))
124            }
125            IntegerBitSize::U128 => MemoryValue::U128(value),
126        }
127    }
128
129    pub fn bit_size(&self) -> BitSize {
130        match self {
131            MemoryValue::Field(_) => BitSize::Field,
132            MemoryValue::U1(_) => BitSize::Integer(IntegerBitSize::U1),
133            MemoryValue::U8(_) => BitSize::Integer(IntegerBitSize::U8),
134            MemoryValue::U16(_) => BitSize::Integer(IntegerBitSize::U16),
135            MemoryValue::U32(_) => BitSize::Integer(IntegerBitSize::U32),
136            MemoryValue::U64(_) => BitSize::Integer(IntegerBitSize::U64),
137            MemoryValue::U128(_) => BitSize::Integer(IntegerBitSize::U128),
138        }
139    }
140
141    /// Expects a `U32` value and converts it into `usize`, otherwise panics.
142    ///
143    /// Primarily a convenience method for using values in memory operations as pointers, sizes and offsets.
144    pub fn to_u32(&self) -> u32 {
145        match self {
146            MemoryValue::U32(value) => *value,
147            other => panic!("value is not typed as Brillig usize: {other}"),
148        }
149    }
150}

152impl<F: AcirField> MemoryValue<F> {
153    /// Builds a memory value from a field element, either field or integer type.
154    ///
155    /// If the bit size indicates an integer type, the value is downcast to fit into the specified size.
156    pub fn new_from_field(value: F, bit_size: BitSize) -> Self {
157        if let BitSize::Integer(bit_size) = bit_size {
158            MemoryValue::new_integer(value.to_u128(), bit_size)
159        } else {
160            MemoryValue::new_field(value)
161        }
162    }
163
164    /// Builds a memory value from a field element, checking that the value is within the bit size,
165    /// otherwise returns `None`.
166    pub fn new_checked(value: F, bit_size: BitSize) -> Option<Self> {
167        if let BitSize::Integer(bit_size) = bit_size {
168            if value.num_bits() > bit_size.into() {
169                return None;
170            }
171        }
172
173        Some(MemoryValue::new_from_field(value, bit_size))
174    }
175
176    /// Converts the memory value to a field element, independent of its type.
177    pub fn to_field(&self) -> F {
178        match self {
179            MemoryValue::Field(value) => *value,
180            MemoryValue::U1(value) => F::from(*value),
181            MemoryValue::U8(value) => F::from(u128::from(*value)),
182            MemoryValue::U16(value) => F::from(u128::from(*value)),
183            MemoryValue::U32(value) => F::from(u128::from(*value)),
184            MemoryValue::U64(value) => F::from(u128::from(*value)),
185            MemoryValue::U128(value) => F::from(*value),
186        }
187    }
188
189    /// Converts the memory value to U128, if the value is an integer.
190    pub fn to_u128(&self) -> Result<u128, MemoryTypeError> {
191        match self {
192            MemoryValue::Field(..) => Err(MemoryTypeError::NotAnInteger),
193            MemoryValue::U1(value) => Ok(u128::from(*value)),
194            MemoryValue::U8(value) => Ok(u128::from(*value)),
195            MemoryValue::U16(value) => Ok(u128::from(*value)),
196            MemoryValue::U32(value) => Ok(u128::from(*value)),
197            MemoryValue::U64(value) => Ok(u128::from(*value)),
198            MemoryValue::U128(value) => Ok(*value),
199        }
200    }
201
202    /// Extracts the field element from the memory value, if it is typed as field element.
203    pub fn expect_field(self) -> Result<F, MemoryTypeError> {
204        if let MemoryValue::Field(field) = self {
205            Ok(field)
206        } else {
207            Err(MemoryTypeError::MismatchedBitSize {
208                value_bit_size: self.bit_size().to_u32::<F>(),
209                expected_bit_size: F::max_num_bits(),
210            })
211        }
212    }
213    pub(crate) fn expect_u1(self) -> Result<bool, MemoryTypeError> {
214        if let MemoryValue::U1(value) = self {
215            Ok(value)
216        } else {
217            Err(MemoryTypeError::MismatchedBitSize {
218                value_bit_size: self.bit_size().to_u32::<F>(),
219                expected_bit_size: 1,
220            })
221        }
222    }
223
224    pub(crate) fn expect_u8(self) -> Result<u8, MemoryTypeError> {
225        if let MemoryValue::U8(value) = self {
226            Ok(value)
227        } else {
228            Err(MemoryTypeError::MismatchedBitSize {
229                value_bit_size: self.bit_size().to_u32::<F>(),

231            })
232        }
233    }
234
235    pub(crate) fn expect_u16(self) -> Result<u16, MemoryTypeError> {
236        if let MemoryValue::U16(value) = self {
237            Ok(value)
238        } else {
239            Err(MemoryTypeError::MismatchedBitSize {

243        }

246    pub(crate) fn expect_u32(self) -> Result<u32, MemoryTypeError> {

369/// The VM's memory.
370///
371/// Memory is internally represented as a vector of values.
372/// We grow the memory when values past the end are set, extending with 0s.
373///
374/// # Capacity Limits
375///
376/// The inner `Vec` is subject to Rust's allocator limit of `isize::MAX` bytes.
377/// This means:
378/// - On 64-bit: Practical limit is available RAM (~200 GB for full u32 range)
379/// - On 32-bit: Hard limit of ~44 million addressable slots
380///
381/// Exceeding these limits will cause a panic with "capacity overflow".
382#[derive(Debug, Clone, Default, PartialEq, Eq)]
383pub struct Memory<F> {
384    // Internal memory representation
385    inner: Vec<MemoryValue<F>>,
386}
387
388impl<F: AcirField> Memory<F> {
389    /// Read the value from slot 0.
390    ///
391    /// Panics if it's not a `U32`.
392    fn get_stack_pointer(&self) -> u32 {
393        self.read(STACK_POINTER_ADDRESS).to_u32()
394    }
395
396    /// Resolve an address to either:
397    /// * itself, if it's a direct address, or
398    /// * the current stack pointer plus the offset, if it's relative.
399    ///
400    /// Returns a memory slot index.
401    fn resolve(&self, address: MemoryAddress) -> u32 {
402        match address {
403            MemoryAddress::Direct(address) => address,
404            MemoryAddress::Relative(offset) => self.get_stack_pointer() + offset,
405        }
406    }
407
408    /// Reads the numeric value at the address.
409    ///
410    /// If the address is beyond the size of memory, a default value is returned.
411    pub fn read(&self, address: MemoryAddress) -> MemoryValue<F> {
412        let resolved_addr = assert_usize(self.resolve(address));
413        self.inner.get(resolved_addr).copied().unwrap_or_default()
414    }
415
416    /// Reads the value at the address and returns it as a direct memory address,
417    /// without dereferencing the pointer itself to a numeric value.
418    pub fn read_ref(&self, ptr: MemoryAddress) -> MemoryAddress {
419        MemoryAddress::direct(self.read(ptr).to_u32())
420    }
421
422    /// Sets `ptr` to point at `address`.
423    pub fn write_ref(&mut self, ptr: MemoryAddress, address: MemoryAddress) {
424        self.write(ptr, MemoryValue::from(address.to_u32()));
425    }
426
427    /// Read a contiguous vector of memory starting at `address`, up to `len` slots.
428    ///
429    /// Panics if the end index is beyond the size of the memory.
430    pub fn read_slice(&self, address: MemoryAddress, len: usize) -> &[MemoryValue<F>] {
431        // Allows to read a vector of uninitialized memory if the length is zero.
432        // Ideally we'd be able to read uninitialized memory in general (as read does)
433        // but that's not possible if we want to return a vector instead of owned data.
434        if len == 0 {
435            return &[];
436        }
437        let resolved_addr = assert_usize(self.resolve(address));
438        &self.inner[resolved_addr..(resolved_addr + len)]
439    }
440
441    /// Sets the value at `address` to `value`
442    pub fn write(&mut self, address: MemoryAddress, value: MemoryValue<F>) {
443        let resolved_addr = assert_usize(self.resolve(address));
444        self.resize_to_fit(resolved_addr + 1);
445        self.inner[resolved_addr] = value;
446    }
447
448    /// Maximum number of memory slots that can be allocated.
449    ///
450    /// This limit is set to `i32::MAX` to ensure deterministic behavior across all architectures.
451    /// On 32-bit systems, Rust's allocator limits allocations to `isize::MAX` bytes, which would
452    /// restrict us to fewer elements anyway. By using `i32::MAX`, we ensure the same behavior
453    /// on both 32-bit and 64-bit systems.
454    ///
455    /// See: <https://github.com/rust-lang/rust/pull/95295> and <https://doc.rust-lang.org/1.81.0/src/core/alloc/layout.rs.html>
456    const MAX_MEMORY_SIZE: usize = i32::MAX as usize;
457
458    /// Increase the size of memory fit `size` elements, or the current length, whichever is bigger.
459    ///
460    /// # Panics
461    ///
462    /// Panics if `size` exceeds [`Self::MAX_MEMORY_SIZE`].
463    fn resize_to_fit(&mut self, size: usize) {
464        assert!(
465            size <= Self::MAX_MEMORY_SIZE,
466            "Memory address space exceeded: requested {size} slots, maximum is {} (i32::MAX)",
467            Self::MAX_MEMORY_SIZE
468        );
469        // Calculate new memory size
470        let new_size = std::cmp::max(self.inner.len(), size);
471        // Expand memory to new size with default values if needed
472        self.inner.resize(new_size, MemoryValue::default());
473    }
474
475    /// Sets the values after `address` to `values`
476    pub fn write_slice(&mut self, address: MemoryAddress, values: &[MemoryValue<F>]) {
477        let resolved_addr = assert_usize(self.resolve(address));
478        let end_addr = resolved_addr + values.len();
479        self.resize_to_fit(end_addr);
480        self.inner[resolved_addr..end_addr].copy_from_slice(values);
481    }
482
483    /// Returns the values of the memory
484    pub fn values(&self) -> &[MemoryValue<F>] {
485        &self.inner
486    }
487}
