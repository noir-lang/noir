18    /// Handles the execution of a single [ForeignCall opcode][acir::brillig::Opcode::ForeignCall].
19    ///
20    /// This method performs the following steps:
21    /// 1. Checks if the foreign call results are already available. If not, it resolves the input
22    ///    values from memory and pauses execution by returning `VMStatus::ForeignCallWait`.
23    ///    For vectors, the preceding `u32` length field is used to truncate the vector input to its semantic length.
24    /// 2. If results are available, it writes them to memory, ensuring that the returned data
25    ///    matches the expected types and sizes:
26    ///     * Nested arrays are reconstructed from flat outputs when necessary.
27    ///     * Nested vectors are an unsupported return type and will trigger an error.
28    ///     * Vectors are written to the heap starting at the free memory pointer, and their address gets stored in the destination.
29    ///     * Update free memory pointer based on how much data (if any) was written to it.
30    /// 3. Increments the foreign call counter and advances the program counter.
31    ///
32    /// # Parameters
33    /// The borrowed fields of a [ForeignCall opcode][acir::brillig::Opcode::ForeignCall].
34    /// They are listed again below:
35    /// - `function`: Name of the foreign function being called.
36    /// - `destinations`: Pointers or heap structures where the return values will be written.
37    /// - `destination_value_types`: Expected type layout for each destination.
38    /// - `inputs`: Pointers or heap structures representing the inputs for the foreign call.
39    /// - `input_value_types`: Expected type layout for each input.
40    ///
41    /// # Returns
42    /// - [VMStatus] indicating the next state of the VM:
43    ///   - [VMStatus::ForeignCallWait] if the results are not yet available.
44    ///   - [VMStatus::Finished] or [VMStatus::Failure] depending on whether writing the results succeeded.
45    ///
46    /// # Panics
47    /// - If `inputs` and `input_value_types` lengths do not match.
48    /// - If `destinations` and `destination_value_types` lengths do not match.
49    pub(super) fn process_foreign_call(
50        &mut self,
51        function: &str,
52        destinations: &[ValueOrArray],
53        destination_value_types: &[HeapValueType],
54        inputs: &[ValueOrArray],
55        input_value_types: &[HeapValueType],
56    ) -> &VMStatus<F> {
57        assert_eq!(inputs.len(), input_value_types.len());
58        assert_eq!(destinations.len(), destination_value_types.len());
59
60        if !self.has_unprocessed_foreign_call_result() {
61            // When this opcode is called, it is possible that the results of a foreign call are
62            // not yet known (not enough entries in `foreign_call_results`).
63            // If that is the case, just resolve the inputs and pause the VM with a status
64            // (VMStatus::ForeignCallWait) that communicates the foreign function name and
65            // resolved inputs back to the caller. Once the caller pushes to `foreign_call_results`,
66            // they can then make another call to the VM that starts at this opcode
67            // but has the necessary results to proceed with execution.
68
69            // With vectors we might have more items in the HeapVector than the semantic length
70            // indicated by the field preceding the pointer to the vector in the inputs.
71            // This happens when SSA merges vectors of different length, which can result in
72            // a vector that has room for the longer of the two, partially filled with items
73            // from the shorter. There are ways to deal with this on the receiver side,
74            // but it is cumbersome, and the cleanest solution is not to send the extra empty
75            // items at all. To do this, however, we need infer which input is the vector length.
76            let mut vector_length: Option<u32> = None;
77
78            let resolved_inputs = inputs
79                .iter()
80                .zip(input_value_types)
81                .map(|(input, input_type)| {
82                    let mut input = self.get_memory_values(*input, input_type);
83                    // Truncate vectors to their semantic length, which we remember from the preceding field.
84                    match input_type {
85                        HeapValueType::Simple(BitSize::Integer(IntegerBitSize::U32)) => {
86                            // If we have a single u32 we may have a vector representation, so store this input.
87                            // On the next iteration, if we have a vector then we know we have the dynamic length
88                            // for that vector.
89                            let ForeignCallParam::Single(length) = input else {
90                                unreachable!("expected u32; got {input:?}");
91                            };
92                            vector_length = Some(length.to_u128() as u32);
93                        }
94                        HeapValueType::Vector { value_types } => {
95                            if let Some(length) = vector_length {
96                                let flattened_length =
97                                    vector_flattened_length(value_types, SemanticLength(length));
98                                let mut fields = input.fields();
99                                fields.truncate(assert_usize(flattened_length.0));
100                                input = ForeignCallParam::Array(fields);
101                            }
102                            vector_length = None;
103                        }
104                        _ => {
105                            // Otherwise we are not dealing with a u32 followed by a vector.
106                            vector_length = None;
107                        }
108                    }
109                    input
110                })
111                .collect::<Vec<_>>();
112
113            return self.wait_for_foreign_call(function.to_owned(), resolved_inputs);
114        }
115
116        let write_result = self.write_foreign_call_result(
117            destinations,
118            destination_value_types,
119            self.foreign_call_counter,
120        );
121
122        if let Err(e) = write_result {
123            return self.fail(e);
124        }
125
126        // Mark the foreign call result as processed.
127        self.foreign_call_counter += 1;
128        self.increment_program_counter()
129    }

131    /// Get input data from memory to pass to foreign calls.
132    fn get_memory_values(
133        &self,
134        input: ValueOrArray,
135        value_type: &HeapValueType,
136    ) -> ForeignCallParam<F> {
137        match (input, value_type) {
138            (ValueOrArray::MemoryAddress(value_addr), HeapValueType::Simple(_)) => {
139                ForeignCallParam::Single(self.memory.read(value_addr).to_field())
140            }
141            (
142                ValueOrArray::HeapArray(HeapArray { pointer, size }),
143                HeapValueType::Array { value_types, size: type_size },
144            ) => {
145                // The array's semi-flattened size must match the expected size
146                let semi_flattened_size =
147                    *type_size * ElementTypesLength(assert_u32(value_types.len()));
148                assert_eq!(semi_flattened_size, size);
149
150                let start = self.memory.read_ref(pointer);
151                self.read_slice_of_values_from_memory(start, size, value_types)
152                    .into_iter()
153                    .map(|mem_value| mem_value.to_field())
154                    .collect::<Vec<_>>()
155                    .into()
156            }
157            (
158                ValueOrArray::HeapVector(HeapVector { pointer, size: size_addr }),
159                HeapValueType::Vector { value_types },
160            ) => {
161                let start = self.memory.read_ref(pointer);
162                let size = self.memory.read(size_addr).to_u32();
163                let size = SemiFlattenedLength(size);
164                self.read_slice_of_values_from_memory(start, size, value_types)
165                    .into_iter()
166                    .map(|mem_value| mem_value.to_field())
167                    .collect::<Vec<_>>()
168                    .into()
169            }
170            _ => {
171                unreachable!("Unexpected value type {value_type:?} for input {input:?}");
172            }
173        }
174    }

176    /// Reads an array/vector from memory but recursively reads pointers to
177    /// nested arrays/vectors according to the sequence of value types.
178    ///
179    /// The given `size` is the total number of `HeapValueType`s to read, which must
180    /// be a multiple of the length of `value_types` (unless `value_types.len()` is 0).
181    fn read_slice_of_values_from_memory(
182        &self,
183        start: MemoryAddress,
184        size: SemiFlattenedLength,
185        value_types: &[HeapValueType],
186    ) -> Vec<MemoryValue<F>> {
187        let size = size.0;
188
189        assert!(start.is_direct(), "read_slice_of_values_from_memory requires direct addresses");
190        if HeapValueType::all_simple(value_types) {
191            self.memory.read_slice(start, assert_usize(size)).to_vec()
192        } else {
193            // Check that the sequence of value types fit an integer number of
194            // times inside the given size.
195            assert!(
196                0 == size % assert_u32(value_types.len()),
197                "array/vector does not contain a whole number of elements"
198            );
199
200            (0..size)
201                .zip(value_types.iter().cycle())
202                .flat_map(|(i, value_type)| {
203                    let value_address = start.offset(i);
204                    match value_type {
205                        HeapValueType::Simple(_) => {
206                            vec![self.memory.read(value_address)]
207                        }
208                        HeapValueType::Array { value_types, size: type_size } => {
209                            let array_address =
210                                ArrayAddress::from(self.memory.read_ref(value_address));
211                            let semi_flattened_size =
212                                *type_size * ElementTypesLength(assert_u32(value_types.len()));
213
214                            self.read_slice_of_values_from_memory(
215                                array_address.items_start(),
216                                semi_flattened_size,
217                                value_types,
218                            )
219                        }
220                        HeapValueType::Vector { .. } => {
221                            unreachable!("Vectors nested in arrays/vectors are not supported");
222                        }
223                    }
224                })
225                .collect::<Vec<_>>()
226        }
227    }

229    /// Sets the status of the VM to `ForeignCallWait`.
230    /// Indicating that the VM is now waiting for a foreign call to be resolved.
231    fn wait_for_foreign_call(
232        &mut self,
233        function: String,
234        inputs: Vec<ForeignCallParam<F>>,
235    ) -> &VMStatus<F> {
236        self.status(VMStatus::ForeignCallWait { function, inputs })
237    }

239    /// Write a foreign call's results to the VM memory.
240    ///
241    /// We match the expected types with the actual results.
242    /// However foreign call results do not support nested structures:
243    /// They are either a single integer value or a vector of integer values (field elements).
244    /// Therefore, nested arrays returned from foreign call results are flattened.
245    /// If the expected array sizes do not match the actual size, we reconstruct the nested
246    /// structure from the flat output array.
247    fn write_foreign_call_result(
248        &mut self,
249        destinations: &[ValueOrArray],
250        destination_value_types: &[HeapValueType],
251        foreign_call_index: usize,
252    ) -> Result<(), String> {
253        // Take ownership of values to allow calling mutating methods on self.
254        let values = std::mem::take(&mut self.foreign_call_results[foreign_call_index].values);
255
256        if destinations.len() != values.len() {
257            return Err(format!(
258                "{} output values were provided as a foreign call result for {} destination slots",
259                values.len(),
260                destinations.len()
261            ));
262        }
263
264        assert_eq!(
265            destinations.len(),
266            destination_value_types.len(),
267            "Number of destinations must match number of value types",
268        );
269
270        for ((destination, value_type), output) in
271            destinations.iter().zip(destination_value_types).zip(&values)
272        {
273            match (destination, value_type) {
274                (ValueOrArray::MemoryAddress(value_addr), HeapValueType::Simple(bit_size)) => {
275                    let output_fields = output.fields();
276                    if value_type.flattened_size().is_some_and(|flattened_size| {
277                        FlattenedLength(assert_u32(output_fields.len())) != flattened_size
278                    }) {
279                        return Err(format!(
280                            "Foreign call return value does not match expected size. Expected {} but got {}",
281                            value_type.flattened_size().unwrap(),
282                            output_fields.len(),
283                        ));
284                    }
285
286                    match output {
287                        ForeignCallParam::Single(value) => {
288                            self.write_value_to_memory(*value_addr, value, *bit_size)?;
289                        }
290                        _ => {
291                            return Err(format!(
292                                "Function result size does not match brillig bytecode. Expected 1 result but got {output:?}"
293                            ));
294                        }
295                    }
296                }
297                (
298                    ValueOrArray::HeapArray(HeapArray { pointer, size }),
299                    HeapValueType::Array { value_types, size: type_size },
300                ) => {
301                    if *type_size * ElementTypesLength(assert_u32(value_types.len())) != *size {
302                        return Err(format!(
303                            "Destination array size of {size} does not match the type size of {type_size}"
304                        ));
305                    }
306                    let output_fields = output.fields();
307                    if value_type.flattened_size().is_some_and(|flattened_size| {
308                        FlattenedLength(assert_u32(output_fields.len())) != flattened_size
309                    }) {
310                        return Err(format!(
311                            "Foreign call return value does not match expected size. Expected {} but got {}",
312                            value_type.flattened_size().unwrap(),
313                            output_fields.len(),
314                        ));
315                    }
316
317                    if HeapValueType::all_simple(value_types) {
318                        let ForeignCallParam::Array(values) = output else {
319                            return Err("Foreign call returned a single value for an array type"
320                                .to_string());
321                        };
322                        assert_eq!(
323                            values.len(),
324                            size.0 as usize,
325                            "Expected values length to be equal to heap array size",
326                        );
327                        self.write_values_to_memory(*pointer, values, value_types)?;
328                    } else {
329                        // foreign call returning flattened values into a nested type, so the sizes do not match
330                        let destination = self.memory.read_ref(*pointer);
331                        let mut flatten_values_idx = 0; //index of values read from flatten_values
332                        self.write_flattened_values_to_memory(
333                            destination,
334                            &output_fields,
335                            &mut flatten_values_idx,
336                            value_type,
337                        )?;
338                        assert_eq!(
339                            flatten_values_idx,
340                            output_fields.len(),
341                            "Not all values were written to memory"
342                        );
343                    }
344                }
345                // We didn't know the length of vectors when we allocated the destination variable, so we pointed
346                // the vector at the start of the free memory; with this technique we can only handle a single vector.
347                // Write the data where the destination points at. It is up to the follow up bytecode to initialize
348                // the meta-data, or move the data somewhere else.
349                (
350                    ValueOrArray::HeapVector(HeapVector { pointer, size: size_addr }),
351                    HeapValueType::Vector { value_types },
352                ) => {
353                    if HeapValueType::all_simple(value_types) {
354                        let ForeignCallParam::Array(values) = output else {
355                            return Err("Foreign call returned a single value for an vector type"
356                                .to_string());
357                        };
358                        if values.len() % value_types.len() != 0 {
359                            return Err(
360                                "Returned data does not match vector element size".to_string()
361                            );
362                        }
363                        // Set the size in the size address.
364                        // Note that unlike `pointer`, we don't treat `size` as a pointer here, even though it is;
365                        // instead we expect the post-call codegen will copy it to the heap.
366                        self.memory.write(*size_addr, assert_u32(values.len()).into());
367                        self.write_values_to_memory(*pointer, values, value_types)?;
368                    } else {
369                        unreachable!("deflattening heap vectors from foreign calls");
370                    }
371                }
372                _ => {
373                    return Err(format!(
374                        "Unexpected value type {value_type:?} for destination {destination:?}"
375                    ));
376                }
377            }
378        }
379
380        self.foreign_call_results[foreign_call_index].values = values;
381
382        Ok(())
383    }

385    /// Write a single numeric value to the destination address, ensuring that the bit size matches the expectation.
386    fn write_value_to_memory(
387        &mut self,
388        destination: MemoryAddress,
389        value: &F,
390        value_bit_size: BitSize,
391    ) -> Result<(), String> {
392        let memory_value = MemoryValue::new_checked(*value, value_bit_size);
393
394        if let Some(memory_value) = memory_value {
395            self.memory.write(destination, memory_value);
396        } else {
397            return Err(format!(
398                "Foreign call result value {value} does not fit in bit size {value_bit_size:?}"
399            ));
400        }
401        Ok(())
402    }

404    /// Write the `values` of an array or vector to an address stored under the `pointer`.
405    fn write_values_to_memory(
406        &mut self,
407        pointer: MemoryAddress,
408        values: &[F],
409        value_types: &[HeapValueType],
410    ) -> Result<(), String> {
411        let bit_sizes_iterator = value_types
412            .iter()
413            .map(|typ| match typ {
414                HeapValueType::Simple(bit_size) => *bit_size,
415                _ => unreachable!("Expected simple value type"),
416            })
417            .cycle();
418
419        // Convert the destination pointer to an address.
420        let destination = self.memory.read_ref(pointer);
421
422        // Write to the destination memory.
423        let memory_values: Option<Vec<_>> = values
424            .iter()
425            .zip(bit_sizes_iterator)
426            .map(|(value, bit_size)| MemoryValue::new_checked(*value, bit_size))
427            .collect();
428        if let Some(memory_values) = memory_values {
429            self.memory.write_slice(destination, &memory_values);
430        } else {
431            return Err(format!(
432                "Foreign call result values {values:?} do not match expected bit sizes",
433            ));
434        }
435        Ok(())
436    }

438    /// Writes flattened values to memory, using the provided type.
439    ///
440    /// The method calls itself recursively in order to work with recursive types (nested arrays).
441    /// `values_idx` is the current index in the `values` vector and is incremented every time
442    /// a value is written to memory.
443    fn write_flattened_values_to_memory(
444        &mut self,
445        destination: MemoryAddress,
446        values: &Vec<F>,
447        values_idx: &mut usize,
448        value_type: &HeapValueType,
449    ) -> Result<(), String> {
450        assert!(
451            destination.is_direct(),
452            "write_flattened_values_to_memory requires direct addresses"
453        );
454        match value_type {
455            HeapValueType::Simple(bit_size) => {
456                self.write_value_to_memory(destination, &values[*values_idx], *bit_size)?;
457                *values_idx += 1;
458                Ok(())
459            }
460            HeapValueType::Array { value_types, size } => {
461                let mut current_pointer = destination;
462                let size = size.0;
463                for _ in 0..size {
464                    for typ in value_types {
465                        match typ {
466                            HeapValueType::Simple(bit_size) => {
467                                self.write_value_to_memory(
468                                    current_pointer,
469                                    &values[*values_idx],
470                                    *bit_size,
471                                )?;
472                                *values_idx += 1;
473                                current_pointer = current_pointer.offset(1);
474                            }
475                            HeapValueType::Array { .. } => {
476                                // The next memory destination is an array, somewhere else in memory where the pointer points to.
477                                let destination =
478                                    ArrayAddress::from(self.memory.read_ref(current_pointer));
479
480                                self.write_flattened_values_to_memory(
481                                    destination.items_start(),
482                                    values,
483                                    values_idx,
484                                    typ,
485                                )?;
486
487                                // Move on to the next slot in *this* array.
488                                current_pointer = current_pointer.offset(1);
489                            }
490                            HeapValueType::Vector { .. } => {
491                                return Err(format!(
492                                    "Unsupported returned type in foreign calls {typ:?}"
493                                ));
494                            }
495                        }
496                    }
497                }
498                Ok(())
499            }
500            HeapValueType::Vector { .. } => {
501                Err(format!("Unsupported returned type in foreign calls {value_type:?}"))
502            }
503        }
504    }
505}

507/// Returns the total number of field elements required to represent the elements in the vector in memory.
508///
509/// Panics if the vector contains nested vectors. Such types are not supported and are rejected by the frontend.
510fn vector_flattened_length(
511    value_types: &[HeapValueType],
512    length: SemanticLength,
513) -> FlattenedLength {
514    let elements_flattened_length: FlattenedLength = value_types
515        .iter()
516        .map(|typ| {
517            typ.flattened_size()
518                .unwrap_or_else(|| panic!("unexpected nested dynamic element type: {typ:?}"))
519        })
520        .sum();
521    ElementsFlattenedLength::from(elements_flattened_length) * length
522}
