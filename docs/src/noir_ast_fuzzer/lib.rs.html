21/// AST generation configuration.
22#[derive(Debug, Clone)]
23pub struct Config {
24    /// Maximum number of global definitions.
25    pub max_globals: usize,
26    /// Minimum number of functions (other than main) to generate.
27    pub min_functions: usize,
28    /// Maximum number of functions (other than main) to generate.
29    pub max_functions: usize,
30    /// Maximum number of arguments a function can have.
31    pub max_function_args: usize,
32    /// Maximum number of statements to aim for in a function body.
33    pub max_function_size: usize,
34    /// Maximum number of statements in a block.
35    pub max_block_size: usize,
36    /// Maximum nesting depth for complex expressions.
37    pub max_depth: usize,
38    /// Maximum number of fields for tuples.
39    pub max_tuple_size: usize,
40    /// Maximum size for arrays.
41    pub max_array_size: usize,
42    /// Maximum size of for loop ranges, which affects unrolling in ACIR.
43    pub max_loop_size: usize,
44    /// Whether to choose the backstop for `loop` and `while` randomly.
45    pub vary_loop_size: bool,
46    /// Maximum number of recursive calls to make at runtime.
47    pub max_recursive_calls: usize,
48    /// Maximum number of match cases.
49    pub max_match_cases: usize,
50    /// Frequency of expressions, which produce a value.
51    pub expr_freqs: Freqs,
52    /// Frequency of statements in ACIR functions.
53    pub stmt_freqs_acir: Freqs,
54    /// Frequency of statements in Brillig functions.
55    pub stmt_freqs_brillig: Freqs,
56    /// Whether to force all functions to be unconstrained.
57    pub force_brillig: bool,
58    /// Try to avoid overflowing operations. Useful when testing the minimal pipeline,
59    /// to avoid trivial failures due to multiplying or adding constants.
60    pub avoid_overflow: bool,
61    /// Try to avoid "Index out of bounds" by using modulo to limit indexing to the
62    /// range that an array or vector is expected to contain.
63    ///
64    /// This is easy to trigger (a random `u32` will most certainly be out of the range
65    /// of the arrays we generate), so by default it is "on". When it's "off", a random
66    /// decision is taken for each index operation whether to apply the modulo or not.
67    pub avoid_index_out_of_bounds: bool,
68    /// Try to avoid operations that can result in error when zero is on the RHS.
69    pub avoid_err_by_zero: bool,
70    /// Avoid using negative integer literals where the frontend expects unsigned types.
71    pub avoid_negative_int_literals: bool,
72    /// Avoid using large integer literals where the frontend expects 32 bits.
73    pub avoid_large_int_literals: bool,
74    /// Avoid using loop control (break/continue).
75    pub avoid_loop_control: bool,
76    /// Avoid using function pointers in parameters.
77    pub avoid_lambdas: bool,
78    /// Avoid print statements.
79    pub avoid_print: bool,
80    /// Avoid using constrain statements.
81    pub avoid_constrain: bool,
82    /// Avoid match statements and expressions.
83    pub avoid_match: bool,
84    /// Avoid using the vector type.
85    pub avoid_vectors: bool,
86    /// Only use comptime friendly expressions.
87    pub comptime_friendly: bool,
88}
