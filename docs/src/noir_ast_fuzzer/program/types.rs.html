15/// Calculate the depth of a type.
16///
17/// Leaf types have a depth of 0.
18pub fn type_depth(typ: &Type) -> usize {
19    match typ {
20        Type::Field | Type::Bool | Type::String(_) | Type::Unit | Type::Integer(_, _) => 0,
21        Type::Array(_, typ) | Type::Vector(typ) => 1 + type_depth(typ),
22        Type::Tuple(types) => 1 + types.iter().map(type_depth).max().unwrap_or_default(),
23        Type::Reference(typ, _) => 1 + type_depth(typ.as_ref()),
24        _ => unreachable!("unexpected type: {typ}"),
25    }
26}

28/// We can only use globals that can be evaluated at comptime.
29/// Some types don't compile in Noir, so avoid those as we couldn't
30/// put any related failures into an integration test.
31pub fn can_be_global(typ: &Type) -> bool {
32    !matches!(
33        typ,
34        Type::Integer(Signedness::Signed, IntegerBitSize::One | IntegerBitSize::HundredTwentyEight)
35            | Type::Integer(Signedness::Unsigned, IntegerBitSize::One)
36    )
37}

39/// Check if a type can be used in the `main` function.
40///
41/// We decided we will avoid 0 length arrays in the main inputs and outputs, because we don't generate
42/// witnesses for them anyway, and they are tricky to handle consistently when they can be regular inputs
43/// as well as part of the databus. They are not expected in real programs as they don't do anything useful.
44pub fn can_be_main(typ: &Type) -> bool {
45    match typ {
46        Type::String(size) => *size > 0,
47        Type::Array(size, typ) => *size > 0 && can_be_main(typ),
48        Type::Tuple(types) => types.iter().all(can_be_main),
49        Type::Bool | Type::Field | Type::Integer(_, _) => true,
50        _ => false,
51    }
52}

59/// Collect all the sub-types produced by a type.
60///
61/// It's like a _power set_ of the type.
62pub fn types_produced(typ: &Type) -> HashSet<Type> {
63    /// Recursively visit subtypes.
64    fn visit(acc: &mut HashSet<Type>, typ: &Type) {
65        if acc.contains(typ) {
66            return;
67        }
68
69        // Trivially produce self.
70        acc.insert(typ.clone());
71
72        match typ {
73            Type::Array(len, item_type) => {
74                if *len > 0 {
75                    visit(acc, item_type);
76                }
77                // Technically we could produce `[T; N]` from `[S; N]` if
78                // we can produce `T` from `S`, but let's ignore that;
79                // instead we will produce `[T; N]` from any source that can
80                // supply `T`, one of which would be the `[S; N]` itself.
81                // So if we have `let foo = [1u32, 2u32];` and we need `[u64; 2]`
82                // we might generate `[foo[1] as u64, 3u64]` instead of "mapping"
83                // over the entire foo. Same goes for tuples.
84            }
85            Type::Tuple(item_types) => {
86                for item_type in item_types {
87                    visit(acc, item_type);
88                }
89            }
90            Type::String(_) => {
91                // Maybe it could produce substrings, but it would be an overkill to enumerate.
92            }
93            Type::Field => {
94                // There are `try_to_*` methods, but let's consider only what is safe.
95                acc.insert(Type::Integer(Signedness::Unsigned, IntegerBitSize::HundredTwentyEight));
96            }
97            Type::Integer(sign, integer_bit_size) => {
98                // Casting up is safe.
99                for size in IntegerBitSize::iter()
100                    .filter(|size| size.bit_size() > integer_bit_size.bit_size())
101                {
102                    // We don't want to produce `i1` or `i128`
103                    if sign.is_signed()
104                        && (size == IntegerBitSize::One
105                            || size == IntegerBitSize::HundredTwentyEight)
106                    {
107                        continue;
108                    }
109
110                    acc.insert(Type::Integer(*sign, size));
111                }
112                // There are `From<u*>` for Field
113                if !sign.is_signed() {
114                    acc.insert(Type::Field);
115                }
116            }
117            Type::Bool => {
118                // Maybe we can also cast to u1 or u8 etc?
119                acc.insert(Type::Field);
120            }
121            Type::Vector(item_type) => {
122                // pop_front -> (T, [T])
123                acc.insert(Type::Tuple(vec![item_type.as_ref().clone(), typ.clone()]));
124                // pop_back -> ([T], T)
125                acc.insert(Type::Tuple(vec![typ.clone(), item_type.as_ref().clone()]));
126                visit(acc, item_type);
127            }
128            Type::Reference(typ, _) => {
129                visit(acc, typ);
130            }
131            Type::Function(_, ret, _, _) => {
132                visit(acc, ret);
133            }
134            Type::Unit | Type::FmtString(_, _) => {}
135        }
136    }
137
138    let mut acc = HashSet::new();
139    visit(&mut acc, typ);
140    acc
141}

183/// Check if the type contains any references.
184pub fn contains_reference(typ: &Type) -> bool {
185    match typ {
186        Type::Reference(_, _) => true,
187        Type::Field
188        | Type::Integer(_, _)
189        | Type::Bool
190        | Type::String(_)
191        | Type::Unit
192        | Type::FmtString(_, _)
193        | Type::Function(_, _, _, _) => false,
194        Type::Array(_, typ) | Type::Vector(typ) => contains_reference(typ),
195        Type::Tuple(types) => types.iter().any(contains_reference),
196    }
197}

215/// Check if the type can be used with a `println` statement.
216pub fn is_printable(typ: &Type) -> bool {
217    match typ {
218        Type::Reference(_, _) => false,
219        Type::Field | Type::Integer(_, _) | Type::String(_) | Type::Bool | Type::Unit => true,
220        Type::Vector(typ) | Type::Array(_, typ) | Type::FmtString(_, typ) => is_printable(typ),
221        Type::Tuple(types) => types.iter().all(is_printable),
222        // Function signatures are printable, although they might differ when we force things to be Brillig.
223        Type::Function(_, _, _, _) => true,
224    }

227/// Can the type be returned by some `UnaryOp`.
228pub fn can_unary_return(typ: &Type) -> bool {
229    match typ {
230        Type::Field => true,
231        Type::Bool => true,
232        Type::Integer(sign, size) => {
233            // What can we apply `UnaryOp::Minus` to.
234            // The number has to be signed, otherwise it doesn't have a negative.
235            sign.is_signed() &&
236            // i1 range is -1..0, so unless it's 0 it will fail
237            size.bit_size() > 1 &&
238            // i128 is not a type the user can declare, but trying to use minus with it
239            // would involve a truncation to 129 bits, which wants to convert to u128,
240            // and 2**129 wouldn't fit into that, and we end up with a division by zero
241            size.bit_size() < 128
242        }
243        _ => false,
244    }
245}

247/// Can the type be returned by some `BinaryOp`.

249    BinaryOp::iter().any(|op| can_binary_op_return(&op, typ))
250}

252/// Check if a certain binary operation can return a target type as output.
253pub fn can_binary_op_return(op: &BinaryOp, typ: &Type) -> bool {
254    use BinaryOpKind::*;
255    match typ {
256        Type::Bool => op.is_comparator(),
257        Type::Field => {
258            matches!(op, Add | Subtract | Multiply | Divide)
259        }
260        Type::Integer(_, _) => {
261            matches!(op, Add | Subtract | Multiply | Divide | ShiftLeft | ShiftRight | Modulo)
262        }
263        _ => false,
264    }
265}

267/// Can the binary operation result in an overflow.
268/// These are operations that commonly fail with random constants.
269pub fn can_binary_op_overflow(op: &BinaryOp) -> bool {
270    use BinaryOpKind::*;
271    matches!(op, Add | Subtract | Multiply | ShiftLeft | ShiftRight)
272}

274/// Can the binary operation fail because the RHS is zero.
275/// These operations can fail because of an unfortunate combination of
276/// expressions, such as `a / (a % a)` or `a % (a % a)`, but it's less
277/// like to occur than the overflows.
278pub fn can_binary_op_err_by_zero(op: &BinaryOp) -> bool {
279    use BinaryOpKind::*;
280    matches!(op, Divide | Modulo)
281}

283/// Check if a certain binary operation can take a type as input and produce the target output.
284pub fn can_binary_op_return_from_input(op: &BinaryOp, input: &Type, output: &Type) -> bool {
285    match (input, output) {
286        (Type::Field, Type::Field) => op.is_valid_for_field_type() && !op.is_equality(),
287        (Type::Field, Type::Bool) => op.is_equality(),
288        (Type::Bool, Type::Bool) => op.is_comparator() || op.is_bitwise(),
289        (Type::Integer(sign, size), Type::Bool) => {
290            // Avoid comparing 128 bit numbers:
291            // `AcirContext::less_than_signed` would cause overflow with i128
292            // `AcirContext::euclidean_division_var` would divide by zero with u128
293            op.is_comparator() && (op.is_equality() || size.bit_size() != 128 || !sign.is_signed())
294        }
295        (Type::Integer(sign_in, size_in), Type::Integer(sign_out, size_out))
296            if sign_in == sign_out =>
297        {
298            let size = size_in.bit_size();
299            // i1 and u1 are very easy to overflow, so we might want to disable those, to not get trivial assertion errors.
300            // i128 is not a type a user can define, and the truncation that gets added after binary operations to
301            // limit it to 129 bits results in division by zero during compilation.
302            (op.is_arithmetic() && size != 1 && size != 128 && size_in <= size_out)
303                || op.is_bitshift() && size != 128 && !(size == 1 && sign_in.is_signed())
304                || op.is_bitwise()
305        }
306        // Reference types need to be dereferenced to participate in binary operations.
307        (Type::Reference(_, _), _) => false,
308        _ => false,
309    }
310}

317/// Convert the type back into a HIR equivalent (not necessarily the original HIR type).
318///
319/// Aims to maintain parity with [Monomorphizer::convert_type](noirc_frontend::monomorphization::Monomorphizer::convert_type).
320pub fn to_hir_type(typ: &Type) -> noirc_frontend::Type {
321    use noirc_frontend::{Kind as HirKind, Type as HirType};
322
323    // Meet the expectations of `Type::evaluate_to_u32`.
324    fn size_const(size: u32) -> Box<HirType> {
325        Box::new(HirType::Constant(
326            SignedField::from(size),
327            HirKind::Numeric(Box::new(HirType::Integer(
328                Signedness::Unsigned,
329                IntegerBitSize::ThirtyTwo,
330            ))),
331        ))
332    }
333
334    // Inverse of HirType::Function -> Type::Tuple([Type::Function, Type::Function])
335    fn maybe_func(items: &[Type]) -> Option<HirType> {
336        if items.len() != 2 {
337            return None;
338        }
339        let Type::Function(args0, ret0, env0, false) = &items[0] else {
340            return None;
341        };
342        let Type::Function(args1, ret1, env1, true) = &items[1] else {
343            return None;
344        };
345        if args0 != args1 || ret0 != ret1 || env0 != env1 {
346            return None;
347        }
348        let func = HirType::Function(
349            vecmap(args0, to_hir_type),
350            Box::new(to_hir_type(ret0)),
351            Box::new(to_hir_type(env0)),
352            // Assume unconstrained.
353            false,
354        );
355        Some(func)
356    }
357
358    match typ {
359        Type::Unit => HirType::Unit,
360        Type::Bool => HirType::Bool,
361        Type::Field => HirType::FieldElement,
362        Type::Integer(signedness, integer_bit_size) => {
363            HirType::Integer(*signedness, *integer_bit_size)
364        }
365        Type::String(size) => HirType::String(size_const(*size)),
366        Type::Array(size, typ) => HirType::Array(size_const(*size), Box::new(to_hir_type(typ))),
367        Type::Reference(typ, mutable) => HirType::Reference(Box::new(to_hir_type(typ)), *mutable),
368        Type::Vector(typ) => HirType::Vector(Box::new(to_hir_type(typ))),
369        Type::FmtString(size, typ) => {
370            HirType::FmtString(size_const(*size), Box::new(to_hir_type(typ)))
371        }
372        Type::Tuple(items) => maybe_func(items)
373            .unwrap_or_else(|| HirType::Tuple(items.iter().map(to_hir_type).collect())),
374        Type::Function(args, ret, env, unconstrained) => HirType::Function(
375            vecmap(args, to_hir_type),
376            Box::new(to_hir_type(ret)),
377            Box::new(to_hir_type(env)),
378            *unconstrained,
379        ),
380    }
