32/// Generate an arbitrary monomorphized AST.
33pub fn arb_program(u: &mut Unstructured, config: Config) -> arbitrary::Result<Program> {
34    let mut ctx = Context::new(config);
35    ctx.gen_globals(u)?;
36    ctx.gen_function_decls(u)?;
37    ctx.gen_functions(u)?;
38    ctx.rewrite_functions(u)?;
39    let program = ctx.finalize();
40    Ok(program)
41}

43/// Generate an arbitrary monomorphized AST to be reversed into a valid comptime
44/// Noir, with a single comptime function called from main with literal arguments.
45pub fn arb_program_comptime(u: &mut Unstructured, config: Config) -> arbitrary::Result<Program> {
46    let mut config = config.clone();
47    // Comptime should use Brillig feature set
48    config.force_brillig = true;
49
50    let mut ctx = Context::new(config);
51
52    // Generate the first (main-wrapped) function declaration
53    let decl_inner = ctx.gen_function_decl(u, 1, true)?;
54    ctx.set_function_decl(FuncId(1), decl_inner.clone());
55
56    // Generate the rest of the declarations
57    let num_extra_fns = u.int_in_range(ctx.config.min_functions..=ctx.config.max_functions)?;
58
59    for i in 0..num_extra_fns {
60        let d = ctx.gen_function_decl(u, i + 2, false)?;
61        ctx.set_function_decl(FuncId((i + 2) as u32), d);
62    }
63
64    // Parameterless main declaration wrapping the inner "main"
65    // function call
66    let decl_main = FunctionDeclaration {
67        name: "main".into(),
68        params: vec![],
69        return_type: decl_inner.return_type.clone(),
70        return_visibility: Visibility::Public,
71        inline_type: InlineType::default(),
72        unconstrained: false,
73    };
74
75    ctx.set_function_decl(FuncId(0), decl_main);
76
77    // Generating functions in this way (after the main wrapper has been
78    // declared) will disallow them from calling the wrapper but not the
79    // main inner function
80    ctx.gen_functions(u)?;
81
82    ctx.gen_function_with_body(u, FuncId(0), |u, function_ctx| {
83        function_ctx.gen_body_with_lit_call(u, FuncId(1))
84    })?;
85    ctx.rewrite_functions(u)?;
86
87    let program = ctx.finalize();
88    Ok(program)
89}

91/// Build a program with the single `main` function returning
92/// the result of a given expression (used for conversion of the
93/// comptime interpreter execution results for comparison)
94pub fn program_wrap_expression(expr: Expression) -> Program {
95    let mut ctx = Context::new(Config::default());
96
97    let decl_main = FunctionDeclaration {
98        name: "main".into(),
99        params: vec![],
100        return_type: expr.return_type().unwrap().into_owned(),
101        return_visibility: Visibility::Public,
102        inline_type: InlineType::default(),
103        unconstrained: true,
104    };
105
106    ctx.set_function_decl(FuncId(0), decl_main);
107    ctx.gen_function_with_body(&mut Unstructured::new(&[]), FuncId(0), |_u, _function_ctx| {
108        Ok(expr)
109    })
110    .expect("shouldn't access any randomness");
111
112    ctx.finalize()
113}

134/// Context to accumulate top level generated item, so we know what we can choose from.
135pub(crate) struct Context {
136    config: Config,
137    /// Global variables.
138    globals: BTreeMap<GlobalId, (Name, Type, Expression)>,
139    /// Function signatures generated up front, so we can call any of them,
140    /// (except `main`), while generating function bodies.
141    function_declarations: BTreeMap<FuncId, FunctionDeclaration>,
142    /// Randomly generated functions that can access the globals and call
143    /// other functions.
144    functions: BTreeMap<FuncId, Function>,
145    /// Random types generated for functions.
146    types: BTreeSet<Type>,
147}

170    /// Generate random global definitions.
171    fn gen_globals(&mut self, u: &mut Unstructured) -> arbitrary::Result<()> {
172        let num_globals = u.int_in_range(0..=self.config.max_globals)?;
173        for i in 0..num_globals {
174            let g = self.gen_global(u, i)?;
175            self.globals.insert(GlobalId(i as u32), g);
176        }
177        Ok(())
178    }
179
180    /// Generate the i-th global variable, which is allowed to use global variables `0..i`.
181    fn gen_global(
182        &mut self,
183        u: &mut Unstructured,
184        i: usize,
185    ) -> arbitrary::Result<(Name, Type, Expression)> {
186        let typ = self.gen_type(
187            u,
188            self.config.max_depth,
189            true,
190            false,
191            self.config.comptime_friendly,
192            true,
193        )?;
194        // By the time we get to the monomorphized AST the compiler will have already turned
195        // complex global expressions into literals.
196        let val = expr::gen_literal(u, &typ, &self.config)?;
197        let name = make_name(i, true);
198        Ok((name, typ, val))
199    }

201    /// Generate random function names and signatures.
202    fn gen_function_decls(&mut self, u: &mut Unstructured) -> arbitrary::Result<()> {
203        let num_non_main_fns =
204            u.int_in_range(self.config.min_functions..=self.config.max_functions)?;
205
206        for i in 0..(1 + num_non_main_fns) {
207            let d = self.gen_function_decl(u, i, i == 0)?;
208            self.function_declarations.insert(FuncId(i as u32), d);
209        }
210        Ok(())
211    }

213    /// Generate a random function declaration.
214    ///
215    /// The `is_abi` parameter tells the generator to only use parameters which are ABI compatible.
216    fn gen_function_decl(
217        &mut self,
218        u: &mut Unstructured,
219        i: usize,
220        is_abi: bool,
221    ) -> arbitrary::Result<FunctionDeclaration> {
222        let id = FuncId(i as u32);
223        let is_main = id == Program::main_id();
224        let num_params = u.int_in_range(0..=self.config.max_function_args)?;
225
226        // If `main` is unconstrained, it won't call ACIR, so no point generating ACIR functions.
227        let unconstrained = self.config.force_brillig
228            || (!is_main
229                && self
230                    .functions
231                    .get(&Program::main_id())
232                    .map(|func| func.unconstrained)
233                    .unwrap_or_default())
234            || bool::arbitrary(u)?;
235
236        // We could return a function as well.
237        let return_type = self.gen_type(
238            u,
239            self.config.max_depth,
240            false,
241            is_main || is_abi,
242            self.config.comptime_friendly,
243            true,
244        )?;
245
246        // Which existing functions we could receive as parameters.
247        let func_param_candidates: Vec<FuncId> = if is_main || self.config.avoid_lambdas {
248            // Main cannot receive function parameters from outside.
249            vec![]
250        } else {
251            self.function_declarations
252                .iter()
253                .filter_map(|(callee_id, callee)| {
254                    can_call(
255                        id,
256                        unconstrained,
257                        types::contains_reference(&return_type),
258                        *callee_id,
259                        callee,
260                    )
261                    .then_some(*callee_id)
262                })
263                .collect()
264        };
265
266        // Choose parameter types.
267        let mut params = Vec::new();
268        for p in 0..num_params {
269            let id = LocalId(p as u32);
270            let name = make_name(p, false);
271            let is_mutable = bool::arbitrary(u)?;
272
273            let typ = if func_param_candidates.is_empty() || u.ratio(7, 10)? {
274                // Take some kind of data type.
275                self.gen_type(
276                    u,
277                    self.config.max_depth,
278                    false,
279                    is_main || is_abi,
280                    self.config.comptime_friendly,
281                    true,
282                )?
283            } else {
284                // Take a function type.
285                let callee_id = u.choose_iter(&func_param_candidates)?;
286                let callee = &self.function_declarations[callee_id];
287                let param_types = callee.params.iter().map(|p| p.3.clone()).collect::<Vec<_>>();
288                let typ = Type::Function(
289                    param_types,
290                    Box::new(callee.return_type.clone()),
291                    Box::new(Type::Unit),
292                    callee.unconstrained,
293                );
294                if u.ratio(2, 5)? { types::ref_mut(typ) } else { typ }
295            };
296
297            let visibility = if is_main {
298                match u.choose_index(5)? {
299                    0 | 1 => Visibility::Public,
300                    2 | 3 => Visibility::Private,
301                    _ => Visibility::CallData(p as u32),
302                }
303            } else {
304                Visibility::Private
305            };
306
307            params.push((id, is_mutable, name, typ, visibility));
308        }
309
310        let return_visibility = if is_main {
311            if types::is_unit(&return_type) {
312                Visibility::Private
313            } else if u.ratio(4, 5)? {
314                Visibility::Public
315            } else {
316                Visibility::ReturnData
317            }
318        } else {
319            Visibility::Private
320        };
321
322        let decl = FunctionDeclaration {
323            name: if is_main { "main".to_string() } else { format!("func_{i}") },
324            params,
325            return_type,
326            return_visibility,
327            inline_type: if is_main {
328                InlineType::default()
329            } else {
330                *u.choose(&[InlineType::Inline, InlineType::InlineAlways])?
331            },
332            unconstrained,
333        };
334
335        Ok(decl)
336    }

345    /// Generate random function bodies.
346    fn gen_functions(&mut self, u: &mut Unstructured) -> arbitrary::Result<()> {
347        let ids = self.function_declarations.keys().cloned().collect::<Vec<_>>();
348        for id in ids {
349            self.gen_function(u, id)?;
350        }
351        Ok(())
352    }

354    /// Generate random function body.
355    fn gen_function(&mut self, u: &mut Unstructured, id: FuncId) -> arbitrary::Result<()> {
356        self.gen_function_with_body(u, id, |u, function_ctx| function_ctx.gen_body(u))
357    }

359    /// Generate function with a specified body generator.
360    fn gen_function_with_body(
361        &mut self,
362        u: &mut Unstructured,
363        id: FuncId,
364        f: impl FnOnce(&mut Unstructured, FunctionContext) -> arbitrary::Result<Expression>,
365    ) -> arbitrary::Result<()> {
366        let function_ctx = FunctionContext::new(self, id);
367        let body = f(u, function_ctx)?;
368        let decl = self.function_decl(id);
369        let func = Function {
370            id,
371            name: decl.name.clone(),
372            parameters: decl.params.clone(),
373            body,
374            return_type: decl.return_type.clone(),
375            return_visibility: decl.return_visibility,
376            unconstrained: decl.unconstrained,
377            inline_type: decl.inline_type,
378            is_entry_point: id == FuncId(0), // we only need main as an entry point
379        };
380        self.functions.insert(id, func);
381        Ok(())
382    }

384    /// Post-processing steps that change functions.
385    fn rewrite_functions(&mut self, u: &mut Unstructured) -> arbitrary::Result<()> {
386        rewrite::remove_unreachable_functions(self);
387        rewrite::add_recursion_limit(self, u)?;
388        Ok(())
389    }

391    /// Return the generated [Program].
392    fn finalize(self) -> Program {
393        let functions = self.functions.into_values().collect::<Vec<_>>();
394        let globals = self.globals.into_iter().collect();
395
396        let program = Program {
397            functions,
398            return_location: None,
399            globals,
400            debug_variables: Default::default(),
401            debug_functions: Default::default(),
402            debug_types: Default::default(),
403        };
404
405        // Carry out the "ownership analysis" here, so the returned program is ready to be turned into SSA.
406        // If we carry out changes that need that analysis to be performed again, we have to make sure
407        // we only run it on functions that haven't been through it before, or it might panic. We could
408        // instead delay the execution to the just before execution, but it's more consistent this way:
409        // for example `CompareSsa` can print the final version without more changes being done to it
410        // while the are being converted to SSA.
411        program.handle_ownership()
412    }

414    /// Generate a random [Type].
415    ///
416    /// Keeps track of types already created, so that we can reuse types instead of always
417    /// creating new ones, to increase the chance of being able to pass variables between
418    /// functions.
419    ///
420    /// With a `max_depth` of 0 only leaf types are created.
421    #[allow(clippy::too_many_arguments)]
422    fn gen_type(
423        &mut self,
424        u: &mut Unstructured,
425        max_depth: usize,
426        is_global: bool,
427        is_main: bool,
428        is_comptime_friendly: bool,
429        is_vector_allowed: bool,
430    ) -> arbitrary::Result<Type> {
431        // See if we can reuse an existing type without going over the maximum depth.
432        if u.ratio(5, 10)? {
433            let existing_types = self
434                .types
435                .iter()
436                .filter(|typ| !is_global || types::can_be_global(typ))
437                .filter(|typ| !is_main || types::can_be_main(typ))
438                .filter(|typ| types::type_depth(typ) <= max_depth)
439                .filter(|typ| is_vector_allowed || !types::contains_vector(typ))
440                .collect::<Vec<_>>();
441
442            if !existing_types.is_empty() {
443                return u.choose(&existing_types).map(|typ| (*typ).clone());
444            }
445        }
446
447        // Once we hit the maximum depth, stop generating composite types.
448        let max_index = if max_depth == 0 { 4 } else { 8 };
449
450        // Generate the inner type for composite types with reduced maximum depth.
451        let gen_inner_type = |this: &mut Self, u: &mut Unstructured, is_vector_allowed: bool| {
452            this.gen_type(
453                u,
454                max_depth - 1,
455                is_global,
456                is_main,
457                is_comptime_friendly,
458                is_vector_allowed,
459            )
460        };
461
462        let mut typ: Type;
463        loop {
464            typ = match u.choose_index(max_index)? {
465                // 4 leaf types
466                0 => Type::Bool,
467                1 => Type::Field,
468                2 => {
469                    // i1 is deprecated, and i128 does not exist yet
470                    let sign = *u.choose(&[Signedness::Signed, Signedness::Unsigned])?;
471                    let sizes = IntegerBitSize::iter()
472                        .filter(|bs| {
473                            // i1 and i128 are rejected by the frontend
474                            (!sign.is_signed() || (bs.bit_size() != 1 && bs.bit_size() != 128)) &&
475                            // Comptime doesn't allow for u1 either
476                            (!is_comptime_friendly || bs.bit_size() != 1)
477                        })
478                        .collect::<Vec<_>>();
479                    Type::Integer(sign, u.choose_iter(sizes)?)
480                }
481                3 => Type::String(u.int_in_range(0..=self.config.max_array_size)? as u32),
482                // 3 composite types
483                4 | 5 => {
484                    // 1-size tuples look strange, so let's make it minimum 2 fields.
485                    let size = u.int_in_range(2..=self.config.max_tuple_size)?;
486                    let types = (0..size)
487                        .map(|_| gen_inner_type(self, u, is_vector_allowed))
488                        .collect::<Result<Vec<_>, _>>()?;
489                    Type::Tuple(types)
490                }
491                6 if is_vector_allowed && !self.config.avoid_vectors => {
492                    let typ = gen_inner_type(self, u, false)?;
493                    Type::Vector(Box::new(typ))
494                }
495                6 | 7 => {
496                    let min_size = 0;
497                    let size = u.int_in_range(min_size..=self.config.max_array_size)?;
498                    let typ = gen_inner_type(self, u, false)?;
499                    Type::Array(size as u32, Box::new(typ))
500                }
501                _ => unreachable!("unexpected arbitrary type index"),
502            };
503            // Looping is kinda dangerous, we could get stuck if we run out of randomness,
504            // so we have to make sure the first type on the list is acceptable.
505            if is_global && !types::can_be_global(&typ) || is_main && !types::can_be_main(&typ) {
506                continue;
507            } else {
508                break;
509            }
510        }
511
512        if !is_main && !is_global && u.ratio(1, 5)? {
513            // Read-only references require the experimental "ownership" feature.
514            typ = types::ref_mut(typ);
515        }
516
517        self.types.insert(typ.clone());
518
519        Ok(typ)
520    }
521}

523/// Derive a variable name from the ID.
524///
525/// Start with `a`, `b`, continuing with `aa`, `ab` if we run out of the alphabet.
526fn make_name(mut id: usize, is_global: bool) -> String {
527    let mut name = Vec::new();
528    let start = if is_global { 65 } else { 97 };
529    loop {
530        let i = id % 26;
531        name.push(char::from(start + i as u8));
532        id -= i;
533        if id == 0 {
534            break;
535        }
536        id /= 26;
537    }
538    name.reverse();
539    let mut name = name.into_iter().collect::<String>();
540    if matches!(name.as_str(), "as" | "if" | "in" | "fn" | "for" | "loop") {
541        name = format!("{name}_");
542    }
543    if is_global { format!("G_{name}") } else { name }
544}

546/// Wrapper around `Program` that prints the AST as close to being able to
547/// copy-paste it as a Noir program as we can get.
548pub struct DisplayAstAsNoir<'a>(pub &'a Program);
549
550impl std::fmt::Display for DisplayAstAsNoir<'_> {
551    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
552        let mut printer = AstPrinter::default();
553        printer.show_id = false;
554        printer.show_clone_and_drop = false;
555        printer.show_specials_as_std = true;
556        printer.show_type_in_let = true;
557        // Most of the time it doesn't affect testing, except the comptime tests where
558        // we parse back the code. For that we use `DisplayAstAsNoirComptime`.
559        // But printing ints with their type makes it much easier to replicate errors in integration tests,
560        // otherwise the frontend rejects negative or large numbers in many contexts.
561        printer.show_type_of_int_literal = true;
562        printer.print_program(self.0, f)
563    }
564}

566/// Wrapper around `Program` that prints its AST as close to
567/// Noir syntax as we can get, making it `comptime`. The AST must
568/// be specifically prepared to include a main function consisting
569/// of a `comptime` wrapped call to a `comptime` (or `unconstrained`)
570/// marked function.
571pub struct DisplayAstAsNoirComptime<'a>(pub &'a Program);
572
573impl std::fmt::Display for DisplayAstAsNoirComptime<'_> {
574    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
575        let mut printer = AstPrinter::default();
576        printer.show_id = false;
577        printer.show_clone_and_drop = false;
578        printer.show_specials_as_std = true;
579        // Declare the type in `let` so that when we parse snippets we can match the types which
580        // the AST had, otherwise a literal which was a `u32` in the AST might be inferred as `Field`.
581        printer.show_type_in_let = true;
582        // Also annotate literals with their type, so we don't have subtle differences in expressions,
583        // for example `for i in (5 / 10) as u32 .. 2` is `0..2` or `1..2` depending on whether 5 and 10
584        // were some number in the AST or `Field` when parsed by the test.
585        printer.show_type_of_int_literal = true;
586
587        for function in &self.0.functions {
588            if function.id == Program::main_id() {
589                let mut function = function.clone();
590                function.return_visibility = Visibility::Public;
591                let fpo = FunctionPrintOptions { comptime_wrap_body: true, ..Default::default() };
592                printer.print_function(&function, f, fpo)?;
593            } else {
594                printer.print_function(function, f, FunctionPrintOptions::default())?;
595            }
596        }
597
598        Ok(())
599    }
