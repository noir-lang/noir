23/// Find the next local ID and ident IDs (in that order) that we can use to add
24/// variables to a [Function] during mutations.
25///
26/// A more sophisticated alternative would be to return metadata along with the `Program`
27/// that contains these values, so we don't have to traverse the AST again, and keep
28/// the logic of what introduces new IDs in sync.
29pub fn next_local_and_ident_id(func: &Function) -> (u32, u32) {
30    let mut next_local_id = func.parameters.iter().map(|p| p.0.0 + 1).max().unwrap_or_default();
31    let mut next_ident_id = 0;
32
33    let mut acc_local_id = |id: LocalId| {
34        next_local_id = next_local_id.max(id.0 + 1);
35    };
36
37    visit_expr(&func.body, &mut |expr| {
38        match expr {
39            Expression::Ident(ident) => {
40                next_ident_id = next_ident_id.max(ident.id.0 + 1);
41            }
42            Expression::Let(let_) => acc_local_id(let_.id),
43            Expression::For(for_) => acc_local_id(for_.index_variable),
44            Expression::Match(match_) => {
45                for case in &match_.cases {
46                    for (id, _) in &case.arguments {
47                        acc_local_id(*id);
48                    }
49                }
50            }
51            _ => {}
52        }
53        true
54    });
55    (next_local_id, next_ident_id)
56}

58/// Turn all ACIR functions into Brillig functions.
59///
60/// This is more involved than flipping the `unconstrained` property because of the
61/// "ownership analysis", which can only run on a function once.
62///
63/// The function also takes care of changing all function pointers into unconstrained ones.
64pub fn change_all_functions_into_unconstrained(mut program: Program) -> Program {
65    for f in &mut program.functions {
66        if f.unconstrained {
67            continue;
68        }
69        // Modify the function.
70        f.unconstrained = true;
71        // Modify any function pointers it takes.
72        for (_, _, _, typ, _) in &mut f.parameters {
73            types::unref_mut_rc(typ, |unref_mut_typ| {
74                if let Type::Function(args, ret, env, _unconstrained) = unref_mut_typ {
75                    Type::Function(args, ret, env, true)
76                } else {
77                    unref_mut_typ
78                }
79            });
80        }
81        // Modify the calls it makes (we don't call ACIR from Brillig).
82        visit_expr_mut(&mut f.body, &mut |expr| {
83            let Expression::Call(Call { func, .. }) = expr else {
84                return true;
85            };
86            let Expression::Ident(Ident { typ, .. }) = expr::unref_mut(func.as_mut()) else {
87                unreachable!("functions are expected to be called by ident; got {func}");
88            };
89
90            types::unref_mut_rc(typ, |unref_mut_typ| {
91                let Type::Function(args, ret, env, _unconstrained) = unref_mut_typ else {
92                    unreachable!(
93                        "function idents are expected to have Function type; got {unref_mut_typ}"
94                    );
95                };
96                Type::Function(args, ret, env, true)
97            });
98            true
99        });
100        f.handle_ownership();
101    }
102    program
103}

105/// Wrap the fuzzer's direct oracle print calls in unconstrained wrapper functions.
106///
107/// In constrained (ACIR) code, direct oracle calls are not allowed. In nargo-compiled
108/// code, `println` goes through unconstrained wrapper functions
109/// (`println` -> `print_unconstrained` -> oracle). The fuzzer generates direct oracle
110/// print calls, so this rewrite wraps them in unconstrained functions to match nargo's
111/// structure, enabling prints in ACIR functions.
112///
113/// Unconstrained functions can call oracles directly, so they are skipped.
114pub(crate) fn wrap_oracle_prints_in_functions(ctx: &mut Context) {
115    let func_ids: Vec<FuncId> = ctx.functions.keys().copied().collect();
116    let mut next_func_id = ctx.functions.len() as u32;
117    let mut wrappers = vec![];
118
119    for &func_id in &func_ids {
120        let func = ctx.functions.get_mut(&func_id).unwrap();
121        // Unconstrained functions can call oracles directly.
122        if func.unconstrained {
123            continue;
124        }
125        visit_expr_mut(&mut func.body, &mut |e| {
126            // Clone to release the borrow on `e` so we can mutate it below.
127            let info =
128                oracle_print_info(e).map(|(vt, nl, ti)| (vt.clone(), nl.clone(), ti.to_vec()));
129            let Some((value_type, newline, type_info)) = info else { return true };
130            // Skip function pointers: no copy-on-write, and their Tuple args
131            // flatten to multiple SSA values which wouldn't match one parameter.
132            if types::is_function(&value_type) {
133                return true;
134            }
135
136            let wrapper_id = FuncId(next_func_id);
137            next_func_id += 1;
138            wrappers.push(make_print_wrapper(wrapper_id, &value_type, newline, &type_info));
139            replace_with_wrapper_call(e, wrapper_id, value_type);
140            true
141        });
142    }
143
144    for w in wrappers {
145        ctx.functions.insert(w.id, w);
146    }
147}

149/// Retarget an oracle print call to a wrapper function, keeping only the value argument.
150fn replace_with_wrapper_call(e: &mut Expression, wrapper_id: FuncId, value_type: Type) {
151    let Expression::Call(call) = e else { unreachable!() };
152    let Expression::Ident(ident) = call.func.as_mut() else { unreachable!() };
153    ident.definition = Definition::Function(wrapper_id);
154    ident.name = format!("print_wrapper_{}", wrapper_id.0);
155    ident.typ =
156        Rc::new(Type::Function(vec![value_type], Rc::new(Type::Unit), Rc::new(Type::Unit), true));
157    // Keep only the value argument (index 1); drop newline and type info.
158    let value_arg = call.arguments.remove(1);
159    call.arguments = vec![value_arg];
160}

162/// If `expr` is a `print` oracle call, return the value type, newline flag, and type-info arguments.
163fn oracle_print_info(expr: &Expression) -> Option<(&Type, &Expression, &[Expression])> {
164    let Expression::Call(Call { func, arguments, .. }) = expr else { return None };
165    let Expression::Ident(Ident { definition: Definition::Oracle(name), typ, .. }) = func.as_ref()
166    else {
167        return None;
168    };
169    if name != "print" {
170        return None;
171    }
172    let Type::Function(params, ..) = typ.as_ref() else { return None };
173    let value_type = params.get(1)?;
174    Some((value_type, &arguments[0], &arguments[2..]))
175}

177/// Build a wrapper: `unconstrained fn(value: T) -> () { print_oracle(newline, value, ...) }`
178fn make_print_wrapper(
179    id: FuncId,
180    value_type: &Type,
181    newline: Expression,
182    type_info_args: &[Expression],
183) -> Function {
184    let value_type_rc = Rc::new(value_type.clone());
185
186    let value_ident = Ident {
187        location: None,
188        definition: Definition::Local(LocalId(0)),
189        mutable: false,
190        name: "value".to_string(),
191        typ: Rc::clone(&value_type_rc),
192        id: IdentId(1),
193    };
194
195    let oracle_ident = Ident {
196        location: None,
197        definition: Definition::Oracle("print".to_string()),
198        mutable: false,
199        name: "print_oracle".to_string(),
200        typ: Rc::new(Type::Function(
201            vec![Type::Bool, value_type.clone()],
202            Rc::new(Type::Unit),
203            Rc::new(Type::Unit),
204            true,
205        )),
206        id: IdentId(0),
207    };
208
209    let mut oracle_args = vec![newline, Expression::Ident(value_ident)];
210    oracle_args.extend_from_slice(type_info_args);
211
212    Function {
213        id,
214        name: format!("print_wrapper_{}", id.0),
215        parameters: vec![(
216            LocalId(0),
217            false,
218            "value".to_string(),
219            value_type_rc,
220            Visibility::Private,
221        )],
222        body: Expression::Call(Call {
223            func: Box::new(Expression::Ident(oracle_ident)),
224            arguments: oracle_args,
225            return_type: Type::Unit,
226            location: Location::dummy(),
227        }),
228        return_type: Type::Unit,
229        return_visibility: Visibility::Private,
230        unconstrained: true,
231        inline_type: InlineType::default(),
232        is_entry_point: false,
233    }
