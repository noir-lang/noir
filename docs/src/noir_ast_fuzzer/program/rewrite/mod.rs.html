14/// Find the next local ID and ident IDs (in that order) that we can use to add
15/// variables to a [Function] during mutations.
16///
17/// A more sophisticated alternative would be to return metadata along with the `Program`
18/// that contains these values, so we don't have to traverse the AST again, and keep
19/// the logic of what introduces new IDs in sync.
20pub fn next_local_and_ident_id(func: &Function) -> (u32, u32) {
21    let mut next_local_id = func.parameters.iter().map(|p| p.0.0 + 1).max().unwrap_or_default();
22    let mut next_ident_id = 0;
23
24    let mut acc_local_id = |id: LocalId| {
25        next_local_id = next_local_id.max(id.0 + 1);
26    };
27
28    visit_expr(&func.body, &mut |expr| {
29        match expr {
30            Expression::Ident(ident) => {
31                next_ident_id = next_ident_id.max(ident.id.0 + 1);
32            }
33            Expression::Let(let_) => acc_local_id(let_.id),
34            Expression::For(for_) => acc_local_id(for_.index_variable),
35            Expression::Match(match_) => {
36                for case in &match_.cases {
37                    for (id, _) in &case.arguments {
38                        acc_local_id(*id);
39                    }
40                }
41            }
42            _ => {}
43        };
44        true
45    });
46    (next_local_id, next_ident_id)
47}

49/// Turn all ACIR functions into Brillig functions.
50///
51/// This is more involved than flipping the `unconstrained` property because of the
52/// "ownership analysis", which can only run on a function once.
53///
54/// The function also takes care of changing all function pointers into unconstrained ones.
55pub fn change_all_functions_into_unconstrained(mut program: Program) -> Program {
56    for f in program.functions.iter_mut() {
57        if f.unconstrained {
58            continue;
59        }
60        // Modify the function.
61        f.unconstrained = true;
62        // Modify any function pointers it takes.
63        for (_, _, _, typ, _) in f.parameters.iter_mut() {
64            if let Type::Function(_, _, _, unconstrained) = types::unref_mut(typ) {
65                *unconstrained = true;
66            }
67        }
68        // Modify the calls it makes (we don't call ACIR from Brillig).
69        visit_expr_mut(&mut f.body, &mut |expr| {
70            let Expression::Call(Call { func, .. }) = expr else {
71                return true;
72            };
73            let Expression::Ident(Ident { typ, .. }) = expr::unref_mut(func.as_mut()) else {
74                unreachable!("functions are expected to be called by ident; got {func}");
75            };
76            let Type::Function(_, _, _, unconstrained) = types::unref_mut(typ) else {
77                unreachable!("function idents are expected to have Function type; got {typ}");
78            };
79            *unconstrained = true;
80            true
81        });
82        f.handle_ownership();
83    }
84    program
85}
