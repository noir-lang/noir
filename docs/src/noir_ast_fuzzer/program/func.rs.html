70/// Help avoid infinite recursion by limiting which function can call which other one.
71pub(super) fn can_call(
72    caller_id: FuncId,
73    caller_unconstrained: bool,
74    caller_returns_ref: bool,
75    callee_id: FuncId,
76    callee_decl: &FunctionDeclaration,
77) -> bool {
78    // Nobody should call `main`.
79    if callee_id == Program::main_id() {
80        return false;
81    }
82
83    // The compiler cannot handle returning references from `if-then-else` in ACIR.
84    // Since the `limit` module currently inserts an `if ctx_limit == 0`,
85    // returning a literal, it would violate this if the return has `&mut`,
86    // therefore we don't make recursive calls from such functions, so the
87    // limit strategy is not applied to them.
88    if caller_returns_ref && caller_unconstrained {
89        return false;
90    }
91
92    // From a Brillig function we restrict ourselves to only call
93    // other Brillig functions. That's because the `Monomorphizer`
94    // would make an unconstrained copy of any ACIR function called
95    // from Brillig, and this is expected by the inliner for example,
96    // but if we did similarly in the generator after we know who
97    // calls who, we would incur two drawbacks:
98    // 1) it would make programs bigger for little benefit
99    // 2) it would skew calibration frequencies as ACIR freqs would overlay Brillig ones
100    if caller_unconstrained {
101        return callee_decl.unconstrained;
102    }
103
104    // When calling ACIR from ACIR, we avoid creating infinite
105    // recursion by only calling functions with lower IDs,
106    // otherwise the inliner could get stuck.
107    if !callee_decl.unconstrained {
108        // Higher calls lower, so we can use this rule to pick function parameters
109        // as we create the declarations: we can pass functions already declared.
110        return callee_id < caller_id;
111    }
112
113    // When calling Brillig from ACIR, we avoid calling functions that take or return
114    // references or vectors, which cannot be passed between the two.
115    !callee_decl.has_refs() && !callee_decl.returns_vectors()
116}

128/// Control what kind of expressions we can generate, depending on the surrounding context.
129#[derive(Debug, Clone, Copy)]
130struct Flags {
131    allow_blocks: bool,
132    allow_if_then: bool,
133}
134
135impl Flags {
136    /// In a top level context, everything is allowed.
137    const TOP: Self = Self { allow_blocks: true, allow_if_then: true };
138    /// In complex nested expressions, avoid generating blocks;
139    /// they would be unreadable and non-idiomatic.
140    const NESTED: Self = Self { allow_blocks: false, allow_if_then: true };
141    /// In `if` conditions avoid nesting more ifs, like `if if if false ...`.
142    const CONDITION: Self = Self { allow_blocks: false, allow_if_then: false };
143    /// In `for` ranges we can use `if` expressions, but let's not do blocks.
144    const RANGE: Self = Self { allow_blocks: false, allow_if_then: true };
145    /// In call arguments we can use `if` expressions, but avoid blocks.
146    /// The arg expressions themselves might call other functions.
147    const CALL: Self = Self { allow_blocks: false, allow_if_then: true };
148}

165/// Context used during the generation of a function body.
166pub(super) struct FunctionContext<'a> {
167    /// Top level context, to access global variables and other functions.
168    ctx: &'a mut Context,
169    /// Self ID.
170    id: FuncId,
171    /// Every variable created in the function will have an increasing ID,
172    /// which does not reset when variables go out of scope.
173    next_local_id: u32,
174    /// Every identifier created in the function will have an increasing ID,
175    /// which does not reset when variables go out of scope.
176    next_ident_id: u32,
177    /// Number of statements remaining to be generated in the function.
178    budget: usize,
179    /// Global variables.
180    globals: Scope<GlobalId>,
181    /// Variables accumulated during the generation of the function body,
182    /// initially consisting of the function parameters, then extended
183    /// by locally defined variables. Block scopes add and remove layers.
184    locals: ScopeStack<LocalId>,
185    /// Indicate which local variables are derived from function inputs.
186    dynamics: Stack<im::HashMap<LocalId, bool>>,
187    /// Indicator of being in a loop (and hence able to generate
188    /// break and continue statements)
189    in_loop: bool,
190    /// Indicator of computing an expression that should not contain dynamic input.
191    in_no_dynamic: bool,
192    /// Indicator of being affected by dynamic input, in which case we should refrain
193    /// from using expression that requires no-dynamic mode.
194    in_dynamic: bool,
195    /// All the functions callable from this one, with the types we can
196    /// produce from their return value.
197    call_targets: BTreeMap<CallableId, HashSet<Type>>,
198    /// Indicate that we have generated a `Call`.
199    has_call: bool,
200}

203    pub(super) fn new(ctx: &'a mut Context, id: FuncId) -> Self {
204        let decl = ctx.function_decl(id);
205        let next_local_id = decl.params.iter().map(|p| p.0.0 + 1).max().unwrap_or_default();
206        let budget = ctx.config.max_function_size;
207
208        let globals = Scope::from_variables(
209            ctx.globals
210                .iter()
211                .map(|(id, (name, typ, _expr))| (*id, false, name.clone(), typ.clone())),
212        );
213
214        // The function parameters are the base layer for local variables.
215        let locals = ScopeStack::from_variables(
216            decl.params
217                .iter()
218                .map(|(id, mutable, name, typ, _vis)| (*id, *mutable, name.clone(), typ.clone())),
219        );
220
221        // Function parameters are by definition considered to be dynamic input.
222        let dynamics = Stack::new(locals.current().variable_ids().map(|id| (*id, true)).collect());
223
224        // Collect all the functions we can call from this one.
225        let mut call_targets = BTreeMap::new();
226
227        // Consider calling any allowed global function.
228        for (callee_id, callee_decl) in &ctx.function_declarations {
229            if !can_call(id, decl.unconstrained, decl.returns_refs(), *callee_id, callee_decl) {
230                continue;
231            }
232            let produces = types::types_produced(&callee_decl.return_type);
233            call_targets.insert(CallableId::Global(*callee_id), produces);
234        }
235
236        // Consider function pointers as callable; they are already filtered during construction.
237        for (callee_id, _, _, typ, _) in &decl.params {
238            let Type::Function(_, return_type, _, _) = types::unref(typ) else {
239                continue;
240            };
241            let produces = types::types_produced(return_type);
242            call_targets.insert(CallableId::Local(*callee_id), produces);
243        }
244
245        Self {
246            ctx,
247            id,
248            next_local_id,
249            budget,
250            globals,
251            locals,
252            dynamics,
253            in_loop: false,
254            in_no_dynamic: false,
255            in_dynamic: false,
256            call_targets,
257            next_ident_id: 0,
258            has_call: false,
259        }
260    }

262    /// Generate the function body.
263    pub(super) fn gen_body(mut self, u: &mut Unstructured) -> arbitrary::Result<Expression> {
264        // If we don't limit the budget according to the available data,
265        // it gives us a lot of `false` and 0 and we end up with deep `!(!false)` if expressions.
266        self.budget = self.budget.min(u.len());
267        let ret = self.decl().return_type.clone();
268        let (mut body, _) = self.gen_expr(u, &ret, self.max_depth(), Flags::TOP)?;
269        if let Some(call) = self.gen_guaranteed_call_from_main(u)? {
270            expr::prepend(&mut body, call);
271        }
272        Ok(body)
273    }

343    /// Mark a variable as dynamic or not dynamic.
344    fn set_dynamic(&mut self, id: LocalId, is_dynamic: bool) {
345        // When a dynamic variable is assigned a constant value, only the current
346        // scope and any future lower scopes are affected. After this scope we
347        // will revert to whatever it was before.
348        let is_new = self.dynamics.current_mut().insert(id, is_dynamic).is_none();
349
350        // Becoming dynamic is contagious: if we assign a dynamic value to a mutable
351        // variable in one of the branches of a conditional statement, we have to
352        // consider it dynamic in the outer scopes as well from then on.
353        if !is_new && is_dynamic {
354            for layer in self.dynamics.iter_mut() {
355                if layer.contains_key(&id) {
356                    layer.insert(id, true);
357                }
358            }
359        }
360    }

372    /// Choose a producer for a type, preferring local variables over global ones.
373    fn choose_producer(
374        &self,
375        u: &mut Unstructured,
376        typ: &Type,
377    ) -> arbitrary::Result<Option<VariableId>> {
378        // Check if we have something that produces this exact type.
379        if u.ratio(7, 10)? {
380            let producer = if self.in_no_dynamic || self.in_dynamic {
381                self.locals.current().choose_producer_filtered(
382                    u,
383                    typ,
384                    |id, (_, _, producer_type)| {
385                        (!self.in_no_dynamic || !self.is_dynamic(id))
386                            && (!self.in_dynamic || self.can_be_used_in_dynamic(producer_type, typ))
387                    },
388                )?
389            } else {
390                self.locals.current().choose_producer(u, typ)?
391            };
392            if let Some(id) = producer {
393                return Ok(Some(VariableId::Local(id)));
394            }
395        }
396        // If we're looking for a mutable reference, we have to choose some
397        // mutable local variable and take a reference over it.
398        // We can't use a global for this, because they are immutable.
399        if let Type::Reference(typ, true) = typ {
400            // Find an underlying mutable variable we can take a reference over.
401            // We cannot have mutable references to array elements.
402            return self
403                .locals
404                .current()
405                .choose_producer_filtered(u, typ.as_ref(), |id, (mutable, _, producer_type)| {
406                    *mutable
407                        && (typ.as_ref() == producer_type
408                            || !types::is_array_or_vector(producer_type))
409                        && (!self.in_no_dynamic || !self.is_dynamic(id))
410                        && (!self.in_dynamic
411                            || self.can_be_used_in_dynamic(producer_type, typ.as_ref()))
412                })
413                .map(|id| id.map(VariableId::Local));
414        }
415        self.globals.choose_producer(u, typ).map(|id| id.map(VariableId::Global))
416    }

438    /// Generate an expression of a certain type.
439    ///
440    /// While doing so, enter and exit blocks, and add variables declared to the context,
441    /// so expressions down the line can refer to earlier variables.
442    ///
443    /// This will always succeed, because we can always return a literal expression.
444    fn gen_expr(
445        &mut self,
446        u: &mut Unstructured,
447        typ: &Type,
448        max_depth: usize,
449        flags: Flags,
450    ) -> arbitrary::Result<TrackedExpression> {
451        // For now if we need a function, return one without further nesting, e.g. avoid `if <cond> { func_1 } else { func_2 }`,
452        // because it makes it harder to rewrite functions to add recursion limit: we would need to replace functions in the
453        // expressions to proxy version if we call Brillig from ACIR, but we would also need to keep track whether we are calling a function,
454        // For example if we could return function pointers, we could have something like this:
455        //  `acir_func_1(if c { brillig_func_2 } else { unsafe { brillig_func_3(brillig_func_4) } })`
456        // We could replace `brillig_func_2` with `brillig_func_2_proxy`, but we wouldn't replace `brillig_func_4` with `brillig_func_4_proxy`
457        // because that is a parameter of another call. But we would have to deal with the return value.
458        // For this reason we handle function parameters directly here.
459        if types::is_function(types::unref(typ)) {
460            if let Type::Reference(typ, _) = typ {
461                // We might have an `&mut &mut fn(...) -> ...`; we recursively peel
462                // off layers of references until we can generate an expression that
463                // returns an `fn`, and then take a reference over it to restore.
464                let (expr, is_dyn) = self.gen_expr(u, typ, max_depth, flags)?;
465                // If the expression is a read-only global ident, then assign a variable first.
466                let expr = if expr::is_immutable_ident(&expr) {
467                    self.indirect_ref_mut((expr, is_dyn), typ.as_ref().clone())
468                } else {
469                    expr::ref_mut(expr, typ.as_ref().clone())
470                };
471                return Ok((expr, is_dyn));
472            } else {
473                // Prefer functions in variables over globals.
474                return match self.gen_expr_from_vars(u, typ, max_depth)? {
475                    Some(expr) => Ok(expr),
476                    None => {
477                        self.find_global_function_with_signature(u, typ).map(|expr| (expr, false))
478                    }
479                };
480            }
481        };
482
483        let mut freq = Freq::new(u, &self.config().expr_freqs)?;
484
485        // Stop nesting if we reached the bottom.
486        let allow_nested = max_depth > 0;
487
488        let allow_blocks = flags.allow_blocks
489            && allow_nested
490            && max_depth == self.config().max_depth
491            && self.budget > 0;
492
493        let allow_if_then = flags.allow_if_then
494            && allow_nested
495            && self.budget > 0
496            && (self.unconstrained() || !types::contains_reference(typ));
497
498        let allow_match = allow_if_then && !self.ctx.config.avoid_match;
499
500        if freq.enabled_when("unary", allow_nested && types::can_unary_return(typ)) {
501            if let Some(expr) = self.gen_unary(u, typ, max_depth)? {
502                return Ok(expr);
503            }
504        }
505
506        if freq.enabled_when("binary", allow_nested && types::can_binary_return(typ)) {
507            if let Some(expr) = self.gen_binary(u, typ, max_depth)? {
508                return Ok(expr);
509            }
510        }
511
512        // if-then-else returning a value
513        // Unlike blocks/loops it can appear in nested expressions.
514        if freq.enabled_when("if", allow_if_then) {
515            return self.gen_if(u, typ, max_depth, flags);
516        }
517
518        // Match expressions, returning a value.
519        // Treating them similarly to if-then-else.
520        if freq.enabled_when("match", allow_match) {
521            // It might not be able to generate the type, if we don't have a suitable variable to match on.
522            if let Some(expr) = self.gen_match(u, typ, max_depth)? {
523                return Ok(expr);
524            }
525        }
526
527        // Block of statements returning a value
528        if freq.enabled_when("block", allow_blocks) {
529            return self.gen_block(u, typ);
530        }
531
532        // Function calls returning a value.
533        if freq.enabled_when("call", allow_nested && self.budget > 0) {
534            // Decreasing the max depth in expression position because it can be very difficult to read.
535            if let Some(expr) = self.gen_call(u, typ, max_depth.saturating_sub(1))? {
536                return Ok(expr);
537            }
538        }
539
540        // We can always try to just derive a value from the variables we have.
541        if freq.enabled("vars") {
542            if let Some(expr) = self.gen_expr_from_vars(u, typ, max_depth)? {
543                return Ok(expr);
544            }
545        }
546
547        // If nothing else worked out we can always produce a random literal.
548        self.gen_literal(u, typ).map(|expr| (expr, false))
549    }

551    /// Try to generate an expression with a certain type out of the variables in scope.
552    fn gen_expr_from_vars(
553        &mut self,
554        u: &mut Unstructured,
555        typ: &Type,
556        max_depth: usize,
557    ) -> arbitrary::Result<Option<TrackedExpression>> {
558        if let Some(id) = self.choose_producer(u, typ)? {
559            let (src_mutable, src_name, src_type) = self.get_variable(&id).clone();
560            let ident_id = self.next_ident_id();
561            let src_expr = expr::ident(id, ident_id, src_mutable, src_name, src_type.clone());
562            let src_dyn = match id {
563                VariableId::Global(_) => false,
564                VariableId::Local(id) => self.is_dynamic(&id),
565            };
566            if let Some(expr) = self.gen_expr_from_source(
567                u,
568                (src_expr, src_dyn),
569                &src_type,
570                src_mutable,
571                typ,
572                max_depth,
573            )? {
574                return Ok(Some(expr));
575            }
576        } else {
577            // If we can't produce the exact we're looking for, maybe we can produce parts of it.
578            match typ {
579                Type::Array(len, item_type) => {
580                    // Randomly choose between Array and Repeated literal
581                    if bool::arbitrary(u)? {
582                        let mut arr = ArrayLiteral {
583                            contents: Vec::with_capacity(*len as usize),
584                            typ: typ.clone(),
585                        };
586                        let mut arr_dyn = false;
587                        for _ in 0..*len {
588                            let (item, item_dyn) =
589                                self.gen_expr(u, item_type, max_depth, Flags::NESTED)?;
590                            arr_dyn |= item_dyn;
591                            arr.contents.push(item);
592                        }
593                        return Ok(Some((Expression::Literal(Literal::Array(arr)), arr_dyn)));
594                    } else {
595                        let (element, elem_dyn) =
596                            self.gen_expr(u, item_type, max_depth, Flags::NESTED)?;
597                        return Ok(Some((
598                            Expression::Literal(Literal::Repeated {
599                                element: Box::new(element),
600                                length: *len,
601                                is_vector: false,
602                                typ: typ.clone(),
603                            }),
604                            elem_dyn,
605                        )));
606                    }
607                }
608                Type::Tuple(items) => {
609                    let mut values = Vec::with_capacity(items.len());
610                    let mut tup_dyn = false;
611                    for item_type in items {
612                        let (item, item_dyn) =
613                            self.gen_expr(u, item_type, max_depth, Flags::NESTED)?;
614                        tup_dyn |= item_dyn;
615                        values.push(item);
616                    }
617                    return Ok(Some((Expression::Tuple(values), tup_dyn)));
618                }
619                _ => {}
620            }
621        }
622        Ok(None)
623    }

625    /// Try to generate an expression that produces a target type from a source,
626    /// e.g. given a source type of `[(u32, bool); 4]` and a target of `u64`
627    /// it might generate `my_var[2].0 as u64`.
628    ///
629    /// The `src_mutable` parameter indicates whether we can take a mutable reference over the source.
630    ///
631    /// Returns `None` if there is no way to produce the target from the source.
632    fn gen_expr_from_source(
633        &mut self,
634        u: &mut Unstructured,
635        (src_expr, src_dyn): TrackedExpression,
636        src_type: &Type,
637        mut src_mutable: bool,
638        tgt_type: &Type,
639        max_depth: usize,
640    ) -> arbitrary::Result<Option<TrackedExpression>> {
641        // If we found our type, we can return it without further ado.
642        if src_type == tgt_type {
643            // If we want a vector, we can push onto it.
644            if let Type::Vector(item_type) = src_type {
645                if bool::arbitrary(u)? {
646                    let (item, item_dyn) = self.gen_expr(u, item_type, max_depth, Flags::TOP)?;
647                    // We can use push_back, push_front, or insert.
648                    if bool::arbitrary(u)? {
649                        let push_expr = self.call_vector_push(
650                            src_type.clone(),
651                            item_type.as_ref().clone(),
652                            src_expr,
653                            bool::arbitrary(u)?,
654                            item,
655                        );
656                        return Ok(Some((push_expr, src_dyn || item_dyn)));
657                    } else {
658                        // Generate a random index and insert the item at it.
659                        return self.gen_vector_access(
660                            u,
661                            (src_expr, src_dyn || item_dyn),
662                            src_type,
663                            src_mutable,
664                            tgt_type,
665                            max_depth,
666                            |this, ident, idx| {
667                                this.call_vector_insert(
668                                    src_type.clone(),
669                                    item_type.as_ref().clone(),
670                                    Expression::Ident(ident),
671                                    idx,
672                                    item,
673                                )
674                            },
675                        );
676                    }
677                }
678            }
679            // Otherwise just return as-is.
680            return Ok(Some((src_expr, src_dyn)));
681        }
682
683        // Some types cannot be accessed in certain contexts.
684        if self.in_dynamic && !self.can_be_used_in_dynamic(src_type, tgt_type) {
685            return Ok(None);
686        }
687
688        // Mutable references to array elements are currently unsupported.
689        if types::is_array_or_vector(src_type) {
690            src_mutable = false;
691        }
692
693        // Cast the source into the target type.
694        let src_as_tgt = || {
695            let expr = expr::cast(src_expr.clone(), tgt_type.clone());
696            Ok(Some((expr, src_dyn)))
697        };
698
699        // See how we can produce tgt from src.
700        match (src_type, tgt_type) {
701            // Simple numeric conversions.
702            (
703                Type::Field,
704                Type::Integer(Signedness::Unsigned, IntegerBitSize::HundredTwentyEight),
705            ) => src_as_tgt(),
706            (Type::Bool, Type::Field) => src_as_tgt(),
707            (Type::Integer(Signedness::Unsigned, _), Type::Field) => src_as_tgt(),
708            (Type::Integer(sign_from, ibs_from), Type::Integer(sign_to, ibs_to))
709                if sign_from == sign_to && ibs_from.bit_size() < ibs_to.bit_size() =>
710            {
711                src_as_tgt()
712            }
713            // Dereference right into the target type.
714            (Type::Reference(typ, _), _) if typ.as_ref() == tgt_type => {
715                let expr = expr::deref(src_expr, tgt_type.clone());
716                Ok(Some((expr, src_dyn)))
717            }
718            // Mutable reference over the source type.
719            (_, Type::Reference(typ, true)) if typ.as_ref() == src_type => {
720                let expr = if src_mutable {
721                    expr::ref_mut(src_expr, typ.as_ref().clone())
722                } else {
723                    self.indirect_ref_mut((src_expr, src_dyn), typ.as_ref().clone())
724                };
725                Ok(Some((expr, src_dyn)))
726            }
727            // Index a non-empty array.
728            (Type::Array(len, item_type), _) if *len > 0 => {
729                // Indexing arrays that contains references with dynamic indexes was banned in #8888
730                // If we are already looking for an index where we can't use dynamic inputs,
731                // don't switch to using them again, as the result can indirectly poison the outer array.
732                // For example this would be wrong:
733                //
734                // fn main(i: u32) -> pub u32 {
735                //     let a = [&mut 0, &mut 1];
736                //     let b = [0, 1];
737                //     *a[b[i]]
738                // }
739                let (idx_expr, idx_dyn) = {
740                    let no_dynamic = self.in_no_dynamic
741                        || !self.unconstrained() && types::contains_reference(item_type);
742                    let was_in_no_dynamic = std::mem::replace(&mut self.in_no_dynamic, no_dynamic);
743
744                    // Choose a random index.
745                    let (idx_expr, idx_dyn) = self.gen_index(u, *len, max_depth)?;
746                    assert!(!(no_dynamic && idx_dyn), "non-dynamic index expected");
747
748                    self.in_no_dynamic = was_in_no_dynamic;
749                    (idx_expr, idx_dyn)
750                };
751
752                // Access the item.
753                let item_expr = Expression::Index(Index {
754                    collection: Box::new(src_expr),
755                    index: Box::new(idx_expr),
756                    element_type: *item_type.clone(),
757                    location: Location::dummy(),
758                });
759                // Produce the target type from the item.
760                self.gen_expr_from_source(
761                    u,
762                    (item_expr, src_dyn || idx_dyn),
763                    item_type,
764                    src_mutable,
765                    tgt_type,
766                    max_depth,
767                )
768            }
769            // Pop from the front of a vector.
770            (Type::Vector(item_type), Type::Tuple(fields))
771                if fields.len() == 2
772                    && &fields[0] == item_type.as_ref()
773                    && &fields[1] == src_type =>
774            {
775                let pop_front = self.call_vector_pop(
776                    src_type.clone(),
777                    item_type.as_ref().clone(),
778                    src_expr,
779                    true,
780                );
781                Ok(Some((pop_front, src_dyn)))
782            }
783            // Pop from the back of a vector, or remove an item.
784            (Type::Vector(item_type), Type::Tuple(fields))
785                if fields.len() == 2
786                    && &fields[0] == src_type
787                    && &fields[1] == item_type.as_ref() =>
788            {
789                if bool::arbitrary(u)? {
790                    let pop_back = self.call_vector_pop(
791                        src_type.clone(),
792                        item_type.as_ref().clone(),
793                        src_expr,
794                        false,
795                    );
796                    Ok(Some((pop_back, src_dyn)))
797                } else {
798                    self.gen_vector_access(
799                        u,
800                        (src_expr, src_dyn),
801                        src_type,
802                        src_mutable,
803                        tgt_type,
804                        max_depth,
805                        |this, ident, idx| {
806                            this.call_vector_remove(
807                                src_type.clone(),
808                                item_type.as_ref().clone(),
809                                Expression::Ident(ident),
810                                idx,
811                            )
812                        },
813                    )
814                }
815            }
816            // Index a vector (might fail at runtime if empty).
817            (Type::Vector(item_type), _) => self.gen_vector_access(
818                u,
819                (src_expr, src_dyn),
820                src_type,
821                src_mutable,
822                tgt_type,
823                max_depth,
824                |_, ident, idx| {
825                    // Index the vector, represented by a variable, using the generated index.
826                    Expression::Index(Index {
827                        collection: Box::new(Expression::Ident(ident)),
828                        index: Box::new(idx),
829                        element_type: *item_type.clone(),
830                        location: Location::dummy(),
831                    })
832                },
833            ),
834            // Extract a tuple field.
835            (Type::Tuple(items), _) => {
836                // Any of the items might be able to produce the target type.
837                let mut opts = Vec::new();
838                for (i, item_type) in items.iter().enumerate() {
839                    let item_expr = Expression::ExtractTupleField(Box::new(src_expr.clone()), i);
840                    if let Some(expr) = self.gen_expr_from_source(
841                        u,
842                        (item_expr, src_dyn),
843                        item_type,
844                        src_mutable,
845                        tgt_type,
846                        max_depth,
847                    )? {
848                        opts.push(expr);
849                    }
850                }
851                if opts.is_empty() { Ok(None) } else { Ok(Some(u.choose_iter(opts)?)) }
852            }
853            _ => {
854                // We have already considered the case when the two types equal.
855                // Normally we would call this function knowing that source can produce the target,
856                // but in case we missed a case, let's return None and let the caller fall back to
857                // a different strategy. In some cases we could return a literal, but it wouldn't
858                // work in the recursive case of producing a type from an array item, which needs
859                // to be wrapped with an accessor.
860                Ok(None)
861            }
862        }
863    }

865    /// Generate code to index an arbitrary item in a vector.
866    ///
867    /// Since we don't know the length of the vector at compile time,
868    /// this can involve creating a temporary variable, getting the length at runtime,
869    /// and using modulo to limit some random index to the runtime length.
870    #[allow(clippy::too_many_arguments)]
871    fn gen_vector_access<F>(
872        &mut self,
873        u: &mut Unstructured,
874        (src_expr, src_dyn): TrackedExpression,
875        src_type: &Type,
876        src_mutable: bool,
877        tgt_type: &Type,
878        max_depth: usize,
879        access_item: F,
880    ) -> arbitrary::Result<Option<TrackedExpression>>
881    where
882        F: FnOnce(&mut Self, Ident, Expression) -> Expression,
883    {
884        let Type::Vector(item_type) = src_type else {
885            unreachable!("only expected to be called with Vector");
886        };
887
888        // Unless the vector is coming from an identifier or literal, we should create a let binding for it
889        // to avoid doubling up any side effects, or double using local variables when it was created.
890        let (let_expr, ident_1) = if let Expression::Ident(ident) = src_expr {
891            (None, ident)
892        } else {
893            let (let_expr, let_ident) = self.let_var_and_ident(
894                false,
895                src_type.clone(),
896                src_expr,
897                false,
898                src_dyn,
899                local_name,
900            );
901            (Some(let_expr), let_ident)
902        };
903
904        // We'll need the ident again to access the item.
905        let ident_2 = Ident { id: self.next_ident_id(), ..ident_1.clone() };
906
907        // Get the runtime length.
908        let len_expr = self.call_array_len(Expression::Ident(ident_1), src_type.clone());
909
910        // The rules around dynamic indexing is the same as for arrays.
911        let (idx_expr, idx_dyn) = if max_depth == 0 || bool::arbitrary(u)? {
912            // Avoid any stack overflow where we look for an index in the vector itself.
913            (self.gen_literal(u, &types::U32)?, false)
914        } else {
915            let no_dynamic =
916                self.in_no_dynamic || !self.unconstrained() && types::contains_reference(item_type);
917            let was_in_no_dynamic = std::mem::replace(&mut self.in_no_dynamic, no_dynamic);
918
919            // Choose a random index.
920            let (mut idx_expr, idx_dyn) =
921                self.gen_expr(u, &types::U32, max_depth.saturating_sub(1), Flags::NESTED)?;
922
923            assert!(!(no_dynamic && idx_dyn), "non-dynamic index expected");
924
925            // Take the modulo, but leave a small chance for index OOB.
926            if self.avoid_index_out_of_bounds(u)? {
927                idx_expr = expr::modulo(idx_expr, len_expr);
928            }
929
930            self.in_no_dynamic = was_in_no_dynamic;
931            (idx_expr, idx_dyn)
932        };
933
934        // Access the item by index
935        let item_expr = access_item(self, ident_2, idx_expr);
936
937        // Produce the target type from the item.
938        let Some((expr, is_dyn)) = self.gen_expr_from_source(
939            u,
940            (item_expr, src_dyn || idx_dyn),
941            item_type,
942            src_mutable,
943            tgt_type,
944            max_depth,
945        )?
946        else {
947            return Ok(None);
948        };
949
950        // Append the let and the final expression if we needed a block,
951        // so we avoid suffixing a block with e.g. indexing, which would
952        // not be parsable by the frontend. Another way to do this would
953        // be to surround the block with parentheses.
954        // So either of this should work:
955        // * { let s = todo!(); s[123 % s.len()][456] }
956        // * ( { let s = todo!(); s[123 % s.len()] } )[456]
957        // But not this:
958        // * { let s = todo!(); s[123 % s.len()] }[123]
959        let expr = if let Some(let_expr) = let_expr {
960            Expression::Block(vec![let_expr, expr])
961        } else {
962            expr
963        };
964
965        Ok(Some((expr, is_dyn)))
966    }

968    /// Generate an arbitrary index for an array.
969    ///
970    /// This can be either a random int literal, or a complex expression that produces an int.
971    fn gen_index(
972        &mut self,
973        u: &mut Unstructured,
974        len: u32,
975        max_depth: usize,
976    ) -> arbitrary::Result<TrackedExpression> {
977        assert!(len > 0, "cannot index empty array");
978        if max_depth > 0 && u.ratio(1, 3)? {
979            let (mut idx, idx_dyn) =
980                self.gen_expr(u, &types::U32, max_depth.saturating_sub(1), Flags::NESTED)?;
981
982            // Limit the index to be in the valid range for the array length, with a small chance of index OOB.
983            if self.avoid_index_out_of_bounds(u)? {
984                idx = expr::index_modulo(idx, len);
985            }
986
987            Ok((idx, idx_dyn))
988        } else {
989            let idx = u.choose_index(len as usize)?;
990            Ok((expr::u32_literal(idx as u32), false))
991        }
992    }

994    /// Try to generate a unary expression of a certain type, if it's amenable to it, otherwise return `None`.
995    fn gen_unary(
996        &mut self,
997        u: &mut Unstructured,
998        typ: &Type,
999        max_depth: usize,
1000    ) -> arbitrary::Result<Option<TrackedExpression>> {
1001        // Negation can cause overflow: for example `-1*i8::MIN` does not fit into `i8`, because `i8` is [-128, 127].
1002        let avoid_overflow = self.config().avoid_overflow || self.in_no_dynamic;
1003
1004        let mut make_unary = |op| {
1005            self.gen_expr(u, typ, max_depth.saturating_sub(1), Flags::NESTED)
1006                .map(|(rhs, is_dyn)| Some((expr::unary(op, rhs, typ.clone()), is_dyn)))
1007        };
1008
1009        if matches!(typ, Type::Field)
1010            || matches!(typ, Type::Integer(Signedness::Signed, _)) && !avoid_overflow
1011        {
1012            make_unary(UnaryOp::Minus)
1013        } else if types::is_bool(typ) {
1014            make_unary(UnaryOp::Not)
1015        } else {
1016            Ok(None)
1017        }
1018    }

1020    /// Try to generate a binary expression of a certain type, if it's amenable to it, otherwise return `None`.
1021    fn gen_binary(
1022        &mut self,
1023        u: &mut Unstructured,
1024        typ: &Type,
1025        max_depth: usize,
1026    ) -> arbitrary::Result<Option<TrackedExpression>> {
1027        // Collect the operations can return the expected type.
1028        // Avoid operations that can fail in no-dynamic mode, otherwise they will be considered non-constant indexes.
1029        let ops = BinaryOp::iter()
1030            .filter(|op| {
1031                types::can_binary_op_return(op, typ)
1032                    && (!(self.config().avoid_overflow || self.in_no_dynamic)
1033                        || !types::can_binary_op_overflow(op))
1034                    && (!(self.config().avoid_err_by_zero || self.in_no_dynamic)
1035                        || !types::can_binary_op_err_by_zero(op))
1036            })
1037            .collect::<Vec<_>>();
1038
1039        // Ideally we checked that the target type can be returned, but just in case.
1040        if ops.is_empty() {
1041            return Ok(None);
1042        }
1043
1044        // Choose a random operation.
1045        let op = u.choose_iter(ops)?;
1046
1047        // Find a type we can produce in the current scope which we can pass as input
1048        // to the operations we selected, and it returns the desired output.
1049        fn collect_input_types<'a, K: Ord>(
1050            op: BinaryOp,
1051            type_out: &Type,
1052            scope: &'a Scope<K>,
1053        ) -> Vec<&'a Type> {
1054            scope
1055                .types_produced()
1056                .filter(|type_in| types::can_binary_op_return_from_input(&op, type_in, type_out))
1057                .collect::<Vec<_>>()
1058        }
1059
1060        // Try local variables first.
1061        let mut lhs_opts = collect_input_types(op, typ, self.locals.current());
1062
1063        // If the locals don't have any type compatible with `op`, try the globals.
1064        if lhs_opts.is_empty() {
1065            lhs_opts = collect_input_types(op, typ, &self.globals);
1066        }
1067
1068        // We might not have any input that works for this operation.
1069        // We could generate literals, but that's not super interesting.
1070        if lhs_opts.is_empty() {
1071            return Ok(None);
1072        }
1073
1074        // Choose a type for the LHS and RHS.
1075        let lhs_type = u.choose_iter(lhs_opts)?.clone();
1076
1077        // Generate expressions for LHS and RHS.
1078        let (lhs_expr, lhs_dyn) =
1079            self.gen_expr(u, &lhs_type, max_depth.saturating_sub(1), Flags::NESTED)?;
1080        let (rhs_expr, rhs_dyn) =
1081            self.gen_expr(u, &lhs_type, max_depth.saturating_sub(1), Flags::NESTED)?;
1082
1083        let mut expr = expr::binary(lhs_expr, op, rhs_expr);
1084
1085        // If we have chosen e.g. u8 and need u32 we need to cast.
1086        if !(lhs_type == *typ || types::is_bool(typ) && op.is_comparator()) {
1087            expr = expr::cast(expr, typ.clone());
1088        }
1089
1090        Ok(Some((expr, lhs_dyn || rhs_dyn)))
1091    }

1093    /// Generate a block of statements, finally returning a target type.
1094    ///
1095    /// This should always succeed, as we can always create a literal in the end.
1096    ///
1097    /// We might use dynamic input in the block and _not_ return a dynamic result from it. For example:
1098    /// ```ignore
1099    /// fn main(a: u32) -> u32 {
1100    ///     let b: u32 = {
1101    ///         let c = a + 1;
1102    ///         2
1103    ///     };
1104    ///     b
1105    /// }
1106    /// ```
1107    fn gen_block(
1108        &mut self,
1109        u: &mut Unstructured,
1110        typ: &Type,
1111    ) -> arbitrary::Result<TrackedExpression> {
1112        // The `max_depth` resets here, because that's only relevant in complex expressions.
1113        let max_depth = self.max_depth();
1114        let max_size = self.config().max_block_size.min(self.budget);
1115
1116        // If we want blocks to be empty, or we don't have a budget for statements, just return an expression.
1117        if max_size == 0 {
1118            return self.gen_expr(u, typ, max_depth, Flags::TOP);
1119        }
1120
1121        // Choose a positive number of statements.
1122        let size = u.int_in_range(1..=max_size)?;
1123        let mut stmts = Vec::with_capacity(size);
1124        // Only the last statement counts into whether the block is dynamic.
1125        let mut is_dyn = false;
1126
1127        self.enter_scope();
1128        self.decrease_budget(1);
1129        for _ in 0..size - 1 {
1130            if self.budget == 0 {
1131                break;
1132            }
1133            self.decrease_budget(1);
1134            stmts.push(self.gen_stmt(u)?);
1135        }
1136        if types::is_unit(typ) && u.ratio(4, 5)? {
1137            // ending a unit block with `<stmt>;` looks better than a `()` but both are valid.
1138            // NB the AST printer puts a `;` between all statements, including after `if` and `for`.
1139            stmts.push(Expression::Semi(Box::new(self.gen_stmt(u)?)));
1140        } else {
1141            let (expr, expr_dyn) = self.gen_expr(u, typ, max_depth, Flags::TOP)?;
1142            is_dyn = expr_dyn;
1143            stmts.push(expr);
1144        }
1145        self.exit_scope();
1146
1147        Ok((Expression::Block(stmts), is_dyn))
1148    }

1150    /// Generate a statement, which is an expression that doesn't return anything,
1151    /// for example loops, variable declarations, etc.
1152    ///
1153    /// Because the statement doesn't return anything (return unit) we don't track
1154    /// whether it used dynamic inputs; there is nothing to propagate.
1155    fn gen_stmt(&mut self, u: &mut Unstructured) -> arbitrary::Result<Expression> {
1156        let mut freq = if self.unconstrained() {
1157            Freq::new(u, &self.config().stmt_freqs_brillig)?
1158        } else {
1159            Freq::new(u, &self.config().stmt_freqs_acir)?
1160        };
1161        // TODO(#7926): Match
1162
1163        // We don't want constraints to get too frequent, as it could dominate all outcome.
1164        if freq.enabled_when("constrain", !self.config().avoid_constrain) {
1165            if let Some(e) = self.gen_constrain(u)? {
1166                return Ok(e);
1167            }
1168        }
1169
1170        // Require a positive budget, so that we have some for the block itself and its contents.
1171        if freq.enabled_when("if", self.budget > 1) {
1172            return self.gen_if(u, &Type::Unit, self.max_depth(), Flags::TOP).map(|(e, _)| e);
1173        }
1174
1175        if freq.enabled_when("match", self.budget > 1 && !self.ctx.config.avoid_match) {
1176            if let Some((e, _)) = self.gen_match(u, &Type::Unit, self.max_depth())? {
1177                return Ok(e);
1178            }
1179        }
1180
1181        if freq.enabled_when("for", self.budget > 1) {
1182            return self.gen_for(u);
1183        }
1184
1185        if freq.enabled_when("call", self.budget > 0) {
1186            if let Some((e, _)) = self.gen_call(u, &Type::Unit, self.max_depth())? {
1187                return Ok(e);
1188            }
1189        }
1190
1191        if self.unconstrained() {
1192            // Get loop out of the way quick, as it's always disabled for ACIR.
1193            if freq.enabled_when("loop", self.budget > 1) {
1194                return self.gen_loop(u);
1195            }
1196
1197            if freq.enabled_when("while", self.budget > 1) {
1198                return self.gen_while(u);
1199            }
1200
1201            if freq.enabled_when("break", self.in_loop && !self.config().avoid_loop_control) {
1202                return Ok(Expression::Break);
1203            }
1204
1205            if freq.enabled_when("continue", self.in_loop && !self.config().avoid_loop_control) {
1206                return Ok(Expression::Continue);
1207            }
1208
1209            // For now only try prints in unconstrained code, were we don't need to create a proxy.
1210            if freq.enabled_when("print", !self.config().avoid_print) {
1211                if let Some(e) = self.gen_print(u)? {
1212                    return Ok(e);
1213                }
1214            }
1215        }
1216
1217        if freq.enabled("assign") {
1218            if let Some(e) = self.gen_assign(u)? {
1219                return Ok(e);
1220            }
1221        }
1222
1223        self.gen_let(u)
1224    }

1226    /// Generate a `Let` statement with arbitrary type and value.
1227    fn gen_let(&mut self, u: &mut Unstructured) -> arbitrary::Result<Expression> {
1228        // Generate a type or choose an existing one.
1229        let max_depth = self.max_depth();
1230        let comptime_friendly = self.config().comptime_friendly;
1231        let mut typ = self.ctx.gen_type(u, max_depth, false, false, comptime_friendly, true)?;
1232
1233        // If we picked the target type to be a vector, we can consider popping from it.
1234        if let Type::Vector(ref item_type) = typ {
1235            if bool::arbitrary(u)? {
1236                let fields = if bool::arbitrary(u)? {
1237                    // ([T], T) <- pop_back or remove
1238                    vec![typ.clone(), item_type.as_ref().clone()]
1239                } else {
1240                    // (T, [T]) <- pop_front
1241                    vec![item_type.as_ref().clone(), typ.clone()]
1242                };
1243                typ = Type::Tuple(fields);
1244            }
1245        }
1246
1247        let (expr, is_dyn) = self.gen_expr(u, &typ, max_depth, Flags::TOP)?;
1248        let mutable = bool::arbitrary(u)?;
1249        Ok(self.let_var(mutable, typ, expr, true, is_dyn, local_name))
1250    }

1252    /// Add a new local variable and return a `Let` expression.
1253    ///
1254    /// If `add_to_scope` is `false`, the value will not be added to the `locals`.
1255    fn let_var(
1256        &mut self,
1257        mutable: bool,
1258        typ: Type,
1259        expr: Expression,
1260        add_to_scope: bool,
1261        is_dynamic: bool,
1262        make_name: impl Fn(LocalId) -> String,
1263    ) -> Expression {
1264        let id = self.next_local_id();
1265        let name = make_name(id);
1266
1267        // Add the variable so we can use it in subsequent expressions.
1268        if add_to_scope {
1269            self.locals.add(id, mutable, name.clone(), typ.clone());
1270        }
1271
1272        self.set_dynamic(id, is_dynamic);
1273
1274        expr::let_var(id, mutable, name, expr)
1275    }

1301    /// Assign to a mutable variable, if we have one in scope.
1302    ///
1303    /// It resets the dynamic flag of the assigned variable.
1304    fn gen_assign(&mut self, u: &mut Unstructured) -> arbitrary::Result<Option<Expression>> {
1305        let opts = self
1306            .locals
1307            .current()
1308            .variables()
1309            .filter(|(_, (mutable, _, typ))| {
1310                // We banned reassigning variables which contain mutable references in ACIR (#8790)
1311                *mutable && (self.unconstrained() || !types::contains_reference(typ))
1312            })
1313            .map(|(id, _)| id)
1314            .collect::<Vec<_>>();
1315
1316        if opts.is_empty() {
1317            return Ok(None);
1318        }
1319
1320        let id = *u.choose_iter(opts)?;
1321        let ident = LValue::Ident(self.local_ident(id));
1322        let typ = self.local_type(id).clone();
1323        let lvalue = self.gen_lvalue(u, ident, typ)?;
1324
1325        // Generate the assigned value.
1326        let (expr, expr_dyn) = self.gen_expr(u, &lvalue.typ, self.max_depth(), Flags::TOP)?;
1327
1328        if lvalue.is_dyn || expr_dyn || self.in_dynamic {
1329            self.set_dynamic(id, true);
1330        } else if !lvalue.is_dyn && !expr_dyn && !lvalue.is_compound {
1331            // This value is no longer considered dynamic, unless we assigned to a member of an array or tuple,
1332            // in which case we don't know if other members have dynamic properties.
1333            self.set_dynamic(id, false);
1334        }
1335
1336        let assign =
1337            Expression::Assign(Assign { lvalue: lvalue.lvalue, expression: Box::new(expr) });
1338
1339        if let Some(mut statements) = lvalue.statements {
1340            statements.push(assign);
1341            Ok(Some(Expression::Block(statements)))
1342        } else {
1343            Ok(Some(assign))
1344        }
1345    }

1347    /// Generate an lvalue to assign to a local variable, or some part of it, if it's a compound type.
1348    ///
1349    /// Say we have an array: `a: [[u32; 2]; 3]`; we call it with `a`, and it might return `a`, `a[i]`, or `a[i][j]`.
1350    fn gen_lvalue(
1351        &mut self,
1352        u: &mut Unstructured,
1353        lvalue: LValue,
1354        typ: Type,
1355    ) -> arbitrary::Result<LValueWithMeta> {
1356        /// Accumulate statements for sub-indexes of multi-dimensional arrays.
1357        /// For example `a[1+2][3+4] = 5;` becomes `let i = 1+2; let j = 3+4; a[i][j] = 5;`
1358        fn merge_statements(
1359            a: Option<Vec<Expression>>,
1360            b: Option<Vec<Expression>>,
1361        ) -> Option<Vec<Expression>> {
1362            match (a, b) {
1363                (x, None) | (None, x) => x,
1364                (Some(mut a), Some(mut b)) => {
1365                    a.append(&mut b);
1366                    Some(a)
1367                }
1368            }
1369        }
1370        // For arrays and tuples we can consider assigning to their items.
1371        let lvalue = match typ {
1372            Type::Array(len, typ) if len > 0 && bool::arbitrary(u)? => {
1373                let (idx, idx_dyn) = self.gen_index(u, len, self.max_depth())?;
1374
1375                // If the index expressions can have side effects, we need to assign it to a
1376                // temporary variable to match the sequencing done by the frontend; see #8384.
1377                // In the compiler the `Elaborator::fresh_definition_for_lvalue_index` decides.
1378                let needs_prefix = !matches!(idx, Expression::Ident(_) | Expression::Literal(_));
1379
1380                let (idx, statements) = if needs_prefix {
1381                    let (let_idx, idx_ident) =
1382                        self.let_var_and_ident(false, types::U32, idx, false, idx_dyn, index_name);
1383                    (Expression::Ident(idx_ident), Some(vec![let_idx]))
1384                } else {
1385                    (idx, None)
1386                };
1387
1388                let typ = typ.as_ref().clone();
1389                let index = LValue::Index {
1390                    array: Box::new(lvalue),
1391                    index: Box::new(idx),
1392                    element_type: typ.clone(),
1393                    location: Location::dummy(),
1394                };
1395
1396                let mut lvalue = self.gen_lvalue(u, index, typ)?;
1397                lvalue.is_compound = true;
1398                lvalue.is_dyn |= idx_dyn;
1399                lvalue.statements = merge_statements(statements, lvalue.statements);
1400                lvalue
1401            }
1402            Type::Tuple(items) if bool::arbitrary(u)? => {
1403                let idx = u.choose_index(items.len())?;
1404                let typ = items[idx].clone();
1405                let member = LValue::MemberAccess { object: Box::new(lvalue), field_index: idx };
1406                let mut lvalue = self.gen_lvalue(u, member, typ)?;
1407                lvalue.is_compound = true;
1408                lvalue
1409            }
1410            typ => {
1411                LValueWithMeta { lvalue, typ, is_dyn: false, is_compound: false, statements: None }
1412            }
1413        };
1414
1415        Ok(lvalue)
1416    }

1418    /// Generate a `println` statement, if there is some printable local variable.
1419    ///
1420    /// For now this only works in unconstrained code. For constrained code we will
1421    /// need to generate a proxy function, which we can do as a follow-up pass,
1422    /// as it has to be done once per function signature.
1423    fn gen_print(&mut self, u: &mut Unstructured) -> arbitrary::Result<Option<Expression>> {
1424        let opts = self
1425            .locals
1426            .current()
1427            .variables()
1428            .filter_map(|(id, (_, _, typ))| types::is_printable(typ).then_some((id, typ)))
1429            // TODO(#10499): comptime function representations are at the moment just "(function)"
1430            // (disable printing functions if comptime_friendly is on)
1431            .filter(|(_, typ)| !types::is_function(typ) || !self.config().comptime_friendly)
1432            .collect::<Vec<_>>();
1433
1434        if opts.is_empty() {
1435            return Ok(None);
1436        }
1437
1438        // Print one of the variables as-is.
1439        let (id, typ) = u.choose_iter(opts)?;
1440        let id = *id;
1441
1442        // The print oracle takes 2 parameters: the newline marker and the value,
1443        // but it takes 2 more arguments: the type descriptor and the format string marker,
1444        // which are inserted automatically by the monomorphizer.
1445        let param_types = vec![Type::Bool, typ.clone()];
1446        let hir_type = types::to_hir_type(typ);
1447        let ident = self.local_ident(id);
1448
1449        // Functions need to be passed as a tuple.
1450        let arg = if types::is_function(&ident.typ) {
1451            Expression::Tuple(vec![
1452                Expression::Ident(ident),
1453                Expression::Ident(self.local_ident(id)),
1454            ])
1455        } else {
1456            Expression::Ident(ident)
1457        };
1458
1459        let mut args = vec![
1460            expr::lit_bool(true), // include newline,
1461            arg,
1462        ];
1463
1464        append_printable_type_info_for_type(hir_type, &mut args);
1465
1466        let print_oracle_ident = Ident {
1467            location: None,
1468            definition: Definition::Oracle("print".to_string()),
1469            mutable: false,
1470            name: "print_oracle".to_string(),
1471            typ: Type::Function(param_types, Box::new(Type::Unit), Box::new(Type::Unit), true),
1472            id: self.next_ident_id(),
1473        };
1474
1475        let call = Expression::Call(Call {
1476            func: Box::new(Expression::Ident(print_oracle_ident)),
1477            arguments: args,
1478            return_type: Type::Unit,
1479            location: Location::dummy(),
1480        });
1481
1482        Ok(Some(call))
1483    }

1504    /// Generate an if-then-else statement or expression.
1505    fn gen_if(
1506        &mut self,
1507        u: &mut Unstructured,
1508        typ: &Type,
1509        max_depth: usize,
1510        flags: Flags,
1511    ) -> arbitrary::Result<TrackedExpression> {
1512        // Decrease the budget so we avoid a potential infinite nesting of if expressions in the arms.
1513        self.decrease_budget(1);
1514
1515        let (cond, cond_dyn) = self.gen_expr(u, &Type::Bool, max_depth, Flags::CONDITION)?;
1516
1517        // If the `if` condition uses dynamic input, we cannot access certain constructs in the body in ACIR.
1518        // Note that this would be the case for `while` and `for` as well, however `while` is not used in ACIR,
1519        // and `for` has its own restriction of having to have compile-time boundaries, so this is only done where necessary.
1520        let in_dynamic = self.in_dynamic || cond_dyn;
1521        let was_in_dynamic = std::mem::replace(&mut self.in_dynamic, in_dynamic);
1522
1523        let (cons, cons_dyn) = {
1524            if flags.allow_blocks {
1525                self.gen_block(u, typ)?
1526            } else {
1527                self.gen_expr(u, typ, max_depth, flags)?
1528            }
1529        };
1530
1531        let alt = if types::is_unit(typ) && bool::arbitrary(u)? {
1532            None
1533        } else {
1534            self.decrease_budget(1);
1535            let expr = if flags.allow_blocks {
1536                self.gen_block(u, typ)?
1537            } else {
1538                self.gen_expr(u, typ, max_depth, flags)?
1539            };
1540            Some(expr)
1541        };
1542
1543        self.in_dynamic = was_in_dynamic;
1544        let alt_dyn = alt.as_ref().is_some_and(|(_, d)| *d);
1545        let is_dyn = cond_dyn || cons_dyn || alt_dyn;
1546
1547        Ok((expr::if_then(cond, cons, alt.map(|(a, _)| a), typ.clone()), is_dyn))
1548    }

1550    /// Generate a `for` loop.
1551    fn gen_for(&mut self, u: &mut Unstructured) -> arbitrary::Result<Expression> {
1552        // The index can be signed or unsigned int, 8 to 128 bits, except i128,
1553        // but currently the frontend expects it to be u32 unless it's declared as a separate variable.
1554        let idx_type = {
1555            let bit_size = if self.config().avoid_large_int_literals {
1556                IntegerBitSize::ThirtyTwo
1557            } else {
1558                u.choose(&[8, 16, 32, 64, 128]).map(|s| IntegerBitSize::try_from(*s).unwrap())?
1559            };
1560
1561            Type::Integer(
1562                if bit_size == IntegerBitSize::HundredTwentyEight
1563                    || self.config().avoid_negative_int_literals
1564                    || bool::arbitrary(u)?
1565                {
1566                    Signedness::Unsigned
1567                } else {
1568                    Signedness::Signed
1569                },
1570                bit_size,
1571            )
1572        };
1573
1574        let (start_range, end_range) = if self.unconstrained() && bool::arbitrary(u)? {
1575            // Choosing a maximum range size because changing it immediately brought out some bug around modulo.
1576            let max_size = u.int_in_range(1..=self.config().max_loop_size)?;
1577            // Generate random expression.
1578            let (s, _) = self.gen_expr(u, &idx_type, self.max_depth(), Flags::RANGE)?;
1579            let (e, _) = self.gen_expr(u, &idx_type, self.max_depth(), Flags::RANGE)?;
1580            // The random expressions might end up being huge to be practical for execution,
1581            // so take the modulo maximum range on both ends.
1582            let s = expr::range_modulo(s, idx_type.clone(), max_size);
1583            let e = expr::range_modulo(e, idx_type.clone(), max_size);
1584            (s, e)
1585        } else {
1586            // `gen_range` will choose a size up to the max.
1587            let max_size = self.config().max_loop_size;
1588            // If the function is constrained, we need a range we can determine at compile time.
1589            // For now do it with literals, although we should be able to use constant variables as well.
1590            let (s, e) = expr::gen_range(u, &idx_type, max_size)?;
1591            // The compiler allows the end to be lower than the start.
1592            if u.ratio(1, 5)? { (e, s) } else { (s, e) }
1593        };
1594
1595        // Declare index variable, but only visible in the loop body, not the range.
1596        let idx_id = self.next_local_id();
1597        let idx_name = index_name(idx_id);
1598
1599        // Add a scope which will hold the index variable.
1600        self.enter_scope();
1601        self.locals.add(idx_id, false, idx_name.clone(), idx_type.clone());
1602
1603        // Decrease budget so we don't nest for loops endlessly.
1604        self.decrease_budget(1);
1605
1606        let was_in_loop = std::mem::replace(&mut self.in_loop, true);
1607        let (block, _) = self.gen_block(u, &Type::Unit)?;
1608        self.in_loop = was_in_loop;
1609
1610        let expr = Expression::For(For {
1611            index_variable: idx_id,
1612            index_name: idx_name,
1613            index_type: idx_type,
1614            start_range: Box::new(start_range),
1615            end_range: Box::new(end_range),
1616            block: Box::new(block),
1617            inclusive: bool::arbitrary(u)?,
1618            start_range_location: Location::dummy(),
1619            end_range_location: Location::dummy(),
1620        });
1621
1622        // Remove the loop scope.
1623        self.exit_scope();
1624
1625        Ok(expr)
1626    }

1628    /// Generate a function call to any function in the global context except `main`,
1629    /// if the function returns the target type, or something we can use to produce that type.
1630    ///
1631    /// Whether a call is dynamic depends on whether it has dynamic arguments,
1632    /// and whether we are crossing an ACIR-to-Brillig boundary.
1633    fn gen_call(
1634        &mut self,
1635        u: &mut Unstructured,
1636        typ: &Type,
1637        max_depth: usize,
1638    ) -> arbitrary::Result<Option<TrackedExpression>> {
1639        // Decrease the budget so we avoid a potential infinite nesting of calls.
1640        self.decrease_budget(1);
1641
1642        let opts = self
1643            .call_targets
1644            .iter()
1645            .filter(|(id, types)|
1646                // We need to be able to generate the type from what the function returns.
1647                types.contains(typ) &&
1648                // We might not be able to call this function, depending on context.
1649                !(self.in_no_dynamic && !self.unconstrained() && self.callable_signature(**id).2))
1650            .map(|(id, _)| id)
1651            .collect::<Vec<_>>();
1652
1653        if opts.is_empty() {
1654            return Ok(None);
1655        }
1656
1657        // Remember that we will have made a call to something.
1658        self.has_call = true;
1659
1660        let callee_id = *u.choose_iter(opts)?;
1661        let callee_expr = self.callable_expr(callee_id);
1662        let (param_types, return_type, callee_unconstrained) = self.callable_signature(callee_id);
1663
1664        // Generate an expression for each argument.
1665        let mut args = Vec::new();
1666        let mut call_dyn = !self.unconstrained() && callee_unconstrained;
1667        for typ in &param_types {
1668            let (arg, arg_dyn) = self.gen_expr(u, typ, max_depth, Flags::CALL)?;
1669            call_dyn |= arg_dyn;
1670            args.push(arg);
1671        }
1672
1673        let call_expr = Expression::Call(Call {
1674            func: Box::new(callee_expr),
1675            arguments: args,
1676            return_type: return_type.clone(),
1677            location: Location::dummy(),
1678        });
1679
1680        // Derive the final result from the call, e.g. by casting, or accessing a member.
1681        self.gen_expr_from_source(
1682            u,
1683            (call_expr, call_dyn),
1684            &return_type,
1685            true,
1686            typ,
1687            self.max_depth(),
1688        )
1689    }

1716    /// Generate a `loop` loop.
1717    fn gen_loop(&mut self, u: &mut Unstructured) -> arbitrary::Result<Expression> {
1718        // Declare break index variable visible in the loop body. Do not include it
1719        // in the locals the generator would be able to manipulate, as it could
1720        // lead to the loop becoming infinite.
1721        let (idx_local_id, idx_name, idx_ident) = self.next_loop_index();
1722        let idx_expr = Expression::Ident(idx_ident.clone());
1723
1724        // Decrease budget so we don't nest endlessly.
1725        self.decrease_budget(1);
1726
1727        // Start building the loop harness, initialize index to 0
1728        let let_idx = expr::let_var(idx_local_id, true, idx_name, expr::u32_literal(0));
1729
1730        // Get the randomized loop body
1731        let was_in_loop = std::mem::replace(&mut self.in_loop, true);
1732        let (mut loop_body, _) = self.gen_block(u, &Type::Unit)?;
1733        self.in_loop = was_in_loop;
1734
1735        // Increment the index in the beginning of the body.
1736        expr::prepend(
1737            &mut loop_body,
1738            expr::assign_ident(
1739                idx_ident,
1740                expr::binary(idx_expr.clone(), BinaryOp::Add, expr::u32_literal(1)),
1741            ),
1742        );
1743
1744        // Put everything into if/else
1745        let max_loop_size = self.gen_loop_size(u)?;
1746        let loop_body = expr::if_else(
1747            expr::binary(idx_expr, BinaryOp::Equal, expr::u32_literal(max_loop_size as u32)),
1748            Expression::Break,
1749            loop_body,
1750            Type::Unit,
1751        );
1752
1753        Ok(Expression::Block(vec![let_idx, Expression::Loop(Box::new(loop_body))]))
1754    }

1818    /// Generate a `match` expression, returning a given type.
1819    ///
1820    /// Match needs a variable; if we don't have one to produce the target type from, it returns `None`.
1821    fn gen_match(
1822        &mut self,
1823        u: &mut Unstructured,
1824        typ: &Type,
1825        max_depth: usize,
1826    ) -> arbitrary::Result<Option<TrackedExpression>> {
1827        // Decrease the budget so we avoid a potential infinite nesting of match expressions in the rows.
1828        self.decrease_budget(1);
1829
1830        // Pick a variable that can produce the type we are looking for.
1831        let id = if types::is_unit(typ) {
1832            // If we are generating a statement (return unit), then let's just pick any local variable.
1833            if self.locals.current().is_empty() {
1834                None
1835            } else {
1836                let id = u.choose_iter(self.locals.current().variable_ids())?;
1837                Some(VariableId::Local(*id))
1838            }
1839        } else {
1840            self.choose_producer(u, typ)?
1841        };
1842
1843        // If we have no viable candidate then do something else.
1844        let Some(id) = id else {
1845            return Ok(None);
1846        };
1847
1848        // If we picked a global variable, we need to create a local binding first,
1849        // because the match only works with local variable IDs.
1850        // We also need to create a secondary local binding for a local variable,
1851        // because of how the ownership analysis works: it only tracks the use of
1852        // identifiers, but the `Match::variable_to_match` only contains a `LocalId`.
1853        // We could change it to an `Ident`, but that's not enough: when the ownership
1854        // inserts `Clone` expressions for all but the last use of an ident, it could
1855        // not do so with the `Match`, because it would need match on an `Expression`.
1856        let (src_id, src_name, src_typ, src_dyn, src_expr) = match id {
1857            VariableId::Global(id) => {
1858                let typ = self.globals.get_variable(&id).2.clone();
1859                // The source is a technical variable that we don't want to access in the match rows.
1860                let (id, name, let_expr) = self.indirect_global(id, false, false);
1861                (id, name, typ, false, let_expr)
1862            }
1863            VariableId::Local(id) => {
1864                let typ = self.local_type(id).clone();
1865                let (id, name, let_expr) = self.indirect_local(id, false, false);
1866                let is_dyn = self.is_dynamic(&id);
1867                (id, name, typ, is_dyn, let_expr)
1868            }
1869        };
1870
1871        // We could add some filtering to `choose_producer`, but it's already complicated; maybe next time.
1872        if !types::can_be_matched(&src_typ) {
1873            return Ok(None);
1874        }
1875
1876        // Similar to an `if` statement, if the source variable is dynamic, we can't do certain things in the body.
1877        let in_dynamic = self.in_dynamic || src_dyn;
1878        let was_in_dynamic = std::mem::replace(&mut self.in_dynamic, in_dynamic);
1879
1880        let mut match_expr = Match {
1881            variable_to_match: (src_id, src_name),
1882            cases: vec![],
1883            default_case: None,
1884            typ: typ.clone(),
1885        };
1886
1887        let num_cases = u.int_in_range(0..=self.ctx.config.max_match_cases)?;
1888
1889        // The dynamic nature of the final expression depends on the source and the rules.
1890        let mut is_dyn = src_dyn;
1891
1892        // Generate a number of rows, depending on what we can do with the source type.
1893        // See `MatchCompiler::compile_rows` for what is currently supported.
1894        let gen_default = match &src_typ {
1895            Type::Bool => {
1896                // There are only two possible values. Repeating one of them results in a warning,
1897                // but let's allow it just so we cover that case, since it's not an error.
1898                for _ in 0..num_cases {
1899                    let constructor = u.choose_iter([Constructor::True, Constructor::False])?;
1900                    let (branch, branch_dyn) = self.gen_expr(u, typ, max_depth, Flags::TOP)?;
1901                    is_dyn |= branch_dyn;
1902                    let case = MatchCase { constructor, arguments: Vec::new(), branch };
1903                    match_expr.cases.push(case);
1904                }
1905
1906                // If we have a non-exhaustive match we have to have a default; otherwise it's optional.
1907                #[allow(clippy::mutable_key_type)]
1908                let cs =
1909                    match_expr.cases.iter().map(|c| c.constructor.clone()).collect::<HashSet<_>>();
1910
1911                if cs.len() < 2 { true } else { bool::arbitrary(u)? }
1912            }
1913            Type::Field | Type::Integer(_, _) => {
1914                for _ in 0..num_cases {
1915                    let constructor = self.gen_num_match_constructor(u, &src_typ)?;
1916                    let (branch, branch_dyn) = self.gen_expr(u, typ, max_depth, Flags::TOP)?;
1917                    is_dyn |= branch_dyn;
1918                    let case = MatchCase { constructor, arguments: Vec::new(), branch };
1919                    match_expr.cases.push(case);
1920                }
1921                // We won't have an exhaustive match with random integers, so we need a default.
1922                true
1923            }
1924            Type::Tuple(item_types) => {
1925                // There is only one case in the AST that we can generate, which is to unpack the tuple
1926                // into its constituent fields. The compiler would do this, and then generate further
1927                // matches on individual fields. We don't do that here, just make the fields available.
1928                let constructor = Constructor::Tuple(vecmap(item_types, types::to_hir_type));
1929                let mut arguments = Vec::new();
1930                self.enter_scope();
1931                for item_type in item_types {
1932                    let item_id = self.next_local_id();
1933                    let item_name = format!("item_{}", local_name(item_id));
1934                    self.locals.add(item_id, false, item_name.clone(), item_type.clone());
1935                    self.set_dynamic(item_id, src_dyn);
1936                    arguments.push((item_id, item_name));
1937                }
1938                // Generate the original expression we wanted with the new arguments in scope.
1939                let (branch, branch_dyn) = self.gen_expr(u, typ, max_depth, Flags::TOP)?;
1940                is_dyn |= branch_dyn;
1941                let case = MatchCase { constructor, arguments, branch };
1942                match_expr.cases.push(case);
1943                self.exit_scope();
1944                // We must not generate a default, or the compiler will panic.
1945                false
1946            }
1947            other => {
1948                unreachable!("unexpected type to generate match for: ${other}");
1949            }
1950        };
1951
1952        // Optionally generate a default case.
1953        if gen_default {
1954            let (default_expr, default_dyn) = self.gen_expr(u, typ, max_depth, Flags::TOP)?;
1955            is_dyn |= default_dyn;
1956            match_expr.default_case = Some(Box::new(default_expr));
1957        }
1958
1959        self.in_dynamic = was_in_dynamic;
1960        let match_expr = Expression::Match(match_expr);
1961        let expr = Expression::Block(vec![src_expr, match_expr]);
1962
1963        Ok(Some((expr, is_dyn)))
1964    }
