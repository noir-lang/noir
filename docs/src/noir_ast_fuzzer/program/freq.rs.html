5/// Frequency distribution of generators.
6#[derive(Debug, Clone)]
7pub struct Freqs {
8    items: im::HashMap<&'static str, usize>,
9    total: usize,
10}
11
12impl Freqs {
13    pub fn new(items: &[(&'static str, usize)]) -> Self {
14        let total = items.iter().map(|i| i.1).sum();
15        Self { items: items.iter().cloned().collect(), total }
16    }
17    pub fn total(&self) -> usize {
18        self.total
19    }
20}
21
22impl Index<&str> for Freqs {
23    type Output = usize;
24
25    fn index(&self, index: &str) -> &Self::Output {
26        self.items.get(index).unwrap_or_else(|| panic!("unknown freq: {index}"))
27    }
28}

30/// Help with cumulative frequency distributions.
31pub(crate) struct Freq {
32    freqs: Freqs,
33    x: usize,
34    accumulated: usize,
35    disabled: usize,
36}
37
38impl Freq {
39    pub(crate) fn new(u: &mut Unstructured, freqs: &Freqs) -> arbitrary::Result<Self> {
40        let x = u.choose_index(freqs.total())?;
41        Ok(Self { freqs: freqs.clone(), x, accumulated: 0, disabled: 0 })
42    }
43
44    /// Check if a key is enabled, based on the already checked cumulative values.
45    ///
46    /// To work correctly, `enabled` calls should come after `enabled_when`s.
47    pub(crate) fn enabled(&mut self, key: &str) -> bool {
48        self.accumulated += self.freqs[key];
49        self.passed()
50    }
51
52    /// Like `enabled`, but if `cond` is `false` it redistributes the probability
53    /// to the remaining keys, pretending to works as if the current one never existed.
54    ///
55    /// To work correctly, `enabled_when` calls should precede `enabled`s.
56    ///
57    /// It's not guaranteed to prevent any distortions; for example it can happen that
58    /// the first key we check is enabled by `cond`, but we skip it because `x` is higher
59    /// than its weight, only to then realize that all following keys are disabled.
60    /// In this case the weight of the first check was different than if we considered
61    /// all conditions up front, summed up the remaining weights, and then chose `x`
62    /// accordingly. Doing so could potentially slow things down.
63    ///
64    /// As a workaround, we should
65    /// * check keys which are more likely to be disabled up front
66    /// * have a default case at the end which is always enabled
67    pub(crate) fn enabled_when(&mut self, key: &str, cond: bool) -> bool {
68        if cond {
69            self.enabled(key)
70        } else {
71            self.disabled += self.freqs[key];
72            false
73        }
74    }
75
76    /// Check if the accumulated weights have passed the random variable `x`,
77    /// adjusted with the total disabled weight.
78    fn passed(&self) -> bool {
79        if self.freqs.total == 0 {
80            return false;
81        }
82        let adj_total = self.freqs.total - self.disabled;
83        let adj_x = self.x * adj_total / self.freqs.total;
84        self.accumulated > adj_x
85    }
86}

88#[cfg(test)]
89mod tests {
90    use super::{Freq, Freqs};
91
92    /// Create a test distribution with a given `x in 0..100`.
93    fn make_test_freq(x: usize) -> Freq {
94        assert!(x < 100, "scale goes to 100");
95        let freqs = Freqs::new(&[("foo", 40), ("bar", 30), ("baz", 20), ("qux", 10)]);
96        Freq { freqs, accumulated: 0, disabled: 0, x }
97    }
98
99    /// Test that it enables baz, because the cumulative freq including baz is 90.
100    #[test]
101    fn test_freq_enabled() {
102        let mut freq = make_test_freq(80);
103        assert!(!freq.enabled("foo"));
104        assert!(!freq.enabled("bar"));
105        assert!(freq.enabled("baz"));
106    }
107
108    /// Test that changing the order results in a different label being enabled,
109    /// because the cross the threshold at a different call.
110    #[test]
111    fn test_freq_enabled_order() {
112        let mut freq = make_test_freq(80);
113        assert!(!freq.enabled("baz"));
114        assert!(!freq.enabled("foo"));
115        assert!(freq.enabled("bar"));
116    }
117
118    /// Test that some labels being disabled redistribute the probability
119    /// on the remaining ones. In this example we choose x=75 out of 100,
120    /// which is the 75th percentile, but disable the first 70 weights,
121    /// which means if we want `x` to be the 75th percentile of the
122    /// remaining 30 we adjust it down to 75*30/100 = 22, which enables
123    /// qux az instead of baz, which would have originally fallen between [70,90).
124    #[test]
125    fn test_freq_redistribution() {
126        let mut freq = make_test_freq(75);
127        assert!(!freq.enabled_when("foo", false));
128        assert!(!freq.enabled_when("bar", false));
129        assert!(!freq.enabled_when("baz", false));
130        assert!(freq.enabled("qux"));
131    }
