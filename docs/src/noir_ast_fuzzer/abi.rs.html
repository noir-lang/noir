9/// Generate the [Abi] interface of a [Program].
10pub fn program_abi(program: &Program) -> Abi {
11    let main = program.main();
12
13    let parameters = main
14        .parameters
15        .iter()
16        .map(|(_id, _is_mutable, name, typ, vis)| AbiParameter {
17            name: name.clone(),
18            typ: to_abi_type(typ),
19            visibility: to_abi_visibility(vis),
20        })
21        .collect();
22
23    let return_type = match &main.return_type {
24        Type::Unit => None,
25        typ => Some(AbiReturnType {
26            abi_type: to_abi_type(typ),
27            visibility: to_abi_visibility(&program.return_visibility()),
28        }),
29    };
30
31    Abi { parameters, return_type, error_types: BTreeMap::default() }
32}
33
34/// Check if a type is valid as an ABI parameter for the `main` function.
35fn is_valid_in_abi(typ: &Type) -> bool {
36    match typ {
37        Type::Unit
38        | Type::FmtString(_, _)
39        | Type::Vector(_)
40        | Type::Reference(_, _)
41        | Type::Function(_, _, _, _) => false,
42
43        Type::Field | Type::Bool | Type::String(_) | Type::Integer(_, _) => true,
44
45        Type::Array(_, typ) => is_valid_in_abi(typ),
46        Type::Tuple(items) => items.iter().all(is_valid_in_abi),
47    }
48}
49
50/// Map an AST type to an ABI type.
51///
52/// Panics if it's called on a type which should not appear in the ABI.
53fn to_abi_type(typ: &Type) -> AbiType {
54    match typ {
55        Type::Field => AbiType::Field,
56        Type::Array(len, typ) => AbiType::Array { length: *len, typ: Box::new(to_abi_type(typ)) },
57        Type::Integer(signedness, integer_bit_size) => AbiType::Integer {
58            sign: if signedness.is_signed() { Sign::Signed } else { Sign::Unsigned },
59            width: integer_bit_size.bit_size().into(),
60        },
61        Type::Bool => AbiType::Boolean,
62        Type::String(len) => AbiType::String { length: *len },
63        Type::Tuple(items) => AbiType::Tuple { fields: items.iter().map(to_abi_type).collect() },
64
65        _ => {
66            if !is_valid_in_abi(typ) {
67                // We should not have generated such a parameter for the main function.
68                panic!("Invalid type in ABI: {typ}");
69            } else {
70                unreachable!("Unexpected type in ABI: {typ}")
71            }
72        }
73    }
74}
75
76fn to_abi_visibility(vis: &Visibility) -> AbiVisibility {
77    match vis {
78        Visibility::Public => AbiVisibility::Public,
79        Visibility::Private => AbiVisibility::Public,
80        Visibility::CallData(_) => AbiVisibility::DataBus,
81        Visibility::ReturnData => AbiVisibility::DataBus,
82    }
83}
