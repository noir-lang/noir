29/// Subset of [SsaEvaluatorOptions] that we want to vary.
30///
31/// It exists to reduce noise in the printed results, compared to showing the full `SsaEvaluatorOptions`.
32#[derive(Debug, Clone, Default)]
33pub struct CompareOptions {
34    pub inliner_aggressiveness: i64,
35}
36
37impl Arbitrary<'_> for CompareOptions {
38    fn arbitrary(u: &mut Unstructured<'_>) -> arbitrary::Result<Self> {
39        Ok(Self { inliner_aggressiveness: *u.choose(&[i64::MIN, 0, i64::MAX])? })
40    }
41}
42
43impl CompareOptions {
44    /// Copy fields into an [SsaEvaluatorOptions] instance.
45    pub fn onto(&self, mut options: SsaEvaluatorOptions) -> SsaEvaluatorOptions {
46        options.inliner_aggressiveness = self.inliner_aggressiveness;
47        options
48    }
49}

51/// Help ignore results and errors we find equivalent between executions.
52pub trait Comparable {
53    fn equivalent(a: &Self, b: &Self) -> bool;
54}
55
56impl<T: Comparable> Comparable for Vec<T> {
57    fn equivalent(a: &Self, b: &Self) -> bool {
58        a.len() == b.len() && a.iter().zip(b).all(|(a, b)| Comparable::equivalent(a, b))
59    }
60}
61
62impl<T: Comparable> Comparable for Option<T> {
63    fn equivalent(a: &Self, b: &Self) -> bool {
64        match (a, b) {
65            (Some(a), Some(b)) => Comparable::equivalent(a, b),
66            (None, None) => true,
67            _ => false,
68        }
69    }
70}
71
72impl<T: Comparable> Comparable for Shared<T> {
73    fn equivalent(a: &Self, b: &Self) -> bool {
74        let a = a.borrow();
75        let b = b.borrow();
76        Comparable::equivalent(a.deref(), b.deref())
77    }
78}

89/// We want to inspect side effects even on failures, so we can treat different failures
90/// as equivalent as long as the other side effects are equivalent.

93    pub error: E,
94    pub print_output: String,

97/// Possible outcomes of the differential execution of two equivalent programs.
98///
99/// Use [CompareResult::return_value_or_err] to do the final comparison between
100/// the execution result.
101pub enum CompareResult<T, E> {
102    BothFailed(FailedOutput<E>, FailedOutput<E>),
103    LeftFailed(FailedOutput<E>, PassedOutput<T>),
104    RightFailed(PassedOutput<T>, FailedOutput<E>),
105    BothPassed(PassedOutput<T>, PassedOutput<T>),
106}
107
108impl<T, E> CompareResult<T, E>
109where
110    E: Comparable + Display + Debug,
111    T: Comparable + Debug,
112{
113    /// Check that two programs agree on a return value.
114    ///
115    /// Returns an error if anything is different.
116    pub fn return_value_or_err(&self) -> eyre::Result<Option<&T>> {
117        match self {
118            CompareResult::BothFailed(e1, e2) => {
119                if !Comparable::equivalent(&e1.error, &e2.error) {
120                    let e1 = &e1.error;
121                    let e2 = &e2.error;
122                    bail!(
123                        "both programs failed in non-equivalent ways:\n{e1}\n!~\n{e2}\n\n{e1:?}\n{e2:?}"
124                    );
125                } else {
126                    let p1 = &e1.print_output;
127                    let p2 = &e2.print_output;
128                    if p1 != p2 {
129                        bail!(
130                            "both programs failed, but disagree on printed output:\n---\n{p1}\n--- != ---\n{p2}\n---",
131                        );
132                    } else {
133                        Ok(None)
134                    }
135                }
136            }
137            CompareResult::LeftFailed(e, _) => {
138                let e = &e.error;
139                bail!("first program failed: {e}\n\n{e:?}")
140            }
141            CompareResult::RightFailed(_, e) => {
142                let e = &e.error;
143                bail!("second program failed: {e}\n\n{e:?}")
144            }
145            CompareResult::BothPassed(o1, o2) => match (&o1.return_value, &o2.return_value) {
146                (Some(r1), Some(r2)) if !Comparable::equivalent(r1, r2) => {
147                    bail!("programs disagree on return value:\n{r1:?}\n!=\n{r2:?}",)
148                }
149                (Some(r1), None) => {
150                    bail!("only the first program returned a value: {r1:?}",)
151                }
152                (None, Some(r2)) => {
153                    bail!("only the second program returned a value: {r2:?}",)
154                }
155                (r1, _) => {
156                    let p1 = &o1.print_output;
157                    let p2 = &o2.print_output;
158                    if p1 != p2 {
159                        bail!(
160                            "both programs passed, but disagree on printed output:\n---\n{p1}\n--- != ---\n{p2}\n---",
161                        )
162                    }
163                    Ok(r1.as_ref())
164                }
165            },
166        }
167    }
168}

170/// We can turn on the logging of artifacts by setting the `RUST_LOG=debug` env var.
171///
172/// This can help reproducing failures. The functions in this module can help cut back some repetition.
173/// Log things as soon as they are available in case the next step fails.
174mod logging {
175    use noirc_abi::{Abi, InputMap};
176    use noirc_evaluator::ssa::{interpreter::value::Value, ssa_gen::Ssa};
177    use noirc_frontend::monomorphization::ast::Program;
178
179    use crate::{DisplayAstAsNoir, compare::CompareOptions};
180
181    fn format_msg(msg: &str) -> String {
182        if msg.is_empty() { String::new() } else { format!(" ({msg})") }
183    }
184
185    pub(super) fn log_program(program: &Program, msg: &str) {
186        log::debug!("AST{}:\n{}\n", format_msg(msg), DisplayAstAsNoir(program));
187    }
188
189    pub(super) fn log_ssa(ssa: &Ssa, msg: &str) {
190        log::debug!("SSA{}:\n{}\n", format_msg(msg), ssa.print_without_locations());
191    }
192
193    pub(super) fn log_comptime(src: &str, msg: &str) {
194        log::debug!("comptime source{}:\n{}\n", format_msg(msg), src);
195    }
196
197    pub(super) fn log_options(options: &CompareOptions, msg: &str) {
198        log::debug!("Options{}:\n{:?}\n", format_msg(msg), options);
199    }
200
201    pub(super) fn log_abi_inputs(abi: &Abi, input_map: &InputMap) {
202        log::debug!(
203            "ABI inputs:\n{}\n",
204            noirc_abi::input_parser::Format::Toml
205                .serialize(input_map, abi)
206                .unwrap_or_else(|e| format!("failed to serialize inputs: {e}"))
207        );
208    }
209
210    pub(super) fn log_ssa_inputs(ssa_args: &[Value]) {
211        log::debug!(
212            "SSA inputs:\n{}\n",
213            ssa_args
214                .iter()
215                .enumerate()
216                .map(|(i, v)| format!("{i}: {v}"))
217                .collect::<Vec<_>>()
218                .join("\n")
219        );
220    }
