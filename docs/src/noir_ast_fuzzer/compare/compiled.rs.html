36/// These are the error types in the `SsaProgramArtifact`, which are not the same as the ones in the ABI,
37/// but they can provide extra information when comparing errors.
38type SsaErrorTypes = BTreeMap<acir::circuit::ErrorSelector, ErrorType>;
39
40/// The execution result is the value returned from the circuit and any output from `println`.
41pub(crate) type ExecResult = (Result<WitnessStack<FieldElement>, NargoError<FieldElement>>, String);
42
43#[derive(Debug)]
44pub struct NargoErrorWithTypes(NargoError<FieldElement>, SsaErrorTypes);
45
46impl NargoErrorWithTypes {
47    /// Copy of `NargoError::user_defined_failure_message` accepting `SsaErrorTypes` instead of ABI errors.
48    fn user_defined_failure_message(&self) -> Option<String> {
49        let unwrap_payload = |payload: &ResolvedAssertionPayload<FieldElement>| {
50            match payload {
51                ResolvedAssertionPayload::String(message) => Some(message.clone()),
52                ResolvedAssertionPayload::Raw(raw) => {
53                    if let Some(ssa_type) = self.1.get(&raw.selector) {
54                        match ssa_type {
55                            ErrorType::String(message) => Some(message.clone()),
56                            ErrorType::Dynamic(hir_type) => {
57                                // A non-literal assert message — for example an `if`-expression that
58                                // produces a string, as the metamorphic rewriter can introduce — is
59                                // recorded by `codegen_constrain_error` as `ErrorType::Dynamic`, even
60                                // when its type is a plain string. Recover the message from the raw
61                                // payload bytes when they encode a string, matching the encoding of
62                                // `ConstrainError::Dynamic { is_string_type: true, .. }`.
63                                //
64                                // A format string (as synthesized for the composite-array
65                                // out-of-bounds error) encodes its template text in the leading
66                                // `length` fields, followed by the field count and the formatted
67                                // values. Decode just the template — it is enough to match the error
68                                // textually and avoids needing full ABI formatting of the values.
69                                if let HirType::FmtString(length, _) = hir_type
70                                    && let HirType::Constant(int) = length.as_ref()
71                                    && let Some(len) = int.as_field().try_to_u64()
72                                {
73                                    return raw
74                                        .data
75                                        .get(..len as usize)
76                                        .and_then(decode_raw_string_payload);
77                                }
78                                decode_raw_string_payload(&raw.data)
79                            }
80                        }
81                    } else {
82                        // `codegen_constrain_error` only records the SSA `ErrorType` when the
83                        // assert message type is *not* a string (see `is_string_type` branch in
84                        // `compiler/noirc_evaluator/src/ssa/ssa_gen/mod.rs`). When a non-literal
85                        // string expression is used as an assert message, the runtime payload
86                        // therefore carries the bytes of the string with a selector that has no
87                        // entry in the SSA error type table. Recover the message from the bytes.
88                        decode_raw_string_payload(&raw.data)
89                    }
90                }
91            }
92        };
93
94        match &self.0 {
95            NargoError::ExecutionError(error) => match error {
96                ExecutionError::AssertionFailed(payload, _, _) => unwrap_payload(payload),
97                ExecutionError::SolvingError(error, _) => match error {
98                    OpcodeResolutionError::BlackBoxFunctionFailed(_, reason) => {
99                        Some(reason.clone())
100                    }
101                    OpcodeResolutionError::BrilligFunctionFailed {
102                        payload: Some(payload), ..
103                    } => unwrap_payload(payload),
104                    // An out-of-bounds array read surfaces as a solving error in ACIR rather than a
105                    // user assert. The equivalence check intentionally treats it as comparable to a
106                    // Brillig "attempt to add/multiply with overflow" (the explicit overflow range
107                    // check on an index can be elided by the redundant-range optimizer because the
108                    // array access bounds-check subsumes it). Surfacing a message here is what lets
109                    // that equivalence fire instead of being reported as a spurious divergence.
110                    OpcodeResolutionError::IndexOutOfBounds { .. } => {
111                        Some("Index out of bounds".to_string())
112                    }
113                    _ => None,
114                },
115            },
116            NargoError::ForeignCallError(error) => Some(error.to_string()),
117            NargoError::CompilationError => None,
118        }
119    }
120}

122/// Try to interpret the raw bytes of an assertion payload as a UTF-8 string,
123/// matching the encoding produced by `ConstrainError::Dynamic { is_string_type: true, .. }`.
124///
125/// Returns `None` if the data is empty, contains a field element wider than a byte,
126/// or the bytes are not valid UTF-8.
127fn decode_raw_string_payload(data: &[FieldElement]) -> Option<String> {
128    if data.is_empty() {
129        return None;
130    }
131    let bytes: Vec<u8> = data
132        .iter()
133        .map(|field| {
134            let be = field.to_be_bytes();
135            let (low, high) = be.split_last().expect("FieldElement::to_be_bytes is non-empty");
136            high.iter().all(|b| *b == 0).then_some(*low)
137        })
138        .collect::<Option<Vec<_>>>()?;
139    String::from_utf8(bytes).ok()
140}

142/// The result of the execution of compiled programs, decoded by their ABI.
143pub type CompareCompiledResult = CompareResult<InputValue, NargoErrorWithTypes>;
144
145impl CompareCompiledResult {
146    pub fn new(
147        abi: &Abi,
148        ets1: &SsaErrorTypes,
149        ets2: &SsaErrorTypes,
150        (res1, print1): ExecResult,
151        (res2, print2): ExecResult,
152    ) -> eyre::Result<Self> {
153        let decode = |ws: WitnessStack<FieldElement>| -> eyre::Result<Option<InputValue>> {
154            let wm = &ws.peek().expect("there should be a main witness").witness;
155            let (_, r) = abi.decode(wm).wrap_err("abi::decode")?;
156            Ok(r)
157        };
158
159        let failed = |e, ets: &SsaErrorTypes, p: String| FailedOutput {
160            error: NargoErrorWithTypes(e, ets.clone()),
161            print_output: p,
162        };
163
164        let passed = |ws, p| decode(ws).map(|r| PassedOutput { return_value: r, print_output: p });
165
166        match (res1, res2) {
167            (Err(e1), Err(e2)) => {
168                Ok(CompareResult::BothFailed(failed(e1, ets1, print1), failed(e2, ets2, print2)))
169            }
170            (Err(e1), Ok(ws2)) => {
171                Ok(CompareResult::LeftFailed(failed(e1, ets1, print1), passed(ws2, print2)?))
172            }
173            (Ok(ws1), Err(e2)) => {
174                Ok(CompareResult::RightFailed(passed(ws1, print1)?, failed(e2, ets2, print2)))
175            }
176            (Ok(ws1), Ok(ws2)) => {
177                Ok(CompareResult::BothPassed(passed(ws1, print1)?, passed(ws2, print2)?))
178            }
179        }
180    }
181}

183impl Comparable for NargoErrorWithTypes {
184    fn equivalent(e1: &Self, e2: &Self) -> bool {
185        use ExecutionError::*;
186
187        // For now consider non-execution errors as failures we need to investigate.
188        let NargoError::ExecutionError(ee1) = &e1.0 else {
189            return false;
190        };
191        let NargoError::ExecutionError(ee2) = &e2.0 else {
192            return false;
193        };
194
195        // We have a notion of treating errors as equivalents as long as the side effects
196        // of the failed program are otherwise the same. For this reason we compare the
197        // prints in `return_value_or_err`. Here we have the option to tweak which errors
198        // we consider equivalents, but that's really just to stay on the conservative
199        // side and give us a chance to inspect new kinds of test failures.
200
201        fn both<F: Fn(&str) -> bool>(s1: &str, s2: &str, f: F) -> bool {
202            f(s1) && f(s2)
203        }
204
205        let msg1 = e1.user_defined_failure_message();
206        let msg2 = e2.user_defined_failure_message();
207        let equiv_msgs = if let (Some(msg1), Some(msg2)) = (&msg1, &msg2) {
208            let msg1 = msg1.to_lowercase();
209            let msg2 = msg2.to_lowercase();
210            msg1 == msg2
211                || both(&msg1, &msg2, |msg| {
212                    msg.contains("overflow") || msg.contains("cannot fit into")
213                })
214                || both(&msg1, &msg2, |msg| {
215                    msg.contains("divide by zero")
216                        || msg.contains("divisor of zero")
217                        || msg.contains("division by zero")
218                })
219                || both(&msg1, &msg2, |msg| {
220                    msg.contains("attempted to shift by")
221                        || msg.contains("shift with overflow")
222                        || msg.contains("shift right with overflow")
223                        || msg.contains("shift left with overflow")
224                })
225                || both(&msg1, &msg2, |msg| {
226                    // In Brillig we have constraints protecting overflows,
227                    // while in ACIR we have checked multiplication unless we know its safe.
228                    msg.contains("multiply with overflow") || msg.contains("index out of bounds")
229                })
230                || both(&msg1, &msg2, |msg| {
231                    msg.contains("add with overflow") || msg.contains("index out of bounds")
232                })
233        } else {
234            false
235        };
236
237        if equiv_msgs {
238            return true;
239        }
240
241        match (ee1, ee2) {
242            (
243                AssertionFailed(ResolvedAssertionPayload::String(c), _, _),
244                AssertionFailed(_, _, _),
245            ) if c.contains("CustomDiagnostic") => {
246                // Looks like the workaround we have for comptime failures originating from overflows and similar assertion failures.
247                true
248            }
249            (
250                AssertionFailed(ResolvedAssertionPayload::Raw(_), _, _),
251                AssertionFailed(ResolvedAssertionPayload::Raw(_), _, _),
252            ) if msg2.as_ref().is_some_and(|msg| msg.contains("overflow"))
253                && msg1.as_ref().is_some_and(|msg| {
254                    msg.len() == crate::program::CONSTRAIN_MSG_LENGTH as usize
255                }) =>
256            {
257                // This is the case where a randomly generated `assert x == const, "MSG"` in ACIR causes
258                // a preceding range constraint to be removed from the bytecode.
259                true
260            }
261            (AssertionFailed(p1, _, _), AssertionFailed(p2, _, _)) => p1 == p2,
262            (SolvingError(s1, _), SolvingError(s2, _)) => format!("{s1}") == format!("{s2}"),
263            (
264                SolvingError(OpcodeResolutionError::UnsatisfiedConstrain { .. }, _),
265                AssertionFailed(_, _, _),
266            ) => msg2.as_ref().is_some_and(|msg| {
267                msg.contains("divide by zero") || msg.contains("divisor of zero")
268            }),
269            (
270                AssertionFailed(_, _, _),
271                SolvingError(OpcodeResolutionError::UnsatisfiedConstrain { .. }, _),
272            ) => msg1.is_some_and(|msg| {
273                msg.contains("divide by zero") || msg.contains("divisor of zero")
274            }),
275            (
276                SolvingError(OpcodeResolutionError::IndexOutOfBounds { .. }, _),
277                AssertionFailed(_, _, _),
278            ) => msg2.is_some_and(|msg| msg.contains("Index out of bounds")),
279            (
280                AssertionFailed(_, _, _),
281                SolvingError(OpcodeResolutionError::IndexOutOfBounds { .. }, _),
282            ) => msg1.is_some_and(|msg| msg.contains("Index out of bounds")),
283            _ => false,
284        }
285    }
286}

304/// Compare the execution of equivalent programs, compiled in different ways.
305pub struct CompareCompiled<P> {
306    pub program: P,
307    pub abi: Abi,
308    pub input_map: InputMap,
309    pub ssa1: CompareArtifact,
310    pub ssa2: CompareArtifact,
311}
312
313impl<P> CompareCompiled<P> {
314    /// Execute the two SSAs and compare the results.
315    pub fn exec(&self) -> eyre::Result<CompareCompiledResult> {
316        let blackbox_solver = Bn254BlackBoxSolver;
317        let initial_witness = self.abi.encode(&self.input_map, None).wrap_err("abi::encode")?;
318
319        let do_exec = |program| {
320            let mut print = Vec::new();
321
322            let mut foreign_call_executor = DefaultForeignCallBuilder::default()
323                .with_mocks(false)
324                .with_output(&mut print)
325                .build();
326
327            let res = nargo::ops::execute_program(
328                program,
329                initial_witness.clone(),
330                &blackbox_solver,
331                &mut foreign_call_executor,
332            );
333
334            let print = String::from_utf8(print).expect("should be valid utf8 string");
335            (res, print)
336        };
337
338        let (res1, print1) = do_exec(&self.ssa1.artifact.program);
339        let (res2, print2) = do_exec(&self.ssa2.artifact.program);
340
341        CompareCompiledResult::new(
342            &self.abi,
343            &self.ssa1.artifact.error_types,
344            &self.ssa2.artifact.error_types,
345            (res1, print1),
346            (res2, print2),
347        )
348    }
349}

351/// Compare the execution the same program compiled in two different ways.
352pub type ComparePipelines = CompareCompiled<Program>;
353
354impl CompareCompiled<Program> {
355    /// Generate a random AST and compile it into SSA in two different ways.
356    pub fn arb(
357        u: &mut Unstructured,
358        c: Config,
359        f: impl FnOnce(
360            &mut Unstructured,
361            Program,
362        ) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
363        g: impl FnOnce(
364            &mut Unstructured,
365            Program,
366        ) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
367    ) -> arbitrary::Result<Self> {
368        let program = arb_program(u, c)?;
369        let abi = program_abi(&program);
370        logging::log_program(&program, "");
371
372        let ssa1 = CompareArtifact::from(f(u, program.clone())?);
373        let ssa2 = CompareArtifact::from(g(u, program.clone())?);
374
375        logging::log_options(&ssa1.options, "1st");
376        logging::log_options(&ssa2.options, "2nd");
377
378        let input_program = &ssa1.artifact.program;
379        let input_map = arb_inputs(u, input_program, &abi)?;
380        logging::log_abi_inputs(&abi, &input_map);
381
382        Ok(Self { program, abi, input_map, ssa1, ssa2 })
383    }
384}

387    fn programs(&self) -> Vec<&Program> {
388        vec![&self.program]
389    }

392/// Compare two equivalent variants of the same program, compiled the same way.
393pub type CompareMorph = CompareCompiled<(Program, Program)>;
394
395impl CompareMorph {
396    /// Generate a random AST, a random metamorph of it, then compile both into SSA with the same options.
397    pub fn arb(
398        u: &mut Unstructured,
399        c: Config,
400        f: impl Fn(&mut Unstructured, Program) -> arbitrary::Result<(Program, CompareOptions)>,
401        g: impl Fn(Program, &CompareOptions) -> SsaProgramArtifact,
402    ) -> arbitrary::Result<Self> {
403        let program1 = arb_program(u, c)?;
404        logging::log_program(&program1, "orig");
405
406        let (program2, options) = f(u, program1.clone())?;
407        logging::log_program(&program2, "morph");
408        logging::log_options(&options, "");
409
410        let abi = program_abi(&program1);
411
412        let ssa1 = g(program1.clone(), &options);
413        let ssa2 = g(program2.clone(), &options);
414
415        let input_program = &ssa1.program;
416        let input_map = arb_inputs(u, input_program, &abi)?;
417        logging::log_abi_inputs(&abi, &input_map);
418
419        Ok(Self {
420            program: (program1, program2),
421            abi,
422            input_map,
423            ssa1: CompareArtifact::new(ssa1, options.clone()),
424            ssa2: CompareArtifact::new(ssa2, options),
425        })
426    }
427}

573    #[test]
574    fn decodes_dynamic_string_error_type_without_panicking() {
575        // Regression test for the `orig_vs_morph` failure with seed 0xee0040bf00100000.
576        //
577        // The metamorphic rewriter turns a literal `assert(_, "PLH")` message into
578        // `if b { "PLH" } else { "PLH" }`. Being non-literal, `codegen_constrain_error`
579        // records it as `ErrorType::Dynamic` (whose inner type is nonetheless a plain
580        // `str<3>`) in the SSA error type table, with the runtime payload carrying the
581        // string bytes. Extracting the message used to `panic!("didn't expect dynamic error
582        // types")`; instead the bytes must be decoded back into the original string so the
583        // morphed program compares equal to the original.
584        let str_type = HirType::String(Box::new(HirType::Constant(Integer::U32(3))));
585        let selector = ErrorSelector::new(9230725515038505495);
586        let plh_payload = vec![
587            FieldElement::from(u32::from(b'P')),
588            FieldElement::from(u32::from(b'L')),
589            FieldElement::from(u32::from(b'H')),
590        ];
591

612    #[test]
613    fn matches_acir_dynamic_fmtstr_oob_with_brillig_string() {
614        // Regression test for the `acir_vs_brillig` failure with seed 0x6c3ad1760003227e.
615        //
616        // A dynamic read of a composite array (element size > 1) out of bounds is reported in
617        // ACIR by a synthesized format-string assertion carrying the logical index and length
618        // (`ConstrainError::Dynamic` with an `ErrorType::Dynamic(fmtstr<_, (u32,)>)`). Its raw
619        // payload is the template bytes, followed by the field count and the formatted index
620        // value. In Brillig the same access surfaces as a plain `"Index out of bounds"` string.
621        // Extracting the template message from the fmtstr payload is what lets the comparator
622        // treat the two as equivalent.
623        let template = "Index out of bounds, array has size 4, but index was {}";
624        let mut data: Vec<FieldElement> =
625            template.bytes().map(|b| FieldElement::from(u32::from(b))).collect();
626        // Format string layout: template bytes, then the field count, then the formatted values.
627        data.push(FieldElement::from(1u32));
628        data.push(FieldElement::from(869191887u32));
629
630        let fmtstr_type = HirType::FmtString(
631            Box::new(HirType::Constant(Integer::U32(template.len() as u32))),
632            Box::new(HirType::Tuple(vec![HirType::u32()])),
633        );
634        let acir_selector = ErrorSelector::new(9034042597070725729);
635        let acir_error = NargoErrorWithTypes(
636            nargo::NargoError::ExecutionError(ExecutionError::AssertionFailed(
637                ResolvedAssertionPayload::Raw(RawAssertionPayload {
638                    selector: acir_selector,
639                    data,
640                }),
641                Vec::new(),
642                None,
643            )),
644            BTreeMap::from_iter([(acir_selector, ErrorType::Dynamic(fmtstr_type))]),
645        );
646
647        let brillig_selector = ErrorSelector::new(16431471497789672479);
648        let brillig_error = NargoErrorWithTypes(
649            nargo::NargoError::ExecutionError(ExecutionError::AssertionFailed(
650                ResolvedAssertionPayload::Raw(RawAssertionPayload {
651                    selector: brillig_selector,
652                    data: vec![],
653                }),
654                Vec::new(),
655                Some(BrilligFunctionId::new(0)),
656            )),
657            BTreeMap::from_iter([(
658                brillig_selector,
659                ErrorType::String("Index out of bounds".to_string()),
660            )]),
661        );
662
663        assert!(NargoErrorWithTypes::equivalent(&acir_error, &brillig_error));
664    }
