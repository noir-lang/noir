35/// These are the error types in the `SsaProgramArtifact`, which are not the same as the ones in the ABI,
36/// but they can provide extra information when comparing errors.
37type SsaErrorTypes = BTreeMap<acir::circuit::ErrorSelector, ErrorType>;
38
39/// The execution result is the value returned from the circuit and any output from `println`.
40pub(crate) type ExecResult = (Result<WitnessStack<FieldElement>, NargoError<FieldElement>>, String);
41
42#[derive(Debug)]
43pub struct NargoErrorWithTypes(NargoError<FieldElement>, SsaErrorTypes);
44
45impl NargoErrorWithTypes {
46    /// Copy of `NargoError::user_defined_failure_message` accepting `SsaErrorTypes` instead of ABI errors.
47    fn user_defined_failure_message(&self) -> Option<String> {
48        let unwrap_payload = |payload: &ResolvedAssertionPayload<FieldElement>| {
49            match payload {
50                ResolvedAssertionPayload::String(message) => Some(message.clone()),
51                ResolvedAssertionPayload::Raw(raw) => {
52                    if let Some(ssa_type) = self.1.get(&raw.selector) {
53                        match ssa_type {
54                            ErrorType::String(message) => Some(message.clone()),
55                            ErrorType::Dynamic(_hir_type) => {
56                                // This would be the case if we have a format string that needs to be filled with the raw payload
57                                // decoded as ABI type. The code generator shouldn't produce this kind. It shouldn't be too difficult
58                                // to map the type, but the mapper in `crate::abi` doesn't handle format strings at the moment.
59                                panic!("didn't expect dynamic error types")
60                            }
61                        }
62                    } else {
63                        // `codegen_constrain_error` only records the SSA `ErrorType` when the
64                        // assert message type is *not* a string (see `is_string_type` branch in
65                        // `compiler/noirc_evaluator/src/ssa/ssa_gen/mod.rs`). When a non-literal
66                        // string expression is used as an assert message, the runtime payload
67                        // therefore carries the bytes of the string with a selector that has no
68                        // entry in the SSA error type table. Recover the message from the bytes.
69                        decode_raw_string_payload(&raw.data)
70                    }
71                }
72            }
73        };
74
75        match &self.0 {
76            NargoError::ExecutionError(error) => match error {
77                ExecutionError::AssertionFailed(payload, _, _) => unwrap_payload(payload),
78                ExecutionError::SolvingError(error, _) => match error {
79                    OpcodeResolutionError::BlackBoxFunctionFailed(_, reason) => {
80                        Some(reason.clone())
81                    }
82                    OpcodeResolutionError::BrilligFunctionFailed {
83                        payload: Some(payload), ..
84                    } => unwrap_payload(payload),
85                    _ => None,
86                },
87            },
88            NargoError::ForeignCallError(error) => Some(error.to_string()),
89            NargoError::CompilationError => None,
90        }
91    }
92}

94/// Try to interpret the raw bytes of an assertion payload as a UTF-8 string,
95/// matching the encoding produced by `ConstrainError::Dynamic { is_string_type: true, .. }`.
96///
97/// Returns `None` if the data is empty, contains a field element wider than a byte,
98/// or the bytes are not valid UTF-8.
99fn decode_raw_string_payload(data: &[FieldElement]) -> Option<String> {
100    if data.is_empty() {
101        return None;
102    }
103    let bytes: Vec<u8> = data
104        .iter()
105        .map(|field| {
106            let be = field.to_be_bytes();
107            let (low, high) = be.split_last().expect("FieldElement::to_be_bytes is non-empty");
108            high.iter().all(|b| *b == 0).then_some(*low)
109        })
110        .collect::<Option<Vec<_>>>()?;
111    String::from_utf8(bytes).ok()
112}

114/// The result of the execution of compiled programs, decoded by their ABI.
115pub type CompareCompiledResult = CompareResult<InputValue, NargoErrorWithTypes>;
116
117impl CompareCompiledResult {
118    pub fn new(
119        abi: &Abi,
120        ets1: &SsaErrorTypes,
121        ets2: &SsaErrorTypes,
122        (res1, print1): ExecResult,
123        (res2, print2): ExecResult,
124    ) -> eyre::Result<Self> {
125        let decode = |ws: WitnessStack<FieldElement>| -> eyre::Result<Option<InputValue>> {
126            let wm = &ws.peek().expect("there should be a main witness").witness;
127            let (_, r) = abi.decode(wm).wrap_err("abi::decode")?;
128            Ok(r)
129        };
130
131        let failed = |e, ets: &SsaErrorTypes, p: String| FailedOutput {
132            error: NargoErrorWithTypes(e, ets.clone()),
133            print_output: p,
134        };
135
136        let passed = |ws, p| decode(ws).map(|r| PassedOutput { return_value: r, print_output: p });
137
138        match (res1, res2) {
139            (Err(e1), Err(e2)) => {
140                Ok(CompareResult::BothFailed(failed(e1, ets1, print1), failed(e2, ets2, print2)))
141            }
142            (Err(e1), Ok(ws2)) => {
143                Ok(CompareResult::LeftFailed(failed(e1, ets1, print1), passed(ws2, print2)?))
144            }
145            (Ok(ws1), Err(e2)) => {
146                Ok(CompareResult::RightFailed(passed(ws1, print1)?, failed(e2, ets2, print2)))
147            }
148            (Ok(ws1), Ok(ws2)) => {
149                Ok(CompareResult::BothPassed(passed(ws1, print1)?, passed(ws2, print2)?))
150            }
151        }
152    }
153}

155impl Comparable for NargoErrorWithTypes {
156    fn equivalent(e1: &Self, e2: &Self) -> bool {
157        use ExecutionError::*;
158
159        // For now consider non-execution errors as failures we need to investigate.
160        let NargoError::ExecutionError(ee1) = &e1.0 else {
161            return false;
162        };
163        let NargoError::ExecutionError(ee2) = &e2.0 else {
164            return false;
165        };
166
167        // We have a notion of treating errors as equivalents as long as the side effects
168        // of the failed program are otherwise the same. For this reason we compare the
169        // prints in `return_value_or_err`. Here we have the option to tweak which errors
170        // we consider equivalents, but that's really just to stay on the conservative
171        // side and give us a chance to inspect new kinds of test failures.
172
173        fn both<F: Fn(&str) -> bool>(s1: &str, s2: &str, f: F) -> bool {
174            f(s1) && f(s2)
175        }
176
177        let msg1 = e1.user_defined_failure_message();
178        let msg2 = e2.user_defined_failure_message();
179        let equiv_msgs = if let (Some(msg1), Some(msg2)) = (&msg1, &msg2) {
180            let msg1 = msg1.to_lowercase();
181            let msg2 = msg2.to_lowercase();
182            msg1 == msg2
183                || both(&msg1, &msg2, |msg| {
184                    msg.contains("overflow") || msg.contains("cannot fit into")
185                })
186                || both(&msg1, &msg2, |msg| {
187                    msg.contains("divide by zero")
188                        || msg.contains("divisor of zero")
189                        || msg.contains("division by zero")
190                })
191                || both(&msg1, &msg2, |msg| {
192                    msg.contains("attempted to shift by")
193                        || msg.contains("shift with overflow")
194                        || msg.contains("shift right with overflow")
195                        || msg.contains("shift left with overflow")
196                })
197                || both(&msg1, &msg2, |msg| {
198                    // In Brillig we have constraints protecting overflows,
199                    // while in ACIR we have checked multiplication unless we know its safe.
200                    msg.contains("multiply with overflow") || msg.contains("index out of bounds")
201                })
202                || both(&msg1, &msg2, |msg| {
203                    msg.contains("add with overflow") || msg.contains("index out of bounds")
204                })
205        } else {
206            false
207        };
208
209        if equiv_msgs {
210            return true;
211        }
212
213        match (ee1, ee2) {
214            (
215                AssertionFailed(ResolvedAssertionPayload::String(c), _, _),
216                AssertionFailed(_, _, _),
217            ) if c.contains("CustomDiagnostic") => {
218                // Looks like the workaround we have for comptime failures originating from overflows and similar assertion failures.
219                true
220            }
221            (
222                AssertionFailed(ResolvedAssertionPayload::Raw(_), _, _),
223                AssertionFailed(ResolvedAssertionPayload::Raw(_), _, _),
224            ) if msg2.as_ref().is_some_and(|msg| msg.contains("overflow"))
225                && msg1.as_ref().is_some_and(|msg| {
226                    msg.len() == crate::program::CONSTRAIN_MSG_LENGTH as usize
227                }) =>
228            {
229                // This is the case where a randomly generated `assert x == const, "MSG"` in ACIR causes
230                // a preceding range constraint to be removed from the bytecode.
231                true
232            }
233            (AssertionFailed(p1, _, _), AssertionFailed(p2, _, _)) => p1 == p2,
234            (SolvingError(s1, _), SolvingError(s2, _)) => format!("{s1}") == format!("{s2}"),
235            (
236                SolvingError(OpcodeResolutionError::UnsatisfiedConstrain { .. }, _),
237                AssertionFailed(_, _, _),
238            ) => msg2.as_ref().is_some_and(|msg| {
239                msg.contains("divide by zero") || msg.contains("divisor of zero")
240            }),
241            (
242                AssertionFailed(_, _, _),
243                SolvingError(OpcodeResolutionError::UnsatisfiedConstrain { .. }, _),
244            ) => msg1.is_some_and(|msg| {
245                msg.contains("divide by zero") || msg.contains("divisor of zero")
246            }),
247            (
248                SolvingError(OpcodeResolutionError::IndexOutOfBounds { .. }, _),
249                AssertionFailed(_, _, _),
250            ) => msg2.is_some_and(|msg| msg.contains("Index out of bounds")),
251            (
252                AssertionFailed(_, _, _),
253                SolvingError(OpcodeResolutionError::IndexOutOfBounds { .. }, _),
254            ) => msg1.is_some_and(|msg| msg.contains("Index out of bounds")),
255            _ => false,
256        }
257    }
258}

276/// Compare the execution of equivalent programs, compiled in different ways.
277pub struct CompareCompiled<P> {
278    pub program: P,
279    pub abi: Abi,
280    pub input_map: InputMap,
281    pub ssa1: CompareArtifact,
282    pub ssa2: CompareArtifact,
283}
284
285impl<P> CompareCompiled<P> {
286    /// Execute the two SSAs and compare the results.
287    pub fn exec(&self) -> eyre::Result<CompareCompiledResult> {
288        let blackbox_solver = Bn254BlackBoxSolver;
289        let initial_witness = self.abi.encode(&self.input_map, None).wrap_err("abi::encode")?;
290
291        let do_exec = |program| {
292            let mut print = Vec::new();
293
294            let mut foreign_call_executor = DefaultForeignCallBuilder::default()
295                .with_mocks(false)
296                .with_output(&mut print)
297                .build();
298
299            let res = nargo::ops::execute_program(
300                program,
301                initial_witness.clone(),
302                &blackbox_solver,
303                &mut foreign_call_executor,
304            );
305
306            let print = String::from_utf8(print).expect("should be valid utf8 string");
307            (res, print)
308        };
309
310        let (res1, print1) = do_exec(&self.ssa1.artifact.program);
311        let (res2, print2) = do_exec(&self.ssa2.artifact.program);
312
313        CompareCompiledResult::new(
314            &self.abi,
315            &self.ssa1.artifact.error_types,
316            &self.ssa2.artifact.error_types,
317            (res1, print1),
318            (res2, print2),
319        )
320    }
321}

323/// Compare the execution the same program compiled in two different ways.
324pub type ComparePipelines = CompareCompiled<Program>;
325
326impl CompareCompiled<Program> {
327    /// Generate a random AST and compile it into SSA in two different ways.
328    pub fn arb(
329        u: &mut Unstructured,
330        c: Config,
331        f: impl FnOnce(
332            &mut Unstructured,
333            Program,
334        ) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
335        g: impl FnOnce(
336            &mut Unstructured,
337            Program,
338        ) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
339    ) -> arbitrary::Result<Self> {
340        let program = arb_program(u, c)?;
341        let abi = program_abi(&program);
342        logging::log_program(&program, "");
343
344        let ssa1 = CompareArtifact::from(f(u, program.clone())?);
345        let ssa2 = CompareArtifact::from(g(u, program.clone())?);
346
347        logging::log_options(&ssa1.options, "1st");
348        logging::log_options(&ssa2.options, "2nd");
349
350        let input_program = &ssa1.artifact.program;
351        let input_map = arb_inputs(u, input_program, &abi)?;
352        logging::log_abi_inputs(&abi, &input_map);
353
354        Ok(Self { program, abi, input_map, ssa1, ssa2 })
355    }
356}

358impl HasPrograms for ComparePipelines {
359    fn programs(&self) -> Vec<&Program> {
360        vec![&self.program]
361    }
362}

364/// Compare two equivalent variants of the same program, compiled the same way.
365pub type CompareMorph = CompareCompiled<(Program, Program)>;
366
367impl CompareMorph {
368    /// Generate a random AST, a random metamorph of it, then compile both into SSA with the same options.
369    pub fn arb(
370        u: &mut Unstructured,
371        c: Config,
372        f: impl Fn(&mut Unstructured, Program) -> arbitrary::Result<(Program, CompareOptions)>,
373        g: impl Fn(Program, &CompareOptions) -> SsaProgramArtifact,
374    ) -> arbitrary::Result<Self> {
375        let program1 = arb_program(u, c)?;
376        logging::log_program(&program1, "orig");
377
378        let (program2, options) = f(u, program1.clone())?;
379        logging::log_program(&program2, "morph");
380        logging::log_options(&options, "");
381
382        let abi = program_abi(&program1);
383
384        let ssa1 = g(program1.clone(), &options);
385        let ssa2 = g(program2.clone(), &options);
386
387        let input_program = &ssa1.program;
388        let input_map = arb_inputs(u, input_program, &abi)?;
389        logging::log_abi_inputs(&abi, &input_map);
390
391        Ok(Self {
392            program: (program1, program2),
393            abi,
394            input_map,
395            ssa1: CompareArtifact::new(ssa1, options.clone()),
396            ssa2: CompareArtifact::new(ssa2, options),
397        })
398    }
399}
400
401impl HasPrograms for CompareMorph {
402    fn programs(&self) -> Vec<&Program> {
403        vec![&self.program.0, &self.program.1]
404    }

454    #[test]
455    fn matches_static_string_with_inline_string_payload() {
456        // Regression test for the `orig_vs_morph` failure with seed 0xbd897d3000100000.
457        //
458        // The metamorphic rule `any_inevitable` wraps a string literal `"KJQ"` used as
459        // an `assert(_, "KJQ")` message into `if a { "KJQ" } else { "KJQ" }`. The original
460        // program lowers the literal to `ConstrainError::StaticString` (selector mapped to
461        // `ErrorType::String("KJQ")` in the SSA error type table, runtime payload has empty
462        // data). The morphed program lowers the `if`-expression to
463        // `ConstrainError::Dynamic(_, is_string_type=true, _)` whose runtime payload carries
464        // the bytes of the string directly and whose selector is *not* recorded in the SSA
465        // error type table (the `is_string_type` branch in `codegen_constrain_error` skips
466        // `record_error_type`). Both encodings represent the same user-visible message, so
467        // the comparator must treat them as equivalent.
468        let static_string_error = NargoErrorWithTypes(
469            nargo::NargoError::ExecutionError(ExecutionError::AssertionFailed(
470                ResolvedAssertionPayload::Raw(RawAssertionPayload {
471                    selector: ErrorSelector::new(17370598246489328139),
472                    data: vec![],
473                }),
474                Vec::new(),
475                Some(BrilligFunctionId(0)),
476            )),
477            BTreeMap::from_iter([(
478                ErrorSelector::new(17370598246489328139),
479                ErrorType::String("KJQ".to_string()),
480            )]),
481        );
482        let inline_string_error = NargoErrorWithTypes(
483            nargo::NargoError::ExecutionError(ExecutionError::AssertionFailed(
484                ResolvedAssertionPayload::Raw(RawAssertionPayload {
485                    selector: ErrorSelector::new(9230725515038505495),
486                    data: vec![
487                        FieldElement::from(u32::from(b'K')),
488                        FieldElement::from(u32::from(b'J')),
489                        FieldElement::from(u32::from(b'Q')),
490                    ],
491                }),
492                Vec::new(),
493                Some(BrilligFunctionId(0)),
494            )),
495            BTreeMap::new(),
496        );
497
498        assert!(NargoErrorWithTypes::equivalent(&static_string_error, &inline_string_error));
499    }
