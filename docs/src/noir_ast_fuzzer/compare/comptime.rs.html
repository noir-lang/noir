36/// Prepare a code snippet.
37/// (copied from nargo_cli/tests/common.rs)
38fn prepare_snippet(source: String) -> (Context<'static, 'static>, CrateId) {
39    let root = Path::new("");
40    let file_name = Path::new("main.nr");
41    let mut file_manager = file_manager_with_stdlib(root);
42    file_manager.add_file_with_source(file_name, source).expect(
43        "Adding source buffer to file manager should never fail when file manager is empty",
44    );
45    let parsed_files = parse_all(&file_manager);
46
47    let mut context = Context::new(file_manager, parsed_files);
48    let root_crate_id = prepare_crate(&mut context, file_name);
49
50    (context, root_crate_id)
51}

53/// Compile the main function in a code snippet.
54///
55/// Use `force_brillig` to test it as an unconstrained function without having to change the code.
56/// This is useful for methods that use the `runtime::is_unconstrained()` method to change their behavior.
57/// (copied from nargo_cli/tests/common.rs)
58fn prepare_and_compile_snippet<W: std::io::Write + 'static>(
59    source: String,
60    force_brillig: bool,
61    output: W,
62) -> (CompilationResult<CompiledProgram>, W) {
63    let output = Rc::new(RefCell::new(output));
64    let (mut context, root_crate_id) = prepare_snippet(source);
65    context.set_comptime_printing(output.clone());
66    let options = CompileOptions {
67        force_brillig,
68        silence_warnings: true,
69        skip_underconstrained_check: true,
70        skip_brillig_constraints_check: true,
71        ..Default::default()
72    };
73    let res = compile_main(&mut context, root_crate_id, &options, None);
74    drop(context);
75    let output = Rc::into_inner(output).expect("context is gone").into_inner();
76    (res, output)
77}

79/// Compare the execution of a Noir program in pure comptime (via interpreter)
80/// vs normal SSA execution.
81pub struct CompareComptime {
82    pub program: Program,
83    pub abi: Abi,
84    pub source: String,
85    pub ssa: CompareArtifact,
86    pub force_brillig: bool,
87}
88
89impl CompareComptime {
90    /// Execute the Noir code with the comptime interpreter
91    /// and prepare SSA returning the result literal,
92    /// then compare SSA execution results.
93    pub fn exec_direct(
94        &self,
95        f_comptime: impl FnOnce(Program) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
96    ) -> eyre::Result<CompareCompiledResult> {
97        let initial_witness = self.input_witness()?;
98        let (res2, print2) =
99            Self::exec_bytecode(&self.ssa.artifact.program, initial_witness.clone());
100
101        // Include the print part of stdlib for the elaborator to be able to use the print oracle
102        let import_print = r#"
103        #[oracle(print)]
104        unconstrained fn print_oracle<T>(with_newline: bool, input: T) {}
105
106        unconstrained fn print_unconstrained<T>(with_newline: bool, input: T) {
107            print_oracle(with_newline, input);
108        }
109
110        pub fn println<T>(input: T) {
111            unsafe {
112                print_unconstrained(true, input);
113            }
114        }
115
116        pub fn print<T>(input: T) {
117            unsafe {
118                print_unconstrained(false, input);
119            }
120        }
121        "#;
122
123        // Add comptime modifier for main
124        let source = format!("comptime {}{}", self.source, import_print);
125        let output = Rc::new(RefCell::new(Vec::new()));
126
127        // Take the printed output.
128        let printed = |output: Rc<RefCell<Vec<u8>>>| {
129            let output = Rc::into_inner(output).expect("context is gone").into_inner();
130            String::from_utf8(output).expect("not UTF-8")
131        };
132
133        let comptime_expr = match interpret(source.as_str(), output.clone()) {
134            Ok(expr) => expr,
135            Err(e) => {
136                let assertion_diagnostic = match &e {
137                    ElaboratorError::Compile(errors) => errors
138                        .iter()
139                        .map(CustomDiagnostic::from)
140                        .find(Self::is_assertion_diagnostic),
141
142                    ElaboratorError::Interpret(e) => {
143                        let e = CompilationError::from(e.clone());
144                        let e = CustomDiagnostic::from(&e);
145                        Self::is_assertion_diagnostic(&e).then_some(e)
146                    }
147                    _ => None,
148                };
149
150                if let Some(e) = assertion_diagnostic {
151                    return self.comptime_failure(&e, printed(output), (res2, print2));
152                } else {
153                    panic!(
154                        "elaborator error while interpreting generated comptime code: {e:?}\n{}",
155                        self.source
156                    );
157                }
158            }
159        };
160
161        let program_comptime = program_wrap_expression(comptime_expr);
162        let comptime_ssa = CompareArtifact::from(f_comptime(program_comptime)?);
163
164        let (res1, _) =
165            Self::exec_bytecode(&comptime_ssa.artifact.program, initial_witness.clone());
166
167        CompareCompiledResult::new(
168            &self.abi,
169            &Default::default(),
170            &self.ssa.artifact.error_types,
171            (res1, printed(output)),
172            (res2, print2),
173        )
174    }
175
176    /// Execute the Noir code (via nargo) and the SSA, then compare the results.
177    pub fn exec(&self) -> eyre::Result<CompareCompiledResult> {
178        let initial_witness = self.input_witness()?;
179
180        // Execute the 2nd (Brillig) program.
181        let (res2, print2) =
182            Self::exec_bytecode(&self.ssa.artifact.program, initial_witness.clone());
183
184        // Try to compile the 1st (comptime) version from string.
185        let (program1, output1) = match prepare_and_compile_snippet(
186            self.source.clone(),
187            self.force_brillig,
188            Vec::new(),
189        ) {
190            (Ok((program, _)), output) => (program, output),
191            (Err(errors), output) => {
192                // If the comptime code failed to compile, it could be because it executed the code
193                // and encountered an overflow, which would be a runtime error in Brillig.
194                let assertion_diagnostic = errors.iter().find(|e| Self::is_assertion_diagnostic(e));
195
196                if let Some(e) = assertion_diagnostic {
197                    return self.comptime_failure(e, Self::decode_print(output), (res2, print2));
198                } else {
199                    panic!("failed to compile program:\n{errors:?}\n{}", self.source);
200                }
201            }
202        };
203
204        // Capture any println that happened during the compilation, which in these tests should be the whole program.
205        let comptime_print = Self::decode_print(output1);
206        // Execute the 1st (comptime) program, capturing the rest of the output.
207        let (res1, print1) = Self::exec_bytecode(&program1.program, initial_witness);
208
209        CompareCompiledResult::new(
210            &self.abi,
211            &Default::default(), // We have a fully compiled program at this point, no access to the SSA error types, just ABI error types.
212            &self.ssa.artifact.error_types,
213            (res1, comptime_print + &print1),
214            (res2, print2),
215        )
216    }

218    /// Generate a random comptime-viable AST, reverse it into
219    /// Noir code and also compile it into SSA.
220    pub fn arb(
221        u: &mut Unstructured,
222        c: Config,
223        f: impl FnOnce(Program) -> arbitrary::Result<(SsaProgramArtifact, CompareOptions)>,
224    ) -> arbitrary::Result<Self> {
225        let force_brillig = c.force_brillig;
226        let program = arb_program_comptime(u, c)?;
227        let abi = program_abi(&program);
228        logging::log_program(&program, "");
229
230        let ssa = CompareArtifact::from(f(program.clone())?);
231        logging::log_options(&ssa.options, "compiled");
232
233        let source = format!("{}", DisplayAstAsNoirComptime(&program));
234        logging::log_comptime(&source, "");
235
236        Ok(Self { program, abi, source, ssa, force_brillig })
237    }

239    /// Execute the program bytecode, returning the execution result along with the captured print output.
240    fn exec_bytecode(
241        program: &acir::circuit::Program<FieldElement>,
242        initial_witness: WitnessMap<FieldElement>,
243    ) -> ExecResult {
244        let blackbox_solver = Bn254BlackBoxSolver;
245        let mut output = Vec::new();
246
247        let mut foreign_call_executor =
248            DefaultForeignCallBuilder::default().with_mocks(false).with_output(&mut output).build();
249
250        let res = nargo::ops::execute_program(
251            program,
252            initial_witness,
253            &blackbox_solver,
254            &mut foreign_call_executor,
255        );
256        let print = Self::decode_print(output);
257
258        (res, print)
259    }
260
261    /// Decode the print output into a string.
262    fn decode_print(output: Vec<u8>) -> String {
263        String::from_utf8(output).expect("should be valid utf8 string")
264    }
265
266    /// Comptime test programs have no inputs.
267    fn input_witness(&self) -> eyre::Result<WitnessMap<FieldElement>> {
268        self.abi.encode(&BTreeMap::new(), None).wrap_err("abi::encode")
269    }
270
271    /// Check if a comptime error is due to some kind of arithmetic or constraint failure.
272    fn is_assertion_diagnostic(e: &CustomDiagnostic) -> bool {
273        let msg = e.message.to_lowercase();
274        e.secondaries.iter().any(|s| s.message == "Assertion failed")
275            || msg.contains("overflow")
276            || msg.contains("cannot fit into") // covers signed overflows
277            || msg.contains("divide by zero")
278            || msg.contains("division by zero")
279            || msg.contains("the remainder with a divisor of zero")
280            || msg.contains("out of bounds")
281    }
282
283    /// Fabricate a result from a comptime `CustomDiagnostic` on the 1st side,
284    /// and a full `ExecResult` on the 2nd side.
285    fn comptime_failure(
286        &self,
287        e: &CustomDiagnostic,
288        print1: String,
289        (res2, print2): ExecResult,
290    ) -> eyre::Result<CompareCompiledResult> {
291        let msg = format!("{e:?}");
292        let err = ExecutionError::AssertionFailed(
293            acvm::pwg::ResolvedAssertionPayload::String(msg),
294            vec![],
295            None,
296        );
297        let res1 = Err(NargoError::ExecutionError(err));
298        CompareCompiledResult::new(
299            &self.abi,
300            &Default::default(), // We failed to compile the program, so no error types.
301            &self.ssa.artifact.error_types,
302            (res1, print1),
303            (res2, print2),
304        )
305    }
306}
