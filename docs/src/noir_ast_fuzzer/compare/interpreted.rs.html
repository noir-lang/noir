24/// The state of the SSA after a particular pass in the pipeline.
25pub struct ComparePass {
26    /// The overall position of this pass in the pipeline.
27    ///
28    /// The Initial SSA is considered step 0.
29    pub step: usize,
30    /// The message (without the counter) of the pass.
31    pub msg: String,
32    /// The state of the SSA after the pass.
33    pub ssa: Ssa,
34}

36type InterpretResult = Result<Vec<Value>, ssa::interpreter::errors::InterpreterError>;
37
38/// The result of the SSA interpreter execution.
39pub type CompareInterpretedResult =
40    CompareResult<Vec<Value>, ssa::interpreter::errors::InterpreterError>;

42/// Inputs for comparing the interpretation of two SSA states of an arbitrary program.
43pub struct CompareInterpreted {
44    pub program: Program,
45    pub abi: Abi,
46    /// ABI inputs, which we map to SSA values as part of the execution.
47    /// We could generate random input for the SSA directly, but it would
48    /// make it more difficult to use it with `nargo` if we find a failure.
49    pub input_map: InputMap,
50
51    /// Inputs for the `main` function in the SSA, mapped from the ABI input.
52    pub ssa_args: Vec<Value>,
53
54    /// Options that influence the pipeline, common to both passes.
55    pub options: CompareOptions,
56    pub ssa1: ComparePass,
57    pub ssa2: ComparePass,
58}
59
60impl CompareInterpreted {
61    /// 1. Generate an arbitrary AST
62    /// 2. Stop the compilation at two arbitrary SSA passes
63    /// 3. Generate input for the main function of the SSA
64    pub fn arb(
65        u: &mut Unstructured,
66        c: Config,
67        f: impl FnOnce(
68            &mut Unstructured,
69            Program,
70        ) -> arbitrary::Result<(CompareOptions, ComparePass, ComparePass)>,
71    ) -> arbitrary::Result<Self> {
72        let program = arb_program(u, c)?;
73        let abi = program_abi(&program);
74        logging::log_program(&program, "");
75
76        let (options, ssa1, ssa2) = f(u, program.clone())?;
77
78        logging::log_options(&options, "");
79        logging::log_ssa(&ssa1.ssa, &format!("after step {} - {}", ssa1.step, ssa1.msg));
80        logging::log_ssa(&ssa2.ssa, &format!("after step {} - {}", ssa2.step, ssa2.msg));
81
82        let input_map = arb_inputs_from_ssa(u, &ssa1.ssa, &abi)?;
83        logging::log_abi_inputs(&abi, &input_map);
84
85        let ssa_args = input_values_to_ssa(&abi, &input_map);
86        logging::log_ssa_inputs(&ssa_args);
87
88        Ok(Self { program, abi, input_map, ssa_args, options, ssa1, ssa2 })
89    }
90
91    pub fn exec(&self) -> eyre::Result<CompareInterpretedResult> {
92        // Interpret an SSA with a fresh copy of the input values.
93        let interpret = |ssa: &Ssa| {
94            let mut output = Vec::new();
95            let res = ssa.interpret_with_options(
96                Value::snapshot_args(&self.ssa_args),
97                InterpreterOptions::default(),
98                &mut output,
99            );
100            (res, output)
101        };
102
103        Ok(CompareInterpretedResult::new(interpret(&self.ssa1.ssa), interpret(&self.ssa2.ssa)))
104    }
105}

107impl CompareInterpretedResult {
108    pub fn new(
109        (res1, print1): (InterpretResult, Vec<u8>),
110        (res2, print2): (InterpretResult, Vec<u8>),
111    ) -> Self {
112        let printed = |p| String::from_utf8(p).expect("from_utf8 of print");
113        let failed = |e, p| FailedOutput { error: e, print_output: printed(p) };
114        let passed = |ret, p| PassedOutput { return_value: Some(ret), print_output: printed(p) };
115
116        match (res1, res2) {
117            (Ok(r1), Ok(r2)) => Self::BothPassed(passed(r1, print1), passed(r2, print2)),
118            (Ok(r1), Err(e2)) => Self::RightFailed(passed(r1, print1), failed(e2, print2)),
119            (Err(e1), Ok(r2)) => Self::LeftFailed(failed(e1, print1), passed(r2, print2)),
120            (Err(e1), Err(e2)) => Self::BothFailed(failed(e1, print1), failed(e2, print2)),
121        }
122    }
123}

125impl Comparable for ssa::interpreter::errors::InterpreterError {
126    fn equivalent(e1: &Self, e2: &Self) -> bool {
127        use ssa::interpreter::errors::InternalError;
128        use ssa::interpreter::errors::InterpreterError::*;
129
130        match (e1, e2) {
131            (
132                Internal(InternalError::ConstantDoesNotFitInType { constant: c1, typ: t1 }),
133                Internal(InternalError::ConstantDoesNotFitInType { constant: c2, typ: t2 }),
134            ) => {
135                // The interpreter represents values in types where the result of some casts cannot be represented, while the ACIR and
136                // Brillig runtime can fit them into Fields, and defer validation later. We could promote this error to a non-internal one,
137                // but the fact remains that the interpreter would fail earlier than ACIR or Brillig.
138                // To deal with this we ignore these errors as long as both passes fail the same way.
139                c1 == c2 && t1 == t2
140            }
141            (
142                Internal(InternalError::ConstantDoesNotFitInType { constant, .. }),
143                RangeCheckFailed { value, .. },
144            )
145            | (
146                RangeCheckFailed { value, .. },
147                Internal(InternalError::ConstantDoesNotFitInType { constant, .. }),
148            ) => {
149                // The value should be a `NumericValue` display format, which is `<type> <value>`.
150                let value = value.split_once(' ').map(|(_, value)| value).unwrap_or(value);
151                value == constant.to_string()
152            }
153            (Internal(_), _) | (_, Internal(_)) => {
154                // We should not get, or ignore, internal errors.
155                // They mean the interpreter got something unexpected that we need to fix.
156                false
157            }
158            (Overflow { instruction: i1, .. }, Overflow { instruction: i2, .. }) => {
159                // Overflows can occur or instructions with different IDs, but in a parentheses it contains the values that caused it.
160                fn details(s: &str) -> Option<&str> {
161                    let start = s.find("(")?;
162                    let end = s.find(")")?;
163                    (start < end).then(|| &s[start..=end])
164                }
165                fn details_or_sanitize(s: &str) -> String {
166                    details(s).map(|s| s.to_string()).unwrap_or_else(|| sanitize_ssa(s))
167                }
168                details_or_sanitize(i1) == details_or_sanitize(i2)
169            }
170
171            // We expand checked operations on signed types into multiple instructions during the `expand_signed_checks`
172            // pass. This results in the error changing from an `Overflow` into a different error type so we match
173            // on the attached error message.
174            (Overflow { operator, .. }, ConstrainEqFailed { msg: Some(msg), .. }) => match operator
175            {
176                BinaryOp::Add { unchecked: false } => msg == "attempt to add with overflow",
177                BinaryOp::Sub { unchecked: false } => msg == "attempt to subtract with overflow",
178                BinaryOp::Mul { unchecked: false } => msg == "attempt to multiply with overflow",
179                BinaryOp::Shl | BinaryOp::Shr => {
180                    msg == "attempt to bit-shift with overflow"
181                        || msg == "attempt to shift right with overflow"
182                        || msg == "attempt to shift left with overflow"
183                }
184                _ => false,
185            },
186            (
187                Overflow { operator: BinaryOp::Mul { unchecked: false }, .. },
188                RangeCheckFailed { msg: Some(msg), .. },
189            ) => msg == "attempt to multiply with overflow",
190
191            (
192                ConstrainEqFailed { msg: msg1, .. } | ConstrainNeFailed { msg: msg1, .. },
193                ConstrainEqFailed { msg: msg2, .. } | ConstrainNeFailed { msg: msg2, .. },
194            ) => {
195                // The `lhs` and `rhs` might change during passes, making direct comparison difficult:
196                // * the sides might be flipped: `u1 0 == u1 1` vs `u1 1 == u1 0`
197                // * the condition might be flipped: `u1 1 != u1 0` vs `Field 0 == Field 0`
198                // * types could change:
199                //      * `Field 313339671284855045676773137498590239475 != Field 0` vs `u128 313339671284855045676773137498590239475 != u128 0`
200                //      * `i64 -1615928006 != i64 -5568658583620095790` vs `u64 18446744072093623610 != u64 12878085490089455826`
201                // So instead of reasoning about the `lhs` and `rhs` formats, let's just compare the message so we know it's the same constraint:
202                msg1 == msg2
203            }
204            (RangeCheckFailed { msg: Some(msg1), .. }, ConstrainEqFailed { msg: msg2, .. }) => {
205                // The removal of unreachable instructions evaluates constant binary operations and can replace
206                // e.g. a `mul` followed by a `range_check` with a `constrain true == false, "attempt to multiple with overflow"`
207                msg2.as_ref().is_some_and(|msg| msg == msg1)
208            }
209            (DivisionByZero { .. }, ConstrainEqFailed { msg, .. }) => {
210                msg.as_ref().is_some_and(|msg| {
211                    msg == "attempt to divide by zero" || msg.contains("divisor of zero")
212                })
213            }
214            (DivisionByZero { .. }, DivisionByZero { .. }) => {
215                // Signed math in ACIR is expanded to unsigned math. We may have two different `DivisionByZero` errors due to differing types.
216                true
217            }
218            (PoppedFromEmptyVector { .. }, ConstrainEqFailed { msg, .. }) => {
219                // The removal of unreachable instructions can replace popping from an empty vector with an always-fail constraint.
220                msg.as_ref().is_some_and(|msg| msg == "Index out of bounds")
221            }
222            (IndexOutOfBounds { .. }, ConstrainEqFailed { msg, .. }) => {
223                msg.as_ref().is_some_and(|msg| msg.contains("Index out of bounds"))
224            }
225            (e1, e2) => {
226                // The format strings contain SSA instructions,
227                // where the only difference might be the value ID.
228                let s1 = format!("{e1}");
229                let s2 = format!("{e2}");
230                sanitize_ssa(&s1) == sanitize_ssa(&s2)
231            }
232        }
233    }
234}

236impl Comparable for Value {
237    fn equivalent(a: &Self, b: &Self) -> bool {
238        match (a, b) {
239            (Value::ArrayOrVector(a), Value::ArrayOrVector(b)) => {
240                // Ignore the RC
241                a.element_types == b.element_types
242                    && Comparable::equivalent(&a.elements, &b.elements)
243                    && a.is_vector == b.is_vector
244            }
245            (Value::Reference(a), Value::Reference(b)) => {
246                // Ignore the original ID
247                a.element_type == b.element_type && Comparable::equivalent(&a.element, &b.element)
248            }
249            (a, b) => a == b,
250        }
251    }
252}

254/// Convert the ABI encoded inputs to what the SSA interpreter expects.
255pub fn input_values_to_ssa(abi: &Abi, input_map: &InputMap) -> Vec<Value> {
256    let mut inputs = Vec::new();
257    for param in &abi.parameters {
258        let input = &input_map
259            .get(&param.name)
260            .unwrap_or_else(|| panic!("parameter not found in input: {}", param.name));
261
262        inputs.extend(input_value_to_ssa(&param.typ, input));
263    }
264    inputs
265}

267/// Convert one ABI encoded input to what the SSA interpreter expects.
268///
269/// Tuple types and structs are flattened.
270pub fn input_value_to_ssa(typ: &AbiType, input: &InputValue) -> Vec<Value> {
271    let mut values = Vec::new();
272    append_input_value_to_ssa(typ, input, &mut values);
273    values
274}
275
276fn append_input_value_to_ssa(typ: &AbiType, input: &InputValue, values: &mut Vec<Value>) {
277    use ssa::interpreter::value::{ArrayValue, NumericValue, Value};
278    use ssa::ir::types::Type;
279    let array_value = |elements: Vec<Value>, types: Vec<Type>| {
280        Value::ArrayOrVector(ArrayValue {
281            elements: Shared::new(elements),
282            rc: Shared::new(1),
283            element_types: Arc::new(types),
284            is_vector: false,
285        })
286    };
287    match input {
288        InputValue::Field(f) => {
289            let num_typ = match typ {
290                AbiType::Field => NumericType::NativeField,
291                AbiType::Boolean => NumericType::Unsigned { bit_size: 1 },
292                AbiType::Integer { sign: Sign::Signed, width } => {
293                    NumericType::Signed { bit_size: *width }
294                }
295                AbiType::Integer { sign: Sign::Unsigned, width } => {
296                    NumericType::Unsigned { bit_size: *width }
297                }
298                other => panic!("unexpected ABY type for Field input: {other:?}"),
299            };
300            let num_val = NumericValue::from_constant(*f, num_typ).expect("cannot create constant");
301            values.push(Value::Numeric(num_val));
302        }
303        InputValue::String(s) => values
304            .push(array_value(vecmap(s.as_bytes(), |b| Value::u8(*b)), vec![Type::unsigned(8)])),
305        InputValue::Vec(input_values) => match typ {
306            AbiType::Array { length, typ } => {
307                assert_eq!(*length as usize, input_values.len(), "array length != input length");
308                let mut elements = Vec::with_capacity(*length as usize);
309                for input in input_values {
310                    append_input_value_to_ssa(typ, input, &mut elements);
311                }
312                values.push(array_value(elements, input_type_to_ssa(typ)));
313            }
314            AbiType::Tuple { fields } => {
315                assert_eq!(fields.len(), input_values.len(), "tuple size != input length");
316
317                // Tuples are flattened
318                for (typ, input) in fields.iter().zip(input_values) {
319                    append_input_value_to_ssa(typ, input, values);
320                }
321            }
322            other => {
323                panic!("unexpected ABI type for vector input: {other:?}")
324            }
325        },
326        InputValue::Struct(field_values) => match typ {
327            AbiType::Struct { path: _, fields } => {
328                assert_eq!(fields.len(), field_values.len(), "struct size != input length");
329
330                // Structs are flattened
331                for (name, typ) in fields {
332                    let input = &field_values[name];
333                    append_input_value_to_ssa(typ, input, values);
334                }
335            }
336            other => {
337                panic!("unexpected ABI type for map input: {other:?}")
338            }
339        },
340    }
341}

343/// Convert an ABI type into SSA.
344fn input_type_to_ssa(typ: &AbiType) -> Vec<ssa::ir::types::Type> {
345    let mut types = Vec::new();
346    append_input_type_to_ssa(typ, &mut types);
347    types
348}
349
350fn append_input_type_to_ssa(typ: &AbiType, types: &mut Vec<ssa::ir::types::Type>) {
351    use ssa::ir::types::Type;
352    match typ {
353        AbiType::Field => types.push(Type::field()),
354        AbiType::Array { length, typ } => {
355            types.push(Type::Array(Arc::new(input_type_to_ssa(typ)), SemanticLength(*length)));
356        }
357        AbiType::Integer { sign: Sign::Signed, width } => types.push(Type::signed(*width)),
358        AbiType::Integer { sign: Sign::Unsigned, width } => types.push(Type::unsigned(*width)),
359        AbiType::Boolean => types.push(Type::bool()),
360        AbiType::Struct { path: _, fields } => {
361            // Structs are flattened
362            for (_, typ) in fields {
363                append_input_type_to_ssa(typ, types);
364            }
365        }
366        AbiType::Tuple { fields } => {
367            // Tuples are flattened
368            for typ in fields {
369                append_input_type_to_ssa(typ, types);
370            }
371        }
372        AbiType::String { length } => {
373            types.push(Type::Array(Arc::new(vec![Type::unsigned(8)]), SemanticLength(*length)));
374        }
375    }
376}

378/// Remove identifiers from the SSA, so we can compare the structure without
379/// worrying about trivial differences like changing IDs of the same variable
380/// between one pass to the next.
381fn sanitize_ssa(ssa: &str) -> String {
382    static RE: OnceLock<Regex> = OnceLock::new();
383    // Capture function ID, value IDs, global IDs.
384    let re = RE.get_or_init(|| Regex::new(r#"(f|b|v|g)\d+"#).expect("ID regex failed"));
385    re.replace_all(ssa, "${1}_").into_owned()
386}
