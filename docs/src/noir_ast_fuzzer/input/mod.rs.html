21/// Generate arbitrary inputs for a compiled program according to the ABI.
22pub fn arb_inputs(
23    u: &mut Unstructured,
24    program: &acir::circuit::Program<FieldElement>,
25    abi: &Abi,
26) -> arbitrary::Result<InputMap> {
27    // Reuse the proptest strategy in `noir_fuzzer` to generate random inputs.
28    let dictionary = build_dictionary_from_program(program);
29    let dictionary = BTreeSet::from_iter(dictionary);
30    let strategy = arb_input_map(abi, &dictionary);
31    arb_value_tree(u, strategy)
32}

34/// Generate arbitrary inputs for an SSA of a program, according to the ABI.
35pub(crate) fn arb_inputs_from_ssa(
36    u: &mut Unstructured,
37    ssa: &Ssa,
38    abi: &Abi,

42    let strategy = arb_input_map(abi, &dictionary);
43    arb_value_tree(u, strategy)

46/// Generate a seed and use it to generate an arbitrary value using the proptest strategy.
47fn arb_value_tree(
48    u: &mut Unstructured,
49    strategy: BoxedStrategy<InputMap>,
50) -> arbitrary::Result<InputMap> {
51    // The strategy needs a runner, although all it really uses is the RNG from it.
52    let seed: [u8; 16] = u.arbitrary()?;
53    let rng = TestRng::from_seed(RngAlgorithm::XorShift, &seed);
54    let mut runner = TestRunner::new_with_rng(Config::default(), rng);
55    let tree = strategy.new_tree(&mut runner).map_err(|_| arbitrary::Error::IncorrectFormat)?;
56    Ok(tree.current())
57}

59/// Given the [Abi] description of a Noir program, generate random [InputValue]s for each circuit parameter.
60///
61/// Use the `dictionary` to draw values from for numeric types.
62fn arb_input_map(abi: &Abi, dictionary: &BTreeSet<FieldElement>) -> BoxedStrategy<InputMap> {
63    let values: Vec<_> = abi
64        .parameters
65        .iter()
66        .map(|param| (Just(param.name.clone()), arb_value_from_abi_type(&param.typ, dictionary)))
67        .collect();
68
69    values
70        .prop_map(|values| {
71            let input_map: InputMap = values.into_iter().collect();
72            input_map
73        })
74        .boxed()
75}

77/// Create a strategy for generating random values for an [AbiType].
78///
79/// Uses the `dictionary` for unsigned integer types.
80fn arb_value_from_abi_type(
81    abi_type: &AbiType,
82    dictionary: &BTreeSet<FieldElement>,
83) -> SBoxedStrategy<InputValue> {
84    match abi_type {
85        AbiType::Field => proptest::collection::vec(any::<u8>(), 32)
86            .prop_map(|bytes| InputValue::Field(FieldElement::from_be_bytes_reduce(&bytes)))
87            .sboxed(),
88        AbiType::Integer { width, sign } if sign == &Sign::Unsigned => {
89            // We've restricted the type system to only allow u64s as the maximum integer type.
90            let width = (*width).min(64);
91            UintStrategy::new(width as usize, dictionary)
92                .prop_map(|uint| InputValue::Field(uint.into()))
93                .sboxed()
94        }
95        AbiType::Integer { width, .. } => {
96            let width = (*width).min(64);
97            // Based on `FieldElement::to_i128`:
98            // * negative integers are represented by the range [p + i128::MIN, p)
99            // * positive integers are represented by the range [0, i128::MAX)
100            let shift = 2i128.pow(width);
101            IntStrategy::new(width as usize)
102                .prop_map(move |mut int| {
103                    if int < 0 {
104                        int += shift;
105                    }
106                    InputValue::Field(int.into())
107                })
108                .sboxed()
109        }
110        AbiType::Boolean => {
111            any::<bool>().prop_map(|val| InputValue::Field(FieldElement::from(val))).sboxed()
112        }
113        AbiType::String { length } => {
114            // Strings only allow ASCII characters as each character must be able to be represented by a single byte.
115            let string_regex = format!("[[:ascii:]]{{{length}}}");
116            proptest::string::string_regex(&string_regex)
117                .expect("parsing of regex should always succeed")
118                .prop_map(InputValue::String)
119                .sboxed()
120        }
121        AbiType::Array { length, typ } => {
122            let length = *length as usize;
123            let elements = proptest::collection::vec(
124                arb_value_from_abi_type(typ, dictionary),
125                length..=length,
126            );
127
128            elements.prop_map(InputValue::Vec).sboxed()
129        }
130        AbiType::Struct { fields, .. } => {
131            let fields: Vec<SBoxedStrategy<(String, InputValue)>> = fields
132                .iter()
133                .map(|(name, typ)| {
134                    (Just(name.clone()), arb_value_from_abi_type(typ, dictionary)).sboxed()
135                })
136                .collect();
137
138            fields
139                .prop_map(|fields| {
140                    let fields: BTreeMap<_, _> = fields.into_iter().collect();
141                    InputValue::Struct(fields)
142                })
143                .sboxed()
144        }
145        AbiType::Tuple { fields } => {
146            let fields: Vec<_> =
147                fields.iter().map(|typ| arb_value_from_abi_type(typ, dictionary)).collect();
148            fields.prop_map(InputValue::Vec).sboxed()
149        }
150    }
151}
