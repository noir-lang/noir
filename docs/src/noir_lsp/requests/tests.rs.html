14pub(crate) fn on_tests_request(
15    state: &mut LspState,
16    params: NargoTestsParams,
17) -> impl Future<Output = Result<NargoTestsResult, ResponseError>> + use<> {
18    future::ready(on_tests_request_inner(state, params))
19}
20
21fn on_tests_request_inner(
22    state: &mut LspState,
23    _params: NargoTestsParams,
24) -> Result<NargoTestsResult, ResponseError> {
25    let root_path = state.root_path.as_deref().ok_or_else(|| {
26        ResponseError::new(ErrorCode::REQUEST_FAILED, "Could not find project root")
27    })?;
28
29    let toml_path = match find_package_manifest(root_path, root_path) {
30        Ok(toml_path) => toml_path,
31        Err(err) => {
32            // If we cannot find a manifest, we log a warning but return no code lenses
33            // We can reconsider this when we can build a file without the need for a Nargo.toml file to resolve deps
34            let _ = state.client.log_message(LogMessageParams {
35                typ: MessageType::WARNING,
36                message: err.to_string(),
37            });
38            return Ok(None);
39        }
40    };
41
42    let workspace = resolve_workspace_from_toml(
43        &toml_path,
44        PackageSelection::All,
45        Some(NOIR_ARTIFACT_VERSION_STRING.to_string()),
46    )
47    .map_err(|err| {
48        // If we found a manifest, but the workspace is invalid, we raise an error about it
49        ResponseError::new(ErrorCode::REQUEST_FAILED, err)
50    })?;
51
52    let mut workspace_file_manager = workspace.new_file_manager();
53    insert_all_files_for_workspace_into_file_manager(
54        state,
55        &workspace,
56        &mut workspace_file_manager,
57    );
58    let parsed_files = parse_diff(&workspace_file_manager, state);
59
60    let package_tests: Vec<_> = workspace
61        .into_iter()
62        .filter_map(|package| {
63            let (mut context, crate_id) =
64                crate::prepare_package(&workspace_file_manager, &parsed_files, package);
65            // We ignore the warnings and errors produced by compilation for producing tests
66            // because we can still get the test functions even if compilation fails
67            let _ = check_crate(&mut context, crate_id, &Default::default());
68
69            // We don't add test headings for a package if it contains no `#[test]` functions
70            get_package_tests_in_crate(&context, &crate_id, &package.name)
71                .map(|tests| NargoPackageTests { package: package.name.to_string(), tests })
72        })
73        .collect();
74
75    if package_tests.is_empty() { Ok(None) } else { Ok(Some(package_tests)) }
76}
