10pub(crate) fn on_references_request(
11    state: &mut LspState,
12    params: ReferenceParams,
13) -> impl Future<Output = Result<Option<Vec<Location>>, ResponseError>> + use<> {
14    let include_declaration = params.context.include_declaration;
15    let result = process_request(state, params.text_document_position, |args| {
16        find_all_references_in_workspace(
17            args.location,
18            args.interner,
19            args.package_cache,
20            args.files,
21            include_declaration,
22            true,
23        )
24    });
25    future::ready(result)
26}

42    async fn check_references_succeeds(
43        directory: &str,
44        name: &str,
45        declaration_index: usize,
46        include_declaration: bool,
47    ) {
48        let (mut state, noir_text_document) = test_utils::init_lsp_server(directory).await;
49
50        // First we find out all of the occurrences of `name` in the main.nr file.
51        // Note that this only works if that name doesn't show up in other places where we don't
52        // expect a rename, but we craft our tests to avoid that.
53        let ranges = search_in_file(noir_text_document.path(), name);
54
55        // Test getting references works on any instance of the symbol.
56        for target_range in &ranges {
57            let target_position = target_range.start;
58
59            let params = ReferenceParams {
60                text_document_position: TextDocumentPositionParams {
61                    text_document: TextDocumentIdentifier { uri: noir_text_document.clone() },
62                    position: target_position,
63                },
64                work_done_progress_params: WorkDoneProgressParams { work_done_token: None },
65                partial_result_params: PartialResultParams { partial_result_token: None },
66                context: ReferenceContext { include_declaration },
67            };
68
69            let locations = on_references_request(&mut state, params)
70                .await
71                .expect("Could not execute on_references_request")
72                .unwrap();
73
74            let mut references_ranges: Vec<_> =
75                locations.iter().map(|location| location.range).collect();
76            references_ranges.sort_by_key(|range| range.start.line);
77
78            if include_declaration {
79                assert_eq!(ranges, references_ranges);
80            } else {
81                let mut ranges_without_declaration = ranges.clone();
82                ranges_without_declaration.remove(declaration_index);
83                assert_eq!(ranges_without_declaration, references_ranges);
84            }
85        }
86    }

103    async fn test_on_references_request_works_across_workspace_packages() {
104        let (mut state, noir_text_document) = test_utils::init_lsp_server("workspace").await;
105
106        // noir_text_document is always `src/main.nr` in the workspace directory, so let's go to the workspace dir
107        let noir_text_document = noir_text_document.to_file_path().unwrap();
108        let workspace_dir = noir_text_document.parent().unwrap().parent().unwrap();
109
110        // Let's check that we can find references to `function_one` by doing that in the package "one"
111        // and getting results in the package "two" too.
112        let one_lib = Url::from_file_path(workspace_dir.join("one/src/lib.nr")).unwrap();
113        let two_lib = Url::from_file_path(workspace_dir.join("two/src/lib.nr")).unwrap();
114
115        // We call this to open the document, so that the entire workspace is analyzed
116        let workspace = workspace_from_document_uri(one_lib.clone()).unwrap();
117
118        notifications::process_workspace(&mut state, &workspace).unwrap();
119
120        let params = ReferenceParams {
121            text_document_position: TextDocumentPositionParams {
122                text_document: TextDocumentIdentifier { uri: one_lib.clone() },
123                position: Position { line: 0, character: 7 },
124            },
125            work_done_progress_params: WorkDoneProgressParams { work_done_token: None },
126            partial_result_params: PartialResultParams { partial_result_token: None },
127            context: ReferenceContext { include_declaration: true },
128        };
129
130        let mut locations = on_references_request(&mut state, params)
131            .await
132            .expect("Could not execute on_references_request")
133            .unwrap();
134
135        // The definition, a use in "two", and a call in "two"
136        assert_eq!(locations.len(), 3);
137
138        locations.sort_by_cached_key(|location| {
139            (location.uri.to_file_path().unwrap(), location.range.start.line)
140        });
141
142        assert_eq!(locations[0].uri, one_lib);
143        assert_eq!(
144            locations[0].range,
145            Range {
146                start: Position { line: 0, character: 7 },
147                end: Position { line: 0, character: 19 },
148            }
149        );
150
151        assert_eq!(locations[1].uri, two_lib);
152        assert_eq!(
153            locations[1].range,
154            Range {
155                start: Position { line: 0, character: 9 },
156                end: Position { line: 0, character: 21 },
157            }
158        );
159
160        assert_eq!(locations[2].uri, two_lib);
161        assert_eq!(
162            locations[2].range,
163            Range {
164                start: Position { line: 3, character: 4 },
165                end: Position { line: 3, character: 16 },
166            }
167        );
168    }

172        let src = "
173        #[foo]
174        struct Fo>|<o {}
175
176        trait Trait {}
177
178        fn main() {}
179
180        comptime fn foo(s: StructDefinition) -> Quoted {

182            quote {
183                impl Trait for $t {}
184            }
185        }
186        ";
187
188        let (mut state, noir_text_document) = test_utils::init_lsp_server("document_symbol").await;
189
190        let (line, column, src) = get_cursor_line_and_column(src);
191
192        let _ = on_did_open_text_document(
193            &mut state,
194            DidOpenTextDocumentParams {
195                text_document: TextDocumentItem {
196                    uri: noir_text_document.clone(),
197                    language_id: "noir".to_string(),
198                    version: 0,
199                    text: src.to_string(),
200                },
201            },
202        );
203
204        let result = on_references_request(
205            &mut state,
206            ReferenceParams {
207                text_document_position: TextDocumentPositionParams {
208                    text_document: TextDocumentIdentifier { uri: noir_text_document },
209                    position: Position { line: line as u32, character: column as u32 },
210                },
211                work_done_progress_params: WorkDoneProgressParams { work_done_token: None },
212                partial_result_params: PartialResultParams { partial_result_token: None },
213                context: ReferenceContext { include_declaration: true },
214            },
215        )
216        .await;
217        let locations = result.unwrap().unwrap();
218        assert_eq!(locations.len(), 1);
