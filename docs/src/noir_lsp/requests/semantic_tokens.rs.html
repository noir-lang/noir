1//! Handles LSP semantic tokens requests.
2//!
3//! Semantic tokens in Noir are provided in two contexts:
4//! - links inside doc comments. For example,a doc comment that has `[println]` in it will
5//!   be colorized as a function reference (only if such function actually exists).
6//! - code blocks inside doc comments. If these are Noir or Rust code blocks, a Lexer
7//!   will be used to colorize keywords and such.

41pub(crate) fn on_semantic_tokens_full_request(
42    state: &mut LspState,
43    params: SemanticTokensParams,
44) -> impl Future<Output = Result<Option<SemanticTokensResult>, ResponseError>> + use<> {
45    let text_document_position_params = TextDocumentPositionParams {
46        text_document: params.text_document.clone(),
47        position: Position { line: 0, character: 0 },
48    };
49
50    let result = process_request(state, text_document_position_params, |args| {
51        let file_id = args.location.file;
52        let file = args.files.get_file(file_id).unwrap();
53        let source = file.source();
54        let (parsed_module, _errors) = noirc_frontend::parse_program(source, file_id);
55
56        let mut collector = SemanticTokenCollector::new(source, file_id, &args);
57        let tokens = collector.collect(&parsed_module);
58        Some(SemanticTokensResult::Tokens(SemanticTokens { result_id: None, data: tokens }))
59    });
60    future::ready(result)
61}

63struct SemanticTokenCollector<'args> {
64    source: &'args str,
65    args: &'args ProcessRequestCallbackArgs<'args>,
66    file_id: FileId,
67    link_finder: LinkFinder,
68    tokens: Vec<SemanticToken>,
69    previous_line: u32,
70    previous_char: u32,
71    token_types: HashMap<SemanticTokenType, usize>,
72}

74#[derive(Copy, Clone, Debug, PartialEq, Eq)]
75enum CodeBlock {
76    None,
77    Noir,
78    Other,
79}

81impl<'args> SemanticTokenCollector<'args> {
82    fn new(
83        source: &'args str,
84        file_id: FileId,
85        args: &'args ProcessRequestCallbackArgs<'args>,
86    ) -> Self {
87        let link_finder = LinkFinder::default();
88        let tokens = Vec::new();
89        let token_types = semantic_token_types_map();
90        SemanticTokenCollector {
91            source,
92            args,
93            file_id,
94            link_finder,
95            tokens,
96            previous_line: 0,
97            previous_char: 0,
98            token_types,
99        }
100    }
101
102    fn collect(&mut self, parsed_module: &noirc_frontend::ParsedModule) -> Vec<SemanticToken> {
103        // Find the module the current file belongs to
104        let krate = self.args.crate_id;
105        let def_map = &self.args.def_maps[&krate];
106        let local_id = if let Some((module_index, _)) = def_map
107            .modules()
108            .iter()
109            .find(|(_, module_data)| module_data.location.file == self.file_id)
110        {
111            LocalModuleId::new(module_index)
112        } else {
113            def_map.root()
114        };
115        let module_id = ModuleId { krate, local_id };
116
117        // Process doc comments on the module itself
118        self.process_reference_id(ReferenceId::Module(module_id));
119
120        parsed_module.accept(self);
121        std::mem::take(&mut self.tokens)
122    }
123
124    /// Checks doc comments on the given ReferenceId. Semantic tokens are produced for any links found,
125    /// so that they can be colorized in the editor.
126    fn process_reference_id(&mut self, id: ReferenceId) {
127        let Some(doc_comments) = self.args.interner.doc_comments(id) else {
128            return;
129        };
130        let Some((current_module_id, current_type)) = current_module_and_type(id, self.args) else {
131            return;
132        };
133
134        let mut code_block = CodeBlock::None;
135
136        self.link_finder.reset();
137        for located_comment in doc_comments {
138            let location = located_comment.location();
139            if location.file != self.file_id {
140                // A module's comments might happen inline in the same file or in a different file.
141                // We should not process comments that are not in the current file.
142                continue;
143            }
144
145            let contents = located_comment.contents.trim();
146            let mut fence = false;
147
148            match code_block {
149                CodeBlock::None => {
150                    if contents == "```" || contents == "```noir" || contents == "```rust" {
151                        code_block = CodeBlock::Noir;
152                        fence = true;
153                    } else if contents.starts_with("```") {
154                        code_block = CodeBlock::Other;
155                        fence = true;
156                    }
157                }
158                CodeBlock::Noir | CodeBlock::Other => {
159                    if contents == "```" {
160                        code_block = CodeBlock::None;
161                        fence = true;
162                    }
163                }
164            }
165
166            let location = located_comment.location();
167            let Some(lsp_location) = to_lsp_location(self.args.files, location.file, location.span)
168            else {
169                continue;
170            };
171            let start_line = lsp_location.range.start.line;
172            let start_char = lsp_location.range.start.character;
173
174            // Read comments from source based on location: the comments in `located_comment` might
175            // have been slightly adjusted.
176            let comments =
177                &self.source[location.span.start() as usize..location.span.end() as usize];
178
179            if code_block == CodeBlock::Noir && !fence {
180                self.colorize_code_block_line(lsp_location, comments);
181            }
182
183            let links = self.link_finder.find_links(
184                comments,
185                current_module_id,
186                current_type,
187                self.args.interner,
188                self.args.def_maps,
189                self.args.crate_graph,
190            );
191            for link in links {
192                let Some(target) = link.target else {
193                    continue;
194                };
195
196                let Some(token_type) = self.link_target_token_type(target) else {
197                    continue;
198                };
199                let token_type = self.token_types[&token_type] as u32;
200                let delta_line = start_line + link.line as u32;
201                let delta_start =
202                    if link.line == 0 { start_char + link.start as u32 } else { link.start as u32 };
203                let length = (link.end - link.start) as u32;
204                let token = SemanticToken {
205                    delta_line,
206                    delta_start,
207                    length,
208                    token_type,
209                    token_modifiers_bitset: 0,
210                };
211                self.push_token(token);
212            }
213        }
214    }

216    fn colorize_code_block_line(&mut self, lsp_location: lsp_types::Location, line: &str) {
217        // The code block line will start with either "///", "//!", or optionally "*".
218        // We remove it, but then we'll need to add an offset when calculating the character position.
219        let (line, offset) = if let Some(line) = line.strip_prefix("///") {
220            (line, 3)
221        } else if let Some(line) = line.strip_prefix("//!") {
222            (line, 3)
223        } else if let Some(line) = line.strip_prefix("*") {
224            (line, 1)
225        } else {
226            (line, 0)
227        };
228        let lexer = Lexer::new_with_dummy_file(line);
229        let mut tokens = Vec::new();
230
231        for token in lexer {
232            let Ok(token) = token else {
233                // If lexing fails, give up
234                return;
235            };
236            if matches!(token.token(), Token::EOF) {
237                break;
238            }
239
240            tokens.push(token);
241        }
242
243        for (index, token) in tokens.iter().enumerate() {
244            let previous_token = if index == 0 { None } else { Some(&tokens[index - 1]) };
245            let next_token = tokens.get(index + 1);
246            let next_next_token = tokens.get(index + 2);
247            self.colorize_token(
248                token,
249                previous_token,
250                next_token,
251                next_next_token,
252                &lsp_location,
253                offset,
254            );
255        }
256    }

258    fn colorize_token(
259        &mut self,
260        token: &LocatedToken,
261        previous_token: Option<&LocatedToken>,
262        next_token: Option<&LocatedToken>,
263        next_next_token: Option<&LocatedToken>,
264        location: &lsp_types::Location,
265        offset: usize,
266    ) {
267        let span = token.span();
268        let token = token.token();
269        let semantic_token_type = match token {
270            Token::Int(..) => SemanticTokenType::NUMBER,
271            Token::Bool(_) => SemanticTokenType::KEYWORD,
272            Token::Str(_) | Token::RawStr(_, _) | Token::FmtStr(..) => SemanticTokenType::STRING,
273            Token::Keyword(_) => SemanticTokenType::KEYWORD,
274            Token::LineComment(..) | Token::BlockComment(..) => SemanticTokenType::COMMENT,
275            Token::Quote(tokens) => {
276                // Colorize "quote"
277                let semantic_token_type = self.token_types[&SemanticTokenType::KEYWORD] as u32;
278                let sematic_token = SemanticToken {
279                    delta_line: location.range.start.line,
280                    delta_start: location.range.start.character + span.start() + offset as u32,
281                    length: 5,
282                    token_type: semantic_token_type,
283                    token_modifiers_bitset: 0,
284                };
285                self.push_token(sematic_token);
286
287                for (index, token) in tokens.0.iter().enumerate() {
288                    let previous_token = if index == 0 { None } else { Some(&tokens.0[index - 1]) };
289                    let next_token = tokens.0.get(index + 1);
290                    let next_next_token = tokens.0.get(index + 2);
291                    self.colorize_token(
292                        token,
293                        previous_token,
294                        next_token,
295                        next_next_token,
296                        location,
297                        offset,
298                    );
299                }
300                return;
301            }
302            Token::Ident(name) => {
303                if name == "self" {
304                    SemanticTokenType::KEYWORD
305                } else if name.chars().next().is_some_and(|char| char.is_ascii_uppercase())
306                    || PrimitiveType::lookup_by_name(name).is_some()
307                {
308                    // Heuristic: if the name starts with an uppercase letter, or it denotes a primitive type,
309                    // colorize it as a struct
310                    SemanticTokenType::STRUCT
311                } else if next_token.is_some_and(|token| matches!(token.token(), Token::LeftParen))
312                    || previous_token
313                        .is_some_and(|token| matches!(token.token(), Token::Keyword(Keyword::Fn)))
314                {
315                    // Heuristic colorize "foo" in "foo(" and "fn foo" as a function
316                    SemanticTokenType::FUNCTION
317                } else if next_token
318                    .is_some_and(|token| matches!(token.token(), Token::DoubleColon))
319                {
320                    // Heuristic: colorize "foo" in "foo::" as a module.
321                    // However, if it's "foo::<", colorize it as either a struct or a function depending
322                    // on whether it starts with uppercase or not.
323                    if next_next_token.is_some_and(|token| matches!(token.token(), Token::Less)) {
324                        if name.chars().next().is_some_and(|char| char.is_ascii_uppercase()) {
325                            SemanticTokenType::STRUCT
326                        } else {
327                            SemanticTokenType::FUNCTION
328                        }
329                    } else {
330                        SemanticTokenType::NAMESPACE
331                    }
332                } else if previous_token.is_some_and(|token| {
333                    matches!(
334                        token.token(),
335                        Token::Keyword(Keyword::Struct)
336                            | Token::Keyword(Keyword::Enum)
337                            | Token::Keyword(Keyword::Impl)
338                            | Token::Keyword(Keyword::Trait)
339                            | Token::Keyword(Keyword::Type)
340                    )
341                }) {
342                    // Heuristic: colorize "foo" in "struct foo", "enum foo", etc., as a struct
343                    SemanticTokenType::STRUCT
344                } else {
345                    SemanticTokenType::VARIABLE
346                }
347            }
348            Token::Less
349            | Token::LessEqual
350            | Token::Greater
351            | Token::GreaterEqual
352            | Token::Equal
353            | Token::NotEqual
354            | Token::Plus
355            | Token::Minus
356            | Token::Star
357            | Token::Slash
358            | Token::Percent
359            | Token::Ampersand
360            | Token::Caret
361            | Token::ShiftLeft
362            | Token::ShiftRight
363            | Token::LeftParen
364            | Token::RightParen
365            | Token::LeftBrace
366            | Token::RightBrace
367            | Token::LeftBracket
368            | Token::RightBracket
369            | Token::Pipe
370            | Token::Assign
371            | Token::Arrow
372            | Token::FatArrow
373            | Token::LogicalAnd
374            | Token::Comma
375            | Token::AttributeStart { .. }
376            | Token::Semicolon => SemanticTokenType::OPERATOR,
377            Token::QuotedType(_) => SemanticTokenType::STRUCT,
378            Token::InternedExpr(..)
379            | Token::InternedStatement(..)
380            | Token::InternedLValue(..)
381            | Token::InternedUnresolvedTypeData(..)
382            | Token::InternedPattern(..)
383            | Token::InternedCrate(..)
384            | Token::Dot
385            | Token::DoubleDot
386            | Token::DoubleDotEqual
387            | Token::Pound
388            | Token::Colon
389            | Token::DoubleColon
390            | Token::Backslash
391            | Token::Bang
392            | Token::DollarSign
393            | Token::At
394            | Token::DeprecatedVectorStart
395            | Token::EOF
396            | Token::Whitespace(_)
397            | Token::UnquoteMarker(_)
398            | Token::Invalid(_) => return,
399        };
400        let semantic_token_type = self.token_types[&semantic_token_type] as u32;
401        let sematic_token = SemanticToken {
402            delta_line: location.range.start.line,
403            delta_start: location.range.start.character + span.start() + offset as u32,
404            length: span.end() - span.start(),
405            token_type: semantic_token_type,
406            token_modifiers_bitset: 0,
407        };
408        self.push_token(sematic_token);
409    }

411    fn link_target_token_type(&self, target: LinkTarget) -> Option<SemanticTokenType> {
412        let token_type = match target {
413            LinkTarget::TopLevelItem(module_def_id) => match module_def_id {
414                ModuleDefId::ModuleId(_) => SemanticTokenType::NAMESPACE,
415                ModuleDefId::FunctionId(func_id) => {
416                    let func_meta = self.args.interner.function_meta(&func_id);
417                    if func_meta.self_type.is_some() {
418                        SemanticTokenType::METHOD
419                    } else {
420                        SemanticTokenType::FUNCTION
421                    }
422                }
423                ModuleDefId::TypeId(_) => SemanticTokenType::STRUCT,
424                ModuleDefId::TypeAliasId(_) => SemanticTokenType::STRUCT,
425                ModuleDefId::TraitId(_) => SemanticTokenType::INTERFACE,
426                ModuleDefId::TraitAssociatedTypeId(_) => return None,
427                ModuleDefId::GlobalId(_) => SemanticTokenType::VARIABLE,
428            },
429            LinkTarget::Method(..) => SemanticTokenType::METHOD,
430            LinkTarget::StructMember(..) => SemanticTokenType::PROPERTY,
431            LinkTarget::PrimitiveType(_) => return None,
432            LinkTarget::PrimitiveTypeFunction(..) => SemanticTokenType::FUNCTION,
433        };
434        Some(token_type)
435    }
436
437    fn push_token(&mut self, mut token: SemanticToken) {
438        let previous_line = std::mem::replace(&mut self.previous_line, token.delta_line);
439        let previous_char = std::mem::replace(&mut self.previous_char, token.delta_start);
440
441        // See https://microsoft.github.io/language-server-protocol/specifications/lsp/3.17/specification/#textDocument_semanticTokens
442        // for an explanation of delta_line and delta_start.
443        let same_line = token.delta_line == previous_line;
444        token.delta_line -= previous_line;
445        if same_line {
446            token.delta_start -= previous_char;
447        }
448
449        self.tokens.push(token);
450    }
451}

453// Visit every AST node that can have doc comments on it. Links are then searched in these.
454impl Visitor for SemanticTokenCollector<'_> {
455    fn visit_parsed_submodule(&mut self, module: &ParsedSubModule, _: Span) -> bool {
456        let name_location = module.name.location();
457        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
458            self.process_reference_id(reference);
459        };
460
461        true
462    }
463
464    fn visit_noir_function(&mut self, function: &NoirFunction, _span: Span) -> bool {
465        let name_location = function.name_ident().location();
466        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
467            self.process_reference_id(reference);
468        };
469
470        false
471    }
472
473    fn visit_noir_struct(&mut self, noir_struct: &NoirStruct, _: Span) -> bool {
474        let name_location = noir_struct.name.location();
475        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
476            self.process_reference_id(reference);
477        };
478
479        for field in noir_struct.fields.iter() {
480            let field_name_location = field.item.name.location();
481            if let Some(reference) = self.args.interner.reference_at_location(field_name_location) {
482                self.process_reference_id(reference);
483            };
484        }
485
486        false
487    }
488
489    fn visit_noir_enum(&mut self, noir_enum: &NoirEnumeration, _: Span) -> bool {
490        let name_location = noir_enum.name.location();
491        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
492            self.process_reference_id(reference);
493        };
494
495        for variant in noir_enum.variants.iter() {
496            let variant_name_location = variant.item.name.location();
497            if let Some(reference) = self.args.interner.reference_at_location(variant_name_location)
498            {
499                self.process_reference_id(reference);
500            };
501        }
502
503        false
504    }
505
506    fn visit_noir_trait(&mut self, noir_trait: &NoirTrait, _: Span) -> bool {
507        let name_location = noir_trait.name.location();
508        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
509            self.process_reference_id(reference);
510        };
511
512        for item in noir_trait.items.iter() {
513            if let TraitItem::Function { name, .. } = &item.item {
514                let func_name_location = name.location();
515                if let Some(reference) =
516                    self.args.interner.reference_at_location(func_name_location)
517                {
518                    self.process_reference_id(reference);
519                };
520            }
521        }
522
523        true
524    }
525
526    fn visit_global(&mut self, let_statement: &LetStatement, _: Span) -> bool {
527        let name_location = let_statement.pattern.location();
528        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
529            self.process_reference_id(reference);
530        };
531        false
532    }
533
534    fn visit_noir_type_alias(&mut self, type_alias: &TypeAlias, _: Span) -> bool {
535        let name_location = type_alias.name.location();
536        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
537            self.process_reference_id(reference);
538        };
539        false
540    }
541}
