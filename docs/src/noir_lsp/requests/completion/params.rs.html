1//! If the cursor is at the end of a function parameter name, suggest parameter names (and their types)
2//! that exists in the same module, impl or trait.
3use std::collections::HashSet;
4
5use noirc_frontend::{
6    ParsedModule,
7    ast::{NoirFunction, NoirTrait, Pattern, TraitItem, TypeImpl, UnresolvedTypeData},
8    parser::ItemKind,
9};
10
11use crate::requests::completion::{NodeFinder, name_matches, variable_completion_item};
12
13impl NodeFinder<'_> {
14    pub(super) fn try_complete_function_param_in_parsed_module(
15        &mut self,
16        parsed_module: &ParsedModule,
17    ) -> bool {
18        let functions = parsed_module.items.iter().filter_map(|item| {
19            if let ItemKind::Function(function) = &item.kind { Some(function) } else { None }
20        });
21
22        let function_and_name =
23            find_function_and_parameter_name_at_byte_index(functions.clone(), self.byte_index);
24        let Some((function, name)) = function_and_name else {
25            return false;
26        };
27
28        let names_to_exclude = names_to_exclude(function, name);
29
30        self.suggest_function_parameters(functions, name, names_to_exclude);
31
32        true
33    }
34
35    pub(super) fn try_complete_function_param_in_type_impl(
36        &mut self,
37        type_impl: &TypeImpl,
38    ) -> bool {
39        let functions =
40            type_impl.methods.iter().map(|(documented_method, _)| &documented_method.item);
41
42        let function_and_name =
43            find_function_and_parameter_name_at_byte_index(functions.clone(), self.byte_index);
44        let Some((function, name)) = function_and_name else {
45            return false;
46        };
47
48        let names_to_exclude = names_to_exclude(function, name);
49
50        self.suggest_function_parameters(functions, name, names_to_exclude);
51
52        true
53    }
54
55    pub(super) fn try_complete_function_param_in_trait(&mut self, trait_: &NoirTrait) -> bool {
56        // Since NoirTrait doesn't hold `NoirFunction`s we have to repeat a bit the code here.
57        let parameters_and_name = trait_.items.iter().find_map(|documented_item| {
58            if let TraitItem::Function { parameters, .. } = &documented_item.item {
59                for (name, _typ) in parameters {
60                    if self.byte_index == name.span().end() as usize {
61                        return Some((parameters, name.as_str()));
62                    }
63                }
64            }
65            None
66        });
67        let Some((parameters, name)) = parameters_and_name else {
68            return false;
69        };
70
71        let mut names_to_exclude = HashSet::new();
72        for (ident, _) in parameters {
73            if ident.as_str() != name {
74                names_to_exclude.insert(ident.to_string());
75            }
76        }
77
78        let mut suggested = HashSet::new();
79        for documented_item in &trait_.items {
80            if let TraitItem::Function { parameters, .. } = &documented_item.item {
81                for (ident, typ) in parameters {
82                    if matches!(typ.typ, UnresolvedTypeData::Error) {
83                        continue;
84                    };
85                    let param_name = ident.as_str();
86                    if names_to_exclude.contains(param_name) {
87                        continue;
88                    }
89
90                    if name_matches(param_name, name) {
91                        let label = format!("{param_name}: {typ}");
92                        if suggested.insert(label.clone()) {
93                            let item = variable_completion_item(label, None);
94                            self.completion_items.push(item);
95                        }
96                    }
97                }
98            }
99        }
100
101        true
102    }
103
104    fn suggest_function_parameters<'a>(
105        &mut self,
106        functions: impl Iterator<Item = &'a NoirFunction>,
107        name: &str,
108        names_to_exclude: HashSet<String>,
109    ) {
110        let mut suggested = HashSet::new();
111        for function in functions {
112            for parameter in function.parameters() {
113                let Pattern::Identifier(ident) = &parameter.pattern else {
114                    continue;
115                };
116                if matches!(parameter.typ.typ, UnresolvedTypeData::Error) {
117                    continue;
118                };
119                let param_name = ident.as_str();
120                if names_to_exclude.contains(param_name) {
121                    continue;
122                }
123
124                if name_matches(param_name, name) {
125                    let label = format!("{param_name}: {}", parameter.typ);
126                    if suggested.insert(label.clone()) {
127                        let item = variable_completion_item(label, None);
128                        self.completion_items.push(item);
129                    }
130                }
131            }
132        }
133    }
134}

136/// Tries to find a function parameter inside `functions` that is being autocompleted.
137/// Returns that function together with the parameter name, if found.
138fn find_function_and_parameter_name_at_byte_index<'a>(
139    mut functions: impl Iterator<Item = &'a NoirFunction>,
140    byte_index: usize,
141) -> Option<(&'a NoirFunction, &'a str)> {
142    functions.find_map(|function| {
143        for parameter in function.parameters() {
144            let Pattern::Identifier(ident) = &parameter.pattern else {
145                return None;
146            };
147            if byte_index == ident.span().end() as usize {
148                return Some((function, ident.as_str()));
149            }
150        }
151        None
152    })
153}

155// Don't suggest names of parameters that already exist in the given function,
156// unless it's the name currently being completed.
157fn names_to_exclude(function: &NoirFunction, name: &str) -> HashSet<String> {
158    let mut names_to_exclude = HashSet::new();
159    for parameter in function.parameters() {
160        let Pattern::Identifier(ident) = &parameter.pattern else {
161            continue;
162        };
163        if ident.as_str() != name {
164            names_to_exclude.insert(ident.to_string());
165        }
166    }
167    names_to_exclude
168}
