1use noirc_frontend::{Type, ast::AttributeTarget};

3/// When suggest a function as a result of completion, whether to autocomplete its name or its name and parameters.
4#[derive(Clone, Copy, PartialEq, Eq, Debug)]
5pub(super) enum FunctionCompletionKind {
6    // Only complete a function's name. This is used in use statement.
7    Name,
8    // Complete a function's name and parameters (as a snippet). This is used in regular code.
9    NameAndParameters,
10}
11
12/// Is there a requirement for suggesting functions?
13#[derive(Clone, Copy, PartialEq, Eq, Debug)]
14pub(super) enum FunctionKind<'a> {
15    /// No requirement: any function is okay to suggest.
16    Any,
17    /// Only show functions that have the given self type.
18    SelfType(&'a Type),
19}
20
21/// When requesting completions, whether to list all items or just types.
22/// For example, when writing `let x: S` we only want to suggest types at this
23/// point (modules too, because they might include types too).
24#[derive(Clone, Copy, PartialEq, Eq, Debug)]
25pub(super) enum RequestedItems {
26    // Suggest any items (types, functions, etc.).
27    AnyItems,
28    // Only suggest types (and modules, because they can contain types).
29    OnlyTypes,
30    // Only suggest traits (and modules, because they can contain traits).
31    OnlyTraits,
32    // Only attribute functions
33    OnlyAttributeFunctions(AttributeTarget),
34}
