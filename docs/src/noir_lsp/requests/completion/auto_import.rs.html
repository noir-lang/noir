20    pub(super) fn complete_auto_imports(
21        &mut self,
22        prefix: &str,
23        requested_items: RequestedItems,
24        function_completion_kind: FunctionCompletionKind,
25    ) {
26        let current_module_parent_id = self.module_id.parent(self.def_maps);
27
28        for (name, entries) in self.interner.get_auto_import_names() {
29            if !name_matches(name, prefix) {
30                continue;
31            }
32
33            for entry in entries {
34                let module_def_id = entry.module_def_id;
35                if self.suggested_module_def_ids.contains(&module_def_id) {
36                    continue;
37                }
38
39                let visibility = entry.visibility;
40                let mut defining_module = entry.defining_module.as_ref().cloned();
41
42                // If the item is offered via a re-export of it's parent module, this holds the name of the reexport.
43                let mut intermediate_name = None;
44
45                let is_visible =
46                    self.module_def_id_is_visible(module_def_id, visibility, defining_module);
47                if !is_visible {
48                    if let Some(reexport) =
49                        self.get_ancestor_module_reexport(module_def_id, visibility)
50                    {
51                        defining_module = Some(reexport.module_id);
52                        intermediate_name = Some(reexport.name);
53                    } else {
54                        continue;
55                    }
56                }
57
58                let completion_items = self.module_def_id_completion_items(
59                    module_def_id,
60                    name.clone(),
61                    function_completion_kind,
62                    FunctionKind::Any,
63                    requested_items,
64                );
65
66                if completion_items.is_empty() {
67                    continue;
68                }
69
70                self.suggested_module_def_ids.insert(module_def_id);
71
72                for mut completion_item in completion_items {
73                    let Some(full_path) = module_def_id_relative_path(
74                        module_def_id,
75                        name,
76                        self.module_id,
77                        current_module_parent_id,
78                        defining_module,
79                        &intermediate_name,
80                        self.interner,
81                        self.def_maps,
82                    ) else {
83                        continue;
84                    };
85
86                    let mut label_details = completion_item.label_details.unwrap();
87                    label_details.detail = Some(format!("(use {full_path})"));
88                    completion_item.label_details = Some(label_details);
89                    completion_item.additional_text_edits =
90                        Some(use_completion_item_additional_text_edits(
91                            UseCompletionItemAdditionTextEditsRequest {
92                                full_path: &full_path,
93                                files: self.files,
94                                file: self.file,
95                                lines: &self.lines,
96                                nesting: self.nesting,
97                                auto_import_line: self.auto_import_line,
98                            },
99                            &self.use_segment_positions,
100                        ));
101                    completion_item.sort_text = Some(auto_import_sort_text());
102
103                    self.completion_items.push(completion_item);
104                }
105            }
106        }
107    }
108
109    fn get_ancestor_module_reexport(

115            module_def_id,
116            visibility,
117            self.module_id,
118            self.interner,
119            self.def_maps,
120            self.dependencies,
121        )
122    }
