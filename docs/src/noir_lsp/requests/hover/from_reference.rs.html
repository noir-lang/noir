32pub(super) fn hover_from_reference(
33    file_id: FileId,
34    position: Position,
35    args: &ProcessRequestCallbackArgs,
36) -> Option<Hover> {
37    utils::position_to_byte_index(args.files, file_id, &position)
38        .and_then(|byte_index| {
39            let file = args.files.get_file(file_id).unwrap();
40            let source = file.source();
41            let (parsed_module, _errors) = noirc_frontend::parse_program(source, file_id);
42
43            let mut finder = VisitorReferenceFinder::new(file_id, source, byte_index, args);
44            finder.find(&parsed_module)
45        })
46        .or_else(|| {
47            args.interner.reference_at_location(args.location).map(|reference| (reference, None))
48        })
49        .and_then(|(reference, link_lsp_location)| {
50            let location = args.interner.reference_location(reference);
51            let lsp_location = link_lsp_location
52                .or_else(|| to_lsp_location(args.files, location.file, location.span));
53            format_reference(reference, args).map(|formatted| Hover {
54                range: lsp_location.map(|location| location.range),
55                contents: HoverContents::Markup(MarkupContent {
56                    kind: MarkupKind::Markdown,
57                    value: formatted,
58                }),
59            })
60        })
61}

63pub(super) fn format_reference(
64    reference: ReferenceId,
65    args: &ProcessRequestCallbackArgs,
66) -> Option<String> {
67    match reference {
68        ReferenceId::Module(id) => format_module(id, args),
69        ReferenceId::Type(id) => Some(format_type(id, args)),
70        ReferenceId::StructMember(id, field_index) => {
71            Some(format_struct_member(id, field_index, args))
72        }
73        ReferenceId::EnumVariant(id, variant_index) => {
74            Some(format_enum_variant(id, variant_index, args))
75        }
76        ReferenceId::Trait(id) => Some(format_trait(id, args)),
77        ReferenceId::TraitAssociatedType(id) => Some(format_trait_associated_type(id, args)),
78        ReferenceId::Global(id) => Some(format_global(id, args)),
79        ReferenceId::Function(id) => Some(format_function(id, args)),
80        ReferenceId::Alias(id) => Some(format_alias(id, args)),
81        ReferenceId::Local(id) => Some(format_local(id, args)),
82        ReferenceId::Reference(location, _) => {
83            format_reference(args.interner.find_referenced(location).unwrap(), args)
84        }
85    }
86}
87fn format_module(id: ModuleId, args: &ProcessRequestCallbackArgs) -> Option<String> {
88    let crate_root = args.def_maps[&id.krate].root();
89
90    let mut string = String::new();
91
92    if id.local_id == crate_root {
93        let dep = args.dependencies().iter().find(|dep| dep.crate_id == id.krate)?;
94        string.push_str("    crate ");
95        string.push_str(&dep.name.to_string());
96    } else {
97        // Note: it's not clear why `try_module_attributes` might return None here, but it happens.
98        // This is a workaround to avoid panicking in that case (which brings the LSP server down).
99        // Cases where this happens are related to generated code, so once that stops happening
100        // this won't be an issue anymore.
101        let module_attributes = args.interner.try_module_attributes(id)?;
102
103        if let Some(parent_local_id) = module_attributes.parent
104            && format_parent_module_from_module_id(
105                ModuleId { krate: id.krate, local_id: parent_local_id },
106                args,
107                &mut string,
108            )
109        {
110            string.push('\n');
111        }
112        string.push_str("    ");
113        string.push_str("mod ");
114        string.push_str(&module_attributes.name);
115    }
116
117    append_doc_comments(ReferenceId::Module(id), &mut string, args);
118
119    Some(string)
120}

122fn format_type(id: TypeId, args: &ProcessRequestCallbackArgs) -> String {
123    let typ = args.interner.get_type(id);
124    let typ = typ.borrow();
125    if let Some(fields) = typ.get_fields_as_written() {
126        format_struct(&typ, fields, args)
127    } else if let Some(variants) = typ.get_variants_as_written() {
128        format_enum(&typ, variants, args)
129    } else {
130        unreachable!("Type should either be a struct or an enum")
131    }
132}

134fn format_struct(
135    typ: &DataType,
136    fields: Vec<StructField>,
137    args: &ProcessRequestCallbackArgs,
138) -> String {
139    let mut string = String::new();
140    if format_parent_module(ModuleDefId::TypeId(typ.id), args, &mut string) {
141        string.push('\n');
142    }
143    string.push_str("    ");
144    string.push_str("struct ");
145    string.push_str(typ.name.as_str());
146    format_generics(&typ.generics, &mut string);
147    string.push_str(" {\n");
148    for field in fields {
149        string.push_str("        ");
150        string.push_str(field.name.as_str());
151        string.push_str(": ");
152        string.push_str(&format!("{}", field.typ));
153        string.push_str(",\n");
154    }
155    string.push_str("    }");
156
157    append_doc_comments(ReferenceId::Type(typ.id), &mut string, args);
158
159    string
160}

162fn format_enum(
163    typ: &DataType,
164    variants: Vec<EnumVariant>,
165    args: &ProcessRequestCallbackArgs,
166) -> String {
167    let mut string = String::new();
168    if format_parent_module(ModuleDefId::TypeId(typ.id), args, &mut string) {
169        string.push('\n');
170    }
171    string.push_str("    ");
172    string.push_str("enum ");
173    string.push_str(typ.name.as_str());
174    format_generics(&typ.generics, &mut string);
175    string.push_str(" {\n");
176    for field in variants {
177        string.push_str("        ");
178        string.push_str(field.name.as_str());
179
180        if !field.params.is_empty() {
181            let types = field.params.iter().map(ToString::to_string).collect::<Vec<_>>();
182            string.push('(');
183            string.push_str(&types.join(", "));
184            string.push(')');
185        }
186
187        string.push_str(",\n");
188    }
189    string.push_str("    }");
190
191    append_doc_comments(ReferenceId::Type(typ.id), &mut string, args);
192
193    string

291fn format_global(id: GlobalId, args: &ProcessRequestCallbackArgs) -> String {
292    let global_info = args.interner.get_global(id);
293    let definition_id = global_info.definition_id;
294    let definition = args.interner.definition(definition_id);
295    let typ = args.interner.definition_type(definition_id);
296
297    let mut string = String::new();
298    if format_parent_module(ModuleDefId::GlobalId(id), args, &mut string) {
299        string.push('\n');
300    }
301
302    let mut print_comptime = definition.comptime;
303
304    if let Some(stmt) = args.interner.get_global_let_statement(id) {
305        print_comptime = stmt.comptime;
306    }
307
308    string.push_str("    ");
309    if print_comptime {
310        string.push_str("comptime ");
311    }
312    if definition.mutable {
313        string.push_str("mut ");
314    }
315    string.push_str("global ");
316    string.push_str(global_info.ident.as_str());
317    string.push_str(": ");
318    string.push_str(&format!("{typ}"));
319
320    if let GlobalValue::Resolved(value) = &global_info.value
321        && let Some(value) = value_to_string(value)
322    {
323        string.push_str(" = ");
324        string.push_str(&value);
325    }
326
327    append_doc_comments(ReferenceId::Global(id), &mut string, args);
328
329    string.push_str(&go_to_type_links(&typ, args.interner, args.files));
330
331    string
332}

334fn format_function(id: FuncId, args: &ProcessRequestCallbackArgs) -> String {
335    let func_meta = args.interner.function_meta(&id);
336
337    // If this points to a trait method, see if we can figure out what's the concrete trait impl method
338    if let Some(func_id) = get_trait_impl_func_id(id, args, func_meta) {
339        return format_function(func_id, args);
340    }
341
342    let func_modifiers = args.interner.function_modifiers(&id);
343
344    let func_name_definition_id = args.interner.definition(func_meta.name.id);
345
346    let enum_variant = match (func_meta.type_id, func_meta.enum_variant_index) {
347        (Some(type_id), Some(index)) => Some((type_id, index)),
348        _ => None,
349    };
350
351    let (reference_id, module_def_id) = if let Some((type_id, variant_index)) = enum_variant {
352        (ReferenceId::EnumVariant(type_id, variant_index), ModuleDefId::TypeId(type_id))
353    } else {
354        (ReferenceId::Function(id), ModuleDefId::FunctionId(id))
355    };
356
357    let mut string = String::new();
358    let formatted_parent_module = format_parent_module(module_def_id, args, &mut string);
359
360    let formatted_parent_type = if let Some(trait_impl_id) = func_meta.trait_impl {
361        let trait_impl = args.interner.get_trait_implementation(trait_impl_id);
362        let trait_impl = trait_impl.borrow();
363        let trait_ = args.interner.get_trait(trait_impl.trait_id);
364
365        let generics = trait_impl
366            .trait_generics
367            .iter()
368            .filter_map(|generic| {
369                if let Type::NamedGeneric(generic) = generic {
370                    Some(generic.name.as_str())
371                } else {
372                    None
373                }
374            })
375            .collect::<Vec<_>>();
376
377        string.push('\n');
378        string.push_str("    impl");
379        if !generics.is_empty() {
380            string.push('<');
381            for (index, generic) in generics.into_iter().enumerate() {
382                if index > 0 {
383                    string.push_str(", ");
384                }
385                string.push_str(generic);
386            }
387            string.push('>');
388        }
389
390        string.push(' ');
391        string.push_str(trait_.name.as_str());
392        if !trait_impl.trait_generics.is_empty() {
393            string.push('<');
394            for (index, generic) in trait_impl.trait_generics.iter().enumerate() {
395                if index > 0 {
396                    string.push_str(", ");
397                }
398                string.push_str(&generic.to_string());
399            }
400            string.push('>');
401        }
402
403        string.push_str(" for ");
404        string.push_str(&trait_impl.typ.to_string());
405
406        true
407    } else if let Some(trait_id) = func_meta.trait_id {
408        let trait_ = args.interner.get_trait(trait_id);
409        string.push('\n');
410        string.push_str("    trait ");
411        string.push_str(trait_.name.as_str());
412        format_generics(&trait_.generics, &mut string);
413
414        true
415    } else if let Some(type_id) = func_meta.type_id {
416        let data_type = args.interner.get_type(type_id);
417        let data_type = data_type.borrow();
418        if formatted_parent_module {
419            string.push_str("::");
420        }
421        string.push_str(data_type.name.as_str());
422        if enum_variant.is_none() {
423            string.push('\n');
424            string.push_str("    ");
425            string.push_str("impl");
426
427            let impl_generics: Vec<_> = func_meta
428                .all_generics
429                .iter()
430                .take(func_meta.all_generics.len() - func_meta.direct_generics.len())
431                .cloned()
432                .collect();
433            format_generics(&impl_generics, &mut string);
434
435            string.push(' ');
436            string.push_str(data_type.name.as_str());
437            format_generic_names(&impl_generics, &mut string);
438        }
439
440        true
441    } else {
442        false
443    };
444    if formatted_parent_module || formatted_parent_type {
445        string.push('\n');
446    }
447    string.push_str("    ");
448
449    if func_modifiers.visibility != ItemVisibility::Private
450        && func_meta.trait_id.is_none()
451        && func_meta.trait_impl.is_none()
452    {
453        string.push_str(&func_modifiers.visibility.to_string());
454        string.push(' ');
455    }
456    if func_modifiers.is_unconstrained {
457        string.push_str("unconstrained ");
458    }
459    if func_modifiers.is_comptime {
460        string.push_str("comptime ");
461    }
462
463    let func_name = &func_name_definition_id.name;
464
465    if enum_variant.is_none() {
466        string.push_str("fn ");
467    }
468    string.push_str(func_name);
469    format_generics(&func_meta.direct_generics, &mut string);
470    string.push('(');
471    let parameters = &func_meta.parameters;
472    for (index, (pattern, typ, visibility)) in parameters.iter().enumerate() {
473        let is_self = pattern_is_self(pattern, args.interner);
474
475        // `&mut self` is represented as a mutable reference type, not as a mutable pattern
476        if is_self && matches!(typ, Type::Reference(..)) {
477            string.push_str("&mut ");
478        }
479
480        if enum_variant.is_some() {
481            string.push_str(&format!("{typ}"));
482        } else {
483            format_pattern(pattern, args.interner, &mut string);
484
485            // Don't add type for `self` param
486            if !is_self {
487                string.push_str(": ");
488                if matches!(visibility, Visibility::Public) {
489                    string.push_str("pub ");
490                }
491                string.push_str(&format!("{typ}"));
492            }
493        }
494
495        if index != parameters.len() - 1 {
496            string.push_str(", ");
497        }
498    }
499
500    string.push(')');
501
502    if enum_variant.is_none() {
503        let return_type = func_meta.return_type();
504        match return_type {
505            Type::Unit => (),
506            _ => {
507                string.push_str(" -> ");
508                string.push_str(&format!("{return_type}"));
509            }
510        }
511    }
512
513    if enum_variant.is_some() {
514        append_doc_comments(reference_id, &mut string, args);
515    } else {
516        let had_doc_comments = append_doc_comments(reference_id, &mut string, args);
517        if !had_doc_comments {
518            // If this function doesn't have doc comments, but it's a trait impl method,
519            // use the trait method doc comments.
520            if let Some(trait_impl_id) = func_meta.trait_impl {
521                let trait_impl = args.interner.get_trait_implementation(trait_impl_id);
522                let trait_impl = trait_impl.borrow();
523                let trait_ = args.interner.get_trait(trait_impl.trait_id);
524                if let Some(func_id) = trait_.method_ids.get(func_name) {
525                    let reference_id = ReferenceId::Function(*func_id);
526                    append_doc_comments(reference_id, &mut string, args);
527                }
528            }
529        }
530    }
531
532    if enum_variant.is_none() {
533        let return_type = func_meta.return_type();
534        string.push_str(&go_to_type_links(return_type, args.interner, args.files));
535    }
536
537    string
538}

540fn get_trait_impl_func_id(
541    id: FuncId,
542    args: &ProcessRequestCallbackArgs,
543    func_meta: &FuncMeta,
544) -> Option<FuncId> {
545    func_meta.trait_id?;
546
547    let index = args.interner.find_location_index(args.location)?;
548    let expr_id = args.interner.get_expr_id_from_index(index)?;
549    let Some(TraitImplKind::Normal(trait_impl_id)) =
550        args.interner.get_selected_impl_for_expression(expr_id)
551    else {
552        return None;
553    };
554
555    let trait_impl = args.interner.get_trait_implementation(trait_impl_id);
556    let trait_impl = trait_impl.borrow();
557
558    let function_name = args.interner.function_name(&id);
559    let mut trait_impl_methods = trait_impl.methods.iter();
560    let func_id =
561        trait_impl_methods.find(|func_id| args.interner.function_name(func_id) == function_name)?;
562    Some(*func_id)
563}

583fn format_local(id: DefinitionId, args: &ProcessRequestCallbackArgs) -> String {
584    let definition_info = args.interner.definition(id);
585    if let DefinitionKind::Global(global_id) = &definition_info.kind {
586        return format_global(*global_id, args);
587    }
588
589    let DefinitionKind::Local(expr_id) = definition_info.kind else {
590        panic!("Expected a local reference to reference a local definition")
591    };
592    let typ = args.interner.definition_type(id);
593
594    let mut string = String::new();
595    string.push_str("    ");
596    if definition_info.comptime {
597        string.push_str("comptime ");
598    }
599    if expr_id.is_some() {
600        string.push_str("let ");
601    }
602    if definition_info.mutable {
603        if expr_id.is_none() {
604            string.push_str("let ");
605        }
606        string.push_str("mut ");
607    }
608    string.push_str(&definition_info.name);
609    if !matches!(typ, Type::Error) {
610        string.push_str(": ");
611        string.push_str(&format!("{typ}"));
612    }
613
614    string.push_str(&go_to_type_links(&typ, args.interner, args.files));
615
616    string
617}

633fn format_generics_impl(generics: &ResolvedGenerics, only_show_names: bool, string: &mut String) {
634    if generics.is_empty() {
635        return;
636    }
637
638    string.push('<');
639    for (index, generic) in generics.iter().enumerate() {
640        if index > 0 {
641            string.push_str(", ");
642        }
643
644        if only_show_names {
645            string.push_str(&generic.name);
646        } else {
647            match generic.kind() {
648                noirc_frontend::Kind::Any | noirc_frontend::Kind::Normal => {
649                    string.push_str(&generic.name);
650                }
651                noirc_frontend::Kind::IntegerOrField | noirc_frontend::Kind::Integer => {
652                    string.push_str("let ");
653                    string.push_str(&generic.name);
654                    string.push_str(": u32");
655                }
656                noirc_frontend::Kind::Numeric(typ) => {
657                    string.push_str("let ");
658                    string.push_str(&generic.name);
659                    string.push_str(": ");
660                    string.push_str(&typ.to_string());
661                }
662            }
663        }
664    }
665    string.push('>');
666}

668fn format_pattern(pattern: &HirPattern, interner: &NodeInterner, string: &mut String) {
669    match pattern {
670        HirPattern::Identifier(ident) => {
671            let definition = interner.definition(ident.id);
672            string.push_str(&definition.name);
673        }
674        HirPattern::Mutable(pattern, _) => {
675            string.push_str("mut ");
676            format_pattern(pattern, interner, string);
677        }
678        HirPattern::Tuple(..) | HirPattern::Struct(..) => {
679            string.push('_');
680        }
681    }
682}
683
684fn pattern_is_self(pattern: &HirPattern, interner: &NodeInterner) -> bool {
685    match pattern {
686        HirPattern::Identifier(ident) => {
687            let definition = interner.definition(ident.id);
688            definition.name == "self"
689        }
690        HirPattern::Mutable(pattern, _) => pattern_is_self(pattern, interner),
691        HirPattern::Tuple(..) | HirPattern::Struct(..) => false,
692    }
693}

728fn go_to_type_links(typ: &Type, interner: &NodeInterner, files: &FileMap) -> String {
729    let mut gatherer = TypeLinksGatherer { interner, files, links: Vec::new() };
730    gatherer.gather_type_links(typ);
731
732    let links = gatherer.links;
733    if links.is_empty() {
734        "".to_string()
735    } else {
736        let mut string = String::new();
737        string.push_str("\n\n");
738        string.push_str("Go to ");
739        for (index, link) in links.iter().enumerate() {
740            if index > 0 {
741                string.push_str(" | ");
742            }
743            string.push_str(link);
744        }
745        string
746    }
747}

749struct TypeLinksGatherer<'a> {
750    interner: &'a NodeInterner,
751    files: &'a FileMap,
752    links: Vec<String>,
753}
754
755impl TypeLinksGatherer<'_> {
756    fn gather_type_links(&mut self, typ: &Type) {
757        match typ {
758            Type::Array(typ, _) => self.gather_type_links(typ),
759            Type::Vector(typ) => self.gather_type_links(typ),
760            Type::Tuple(types) => {
761                for typ in types {
762                    self.gather_type_links(typ);
763                }
764            }
765            Type::DataType(data_type, generics) => {
766                self.gather_struct_type_links(data_type);
767                for generic in generics {
768                    self.gather_type_links(generic);
769                }
770            }
771            Type::Alias(type_alias, generics) => {
772                self.gather_type_alias_links(type_alias);
773                for generic in generics {
774                    self.gather_type_links(generic);
775                }
776            }
777            Type::TypeVariable(var) => {
778                self.gather_type_variable_links(var);
779            }
780            Type::TraitAsType(trait_id, _, generics) => {
781                let some_trait = self.interner.get_trait(*trait_id);
782                self.gather_trait_links(some_trait);
783                for generic in &generics.ordered {
784                    self.gather_type_links(generic);
785                }
786                for named_type in &generics.named {
787                    self.gather_type_links(&named_type.typ);
788                }
789            }
790            Type::NamedGeneric(NamedGeneric { type_var, .. }) => {
791                self.gather_type_variable_links(type_var);
792            }
793            Type::Function(args, return_type, env, _) => {
794                for arg in args {
795                    self.gather_type_links(arg);
796                }
797                self.gather_type_links(return_type);
798                self.gather_type_links(env);
799            }
800            Type::Reference(typ, _) => self.gather_type_links(typ),
801            Type::InfixExpr(lhs, _, rhs, _) => {
802                self.gather_type_links(lhs);
803                self.gather_type_links(rhs);
804            }
805            Type::CheckedCast { to, .. } => self.gather_type_links(to),
806            Type::FieldElement
807            | Type::Integer(..)
808            | Type::Bool
809            | Type::String(_)
810            | Type::FmtString(_, _)
811            | Type::Unit
812            | Type::Forall(_, _)
813            | Type::Constant(..)
814            | Type::Quoted(_)
815            | Type::Error => (),
816        }
817    }
818
819    fn gather_struct_type_links(&mut self, struct_type: &Shared<DataType>) {
820        let struct_type = struct_type.borrow();
821        if let Some(lsp_location) =
822            to_lsp_location(self.files, struct_type.location.file, struct_type.name.span())
823        {
824            self.push_link(format_link(struct_type.name.as_str(), lsp_location));
825        }
826    }
827
828    fn gather_type_alias_links(&mut self, type_alias: &Shared<TypeAlias>) {
829        let type_alias = type_alias.borrow();
830        if let Some(lsp_location) =
831            to_lsp_location(self.files, type_alias.location.file, type_alias.name.span())
832        {
833            self.push_link(format_link(type_alias.name.as_str(), lsp_location));
834        }
835    }
836
837    fn gather_trait_links(&mut self, some_trait: &Trait) {
838        if let Some(lsp_location) =
839            to_lsp_location(self.files, some_trait.location.file, some_trait.name.span())
840        {
841            self.push_link(format_link(some_trait.name.as_str(), lsp_location));
842        }
843    }
844
845    fn gather_type_variable_links(&mut self, var: &TypeVariable) {
846        let var = &*var.borrow();
847        match var {
848            TypeBinding::Bound(typ) => {
849                self.gather_type_links(typ);
850            }
851            TypeBinding::Unbound(..) => (),
852        }
853    }
854
855    fn push_link(&mut self, link: String) {
856        if !self.links.contains(&link) {
857            self.links.push(link);
858        }
859    }
860}
861
862fn format_link(name: &str, location: lsp_types::Location) -> String {
863    format!(
864        "[{}]({}#L{},{}-{},{})",
865        name,
866        location.uri,
867        location.range.start.line + 1,
868        location.range.start.character + 1,
869        location.range.end.line + 1,
870        location.range.end.character + 1
871    )
872}

874fn append_doc_comments(
875    id: ReferenceId,
876    string: &mut String,
877    args: &ProcessRequestCallbackArgs,
878) -> bool {
879    let Some(doc_comments) = args.interner.doc_comments(id) else {
880        return false;
881    };
882
883    string.push_str("\n\n---\n\n");
884
885    let doc_comments = vecmap(doc_comments, |comment| comment.contents.clone()).join("\n");
886    let doc_comments = process_doc_comments_links(doc_comments, id, args);
887
888    string.push_str(&doc_comments);
889    string.push('\n');
890
891    true
892}

894/// Replaces markdown links inside doc comments that point to Noir items to point at code locations
895/// where those items are defined.
896fn process_doc_comments_links(
897    comments: String,
898    id: ReferenceId,
899    args: &ProcessRequestCallbackArgs,
900) -> String {
901    let Some((current_module_id, current_type)) = current_module_and_type(id, args) else {
902        return comments;
903    };
904
905    let links = LinkFinder::default().find_links(
906        &comments,
907        current_module_id,
908        current_type,
909        args.interner,
910        args.def_maps,
911        args.crate_graph,
912    );
913    if links.is_empty() {
914        return comments;
915    }
916
917    let mut lines = comments.lines().map(|line| line.to_string()).collect::<Vec<_>>();
918
919    for link in links.into_iter().rev() {
920        let Some(target) = link.target else {
921            continue;
922        };
923
924        let Some(location) = link_target_location(target, args) else {
925            continue;
926        };
927        let mut line = lines[link.line].to_string();
928        let replacement = format_link(&link.name, location);
929        line.replace_range(link.start..link.end, &replacement);
930        lines[link.line] = line;
931    }
932
933    lines.join("\n")
934}

936/// Returns the Location where a link target exists.
937/// The Location might not exist. For example, primitive types have no definition location.
938fn link_target_location(
939    target: LinkTarget,
940    args: &ProcessRequestCallbackArgs,
941) -> Option<lsp_types::Location> {
942    match target {
943        LinkTarget::TopLevelItem(module_def_id) => match module_def_id {
944            ModuleDefId::ModuleId(module_id) => {
945                let module_attributes = args.interner.try_module_attributes(module_id)?;
946                let location = module_attributes.location;
947                to_lsp_location(args.files, location.file, location.span)
948            }
949            ModuleDefId::FunctionId(func_id) => {
950                let func_meta = args.interner.function_meta(&func_id);
951                let location = func_meta.location;
952                to_lsp_location(args.files, location.file, location.span)
953            }
954            ModuleDefId::TypeId(type_id) => {
955                let typ = args.interner.get_type(type_id);
956                let typ = typ.borrow();
957                let location = typ.location;
958                to_lsp_location(args.files, location.file, location.span)
959            }
960            ModuleDefId::TypeAliasId(type_alias_id) => {
961                let type_alias = args.interner.get_type_alias(type_alias_id);
962                let type_alias = type_alias.borrow();
963                let location = type_alias.location;
964                to_lsp_location(args.files, location.file, location.span)
965            }
966            ModuleDefId::TraitId(trait_id) => {
967                let some_trait = args.interner.get_trait(trait_id);
968                let location = some_trait.location;
969                to_lsp_location(args.files, location.file, location.span)
970            }
971            ModuleDefId::TraitAssociatedTypeId(..) => None,
972            ModuleDefId::GlobalId(global_id) => {
973                let global_info = args.interner.get_global(global_id);
974                let location = global_info.location;
975                to_lsp_location(args.files, location.file, location.span)
976            }
977        },
978        LinkTarget::Method(_, func_id) | LinkTarget::PrimitiveTypeFunction(_, func_id) => {
979            let func_meta = args.interner.function_meta(&func_id);
980            let location = func_meta.location;
981            to_lsp_location(args.files, location.file, location.span)
982        }
983        LinkTarget::StructMember(type_id, index) => {
984            let struct_type = args.interner.get_type(type_id);
985            let struct_type = struct_type.borrow();
986            let field = struct_type.field_at(index);
987            let location = field.name.location();
988            to_lsp_location(args.files, location.file, location.span)
989        }
990        LinkTarget::PrimitiveType(_) => {
991            // Can't link to primitive types
992            None
993        }
994    }
995}

997fn value_to_string(value: &Value) -> Option<String> {
998    let mut string = String::new();
999    append_value_to_string(value, &mut string)?;
1000    Some(string)
1001}
1002
1003fn append_value_to_string(value: &Value, string: &mut String) -> Option<()> {
1004    match value {
1005        Value::Unit => string.push_str("()"),
1006        Value::Bool(value) => string.push_str(&value.to_string()),
1007        Value::Field(field_element) => string.push_str(&field_element.to_string()),
1008        Value::I8(value) => string.push_str(&value.to_string()),
1009        Value::I16(value) => string.push_str(&value.to_string()),
1010        Value::I32(value) => string.push_str(&value.to_string()),
1011        Value::I64(value) => string.push_str(&value.to_string()),
1012        Value::U1(value) => string.push_str(&value.to_string()),
1013        Value::U8(value) => string.push_str(&value.to_string()),
1014        Value::U16(value) => string.push_str(&value.to_string()),
1015        Value::U32(value) => string.push_str(&value.to_string()),
1016        Value::U64(value) => string.push_str(&value.to_string()),
1017        Value::U128(value) => string.push_str(&value.to_string()),
1018        Value::String(value) | Value::CtString(value) => string.push_str(&value.to_string()),
1019        Value::Tuple(values) => {
1020            let len = values.iter().len();
1021            string.push('(');
1022            for (index, value) in values.iter().enumerate() {
1023                if index > 0 {
1024                    string.push_str(", ");
1025                }
1026                append_value_to_string(&value.borrow(), string)?;
1027            }
1028            if len == 1 {
1029                string.push(',');
1030            }
1031            string.push(')');
1032        }
1033        Value::Array(values, _) => {
1034            string.push('[');
1035            for (index, value) in values.iter().enumerate() {
1036                if index > 0 {
1037                    string.push_str(", ");
1038                }
1039                append_value_to_string(value, string)?;
1040            }
1041            string.push(']');
1042        }
1043        Value::Vector(values, _) => {
1044            string.push_str("@[");
1045            for (index, value) in values.iter().enumerate() {
1046                if index > 0 {
1047                    string.push_str(", ");
1048                }
1049                append_value_to_string(value, string)?;
1050            }
1051            string.push(']');
1052        }
1053        // We could turn these into strings but the output wouldn't be very useful to users
1054        Value::FormatString(..)
1055        | Value::Function(..)
1056        | Value::Closure(..)
1057        | Value::Struct(..)
1058        | Value::Enum(..)
1059        | Value::Pointer(..)
1060        | Value::Quoted(..)
1061        | Value::TypeDefinition(..)
1062        | Value::TraitConstraint(..)
1063        | Value::TraitDefinition(..)
1064        | Value::TraitImpl(..)
1065        | Value::FunctionDefinition(..)
1066        | Value::ModuleDefinition(..)
1067        | Value::Type(..)
1068        | Value::Zeroed(..)
1069        | Value::Expr(..)
1070        | Value::TypedExpr(..)
1071        | Value::UnresolvedType(..) => return None,
1072    }
1073
1074    Some(())
1075}
