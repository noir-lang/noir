54pub(crate) fn on_code_lens_request(
55    state: &mut LspState,
56    params: CodeLensParams,
57) -> impl Future<Output = Result<CodeLensResult, ResponseError>> + use<> {
58    future::ready(on_code_lens_request_inner(state, params))
59}
60
61fn on_code_lens_request_inner(
62    state: &mut LspState,
63    params: CodeLensParams,
64) -> Result<CodeLensResult, ResponseError> {
65    let Ok(file_path) = params.text_document.uri.to_file_path() else {
66        return Ok(None);
67    };
68
69    let text_document_position_params = TextDocumentPositionParams {
70        text_document: params.text_document,
71        position: Position::new(0, 0),
72    };
73    process_request(state, text_document_position_params, |args| {
74        let file_id = args.files.get_file_id(&PathString::from_path(file_path))?;
75        let file = args.files.get_file(file_id).unwrap();
76        let source = file.source();
77        let (parsed_module, _errors) = noirc_frontend::parse_program(source, file_id);
78
79        let collected_lenses = collect_lenses_for_file(
80            parsed_module,
81            args.workspace,
82            args.package,
83            args.interner,
84            args.def_maps,
85            args.crate_graph,
86            args.files,
87        );
88        if collected_lenses.is_empty() { None } else { Some(collected_lenses) }
89    })
90}

92pub(crate) fn collect_lenses_for_file(
93    parsed_module: ParsedModule,
94    workspace: &Workspace,
95    package: &Package,
96    interner: &NodeInterner,
97    def_maps: &DefMaps,
98    crate_graph: &CrateGraph,
99    files: &FileMap,
100) -> Vec<CodeLens> {
101    let mut visitor = CodeLensVisitor {
102        workspace,
103        package,
104        interner,
105        def_maps,
106        crate_graph,
107        files,
108        nesting: 0,
109        lenses: vec![],
110    };
111    parsed_module.accept(&mut visitor);
112    visitor.lenses
113}
114
115struct CodeLensVisitor<'a> {
116    workspace: &'a Workspace,
117    package: &'a Package,
118    interner: &'a NodeInterner,
119    def_maps: &'a DefMaps,
120    crate_graph: &'a CrateGraph,
121    files: &'a FileMap,
122    nesting: usize,
123    lenses: Vec<CodeLens>,
124}
125
126impl Visitor for CodeLensVisitor<'_> {
127    fn visit_noir_function(&mut self, function: &NoirFunction, _: Span) -> bool {
128        let location = function.name_ident().location();
129
130        // Check if it's a main function
131        if self.nesting == 0 && self.package.is_binary() && function.name() == "main" {
132            let range = byte_span_to_range(self.files, location.file, location.span.into())
133                .unwrap_or_default();
134            self.lenses.push(compile_lens(self.workspace, self.package, range));
135            self.lenses.push(info_lens(self.workspace, self.package, range));
136            self.lenses.push(execute_lens(self.workspace, self.package, range));
137            self.lenses.push(debug_lens(self.workspace, self.package, range));
138        }
139
140        // Check if it's a test function
141        if let Some(ReferenceId::Function(func_id)) = self.interner.reference_at_location(location)
142        {
143            if self.interner.function_modifiers(&func_id).attributes.is_test_function() {
144                let func_meta = self.interner.function_meta(&func_id);
145                let local_module_id = func_meta.source_module;
146                let crate_id = func_meta.source_crate;
147                let module_id = ModuleId { krate: crate_id, local_id: local_module_id };
148                let module_path = fully_qualified_module_path(
149                    self.def_maps,
150                    self.crate_graph,
151                    &crate_id,
152                    module_id,
153                );
154                let func_name = if module_path.is_empty() {
155                    function.name().to_string()
156                } else {
157                    format!("{}::{}", module_path, function.name())
158                };
159
160                let range = byte_span_to_range(self.files, location.file, location.span.into())
161                    .unwrap_or_default();
162                self.lenses.push(test_lens(self.workspace, self.package, &func_name, range));
163                self.lenses.push(debug_test_lens(self.workspace, self.package, func_name, range));
164            }
165        };
166
167        false
168    }
169
170    fn visit_parsed_submodule(&mut self, parsed_sub_module: &ParsedSubModule, _: Span) -> bool {
171        // Check if this is a contract
172        if parsed_sub_module.is_contract && self.package.is_contract() {
173            let location = parsed_sub_module.name.location();
174            let range = byte_span_to_range(self.files, location.file, location.span.into())
175                .unwrap_or_default();
176            self.lenses.push(compile_lens(self.workspace, self.package, range));
177            self.lenses.push(info_lens(self.workspace, self.package, range));
178        }
179
180        self.nesting += 1;
181        parsed_sub_module.accept_children(self);
182        self.nesting -= 1;
183        false
184    }
185
186    fn visit_type_impl(&mut self, _: &TypeImpl, _: Span) -> bool {
187        // We don't want to find functions inside impl blocks
188        false
189    }
190
191    fn visit_noir_trait(&mut self, _: &NoirTrait, _: Span) -> bool {
192        // We don't want to find functions inside traits
193        false
194    }
195
196    fn visit_noir_trait_impl(&mut self, _: &NoirTraitImpl, _: Span) -> bool {
197        // We don't want to find functions inside trait impls
198        false
199    }
200}

307    async fn get_code_lens(src: &str, directory: &str) -> CodeLensResult {
308        let (mut state, noir_text_document) = test_utils::init_lsp_server(directory).await;
309
310        let _ = on_did_open_text_document(
311            &mut state,
312            DidOpenTextDocumentParams {
313                text_document: TextDocumentItem {
314                    uri: noir_text_document.clone(),
315                    language_id: "noir".to_string(),
316                    version: 0,
317                    text: src.to_string(),
318                },
319            },
320        );
321
322        on_code_lens_request(
323            &mut state,
324            CodeLensParams {
325                text_document: TextDocumentIdentifier { uri: noir_text_document },
326                work_done_progress_params: WorkDoneProgressParams { work_done_token: None },
327                partial_result_params: PartialResultParams { partial_result_token: None },

329        )

357    async fn test_main_code_lens() {
358        let src = r#"fn main() {}"#;
359
360        let code_lens = get_code_lens(src, "document_symbol").await.unwrap();
361        assert_eq!(code_lens.len(), 4);
362
363        for lens in &code_lens {
364            assert_eq!(lens.range.start.line, 0);
365            assert_eq!(lens.range.end.line, 0);
366            assert_eq!(
367                &src[lens.range.start.character as usize..lens.range.end.character as usize],
368                "main"
369            );
370        }
371
372        let mut titles = vecmap(code_lens, |lens| lens.command.unwrap().title);
373        titles.sort();
