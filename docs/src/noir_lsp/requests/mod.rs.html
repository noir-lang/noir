418    state: &mut LspState,
419    params: lsp_types::DocumentFormattingParams,
420) -> impl Future<Output = Result<Option<Vec<lsp_types::TextEdit>>, ResponseError>> + use<> {
421    std::future::ready(on_formatting_inner(state, params))

423
424fn on_formatting_inner(
425    state: &LspState,
426    params: lsp_types::DocumentFormattingParams,
427) -> Result<Option<Vec<lsp_types::TextEdit>>, ResponseError> {
428    // The file_path might be Err/None if the action runs against an unsaved file
429    let file_path = params.text_document.uri.to_file_path().ok();
430    let directory_path = file_path.as_ref().and_then(|path| path.parent());
431
432    let path = params.text_document.uri.to_string();
433
434    if let Some(source) = state.input_files.get(&path) {
435        let (module, errors) = noirc_frontend::parse_program_with_dummy_file(source);
436        let is_all_warnings = errors.iter().all(ParserError::is_warning);
437        if !is_all_warnings {
438            return Ok(None);
439        }
440
441        let config = read_format_config(directory_path);
442        let new_text = nargo_fmt::format(source, module, &config);
443
444        let start_position = Position { line: 0, character: 0 };
445        let end_position = Position {
446            line: source.lines().count() as u32,
447            character: source.chars().count() as u32,
448        };
449
450        Ok(Some(vec![lsp_types::TextEdit {
451            range: lsp_types::Range::new(start_position, end_position),
452            new_text,
453        }]))
454    } else {
455        Ok(None)
456    }
457}

459fn read_format_config(file_path: Option<&Path>) -> Config {
460    match file_path {
461        Some(file_path) => match Config::read(file_path) {
462            Ok(config) => config,
463            Err(err) => {
464                eprintln!("{err}");
465                Config::default()
466            }
467        },
468        None => Config::default(),
469    }
470}

472pub(crate) fn position_to_byte_index<'a, F>(
473    files: &'a F,
474    file_id: F::FileId,
475    position: &Position,
476) -> Result<usize, Error>
477where
478    F: fm::codespan_files::Files<'a> + ?Sized,
479{
480    let source = files.source(file_id)?;
481    let source = source.as_ref();
482
483    let line_span = files.line_range(file_id, position.line as usize)?;
484
485    let line_str = source.get(line_span.clone());
486
487    if let Some(line_str) = line_str {
488        let byte_offset = character_to_line_offset(line_str, position.character)?;
489        Ok(line_span.start + byte_offset)
490    } else {
491        Err(Error::InvalidCharBoundary { given: position.line as usize })
492    }
493}

495fn position_to_location(
496    files: &FileMap,
497    file_path: &PathString,
498    position: &Position,
499) -> Result<noirc_errors::Location, ResponseError> {
500    let file_id = file_path_to_file_id(files, file_path)?;
501    let byte_index = position_to_byte_index(files, file_id, position).map_err(|err| {
502        ResponseError::new(
503            ErrorCode::REQUEST_FAILED,
504            format!("Could not convert position to byte index. Error: {err:?}"),
505        )
506    })?;
507
508    let location = noirc_errors::Location {
509        file: file_id,
510        span: noirc_errors::Span::single_char(byte_index as u32),
511    };
512
513    Ok(location)
514}

526fn character_to_line_offset(line: &str, character: u32) -> Result<usize, Error> {
527    let line_len = line.len();
528    let mut character_offset = 0;
529
530    let mut chars = line.chars();
531    while let Some(ch) = chars.next() {
532        if character_offset == character {
533            let chars_off = chars.as_str().len();
534            let ch_off = ch.len_utf8();
535
536            return Ok(line_len - chars_off - ch_off);
537        }
538
539        character_offset += ch.len_utf16() as u32;
540    }
541
542    // Handle positions after the last character on the line
543    if character_offset == character {
544        Ok(line_len)
545    } else {
546        Err(Error::ColumnTooLarge { given: character_offset as usize, max: line.len() })
547    }
548}

550pub(crate) fn to_lsp_location(
551    files: &FileMap,
552    file_id: FileId,
553    definition_span: noirc_errors::Span,
554) -> Option<Location> {
555    let range = crate::byte_span_to_range(files, file_id, definition_span.into())?;
556    let file_name = files.get_absolute_name(file_id).ok()?;
557    let path = file_name.to_string();
558    if let Ok(uri) = Url::from_file_path(path.clone()) {
559        Some(Location { uri, range })
560    } else if path.starts_with("std/") {
561        Some(Location { uri: Url::from_str(&format!("noir-std://{path}")).unwrap(), range })
562    } else {
563        None
564    }

594pub(crate) fn process_request<F, T>(
595    state: &mut LspState,
596    text_document_position_params: TextDocumentPositionParams,
597    callback: F,
598) -> Result<T, ResponseError>
599where
600    F: FnOnce(ProcessRequestCallbackArgs) -> T,
601{
602    let uri = text_document_position_params.text_document.uri.clone();
603    let file_path = uri_to_file_path(&uri)?;
604    let workspace = if uri.scheme() == "noir-std" {
605        fake_stdlib_workspace()
606    } else {
607        resolve_workspace_for_source_path(file_path.as_path()).unwrap()
608    };
609
610    let package = crate::workspace_package_for_file(&workspace, &file_path).ok_or_else(|| {
611        ResponseError::new(ErrorCode::REQUEST_FAILED, "Could not find package for file")
612    })?;
613
614    // In practice when `process_request` is called, a document in the project should already have been
615    // open so both the workspace and package cache will have data. However, just in case this isn't true
616    // for some reason, and also for tests (some tests just test a request without going through the full
617    // LSP workflow), we have a fallback where we type-check the workspace/package, then continue with
618    // processing the request.
619    let Some(workspace_cache_data) = state.workspace_cache.get(&workspace.root_dir) else {
620        return process_request_no_workspace_cache(state, text_document_position_params, callback);
621    };
622
623    let Some(package_cache_data) = state.package_cache.get(&package.root_dir) else {
624        return process_request_no_workspace_cache(state, text_document_position_params, callback);
625    };
626
627    let file_manager = &workspace_cache_data.file_manager;
628    let interner = &package_cache_data.node_interner;
629    let def_maps = &package_cache_data.def_maps;
630    let usage_tracker = &package_cache_data.usage_tracker;
631    let crate_graph = &package_cache_data.crate_graph;
632    let crate_id = package_cache_data.crate_id;
633
634    let files = file_manager.as_file_map();
635
636    let location = position_to_location(
637        files,
638        &PathString::from(file_path),
639        &text_document_position_params.position,
640    )?;
641
642    Ok(callback(ProcessRequestCallbackArgs {
643        location,
644        workspace: &workspace,
645        package,
646        files,
647        interner,
648        package_cache: &state.package_cache,
649        crate_id,
650        crate_name: package.name.to_string(),
651        crate_graph,
652        def_maps,
653        usage_tracker,
654    }))
655}

657pub(crate) fn process_request_no_workspace_cache<F, T>(
658    state: &mut LspState,
659    text_document_position_params: TextDocumentPositionParams,
660    callback: F,
661) -> Result<T, ResponseError>
662where
663    F: FnOnce(ProcessRequestCallbackArgs) -> T,
664{
665    let file_path =
666        text_document_position_params.text_document.uri.to_file_path().map_err(|_| {
667            ResponseError::new(ErrorCode::REQUEST_FAILED, "URI is not a valid file path")
668        })?;
669
670    let workspace = resolve_workspace_for_source_path(file_path.as_path()).unwrap();
671    let package = crate::workspace_package_for_file(&workspace, &file_path).ok_or_else(|| {
672        ResponseError::new(ErrorCode::REQUEST_FAILED, "Could not find package for file")
673    })?;
674
675    let mut workspace_file_manager = workspace.new_file_manager();
676    insert_all_files_for_workspace_into_file_manager(
677        state,
678        &workspace,
679        &mut workspace_file_manager,
680    );
681    let parsed_files = parse_diff(&workspace_file_manager, state);
682
683    let (mut context, crate_id) =
684        crate::prepare_package(&workspace_file_manager, &parsed_files, package);
685
686    let interner;
687    let def_maps;
688    let usage_tracker;
689    if let Some(package_cache) = state.package_cache.get(&package.root_dir) {
690        interner = &package_cache.node_interner;
691        def_maps = &package_cache.def_maps;
692        usage_tracker = &package_cache.usage_tracker;
693    } else {
694        // We ignore the warnings and errors produced by compilation while resolving the definition
695        let _ = noirc_driver::check_crate(&mut context, crate_id, &Default::default());
696        interner = &context.def_interner;
697        def_maps = &context.def_maps;
698        usage_tracker = &context.usage_tracker;
699    }
700
701    let files = workspace_file_manager.as_file_map();
702
703    let location = position_to_location(
704        files,
705        &PathString::from(file_path),
706        &text_document_position_params.position,
707    )?;
708
709    Ok(callback(ProcessRequestCallbackArgs {
710        location,
711        workspace: &workspace,
712        package,
713        files,
714        interner,
715        package_cache: &state.package_cache,
716        crate_id,
717        crate_name: package.name.to_string(),
718        crate_graph: &context.crate_graph,
719        def_maps,
720        usage_tracker,
721    }))
722}

724pub(crate) fn uri_to_file_path(uri: &Url) -> Result<PathBuf, ResponseError> {
725    if uri.scheme() == "noir-std" {
726        Ok(PathBuf::from_str(&format!("{}{}", uri.host().unwrap(), uri.path())).unwrap())
727    } else {
728        uri.to_file_path().map_err(|_| {
729            ResponseError::new(ErrorCode::REQUEST_FAILED, "URI is not a valid file path")
730        })
731    }

734pub(crate) fn find_all_references_in_workspace(
735    location: noirc_errors::Location,
736    interner: &NodeInterner,
737    package_cache: &HashMap<PathBuf, PackageCacheData>,
738    files: &FileMap,
739    include_declaration: bool,
740    include_self_type_name: bool,
741) -> Option<Vec<Location>> {
742    // First find the node that's referenced by the given location, if any
743    let referenced = interner.find_referenced(location)?;
744    let name = get_reference_name(referenced, interner)?;
745
746    // If we found the referenced node, find its location
747    let referenced_location = interner.reference_location(referenced);
748
749    // Now we find all references that point to this location, in all interners
750    // (there's one interner per package, and all interners in a workspace rely on the
751    // same FileManager so a Location/FileId in one package is the same as in another package)
752    let mut locations = find_all_references(
753        referenced_location,
754        interner,
755        include_declaration,
756        include_self_type_name,
757    );
758    for cache_data in package_cache.values() {
759        locations.extend(find_all_references(
760            referenced_location,
761            &cache_data.node_interner,
762            include_declaration,
763            include_self_type_name,
764        ));
765    }
766
767    // Only keep locations whose span, when read from the file, matches "name"
768    // (it might not match because of macro expansions)
769    locations.retain(|location| {
770        let Some(file) = files.get_file(location.file) else {
771            return false;
772        };
773
774        let Some(substring) =
775            file.source().get(location.span.start() as usize..location.span.end() as usize)
776        else {
777            return false;
778        };
779
780        substring == name
781    });
782
783    let mut locations = locations
784        .iter()
785        .filter_map(|location| to_lsp_location(files, location.file, location.span))
786        .collect::<Vec<_>>();
787
788    // The LSP client usually removes duplicate locations, but we do it here just in case they don't
789    locations.sort_by_key(|location| {
790        (
791            location.uri.to_string(),
792            location.range.start.line,
793            location.range.start.character,
794            location.range.end.line,
795            location.range.end.character,
796        )
797    });
798    locations.dedup();
799
800    if locations.is_empty() { None } else { Some(locations) }
801}

803pub(crate) fn find_all_references(
804    referenced_location: noirc_errors::Location,
805    interner: &NodeInterner,
806    include_declaration: bool,
807    include_self_type_name: bool,
808) -> Vec<noirc_errors::Location> {
809    interner
810        .find_all_references(referenced_location, include_declaration, include_self_type_name)
811        .unwrap_or_default()
812}
813
814fn get_reference_name(reference: ReferenceId, interner: &NodeInterner) -> Option<String> {
815    match reference {
816        ReferenceId::Module(module_id) => {
817            Some(interner.try_module_attributes(module_id)?.name.clone())
818        }
819        ReferenceId::Type(type_id) => Some(interner.get_type(type_id).borrow().name.to_string()),
820        ReferenceId::StructMember(type_id, index) => {
821            Some(interner.get_type(type_id).borrow().field_at(index).name.to_string())
822        }
823        ReferenceId::EnumVariant(type_id, index) => {
824            Some(interner.get_type(type_id).borrow().variant_at(index).name.to_string())
825        }
826        ReferenceId::Trait(trait_id) => Some(interner.get_trait(trait_id).name.to_string()),
827        ReferenceId::TraitAssociatedType(id) => {
828            Some(interner.get_trait_associated_type(id).name.to_string())
829        }
830        ReferenceId::Global(global_id) => Some(interner.get_global(global_id).ident.to_string()),
831        ReferenceId::Function(func_id) => Some(interner.function_name(&func_id).to_string()),
832        ReferenceId::Alias(type_alias_id) => {
833            Some(interner.get_type_alias(type_alias_id).borrow().name.to_string())
834        }
835        ReferenceId::Local(definition_id) => {
836            Some(interner.definition_name(definition_id).to_string())
837        }
838        ReferenceId::Reference(location, _) => {
839            get_reference_name(interner.find_referenced(location)?, interner)
840        }
841    }
842}
