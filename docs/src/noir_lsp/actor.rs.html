1//! The compiler actor owns the [`LspState`] (source overrides, parse caches, type-checked
2//! packages) and is the only place compiler work runs. The LSP main loop forwards every
3//! request and notification to it as a message and immediately returns to serving protocol
4//! traffic, so the editor is never blocked behind a long type-check.
5//!
6//! Messages are processed strictly in order. This gives the freshness guarantee LSP needs
7//! without any bookkeeping: a request enqueued after a document change is processed after
8//! the re-check that change triggered, so it always sees up-to-date compiler state.
9//!
10//! Compiler state cannot cross threads (`NodeInterner` transitively contains `Rc`), which
11//! rules out type-checking on a helper thread and handing the result back. Instead the
12//! state is created on the thread that processes messages and never leaves it; only
13//! messages (jobs and their `Send` results) cross the thread boundary.
14
15use std::future::Future;
16use std::pin::Pin;
17use std::sync::mpsc;
18use std::task::{Context, Poll};

20use acvm::{BlackBoxFunctionSolver, FieldElement};
21use async_lsp::lsp_types::DidChangeTextDocumentParams;
22use async_lsp::{ClientSocket, ErrorCode, ResponseError};
23use tokio::sync::oneshot;
24
25use crate::LspState;
26use crate::notifications::on_did_change_text_document;

28/// A unit of work for the compiler actor.
29pub(crate) enum ActorMessage {
30    /// An arbitrary job to run against the compiler state.
31    Job(Box<dyn FnOnce(&mut LspState) + Send>),
32    /// A document's contents changed. This is a dedicated variant so that consecutive
33    /// changes to the same document can be coalesced, type-checking only the latest text.
34    FileChanged(DidChangeTextDocumentParams),
35}
36
37pub(crate) struct CompilerActor {
38    sender: mpsc::Sender<ActorMessage>,
39}
40
41impl CompilerActor {
42    /// Spawns a dedicated thread owning the compiler state and processing messages in order.
43    pub(crate) fn spawn(
44        client: ClientSocket,
45        solver: impl BlackBoxFunctionSolver<FieldElement> + Send + 'static,
46    ) -> Self {
47        let (sender, receiver) = mpsc::channel();
48        std::thread::Builder::new()
49            .name("noir-compiler".to_string())
50            .spawn(move || {
51                let mut state = LspState::new(&client, solver);
52                run_actor_loop(&mut state, receiver);
53            })
54            .expect("failed to spawn the compiler actor thread");
55        Self { sender }
56    }
57
58    /// Runs a job against the compiler state, resolving with its result once it's processed.
59    pub(crate) fn request<T, F>(&self, job: F) -> ActorResponse<T>
60    where
61        T: Send + 'static,
62        F: FnOnce(&mut LspState) -> Result<T, ResponseError> + Send + 'static,
63    {
64        let (tx, rx) = oneshot::channel();
65        self.send(ActorMessage::Job(Box::new(move |state| {
66            let _ = tx.send(job(state));
67        })));
68        ActorResponse(rx)
69    }
70
71    /// Enqueues a job that produces no reply.
72    pub(crate) fn notify<F>(&self, job: F)
73    where
74        F: FnOnce(&mut LspState) + Send + 'static,
75    {
76        self.send(ActorMessage::Job(Box::new(job)));
77    }
78
79    pub(crate) fn send(&self, message: ActorMessage) {
80        // If the actor thread is gone the message's reply channel (if any) is dropped with
81        // the message, so the corresponding request resolves with an error instead of hanging.
82        if self.sender.send(message).is_err() {
83            eprintln!("the compiler actor thread is gone; dropping message");
84        }
85    }
86}
87
88fn run_actor_loop(state: &mut LspState, receiver: mpsc::Receiver<ActorMessage>) {
89    while let Ok(message) = receiver.recv() {
90        // Drain whatever queued up while the previous message was being processed, so
91        // bursts of document changes (fast typing during a slow re-check) can be coalesced
92        // instead of type-checked one by one.
93        let mut batch = vec![message];
94        while let Ok(message) = receiver.try_recv() {
95            batch.push(message);
96        }
97
98        for message in coalesce_file_changes(batch) {
99            // A panicking job (e.g. an ICE during type-checking) must not take this thread
100            // down with it: that would leave every queued and future request unanswered.
101            // The unwind drops the job's reply channel, so its request resolves with an
102            // error, and the loop continues with the next message. Caches the panic left
103            // incomplete are rebuilt from sources the next time they are found missing.
104            let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
105                process_message(state, message);
106            }));
107            if result.is_err() {
108                eprintln!("a compiler actor job panicked; continuing with the next message");
109            }
110        }
111    }
112}
113
114/// Merges every run of consecutive `FileChanged` messages for the same document into the
115/// last change of the run, which carries the newest text (the server uses full-document
116/// sync). Only adjacent messages are merged, so a message ordered between two changes
117/// (e.g. a request) still observes the document text of the change that preceded it.
118fn coalesce_file_changes(batch: Vec<ActorMessage>) -> Vec<ActorMessage> {
119    let mut messages: Vec<ActorMessage> = Vec::with_capacity(batch.len());
120    for message in batch {
121        match (messages.last_mut(), message) {
122            (Some(ActorMessage::FileChanged(previous)), ActorMessage::FileChanged(next))
123                if previous.text_document.uri == next.text_document.uri =>
124            {
125                *previous = next;
126            }
127            (_, message) => messages.push(message),
128        }
129    }
130    messages
131}
132
133fn process_message(state: &mut LspState, message: ActorMessage) {
134    match message {
135        ActorMessage::Job(job) => job(state),
136        ActorMessage::FileChanged(params) => {
137            if let Err(error) = on_did_change_text_document(state, params) {
138                eprintln!("error processing document change: {error}");
139            }
140        }
141    }
142}

144/// The pending reply to a job sent with [`CompilerActor::request`].
145///
146/// Resolves with an error if the actor goes away without replying (for example if the job
147/// panicked), so a lost job surfaces as a failed request instead of a hang.
148pub(crate) struct ActorResponse<T>(oneshot::Receiver<Result<T, ResponseError>>);
149
150impl<T> Future for ActorResponse<T> {
151    type Output = Result<T, ResponseError>;
152
153    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
154        Pin::new(&mut self.0).poll(cx).map(|result| {
155            result.unwrap_or_else(|_| {
156                Err(ResponseError::new(
157                    ErrorCode::INTERNAL_ERROR,
158                    "the compiler actor did not reply to the request",
159                ))
160            })
161        })
162    }
163}

165#[cfg(test)]
166mod tests {
167    use super::*;
168    use acvm::blackbox_solver::StubbedBlackBoxSolver;
169    use async_lsp::lsp_types::{
170        TextDocumentContentChangeEvent, Url, VersionedTextDocumentIdentifier,
171    };
172
173    fn spawn_actor() -> CompilerActor {
174        CompilerActor::spawn(ClientSocket::new_closed(), StubbedBlackBoxSolver)
175    }
176
177    #[tokio::test]
178    async fn processes_messages_in_order() {
179        let actor = spawn_actor();
180
181        // A slow fire-and-forget job followed by a request: the request must observe the
182        // job's effect even though the job is still running when the request is enqueued.
183        actor.notify(|state| {
184            std::thread::sleep(std::time::Duration::from_millis(100));
185            state.options.enable_code_lens = false;
186        });
187        let enabled = actor.request(|state| Ok(state.options.enable_code_lens)).await.unwrap();
188        assert!(!enabled);
189    }
190
191    #[tokio::test]
192    async fn survives_a_panicking_job() {
193        let actor = spawn_actor();
194
195        let result =
196            actor.request(|_state| -> Result<i32, ResponseError> { panic!("job panicked") }).await;
197        assert!(result.is_err(), "a panicked request should resolve with an error, not hang");
198
199        let answer = actor.request(|_state| Ok(42)).await;
200        assert_eq!(answer.unwrap(), 42, "the actor should keep serving requests after a panic");
201    }
202
203    fn file_changed(uri: &Url, text: &str) -> ActorMessage {
204        ActorMessage::FileChanged(DidChangeTextDocumentParams {
205            text_document: VersionedTextDocumentIdentifier { uri: uri.clone(), version: 0 },
206            content_changes: vec![TextDocumentContentChangeEvent {
207                range: None,
208                range_length: None,
209                text: text.to_string(),
210            }],
211        })
212    }
213
214    /// Renders a message as `uri=text` for `FileChanged` and `job` for `Job`, so tests can
215    /// assert on whole batches at a glance.
216    fn describe(message: &ActorMessage) -> String {
217        match message {
218            ActorMessage::Job(_) => "job".to_string(),
219            ActorMessage::FileChanged(params) => {
220                format!("{}={}", params.text_document.uri, params.content_changes[0].text)
221            }
222        }
223    }
224
225    #[test]
226    fn coalesces_consecutive_changes_to_the_same_document() {
227        let one = Url::parse("file:///one.nr").unwrap();
228        let two = Url::parse("file:///two.nr").unwrap();
229
230        let batch = vec![
231            file_changed(&one, "a"),
232            file_changed(&one, "b"),
233            file_changed(&two, "c"),
234            file_changed(&one, "d"),
235        ];
236        let messages = coalesce_file_changes(batch);
237        let descriptions: Vec<_> = messages.iter().map(describe).collect();
238        assert_eq!(descriptions, vec!["file:///one.nr=b", "file:///two.nr=c", "file:///one.nr=d"]);
239    }
240
241    #[test]
242    fn does_not_coalesce_changes_across_other_messages() {
243        let one = Url::parse("file:///one.nr").unwrap();
244
245        let batch = vec![
246            file_changed(&one, "a"),
247            ActorMessage::Job(Box::new(|_state| ())),
248            file_changed(&one, "b"),
249        ];
250        let messages = coalesce_file_changes(batch);
251        let descriptions: Vec<_> = messages.iter().map(describe).collect();
252        assert_eq!(descriptions, vec!["file:///one.nr=a", "job", "file:///one.nr=b"]);
253    }
