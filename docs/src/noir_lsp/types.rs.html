206#[derive(Debug, Clone, Serialize, Deserialize)]
207#[serde(try_from = "String", into = "String")]
208pub(crate) struct NargoTestId {
209    package: CrateName,
210    fully_qualified_path: String,
211}
212
213impl TryFrom<String> for NargoTestId {
214    type Error = String;
215
216    fn try_from(value: String) -> Result<Self, Self::Error> {
217        if let Some((crate_name, function_name)) = value.split_once('/') {
218            let crate_name = crate_name.parse()?;
219            Ok(Self { package: crate_name, fully_qualified_path: function_name.to_string() })
220        } else {
221            Err("NargoTestId should be serialized as package_name/fully_qualified_path".to_string())
222        }
223    }
224}
225
226impl From<NargoTestId> for String {
227    fn from(value: NargoTestId) -> Self {
228        format!("{}/{}", value.package, value.fully_qualified_path)
229    }
230}
231
232impl NargoTestId {
233    pub(crate) fn new(crate_name: CrateName, function_name: String) -> Self {
234        Self { package: crate_name, fully_qualified_path: function_name }
235    }
236
237    pub(crate) fn crate_name(&self) -> &CrateName {
238        &self.package
239    }
240
241    pub(crate) fn function_name(&self) -> &String {
242        &self.fully_qualified_path
243    }
244}
