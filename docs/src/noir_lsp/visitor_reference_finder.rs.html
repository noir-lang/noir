27/// Traverses an AST to find a ReferenceId at the given byte index.
28/// This searches:
29/// - references in doc comment links
30/// - attribute references (see `visit_meta_attribute`).
31pub(crate) struct VisitorReferenceFinder<'a> {
32    source: &'a str,
33    file_id: FileId,
34    byte_index: usize,
35    /// The module ID in scope. This might change as we traverse the AST
36    /// if we are analyzing something inside an inline module declaration.
37    module_id: ModuleId,
38    args: &'a ProcessRequestCallbackArgs<'a>,
39    link_finder: LinkFinder,
40
41    /// The found ReferenceId, if any, along with an LSP location that covers
42    /// the range of the text that points to that reference (None if the range
43    /// should be the word under the cursor, or Some if the range is not that word,
44    /// which is the case of doc comment links where the entire `[...]` is the range)
45    reference_id: Option<(ReferenceId, Option<lsp_types::Location>)>,
46}
47
48impl<'a> VisitorReferenceFinder<'a> {
49    #[allow(clippy::too_many_arguments)]
50    pub(crate) fn new(
51        file_id: FileId,
52        source: &'a str,
53        byte_index: usize,
54        args: &'a ProcessRequestCallbackArgs<'a>,
55    ) -> Self {
56        // Find the module the current file belongs to
57        let krate = args.crate_id;
58        let def_map = &args.def_maps[&krate];
59        let local_id = if let Some((module_index, _)) =
60            def_map.modules().iter().find(|(_, module_data)| module_data.location.file == file_id)
61        {
62            LocalModuleId::new(module_index)
63        } else {
64            def_map.root()
65        };
66        let module_id = ModuleId { krate, local_id };
67        let link_finder = LinkFinder::default();
68        Self { source, file_id, byte_index, module_id, args, link_finder, reference_id: None }
69    }
70
71    pub(crate) fn find(
72        &mut self,
73        parsed_module: &ParsedModule,
74    ) -> Option<(ReferenceId, Option<lsp_types::Location>)> {
75        // Find in the doc comments on the crate root, if we are in the crate root
76        self.find_in_reference_doc_comments(ReferenceId::Module(self.module_id));
77
78        parsed_module.accept(self);
79
80        std::mem::take(&mut self.reference_id)
81    }
82
83    /// Checks if the cursor is on a link inside the doc comments of the given ReferenceId.
84    fn find_in_reference_doc_comments(&mut self, id: ReferenceId) {
85        let Some(doc_comments) = self.args.interner.doc_comments(id) else {
86            return;
87        };
88
89        if !doc_comments.iter().any(|doc_comment| self.intersects_span(doc_comment.span())) {
90            return;
91        }
92
93        let Some((current_module_id, current_type)) = current_module_and_type(id, self.args) else {
94            return;
95        };
96
97        let Some(byte_lsp_location) = to_lsp_location(
98            self.args.files,
99            self.args.location.file,
100            Span::single_char(self.byte_index as u32),
101        ) else {
102            return;
103        };
104
105        self.link_finder.reset();
106        for located_comment in doc_comments {
107            let location = located_comment.location();
108            if location.file != self.file_id {
109                // A module's comments might happen inline in the same file or in a different file.
110                // We should not process comments that are not in the current file.
111                continue;
112            }
113
114            let Some(lsp_location) = to_lsp_location(self.args.files, location.file, location.span)
115            else {
116                continue;
117            };
118            let start_line = lsp_location.range.start.line;
119            let start_char = lsp_location.range.start.character;
120
121            // Read comments from source based on location: the comments in `located_comment` might
122            // have been slightly adjusted.
123            let comments =
124                &self.source[location.span.start() as usize..location.span.end() as usize];
125
126            let links = self.link_finder.find_links(
127                comments,
128                current_module_id,
129                current_type,
130                self.args.interner,
131                self.args.def_maps,
132                self.args.crate_graph,
133            );
134            for link in links {
135                let Some(target) = link.target else {
136                    continue;
137                };
138
139                let line = start_line + link.line as u32;
140                let start =
141                    if link.line == 0 { start_char + link.start as u32 } else { link.start as u32 };
142                let length = (link.end - link.start) as u32;
143                let end = start + length;
144                if byte_lsp_location.range.start.line == line
145                    && start <= byte_lsp_location.range.start.character
146                    && byte_lsp_location.range.start.character <= end
147                {
148                    let reference = match target {
149                        LinkTarget::TopLevelItem(module_def_id) => {
150                            module_def_id_to_reference_id(module_def_id)
151                        }
152                        LinkTarget::Method(_, func_id)
153                        | LinkTarget::PrimitiveTypeFunction(_, func_id) => {
154                            ReferenceId::Function(func_id)
155                        }
156                        LinkTarget::StructMember(type_id, index) => {
157                            ReferenceId::StructMember(type_id, index)
158                        }
159                        LinkTarget::PrimitiveType(_) => {
160                            continue;
161                        }
162                    };
163                    let location = lsp_types::Location {
164                        uri: byte_lsp_location.uri.clone(),
165                        range: lsp_types::Range {
166                            start: lsp_types::Position { line, character: start },
167                            end: lsp_types::Position { line, character: end },
168                        },
169                    };
170
171                    self.reference_id = Some((reference, Some(location)));
172                    return;
173                }
174            }
175        }
176    }
177
178    fn intersects_span(&self, span: Span) -> bool {
179        span.start() as usize <= self.byte_index && self.byte_index <= span.end() as usize
180    }
181}

183impl Visitor for VisitorReferenceFinder<'_> {
184    fn visit_parsed_submodule(&mut self, parsed_sub_module: &ParsedSubModule, _span: Span) -> bool {
185        let name_location = parsed_sub_module.name.location();
186        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
187            self.find_in_reference_doc_comments(reference);
188        };
189
190        // Switch `self.module_id` to the submodule
191        let previous_module_id = self.module_id;
192
193        let def_map = &self.args.def_maps[&self.module_id.krate];
194        if let Some(module_data) = def_map.get(self.module_id.local_id)
195            && let Some(child_module) = module_data.children.get(&parsed_sub_module.name)
196        {
197            self.module_id = ModuleId { krate: self.module_id.krate, local_id: *child_module };
198        }
199
200        parsed_sub_module.accept_children(self);
201
202        // Restore the old module before continuing
203        self.module_id = previous_module_id;
204
205        false
206    }

208    fn visit_noir_function(&mut self, function: &NoirFunction, span: Span) -> bool {
209        let name_location = function.name_ident().location();
210        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
211            self.find_in_reference_doc_comments(reference);
212        };
213
214        self.intersects_span(span)
215    }

217    fn visit_noir_struct(&mut self, noir_struct: &NoirStruct, _: Span) -> bool {
218        let name_location = noir_struct.name.location();
219        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
220            self.find_in_reference_doc_comments(reference);
221        };
222
223        for field in noir_struct.fields.iter() {
224            let field_name_location = field.item.name.location();
225            if let Some(reference) = self.args.interner.reference_at_location(field_name_location) {
226                self.find_in_reference_doc_comments(reference);
227            };
228        }
229
230        false
231    }

233    fn visit_noir_enum(&mut self, noir_enum: &NoirEnumeration, _: Span) -> bool {
234        let name_location = noir_enum.name.location();
235        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
236            self.find_in_reference_doc_comments(reference);
237        };
238
239        for variant in noir_enum.variants.iter() {
240            let variant_name_location = variant.item.name.location();
241            if let Some(reference) = self.args.interner.reference_at_location(variant_name_location)
242            {
243                self.find_in_reference_doc_comments(reference);
244            };
245        }
246
247        false
248    }

250    fn visit_noir_trait(&mut self, noir_trait: &NoirTrait, _: Span) -> bool {
251        let name_location = noir_trait.name.location();
252        if let Some(reference) = self.args.interner.reference_at_location(name_location) {
253            self.find_in_reference_doc_comments(reference);
254        };
255
256        for item in noir_trait.items.iter() {
257            if let TraitItem::Function { name, .. } = &item.item {
258                let func_name_location = name.location();
259                if let Some(reference) =
260                    self.args.interner.reference_at_location(func_name_location)
261                {
262                    self.find_in_reference_doc_comments(reference);
263                };
264            }
265        }
266
267        true
268    }

286    /// If the cursor is on an custom attribute, try to resolve its
287    /// underlying function and return a ReferenceId to it.
288    /// This is needed in hover and go-to-definition because when an annotation generates
289    /// code, that code ends up residing in the attribute definition (it ends up having the
290    /// attribute's span) so using the usual graph to locate what points to that location
291    /// will give not only the attribute function but also any type generated by it.use fm::FileId;
292    fn visit_meta_attribute(
293        &mut self,
294        attribute: &MetaAttribute,
295        _target: AttributeTarget,
296        span: Span,
297    ) -> bool {
298        if !self.intersects_span(span) {
299            return false;
300        }
301
302        let MetaAttributeName::Path(path) = attribute.name.clone() else {
303            return false;
304        };
305
306        // The path here must resolve to a function and it's a simple path (can't have turbofish)
307        // so it can (and must) be solved as an import.
308        let Ok(Some((module_def_id, _, _))) = resolve_import(
309            path,
310            self.module_id,
311            self.args.def_maps,
312            &mut UsageTracker::default(),
313            None, // references tracker
314        )
315        .map(|result| result.namespace.values) else {
316            return true;
317        };
318
319        self.reference_id = Some((module_def_id_to_reference_id(module_def_id), None));
320
321        true
322    }
