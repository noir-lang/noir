18pub(crate) fn position_to_byte_index(
19    files: &FileMap,
20    file_id: FileId,
21    position: &async_lsp::lsp_types::Position,
22) -> Option<usize> {
23    let Ok(source) = files.source(file_id) else {
24        return None;
25    };
26
27    let Ok(line_span) = files.line_range(file_id, position.line as usize) else {
28        return None;
29    };
30    let line_str = source.get(line_span.clone())?;
31
32    let byte_offset = character_to_line_offset(line_str, position.character)?;
33
34    Some(line_span.start + byte_offset)
35}

37pub(crate) fn character_to_line_offset(line: &str, character: u32) -> Option<usize> {
38    let line_len = line.len();
39    let mut character_offset = 0;
40
41    let mut chars = line.chars();
42    while let Some(ch) = chars.next() {
43        if character_offset == character {
44            let chars_off = chars.as_str().len();
45            let ch_off = ch.len_utf8();
46
47            return Some(line_len - chars_off - ch_off);
48        }
49
50        character_offset += ch.len_utf16() as u32;
51    }
52
53    // Handle positions after the last character on the line
54    if character_offset == character { Some(line_len) } else { None }
55}

57/// Given a string with a single ">|<" (cursor) in it, returns:
58/// 1. The line where the cursor is (zero-based)
59/// 2. The column where the cursor is (zero-based)
60/// 3. that string with ">|<" removed
61#[cfg(test)]
62pub(crate) fn get_cursor_line_and_column(src: &str) -> (usize, usize, String) {
63    let (line, column) = src
64        .lines()
65        .enumerate()
66        .find_map(|(line_index, line)| line.find(">|<").map(|char_index| (line_index, char_index)))
67        .expect("Expected to find one >|< in the source code");
68
69    let src = src.replace(">|<", "");
70    (line, column, src)
71}
