29// Global atomic counter used to generate unique sequential IDs for testcases
30static NEXT_TESTCASE_ID: AtomicU64 = AtomicU64::new(0xdead);
31
32// Type alias for testcase IDs which are 64-bit integers
33pub type TestCaseId = u64;
34
35/// Generates a unique sequential ID for a testcase by atomically incrementing a global counter
36fn generate_testcase_id() -> TestCaseId {
37    NEXT_TESTCASE_ID.fetch_add(1, Ordering::SeqCst)
38}

40/// Represents a single test case in the corpus, containing both the input values and a unique identifier
41pub struct TestCase<'a> {
42    // The actual input values for this test case
43    value: &'a InputMap,
44    // Unique identifier for this test case
45    id: TestCaseId,
46}
47
48impl<'a> TestCase<'a> {
49    /// Returns a reference to the input values for this test case
50    pub fn value(&self) -> &InputMap {
51        self.value
52    }
53
54    /// Returns the unique identifier for this test case
55    pub fn id(&self) -> TestCaseId {
56        self.id
57    }
58
59    /// Creates a new TestCase with a specified ID and input values
60    pub fn with_id(id: TestCaseId, value: &'a InputMap) -> Self {
61        Self { id, value }
62    }
63}
64
65impl<'a> From<&'a InputMap> for TestCase<'a> {
66    /// Converts an InputMap into a TestCase by generating a new unique ID
67    fn from(value: &'a InputMap) -> Self {
68        Self { value, id: generate_testcase_id() }
69    }
70}

72/// Manages reading and writing corpus files to the filesystem, including parsing JSON inputs
73pub struct CorpusFileManager {
74    // Manages file operations and tracks file metadata
75    file_manager: FileManager,
76    // Path to the directory containing corpus files
77    corpus_path: PathBuf,
78    // ABI definition used for parsing inputs
79    abi: Abi,
80    // Maps file IDs to their parsed input values
81    parsed_map: HashMap<FileId, InputMap>,
82}
83
84impl CorpusFileManager {
85    /// Creates a new CorpusFileManager for a specific package and test harness
86    pub fn new(root: &Path, package_name: &str, harness_name: &str, abi: Abi) -> Self {
87        let corpus_path = root.join(package_name).join(harness_name);
88
89        Self { file_manager: FileManager::new(root), corpus_path, abi, parsed_map: HashMap::new() }
90    }
91
92    /// Loads all corpus files from disk into memory, parsing them as JSON inputs.
93    /// Creates the corpus directory if it doesn't exist.
94    /// Returns an error if directory creation fails or if any files can't be parsed.
95    pub fn load_corpus_from_disk(&mut self) -> Result<(), String> {
96        let mut builder = DirBuilder::new();
97        match builder.recursive(true).create(&self.corpus_path) {
98            Ok(_) => (),
99            Err(_) => {
100                return Err(format!(
101                    "Couldn't create directory {:?}",
102                    self.corpus_path.as_os_str()
103                ));
104            }
105        }
106        // Go through all files in the corpus directory recursively
107        for entry in WalkDir::new(&self.corpus_path) {
108            let Ok(entry) = entry else {
109                continue;
110            };
111            if !entry.file_type().is_file() {
112                continue;
113            }
114            if entry.path().extension().is_none_or(|ext| ext != CORPUS_FILE_EXTENSION) {
115                continue;
116            };
117            let path = entry.into_path();
118            // Skip files that are already loaded
119            if self.file_manager.has_file(&path) {
120                continue;
121            }
122            let source = std::fs::read_to_string(path.as_path())
123                .unwrap_or_else(|_| panic!("could not read file {path:?} into string"));
124
125            let parsed_source = parse_json(&source, &self.abi).map_err(|parsing_error| {
126                format!("Error while parsing file {:?}: {:?}", path.as_os_str(), parsing_error)
127            })?;
128
129            // Add the file and its parsed contents to our tracking maps
130            let file_id = self.file_manager.add_file_with_source(path.as_path(), source).unwrap();
131            self.parsed_map.insert(file_id, parsed_source);
132        }
133        Ok(())
134    }
135
136    /// Returns a vector containing all parsed inputs from the corpus
137    pub fn get_full_corpus(&self) -> Vec<&InputMap> {
138        let file_ids = self.file_manager.as_file_map().all_file_ids();
139        file_ids.map(|file_id| &self.parsed_map[file_id]).collect()
140    }
141
142    /// Returns the path to the corpus directory
143    pub fn get_corpus_path(&self) -> &Path {
144        &self.corpus_path
145    }
146
147    /// Saves a new testcase to disk and adds it to the file manager.
148    /// The filename is generated from the SHA-256 hash of the contents.
149    /// Returns an error if file creation or writing fails.
150    pub fn save_testcase_to_disk(&mut self, contents: &str) -> Result<(), String> {
151        let mut builder = DirBuilder::new();
152        match builder.recursive(true).create(&self.corpus_path) {
153            Ok(_) => (),
154            Err(_) => {
155                return Err(format!(
156                    "Couldn't create directory {:?}",
157                    self.corpus_path.as_os_str()
158                ));
159            }
160        }
161        let file_name = Path::new(&digest(contents)).with_extension(CORPUS_FILE_EXTENSION);
162        let full_file_path = self.corpus_path.join(file_name);
163        let mut file = File::create(&full_file_path)
164            .map_err(|_x| format!("Couldn't create file {:?}", &full_file_path))?;
165
166        file.write_all(contents.as_bytes())
167            .map_err(|_x| format!("Couldn't write to file {:?}", &full_file_path))?;
168        self.file_manager.add_file_with_source(&full_file_path, String::from(contents));
169        Ok(())
170    }
171}

173/// Tracks information about a sequence of executions for a particular testcase,
174/// including how many executions remain and which testcase is being used
175#[derive(Debug)]
176struct Sequence {
177    // ID of the testcase being executed in sequence
178    testcase_id: TestCaseId,
179    // Number of remaining executions in this sequence
180    executions_left: u64,
181}
182
183impl Sequence {
184    /// Creates a new empty sequence
185    pub fn new() -> Self {
186        Self { testcase_id: 0, executions_left: 0 }
187    }
188
189    /// Returns true if there are no executions remaining in this sequence
190    pub fn is_empty(&self) -> bool {
191        self.executions_left == 0
192    }
193
194    /// Resets the sequence by setting remaining executions to 0
195    pub fn clear(&mut self) {
196        self.executions_left = 0;
197    }
198
199    /// Decrements the number of remaining executions by 1
200    pub fn decrement(&mut self) {
201        self.executions_left -= 1;
202    }
203}

205/// Manages the selection of testcases for fuzzing, tracking execution counts and scheduling
206pub struct TestCaseOrchestrator {
207    // Maps testcase IDs to their total execution count
208    executions_per_testcase: HashMap<TestCaseId, u64>,
209    // Maps testcase IDs to their sequence number (how many times they've been selected for sequential fuzzing)
210    sequence_number: HashMap<TestCaseId, u32>,
211    // Total number of executions across all testcases
212    total_executions: u64,
213    // Currently active sequence of executions
214    current_sequence: Sequence,
215}
216
217/// Represents the next testcase(s) selected for fuzzing:
218/// - First element is the primary testcase ID
219/// - Second optional element is an additional testcase ID for splicing mutations
220type NextSelection = (TestCaseId, Option<TestCaseId>);
221
222impl TestCaseOrchestrator {
223    /// Creates a new empty TestCaseOrchestrator
224    pub fn new() -> Self {
225        Self {
226            executions_per_testcase: HashMap::new(),
227            sequence_number: HashMap::new(),
228            total_executions: 0,
229            current_sequence: Sequence::new(),
230        }
231    }
232
233    /// Adds a new testcase to be managed by the orchestrator
234    pub fn new_testcase(&mut self, testcase_id: TestCaseId) {
235        self.executions_per_testcase.insert(testcase_id, 0);
236        self.sequence_number.insert(testcase_id, 0);
237    }
238
239    /// Removes a testcase from being managed by the orchestrator.
240    /// Updates execution counts and clears current sequence if it was using this testcase.
241    pub fn remove(&mut self, testcase_id: TestCaseId) {
242        let executions = self.executions_per_testcase[&testcase_id];
243        self.executions_per_testcase.remove(&testcase_id);
244        self.sequence_number.remove(&testcase_id);
245        self.total_executions -= executions;
246        if self.current_sequence.testcase_id == testcase_id {
247            self.current_sequence.clear();
248        }
249    }
250
251    /// Selects the next testcase(s) to use for fuzzing based on execution counts.
252    /// Prioritizes testcases that have been executed least often.
253    /// Returns both a primary testcase and optionally a second one for splicing.
254    pub fn get_next_testcase(&mut self, prng: &mut XorShiftRng) -> NextSelection {
255        let testcase_count = self.executions_per_testcase.len();
256        // If we're in the middle of a sequence, continue using the current testcase
257        if !self.current_sequence.is_empty() {
258            // Update execution counts
259            self.current_sequence.decrement();
260        } else {
261            // Calculate average executions per testcase
262            let average = self.total_executions / testcase_count as u64;
263
264            // Select the testcase with the lowest execution count
265            let chosen_id = self
266                .executions_per_testcase
267                .iter()
268                .min_by_key(|&(_id, value)| value)
269                .map(|(&id, _)| id)
270                .unwrap();
271
272            // Increment sequence number for chosen testcase
273            self.sequence_number
274                .entry(chosen_id)
275                .and_modify(|sequence_number| *sequence_number += 1);
276
277            // Create new sequence with exponentially increasing length based on sequence number
278            self.current_sequence = Sequence {
279                testcase_id: chosen_id,
280                executions_left: min(1u64 << self.sequence_number[&chosen_id], average / 2),
281            };
282        }
283        // Update execution counts
284        self.executions_per_testcase
285            .entry(self.current_sequence.testcase_id)
286            .and_modify(|executions| *executions += 1);
287        self.total_executions += 1;
288        // If we have multiple testcases, randomly select a different one for splicing
289        if testcase_count > 1 {
290            let mut additional_id = self.current_sequence.testcase_id;
291            // Keep sampling until we get a different testcase
292            while additional_id == self.current_sequence.testcase_id {
293                additional_id =
294                    self.executions_per_testcase.iter().map(|(&id, _)| id).choose(prng).unwrap();
295            }
296            (self.current_sequence.testcase_id, Some(additional_id))
297        } else {
298            (self.current_sequence.testcase_id, None)
299        }
300    }
301}

303/// Main corpus manager that tracks all discovered testcases and orchestrates their selection for fuzzing
304pub struct Corpus {
305    // Currently active testcases used for fuzzing
306    discovered_testcases: HashMap<TestCaseId, InputMap>,
307
308    // Testcases that are stored but not actively used in fuzzing
309    cached_testcases: HashMap<TestCaseId, InputMap>,
310
311    // Orchestrator for selecting testcases for Brillig execution
312    brillig_orchestrator: TestCaseOrchestrator,
313
314    // Orchestrator for selecting testcases for ACIR execution
315    acir_orchestrator: TestCaseOrchestrator,
316
317    // Manages reading/writing corpus files to disk
318    corpus_file_manager: CorpusFileManager,
319}

321impl Corpus {
322    /// Creates a new empty corpus for a specific package and function
323    pub fn new(base_folder: &Path, package_name: &str, function_name: &str, abi: &Abi) -> Self {
324        Self {
325            discovered_testcases: HashMap::new(),
326            cached_testcases: HashMap::new(),
327            brillig_orchestrator: TestCaseOrchestrator::new(),
328            acir_orchestrator: TestCaseOrchestrator::new(),
329            corpus_file_manager: CorpusFileManager::new(
330                base_folder,
331                package_name,
332                function_name,
333                abi.clone(),
334            ),
335        }
336    }

338    /// Attempts to load existing corpus files from disk

340        self.corpus_file_manager.load_corpus_from_disk()
341    }
342
343    /// Returns all testcases that have ever been added to the corpus, including cached ones
344    pub fn get_full_stored_corpus(&mut self) -> Vec<TestCase> {
345        let stored_corpus: Vec<_> =
346            self.corpus_file_manager.get_full_corpus().into_iter().map(TestCase::from).collect();
347        let id_testcase_pair: Vec<_> =
348            stored_corpus.iter().map(|x| (x.id(), x.value().clone())).collect();
349        for (id, input_map) in id_testcase_pair.iter() {
350            self.insert_into_cache(*id, input_map.clone());
351        }
352        id_testcase_pair
353            .iter()
354            .map(|(id, _)| TestCase::with_id(*id, &self.cached_testcases[id]))
355            .collect()
356    }
357
358    /// Returns the path where corpus files are stored
359    pub fn get_corpus_storage_path(&self) -> &Path {
360        self.corpus_file_manager.get_corpus_path()
361    }
362
363    /// Adds a testcase to the cache (stored but not active)
364    fn insert_into_cache(&mut self, testcase_id: TestCaseId, new_testcase_value: InputMap) {
365        self.cached_testcases.insert(testcase_id, new_testcase_value);
366    }
367
368    /// Adds a new testcase to the active corpus.
369    /// Optionally saves it to disk and updates orchestrators.
370    pub fn insert(
371        &mut self,
372        testcase_id: TestCaseId,
373        new_testcase_value: InputMap,
374        save_to_disk: bool,
375    ) -> Result<TestCaseId, String> {
376        if save_to_disk {
377            self.corpus_file_manager.save_testcase_to_disk(
378                &serialize_to_json(&new_testcase_value, &self.corpus_file_manager.abi)
379                    .expect("Shouldn't be any issues with serializing input map"),
380            )?;
381        }
382        self.brillig_orchestrator.new_testcase(testcase_id);
383        self.acir_orchestrator.new_testcase(testcase_id);
384        self.discovered_testcases.insert(testcase_id, new_testcase_value.clone());
385        Ok(testcase_id)
386    }
387
388    /// Removes a testcase from the active corpus and both orchestrators
389    pub fn remove(&mut self, testcase_id: TestCaseId) {
390        self.brillig_orchestrator.remove(testcase_id);
391        self.acir_orchestrator.remove(testcase_id);
392        self.discovered_testcases.remove(&testcase_id);
393    }
394
395    /// Returns the input values for a testcase by its ID, checking both active and cached testcases
396    pub fn get_testcase_by_id(&self, id: TestCaseId) -> &InputMap {
397        self.discovered_testcases.get(&id).unwrap_or_else(|| &self.cached_testcases[&id])
398    }

400    /// Selects the next testcase(s) for ACIR execution
401    pub fn get_next_testcase_for_acir(&mut self, prng: &mut XorShiftRng) -> NextSelection {

403    }

405    /// Selects the next testcase(s) for Brillig execution
406    pub fn get_next_testcase_for_brillig(&mut self, prng: &mut XorShiftRng) -> NextSelection {
407        self.brillig_orchestrator.get_next_testcase(prng)
408    }

410    /// Returns the number of testcases currently active in the corpus

412        self.discovered_testcases.len()
413    }
