23    /// Create a spliced version of 2 buffers, where each element in the result is at the same index as in the original ones
24    fn structured_splice(
25        &mut self,
26        first_buffer: &[InputValue],
27        second_buffer: &[InputValue],
28    ) -> Vec<InputValue> {
29        let mut result = first_buffer.to_vec();
30        let mut index = 0;
31        let buffer_length = first_buffer.len();
32        while index != buffer_length {
33            // Pick the length of the sequence from 1 to maximum available
34            let sequence_length = self.prng.random_range(1..=(buffer_length - index));
35
36            // If first buffer is selected for the chunk, do nothing (we already have that part in the result)
37            // If the second is selected, copy the chunk into result
38            match BASIC_SPLICE_CANDIDATE_PRIORITIZATION_CONFIGURATION.select(self.prng) {
39                SpliceCandidateOptions::First => {}
40                SpliceCandidateOptions::Second => {
41                    result.splice(
42                        index..(index + sequence_length),
43                        second_buffer[index..(index + sequence_length)].iter().cloned(),
44                    );
45                }
46            }
47            index += sequence_length;
48        }
49        result
50    }

52    /// Create buffer from random chunks of 2 buffers
53    fn chaotic_splice(
54        &mut self,
55        first_buffer: &[InputValue],
56        second_buffer: &[InputValue],
57    ) -> Vec<InputValue> {
58        let mut result = Vec::new();
59        let mut index = 0;
60        let buffer_length = first_buffer.len();
61        while index != buffer_length {
62            // Pick the length of the sequence from 1 to maximum available
63            let sequence_length = self.prng.random_range(1..=(buffer_length - index));
64
65            let source_position = self.prng.random_range(0..=(buffer_length - sequence_length));
66            // If first buffer is selected for the chunk, do nothing (we already have that part in the result)
67            // If the second is selected, copy the chunk into result
68            match BASIC_SPLICE_CANDIDATE_PRIORITIZATION_CONFIGURATION.select(self.prng) {
69                SpliceCandidateOptions::First => {
70                    result.extend_from_slice(
71                        &first_buffer[source_position..(source_position + sequence_length)],
72                    );
73                }
74                SpliceCandidateOptions::Second => {
75                    result.extend_from_slice(
76                        &second_buffer[source_position..(source_position + sequence_length)],
77                    );
78                }
79            }
80            index += sequence_length;
81        }
82        result
83    }

85    /// Generate a combination of two string inputs
86    pub fn splice(&mut self, first_input: &InputValue, second_input: &InputValue) -> InputValue {
87        let first_buffer = match first_input {
88            InputValue::Vec(inner_vec) => inner_vec,
89            _ => panic!("Shouldn't be used with other input value types"),
90        };
91
92        let second_buffer = match second_input {
93            InputValue::Vec(inner_vec) => inner_vec,
94            _ => panic!("Shouldn't be used with other input value types"),
95        };
96
97        assert!(!first_buffer.is_empty());
98        assert!(second_buffer.len() == first_buffer.len());
99
100        let result = match BASIC_SPLICE_MUTATION_CONFIGURATION.select(self.prng) {
101            SpliceMutationOptions::PositionPreserving => {
102                self.structured_splice(first_buffer, second_buffer)
103            }
104            SpliceMutationOptions::RandomChunks => self.chaotic_splice(first_buffer, second_buffer),
105        };
106        InputValue::Vec(result)
107    }

109    /// Perform one of structural mutations on the vector of input values
110    pub fn perform_structure_mutation_on_vector(
111        &mut self,
112        input_buffer: &[InputValue],
113    ) -> Vec<InputValue> {
114        match BASIC_VECTOR_STRUCTURE_MUTATION_CONFIGURATION.select(self.prng) {
115            StructuralMutationOptions::ChaoticSelfSplice => {
116                self.chaotic_splice(input_buffer, input_buffer)
117            }
118            StructuralMutationOptions::ChunkDuplication => self.duplicate_chunk(input_buffer),
119            StructuralMutationOptions::Swap => self.swap(input_buffer),
120            StructuralMutationOptions::RandomValueDuplication => {
121                panic!("Vector mutations should have a value duplication weight of zero")
122            }
123        }
124    }

126    /// Swap 2 random chunks in the buffer
127    fn swap(&mut self, buffer: &[InputValue]) -> Vec<InputValue> {
128        let mut result = Vec::new();
129        let buffer_length = buffer.len();
130
131        // We need to leave at least the last byte for the second chunk
132        let first_chunk_position = self.prng.random_range(0..(buffer_length - 1));
133
134        // The second chunk starts after the first
135        let second_chunk_position =
136            self.prng.random_range((first_chunk_position + 1)..buffer_length);
137
138        let first_chunk_end =
139            self.prng.random_range((first_chunk_position + 1)..=second_chunk_position);
140
141        let second_chunk_end = self.prng.random_range((second_chunk_position + 1)..=buffer_length);
142
143        // Leave the start in place
144        result.extend_from_slice(&buffer[0..first_chunk_position]);
145
146        // Insert second chunk
147        result.extend_from_slice(&buffer[second_chunk_position..(second_chunk_end)]);
148
149        // Insert what's in between the chunks
150        result.extend_from_slice(&buffer[first_chunk_end..(second_chunk_position)]);
151
152        // Insert first chunk
153        result.extend_from_slice(&buffer[first_chunk_position..first_chunk_end]);
154
155        // Insert the tail
156        result.extend_from_slice(&buffer[second_chunk_end..buffer_length]);
157
158        result
159    }

161    /// Take a random chunk of the input and insert it several times into the input
162    fn duplicate_chunk(&mut self, input_buffer: &[InputValue]) -> Vec<InputValue> {
163        let mut result = input_buffer.to_vec();
164        let buffer_length = input_buffer.len();
165        // The maximum length of the chunk is half the total length
166        let maximum_chunk_length = buffer_length / 2;
167
168        // Get a random position for the chunk
169        let chunk_position = self.prng.random_range(0..=buffer_length - 1);
170
171        // Pick size
172        let chunk_size =
173            self.prng.random_range(1..=min(buffer_length - chunk_position, maximum_chunk_length));
174
175        // Find an insertion position with enough space
176        let insertion_position = self.prng.random_range(0..(buffer_length - chunk_size));
177
178        // Determine how many times to repeat
179        let maximum_insertion_count = (buffer_length - insertion_position) / chunk_size;
180        let insertion_count = self.prng.random_range(0..=maximum_insertion_count);
181        for i in 0..insertion_count {
182            result.splice(
183                (insertion_position + i * chunk_size)..(insertion_position + (i + 1) * chunk_size),
184                input_buffer[chunk_position..(chunk_position + chunk_size)].iter().cloned(),
185            );
186        }
187        result
188    }

191pub fn splice_array_structure(
192    first_input: &InputValue,
193    second_input: &InputValue,
194    prng: &mut XorShiftRng,
195) -> InputValue {
196    let mut array_splicer = ArrayMutator::new(prng);

200pub fn mutate_vector_structure(input: &[InputValue], prng: &mut XorShiftRng) -> Vec<InputValue> {
201    let mut array_mutator = ArrayMutator::new(prng);
202    array_mutator.perform_structure_mutation_on_vector(input)

206mod tests {
207    use super::*;
208    use acvm::{AcirField, FieldElement};
209    use noirc_abi::input_parser::InputValue;
210    use rand::SeedableRng;
211    use rand_xorshift::XorShiftRng;

213    #[test]
214    fn test_swap() {
215        // Create a deterministic PRNG for testing
216        let seed = [42u8; 16];
217        let mut prng = XorShiftRng::from_seed(seed);
218
219        // Create a test array mutator
220        let mut array_mutator = ArrayMutator::new(&mut prng);
221
222        // Create a test buffer with distinct values for easy verification
223        let buffer: Vec<InputValue> =
224            (0..10).map(|i| InputValue::Field(FieldElement::from(i as u128))).collect();
225
226        // Perform the swap operation
227        let result = array_mutator.swap(&buffer);
228
229        // Verify the result has the same length as the input
230        assert_eq!(result.len(), buffer.len());
231
232        // Verify that all elements from the original buffer are present in the result
233        // (though potentially in a different order)
234        let mut original_elements: Vec<u128> = buffer
235            .iter()
236            .map(|v| match v {
237                InputValue::Field(i) => i.to_u128(),
238                _ => panic!("Unexpected input value type"),
239            })
240            .collect();
241
242        let mut result_elements: Vec<u128> = result
243            .iter()
244            .map(|v| match v {
245                InputValue::Field(i) => i.to_u128(),
246                _ => panic!("Unexpected input value type"),
247            })
248            .collect();
249
250        original_elements.sort();
251        result_elements.sort();
252
253        assert_eq!(original_elements, result_elements);
254
255        // Verify that the order has changed (the swap actually did something)
256        // This might rarely fail if the random swap happens to produce the same order,
257        // but with our seed it should be consistent
258        assert_ne!(
259            buffer
260                .iter()
261                .map(|v| match v {
262                    InputValue::Field(i) => i.to_u128(),
263                    _ => panic!("Unexpected input value type"),
264                })
265                .collect::<Vec<_>>(),
266            result
267                .iter()
268                .map(|v| match v {
269                    InputValue::Field(i) => i.to_u128(),
270                    _ => panic!("Unexpected input value type"),
271                })
272                .collect::<Vec<_>>()
273        );
274    }
