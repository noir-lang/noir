56/// Compile a monomorphized [Program] into circuit using the given SSA passes or panic.
57///
58/// If there is a seed in the environment, then it prints the AST when an error is encountered.
59pub fn compile_into_circuit_with_ssa_passes_or_die(
60    program: Program,
61    options: &SsaEvaluatorOptions,
62    primary: &[SsaPass],
63    msg: Option<&str>,
64) -> SsaProgramArtifact {
65    // If we are using a seed, we are probably trying to reproduce some failure;
66    // in that case let's clone the program for printing if it panicked here,
67    // otherwise try to keep things faster by not cloning.
68    let for_print = std::env::var("NOIR_AST_FUZZER_SEED").is_ok().then(|| program.clone());
69
70    // We expect the programs generated should compile, but sometimes SSA passes panic, or return an error.
71    // Turn everything into a panic, catch it, print the AST, then resume panicking.
72    let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
73        ssa::create_program_with_passes(program.clone(), options, primary, None).unwrap_or_else(
74            |e| {
75                panic!(
76                    "failed to compile program: {}{e}",
77                    msg.map(|s| format!("{s}: ")).unwrap_or_default()
78                )
79            },
80        )
81    }));
82
83    match result {
84        Ok(ssa) => ssa,
85        Err(payload) => {
86            if let Some(program) = for_print {
87                eprintln!("--- Failing AST:\n{}\n---", DisplayAstAsNoir(&program));
88            }
89            std::panic::resume_unwind(payload);
90        }
91    }
92}

94/// Compare the execution result and print the inputs if the result is a failure.
95pub fn compare_results_compiled<P>(
96    inputs: &CompareCompiled<P>,
97    result: &CompareCompiledResult,
98) -> eyre::Result<()>
99where
100    CompareCompiled<P>: HasPrograms,
101{
102    let res = result.return_value_or_err();
103
104    if let Err(report) = res {
105        eprintln!("---\nComparison failed:");
106        eprintln!("{report:#}");
107
108        // Showing the AST as Noir so we can easily create integration tests.
109        let asts = inputs.programs();
110        let has_many = asts.len() > 1;
111        for (i, ast) in asts.into_iter().enumerate() {
112            if has_many {
113                eprintln!("---\nAST {}:\n{}", i + 1, DisplayAstAsNoir(ast));
114            } else {
115                eprintln!("---\nAST:\n{}", DisplayAstAsNoir(ast));
116            }
117        }
118        // Showing the inputs as TOML so we can easily create a Prover.toml file.
119        eprintln!(
120            "---\nInputs:\n{}",
121            Format::Toml
122                .serialize(&inputs.input_map, &inputs.abi)
123                .unwrap_or_else(|e| format!("failed to serialize inputs: {e}"))
124        );
125
126        // Display a Program without the Brillig opcodes, which are unreadable.
127        fn display_program(artifact: &SsaProgramArtifact) {
128            for (func_index, function) in artifact.program.functions.iter().enumerate() {
129                eprintln!("func {func_index}");
130                eprintln!("{function}");
131            }
132            for (func_index, function) in
133                artifact.program.unconstrained_functions.iter().enumerate()
134            {
135                eprintln!("unconstrained func {func_index}");
136                eprintln!("opcode count: {}", function.bytecode.len());
137            }
138        }
139
140        eprintln!("---\nOptions 1:\n{:?}", inputs.ssa1.options);
141        eprintln!("---\nProgram 1:");
142        display_program(&inputs.ssa1.artifact);
143
144        eprintln!("---\nOptions 2:\n{:?}", inputs.ssa2.options);
145        eprintln!("---\nProgram 2:");
146        display_program(&inputs.ssa1.artifact);
147
148        // Returning it as-is, so we can see the error message at the bottom as well.
149        Err(report)
150    } else {
151        Ok(())
152    }
153}

155/// Compare the execution result for comptime fuzzing and print the inputs if the result is a failure.
156pub fn compare_results_comptime(
157    inputs: &CompareComptime,
158    result: &CompareCompiledResult,
159) -> eyre::Result<()> {
160    let res = result.return_value_or_err();
161
162    if let Err(report) = res {
163        eprintln!("---\nComparison failed:");
164        eprintln!("{report:#}");
165
166        // Showing the AST as Noir so we can easily create integration tests.
167        eprintln!("---\nComptime source:\n{}", &inputs.source);
168        eprintln!("---\nAST:\n{}", DisplayAstAsNoir(&inputs.program));
169
170        eprintln!("---\nCompile options:\n{:?}", inputs.ssa.options);
171        eprintln!("---\nCompiled program:\n{}", inputs.ssa.artifact.program);
172
173        // Returning it as-is, so we can see the error message at the bottom as well.
174        Err(report)
175    } else {
176        Ok(())
177    }
178}

180/// Compare the execution result and print the inputs if the result is a failure.
181pub fn compare_results_interpreted(
182    inputs: &CompareInterpreted,
183    result: &CompareInterpretedResult,
184) -> eyre::Result<()> {
185    let res = result.return_value_or_err();
186
187    if let Err(report) = res {
188        eprintln!("---\nComparison failed:");
189        eprintln!("{report:#}");
190
191        // Showing the AST as Noir so we can easily create integration tests.
192        eprintln!("---\nAST:\n{}", DisplayAstAsNoir(&inputs.program));
193
194        // Showing the inputs as TOML so we can easily create a Prover.toml file.
195        eprintln!(
196            "---\nABI Inputs:\n{}",
197            Format::Toml
198                .serialize(&inputs.input_map, &inputs.abi)
199                .unwrap_or_else(|e| format!("failed to serialize inputs: {e}"))
200        );
201
202        // Show the SSA inputs as well so we can see any discrepancy in encoding.
203        eprintln!(
204            "---\nSSA Inputs:\n{}",
205            inputs
206                .ssa_args
207                .iter()
208                .enumerate()
209                .map(|(i, v)| format!("{i}: {v}"))
210                .collect::<Vec<_>>()
211                .join("\n")
212        );
213
214        // Common options for the SSA passes.
215        eprintln!("---\nOptions:\n{:?}", inputs.options);
216
217        eprintln!(
218            "---\nSSA 1 after step {} ({}):\n{}",
219            inputs.ssa1.step,
220            inputs.ssa1.msg,
221            inputs.ssa1.ssa.print_without_locations()
222        );
223        eprintln!(
224            "---\nSSA 2 after step {} ({}):\n{}",
225            inputs.ssa2.step,
226            inputs.ssa2.msg,
227            inputs.ssa2.ssa.print_without_locations()
228        );
229
230        // Returning it as-is, so we can see the error message at the bottom as well.
231        Err(report)
232    } else {
233        Ok(())
234    }
