33/// Minimal `Nargo.toml` for a reproduction package emitted on failure.
34const REPRO_NARGO_TOML: &str = "[package]\nname = \"fuzz_repro\"\ntype = \"bin\"\nauthors = []\n";
35
36/// Target directory for reproduction projects, taken from `NOIR_AST_FUZZER_EMIT_PROJECT`.
37///
38/// When set, a failing target writes a runnable `nargo` package so the failure can be
39/// replayed with `nargo execute` instead of rebuilding the project by hand.
40fn emit_project_dir() -> Option<std::path::PathBuf> {
41    std::env::var_os("NOIR_AST_FUZZER_EMIT_PROJECT").map(Into::into)
42}
43
44/// Write a single `nargo` package: `Nargo.toml`, `src/main.nr`, and an optional `Prover.toml`.
45///
46/// The `main.nr` is the fuzzer's best-effort Noir rendering of the AST; for the compiled
47/// targets it is not guaranteed to parse back, but it is a starting point that is otherwise
48/// reconstructed by hand from the printed output.
49fn write_nargo_package(dir: &Path, main_nr: &str, prover_toml: Option<&str>) {
50    if let Err(e) = std::fs::create_dir_all(dir.join("src")) {
51        eprintln!("failed to create project dir {}: {e}", dir.display());
52        return;
53    }
54    let mut files = vec![("Nargo.toml", REPRO_NARGO_TOML), ("src/main.nr", main_nr)];
55    if let Some(toml) = prover_toml {
56        files.push(("Prover.toml", toml));
57    }
58    for (rel, contents) in files {
59        let path = dir.join(rel);
60        if let Err(e) = std::fs::write(&path, contents) {
61            eprintln!("failed to write {}: {e}", path.display());
62        }
63    }
64    eprintln!("--- Wrote nargo project to {}", dir.display());
65}
66
67/// Emit one `nargo` package per program. A single program is written directly under `dir`;
68/// multiple programs are written under `dir/ast_1`, `dir/ast_2`, ..., all sharing the same
69/// `Prover.toml` inputs.
70fn emit_nargo_projects(dir: &Path, mains: &[String], prover_toml: Option<&str>) {
71    match mains {
72        [] => {}
73        [main] => write_nargo_package(dir, main, prover_toml),
74        many => {
75            for (i, main) in many.iter().enumerate() {
76                write_nargo_package(&dir.join(format!("ast_{}", i + 1)), main, prover_toml);
77            }
78        }
79    }
80}

82/// Compile a monomorphized [Program] into circuit or panic.
83pub fn compile_into_circuit_or_die(
84    program: Program,
85    options: &SsaEvaluatorOptions,
86    msg: Option<&str>,
87) -> SsaProgramArtifact {
88    compile_into_circuit_with_ssa_passes_or_die(program, options, &primary_passes(options), msg)
89}
90
91/// Compile a monomorphized [Program] into circuit using the given SSA passes or panic.
92///
93/// If there is a seed in the environment, then it prints the AST when an error is encountered.
94pub fn compile_into_circuit_with_ssa_passes_or_die(
95    program: Program,
96    options: &SsaEvaluatorOptions,
97    primary: &[SsaPass],
98    msg: Option<&str>,
99) -> SsaProgramArtifact {
100    // If we are using a seed, we are probably trying to reproduce some failure;
101    // in that case let's clone the program for printing if it panicked here,
102    // otherwise try to keep things faster by not cloning.
103    let for_print = std::env::var("NOIR_AST_FUZZER_SEED").is_ok().then(|| program.clone());
104
105    // We expect the programs generated should compile, but sometimes SSA passes panic, or return an error.
106    // Turn everything into a panic, catch it, print the AST, then resume panicking.
107    let result = std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
108        ssa::create_program_with_passes(program.clone(), options, primary, None).unwrap_or_else(
109            |e| {
110                panic!(
111                    "failed to compile program: {}{e}",
112                    msg.map(|s| format!("{s}: ")).unwrap_or_default()
113                )
114            },
115        )
116    }));
117
118    match result {
119        Ok(ssa) => ssa,
120        Err(payload) => {
121            if let Some(program) = for_print {
122                eprintln!("--- Failing AST:\n{}\n---", DisplayAstAsNoir(&program));
123            }
124            if let Some(dir) = emit_project_dir() {
125                // A compile-time failure has no ABI inputs, so emit `main.nr` only.
126                write_nargo_package(&dir, &DisplayAstAsNoir(&program).to_string(), None);
127            }
128            std::panic::resume_unwind(payload);
129        }
130    }
131}

133/// Compare the execution result and print the inputs if the result is a failure.
134pub fn compare_results_compiled<P>(
135    inputs: &CompareCompiled<P>,
136    result: &CompareCompiledResult,
137) -> eyre::Result<()>
138where
139    CompareCompiled<P>: HasPrograms,
140{
141    let res = result.return_value_or_err();
142
143    if let Err(report) = res {
144        eprintln!("---\nComparison failed:");
145        eprintln!("{report:#}");
146
147        // Showing the AST as Noir so we can easily create integration tests.
148        let asts = inputs.programs();
149        let has_many = asts.len() > 1;
150        for (i, &ast) in asts.iter().enumerate() {
151            if has_many {
152                eprintln!("---\nAST {}:\n{}", i + 1, DisplayAstAsNoir(ast));
153            } else {
154                eprintln!("---\nAST:\n{}", DisplayAstAsNoir(ast));
155            }
156        }
157        // Showing the inputs as TOML so we can easily create a Prover.toml file.
158        let inputs_toml = Format::Toml.serialize(&inputs.input_map, &inputs.abi);
159        eprintln!(
160            "---\nInputs:\n{}",
161            match &inputs_toml {
162                Ok(toml) => toml.clone(),
163                Err(e) => format!("failed to serialize inputs: {e}"),
164            }
165        );
166
167        if let Some(dir) = emit_project_dir() {
168            let mains =
169                asts.iter().map(|&ast| DisplayAstAsNoir(ast).to_string()).collect::<Vec<_>>();
170            emit_nargo_projects(&dir, &mains, inputs_toml.as_deref().ok());
171        }
172
173        // Display a Program without the Brillig opcodes, which are unreadable.
174        fn display_program(artifact: &SsaProgramArtifact) {
175            for (func_index, function) in artifact.program.functions.iter().enumerate() {
176                eprintln!("func {func_index}");
177                eprintln!("{function}");
178            }
179            for (func_index, function) in
180                artifact.program.unconstrained_functions.iter().enumerate()
181            {
182                eprintln!("unconstrained func {func_index}");
183                eprintln!("opcode count: {}", function.bytecode.len());
184            }
185        }
186
187        eprintln!("---\nOptions 1:\n{:?}", inputs.ssa1.options);
188        eprintln!("---\nProgram 1:");
189        display_program(&inputs.ssa1.artifact);
190
191        eprintln!("---\nOptions 2:\n{:?}", inputs.ssa2.options);
192        eprintln!("---\nProgram 2:");
193        display_program(&inputs.ssa1.artifact);
194
195        // Returning it as-is, so we can see the error message at the bottom as well.
196        Err(report)
197    } else {
198        Ok(())
199    }
200}

202/// Compare the execution result for comptime fuzzing and print the inputs if the result is a failure.
203pub fn compare_results_comptime(
204    inputs: &CompareComptime,
205    result: &CompareCompiledResult,
206) -> eyre::Result<()> {
207    let res = result.return_value_or_err();
208
209    if let Err(report) = res {
210        eprintln!("---\nComparison failed:");
211        eprintln!("{report:#}");
212
213        // Showing the AST as Noir so we can easily create integration tests.
214        eprintln!("---\nComptime source:\n{}", inputs.source);
215        eprintln!("---\nAST:\n{}", DisplayAstAsNoir(&inputs.program));
216
217        eprintln!("---\nCompile options:\n{:?}", inputs.ssa.options);
218        eprintln!("---\nCompiled program:\n{}", inputs.ssa.artifact.program);
219
220        if let Some(dir) = emit_project_dir() {
221            // The comptime source is already valid Noir; the comptime call bakes in its inputs,
222            // so there is no `Prover.toml` to emit.
223            write_nargo_package(&dir, &inputs.source, None);
224        }
225
226        // Returning it as-is, so we can see the error message at the bottom as well.
227        Err(report)
228    } else {
229        Ok(())
230    }
231}

233/// Compare the execution result and print the inputs if the result is a failure.
234pub fn compare_results_interpreted(
235    inputs: &CompareInterpreted,
236    result: &CompareInterpretedResult,
237) -> eyre::Result<()> {
238    let res = result.return_value_or_err();
239
240    if let Err(report) = res {
241        eprintln!("---\nComparison failed:");
242        eprintln!("{report:#}");
243
244        // Showing the AST as Noir so we can easily create integration tests.
245        eprintln!("---\nAST:\n{}", DisplayAstAsNoir(&inputs.program));
246
247        // Showing the inputs as TOML so we can easily create a Prover.toml file.
248        eprintln!(
249            "---\nABI Inputs:\n{}",
250            Format::Toml
251                .serialize(&inputs.input_map, &inputs.abi)
252                .unwrap_or_else(|e| format!("failed to serialize inputs: {e}"))
253        );
254
255        // Show the SSA inputs as well so we can see any discrepancy in encoding.
256        eprintln!(
257            "---\nSSA Inputs:\n{}",
258            inputs
259                .ssa_args
260                .iter()
261                .enumerate()
262                .map(|(i, v)| format!("{i}: {v}"))
263                .collect::<Vec<_>>()
264                .join("\n")
265        );
266
267        // Common options for the SSA passes.
268        eprintln!("---\nOptions:\n{:?}", inputs.options);
269
270        eprintln!(
271            "---\nSSA 1 after step {} ({}):\n{}",
272            inputs.ssa1.step,
273            inputs.ssa1.msg,
274            inputs.ssa1.ssa.print_without_locations()
275        );
276        eprintln!(
277            "---\nSSA 2 after step {} ({}):\n{}",
278            inputs.ssa2.step,
279            inputs.ssa2.msg,
280            inputs.ssa2.ssa.print_without_locations()
281        );
282
283        if let Some(dir) = emit_project_dir() {
284            let toml = Format::Toml.serialize(&inputs.input_map, &inputs.abi).ok();
285            write_nargo_package(
286                &dir,
287                &DisplayAstAsNoir(&inputs.program).to_string(),
288                toml.as_deref(),
289            );
290        }
291
292        // Returning it as-is, so we can see the error message at the bottom as well.
293        Err(report)
294    } else {
295        Ok(())
296    }
297}

299#[cfg(test)]
300mod tests {
301    use std::path::PathBuf;
302    use std::sync::atomic::{AtomicU32, Ordering};
303
304    use super::{emit_nargo_projects, write_nargo_package};
305
306    static COUNTER: AtomicU32 = AtomicU32::new(0);
307
308    /// A fresh, unique temp directory that does not yet exist on disk.
309    fn temp_dir() -> PathBuf {
310        let n = COUNTER.fetch_add(1, Ordering::Relaxed);
311        let dir =
312            std::env::temp_dir().join(format!("noir_ast_fuzzer_emit_{}_{n}", std::process::id()));
313        let _ = std::fs::remove_dir_all(&dir);
314        dir
315    }
316
317    #[test]
318    fn writes_single_package_with_inputs() {
319        let dir = temp_dir();
320        write_nargo_package(&dir, "fn main() {}", Some("x = \"1\"\n"));
321
322        assert!(dir.join("Nargo.toml").is_file());
323        assert_eq!(std::fs::read_to_string(dir.join("src/main.nr")).unwrap(), "fn main() {}");
324        assert_eq!(std::fs::read_to_string(dir.join("Prover.toml")).unwrap(), "x = \"1\"\n");
325
326        std::fs::remove_dir_all(&dir).unwrap();
327    }
328
329    #[test]
330    fn omits_prover_toml_when_no_inputs() {
331        let dir = temp_dir();
332        write_nargo_package(&dir, "fn main() {}", None);
333
334        assert!(dir.join("src/main.nr").is_file());
335        assert!(!dir.join("Prover.toml").exists());
336
337        std::fs::remove_dir_all(&dir).unwrap();
338    }
339
340    #[test]
341    fn single_ast_is_written_directly_under_dir() {
342        let dir = temp_dir();
343        emit_nargo_projects(&dir, &["fn main() {}".to_string()], None);
344
345        assert!(dir.join("src/main.nr").is_file());
346        assert!(!dir.join("ast_1").exists());
347
348        std::fs::remove_dir_all(&dir).unwrap();
349    }
350
351    #[test]
352    fn multiple_asts_are_written_under_numbered_subdirs() {
353        let dir = temp_dir();
354        let mains = ["fn main() { 1 }".to_string(), "fn main() { 2 }".to_string()];
355        emit_nargo_projects(&dir, &mains, Some("y = \"2\"\n"));
356
357        assert_eq!(
358            std::fs::read_to_string(dir.join("ast_1/src/main.nr")).unwrap(),
359            "fn main() { 1 }"
360        );
361        assert_eq!(
362            std::fs::read_to_string(dir.join("ast_2/src/main.nr")).unwrap(),
363            "fn main() { 2 }"
364        );
365        // Shared inputs are written into every package.
366        assert_eq!(std::fs::read_to_string(dir.join("ast_1/Prover.toml")).unwrap(), "y = \"2\"\n");
367        assert_eq!(std::fs::read_to_string(dir.join("ast_2/Prover.toml")).unwrap(), "y = \"2\"\n");
368
369        std::fs::remove_dir_all(&dir).unwrap();
370    }
