11/// Create a default configuration instance, with some common flags randomized.
12fn default_config(u: &mut Unstructured) -> arbitrary::Result<Config> {
13    // Some errors such as overflows and OOB are easy to trigger, so in half
14    // the cases we avoid all of them, to make sure they don't mask other errors.
15    let avoid_frequent_errors = u.arbitrary()?;
16    let config = Config {
17        avoid_overflow: avoid_frequent_errors,
18        avoid_index_out_of_bounds: avoid_frequent_errors,
19        ..Default::default()
20    };
21    Ok(config)
22}

24/// Common functions used in the test modules of targets.
25#[cfg(test)]
26mod tests {
27
28    const BUDGET: Duration = Duration::from_secs(20);
29    const MIN_SIZE: u32 = 1 << 12;
30    const MAX_SIZE: u32 = 1 << 20;
31
32    use std::time::Duration;
33
34    use arbitrary::Unstructured;
35    use color_eyre::eyre;
36    use proptest::prelude::*;
37
38    use crate::bool_from_env;
39
40    fn seed_from_env() -> Option<u64> {
41        let Ok(seed) = std::env::var("NOIR_AST_FUZZER_SEED") else { return None };
42        let seed = u64::from_str_radix(seed.trim_start_matches("0x"), 16)
43            .unwrap_or_else(|e| panic!("failed to parse seed '{seed}': {e}"));
44        Some(seed)
45    }
46
47    /// How long to let non-deterministic tests run for.
48    fn budget() -> Duration {
49        std::env::var("NOIR_AST_FUZZER_BUDGET_SECS").ok().map_or(BUDGET, |b| {
50            Duration::from_secs(
51                b.parse().unwrap_or_else(|e| panic!("failed to parse budget; got {b}: {e}")),
52            )
53        })
54    }
55
56    /// Check if we are running on CI.
57    fn is_running_in_ci() -> bool {
58        std::env::var("CI").is_ok()
59    }
60
61    /// Check if we explicitly want non-deterministic behavior, even on CI.
62    fn force_non_deterministic() -> bool {
63        bool_from_env("NOIR_AST_FUZZER_FORCE_NON_DETERMINISTIC")
64    }
65
66    /// `cargo fuzz` takes a long time to ramp up the complexity.
67    /// This test catches crash bugs much faster.
68    ///
69    /// Run it with for example:
70    /// ```ignore
71    /// NOIR_AST_FUZZER_SEED=0x6819c61400001000 \
72    /// cargo test -p noir_ast_fuzzer_fuzz acir_vs_brillig
73    /// ```
74    ///
75    /// The `cases` determine how many tests to run on CI.
76    /// Tune this so that we can expect CI to be able to get through all cases in reasonable time.
77    pub fn fuzz_with_arbtest(f: impl Fn(&mut Unstructured) -> eyre::Result<()>, cases: u32) {
78        let _ = env_logger::try_init();
79
80        if let Some(seed) = seed_from_env() {
81            run_reproduce(f, seed);
82        } else if is_running_in_ci() && !force_non_deterministic() {
83            run_deterministic(f, cases);
84        } else {
85            run_nondeterministic(f);
86        }
87    }
88
89    /// Reproduce the result of a single seed.
90    fn run_reproduce(f: impl Fn(&mut Unstructured) -> eyre::Result<()>, seed: u64) {
91        arbtest::arbtest(|u| {
92            f(u).unwrap();
93            Ok(())
94        })
95        .seed(seed)
96        .run();
97    }
98
99    /// Run the tests non-deterministically until the timeout.
100    ///
101    /// This is the local behavior.
102    fn run_nondeterministic(f: impl Fn(&mut Unstructured) -> eyre::Result<()>) {
103        arbtest::arbtest(|u| {
104            f(u).unwrap();
105            Ok(())
106        })
107        .size_min(MIN_SIZE)
108        .size_max(MAX_SIZE)
109        .budget(budget())
110        .run();
111    }
112
113    /// Run multiple tests with a deterministic RNG.
114    ///
115    /// This is the behavior on CI.
116    fn run_deterministic(f: impl Fn(&mut Unstructured) -> eyre::Result<()>, cases: u32) {
117        let config = proptest::test_runner::Config {
118            cases,
119            failure_persistence: None,
120            max_shrink_iters: 0,
121            ..Default::default()
122        };
123        let rng = proptest::test_runner::TestRng::deterministic_rng(config.rng_algorithm);
124        let mut runner = proptest::test_runner::TestRunner::new_with_rng(config, rng);
125
126        runner
127            .run(&seed_strategy(), |seed| {
128                run_reproduce(&f, seed);
129                Ok(())
130            })
131            .unwrap();
132    }
133
134    /// Generate seeds for `arbtest` where the top 32 bits are random and the lower 32 bits represent the input size.
135    fn seed_strategy() -> proptest::strategy::BoxedStrategy<u64> {
136        (MIN_SIZE..MAX_SIZE)
137            .prop_flat_map(move |size| {
138                any::<u64>().prop_map(move |raw| (size as u64) | (raw << u32::BITS))
139            })
140            .boxed()
141    }
142}
