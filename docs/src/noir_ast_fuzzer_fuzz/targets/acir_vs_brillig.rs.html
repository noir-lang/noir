1//! Compare the execution of random ASTs between the normal execution
2//! vs when everything is forced to be Brillig.
3use crate::targets::default_config;
4use crate::{compare_results_compiled, compile_into_circuit_or_die, default_ssa_options};
5use arbitrary::Arbitrary;
6use arbitrary::Unstructured;
7use color_eyre::eyre;
8use noir_ast_fuzzer::compare::{CompareOptions, ComparePipelines};
9use noir_ast_fuzzer::rewrite::change_all_functions_into_unconstrained;
10
11pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
12    let config = default_config(u)?;
13
14    let inputs = ComparePipelines::arb(
15        u,
16        config,
17        |u, program| {
18            let options = CompareOptions::arbitrary(u)?;
19            let ssa = compile_into_circuit_or_die(
20                program,
21                &options.onto(default_ssa_options()),
22                Some("acir"),
23            );
24            Ok((ssa, options))
25        },
26        |u, program| {
27            let options = CompareOptions::arbitrary(u)?;
28            let ssa = compile_into_circuit_or_die(
29                change_all_functions_into_unconstrained(program),
30                &options.onto(default_ssa_options()),
31                Some("brillig"),
32            );
33            Ok((ssa, options))
34        },
35    )?;
36
37    let result = inputs.exec()?;
38
39    compare_results_compiled(&inputs, &result)
40}
