1//! Compare the execution of an SSA at two arbitrary points in the
2//! optimization pipeline using the SSA interpreter.
3//!
4//! The two points are chosen independently per seed and may be any
5//! distance apart — including adjacent (the original behavior) and
6//! all the way from the initial SSA to the fully-optimized form. If
7//! their interpretation results differ, semantic preservation has been
8//! broken by some pass between them; the offending pass can then be
9//! pinpointed with `nargo interpret` (or the `bisect-ssa-pass` skill).
10//!
11//! Comparing arbitrary-distance pairs (rather than only adjacent ones)
12//! roughly multiplies the per-seed odds of catching a bug whose
13//! "toggling" pass is at a fixed pipeline position: adjacent-only
14//! comparison catches the bug with probability `1/N` for `N` passes,
15//! while arbitrary-distance comparison catches it whenever the two
16//! sampled steps straddle the toggling pass.
17//!
18//! By using the SSA interpreter we can execute any pass in the pipeline,
19//! as opposed to the Brillig runtime, which requires a minimum number
20//! of passes to be carried out to work.
21use crate::{compare_results_interpreted, default_ssa_options};
22use arbitrary::{Arbitrary, Unstructured};
23use color_eyre::eyre;
24use noir_ast_fuzzer::Config;
25use noir_ast_fuzzer::compare::{CompareInterpreted, CompareOptions, ComparePass};
26use noirc_evaluator::ssa::ssa_gen::Ssa;
27use noirc_evaluator::ssa::{SsaPass, primary_passes, ssa_gen};
28
29pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
30    let config = Config { avoid_overflow: u.arbitrary()?, ..Config::default() };
31
32    let inputs = CompareInterpreted::arb(u, config, |u, program| {
33        let options = CompareOptions::arbitrary(u)?;
34        let ssa_options = options.onto(default_ssa_options());
35        let ssa_passes = primary_passes(&ssa_options);
36
37        // Step 0 = the initial SSA (no passes run yet); step N = the SSA
38        // after running the first N passes in `ssa_passes`. We pick two
39        // distinct steps `step_a < step_b` from the closed interval
40        // `[0, max_passes]` and compare them.
41        let max_passes = ssa_passes.len();
42        let step_a = u.int_in_range(0..=max_passes - 1)?;
43        let step_b = u.int_in_range(step_a + 1..=max_passes)?;
44
45        // Generate the initial SSA, which is considered to be step 0.
46        let ssa = ssa_gen::generate_ssa(program).expect("failed to generate initial SSA");
47
48        // Run passes up to (and not including) step_a to get the state
49        // at step_a. The remaining passes [step_a..step_b) are then
50        // applied on a clone of that state to get step_b.
51        let ssa_at_a = ssa_passes[..step_a].iter().fold(ssa, run_pass_or_die);
52        let ssa1 = ComparePass {
53            step: step_a,
54            msg: step_msg(&ssa_passes, step_a),
55            ssa: clone_ssa(&ssa_at_a),
56        };
57        let ssa2 = ComparePass {
58            step: step_b,
59            msg: step_msg(&ssa_passes, step_b),
60            ssa: ssa_passes[step_a..step_b].iter().fold(ssa_at_a, run_pass_or_die),
61        };
62
63        Ok((options, ssa1, ssa2))
64    })?;
65
66    let result = inputs.exec()?;
67
68    compare_results_interpreted(&inputs, &result)
69}
70
71/// Human-readable label for a given pipeline step:
72/// - step 0 is the initial SSA, before any pass has run;
73/// - step N (1 ≤ N ≤ passes.len()) is the SSA after running
74///   `passes[N - 1]`.
75fn step_msg(passes: &[SsaPass<'_>], step: usize) -> String {
76    if step == 0 { "Initial SSA".to_string() } else { passes[step - 1].msg().to_string() }
77}
78
79fn run_pass_or_die(ssa: Ssa, pass: &SsaPass) -> Ssa {
80    pass.run(ssa).unwrap_or_else(|e| panic!("failed to run pass {}: {e}", pass.msg()))
81}
82
83fn clone_ssa(ssa: &Ssa) -> Ssa {
84    Ssa::new(ssa.functions.values().cloned().collect(), ssa.error_selector_to_type.clone())
85}
