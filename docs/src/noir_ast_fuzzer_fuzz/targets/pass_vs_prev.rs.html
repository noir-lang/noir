1//! Compare the execution of an SSA pass to the one preceding it
2//! using the SSA interpreter.
3//!
4//! By using the SSA interpreter we can execute any pass in the pipeline,
5//! as opposed to the Brillig runtime, which requires a minimum number
6//! of passes to be carried out to work.
7use crate::{compare_results_interpreted, default_ssa_options};
8use arbitrary::{Arbitrary, Unstructured};
9use color_eyre::eyre;
10use noir_ast_fuzzer::Config;
11use noir_ast_fuzzer::compare::{CompareInterpreted, CompareOptions, ComparePass};
12use noirc_evaluator::ssa::ssa_gen::Ssa;
13use noirc_evaluator::ssa::{SsaPass, primary_passes, ssa_gen};
14
15pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
16    let config = Config { avoid_overflow: u.arbitrary()?, ..Config::default() };
17
18    let inputs = CompareInterpreted::arb(u, config, |u, program| {
19        let options = CompareOptions::arbitrary(u)?;
20        let ssa_options = options.onto(default_ssa_options());
21        let ssa_passes = primary_passes(&ssa_options);
22
23        // Choose the number of passes we run on top of the initial SSA.
24        let max_passes = ssa_passes.len();
25        let run_passes = u.int_in_range(1..=max_passes)?;
26
27        // Generate the initial SSA, which is considered to be step 0.
28        let ssa = ssa_gen::generate_ssa(program).expect("failed to generate initial SSA");
29
30        // Compare two consecutive passes.
31        let last_pass = &ssa_passes[run_passes - 1];
32        let ssa1 = if run_passes == 1 {
33            ComparePass { step: 0, msg: "Initial SSA".to_string(), ssa }
34        } else {
35            let prev_step = run_passes - 1;
36            let prev_idx = prev_step - 1;
37            ComparePass {
38                step: prev_step,
39                msg: ssa_passes[prev_idx].msg().to_string(),
40                ssa: ssa_passes[..=prev_idx].iter().fold(ssa, run_pass_or_die),
41            }
42        };
43        let ssa2 = ComparePass {
44            step: run_passes,
45            msg: last_pass.msg().to_string(),
46            ssa: run_pass_or_die(clone_ssa(&ssa1.ssa), last_pass),
47        };
48
49        Ok((options, ssa1, ssa2))
50    })?;
51
52    let result = inputs.exec()?;
53
54    compare_results_interpreted(&inputs, &result)
55}
56
57fn run_pass_or_die(ssa: Ssa, pass: &SsaPass) -> Ssa {
58    pass.run(ssa).unwrap_or_else(|e| panic!("failed to run pass {}: {e}", pass.msg()))
59}
60
61fn clone_ssa(ssa: &Ssa) -> Ssa {
62    Ssa::new(ssa.functions.values().cloned().collect(), ssa.error_selector_to_type.clone())
63}
