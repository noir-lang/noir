19pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
20    let config = default_config(u)?;
21    let rules = rules::collect(&config);
22    let max_rewrites = 10;
23    let inputs = CompareMorph::arb(
24        u,
25        config,
26        |u, mut program| {
27            let options = CompareOptions::arbitrary(u)?;
28            rewrite_program(u, &mut program, &rules, max_rewrites);
29            Ok((program, options))
30        },
31        |program, options| {
32            compile_into_circuit_or_die(program, &options.onto(default_ssa_options()), None)
33        },
34    )?;
35
36    let result = inputs.exec()?;
37
38    compare_results_compiled(&inputs, &result)
39}

41fn rewrite_program(
42    u: &mut Unstructured,
43    program: &mut Program,
44    rules: &[rules::Rule],
45    max_rewrites: usize,
46) {
47    for func in program.functions.iter_mut() {
48        if func.name.ends_with("_proxy") {
49            continue;
50        }
51        rewrite_function(u, func, rules, max_rewrites);
52    }
53}
54
55fn rewrite_function(
56    u: &mut Unstructured,
57    func: &mut Function,
58    rules: &[rules::Rule],
59    max_rewrites: usize,
60) {
61    // We can call `rewrite::next_local_and_ident_id` and pass the results to the rewrite rules,
62    // if they want to add new variables with new local IDs.
63    let ctx = rules::Context { unconstrained: func.unconstrained, ..Default::default() };
64
65    let estimate = estimate_applicable_rules(&ctx, &func.body, rules);
66    let morph = MorphContext {
67        target: max_rewrites.min(estimate),
68        estimate,
69        count: Cell::new(0),
70        rules,
71        vars: RefCell::new(VariableContext::new(func)),
72    };
73
74    morph.rewrite_expr(&ctx, u, &mut func.body);
75}

77/// Context necessary to generate new local IDs during rewrites.
78///
79/// Potentially a place to reconstruct local variable scopes.
80struct VariableContext {
81    next_local_id: u32,
82    next_ident_id: u32,
83    locals: ScopeStack<LocalId>,
84}
85
86impl VariableContext {
87    fn new(func: &Function) -> Self {
88        let (next_local_id, next_ident_id) = rewrite::next_local_and_ident_id(func);
89
90        let locals = ScopeStack::from_variables(
91            func.parameters
92                .iter()
93                .map(|(id, mutable, name, typ, _vis)| (*id, *mutable, name.clone(), typ.clone())),
94        );
95
96        Self { next_local_id, next_ident_id, locals }
97    }
98
99    fn next_local_id(&mut self) -> LocalId {
100        let id = self.next_local_id;
101        self.next_local_id += 1;
102        LocalId(id)
103    }
104
105    fn next_ident_id(&mut self) -> IdentId {
106        let id = self.next_ident_id;
107        self.next_ident_id += 1;
108        IdentId(id)
109    }
110}

112/// Recursively apply rules while keeping a tally on how many we have done.
113struct MorphContext<'a> {
114    /// Number of rewrites we want to achieve.
115    target: usize,
116    /// (Over)estimate of the maximum number we could hope to apply.
117    estimate: usize,
118    /// Number of rewrites applied so far, up to the `target`.
119    count: Cell<usize>,
120    /// Rules to apply.
121    rules: &'a [rules::Rule],
122    /// Book keeping for local variables.
123    vars: RefCell<VariableContext>,
124}
125
126impl MorphContext<'_> {
127    /// Check if we have reached the target.
128    fn limit_reached(&self) -> bool {
129        self.target == 0 || self.estimate == 0 || self.count.get() == self.target
130    }
131
132    fn rewrite_expr(&self, ctx: &rules::Context, u: &mut Unstructured, expr: &mut Expression) {
133        visit_expr_be_mut(
134            expr,
135            &mut |expr: &mut Expression| {
136                if self.limit_reached() {
137                    return (false, false);
138                }
139
140                // Check if we are entering a new scope.
141                let entered = matches!(
142                    expr,
143                    Expression::Block(_)
144                        | Expression::For(_)
145                        | Expression::Loop(_)
146                        | Expression::While(_)
147                        | Expression::If(_)
148                );
149
150                if entered {
151                    self.vars.borrow_mut().locals.enter();
152                }
153
154                // We apply the rules on this expression, but its
155                // children will be visited after this call.
156                let cont = self.apply_rules(ctx, u, expr);
157
158                (cont, entered)
159            },
160            &mut |expr, entered| {
161                // A `let` variable becomes visible *after* we have have processed all its children,
162                // so, only its siblings can see it.
163                if let Expression::Let(let_) = expr {
164                    let typ = let_.expression.return_type().expect("let should have a type");
165                    self.vars.borrow_mut().locals.add(
166                        let_.id,
167                        let_.mutable,
168                        let_.name.clone(),
169                        typ.into_owned(),
170                    );
171                }
172                if entered {
173                    self.vars.borrow_mut().locals.exit();
174                }
175            },
176            &mut |_| {},
177        );
178    }
179
180    fn apply_rules(
181        &self,
182        ctx: &rules::Context,
183        u: &mut Unstructured,
184        expr: &mut Expression,
185    ) -> bool {
186        match expr {
187            Expression::For(for_) => {
188                // Separate context for just the ranges.
189                let range_ctx = rules::Context { is_in_range: true, ..*ctx };
190                self.rewrite_expr(&range_ctx, u, &mut for_.start_range);
191                self.rewrite_expr(&range_ctx, u, &mut for_.end_range);
192                // Original context for the body.
193                self.vars.borrow_mut().locals.add(
194                    for_.index_variable,
195                    false,
196                    for_.index_name.clone(),
197                    for_.index_type.clone(),
198                );
199                self.rewrite_expr(ctx, u, &mut for_.block);
200                self.vars.borrow_mut().locals.remove(&for_.index_variable);
201                // No need to visit children, we just visited them.
202                false
203            }
204            Expression::Unary(
205                unary @ Unary { operator: UnaryOp::Reference { mutable: true }, .. },
206            ) => {
207                let ctx = rules::Context { is_in_ref_mut: true, ..*ctx };
208                self.rewrite_expr(&ctx, u, &mut unary.rhs);
209                false
210            }
211            Expression::Call(call) if is_special_call(call) => {
212                let ctx = rules::Context { is_in_special_call: true, ..*ctx };
213                for arg in call.arguments.iter_mut() {
214                    self.rewrite_expr(&ctx, u, arg);
215                }
216                false
217            }
218            // The rest can just have the rules applied on them, using the same context.
219            _ => {
220                for rule in self.rules {
221                    match self.try_apply_rule(ctx, u, expr, rule) {
222                        Ok(false) => {
223                            // We couldn't, or decided not to apply this rule; try the next one.
224                            continue;
225                        }
226                        Err(_) => {
227                            // We ran out of randomness; stop visiting the AST.
228                            return false;
229                        }
230
231                        Ok(true) => {
232                            // We applied a rule on this expression.
233                            self.count.set(self.count.get() + 1);
234                            // We could visit the children of this morphed expression, which could result in repeatedly applying
235                            // the same rule over and over again. When we have 100% application rate (e.g. a small function),
236                            // it would be wasting all the potential on the first rule that matched, e.g. `(x - (0 + (0 - 0)))`.
237                            // It would also throw off the estimate if we introduce new items on which we can apply rules.
238                            return false;
239                        }
240                    }
241                }
242                // If we made it this far, we did not apply any rule, so look deeper in the AST.
243                true
244            }
245        }
246    }
247
248    /// Check if a rule can be applied on an expression. If it can, apply it based on some arbitrary
249    /// criteria, returning a flag showing whether it was applied.
250    fn try_apply_rule(
251        &self,
252        ctx: &rules::Context,
253        u: &mut Unstructured,
254        expr: &mut Expression,
255        rule: &rules::Rule,
256    ) -> arbitrary::Result<bool> {
257        if rule.matches(ctx, expr) && u.ratio(self.target, self.estimate)? {
258            rule.rewrite(u, &mut self.vars.borrow_mut(), expr)?;
259            Ok(true)
260        } else {
261            Ok(false)
262        }
263    }
264}

266/// Provide a rough estimate for how many rules can be applied.
267///
268/// It will overestimate, because it ignores the finer points of when a rule can match.
269fn estimate_applicable_rules(
270    ctx: &rules::Context,
271    expr: &Expression,
272    rules: &[rules::Rule],
273) -> usize {
274    let mut count = 0;
275    visit_expr(expr, &mut |expr| {
276        for rule in rules {
277            if rule.matches(ctx, expr) {
278                count += 1;
279            }
280        }
281        true
282    });
283    count
284}

286/// Check if we are calling an oracle or builtin function.
287fn is_special_call(call: &Call) -> bool {
288    matches!(
289        call.func.as_ref(),
290        Expression::Ident(Ident {
291            definition: Definition::Oracle(_) | Definition::Builtin(_) | Definition::LowLevel(_),
292            ..
293        })
294    )
295}

332    /// Metamorphic transformation rule.
333    pub struct Rule {
334        pub matches: Box<MatchFn>,
335        pub rewrite: Box<RewriteFn>,
336    }

338    impl Rule {
339        pub fn new(
340            matches: impl Fn(&Context, &Expression) -> bool + 'static,
341            rewrite: impl Fn(
342                &mut Unstructured,
343                &mut VariableContext,
344                &mut Expression,
345            ) -> arbitrary::Result<()>
346            + 'static,
347        ) -> Self {
348            Self { matches: Box::new(matches), rewrite: Box::new(rewrite) }
349        }
350
351        /// Check if the rule can be applied on an expression.
352        pub fn matches(&self, ctx: &Context, expr: &Expression) -> bool {
353            (self.matches)(ctx, expr)
354        }
355
356        /// Apply the rule on an expression, mutating/replacing it in-place.
357        pub fn rewrite(
358            &self,
359            u: &mut Unstructured,
360            vars: &mut VariableContext,
361            expr: &mut Expression,
362        ) -> arbitrary::Result<()> {
363            (self.rewrite)(u, vars, expr)
364        }

367    /// Construct all rules that we can apply on a program.
368    pub fn collect(config: &Config) -> Vec<Rule> {
369        let mut rules = vec![
370            num_add_zero(),
371            num_sub_zero(),
372            num_mul_one(),
373            num_div_one(),
374            bool_or_self(),
375            bool_xor_self(),
376            bool_xor_rand(),
377            any_inevitable(),
378            int_break_up(),
379        ];
380        if config.avoid_overflow {
381            // When we can overflowing instruction, then swapping around the LHS and RHS
382            // of a binary operation can swap failures. We could visit the expressions to rule
383            // out a potential failure on both sides at the same time, or just skip this rule.
384            rules.push(num_commute());
385        }
386        rules
387    }

389    /// Transform any numeric value `x` into `x <op> <rhs>`
390    fn num_op(op: BinaryOpKind, rhs: u32) -> Rule {
391        Rule::new(num_rule_matches, move |_u, _locals, expr| {
392            let typ = expr.return_type().expect("only called on matching type").into_owned();
393            expr::replace(expr, |expr| expr::binary(expr, op, expr::int_literal(rhs, false, typ)));
394            Ok(())
395        })
396    }

418    /// Break an integer literal `a` into `b + c`.
419    pub fn int_break_up() -> Rule {
420        Rule::new(
421            |ctx, expr| {
422                if ctx.is_in_range && !ctx.unconstrained || ctx.is_in_ref_mut {
423                    return false;
424                }
425                matches!(expr, Expression::Literal(Literal::Integer(_, Type::Integer(_, _), _)))
426            },
427            |u, _locals, expr| {
428                let Expression::Literal(Literal::Integer(a, typ, loc)) = expr else {
429                    unreachable!("matched only integer literals, got {expr}");
430                };
431                let mut b_expr = expr::gen_literal(u, typ, &Config::default())?;
432                let Expression::Literal(Literal::Integer(b, _, _)) = &mut b_expr else {
433                    unreachable!("generated a literal of the same type");
434                };
435
436                // Make them have the same sign, so they are on the same side of 0 and a single number
437                // can add up to them without overflow. (e.g. there is no x such that `i32::MIN + x == i32::MAX`)
438                if a.is_negative() && !b.is_negative() {
439                    *b = SignedField::negative(b.absolute_value());
440                } else if !a.is_negative() && b.is_negative() {
441                    *b = SignedField::positive(b.absolute_value() - FieldElement::one()); // -1 just to avoid the potential of going from e.g. i8 -128 to 128 where the maximum is 127.
442                }
443
444                let (op, c) = if *a >= *b {
445                    (BinaryOpKind::Add, (*a - *b))
446                } else {
447                    (BinaryOpKind::Subtract, (*b - *a))
448                };
449
450                let c_expr = Expression::Literal(Literal::Integer(c, typ.clone(), *loc));
451
452                *expr = expr::binary(b_expr, op, c_expr);
453
454                Ok(())
455            },
456        )
457    }

468    pub fn bool_xor_self() -> Rule {
469        Rule::new(bool_rule_matches, |_u, _locals, expr| {
470            expr::replace(expr, |expr| {
471                let rhs = expr::binary(expr.clone(), BinaryOpKind::Xor, expr.clone());
472                expr::binary(expr, BinaryOpKind::Xor, rhs)

478    /// Transform boolean value `x` into `rnd ^ x ^ rnd`.
479    pub fn bool_xor_rand() -> Rule {
480        Rule::new(bool_rule_matches, |u, _locals, expr| {
481            // This is where we could access the scope to look for a random bool variable.
482            let rnd = expr::gen_literal(u, &Type::Bool, &Config::default())?;
483            expr::replace(expr, |expr| {
484                let rhs = expr::binary(expr, BinaryOpKind::Xor, rnd.clone());

491    /// Transform commutative arithmetic operations:
492    /// * `a + b` into `b + a`
493    /// * `a * b` into `b * a`
494    pub fn num_commute() -> Rule {
495        Rule::new(
496            |_ctx, expr| {
497                matches!(
498                    expr,
499                    Expression::Binary(Binary {
500                        operator: BinaryOpKind::Add | BinaryOpKind::Multiply,
501                        ..
502                    })
503                ) && !has_side_effect(expr)
504            },
505            |_u, _locals, expr| {
506                let Expression::Binary(binary) = expr else {
507                    unreachable!("the rule only matches Binary expressions");
508                };
509
510                std::mem::swap(&mut binary.lhs, &mut binary.rhs);
511
512                Ok(())
513            },
514        )
515    }

517    /// Transform any expression into an if-then-else with the itself
518    /// repeated in the _then_ and _else_ branch:
519    /// * `x` into `if c { x } else { x }`
520    pub fn any_inevitable() -> Rule {
521        Rule::new(
522            |ctx, expr| {
523                !ctx.is_in_special_call
524                    && !ctx.is_in_ref_mut
525                    // If we're in ACIR then don't turn loop ranges into non-constant expressions.
526                    && (ctx.unconstrained || !ctx.is_in_range)
527                    // `let x = 1;` transformed into `if true { let x = 1; } else { let x = 1; }` would leave `x` undefined.
528                    && !matches!(expr, Expression::Let(_))
529                    // We can't return references from an `if` statement
530                    && expr.return_type().map(|typ| !types::contains_reference(typ.as_ref())).unwrap_or(true)
531            },
532            |u, vars, expr| {
533                let typ = expr.return_type().map(|typ| typ.into_owned()).unwrap_or(Type::Unit);
534
535                // Find a bool expression we can use. For simplicity just consider actual bool variables,
536                // not things that can produce variables, so we have less logic to repeat for the `FunctionContext`.
537                let bool_vars = vars
538                    .locals
539                    .current()
540                    .variables()
541                    .filter_map(|(id, (_, _, typ))| (*typ == Type::Bool).then_some(id))
542                    .collect::<Vec<_>>();
543
544                // If we don't have a bool variable, generate some random expression.
545                let cond = if bool_vars.is_empty() {
546                    gen_expr(u, &Type::Bool, 2)?
547                } else {
548                    let id = u.choose_iter(bool_vars)?;
549                    let (mutable, name, typ) = vars.locals.current().get_variable(id);
550                    Expression::Ident(Ident {
551                        location: None,
552                        definition: Definition::Local(*id),
553                        mutable: *mutable,
554                        name: name.clone(),
555                        typ: typ.clone(),
556                        id: vars.next_ident_id(),
557                    })
558                };
559
560                // Duplicate the expression, then assign new IDs to all variables created in it.
561                let mut alt = expr.clone();
562
563                reassign_ids(vars, &mut alt);
564
565                expr::replace(expr, |expr| expr::if_else(cond, expr, alt, typ));
566                Ok(())
567            },
568        )
569    }

571    /// Common match condition for boolean rules.
572    fn bool_rule_matches(ctx: &Context, expr: &Expression) -> bool {
573        // If we rewrite `&mut x` into `&mut (x | x)` we will alter the semantics.
574        if ctx.is_in_ref_mut {
575            return false;
576        }
577        // We don't want to mess with the arguments of a `println`, because the printer assumes they are bool literals.
578        // Similarly a `constrain` call is expected to have a single boolean expression.
579        if ctx.is_in_special_call {
580            return false;
581        }
582        // We can apply boolean rule on anything that returns a bool,
583        // unless the expression can have a side effect, which we don't want to duplicate.
584        if let Some(typ) = expr.return_type() {
585            matches!(typ.as_ref(), Type::Bool)
586                && !has_side_effect(expr)
587                && !expr::exists(expr, |expr| {
588                    matches!(
589                        expr,
590                        Expression::Let(_)     // Creating a variable needs a new ID
591                        | Expression::Match(_) // Match creates variables which would need new IDs
592                        | Expression::Block(_) // Applying logical operations on blocks would look odd
593                    )
594                })
595        } else {
596            false
597        }
598    }
599
600    /// Common condition for numeric rules
601    fn num_rule_matches(ctx: &Context, expr: &Expression) -> bool {
602        // Because of #8305 we can't reliably use expressions in ranges in ACIR.
603        if ctx.is_in_range && !ctx.unconstrained {
604            return false;
605        }
606        // If we rewrite `&mut x` into `&mut (x - 0)` we will alter the semantics.
607        if ctx.is_in_ref_mut {
608            return false;
609        }
610        // Appending 0 to a block would look odd.
611        if matches!(expr, Expression::Block(_)) {
612            return false;
613        }
614        // We can apply this rule on anything that returns a number.
615        if let Some(typ) = expr.return_type() {
616            matches!(typ.as_ref(), Type::Field | Type::Integer(_, _))
617        } else {
618            false
619        }
620    }

640    /// Check if an expression can have a side effect, in which case duplicating or reordering it could
641    /// change the behavior of the program. This doesn't concern about failures, just observable changes
642    /// the state of the program.
643    pub(super) fn has_side_effect(expr: &Expression) -> bool {
644        expr::exists(expr, |expr| {
645            matches!(
646                expr,
647                Expression::Call(_) // Functions can have side effects, maybe mutating some reference, printing
648                | Expression::Assign(_) // Assignment to a mutable variable could double up effects
649            )
650        })
651    }

653    /// Generate an arbitrary pure (free of side effects) expression, returning a specific type.
654    pub(super) fn gen_expr(
655        u: &mut Unstructured,
656        typ: &Type,
657        max_depth: usize,
658    ) -> arbitrary::Result<Expression> {
659        if max_depth > 0 {
660            let idx = u.choose_index(3)?;
661            if idx == 0 {
662                if let Some(expr) = gen_unary(u, typ, max_depth)? {
663                    return Ok(expr);
664                }
665            }
666            if idx == 1 {
667                if let Some(expr) = gen_binary(u, typ, max_depth)? {
668                    return Ok(expr);
669                }
670            }
671        }
672        expr::gen_literal(u, typ, &Config::default())
673    }

675    /// Generate an arbitrary unary expression, returning a specific type.
676    pub(super) fn gen_unary(
677        u: &mut Unstructured,
678        typ: &Type,
679        max_depth: usize,
680    ) -> arbitrary::Result<Option<Expression>> {
681        if !types::can_unary_return(typ) {
682            return Ok(None);
683        }
684        let mut make_unary = |op| {
685            let expr = gen_expr(u, typ, max_depth.saturating_sub(1))?;
686            Ok(Some(expr::unary(op, expr, typ.clone())))
687        };
688        if types::is_numeric(typ) {
689            make_unary(UnaryOp::Minus)
690        } else if types::is_bool(typ) {
691            make_unary(UnaryOp::Not)
692        } else {
693            Ok(None)
694        }
695    }

697    /// Generate an arbitrary binary expression, returning a specific type.
698    ///
699    /// The operation should not have any side effects, ie. it must not fail.
700    pub(super) fn gen_binary(
701        u: &mut Unstructured,
702        typ: &Type,
703        max_depth: usize,
704    ) -> arbitrary::Result<Option<Expression>> {
705        // Collect the operations can return the expected type.
706        // Do not introduce new errors in randomly generated code that only exists in the morph.
707        let ops = BinaryOp::iter()
708            .filter(|op| {
709                types::can_binary_op_return(op, typ)
710                    && !types::can_binary_op_overflow(op)
711                    && !types::can_binary_op_err_by_zero(op)
712            })
713            .collect::<Vec<_>>();
714
715        // Ideally we checked that the target type can be returned, but just in case.
716        if ops.is_empty() {
717            return Ok(None);
718        }
719
720        // Choose a random operation.
721        let op = u.choose_iter(ops)?;
722
723        let type_options = TYPES.get_or_init(|| {
724            let mut types = vec![Type::Bool, Type::Field];
725
726            for sign in [Signedness::Signed, Signedness::Unsigned] {
727                for size in IntegerBitSize::iter() {
728                    if sign.is_signed() && size.bit_size() == 1 {
729                        continue;
730                    }
731                    // Avoid negative literals; the frontend makes them difficult to work with in expressions
732                    // where no type inference information is available.
733                    if sign.is_signed() {
734                        continue;
735                    }
736                    // Avoid large integers; frontend doesn't like them.
737                    if size.bit_size() > 32 {
738                        continue;
739                    }
740                    types.push(Type::Integer(sign, size));
741                }
742            }
743            types
744        });
745
746        // Select input types that can produce the output we want.
747        let type_options = type_options
748            .iter()
749            .filter(|input| types::can_binary_op_return_from_input(&op, input, typ))
750            .collect::<Vec<_>>();
751
752        // Choose a type for the LHS and RHS.
753        let lhs_type = u.choose_iter(type_options)?;
754
755        // Generate expressions for LHS and RHS.
756        let lhs_expr = gen_expr(u, lhs_type, max_depth.saturating_sub(1))?;
757        let rhs_expr = gen_expr(u, lhs_type, max_depth.saturating_sub(1))?;
758
759        let mut expr = expr::binary(lhs_expr, op, rhs_expr);
760
761        // If we have chosen e.g. u8 and need u32 we need to cast.
762        if !(lhs_type == typ || types::is_bool(typ) && op.is_comparator()) {
763            expr = expr::cast(expr, typ.clone());
764        }
765
766        Ok(Some(expr))
767    }

769    /// Types we can consider using in this context.
770    static TYPES: OnceLock<Vec<Type>> = OnceLock::new();
771
772    /// Assign new IDs to variables and identifiers created in the expression.
773    pub(super) fn reassign_ids(vars: &mut VariableContext, expr: &mut Expression) {
774        fn replace_local_id(
775            vars: &mut VariableContext,
776            replacements: &mut HashMap<LocalId, LocalId>,
777            id: &mut LocalId,
778        ) {
779            let curr = *id;
780            let next = vars.next_local_id();
781            replacements.insert(curr, next);
782            *id = next;
783        }
784
785        let replacements = RefCell::new(HashMap::new());
786
787        visit_expr_be_mut(
788            expr,
789            // Assign a new ID where variables are created, and remember what original value they replaced.
790            &mut |expr| {
791                match expr {
792                    Expression::Ident(ident) => {
793                        ident.id = vars.next_ident_id();
794                    }
795                    Expression::Let(let_) => {
796                        replace_local_id(vars, &mut replacements.borrow_mut(), &mut let_.id)
797                    }
798                    Expression::For(for_) => replace_local_id(
799                        vars,
800                        &mut replacements.borrow_mut(),
801                        &mut for_.index_variable,
802                    ),
803                    Expression::Match(match_) => {
804                        let mut replacements = replacements.borrow_mut();
805                        if let Some(replacement) = replacements.get(&match_.variable_to_match.0) {
806                            match_.variable_to_match.0 = *replacement;
807                        }
808                        for case in match_.cases.iter_mut() {
809                            for (arg, _) in case.arguments.iter_mut() {
810                                replace_local_id(vars, &mut replacements, arg);
811                            }
812                        }
813                    }
814                    _ => (),
815                }
816                (true, ())
817            },
818            &mut |_, _| {},
819            // Update the IDs in identifiers based on the replacements we remember from above.
820            &mut |ident| {
821                if let Definition::Local(id) = &mut ident.definition {
822                    if let Some(replacement) = replacements.borrow().get(id) {
823                        *id = *replacement;
824                    }
825                }
826            },
827        );
828    }
