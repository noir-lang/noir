19pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
20    let config = default_config(u)?;
21    let rules = rules::collect(&config);
22    let max_rewrites = 10;
23    let inputs = CompareMorph::arb(
24        u,
25        config,
26        |u, mut program| {
27            let options = CompareOptions::arbitrary(u)?;
28            rewrite_program(u, &mut program, &rules, max_rewrites);
29            Ok((program, options))
30        },
31        |program, options| {
32            compile_into_circuit_or_die(program, &options.onto(default_ssa_options()), None)
33        },
34    )?;
35
36    let result = inputs.exec()?;
37
38    compare_results_compiled(&inputs, &result)
39}

41fn rewrite_program(
42    u: &mut Unstructured,
43    program: &mut Program,
44    rules: &[rules::Rule],
45    max_rewrites: usize,
46) {
47    for func in program.functions.iter_mut() {
48        if func.name.ends_with("_proxy") {
49            continue;
50        }
51        rewrite_function(u, func, rules, max_rewrites);
52    }
53}
54
55fn rewrite_function(
56    u: &mut Unstructured,
57    func: &mut Function,
58    rules: &[rules::Rule],
59    max_rewrites: usize,
60) {
61    // We can call `rewrite::next_local_and_ident_id` and pass the results to the rewrite rules,
62    // if they want to add new variables with new local IDs.
63    let ctx = rules::Context { unconstrained: func.unconstrained, ..Default::default() };
64
65    let estimate = estimate_applicable_rules(&ctx, &func.body, rules);
66    let morph = MorphContext {
67        target: max_rewrites.min(estimate),
68        estimate,
69        count: Cell::new(0),
70        rules,
71        vars: RefCell::new(VariableContext::new(func)),
72    };
73
74    morph.rewrite_expr(&ctx, u, &mut func.body);
75}

77/// Context necessary to generate new local IDs during rewrites.
78///
79/// Potentially a place to reconstruct local variable scopes.
80struct VariableContext {
81    next_local_id: u32,
82    next_ident_id: u32,
83    locals: ScopeStack<LocalId>,
84}
85
86impl VariableContext {
87    fn new(func: &Function) -> Self {
88        let (next_local_id, next_ident_id) = rewrite::next_local_and_ident_id(func);
89
90        let locals = ScopeStack::from_variables(func.parameters.iter().map(
91            |(id, mutable, name, typ, _vis)| (*id, *mutable, name.clone(), typ.as_ref().clone()),
92        ));
93
94        Self { next_local_id, next_ident_id, locals }
95    }
96
97    fn next_local_id(&mut self) -> LocalId {
98        let id = self.next_local_id;
99        self.next_local_id += 1;
100        LocalId(id)
101    }
102
103    fn next_ident_id(&mut self) -> IdentId {
104        let id = self.next_ident_id;
105        self.next_ident_id += 1;
106        IdentId(id)
107    }
108}

110/// Recursively apply rules while keeping a tally on how many we have done.
111struct MorphContext<'a> {
112    /// Number of rewrites we want to achieve.
113    target: usize,
114    /// (Over)estimate of the maximum number we could hope to apply.
115    estimate: usize,
116    /// Number of rewrites applied so far, up to the `target`.
117    count: Cell<usize>,
118    /// Rules to apply.
119    rules: &'a [rules::Rule],
120    /// Book keeping for local variables.
121    vars: RefCell<VariableContext>,
122}
123
124impl MorphContext<'_> {
125    /// Check if we have reached the target.
126    fn limit_reached(&self) -> bool {
127        self.target == 0 || self.estimate == 0 || self.count.get() == self.target
128    }
129
130    fn rewrite_expr(&self, ctx: &rules::Context, u: &mut Unstructured, expr: &mut Expression) {
131        visit_expr_be_mut(
132            expr,
133            &mut |expr: &mut Expression| {
134                if self.limit_reached() {
135                    return (false, false);
136                }
137
138                // Check if we are entering a new scope.
139                let entered = matches!(
140                    expr,
141                    Expression::Block(_)
142                        | Expression::For(_)
143                        | Expression::Loop(_)
144                        | Expression::While(_)
145                        | Expression::If(_)
146                );
147
148                if entered {
149                    self.vars.borrow_mut().locals.enter();
150                }
151
152                // We apply the rules on this expression, but its
153                // children will be visited after this call.
154                let cont = self.apply_rules(ctx, u, expr);
155
156                (cont, entered)
157            },
158            &mut |expr, entered| {
159                // A `let` variable becomes visible *after* we have have processed all its children,
160                // so, only its siblings can see it.
161                if let Expression::Let(let_) = expr {
162                    let typ = let_.expression.return_type().expect("let should have a type");
163                    self.vars.borrow_mut().locals.add(
164                        let_.id,
165                        let_.mutable,
166                        let_.name.clone(),
167                        typ.into_owned(),
168                    );
169                }
170                if entered {
171                    self.vars.borrow_mut().locals.exit();
172                }
173            },
174            &mut |_| {},
175        );
176    }
177
178    fn apply_rules(
179        &self,
180        ctx: &rules::Context,
181        u: &mut Unstructured,
182        expr: &mut Expression,
183    ) -> bool {
184        match expr {
185            Expression::For(for_) => {
186                // Separate context for just the ranges.
187                let range_ctx = rules::Context { is_in_range: true, ..*ctx };
188                self.rewrite_expr(&range_ctx, u, &mut for_.start_range);
189                self.rewrite_expr(&range_ctx, u, &mut for_.end_range);
190                // Original context for the body.
191                self.vars.borrow_mut().locals.add(
192                    for_.index_variable,
193                    false,
194                    for_.index_name.clone(),
195                    for_.index_type.clone(),
196                );
197                self.rewrite_expr(ctx, u, &mut for_.block);
198                self.vars.borrow_mut().locals.remove(&for_.index_variable);
199                // No need to visit children, we just visited them.
200                false
201            }
202            Expression::Unary(
203                unary @ Unary { operator: UnaryOp::Reference { mutable: true }, .. },
204            ) => {
205                let ctx = rules::Context { is_in_ref_mut: true, ..*ctx };
206                self.rewrite_expr(&ctx, u, &mut unary.rhs);
207                false
208            }
209            Expression::Call(call) if is_special_call(call) => {
210                let ctx = rules::Context { is_in_special_call: true, ..*ctx };
211                for arg in call.arguments.iter_mut() {
212                    self.rewrite_expr(&ctx, u, arg);
213                }
214                false
215            }
216            // The rest can just have the rules applied on them, using the same context.
217            _ => {
218                for rule in self.rules {
219                    match self.try_apply_rule(ctx, u, expr, rule) {
220                        Ok(false) => {
221                            // We couldn't, or decided not to apply this rule; try the next one.
222                            continue;
223                        }
224                        Err(_) => {
225                            // We ran out of randomness; stop visiting the AST.
226                            return false;
227                        }
228
229                        Ok(true) => {
230                            // We applied a rule on this expression.
231                            self.count.set(self.count.get() + 1);
232                            // We could visit the children of this morphed expression, which could result in repeatedly applying
233                            // the same rule over and over again. When we have 100% application rate (e.g. a small function),
234                            // it would be wasting all the potential on the first rule that matched, e.g. `(x - (0 + (0 - 0)))`.
235                            // It would also throw off the estimate if we introduce new items on which we can apply rules.
236                            return false;
237                        }
238                    }
239                }
240                // If we made it this far, we did not apply any rule, so look deeper in the AST.
241                true
242            }
243        }
244    }
245
246    /// Check if a rule can be applied on an expression. If it can, apply it based on some arbitrary
247    /// criteria, returning a flag showing whether it was applied.
248    fn try_apply_rule(
249        &self,
250        ctx: &rules::Context,
251        u: &mut Unstructured,
252        expr: &mut Expression,
253        rule: &rules::Rule,
254    ) -> arbitrary::Result<bool> {
255        if rule.matches(ctx, expr) && u.ratio(self.target, self.estimate)? {
256            rule.rewrite(u, &mut self.vars.borrow_mut(), expr)?;
257            Ok(true)
258        } else {
259            Ok(false)
260        }
261    }
262}

264/// Provide a rough estimate for how many rules can be applied.
265///
266/// It will overestimate, because it ignores the finer points of when a rule can match.
267fn estimate_applicable_rules(
268    ctx: &rules::Context,
269    expr: &Expression,
270    rules: &[rules::Rule],
271) -> usize {
272    let mut count = 0;
273    visit_expr(expr, &mut |expr| {
274        for rule in rules {
275            if rule.matches(ctx, expr) {
276                count += 1;
277            }
278        }
279        true
280    });
281    count
282}

284/// Check if we are calling an oracle or builtin function.
285fn is_special_call(call: &Call) -> bool {
286    matches!(
287        call.func.as_ref(),
288        Expression::Ident(Ident {
289            definition: Definition::Oracle(_) | Definition::Builtin(_) | Definition::LowLevel(_),
290            ..
291        })
292    )
293}

332    /// Metamorphic transformation rule.
333    pub struct Rule {
334        pub matches: Box<MatchFn>,
335        pub rewrite: Box<RewriteFn>,
336    }

338    impl Rule {
339        pub fn new(
340            matches: impl Fn(&Context, &Expression) -> bool + 'static,
341            rewrite: impl Fn(
342                &mut Unstructured,
343                &mut VariableContext,
344                &mut Expression,
345            ) -> arbitrary::Result<()>
346            + 'static,
347        ) -> Self {
348            Self { matches: Box::new(matches), rewrite: Box::new(rewrite) }
349        }
350
351        /// Check if the rule can be applied on an expression.
352        pub fn matches(&self, ctx: &Context, expr: &Expression) -> bool {
353            (self.matches)(ctx, expr)
354        }

356        /// Apply the rule on an expression, mutating/replacing it in-place.
357        pub fn rewrite(
358            &self,
359            u: &mut Unstructured,
360            vars: &mut VariableContext,
361            expr: &mut Expression,
362        ) -> arbitrary::Result<()> {
363            (self.rewrite)(u, vars, expr)
364        }

367    /// Construct all rules that we can apply on a program.
368    pub fn collect(config: &Config) -> Vec<Rule> {
369        let mut rules = vec![
370            num_add_zero(),
371            num_sub_zero(),
372            num_mul_one(),
373            num_div_one(),
374            bool_or_self(),
375            bool_xor_self(),
376            bool_xor_rand(),
377            any_inevitable(),
378            int_break_up(),
379        ];
380        if config.avoid_overflow {
381            // When we can overflowing instruction, then swapping around the LHS and RHS
382            // of a binary operation can swap failures. We could visit the expressions to rule
383            // out a potential failure on both sides at the same time, or just skip this rule.
384            rules.push(num_commute());
385        }
386        rules
387    }

389    /// Transform any numeric value `x` into `x <op> <rhs>`
390    fn num_op(op: BinaryOpKind, rhs: u32) -> Rule {
391        Rule::new(num_rule_matches, move |_u, _locals, expr| {
392            let typ = expr.return_type().expect("only called on matching type").into_owned();
393            expr::replace(expr, |expr| expr::binary(expr, op, expr::int_literal(rhs, typ)));
394            Ok(())
395        })
396    }

418    /// Break an integer literal `a` into `b + c`.
419    pub fn int_break_up() -> Rule {
420        Rule::new(
421            |ctx, expr| {
422                if ctx.is_in_range && !ctx.unconstrained || ctx.is_in_ref_mut {
423                    return false;
424                }
425                matches!(expr, Expression::Literal(Literal::Integer(_, Type::Integer(_, _), _)))
426            },
427            |u, _locals, expr| {
428                let Expression::Literal(Literal::Integer(a, typ, loc)) = expr else {
429                    unreachable!("matched only integer literals, got {expr}");
430                };
431                let mut b_expr = expr::gen_literal(u, typ, &Config::default())?;
432                let Expression::Literal(Literal::Integer(b, _, _)) = &mut b_expr else {
433                    unreachable!("generated a literal of the same type");
434                };
435
436                let Type::Integer(sign, bits) = typ else {
437                    return Ok(());
438                };
439                let hir_type = HirType::Integer(*sign, *bits);
440
441                // Convert to Integer for correct signed/unsigned comparison.
442                // FieldElement ordering does not match signed integer ordering.
443                let Some(a_int) = Integer::try_from_type(*a, &hir_type) else {
444                    return Ok(());
445                };
446                let Some(b_int) = Integer::try_from_type(*b, &hir_type) else {
447                    return Ok(());
448                };
449
450                let (op, c) = if a_int >= b_int {
451                    (BinaryOpKind::Add, *a - *b)
452                } else {
453                    (BinaryOpKind::Subtract, *b - *a)
454                };
455
456                // Verify c fits in the type (modular subtraction can yield out-of-range values
457                // for signed integers, e.g. 100_i8 - (-28_i8) = 128 which overflows i8).
458                if Integer::try_from_type(c, &hir_type).is_none() {
459                    return Ok(());
460                }
461
462                let c_expr = Expression::Literal(Literal::Integer(c, typ.clone(), *loc));
463                *expr = expr::binary(b_expr, op, c_expr);
464                Ok(())
465            },
466        )
467    }

478    pub fn bool_xor_self() -> Rule {
479        Rule::new(bool_rule_matches, |_u, _locals, expr| {
480            expr::replace(expr, |expr| {
481                let rhs = expr::binary(expr.clone(), BinaryOpKind::Xor, expr.clone());
482                expr::binary(expr, BinaryOpKind::Xor, rhs)

488    /// Transform boolean value `x` into `rnd ^ x ^ rnd`.
489    pub fn bool_xor_rand() -> Rule {

492            let rnd = expr::gen_literal(u, &Type::Bool, &Config::default())?;

501    /// Transform commutative arithmetic operations:
502    /// * `a + b` into `b + a`
503    /// * `a * b` into `b * a`
504    pub fn num_commute() -> Rule {
505        Rule::new(
506            |_ctx, expr| {
507                matches!(
508                    expr,
509                    Expression::Binary(Binary {
510                        operator: BinaryOpKind::Add | BinaryOpKind::Multiply,
511                        ..
512                    })
513                ) && !has_side_effect(expr)
514            },
515            |_u, _locals, expr| {
516                let Expression::Binary(binary) = expr else {
517                    unreachable!("the rule only matches Binary expressions");
518                };
519
520                std::mem::swap(&mut binary.lhs, &mut binary.rhs);
521
522                Ok(())
523            },
524        )
525    }

527    /// Transform any expression into an if-then-else with the itself
528    /// repeated in the _then_ and _else_ branch:
529    /// * `x` into `if c { x } else { x }`
530    pub fn any_inevitable() -> Rule {
531        Rule::new(
532            |ctx, expr| {
533                !ctx.is_in_special_call
534                    && !ctx.is_in_ref_mut
535                    // If we're in ACIR then don't turn loop ranges into non-constant expressions.
536                    && (ctx.unconstrained || !ctx.is_in_range)
537                    // `let x = 1;` transformed into `if true { let x = 1; } else { let x = 1; }` would leave `x` undefined.
538                    && !matches!(expr, Expression::Let(_))
539                    // We can't return references from an `if` statement
540                    && expr.return_type().map(|typ| !types::contains_reference(typ.as_ref())).unwrap_or(true)
541            },
542            |u, vars, expr| {
543                let typ = expr.return_type().map(|typ| typ.into_owned()).unwrap_or(Type::Unit);
544
545                // Find a bool expression we can use. For simplicity just consider actual bool variables,
546                // not things that can produce variables, so we have less logic to repeat for the `FunctionContext`.
547                let bool_vars = vars
548                    .locals
549                    .current()
550                    .variables()
551                    .filter_map(|(id, (_, _, typ))| (*typ == Type::Bool).then_some(id))
552                    .collect::<Vec<_>>();
553
554                // If we don't have a bool variable, generate some random expression.
555                let cond = if bool_vars.is_empty() {
556                    gen_expr(u, &Type::Bool, 2)?
557                } else {
558                    let id = u.choose_iter(bool_vars)?;
559                    let (mutable, name, typ) = vars.locals.current().get_variable(id);
560                    Expression::Ident(Ident {
561                        location: None,
562                        definition: Definition::Local(*id),
563                        mutable: *mutable,
564                        name: name.clone(),
565                        typ: Rc::new(typ.clone()),
566                        id: vars.next_ident_id(),
567                    })
568                };
569
570                // Duplicate the expression, then assign new IDs to all variables created in it.
571                let mut alt = expr.clone();
572
573                reassign_ids(vars, &mut alt);
574
575                expr::replace(expr, |expr| expr::if_else(cond, expr, alt, typ));
576                Ok(())
577            },
578        )
579    }

581    /// Common match condition for boolean rules.
582    fn bool_rule_matches(ctx: &Context, expr: &Expression) -> bool {
583        // If we rewrite `&mut x` into `&mut (x | x)` we will alter the semantics.
584        if ctx.is_in_ref_mut {
585            return false;
586        }
587        // We don't want to mess with the arguments of a `println`, because the printer assumes they are bool literals.
588        // Similarly a `constrain` call is expected to have a single boolean expression.
589        if ctx.is_in_special_call {
590            return false;
591        }
592        // We can apply boolean rule on anything that returns a bool,
593        // unless the expression can have a side effect, which we don't want to duplicate.
594        if let Some(typ) = expr.return_type() {
595            matches!(typ.as_ref(), Type::Bool)
596                && !has_side_effect(expr)
597                && !expr::exists(expr, |expr| {
598                    matches!(
599                        expr,
600                        Expression::Let(_)     // Creating a variable needs a new ID
601                        | Expression::Match(_) // Match creates variables which would need new IDs
602                        | Expression::Block(_) // Applying logical operations on blocks would look odd
603                    )
604                })
605        } else {
606            false
607        }
608    }
609
610    /// Common condition for numeric rules
611    fn num_rule_matches(ctx: &Context, expr: &Expression) -> bool {
612        // Because of #8305 we can't reliably use expressions in ranges in ACIR.
613        if ctx.is_in_range && !ctx.unconstrained {
614            return false;
615        }
616        // If we rewrite `&mut x` into `&mut (x - 0)` we will alter the semantics.
617        if ctx.is_in_ref_mut {
618            return false;
619        }
620        // Appending 0 to a block would look odd.
621        if matches!(expr, Expression::Block(_)) {
622            return false;
623        }
624        // We can apply this rule on anything that returns a number.
625        if let Some(typ) = expr.return_type() {
626            matches!(typ.as_ref(), Type::Field | Type::Integer(_, _))
627        } else {
628            false
629        }
630    }

650    /// Check if an expression can have a side effect, in which case duplicating or reordering it could
651    /// change the behavior of the program. This doesn't concern about failures, just observable changes
652    /// the state of the program.
653    pub(super) fn has_side_effect(expr: &Expression) -> bool {
654        expr::exists(expr, |expr| {
655            matches!(
656                expr,
657                Expression::Call(_) // Functions can have side effects, maybe mutating some reference, printing
658                | Expression::Assign(_) // Assignment to a mutable variable could double up effects
659            )
660        })
661    }

663    /// Generate an arbitrary pure (free of side effects) expression, returning a specific type.
664    pub(super) fn gen_expr(
665        u: &mut Unstructured,
666        typ: &Type,
667        max_depth: usize,
668    ) -> arbitrary::Result<Expression> {
669        if max_depth > 0 {
670            let idx = u.choose_index(3)?;
671            if idx == 0
672                && let Some(expr) = gen_unary(u, typ, max_depth)?
673            {
674                return Ok(expr);
675            }
676            if idx == 1
677                && let Some(expr) = gen_binary(u, typ, max_depth)?
678            {
679                return Ok(expr);
680            }
681        }
682        expr::gen_literal(u, typ, &Config::default())
683    }
684
685    /// Generate an arbitrary unary expression, returning a specific type.
686    pub(super) fn gen_unary(
687        u: &mut Unstructured,
688        typ: &Type,
689        max_depth: usize,
690    ) -> arbitrary::Result<Option<Expression>> {
691        if !types::can_unary_return(typ) {
692            return Ok(None);
693        }
694        let mut make_unary = |op| {
695            let expr = gen_expr(u, typ, max_depth.saturating_sub(1))?;
696            Ok(Some(expr::unary(op, expr, typ.clone())))
697        };
698        if types::is_numeric(typ) {
699            make_unary(UnaryOp::Minus)
700        } else if types::is_bool(typ) {
701            make_unary(UnaryOp::Not)
702        } else {
703            Ok(None)
704        }
705    }
706
707    /// Generate an arbitrary binary expression, returning a specific type.
708    ///
709    /// The operation should not have any side effects, ie. it must not fail.
710    pub(super) fn gen_binary(
711        u: &mut Unstructured,
712        typ: &Type,
713        max_depth: usize,
714    ) -> arbitrary::Result<Option<Expression>> {
715        // Collect the operations can return the expected type.
716        // Do not introduce new errors in randomly generated code that only exists in the morph.
717        let ops = BinaryOp::iter()
718            .filter(|op| {
719                types::can_binary_op_return(op, typ)
720                    && !types::can_binary_op_overflow(op)
721                    && !types::can_binary_op_err_by_zero(op)
722            })
723            .collect::<Vec<_>>();
724
725        // Ideally we checked that the target type can be returned, but just in case.
726        if ops.is_empty() {
727            return Ok(None);
728        }
729
730        // Choose a random operation.
731        let op = u.choose_iter(ops)?;
732
733        // let type_options = TYPES.get_or_init(|| {
734        //     let mut types = vec![Type::Bool, Type::Field];
735
736        //     for sign in [Signedness::Signed, Signedness::Unsigned] {
737        //         for size in IntegerBitSize::iter() {
738        //             if sign.is_signed() && size.bit_size() == 1 {
739        //                 continue;
740        //             }
741        //             // Avoid negative literals; the frontend makes them difficult to work with in expressions
742        //             // where no type inference information is available.
743        //             if sign.is_signed() {
744        //                 continue;
745        //             }
746        //             // Avoid large integers; frontend doesn't like them.
747        //             if size.bit_size() > 32 {
748        //                 continue;
749        //             }
750        //             types.push(Type::Integer(sign, size));
751        //         }
752        //     }
753        //     types
754        // });
755
756        TYPES.with(|types| {
757            // Select input types that can produce the output we want.
758            let type_options = types
759                .iter()
760                .filter(|input| types::can_binary_op_return_from_input(&op, input, typ))
761                .collect::<Vec<_>>();
762            // Choose a type for the LHS and RHS.
763            let lhs_type = u.choose_iter(type_options)?;
764
765            // Generate expressions for LHS and RHS.
766            let lhs_expr = gen_expr(u, lhs_type, max_depth.saturating_sub(1))?;
767            let rhs_expr = gen_expr(u, lhs_type, max_depth.saturating_sub(1))?;
768
769            let mut expr = expr::binary(lhs_expr, op, rhs_expr);
770
771            // If we have chosen e.g. u8 and need u32 we need to cast.
772            if !(lhs_type == typ || types::is_bool(typ) && op.is_comparator()) {
773                expr = expr::cast(expr, typ.clone());
774            }
775
776            Ok(Some(expr))
777        })
778    }
779
780    thread_local! {
781        /// Types we can consider using in this context.
782        static TYPES: Vec<Type> = {
783            let mut types = vec![Type::Bool, Type::Field];
784
785            for sign in [Signedness::Signed, Signedness::Unsigned] {
786                for size in IntegerBitSize::iter() {
787                    if sign.is_signed() && size.bit_size() == 1 {
788                        continue;
789                    }
790                    // Avoid negative literals; the frontend makes them difficult to work with in expressions
791                    // where no type inference information is available.
792                    if sign.is_signed() {
793                        continue;
794                    }
795                    // Avoid large integers; frontend doesn't like them.
796                    if size.bit_size() > 32 {
797                        continue;
798                    }
799                    types.push(Type::Integer(sign, size));
800                }
801            }
802            types
803        };
804    }

806    /// Assign new IDs to variables and identifiers created in the expression.
807    pub(super) fn reassign_ids(vars: &mut VariableContext, expr: &mut Expression) {
808        fn replace_local_id(
809            vars: &mut VariableContext,
810            replacements: &mut HashMap<LocalId, LocalId>,
811            id: &mut LocalId,
812        ) {
813            let curr = *id;
814            let next = vars.next_local_id();
815            replacements.insert(curr, next);
816            *id = next;
817        }
818
819        let replacements = RefCell::new(HashMap::new());
820
821        visit_expr_be_mut(
822            expr,
823            // Assign a new ID where variables are created, and remember what original value they replaced.
824            &mut |expr| {
825                match expr {
826                    Expression::Ident(ident) => {
827                        ident.id = vars.next_ident_id();
828                    }
829                    Expression::Let(let_) => {
830                        replace_local_id(vars, &mut replacements.borrow_mut(), &mut let_.id)
831                    }
832                    Expression::For(for_) => replace_local_id(
833                        vars,
834                        &mut replacements.borrow_mut(),
835                        &mut for_.index_variable,
836                    ),
837                    Expression::Match(match_) => {
838                        let mut replacements = replacements.borrow_mut();
839                        if let Some(replacement) = replacements.get(&match_.variable_to_match.0) {
840                            match_.variable_to_match.0 = *replacement;
841                        }
842                        for case in match_.cases.iter_mut() {
843                            for (arg, _) in case.arguments.iter_mut() {
844                                replace_local_id(vars, &mut replacements, arg);
845                            }
846                        }
847                    }
848                    _ => (),
849                }
850                (true, ())
851            },
852            &mut |_, _| {},
853            // Update the IDs in identifiers based on the replacements we remember from above.
854            &mut |ident| {
855                if let Definition::Local(id) = &mut ident.definition
856                    && let Some(replacement) = replacements.borrow().get(id)
857                {
858                    *id = *replacement;
859                }
860            },
861        );
862    }
