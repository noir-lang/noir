1//! Verify that every SSA pass produces well-formed SSA.
2//!
3//! 1. generate an arbitrary AST
4//! 2. lower it to the initial SSA
5//! 3. run the primary pass pipeline one pass at a time
6//! 4. validate the SSA after each pass
7//!
8//! A failure means some pass turned valid SSA into malformed SSA — e.g. a pass
9//! that produces a `truncate` following a signed-integer `unchecked_sub`, which
10//! [`noirc_evaluator::ssa::ssa_gen::validate_ssa`] rejects. This is a bug
11//! independent of whether the interpreter or the backends happen to cope with it.
12//!
13//! This complements [`super::pass_vs_prev`], which checks that passes preserve
14//! *behavior*; here we check that they preserve *well-formed-ness*. Running the
15//! full validator after every pass is too costly to leave on for every
16//! compilation, but is affordable in a fuzzer.
17use crate::default_ssa_options;
18use arbitrary::Unstructured;
19use color_eyre::eyre;
20use noir_ast_fuzzer::{Config, DisplayAstAsNoir, arb_program};
21use noirc_evaluator::ssa::primary_passes;
22use noirc_evaluator::ssa::ssa_gen;
23use noirc_frontend::monomorphization::ast::Program;
24
25pub fn fuzz(u: &mut Unstructured) -> eyre::Result<()> {
26    let config = Config { avoid_overflow: u.arbitrary()?, ..Config::default() };
27    let program = arb_program(u, config)?;
28
29    let result = validate_each_pass(&program);
30    if result.is_err() {
31        // Show the AST as Noir so the failure can be replicated with other `nargo` tools.
32        eprintln!("---\nAST:\n{}", DisplayAstAsNoir(&program));
33    }
34    result
35}
36
37fn validate_each_pass(program: &Program) -> eyre::Result<()> {
38    let ssa_options = default_ssa_options();
39    let passes = primary_passes(&ssa_options);
40
41    // `generate_ssa` already validates the initial SSA, so we only check each pass's output.
42    // Clone so `program` survives for the AST printout on failure.
43    let mut ssa = ssa_gen::generate_ssa(program.clone()).expect("failed to generate initial SSA");
44
45    for pass in &passes {
46        ssa = pass.run(ssa).map_err(|e| eyre::eyre!("pass '{}' failed to run: {e}", pass.msg()))?;
47        // Prints the SSA when `NOIR_SHOW_INVALID_SSA` is set.
48        ssa = ssa_gen::validate_ssa_or_err(ssa, false)
49            .map_err(|e| eyre::eyre!("SSA invalid after '{}': {e}", pass.msg()))?;
50    }
51
52    Ok(())
53}
