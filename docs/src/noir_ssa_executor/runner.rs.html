14/// Errors that can occur during execution of the program
15/// It can be NargoError or rust panic
16#[derive(Debug, Error)]
17pub enum SsaExecutionError {
18    #[error("Execution failed: {0}")]
19    ExecutionFailed(NargoError<FieldElement>),
20    #[error("Execution panicked: {0}")]
21    ExecutionPanicked(String),
22    #[error("SSA compilation failed: {0}")]
23    SsaCompilationFailed(String),
24    #[error("SSA parsing failed: {0}")]
25    SsaParsingFailed(String),
26}

28/// Low level function to execute the given program with the given initial witness
29/// It uses nargo execute_program to run the program
30fn execute<B: BlackBoxFunctionSolver<FieldElement> + Default>(
31    program: &Program<FieldElement>,
32    initial_witness: WitnessMap<FieldElement>,
33) -> Result<WitnessStack<FieldElement>, NargoError<FieldElement>> {
34    execute_program(
35        program,
36        initial_witness.clone(),
37        &B::default(),
38        &mut DefaultForeignCallBuilder::default().build(),
39    )
40}

42/// High level function to execute the given program with the given initial witness
43/// It returns the result of the program execution
44pub fn execute_single(
45    program: &Program<FieldElement>,
46    initial_witness: WitnessMap<FieldElement>,
47) -> Result<WitnessStack<FieldElement>, SsaExecutionError> {
48    // Save the current panic hook
49    let previous_hook = std::panic::take_hook();
50    let panic_message = std::sync::Arc::new(std::sync::Mutex::new(String::new()));
51    let hook_message = panic_message.clone();
52
53    // Set a custom panic hook to capture panic messages
54    std::panic::set_hook(Box::new(move |panic_info| {
55        let message = if let Some(s) = panic_info.payload().downcast_ref::<&str>() {
56            format!("Panic: {s}")
57        } else if let Some(s) = panic_info.payload().downcast_ref::<String>() {
58            format!("Panic: {s}")
59        } else {
60            format!("Unknown panic: {panic_info:?}")
61        };
62
63        if let Some(location) = panic_info.location() {
64            let loc_info = format!(" at {}:{}", location.file(), location.line());
65            *hook_message.lock().unwrap() = message + &loc_info;
66        } else {
67            *hook_message.lock().unwrap() = message;
68        }
69    }));
70
71    let result =
72        std::panic::catch_unwind(|| execute::<Bn254BlackBoxSolver>(program, initial_witness));
73
74    // Restore the previous panic hook
75    std::panic::set_hook(previous_hook);
76
77    match result {
78        Ok(result) => match result {
79            Ok(result) => Ok(result),
80            Err(e) => Err(SsaExecutionError::ExecutionFailed(e)),
81        },
82        Err(_) => {
83            let error_msg = panic_message.lock().unwrap().clone();
84            if error_msg.is_empty() {
85                Err(SsaExecutionError::ExecutionPanicked("Unknown error".to_string()))
86            } else {
87                Err(SsaExecutionError::ExecutionPanicked(error_msg))
88            }
89        }
90    }
91}
