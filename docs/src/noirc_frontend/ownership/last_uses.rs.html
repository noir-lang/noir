1//! This module contains the last use analysis pass which is run on each function before
2//! the ownership pass.
3//!
4//! The purpose of this pass is to find which instance of a variable is the variable's
5//! last use. Note that a variable may have multiple last uses. This can happen if the
6//! variable's last use is within an `if` expression or similar. It could be last used
7//! in one place in the `then` branch and in another place in the `else` branch as long
8//! as no code after the `if` expression uses the same variable.
9//!
10//! This pass works by traversing the expression tree in **reverse** order. The first
11//! encounter of each variable (in reverse) is its last use. For if/match branches,
12//! each branch is processed independently (starting from a shared "seen" set), so a
13//! variable can accumulate multiple last uses — one per branch.
14//!
15//! Loop handling: variables declared outside a loop cannot be moved inside it (the loop
16//! may execute multiple times). When leaving a loop, any pending last uses for variables
17//! declared outside the loop are removed.
18//!
19//! Assignment handling: when `x = expr` is encountered (in reverse), `x` is removed from
20//! the "seen" set so that code before the assignment can independently identify the last
21//! use of the old value of `x`.
22//!
23//! This pass is not sophisticated with regard to struct and tuple fields. It currently
24//! ignores these entirely and counts each use as a use of the entire variable. This is an
25//! area for future optimization. E.g. the program `a.b.c; a.e.f` will result in `a` being
26//! cloned in its entirety in the first statement. Note that this is lessened in the overall
27//! ownership pass such that only `.c` is cloned but it is still an area for improvement.
28
29use crate::ast::UnaryOp;
30use crate::monomorphization::ast::{self, Definition, IdentId, LocalId};
31use crate::monomorphization::ast::{Expression, Function, Literal};
32use rustc_hash::{FxHashMap as HashMap, FxHashSet as HashSet};

34use super::Context;

36struct LastUseContext {
37    /// Variables already encountered in the reverse traversal.
38    /// The first encounter of a variable (in reverse) is its last use in forward order.
39    seen: HashSet<LocalId>,
40
41    /// Accumulated last uses for each variable. May be truncated on loop exit
42    /// for variables declared outside the loop.
43    pending_last_uses: HashMap<LocalId, Vec<IdentId>>,
44
45    /// Last uses confirmed by an assignment (`x = f(x)`). The use of `x` in `f(x)` is
46    /// confirmed because the assignment kills the old value — these survive loop truncation.
47    confirmed_moves: HashMap<LocalId, Vec<IdentId>>,
48
49    /// Variables unconditionally reassigned in the current scope.
50    ///
51    /// A variable is in this set only if it is reassigned on ALL paths through the
52    /// current scope. The `if`/`match` handlers enforce this via intersection of each
53    /// branch's killed set.
54    ///
55    /// Used by loop truncation: normally, pending last uses for variables declared
56    /// outside a loop are truncated (the loop may execute multiple times, so moving
57    /// would consume the value). But if a variable is killed inside the loop, its value
58    /// is freshly created and consumed each iteration, so pending uses can survive.
59    killed: HashSet<LocalId>,
60
61    /// Loop depth at which each variable was declared.
62    /// 0 = function body (not in any loop).
63    declaration_depth: HashMap<LocalId, usize>,
64
65    /// Current loop nesting depth.
66    loop_depth: usize,
67
68    /// Variables that have been aliased via a reference expression (`&var` or `&mut var`).
69    ///
70    /// When a variable is referenced, any subsequent copy of it must be a clone (not a move).
71    /// This is because the reference creates an invisible alias: if the variable were moved
72    /// (sharing the same array pointer with refcount=1), a later write through the reference
73    /// would mutate the "moved" copy in place (bypassing copy-on-write semantics), since the
74    /// refcount would be 1 and no COW would be triggered.
75    ///
76    /// By preventing moves of aliased variables, we ensure that subsequent copies increment
77    /// the refcount, so that writes through the reference correctly trigger COW.
78    referenced_variables: HashSet<LocalId>,
79}
80
81impl Context {
82    /// Traverse the given function and return the last use(s) of each local variable.
83    /// A variable may have multiple last uses if it was last used within a conditional expression.
84    pub(super) fn find_last_uses_of_variables(
85        function: &Function,
86    ) -> HashMap<LocalId, Vec<IdentId>> {
87        let mut context = LastUseContext {
88            seen: HashSet::default(),
89            pending_last_uses: HashMap::default(),
90            confirmed_moves: HashMap::default(),
91            declaration_depth: HashMap::default(),
92            loop_depth: 0,
93            killed: HashSet::default(),
94            referenced_variables: HashSet::default(),
95        };
96
97        for (parameter, ..) in &function.parameters {
98            context.declare_variable(*parameter);
99        }
100        context.find_last_uses_in_expression(&function.body);
101        context.into_variables_to_move()
102    }
103}
104
105impl LastUseContext {
106    fn declare_variable(&mut self, id: LocalId) {
107        self.declaration_depth.insert(id, self.loop_depth);
108    }
109
110    /// Record a use of a local variable. If this is the first encounter in the
111    /// reverse traversal (i.e. the last use in forward order), add it to pending_last_uses.
112    fn use_variable(&mut self, id: LocalId, ident_id: IdentId) {
113        if self.seen.insert(id) {
114            self.pending_last_uses.entry(id).or_default().push(ident_id);
115        }
116    }
117
118    /// Collect all last uses, excluding variables that have been aliased via references.
119    fn into_variables_to_move(self) -> HashMap<LocalId, Vec<IdentId>> {
120        let mut moves = self.confirmed_moves;
121        for (id, uses) in self.pending_last_uses {
122            if !uses.is_empty() {
123                moves.entry(id).or_default().extend(uses);
124            }
125        }
126        moves.retain(|id, _| !self.referenced_variables.contains(id));
127        moves
128    }
129
130    // --- Expression traversal (reverse order) ---
131    //
132    // Sub-expressions are processed in reverse of their forward evaluation order
133    // so that the first encounter in our traversal corresponds to the last use.
134
135    fn find_last_uses_in_expression(&mut self, expr: &Expression) {
136        match expr {
137            Expression::Ident(ident) => self.find_last_uses_in_ident(ident),
138            Expression::Literal(literal) => self.find_last_uses_in_literal(literal),
139            Expression::Block(exprs) => {
140                for expr in exprs.iter().rev() {
141                    self.find_last_uses_in_expression(expr);
142                }
143            }
144            Expression::Unary(unary) => self.find_last_uses_in_unary(unary),
145            Expression::Binary(binary) => {
146                self.find_last_uses_in_expression(&binary.rhs);
147                self.find_last_uses_in_expression(&binary.lhs);
148            }
149            Expression::Index(index) => {
150                // SSA codegen evaluates the index before the collection, so in forward
151                // order the collection is used *after* the index. Reverse that here: visit
152                // the collection first so it (not the index) becomes the last use.
153                self.find_last_uses_in_expression(&index.collection);
154                self.find_last_uses_in_expression(&index.index);
155            }
156            Expression::Cast(cast) => self.find_last_uses_in_expression(&cast.lhs),
157            Expression::For(for_expr) => self.find_last_uses_in_for(for_expr),
158            Expression::Loop(body) => self.find_last_uses_in_loop_body(&[body]),
159            Expression::While(while_expr) => {
160                // Both condition and body are re-evaluated each iteration
161                self.find_last_uses_in_loop_body(&[&while_expr.body, &while_expr.condition]);
162            }
163            Expression::If(if_expr) => self.find_last_uses_in_if(if_expr),
164            Expression::Match(match_expr) => self.find_last_uses_in_match(match_expr),
165            Expression::Tuple(elements) => {
166                for elem in elements.iter().rev() {
167                    self.find_last_uses_in_expression(elem);
168                }
169            }
170            Expression::ExtractTupleField(tuple, _) => {
171                self.find_last_uses_in_expression(tuple);
172            }
173            Expression::Call(call) => self.find_last_uses_in_call(call),
174            Expression::Let(let_expr) => self.find_last_uses_in_let(let_expr),
175            Expression::Constrain(boolean, _, msg) => {
176                if let Some(msg) = msg {
177                    self.find_last_uses_in_expression(&msg.0);
178                }
179                self.find_last_uses_in_expression(boolean);
180            }
181            Expression::Assign(assign) => self.find_last_uses_in_assign(assign),
182            Expression::Semi(expr) => self.find_last_uses_in_expression(expr),
183            Expression::Clone(_) => unreachable!("last_uses is called before clones are inserted"),
184            Expression::Drop(_) => unreachable!("last_uses is called before drops are inserted"),
185            Expression::Break | Expression::Continue => {
186                // break/continue can skip assignments that were processed earlier in
187                // the reverse traversal, making those assignments conditional.
188                self.killed.clear();
189            }
190        }
191    }
192
193    fn find_last_uses_in_ident(&mut self, ident: &ast::Ident) {
194        if let Definition::Local(local_id) = &ident.definition {
195            self.use_variable(*local_id, ident.id);
196        }
197    }
198
199    fn find_last_uses_in_literal(&mut self, literal: &Literal) {
200        match literal {
201            Literal::Integer(..) | Literal::Bool(_) | Literal::Unit | Literal::Str(_) => (),
202            Literal::FmtStr(_, _, captures) => self.find_last_uses_in_expression(captures),
203            Literal::Array(array) | Literal::Vector(array) => {
204                for element in array.contents.iter().rev() {
205                    self.find_last_uses_in_expression(element);
206                }
207            }
208            Literal::Repeated { element, .. } => self.find_last_uses_in_expression(element),
209        }
210    }
211
212    fn find_last_uses_in_unary(&mut self, unary: &ast::Unary) {
213        if matches!(unary.operator, UnaryOp::Reference { .. }) {
214            // When a reference is taken to a local variable or one of its fields (e.g. `&mut x`
215            // or `&mut x.field`), the variable `x` is now aliased. Mark it so that any future
216            // copy of `x` must clone rather than move.
217            if let Some(local_id) = base_ident_of_field_access(&unary.rhs) {
218                self.referenced_variables.insert(local_id);
219            }
220        }
221        self.find_last_uses_in_expression(&unary.rhs);
222    }
223
224    fn find_last_uses_in_for(&mut self, for_expr: &ast::For) {
225        // The body is inside the loop scope; start/end ranges are evaluated once before
226        // the loop begins and are tracked outside the loop scope.
227        self.find_last_uses_in_loop_body(&[&for_expr.block]);
228        self.find_last_uses_in_expression(&for_expr.end_range);
229        self.find_last_uses_in_expression(&for_expr.start_range);
230    }
231
232    /// Process loop body expressions at an increased loop depth.
233    /// After processing, remove any pending last uses for variables declared outside the loop
234    /// since they cannot be safely moved inside a loop that may execute multiple times.
235    fn find_last_uses_in_loop_body(&mut self, body_exprs: &[&Expression]) {
236        let pending_lengths: HashMap<LocalId, usize> =
237            self.pending_last_uses.iter().map(|(id, uses)| (*id, uses.len())).collect();
238        let saved_seen = self.seen.clone();
239
240        // Kills inside the loop don't propagate outward (the loop may execute 0 times).
241        let saved_killed = std::mem::take(&mut self.killed);
242
243        let loop_body_depth = self.loop_depth + 1;
244        self.loop_depth = loop_body_depth;
245        for expr in body_exprs.iter().rev() {
246            self.find_last_uses_in_expression(expr);
247        }
248        self.loop_depth = loop_body_depth - 1;
249
250        // Variables declared outside this loop cannot be moved inside it — unless
251        // they are unconditionally reassigned (killed) within the loop body, in which
252        // case their value is freshly created and consumed each iteration.
253        for (id, uses) in &mut self.pending_last_uses {
254            let decl_depth = self.declaration_depth.get(id).copied().unwrap_or(0);
255            if decl_depth < loop_body_depth && !self.killed.contains(id) {
256                let before_len = pending_lengths.get(id).copied().unwrap_or(0);
257                uses.truncate(before_len);
258            }
259        }
260        // Reinsert anything we have seen before, so nothing before the loop becomes a new last use.
261        for id in &saved_seen {
262            let decl_depth = self.declaration_depth.get(id).copied().unwrap_or(0);
263            if decl_depth < loop_body_depth {
264                self.seen.insert(*id);
265            }
266        }
267        self.killed = saved_killed;
268    }
269
270    fn find_last_uses_in_if(&mut self, if_expr: &ast::If) {
271        let saved_seen = self.seen.clone();
272        let saved_killed = self.killed.clone();
273
274        // Process the then-branch
275        self.find_last_uses_in_expression(&if_expr.consequence);
276        let mut merged = std::mem::replace(&mut self.seen, saved_seen.clone());
277        let then_killed = std::mem::replace(&mut self.killed, saved_killed);
278
279        // Process the else-branch, or use saved_seen for the implicit "do nothing" path
280        if let Some(alt) = &if_expr.alternative {
281            self.find_last_uses_in_expression(alt);
282            merged.extend(&self.seen);
283        } else {
284            merged.extend(&saved_seen);
285        }
286
287        // A variable is killed only if it is reassigned on ALL paths
288        self.killed.retain(|id| then_killed.contains(id));
289
290        self.seen = merged;
291
292        // The condition is evaluated before either branch
293        self.find_last_uses_in_expression(&if_expr.condition);
294    }
295
296    fn find_last_uses_in_match(&mut self, match_expr: &ast::Match) {
297        // Note: We don't track `variable_to_match` as a use here because it's just a LocalId
298        // that references a variable defined earlier. The last-use analysis for that variable
299        // happens at its actual use sites.
300        let saved_seen = self.seen.clone();
301        let saved_killed = self.killed.clone();
302        let mut merged = HashSet::default();
303        let mut all_killed: Option<HashSet<LocalId>> = None;
304
305        for case in &match_expr.cases {
306            self.seen = saved_seen.clone();
307            self.killed = saved_killed.clone();
308            for (argument, _) in &case.arguments {
309                self.declare_variable(*argument);
310            }
311            self.find_last_uses_in_expression(&case.branch);
312            merged.extend(&self.seen);
313            match &mut all_killed {
314                None => all_killed = Some(self.killed.clone()),
315                Some(acc) => acc.retain(|id| self.killed.contains(id)),
316            }
317        }
318
319        if let Some(default_case) = &match_expr.default_case {
320            self.seen = saved_seen;
321            self.killed = saved_killed.clone();
322            self.find_last_uses_in_expression(default_case);
323            merged.extend(&self.seen);
324            match &mut all_killed {
325                None => all_killed = Some(self.killed.clone()),
326                Some(acc) => acc.retain(|id| self.killed.contains(id)),
327            }
328        } else {
329            // No default case: conservatively include saved_seen
330            merged.extend(&saved_seen);
331            // No default path kills nothing new
332            if let Some(acc) = &mut all_killed {
333                acc.retain(|id| saved_killed.contains(id));
334            }
335        }
336
337        self.seen = merged;
338        self.killed = all_killed.unwrap_or(saved_killed);
339    }

341    fn find_last_uses_in_call(&mut self, call: &ast::Call) {
342        // A reference passed directly as a call argument (e.g. `foo(&mut x)`) is temporary:
343        // it only lives for the duration of the call. After the call returns, `x` is no longer
344        // aliased, so future copies of `x` don't need to clone.
345        //
346        // We must fall back to conservative (mark `x` as aliased) if the reference could escape:
347        // 1. The call returns a reference type — the passed reference might be returned.
348        // 2. Another argument has type `&mut T` where `T` contains a reference — the function
349        //    could write the passed reference into `*that_arg`, making it escape without returning.
350        let conservative = type_contains_reference(&call.return_type)
351            || call.arguments.iter().any(arg_can_store_reference);
352
353        for arg in call.arguments.iter().rev() {
354            if !conservative
355                && let Expression::Unary(unary) = arg
356                && matches!(unary.operator, UnaryOp::Reference { .. })
357                && base_ident_of_field_access(&unary.rhs).is_some()
358            {
359                // Track the use of the variable inside the reference (for last-use analysis)
360                // but skip the unary handler, which would mark the variable as aliased.
361                self.find_last_uses_in_expression(&unary.rhs);
362            } else {
363                self.find_last_uses_in_expression(arg);
364            }
365        }
366        self.find_last_uses_in_expression(&call.func);
367    }
368
369    fn find_last_uses_in_let(&mut self, let_expr: &ast::Let) {
370        self.declare_variable(let_expr.id);
371        self.find_last_uses_in_expression(&let_expr.expression);
372    }

374    fn find_last_uses_in_assign(&mut self, assign: &ast::Assign) {
375        // See if we are reassigning a variable, killing the reference to its previous value.
376        // Remove it from `seen` so that earlier code can independently identify the last use
377        // of the old value.
378        if let ast::LValue::Ident(ast::Ident { definition: Definition::Local(local_id), .. }) =
379            &assign.lvalue
380        {
381            self.seen.remove(local_id);
382            let pending_before = self.pending_last_uses.get(local_id).map_or(0, |v| v.len());
383
384            self.find_last_uses_in_expression(&assign.expression);
385
386            // Any uses of the variable added while processing the RHS (e.g. `x = f(x)`)
387            // are confirmed as last uses of the old value being killed by this assignment.
388            // Confirmed moves survive loop-exit truncation.
389            if let Some(uses) = self.pending_last_uses.get_mut(local_id)
390                && uses.len() > pending_before
391            {
392                let confirmed: Vec<_> = uses.drain(pending_before..).collect();
393                self.confirmed_moves.entry(*local_id).or_default().extend(confirmed);
394            } else {
395                // x does not appear in RHS: fresh value, safe to treat as killed
396                self.killed.insert(*local_id);
397            }
398            return;
399        }
400
401        // For compound lvalues (e.g. `a[i] = expr`), process in reverse evaluation order:
402        // the lvalue is accessed after the RHS is evaluated.
403        self.find_last_uses_in_lvalue(&assign.lvalue, false);
404        self.find_last_uses_in_expression(&assign.expression);
405    }
406
407    /// A variable in an lvalue position is never moved (otherwise you wouldn't
408    /// be able to access the variable you assigned to afterward). However, the
409    /// index in an array expression `a[i] = ...` is an arbitrary expression that
410    /// is actually in an rvalue position and can thus be moved.
411    ///
412    /// The `nested` parameter indicates whether this lvalue is nested inside another lvalue.
413    /// For top-level identifiers there's nothing to track, but for an identifier happening
414    /// as part of an index (`ident[index] = ...`) we do want to consider `ident` as used,
415    /// which should preclude any previous last uses that could result in it being moved.
416    fn find_last_uses_in_lvalue(&mut self, lvalue: &ast::LValue, nested: bool) {
417        match lvalue {
418            ast::LValue::Ident(ident) => {
419                if nested {
420                    self.find_last_uses_in_ident(ident);
421                }
422            }
423            ast::LValue::Index { array, index, .. } => {
424                // As in the rvalue Index case, SSA codegen evaluates the index before
425                // touching the array in an lvalue position, so visit the array first in
426                // the reverse traversal to make it the last use.
427                self.find_last_uses_in_lvalue(array, true);
428                self.find_last_uses_in_expression(index);
429            }
430            ast::LValue::MemberAccess { object, .. } => {
431                self.find_last_uses_in_lvalue(object, true);
432            }
433            ast::LValue::Dereference { reference, .. } => {
434                self.find_last_uses_in_lvalue(reference, true);
435            }
436            ast::LValue::Clone(_) => {
437                unreachable!("LValue::Clone should only be inserted by the ownership pass")
438            }
439        }
440    }
441}

443/// Given an expression that is the operand of a reference (`&expr` or `&mut expr`),
444/// walk through any chain of struct-field accesses (`expr.field` = `ExtractTupleField`)
445/// and return the `LocalId` of the base variable, if it is a local variable.
446///
447/// For example:
448/// - `&mut x`         → `Some(x_id)`
449/// - `&mut x.field`   → `Some(x_id)`  (field is `ExtractTupleField(x, _)`)
450/// - `&mut x.a.b`     → `Some(x_id)`
451/// - `&mut some_call()` → `None`
452fn base_ident_of_field_access(expr: &Expression) -> Option<LocalId> {
453    match expr {
454        Expression::Ident(ident) => {
455            if let Definition::Local(local_id) = ident.definition {
456                Some(local_id)
457            } else {
458                None
459            }
460        }
461        Expression::ExtractTupleField(inner, _) => base_ident_of_field_access(inner),
462        _ => None,
463    }
464}

466/// Returns `true` if `arg`'s type can be used to store a reference, i.e. the type
467/// contains a `&mut T` (at any depth) where `T` itself contains a reference.
468///
469/// When this is true for any argument, all `&mut x` arguments in the call must conservatively
470/// be treated as aliasing `x`, because the callee might write the reference into the location
471/// reachable through that argument.
472fn arg_can_store_reference(arg: &Expression) -> bool {
473    match arg.return_type() {
474        Some(typ) => type_can_store_reference(&typ),
475        None => true,
476    }
477}
478
479/// Returns `true` if `typ` contains — at any depth — a `&mut T` where `T` itself contains
480/// a reference. Such a type allows a reference to be written somewhere persistent.
481fn type_can_store_reference(typ: &ast::Type) -> bool {
482    use ast::Type;
483    match typ {
484        Type::Reference(inner, true /* mutable */) => type_contains_reference(inner),
485        Type::Reference(inner, false) => type_can_store_reference(inner),
486        Type::Tuple(elements) => elements.iter().any(type_can_store_reference),
487        Type::Array(_, elem) | Type::Vector(elem) | Type::FmtString(_, elem) => {
488            type_can_store_reference(elem)
489        }
490        Type::Function(args, ret, env, _) => {
491            args.iter().any(type_can_store_reference)
492                || type_can_store_reference(ret)
493                || type_can_store_reference(env)
494        }
495        Type::Field | Type::Integer(..) | Type::Bool | Type::String(..) | Type::Unit => false,
496    }
497}

499/// Returns `true` if the type contains a `Reference` anywhere (directly or nested within
500/// tuples, arrays, or function types). Used to decide whether a call might return a
501/// reference that aliases a variable passed by `&mut` to that call.
502fn type_contains_reference(typ: &ast::Type) -> bool {
503    use ast::Type;
504    match typ {
505        Type::Reference(..) => true,
506        Type::Tuple(elements) => elements.iter().any(type_contains_reference),
507        Type::Array(_, elem) | Type::Vector(elem) | Type::FmtString(_, elem) => {
508            type_contains_reference(elem)
509        }
510        Type::Function(args, ret, env, _) => {
511            args.iter().any(type_contains_reference)
512                || type_contains_reference(ret)
513                || type_contains_reference(env)
514        }
515        Type::Field | Type::Integer(..) | Type::Bool | Type::String(..) | Type::Unit => false,
516    }
