1//! This module implements the "ownership analysis" compiler pass on the
2//! monomorphized AST. It is run after monomorphization and before SSA-gen.
3//! At this point the monomorphized AST has no polymorphic types and all functions
4//! are specialized into either constrained or unconstrained versions. This pass
5//! only operates on unconstrained functions since only Noir's unconstrained runtime Brillig
6//! has any notion of cloning a value - specifically arrays with their brillig-only reference counts.
7//!
8//! Note that this documentation may refer to cloning a value or incrementing an array's reference
9//! count. These operations are equivalent on arrays. Cloning may be applied to any value and only
10//! increments the reference counts of any arrays contained within (but not behind references or
11//! inside nested arrays). This document also focuses on arrays but all reference count operations
12//! on arrays are also performed on vectors.
13//!
14//! Arrays in brillig have copy on write semantics which relies on us incrementing their
15//! reference counts when they are shared in multiple places. Note that while Noir has references,
16//! arrays can also be shared by value and we want to avoid clones when possible. This pass
17//! clones arrays (increments their reference counts) in the following situations which roughly
18//! correspond to where a `Copy` variable in Rust would be copied:
19//! - Variables are copied on each use, except for the last use where they are moved.
20//!   - If a variable's last use is in a loop that it was not defined in, it is copied instead of moved,
21//!     except if the last use is also reassigning the variable, killing the reference to its previous value.
22//!   - The last use analysis isn't sophisticated on struct fields. It will count `a.b` and `a.c`
23//!     both as uses of `a`. Even if both could conceptually be moved, only the last usage will be
24//!     moved and the first (say `a.b`) will still be cloned.
25//! - Dereferences always clone.
26//! - Certain expressions will avoid cloning or delay where clones are performed:
27//!   - Reference expressions `&e` will not clone `e` but may still clone variables used within.
28//!     - E.g. `&foo.bar` will not clone but `&foo(bar)` may still clone `foo` or `bar`.
29//!   - Dereferences will attempt to extract a field first if possible.
30//!     - E.g. `(*self).b.c` is transformed to `*(self.b.c)` where the `*` operation also clones.
31//!   - Ordinary member access will also delay clones.
32//!     - E.g. `self.b.c` is compiled as `self.b.c.clone()` over `self.clone().b.c`
33//!   - Array indexing `a[i]` will avoid cloning `a`. The extracted element is always cloned.
34//!
35//! Most of this logic is contained in this file except for the last use analysis which is in the
36//! `last_uses` module. That module contains a separate pass run on each function before this pass
37//! to find the last use of each local variable to identify where moves can occur.

44};

51impl Program {
52    /// Perform "ownership analysis".
53    ///
54    /// See [ownership](crate::ownership) for details.
55    ///
56    /// This should only be called once, before converting to SSA.
57    pub fn handle_ownership(mut self) -> Self {
58        for function in self.functions.iter_mut() {
59            function.handle_ownership();
60        }
61        self
62    }
63}

65impl Function {
66    /// Perform "ownership analysis".
67    ///
68    /// See [ownership](crate::ownership) for details.
69    ///
70    /// This should only be called on a function once.
71    pub fn handle_ownership(&mut self) {
72        let mut context = Context { variables_to_move: Default::default() };
73        context.handle_ownership_in_function(self);
74    }
75}
76
77struct Context {
78    /// This contains each instance of a variable we should move instead of cloning.
79    variables_to_move: HashMap<LocalId, Vec<IdentId>>,
80}
81
82impl Context {
83    fn should_move(&self, definition: LocalId, variable: IdentId) -> bool {
84        self.variables_to_move
85            .get(&definition)
86            .is_some_and(|instances_to_move| instances_to_move.contains(&variable))
87    }
88
89    fn handle_ownership_in_function(&mut self, function: &mut Function) {
90        if !function.unconstrained {
91            return;
92        }
93
94        self.variables_to_move = Self::find_last_uses_of_variables(function);
95        self.handle_expression(&mut function.body);
96    }
97
98    fn handle_expression(&mut self, expr: &mut Expression) {
99        match expr {
100            Expression::Ident(_) => self.handle_ident(expr),
101            Expression::Literal(literal) => self.handle_literal(literal),
102            Expression::Block(exprs) => {
103                exprs.iter_mut().for_each(|expr| self.handle_expression(expr));
104            }
105            Expression::Unary(_) => self.handle_unary(expr),
106            Expression::Binary(binary) => self.handle_binary(binary),
107            Expression::Index(_) => self.handle_index(expr),
108            Expression::Cast(cast) => self.handle_cast(cast),
109            Expression::For(for_expr) => self.handle_for(for_expr),
110            Expression::Loop(loop_expr) => self.handle_expression(loop_expr),
111            Expression::While(while_expr) => self.handle_while(while_expr),
112            Expression::If(if_expr) => self.handle_if(if_expr),
113            Expression::Match(match_expr) => self.handle_match(match_expr),
114            Expression::Tuple(elements) => self.handle_tuple(elements),
115            Expression::ExtractTupleField(..) => self.handle_extract_expression(expr),
116            Expression::Call(call) => self.handle_call(call),
117            Expression::Let(let_expr) => self.handle_let(let_expr),
118            Expression::Constrain(boolean, _location, msg) => self.handle_constrain(boolean, msg),
119            Expression::Assign(assign) => self.handle_assign(assign),
120            Expression::Semi(expr) => self.handle_expression(expr),
121            // Clones & Drops are only inserted by this pass so we can assume any code they
122            // contain is already handled
123            Expression::Clone(_) => (),
124            Expression::Drop(_) => (),
125            Expression::Break => (),
126            Expression::Continue => (),
127        }
128    }
129
130    /// Handle the RHS of a `&expr` unary expression.
131    /// Variables and field accesses in these expressions are exempt from clones.
132    ///
133    /// Note that this also matches on dereference operations to exempt their LHS from clones,
134    /// but their LHS is always exempt from clones so this is unchanged.
135    fn handle_reference_expression(&mut self, expr: &mut Expression) {
136        match expr {
137            Expression::Ident(_) => (),
138            Expression::Block(exprs) => {
139                let len_minus_one = exprs.len().saturating_sub(1);
140                for expr in exprs.iter_mut().take(len_minus_one) {
141                    // In `&{ a; b; ...; z }` we're only taking the reference of `z`.
142                    self.handle_expression(expr);
143                }
144                if let Some(expr) = exprs.last_mut() {
145                    self.handle_reference_expression(expr);
146                }
147            }
148            Expression::Unary(Unary { rhs, operator: UnaryOp::Dereference { .. }, .. }) => {
149                self.handle_reference_expression(rhs);
150            }
151            Expression::ExtractTupleField(tuple, _index) => self.handle_reference_expression(tuple),
152
153            Expression::Index(index) => {
154                self.handle_reference_expression(&mut index.collection);
155                self.handle_expression(&mut index.index);
156            }
157
158            // If we have something like `f(arg)` then we want to treat those variables normally
159            // rather than avoid cloning them. So we shouldn't recur in `handle_reference_expression`.
160            other => self.handle_expression(other),
161        }
162    }
163
164    /// Handle an [Expression::ExtractTupleField] by moving the cloning to limit its scope to the
165    /// innermost item it needs to be applied to.
166    ///
167    /// Panics if called on a different kind of expression.
168    fn handle_extract_expression(&mut self, expr: &mut Expression) {
169        let Expression::ExtractTupleField(tuple, index) = expr else {
170            panic!("handle_extract_expression given non-extract expression {expr}");
171        };
172
173        // We want to avoid cloning the entire object if we're only accessing one field of it
174        // so we check here to move the clone to the outermost extract expression instead.
175        // E.g. we want to change `a.clone().b.c` to `a.b.c.clone()`.
176        if let Some((should_clone, tuple_type)) = self.handle_extract_expression_rec(tuple) {
177            if let Some(elements) = unwrap_tuple_type(tuple_type)
178                && should_clone
179                && contains_array_or_str_type(&elements[*index])
180            {
181                clone_expr(expr);
182            }
183        } else {
184            self.handle_expression(tuple);
185        }
186    }
187
188    /// Traverse an expression comprised of only identifiers, tuple field extractions, and
189    /// dereferences returning whether we should clone the result and the type of that result.
190    ///
191    /// Returns None if a different expression variant was found.
192    fn handle_extract_expression_rec(&mut self, expr: &mut Expression) -> Option<(bool, Type)> {
193        match expr {
194            Expression::Ident(ident) => {
195                let should_clone = self.should_clone_ident(ident);
196                Some((should_clone, ident.typ.clone()))
197            }
198            // Delay dereferences as well so we change `(*self).foo.bar` to `*(self.foo.bar)`
199            Expression::Unary(Unary {
200                rhs,
201                operator: UnaryOp::Dereference { .. },
202                result_type,
203                ..
204            }) => {
205                self.handle_reference_expression(rhs);
206                Some((true, result_type.clone()))
207            }
208            Expression::ExtractTupleField(tuple, index) => {
209                let (should_clone, typ) = self.handle_extract_expression_rec(tuple)?;
210                let mut elements = unwrap_tuple_type(typ)?;
211                Some((should_clone, elements.swap_remove(*index)))
212            }
213            _ => None,
214        }
215    }
216
217    /// Whenever an ident is used it is always cloned unless it is the last use of the ident (not in a loop, unless it's also reassigning the ident).
218    fn should_clone_ident(&self, ident: &Ident) -> bool {
219        match &ident.definition {
220            Definition::Local(local_id) => {
221                contains_array_or_str_type(&ident.typ) && !self.should_move(*local_id, ident.id)
222            }
223            // Globals are always cloned if they contain arrays
224            Definition::Global(_) => contains_array_or_str_type(&ident.typ),
225            _ => false,
226        }
227    }
228
229    fn handle_ident(&self, expr: &mut Expression) {
230        let ident = match expr {
231            Expression::Ident(ident) => ident,
232            other => panic!("handle_ident given non-ident expr: {other}"),
233        };
234
235        if self.should_clone_ident(ident) {
236            clone_expr(expr);
237        }
238    }
239
240    fn handle_literal(&mut self, literal: &mut Literal) {
241        match literal {
242            Literal::Integer(..) | Literal::Bool(_) | Literal::Unit | Literal::Str(_) => (),
243
244            Literal::FmtStr(_, _, captures) => self.handle_expression(captures),
245
246            Literal::Array(array) | Literal::Vector(array) => {
247                for element in array.contents.iter_mut() {
248                    self.handle_expression(element);
249                }
250            }
251
252            Literal::Repeated { element, .. } => {
253                self.handle_expression(element);
254                // Reference counting for repeated arrays is handled in SSA via inc_rc instructions
255            }
256        }
257    }
258
259    fn handle_unary(&mut self, expr: &mut Expression) {
260        let unary = match expr {
261            Expression::Unary(unary) => unary,
262            other => panic!("handle_unary given non-unary expression: {other}"),
263        };
264
265        // Don't clone `rhs` if this is a reference or dereference expression.
266        // - If this is a reference expression `&rhs`, `rhs` by definition shouldn't be cloned
267        // - If this is `*rhs` we're going to clone the extracted element instead
268        if matches!(unary.operator, UnaryOp::Reference { .. } | UnaryOp::Dereference { .. }) {
269            self.handle_reference_expression(&mut unary.rhs);
270        } else {
271            self.handle_expression(&mut unary.rhs);
272        }
273
274        if matches!(unary.operator, UnaryOp::Dereference { .. })
275            && contains_array_or_str_type(&unary.result_type)
276        {
277            clone_expr(expr);
278        }
279    }
280
281    fn handle_binary(&mut self, binary: &mut crate::monomorphization::ast::Binary) {
282        self.handle_expression(&mut binary.lhs);
283        self.handle_expression(&mut binary.rhs);
284    }
285
286    fn handle_index(&mut self, index_expr: &mut Expression) {
287        let Expression::Index(index) = index_expr else {
288            panic!("handle_index given non-index expression: {index_expr}");
289        };
290
291        // Don't clone the collection, cloning only the resulting element is cheaper.
292        self.handle_reference_expression(&mut index.collection);
293        self.handle_expression(&mut index.index);
294
295        // If the index collection is being borrowed we need to clone the result.
296        if contains_array_or_str_type(&index.element_type) {
297            clone_expr(index_expr);
298        }
299    }
300
301    fn handle_cast(&mut self, cast: &mut crate::monomorphization::ast::Cast) {
302        self.handle_expression(&mut cast.lhs);
303    }
304
305    fn handle_for(&mut self, for_expr: &mut crate::monomorphization::ast::For) {
306        self.handle_expression(&mut for_expr.start_range);
307        self.handle_expression(&mut for_expr.end_range);
308        self.handle_expression(&mut for_expr.block);
309    }
310
311    fn handle_while(&mut self, while_expr: &mut crate::monomorphization::ast::While) {
312        self.handle_expression(&mut while_expr.condition);
313        self.handle_expression(&mut while_expr.body);
314    }
315
316    fn handle_if(&mut self, if_expr: &mut crate::monomorphization::ast::If) {
317        self.handle_expression(&mut if_expr.condition);
318        self.handle_expression(&mut if_expr.consequence);
319        if let Some(alt) = &mut if_expr.alternative {
320            self.handle_expression(alt);
321        }
322    }
323
324    fn handle_match(&mut self, match_expr: &mut crate::monomorphization::ast::Match) {
325        // Note: We don't need to explicitly handle `Match::variable_to_match` here.
326        // The matched variable is just a LocalId reference to a variable that was assigned earlier.
327        // Cloning for that variable happens at its use sites (e.g., when passed to the enum
328        // constructor or used after the match), not at the match expression itself.
329        // The match will only destructure the value; it doesn't "use" the variable in a way that
330        // requires additional cloning beyond what the last-use analysis already handles.
331        for case in &mut match_expr.cases {
332            self.handle_expression(&mut case.branch);
333        }
334
335        if let Some(default_case) = &mut match_expr.default_case {
336            self.handle_expression(default_case);
337        }
338    }
339
340    fn handle_tuple(&mut self, elements: &mut [Expression]) {
341        for elem in elements {
342            self.handle_expression(elem);
343        }
344    }
345
346    fn handle_call(&mut self, call: &mut crate::monomorphization::ast::Call) {
347        self.handle_expression(&mut call.func);
348        for arg in &mut call.arguments {
349            self.handle_expression(arg);
350        }
351    }
352
353    fn handle_let(&mut self, let_expr: &mut crate::monomorphization::ast::Let) {
354        self.handle_expression(&mut let_expr.expression);
355    }
356
357    fn handle_constrain(
358        &mut self,
359        boolean: &mut Expression,
360        msg: &mut Option<Box<(Expression, crate::hir_def::types::Type)>>,
361    ) {
362        self.handle_expression(boolean);
363
364        if let Some(msg) = msg {
365            self.handle_expression(&mut msg.0);
366        }
367    }
368
369    fn handle_assign(&mut self, assign: &mut crate::monomorphization::ast::Assign) {
370        self.handle_expression(&mut assign.expression);
371        self.handle_lvalue(&mut assign.lvalue);
372    }
373
374    fn handle_lvalue(&mut self, lvalue: &mut LValue) {
375        match lvalue {
376            // A variable can never be moved into an LValue position so it doesn't
377            // need to be cloned or checked here.
378            LValue::Ident(_) => (),
379            LValue::Index { array, index, element_type: _, location: _ } => {
380                self.handle_expression(index);
381                self.handle_lvalue(array);
382
383                if contains_index(array) {
384                    **array = LValue::Clone(array.clone());
385                }
386            }
387            LValue::MemberAccess { object, field_index: _ } => {
388                self.handle_lvalue(object);
389            }
390            LValue::Dereference { reference, element_type: _ } => {
391                self.handle_lvalue(reference);
392            }
393            // LValue::Clone isn't present before this pass and is only inserted after we already
394            // handle the corresponding lvalue
395            LValue::Clone(_) => unreachable!("LValue::Clone should only be inserted by this pass"),
396        }
397    }
398}
399
400fn contains_index(lvalue: &LValue) -> bool {
401    use LValue::*;
402    match lvalue {
403        Ident(_) => false,
404        Index { .. } => true,
405        Dereference { reference: lvalue, .. }
406        | MemberAccess { object: lvalue, .. }
407        | Clone(lvalue) => contains_index(lvalue),
408    }
409}

411/// Adds a `.clone()` to the given expression.
412/// Note that this method should be careful not to actually duplicate the given expression
413/// so that we do not duplicate any side-effects.
414fn clone_expr(expr: &mut Expression) {
415    let unit = Expression::Literal(Literal::Unit);
416    let old_expr = std::mem::replace(expr, unit);
417    *expr = Expression::Clone(Box::new(old_expr));
418}

420/// Returns `true` if the type contains an `Array`, `Vector`, `String` or `FmtString`,
421/// directly or as part of a `Tuple`, but _not_ through a reference.
422fn contains_array_or_str_type(typ: &Type) -> bool {
423    match typ {
424        Type::Field
425        | Type::Integer(..)
426        | Type::Bool
427        | Type::Unit
428        | Type::Function(..)
429        | Type::Reference(..) => false,
430
431        Type::Array(_, _) | Type::String(_) | Type::FmtString(_, _) | Type::Vector(_) => true,
432
433        Type::Tuple(elements) => elements.iter().any(contains_array_or_str_type),
434    }
435}

437/// Returns the element types of a [Type::Tuple], or a reference to a tuple.
438///
439/// Returns `None` for any other type.
440fn unwrap_tuple_type(typ: Type) -> Option<Vec<Type>> {
441    match typ {
442        Type::Tuple(elements) => Some(elements),
443        // array accesses will automatically dereference so we do too
444        Type::Reference(element, _) => unwrap_tuple_type(*element),
445        _ => None,
446    }
