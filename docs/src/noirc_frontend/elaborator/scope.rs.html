51    /// Runs `f` with `self.local_module` set to `module`, restoring the previous value
52    /// afterwards (on every exit path, including early returns inside `f`). This is the
53    /// module-scope analogue of [`Self::recover_generics`] and should be used instead of a
54    /// bare `self.local_module = Some(..)` so that the caller's module context is never left
55    /// dangling.
56    #[tracing::instrument(level = "trace", skip_all)]
57    pub(super) fn in_local_module<T>(
58        &mut self,
59        module: LocalModuleId,
60        f: impl FnOnce(&mut Self) -> T,
61    ) -> T {
62        let previous = self.replace_local_module(module);
63        let result = f(self);
64        self.local_module = previous;
65        result
66    }
67
68    #[must_use]
69    pub(super) fn replace_local_module(&mut self, module: LocalModuleId) -> Option<LocalModuleId> {
70        self.local_module.replace(module)
71    }

82    /// For each [`crate::elaborator::LambdaContext`] on the lambda stack with a scope index higher than that
83    /// of the variable, add the [`crate::elaborator::HirIdent`] to the list of captures.
84    #[tracing::instrument(level = "trace", skip_all)]
85    pub(super) fn check_if_variable_is_captured_by_closure(&mut self, variable: &Variable) {
86        // Only local variables can be captured by closures.
87        // (the variable might point to a numeric generic like `let N: u32`, which is not captured)
88        let DefinitionKind::Local(..) = self.interner.definition(variable.ident.id).kind else {
89            return;
90        };
91
92        let mut transitive_capture_index: Option<usize> = None;
93
94        for lambda_index in 0..self.lambda_stack.len() {
95            if self.lambda_stack[lambda_index].scope_index > variable.scope {
96                // Beware: the same variable may be captured multiple times, so we check
97                // for its presence before adding the capture below.
98                let position = self.lambda_stack[lambda_index]
99                    .captures
100                    .iter()
101                    .position(|capture| capture.ident.id == variable.ident.id);
102
103                let capture_index = position.or_else(|| {
104                    // In a comptime context we capture comptime and non-comptime variables
105                    // (the latter will be an error).
106                    // In a non-comptime context we don't capture comptime variables.
107                    if self.in_comptime_context()
108                        || !self.interner.definition(variable.ident.id).is_comptime_local()
109                    {
110                        self.lambda_stack[lambda_index].captures.push(HirCapturedVar {
111                            ident: variable.ident.clone(),
112                            transitive_capture_index,
113                        });
114                        // If this was a fresh capture, we added it to the end of
115                        // the captures vector:
116                        Some(self.lambda_stack[lambda_index].captures.len() - 1)
117                    } else {
118                        None
119                    }
120                });
121
122                if lambda_index + 1 < self.lambda_stack.len() {
123                    // There is more than one closure between the current scope and
124                    // the scope of the variable, so this is a propagated capture.
125                    // We need to track the transitive capture index as we go up in
126                    // the closure stack.
127                    transitive_capture_index = capture_index;
128                }
129            }
130        }
131    }

133    /// Try to look up a [`TypedPath`] as a value (a global, a numeric type alias or a function).
134    /// If the path resolves to an item that is not a value (for example a struct, an enum,
135    /// a type alias, etc.), returns a `ResolverError`. `ResolverError` is also returned
136    /// when no item is found.
137    ///
138    /// Unlike [`Self::resolve_path_as_value_or_error`] this never tries a local variable, so it is
139    /// what a multi-segment path (which can never name a local variable) needs.
140    #[tracing::instrument(level = "trace", skip_all)]
141    pub(super) fn lookup_path_as_value(
142        &mut self,
143        path: TypedPath,
144    ) -> Result<PathValue, ResolverError> {
145        let location = path.location;
146        let item = self.use_path_or_error(path, PathResolutionTarget::Value)?;
147        self.path_resolution_item_as_value(item, location)
148    }
149
150    /// Like [`Self::lookup_path_as_value`], but resolves `segment` directly in the already-resolved
151    /// `module_id` rather than from the current module.
152    pub(super) fn lookup_path_as_value_in_module(
153        &mut self,
154        segment: TypedPathSegment,
155        module_id: ModuleId,
156    ) -> Result<PathValue, ResolverError> {
157        let location = segment.ident.location();
158        let item =
159            self.use_value_in_module(TypedPath::plain(vec![segment], location), module_id)?;
160        self.path_resolution_item_as_value(item, location)
161    }

163    /// Like [`Self::lookup_path_as_value`], but resolves `segment` directly as a value member (an
164    /// enum variant or associated constant) of the already-resolved type `typ`.
165    pub(super) fn lookup_path_as_value_in_type(
166        &mut self,
167        segment: &TypedPathSegment,
168        typ: &Type,
169        turbofish: Option<Turbofish>,
170    ) -> Result<PathValue, ResolverError> {
171        let location = segment.ident.location();
172        let item = self.use_value_in_type(segment, typ, turbofish)?;
173        self.path_resolution_item_as_value(item, location)
174    }

176    /// Interpret an already-resolved [`PathResolutionItem`] as a value (the definition it refers
177    /// to), or return an `Expected` error when it is not a value. `location` is used for that error.
178    fn path_resolution_item_as_value(
179        &self,
180        item: PathResolutionItem,
181        location: Location,
182    ) -> Result<PathValue, ResolverError> {
183        if let Some(function) = item.function_id() {
184            let definition_id = self.interner.function_definition_id(function);
185            return Ok(PathValue::Definition { id: definition_id, item });
186        }
187
188        let expected = "value";
189        match item {
190            PathResolutionItem::Global(global) | PathResolutionItem::EnumVariant(global) => {
191                let global = self.interner.get_global(global);
192                Ok(PathValue::Definition { id: global.definition_id, item })
193            }
194            PathResolutionItem::TypeAlias(type_alias_id) => {
195                let type_alias = self.interner.get_type_alias(type_alias_id);
196
197                // Type alias is returned
198                if type_alias.borrow().numeric_expr.is_some() {
199                    // Type alias to numeric generics are aliases to some global value
200                    // Therefore we allow this case although we cannot provide the value yet
201                    return Ok(PathValue::TypeAlias(type_alias_id));
202                }
203                if matches!(type_alias.borrow().typ, Type::Alias(_, _))
204                    || matches!(type_alias.borrow().typ, Type::Error)
205                {
206                    // Type alias to a type alias is not supported, but the error is handled in define_type_alias()
207                    return Ok(PathValue::TypeAlias(type_alias_id));
208                }
209                Err(ResolverError::Expected {
210                    location,
211                    expected,
212                    found: item.description(self.interner),
213                })
214            }
215            PathResolutionItem::TraitConstant(_, _, def_id) => {
216                // TraitConstant is returned, item is Some
217                Ok(PathValue::Definition { id: def_id, item })
218            }
219            item => Err(ResolverError::Expected {
220                location,
221                expected,
222                found: item.description(self.interner),
223            }),
224        }
225    }

227    #[tracing::instrument(level = "trace", skip_all)]
228    pub fn push_scope(&mut self) {
229        self.scopes.start_scope();

233    #[tracing::instrument(level = "trace", skip_all)]
234    pub fn pop_scope(&mut self) {
235        let scope = self.scopes.end_scope();
236        self.interner.comptime_scopes.pop();
237        let scope_decls = scope.into();
238        self.check_for_unused_variables_in_scope_tree(&scope_decls);
239        self.check_for_unnecessary_mut_variables_in_scope_tree(&scope_decls);
240    }

242    #[tracing::instrument(level = "trace", skip_all)]
243    pub fn check_for_unused_variables_in_scope_tree(&mut self, scope_decls: &ScopeTree) {
244        let mut unused_vars = Vec::new();
245
246        for scope in &scope_decls.0 {
247            for (variable_name, metadata) in scope.iter() {
248                if !metadata.warn_if_unused || metadata.used || variable_name.starts_with('_') {
249                    continue;
250                }
251
252                let unused_var = &metadata.ident;
253                let definition_info = self.interner.definition(unused_var.id);
254                let name = &definition_info.name;
255                if name != ERROR_IDENT && !definition_info.is_global() {
256                    let ident = Ident::new(name.to_owned(), unused_var.location);
257                    unused_vars.push(ident);
258                }
259            }
260        }
261
262        unused_vars.sort_by_key(|ident| ident.location());
263
264        for ident in unused_vars {
265            self.push_err(ResolverError::UnusedVariable { ident });
266        }
267    }
268
269    #[tracing::instrument(level = "trace", skip_all)]
270    pub fn check_for_unnecessary_mut_variables_in_scope_tree(&mut self, scope_decls: &ScopeTree) {
271        let mut unnecessary_mut_vars = Vec::new();
272
273        for scope in &scope_decls.0 {
274            for (variable_name, metadata) in scope.iter() {
275                if !metadata.warn_if_not_mutated
276                    || metadata.mutated
277                    || variable_name.starts_with('_')
278                {
279                    continue;
280                }
281
282                let ident = &metadata.ident;
283                let definition_info = self.interner.definition(ident.id);
284                if definition_info.mutable && !definition_info.is_global() {
285                    let ident = Ident::new(variable_name.to_owned(), ident.location);
286                    unnecessary_mut_vars.push(ident);
287                }
288            }
289        }
290
291        unnecessary_mut_vars.sort_by_key(|ident| ident.location());
292
293        for ident in unnecessary_mut_vars {
294            self.push_err(ResolverError::VariableDoesNotNeedToBeMutable { ident });
295        }
296    }

298    /// Lookup a given trait by name/path.
299    #[tracing::instrument(level = "trace", skip_all)]
300    pub(crate) fn lookup_trait_or_error(&mut self, path: TypedPath) -> Option<&Trait> {
301        let location = path.location;
302        match self.resolve_path_or_error(path, PathResolutionTarget::Type) {
303            Ok(item) => {
304                if let PathResolutionItem::Trait(trait_id) = item {
305                    Some(self.get_trait(trait_id))
306                } else {
307                    self.push_err(ResolverError::Expected {
308                        expected: "trait",
309                        found: item.description(self.interner),
310                        location,
311                    });
312                    None
313                }
314            }
315            Err(err) => {
316                self.push_err(err);
317                None
318            }
319        }
320    }

322    /// Looks up a given [Type] by name.
323    ///
324    /// This will also instantiate any struct types found.
325    #[tracing::instrument(level = "trace", skip_all)]
326    pub(super) fn lookup_type_or_error(&mut self, path: TypedPath) -> Option<Type> {
327        let segment = path.as_single_segment();
328        if let Some(segment) = segment
329            && segment.ident.is_self_type_name()
330            && let Some(typ) = &self.self_type
331        {
332            return Some(typ.clone());
333        }
334
335        let location = path.location;
336        match self.use_path_or_error(path, PathResolutionTarget::Type) {
337            Ok(PathResolutionItem::Type(struct_id)) => {
338                let struct_type = self.get_type(struct_id);
339                let generics = struct_type.borrow().instantiate(self.interner);
340                let typ = Type::DataType(struct_type, generics);
341                self.check_comptime_type_in_non_comptime_item(&typ, location);
342                Some(typ)
343            }
344            Ok(PathResolutionItem::TypeAlias(alias_id)) => {
345                let alias = self.interner.get_type_alias(alias_id);
346                self.check_comptime_type_in_non_comptime_item(
347                    &Type::Alias(alias.clone(), Vec::new()),
348                    location,
349                );
350                let alias = alias.borrow();
351                Some(alias.instantiate(self.interner))
352            }
353            Ok(other) => {
354                self.push_err(ResolverError::Expected {
355                    expected: "type",
356                    found: other.description(self.interner),
357                    location,
358                });
359                None
360            }
361            Err(error) => {
362                self.push_err(error);
363                None
364            }
365        }
366    }

368    #[tracing::instrument(level = "trace", skip_all)]
369    pub(super) fn lookup_type_alias(
370        &mut self,
371        path: TypedPath,
372        mode: PathResolutionMode,
373    ) -> Option<Shared<TypeAlias>> {
374        match self.resolve_path_inner(path, PathResolutionTarget::Type, mode) {
375            Ok(PathResolution { item: PathResolutionItem::TypeAlias(type_alias_id), errors }) => {
376                self.push_errors(errors);
377                Some(self.interner.get_type_alias(type_alias_id))
378            }
379            _ => None,
380        }
381    }
