46    /// For each [crate::elaborator::LambdaContext] on the lambda stack with a scope index higher than that
47    /// of the variable, add the [HirIdent] to the list of captures.
48    pub(super) fn resolve_local_variable(&mut self, hir_ident: HirIdent, var_scope_index: usize) {
49        let mut transitive_capture_index: Option<usize> = None;
50
51        for lambda_index in 0..self.lambda_stack.len() {
52            if self.lambda_stack[lambda_index].scope_index > var_scope_index {
53                // Beware: the same variable may be captured multiple times, so we check
54                // for its presence before adding the capture below.
55                let position = self.lambda_stack[lambda_index]
56                    .captures
57                    .iter()
58                    .position(|capture| capture.ident.id == hir_ident.id);
59
60                if position.is_none() {
61                    // In a comptime context we capture comptime and non-comptime variables
62                    // (the latter will be an error).
63                    // In a non-comptime context we don't capture comptime variables.
64                    if self.in_comptime_context()
65                        || !self.interner.definition(hir_ident.id).is_comptime_local()
66                    {
67                        self.lambda_stack[lambda_index].captures.push(HirCapturedVar {
68                            ident: hir_ident.clone(),
69                            transitive_capture_index,
70                        });
71                    }
72                }
73
74                if lambda_index + 1 < self.lambda_stack.len() {
75                    // There is more than one closure between the current scope and
76                    // the scope of the variable, so this is a propagated capture.
77                    // We need to track the transitive capture index as we go up in
78                    // the closure stack.
79                    transitive_capture_index = Some(position.unwrap_or(
80                        // If this was a fresh capture, we added it to the end of
81                        // the captures vector:
82                        self.lambda_stack[lambda_index].captures.len() - 1,
83                    ));
84                }
85            }
86        }
87    }

89    /// Try to look up a [TypedPath] as a value (a global, a numeric type alias or a function).
90    /// If the path resolves to an item that is not a value (for example a struct, an enum,
91    /// a type alias, etc.), returns a `ResolverError`. `ResolverError` is also returned
92    /// when no item is found.
93    pub(super) fn lookup_item_as_value(
94        &mut self,
95        path: TypedPath,
96    ) -> Result<(DefinitionId, PathResolutionItem), ResolverError> {
97        let location = path.location;
98        let item = self.use_path_or_error(path, PathResolutionTarget::Value)?;
99
100        if let Some(function) = item.function_id() {
101            return Ok((self.interner.function_definition_id(function), item));
102        }
103
104        let expected = "value";
105        match item {
106            PathResolutionItem::Global(global) => {
107                let global = self.interner.get_global(global);
108                Ok((global.definition_id, item))
109            }
110            PathResolutionItem::TypeAlias(type_alias_id) => {
111                let type_alias = self.interner.get_type_alias(type_alias_id);
112
113                if type_alias.borrow().numeric_expr.is_some() {
114                    // Type alias to numeric generics are aliases to some global value
115                    // Therefore we allow this case although we cannot provide the value yet
116                    return Ok((DefinitionId::dummy_id(), item));
117                }
118                if matches!(type_alias.borrow().typ, Type::Alias(_, _))
119                    || matches!(type_alias.borrow().typ, Type::Error)
120                {
121                    // Type alias to a type alias is not supported, but the error is handled in define_type_alias()
122                    return Ok((DefinitionId::dummy_id(), item));
123                }
124                Err(ResolverError::Expected {
125                    location,
126                    expected,
127                    found: item.description(self.interner),
128                })
129            }
130            PathResolutionItem::TraitConstant(_, _, def_id) => Ok((def_id, item)),
131            item => Err(ResolverError::Expected {
132                location,
133                expected,
134                found: item.description(self.interner),
135            }),
136        }
137    }

139    pub fn push_scope(&mut self) {
140        self.scopes.start_scope();
141        self.interner.comptime_scopes.push(Default::default());
142    }
143
144    pub fn pop_scope(&mut self) {
145        let scope = self.scopes.end_scope();
146        self.interner.comptime_scopes.pop();
147        self.check_for_unused_variables_in_scope_tree(scope.into());
148    }
149
150    pub fn check_for_unused_variables_in_scope_tree(&mut self, scope_decls: ScopeTree) {
151        let mut unused_vars = Vec::new();
152        for scope in scope_decls.0.into_iter() {
153            Self::check_for_unused_variables_in_local_scope(scope, &mut unused_vars);
154        }
155
156        for unused_var in unused_vars.iter() {
157            if let Some(definition_info) = self.interner.try_definition(unused_var.id) {
158                let name = &definition_info.name;
159                if name != ERROR_IDENT && !definition_info.is_global() {
160                    let ident = Ident::new(name.to_owned(), unused_var.location);
161                    self.push_err(ResolverError::UnusedVariable { ident });
162                }
163            }
164        }
165    }
166
167    fn check_for_unused_variables_in_local_scope(decl_map: Scope, unused_vars: &mut Vec<HirIdent>) {
168        let unused_variables = decl_map.filter(|(variable_name, metadata)| {
169            let has_underscore_prefix = variable_name.starts_with('_'); // XXX: This is used for development mode, and will be removed
170            metadata.warn_if_unused && metadata.num_times_used == 0 && !has_underscore_prefix
171        });
172        unused_vars.extend(unused_variables.map(|(_, meta)| meta.ident.clone()));
173    }

175    /// Lookup a given trait by name/path.
176    pub(crate) fn lookup_trait_or_error(&mut self, path: TypedPath) -> Option<&Trait> {
177        let location = path.location;
178        match self.resolve_path_or_error(path, PathResolutionTarget::Type) {
179            Ok(item) => {
180                if let PathResolutionItem::Trait(trait_id) = item {
181                    Some(self.get_trait(trait_id))
182                } else {
183                    self.push_err(ResolverError::Expected {
184                        expected: "trait",
185                        found: item.description(self.interner),
186                        location,
187                    });
188                    None
189                }
190            }
191            Err(err) => {
192                self.push_err(err);
193                None
194            }
195        }
196    }

198    /// Looks up a given [Type] by name.
199    ///
200    /// This will also instantiate any struct types found.
201    pub(super) fn lookup_type_or_error(&mut self, path: TypedPath) -> Option<Type> {
202        let segment = path.as_single_segment();
203        if let Some(segment) = segment {
204            if segment.ident.is_self_type_name() {
205                if let Some(typ) = &self.self_type {
206                    return Some(typ.clone());
207                }
208            }
209        }
210
211        let location = path.location;
212        match self.use_path_or_error(path, PathResolutionTarget::Type) {
213            Ok(PathResolutionItem::Type(struct_id)) => {
214                let struct_type = self.get_type(struct_id);
215                let generics = struct_type.borrow().instantiate(self.interner);
216                Some(Type::DataType(struct_type, generics))
217            }
218            Ok(PathResolutionItem::TypeAlias(alias_id)) => {
219                let alias = self.interner.get_type_alias(alias_id);
220                let alias = alias.borrow();
221                Some(alias.instantiate(self.interner))
222            }
223            Ok(other) => {
224                self.push_err(ResolverError::Expected {
225                    expected: "type",
226                    found: other.description(self.interner),
227                    location,
228                });
229                None
230            }
231            Err(error) => {
232                self.push_err(error);
233                None
234            }
235        }
236    }

238    pub(super) fn lookup_type_alias(
239        &mut self,
240        path: TypedPath,
241        mode: PathResolutionMode,
242    ) -> Option<Shared<TypeAlias>> {
243        match self.resolve_path_inner(path, PathResolutionTarget::Type, mode) {
244            Ok(PathResolution { item: PathResolutionItem::TypeAlias(type_alias_id), errors }) => {
245                self.push_errors(errors);
246                Some(self.interner.get_type_alias(type_alias_id))
247            }
248            _ => None,
249        }
250    }
