1//! Function metadata and function body elaboration.
2//!
3//! This module handles the definition and elaboration of functions:
4//! - Collect and resolve function metadata (i.e., signatures with type information). Type information includes:
5//!   - Generics, parameters, trait constraints which will all be resolved while collecting function metas.
6//!   - This metadata is also used for elaboration of impls and trait impls
7//! - Second stage elaboration strategy of function bodies and their return type.
8//!   - Shared strategy for all types of functions (standalone, impl, trait impl)
9

43    /// Defines function metadata for all functions, impl methods, and trait impl methods.
44    /// This is the first pass of function elaboration that extracts type signatures and
45    /// resolves generics before the function bodies are elaborated.
46    pub(super) fn define_function_metas(
47        &mut self,
48        functions: &mut [UnresolvedFunctions],
49        impls: &mut ImplMap,
50        trait_impls: &mut [UnresolvedTraitImpl],
51    ) {
52        // Define metas for regular functions
53        for function_set in functions {
54            self.define_function_metas_for_functions(function_set, &[]);
55        }
56
57        // Define metas for impl functions
58        for ((self_type, local_module), function_sets) in impls {
59            self.define_function_metas_for_impl(self_type, *local_module, function_sets);
60        }
61
62        // Define metas for trait impl functions
63        for trait_impl in trait_impls {
64            self.define_function_metas_for_trait_impl(trait_impl);
65        }
66    }

68    /// Defines function metadata for a set of functions with optional extra trait constraints.
69    /// This is used for both standalone functions and methods within impls/trait impls.
70    fn define_function_metas_for_functions(
71        &mut self,
72        function_set: &mut UnresolvedFunctions,
73        extra_constraints: &[(TraitConstraint, Location)],
74    ) {
75        for (local_module, id, func) in &mut function_set.functions {
76            self.local_module = Some(*local_module);
77            self.recover_generics(|this| {
78                this.define_function_meta(func, *id, None, extra_constraints);
79            });
80        }
81    }

83    /// Defines function metadata for all methods within an impl block.
84    /// Resolves the self type and adds it to scope for method resolution.
85    fn define_function_metas_for_impl(
86        &mut self,
87        self_type: &UnresolvedType,
88        local_module: crate::hir::def_map::LocalModuleId,
89        function_sets: &mut Vec<(UnresolvedGenerics, Location, UnresolvedFunctions)>,
90    ) {
91        self.local_module = Some(local_module);
92
93        for (generics, _, function_set) in function_sets {
94            // Prepare the impl
95            // Adds the impl generics to the generics state and resolve the impl's self type
96            self.add_generics(generics);
97
98            let wildcard_allowed = WildcardAllowed::No(WildcardDisallowedContext::ImplType);
99            let self_type = self.resolve_type(self_type.clone(), wildcard_allowed);
100            function_set.self_type = Some(self_type.clone());
101            self.self_type = Some(self_type);
102
103            self.define_function_metas_for_functions(function_set, &[]);
104
105            // Cleanup
106            self.self_type = None;
107            self.generics.clear();
108        }
109    }

111    /// Defines function metadata for all methods within a trait impl.
112    /// This handles trait resolution, generics, associated types, and constraint checking.
113    fn define_function_metas_for_trait_impl(&mut self, trait_impl: &mut UnresolvedTraitImpl) {
114        // Prepare the trait impl
115        let new_generics_trait_constraints =
116            self.prepare_trait_impl_for_function_meta_definition(trait_impl);
117
118        // Set up trait impl state
119        self.current_trait_impl = trait_impl.impl_id;
120        self.self_type = trait_impl.methods.self_type.clone();
121
122        // Now define the function metas with the constraints from where clause desugaring
123        self.define_function_metas_for_functions(
124            &mut trait_impl.methods,
125            &new_generics_trait_constraints,
126        );
127
128        // Cleanup
129        self.self_type = None;
130        self.current_trait_impl = None;
131        self.generics.clear();
132    }

134    /// Extracts and stores metadata from a function definition.
135    ///
136    /// This resolves the function's signature including generics, parameters, return type,
137    /// and trait constraints. The function body is stored unresolved and will be elaborated
138    /// later by [Self::elaborate_function].
139    ///
140    /// Prerequisite: any implicit generics from enclosing impls have already been added
141    /// to scope via [Self::add_generics].
142    pub(super) fn define_function_meta(
143        &mut self,
144        func: &mut NoirFunction,
145        func_id: FuncId,
146        trait_id: Option<TraitId>,
147        extra_trait_constraints: &[(TraitConstraint, Location)],
148    ) {
149        self.scopes.start_function();
150        self.current_item = Some(DependencyId::Function(func_id));
151        let old_comptime_value =
152            std::mem::replace(&mut self.in_comptime_context, func.def.is_comptime);
153
154        let location = func.name_ident().location();
155        let id = self.interner.function_definition_id(func_id);
156        let name_ident = HirIdent::non_trait_method(id, location);
157
158        // Add generics to scope
159        let (mut generics, associated_generics_trait_constraints) =
160            self.add_function_generics_to_scope(&func.def.generics, &mut func.def.where_clause);
161
162        // Setup trait constraints
163        for (extra_constraint, location) in extra_trait_constraints {
164            let bound = &extra_constraint.trait_bound;
165            self.add_trait_bound_to_scope(*location, &extra_constraint.typ, bound, bound.trait_id);
166        }
167
168        let mut trait_constraints =
169            self.resolve_trait_constraints_and_add_to_scope(&func.def.where_clause);
170        let mut extra_trait_constraints =
171            vecmap(extra_trait_constraints, |(constraint, _)| constraint.clone());
172        extra_trait_constraints.extend(associated_generics_trait_constraints);
173
174        // Resolve parameters
175        let (parameters, parameter_types, parameter_idents) =
176            self.resolve_function_parameters(func, &mut generics, &mut trait_constraints);
177
178        // Resolve return type
179        let wildcard_allowed = WildcardAllowed::No(WildcardDisallowedContext::FunctionReturn);
180        let return_type = Box::new(self.use_type(func.return_type(), wildcard_allowed));
181
182        let is_crate_root = self.is_at_crate_root();
183        let is_entry_point = func.is_entry_point(self.is_function_in_contract(), is_crate_root);
184        // Temporary allow vectors for contract functions, until contracts are re-factored.
185        if !func.attributes().has_contract_library_method() {
186            let output = true;
187            if let Err(err) = Self::check_if_type_is_valid_for_program(
188                &return_type,
189                is_entry_point || func.is_test_or_fuzz(),
190                func.has_inline_attribute(),
191                output,
192                func.return_type().location,
193            ) {
194                self.push_err(err);
195            }
196        }
197
198        // Build function type
199        let mut typ = Type::Function(
200            parameter_types,
201            return_type,
202            Box::new(Type::Unit),
203            func.def.is_unconstrained,
204        );
205        if !generics.is_empty() {
206            typ = Type::Forall(generics, Box::new(typ));
207        }
208        self.interner.push_definition_type(name_ident.id, typ.clone());
209
210        // Set up metadata to place on main FuncMeta structure
211        let direct_generics = func.def.generics.iter();
212        let direct_generics = direct_generics
213            .filter_map(|generic| {
214                generic.ident().ident().and_then(|name| self.find_generic(name.as_str())).cloned()
215            })
216            .collect();
217
218        let statements = std::mem::take(&mut func.def.body.statements);
219        let body = BlockExpression { statements };
220
221        let struct_id = if let Some(Type::DataType(struct_type, _)) = &self.self_type {
222            Some(struct_type.borrow().id)
223        } else {
224            None
225        };
226
227        // Remove the traits assumed by `resolve_trait_constraints` from scope
228        self.remove_trait_constraints_from_scope(
229            trait_constraints.iter().chain(extra_trait_constraints.iter()),
230        );
231
232        let meta = FuncMeta {
233            name: name_ident,
234            kind: func.kind,
235            location,
236            typ,
237            direct_generics,
238            all_generics: self.generics.clone(),
239            type_id: struct_id,
240            trait_id,
241            trait_impl: self.current_trait_impl,
242            enum_variant_index: None,
243            parameters: parameters.into(),
244            parameter_idents,
245            return_type: func.def.return_type.clone(),
246            return_visibility: func.def.return_visibility,
247            has_body: !func.def.body.is_empty(),
248            trait_constraints,
249            extra_trait_constraints,
250            is_entry_point,
251            has_inline_attribute: func.has_inline_attribute(),
252            source_crate: self.crate_id,
253            source_module: self.local_module(),
254            function_body: FunctionBody::Unresolved(func.kind, body, func.def.location),
255            self_type: self.self_type.clone(),
256            source_file: location.file,
257        };
258
259        self.interner.push_fn_meta(meta, func_id);
260        self.scopes.end_function();
261        self.current_item = None;
262        self.in_comptime_context = old_comptime_value;
263    }

265    /// Adds function generics and associated generics (from where clause) to scope.
266    ///
267    /// Returns (generics, associated_generics_trait_constraints) where generics contains
268    /// both associated and explicit generics in the correct order (associated first, then explicit function generics).
269    fn add_function_generics_to_scope(
270        &mut self,
271        func_generics: &UnresolvedGenerics,
272        where_clause: &mut [UnresolvedTraitConstraint],
273    ) -> (Vec<TypeVariable>, Vec<TraitConstraint>) {
274        self.add_generics(func_generics);
275
276        let func_generics = vecmap(&self.generics, |generic| generic.type_var.clone());
277
278        let associated_generics = self.desugar_trait_constraints(where_clause);
279
280        let mut generics = Vec::with_capacity(associated_generics.len());
281        let mut associated_generics_trait_constraints = Vec::new();
282
283        for (associated_generic, bounds) in associated_generics {
284            for bound in bounds {
285                let typ = Type::TypeVariable(associated_generic.type_var.clone());
286                let location = associated_generic.location;
287                self.add_trait_bound_to_scope(location, &typ, &bound, bound.trait_id);
288                associated_generics_trait_constraints
289                    .push(TraitConstraint { typ, trait_bound: bound });
290            }
291
292            generics.push(associated_generic.type_var);
293        }
294
295        // We put associated generics first, as they are implicit and implicit generics
296        // come before explicit generics (see `Type::instantiate_with`).
297        generics.extend(func_generics);
298
299        (generics, associated_generics_trait_constraints)
300    }

302    fn is_function_in_contract(&self) -> bool {
303        if self.self_type.is_some() {
304            // Without this, impl methods can accidentally be placed in contracts.
305            // See: https://github.com/noir-lang/noir/issues/3254
306            false
307        } else {
308            self.in_contract()
309        }
310    }

312    /// True if the `pub` keyword is allowed on parameters in this function
313    /// `pub` on function parameters is only allowed for entry point functions
314    fn pub_allowed(&self, func: &NoirFunction, in_contract: bool, is_crate_root: bool) -> bool {
315        func.is_entry_point(in_contract, is_crate_root) || func.attributes().is_foldable()
316    }

318    /// Resolves function parameters and validates their types for entry points.
319    ///
320    /// Returns (parameters, parameter_types, parameter_idents) where generics and
321    /// trait_constraints may be extended due to `impl Trait` desugaring.
322    fn resolve_function_parameters(
323        &mut self,
324        func: &NoirFunction,
325        generics: &mut Vec<TypeVariable>,
326        trait_constraints: &mut Vec<TraitConstraint>,
327    ) -> ResolvedParametersInfo {
328        let is_crate_root = self.is_at_crate_root();
329        let is_entry_point = func.is_entry_point(self.is_function_in_contract(), is_crate_root);
330        let is_test_or_fuzz = func.is_test_or_fuzz();
331
332        let has_inline_attribute = func.has_inline_attribute();
333        let is_pub_allowed = self.pub_allowed(func, self.is_function_in_contract(), is_crate_root);
334
335        let mut parameters = Vec::new();
336        let mut parameter_types = Vec::new();
337        let mut parameter_idents = Vec::new();
338        let mut parameter_names_in_list = rustc_hash::FxHashMap::default();
339        let wildcard_allowed = WildcardAllowed::No(WildcardDisallowedContext::FunctionParameter);
340
341        for Param { visibility, pattern, typ, location: _ } in func.parameters().iter().cloned() {
342            self.run_lint(|_| {
343                lints::unnecessary_pub_argument(func, visibility, is_pub_allowed).map(Into::into)
344            });
345            self.run_lint(|_| {
346                lints::databus_on_non_entry_point(func, visibility, is_entry_point).map(Into::into)
347            });
348            let type_location = typ.location;
349            let typ = match typ.typ {
350                UnresolvedTypeData::TraitAsType(path, args) => {
351                    self.use_unstable_feature(UnstableFeature::TraitAsType, path.location);
352                    self.desugar_impl_trait_arg(path, args, generics, trait_constraints)
353                }
354                // Function parameters have Kind::Normal
355                _ => self.resolve_type_with_kind(typ, &Kind::Normal, wildcard_allowed),
356            };
357
358            let output = false;
359            if let Err(err) = Self::check_if_type_is_valid_for_program(
360                &typ,
361                is_entry_point || is_test_or_fuzz,
362                has_inline_attribute,
363                output,
364                type_location,
365            ) {
366                self.push_err(err);
367            }
368
369            if is_entry_point || is_test_or_fuzz {
370                self.mark_type_as_used(&typ);
371            }
372
373            let pattern = self.elaborate_pattern_and_store_ids(
374                pattern,
375                typ.clone(),
376                DefinitionKind::Local(None),
377                &mut parameter_idents,
378                true, // warn_if_unused
379                &mut parameter_names_in_list,
380            );
381
382            parameters.push((pattern, typ.clone(), visibility));
383            parameter_types.push(typ);
384        }
385
386        (parameters, parameter_types, parameter_idents)
387    }

389    /// Only sized types are valid to be used as main's parameters or the parameters to a contract
390    /// function. If the given type is not sized (e.g. contains a vector or NamedGeneric type), an
391    /// error is issued.
392    fn check_if_type_is_valid_for_program(
393        typ: &Type,
394        is_entry_point: bool,
395        has_inline_attribute: bool,
396        output: bool,
397        location: Location,
398    ) -> Result<(), TypeCheckError> {
399        if is_entry_point && let Some(invalid_type) = typ.program_validity(output) {
400            return Err(TypeCheckError::InvalidTypeForEntryPoint { invalid_type, location });
401        }
402
403        if has_inline_attribute
404            && !output
405            && let Some(invalid_type) = typ.non_inlined_function_input_validity()
406        {
407            return Err(TypeCheckError::InvalidTypeForEntryPoint { invalid_type, location });
408        }
409
410        Ok(())
411    }

413    fn run_function_lints(&mut self, func: &FuncMeta, modifiers: &FunctionModifiers) {
414        self.run_lint(|_| lints::inlining_attributes(func, modifiers).map(Into::into));
415        self.run_lint(|_| lints::no_predicates_on_entry_point(func, modifiers).map(Into::into));
416        self.run_lint(|_| lints::missing_pub(func, modifiers).map(Into::into));
417        self.run_lint(|_| {
418            let pub_allowed = func.is_entry_point || modifiers.attributes.is_foldable();
419            lints::unnecessary_pub_return(func, modifiers, pub_allowed).map(Into::into)
420        });
421        self.run_lint(|_| lints::oracle_not_marked_unconstrained(func, modifiers).map(Into::into));
422        self.run_lint(|_| lints::oracle_returns_multiple_vectors(func, modifiers).map(Into::into));
423        self.run_lint(|_| lints::oracle_returns_reference(func, modifiers).map(Into::into));
424        self.run_lint(|_| {
425            lints::oracle_returns_vector_with_nested_array(func, modifiers).map(Into::into)
426        });
427        self.run_lint(|elaborator| {
428            lints::low_level_function_outside_stdlib(modifiers, elaborator.crate_id).map(Into::into)
429        });
430        self.run_lint(|elaborator| {
431            lints::oracle_name_clashes_with_stdlib(modifiers, elaborator.crate_id).map(Into::into)
432        });
433        self.run_lint(|_| lints::check_varargs(func, modifiers).map(Into::into));
434    }

436    /// Elaborates a function's body and performs type checking.
437    ///
438    /// This is the second pass of function elaboration that processes the function body,
439    /// resolves all expressions and statements, performs type checking, and verifies
440    /// trait constraints. The function metadata must already be defined by [Self::define_function_meta].
441    pub(crate) fn elaborate_function(&mut self, id: FuncId) {
442        let func_meta = self.interner.func_meta.get_mut(&id);
443        let func_meta =
444            func_meta.expect("FuncMetas should be declared before a function is elaborated");
445
446        let (kind, body, body_location) = match func_meta.take_body() {
447            FunctionBody::Unresolved(kind, body, location) => (kind, body, location),
448            FunctionBody::Resolved => return,
449            // Do not error for the still-resolving case. If there is a dependency cycle,
450            // the dependency cycle check will find it later on.
451            FunctionBody::Resolving => return,
452        };
453
454        let func_meta = func_meta.clone();
455
456        assert_eq!(
457            self.crate_id, func_meta.source_crate,
458            "Functions in other crates should be already elaborated"
459        );
460
461        self.local_module = Some(func_meta.source_module);
462        self.self_type = func_meta.self_type.clone();
463        self.current_trait_impl = func_meta.trait_impl;
464
465        self.scopes.start_function();
466        let old_item = self.current_item.replace(DependencyId::Function(id));
467
468        self.trait_bounds = func_meta.all_trait_constraints().cloned().collect();
469        self.push_function_context();
470
471        // Lints and visibility must be separately from function meta resolution as comptime attribute
472        // may possibly update a function's modifiers.
473        let modifiers = self.interner.function_modifiers(&id).clone();
474        self.run_function_lints(&func_meta, &modifiers);
475        let name = Ident::new(
476            self.interner.definition_name(func_meta.name.id).to_string(),
477            func_meta.name.location,
478        );
479        self.check_function_visibility(&func_meta, &modifiers, &name, func_meta.location);
480
481        self.introduce_generics_into_scope(func_meta.all_generics.clone());
482
483        // The DefinitionIds for each parameter were already created in define_function_meta
484        // so we need to reintroduce the same IDs into scope here.
485        for parameter in &func_meta.parameter_idents {
486            let name = self.interner.definition_name(parameter.id).to_owned();
487            let warn_if_unused = !(func_meta.trait_impl.is_some() && name == "self");
488            // We allow shadowing here because there's no outer scope to shadow
489            // (duplicate parameter names were already checked in `resolve_function_parameters`)
490            let allow_shadowing = true;
491            self.add_existing_variable_to_scope(
492                name,
493                parameter.clone(),
494                warn_if_unused,
495                allow_shadowing,
496            );
497        }
498
499        self.add_trait_constraints_to_scope(func_meta.all_trait_constraints(), func_meta.location);
500
501        let (hir_func, body_type) = match kind {
502            FunctionKind::Builtin
503            | FunctionKind::LowLevel
504            | FunctionKind::TraitFunctionWithoutBody => {
505                if !body.statements.is_empty() {
506                    self.push_err(ResolverError::BuiltinWithBody {
507                        location: func_meta.name.location,
508                    });
509                }
510                (HirFunction::empty(), Type::Error)
511            }
512            FunctionKind::Oracle => {
513                if !body.statements.is_empty() {
514                    self.push_err(ResolverError::OracleWithBody {
515                        location: func_meta.name.location,
516                    });
517                }
518                (HirFunction::empty(), Type::Error)
519            }
520            FunctionKind::Normal => {
521                let return_type = func_meta.return_type();
522                let (block, body_type) = self.elaborate_block(body, Some(return_type));
523                let expr_id = self.interner.push_expr_full(block, body_location, body_type.clone());
524                (HirFunction::unchecked_from_expr(expr_id), body_type)
525            }
526        };
527
528        // Don't verify the return type for builtin functions & trait function declarations
529        if !func_meta.is_stub() {
530            self.type_check_function_body(body_type, &func_meta, hir_func.as_expr());
531        }
532
533        // Default any type variables that still need defaulting and
534        // verify any remaining trait constraints arising from the function body.
535        // This is done before trait impl search since leaving them bindable can lead to errors
536        // when multiple impls are available. Instead we default first to choose the Field or u64 impl.
537        self.check_and_pop_function_context();
538
539        self.remove_trait_constraints_from_scope(func_meta.all_trait_constraints());
540
541        let func_scope_tree = self.scopes.end_function();
542
543        // The arguments to low-level and oracle functions are always unused so we do not produce warnings for them.
544        if !func_meta.is_stub() {
545            self.check_for_unused_variables_in_scope_tree(func_scope_tree);
546        }
547
548        // Check that the body can return without calling the function.
549        if let FunctionKind::Normal = kind {
550            self.run_lint(|elaborator| {
551                lints::unbounded_recursion(
552                    elaborator.interner,
553                    id,
554                    || elaborator.interner.definition_name(func_meta.name.id),
555                    func_meta.name.location,
556                    hir_func.as_expr(),
557                )
558                .map(Into::into)
559            });
560        }
561
562        let meta = self
563            .interner
564            .func_meta
565            .get_mut(&id)
566            .expect("FuncMetas should be declared before a function is elaborated");
567
568        meta.function_body = FunctionBody::Resolved;
569
570        self.trait_bounds.clear();
571        self.interner.update_fn(id, hir_func);
572        self.current_item = old_item;
573    }
