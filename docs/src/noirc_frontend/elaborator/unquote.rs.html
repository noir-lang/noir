1//! Token stream processing for macro unquoting and variable interpolation.
2
3use crate::{
4    ast::Path,
5    hir::comptime::Interpreter,
6    token::{Keyword, LocatedToken, Token, Tokens},
7};
8
9use super::Elaborator;
10
11impl Elaborator<'_> {
12    /// Go through the given tokens looking for a '$' token followed by a variable to unquote.
13    /// Each time these two tokens are found, they are replaced by a new UnquoteMarker token
14    /// containing the ExprId of the resolved variable to unquote.
15    pub fn find_unquoted_exprs_tokens(&mut self, tokens: Tokens) -> Tokens {
16        let token_count = tokens.0.len();
17        let mut new_tokens = Vec::with_capacity(token_count);
18        let mut tokens = tokens.0.into_iter();
19
20        while let Some(token) = tokens.next() {
21            let location = token.location();
22            let token = token.into_token();
23
24            // If we find `quote { ... }` it means this is nested inside another `quote { ... }`.
25            // We need to replace `$...` inside the `quote` tokens as well.
26            if let Token::Quote(tokens) = token {
27                let tokens = self.find_unquoted_exprs_tokens(tokens);
28                new_tokens.push(LocatedToken::new(Token::Quote(tokens), location));
29                continue;
30            }
31
32            if Token::Backslash == token {
33                if let Err(error) =
34                    Interpreter::escape_token(&mut tokens, &mut new_tokens, location)
35                {
36                    self.push_err(error);
37                }
38                continue;
39            }
40
41            let is_unquote = matches!(token, Token::DollarSign);
42            new_tokens.push(LocatedToken::new(token, location));
43
44            if is_unquote {
45                if let Some(next) = tokens.next() {
46                    let location = next.location();
47
48                    match next.into_token() {
49                        Token::Ident(name) => {
50                            // Don't want the leading `$` anymore
51                            new_tokens.pop();
52                            let path = Path::from_single(name, location);
53                            let (expr_id, _) = self.elaborate_variable(path);
54                            new_tokens
55                                .push(LocatedToken::new(Token::UnquoteMarker(expr_id), location));
56                        }
57                        // If we see `$crate` resolve the crate to the current crate now so it
58                        // stays even if the rest of the quote is unquoted elsewhere.
59                        Token::Keyword(Keyword::Crate) => {
60                            new_tokens.pop();
61                            let token = Token::InternedCrate(self.crate_id);
62                            new_tokens.push(LocatedToken::new(token, location));
63                        }
64                        Token::Quote(tokens) => {
65                            new_tokens.pop();
66                            let tokens = self.find_unquoted_exprs_tokens(tokens);
67                            new_tokens.push(LocatedToken::new(Token::Quote(tokens), location));
68                        }
69                        other_next => new_tokens.push(LocatedToken::new(other_next, location)),
70                    }
71                }
72            }
73        }
74
75        Tokens(new_tokens)
76    }
77}
