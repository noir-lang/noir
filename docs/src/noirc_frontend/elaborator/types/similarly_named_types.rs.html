5/// A type that is similarly named to another type, but is actually a different type.
6/// The other type is given in the other element of the tuple returned by [`Elaborator::compute_similarly_named_types`].
7#[derive(Debug, Clone, PartialEq, Eq)]
8pub struct SimilarlyNamedType {
9    /// The fully qualified name of the type.
10    pub name: String,
11    /// The location of the type's name.
12    pub location: Location,
13    /// If the type is from the root crate, this is `None`. Otherwise it's the name of that crate.
14    pub external_crate: Option<String>,
15}
16
17impl Elaborator<'_> {
18    /// Computes a list of types that appear in `actual` and `expected` that have the same base name but
19    /// are actually different types. For example, given two types `foo::Bar` and `baz::Bar`, if a function
20    /// is expected to take `foo::Bar` but is given `baz::Bar`, this function will return `[("foo::Bar", "baz::Bar")]`.
21    /// The check is done recursively, so if the function is expected to take `Generic<foo::Bar>` but is given
22    /// `Generic<baz::Bar>`, this function will also return `[("foo::Bar", "baz::Bar")]`.
23    pub(super) fn compute_similarly_named_types(
24        &self,
25        actual: &Type,
26        expected: &Type,
27    ) -> Vec<(SimilarlyNamedType, SimilarlyNamedType)> {
28        let mut similar_types = Vec::new();
29        self.compute_similarly_named_types_helper(actual, expected, &mut similar_types);
30        similar_types
31    }
32
33    fn compute_similarly_named_types_helper(
34        &self,
35        actual: &Type,
36        expected: &Type,
37        similar_types: &mut Vec<(SimilarlyNamedType, SimilarlyNamedType)>,
38    ) {
39        let actual = actual.follow_bindings_shallow();
40        let expected = expected.follow_bindings_shallow();
41
42        match (actual.as_ref(), expected.as_ref()) {
43            (Type::DataType(data_type_1, generics_1), Type::DataType(data_type_2, generics_2)) => {
44                let data_type_1 = data_type_1.borrow();
45                let data_type_2 = data_type_2.borrow();
46                if data_type_1.id != data_type_2.id
47                    && data_type_1.name.as_str() == data_type_2.name.as_str()
48                {
49                    let actual_similar_type = data_type_to_similarly_named_type(&data_type_1, self);
50                    let expected_similar_type =
51                        data_type_to_similarly_named_type(&data_type_2, self);
52                    similar_types.push((actual_similar_type, expected_similar_type));
53                }
54
55                for (actual, expected) in generics_1.iter().zip(generics_2) {
56                    self.compute_similarly_named_types_helper(actual, expected, similar_types);
57                }
58            }
59            (Type::Tuple(types_1), Type::Tuple(types_2)) => {
60                for (actual, expected) in types_1.iter().zip(types_2) {
61                    self.compute_similarly_named_types_helper(actual, expected, similar_types);
62                }
63            }
64            (Type::Array(_, type_1), Type::Array(_, type_2))
65            | (Type::Vector(type_1), Type::Vector(type_2))
66            | (Type::FmtString(_, type_1), Type::FmtString(_, type_2))
67            | (Type::Reference(type_1, _), Type::Reference(type_2, _)) => {
68                self.compute_similarly_named_types_helper(type_1, type_2, similar_types);
69            }
70            (Type::Function(arg_1, ret_1, env_1, _), Type::Function(arg_2, ret_2, env_2, _)) => {
71                for (actual, expected) in arg_1.iter().zip(arg_2) {
72                    self.compute_similarly_named_types_helper(actual, expected, similar_types);
73                }
74                self.compute_similarly_named_types_helper(ret_1, ret_2, similar_types);
75                self.compute_similarly_named_types_helper(env_1, env_2, similar_types);
76            }
77            _ => {}
78        }
79    }
80}
81
82fn data_type_to_similarly_named_type(
83    data_type: &DataType,
84    elaborator: &Elaborator,
85) -> SimilarlyNamedType {
86    let name = data_type_fully_qualified_name(data_type, elaborator);
87    let location = data_type.name.location();
88    let external_crate = elaborator
89        .crate_graph
90        .find_dependencies(elaborator.crate_graph.root_crate_id(), &data_type.id.module_id().krate)
91        .map(|crates| crates.join("::"));
92    SimilarlyNamedType { name, location, external_crate }
93}
94
95fn data_type_fully_qualified_name(data_type: &DataType, elaborator: &Elaborator) -> String {
96    let module_id = data_type.id.module_id();
97    fully_qualified_module_path(
98        elaborator.def_maps,
99        elaborator.crate_graph,
100        &elaborator.crate_id,
101        module_id,
102    )
