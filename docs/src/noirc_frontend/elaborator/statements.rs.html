41    #[tracing::instrument(level = "trace", skip_all)]
42    fn elaborate_statement_value_with_target_type(
43        &mut self,
44        statement: Statement,
45        target_type: Option<&Type>,
46    ) -> (HirStatement, Type) {
47        match statement.kind {
48            StatementKind::Let(let_stmt) => self.elaborate_local_let(let_stmt),
49            StatementKind::Assign(assign) => self.elaborate_assign(assign),
50            StatementKind::AssignOp(assign_op) => self.elaborate_assign_op(assign_op),
51            StatementKind::For(for_stmt) => self.elaborate_for(for_stmt),
52            StatementKind::Loop(loop_) => self.elaborate_loop(loop_),
53            StatementKind::While(while_) => self.elaborate_while(while_),
54            StatementKind::Break => self.elaborate_jump(true, statement.location),
55            StatementKind::Continue => self.elaborate_jump(false, statement.location),
56            StatementKind::Comptime(statement) => {
57                if self.in_comptime_context() {
58                    // Treat a nested comptime block as a regular block. Nested comptime blocks
59                    // can happen as a result of macro expansion so it wouldn't be good to produce
60                    // a warning in that case.
61                    self.elaborate_statement_value_with_target_type(*statement, target_type)
62                } else {
63                    self.elaborate_comptime_statement(*statement, target_type)
64                }
65            }
66            StatementKind::Expression(expr) => {
67                let (expr, typ) = self.elaborate_expression_with_target_type(expr, target_type);
68                (HirStatement::Expression(expr), typ)
69            }
70            StatementKind::Semi(expr) => {
71                let (expr, _typ) = self.elaborate_expression(expr);
72                (HirStatement::Semi(expr), Type::Unit)
73            }
74            StatementKind::Interned(id) => {
75                let kind = self.interner.get_statement_kind(id);
76                let statement = Statement { kind: kind.clone(), location: statement.location };
77                self.elaborate_statement_value_with_target_type(statement, target_type)
78            }
79            StatementKind::Error => (HirStatement::Error, Type::Error),
80        }
81    }

88    #[tracing::instrument(level = "trace", skip_all)]
89    pub(crate) fn elaborate_statement_with_target_type(
90        &mut self,
91        statement: Statement,
92        target_type: Option<&Type>,
93    ) -> (StmtId, Type) {
94        let ((id, typ), has_errors) =
95            self.with_error_guard(|this| this.elaborate_statement_inner(statement, target_type));
96
97        if has_errors {
98            self.interner.stmts_with_errors.insert(id);
99        }
100
101        (id, typ)
102    }

105    fn elaborate_statement_inner(
106        &mut self,
107        statement: Statement,
108        target_type: Option<&Type>,
109    ) -> (StmtId, Type) {
110        let location = statement.location;
111        let (hir_statement, typ) =

114        (id, typ)

123    /// Elaborate a local or global let statement.
124    /// If this is a global let, the DefinitionId of the global is specified so that
125    /// elaborate_pattern can create a Global definition kind with the correct ID
126    /// instead of a local one with a fresh ID.
127    #[tracing::instrument(level = "trace", skip_all)]
128    pub(super) fn elaborate_let(
129        &mut self,
130        let_stmt: LetStatement,
131        global_id: Option<GlobalId>,
132    ) -> (HirLetStatement, Type) {
133        let previous_in_comptime_context = self.in_comptime_context;
134
135        // If this is a global we need to evaluate its type and value in a new function context.
136        // Also, if it's a comptime global we check the type in a comptime context (so Quoted
137        // and other comptime-only types are allowed).
138        if let Some(global_id) = global_id {
139            self.in_comptime_context = self.interner.get_global_definition(global_id).comptime;
140            self.push_function_context();
141        }
142
143        let no_type = let_stmt.r#type.is_none();
144        let wildcard_allowed = if global_id.is_some() {
145            WildcardAllowed::No(WildcardDisallowedContext::Global)
146        } else {
147            WildcardAllowed::Yes
148        };
149
150        let previous_impl_trait_context = if global_id.is_some() {
151            self.impl_trait_is_disallowed.replace(super::types::ImplTraitDisallowedContext::Global)
152        } else {
153            self.impl_trait_is_disallowed
154        };
155        let annotated_type = self.resolve_inferred_type(let_stmt.r#type, wildcard_allowed);
156        self.impl_trait_is_disallowed = previous_impl_trait_context;
157
158        // After resolving the type we'll elaborate the global's value. The value is interpreted
159        // at compile time, so we need to switch to a comptime context. For example using `Quoted`
160        // is allowed in global values, even in non-comptime globals, as long as these comptime-only
161        // values do not survive until runtime (if they do survive, the monomorphizer will catch this).
162        if global_id.is_some() {
163            self.in_comptime_context = true;
164        }
165
166        let pattern_location = let_stmt.pattern.location();
167        let expr_location = let_stmt.expression.location;
168
169        // If this is a global, the global's value will be interpreted at compile time later on,
170        // which means the expression must be elaborated in a comptime context. For example
171        // Quoted and other comptime-only types are allowed here (though if they survive until
172        // runtime the monomorphizer will detect this).
173        let (expression, expr_type) = if no_type {
174            self.elaborate_expression(let_stmt.expression)
175        } else {
176            self.elaborate_expression_with_target_type(let_stmt.expression, Some(&annotated_type))
177        };
178
179        // Require the top-level of a global's type to be fully-specified
180        if global_id.is_some() && (no_type || annotated_type.contains_type_variable()) {
181            let expected_type = annotated_type.clone();
182            let error = ResolverError::UnspecifiedGlobalType {
183                pattern_location,
184                expr_location,
185                expected_type,
186            };
187            self.push_err(error);
188        }
189
190        let definition = match global_id {
191            None => {
192                assert!(!let_stmt.is_global_let);
193                DefinitionKind::Local(Some(expression))
194            }
195            Some(id) => {
196                assert!(let_stmt.is_global_let);
197                DefinitionKind::Global(id)
198            }
199        };
200
201        // Now check if LHS is the same type as the RHS
202        // Importantly, we do not coerce any types implicitly
203        self.unify_with_coercions(
204            &expr_type,
205            &annotated_type,
206            expression,
207            expr_location,
208            |elaborator| {
209                CompilationError::TypeError(elaborator.new_type_mismatch_error(
210                    &expr_type,
211                    &annotated_type,
212                    expr_location,
213                ))
214            },
215        );
216
217        let warn_if_unused =
218            !let_stmt.attributes.iter().any(|attr| attr.kind.is_allow("unused_variables"));
219        let warn_if_not_mutated =
220            !let_stmt.attributes.iter().any(|attr| attr.kind.is_allow("unused_mut"));
221
222        let r#type = annotated_type;
223        let mut parameter_names_in_list = rustc_hash::FxHashMap::default();
224        let pattern = self.elaborate_pattern(
225            let_stmt.pattern,
226            r#type.clone(),
227            definition,
228            warn_if_unused,
229            warn_if_not_mutated,
230            &mut parameter_names_in_list,
231        );
232
233        let attributes = let_stmt.attributes;
234        let comptime = let_stmt.comptime;
235        let is_global_let = let_stmt.is_global_let;
236        let let_ =
237            HirLetStatement::new(pattern, r#type, expression, attributes, comptime, is_global_let);
238
239        if global_id.is_some() {
240            self.check_and_pop_function_context();
241            self.in_comptime_context = previous_in_comptime_context;
242        }
243
244        (let_, Type::Unit)
245    }

248    pub(super) fn elaborate_assign(&mut self, assign: AssignStatement) -> (HirStatement, Type) {
249        let expr_location = assign.expression.location;
250        let (expression, expr_type) = self.elaborate_expression(assign.expression);
251        let (lvalue, lvalue_type, mutable, new_statements) = self.elaborate_lvalue(assign.lvalue);

258            expr_location,
259            new_statements,

263    #[tracing::instrument(level = "trace", skip_all)]
264    pub(super) fn elaborate_assign_op(
265        &mut self,
266        assign_op: AssignOpStatement,
267    ) -> (HirStatement, Type) {
268        // Transform `lvalue <op>= rhs` into `lvalue = lvalue <op> rhs`.
269        //
270        // We must elaborate the lvalue first and reuse its HIR form for the read side of the
271        // infix expression. This prevents index sub-expressions from being evaluated twice.
272        // For example, `x[f()] += 1` must call `f()` only once.
273        //
274        // `elaborate_lvalue` already extracts any side-effectful index expressions into fresh
275        // let bindings (new_statements) and replaces them with simple ident references, so the
276        // HIR lvalue we get back is safe to convert into a read expression without re-evaluating
277        // any side effects.
278        let expression_location = assign_op.expression.location;
279
280        let (hir_lvalue, lvalue_type, mutable, new_statements) =
281            self.elaborate_lvalue(assign_op.lvalue);
282
283        // Convert the HIR lvalue into a read expression, reusing the same ident ExprIds
284        // that were already bound by the index let-statements in new_statements.
285        let lhs_expr = self.hir_lvalue_as_expr(&hir_lvalue);
286        let lhs_type = lvalue_type.clone();
287
288        // Elaborate the right-hand side of the operator.
289        let (rhs_expr, rhs_type) = self.elaborate_expression(assign_op.expression);
290
291        let op_kind = assign_op.op.contents.to_binary_op_kind();
292        let operator = HirBinaryOp { kind: op_kind, location: assign_op.op.location() };
293        let (expression, expression_type) = self.finish_infix(
294            lhs_expr,
295            lhs_type,
296            operator,
297            rhs_expr,
298            rhs_type,
299            expression_location,
300        );
301
302        self.finish_assign(
303            hir_lvalue,
304            lvalue_type,
305            mutable,
306            expression,
307            expression_type,
308            expression_location,
309            new_statements,
310        )
311    }

313    /// Complete assignment elaboration given an already-elaborated lvalue and rhs expression.
314    /// Handles mutability checks, type unification, and wrapping in a block when index
315    /// let-bindings were emitted by `elaborate_lvalue`.
316    #[allow(clippy::too_many_arguments)]
317    fn finish_assign(
318        &mut self,
319        lvalue: HirLValue,
320        lvalue_type: Type,
321        mutable: bool,
322        expression: ExprId,
323        expr_type: Type,
324        location: Location,
325        mut new_statements: Vec<StmtId>,
326    ) -> (HirStatement, Type) {
327        self.mark_lvalue_variables_as_mutated(&lvalue);
328
329        if !mutable {
330            self.push_assign_to_immutable_lvalue_error(&lvalue, &lvalue);
331        } else {
332            let (id, name, loc) = self.get_lvalue_error_info(&lvalue);
333            if let Some(id) = id {
334                self.check_can_mutate_lambda_capture(id, name, loc);
335            }
336        }
337
338        self.unify_with_coercions(&expr_type, &lvalue_type, expression, location, |elaborator| {
339            CompilationError::TypeError(elaborator.new_type_mismatch_with_source_error(
340                &expr_type,
341                &lvalue_type,
342                Source::Assignment,
343                location,
344            ))
345        });
346
347        let assign = HirAssignStatement { lvalue, expression };
348        let assign = HirStatement::Assign(assign);
349
350        if new_statements.is_empty() {
351            (assign, Type::Unit)
352        } else {
353            let assign = self.interner.push_stmt_full(assign, location);
354            new_statements.push(assign);
355            let block = HirExpression::Block(HirBlockExpression { statements: new_statements });
356            let block = self.interner.push_expr_full(block, location, Type::Unit);
357            (HirStatement::Expression(block), Type::Unit)
358        }
359    }

361    /// Convert a HIR lvalue into a read expression, reusing the same ident ExprIds.
362    /// This is used to build the read side of a desugared op-assign without re-evaluating
363    /// any index sub-expressions.
364    fn hir_lvalue_as_expr(&mut self, lvalue: &HirLValue) -> ExprId {
365        let location = lvalue.location();
366        match lvalue {
367            HirLValue::Ident(ident, typ) => {
368                let expr = HirExpression::Ident(ident.clone(), None);
369                self.interner.push_expr_full(expr, location, typ.clone())
370            }
371            HirLValue::MemberAccess { object, field_name, field_index, typ, location } => {
372                let lhs = self.hir_lvalue_as_expr(object);
373                let expr = HirExpression::MemberAccess(HirMemberAccess {
374                    lhs,
375                    rhs: field_name.clone(),
376                    is_offset: false,
377                });
378                let expr_id = self.interner.push_expr_full(expr, *location, typ.clone());
379                if let Some(index) = field_index {
380                    self.interner.set_field_index(expr_id, *index);
381                }
382                expr_id
383            }
384            HirLValue::Index { array, index, typ, location } => {
385                let collection = self.hir_lvalue_as_expr(array);
386                let expr = HirExpression::Index(HirIndexExpression { collection, index: *index });
387                self.interner.push_expr_full(expr, *location, typ.clone())
388            }
389            HirLValue::Dereference { lvalue, element_type, implicitly_added, location } => {
390                let inner = self.hir_lvalue_as_expr(lvalue);
391                let expr = HirExpression::Prefix(HirPrefixExpression::new(
392                    UnaryOp::Dereference { implicitly_added: *implicitly_added },
393                    inner,
394                ));
395                self.interner.push_expr_full(expr, *location, element_type.clone())
396            }
397            HirLValue::Error { location } => {
398                self.interner.push_expr_full(HirExpression::Error, *location, Type::Error)
399            }
400        }
401    }

403    #[tracing::instrument(level = "trace", skip_all)]
404    fn push_assign_to_immutable_lvalue_error(
405        &mut self,
406        lvalue: &HirLValue,
407        main_lvalue: &HirLValue,
408    ) {
409        match lvalue {
410            HirLValue::Ident(ident, _) => {
411                let definition = self.interner.definition(ident.id);
412                let name = definition.name.clone();
413                let location = ident.location;
414                self.push_err(TypeCheckError::VariableMustBeMutable { name, location });
415            }
416
417            HirLValue::MemberAccess { object, .. } => {
418                self.push_assign_to_immutable_lvalue_error(object, main_lvalue);
419            }
420            HirLValue::Index { array, .. } => {
421                self.push_assign_to_immutable_lvalue_error(array, main_lvalue);
422            }
423            HirLValue::Dereference { lvalue, .. } => {
424                if let HirLValue::Ident(ident, _) = lvalue.as_ref() {
425                    let definition = self.interner.definition(ident.id);
426                    let name = definition.name.clone();
427                    let location = ident.location;
428                    self.push_err(TypeCheckError::CannotAssignToReference { name, location });
429                } else {
430                    let lvalue = main_lvalue.to_display_ast(self.interner).to_string();
431                    let location = main_lvalue.location();
432                    self.push_err(TypeCheckError::CannotAssignToLValueBehindReference {
433                        lvalue,
434                        location,
435                    });
436                }
437            }
438            HirLValue::Error { .. } => {
439                // An error was already pushed for this
440            }
441        }
442    }

444    #[tracing::instrument(level = "trace", skip_all)]
445    pub(super) fn elaborate_for(&mut self, for_loop: ForLoopStatement) -> (HirStatement, Type) {
446        let (start, end, inclusive) = match for_loop.range {
447            ForRange::Range(bounds) => (bounds.start, bounds.end, bounds.inclusive),
448            ForRange::Array(_) => {
449                let for_stmt =
450                    for_loop.range.into_for(for_loop.identifier, for_loop.block, for_loop.location);
451
452                return self.elaborate_statement_value(for_stmt);
453            }
454        };
455
456        let start_location = start.location;
457        let end_location = end.location;
458
459        let (start_range, start_range_type) = self.elaborate_expression(start);
460        let (end_range, end_range_type) = self.elaborate_expression(end);
461        let (identifier, block) = (for_loop.identifier, for_loop.block);
462
463        let old_loop = std::mem::take(&mut self.current_loop);
464
465        self.current_loop = Some(Loop { is_for: true, has_break: false });
466        self.push_scope();
467
468        let kind = DefinitionKind::Local(None);
469        let identifier = self.add_variable_decl(
470            identifier, false, // mutable
471            true,  // allow_shadowing
472            true,  // warn_if_unused
473            true,  // warn_if_not_mutated
474            kind,
475        );
476
477        // Check that start range and end range have the same types
478        self.unify_or_type_mismatch(&end_range_type, &start_range_type, end_location);
479
480        let expected_type = self.polymorphic_integer();
481
482        self.unify(&start_range_type, &expected_type, |_| TypeCheckError::TypeCannotBeUsed {
483            typ: start_range_type.clone(),
484            place: "for loop",
485            location: start_location,
486        });
487
488        self.interner.push_definition_type(identifier.id, start_range_type);
489
490        let block_location = block.type_location();
491        let (block, block_type) = self.elaborate_expression(block);
492
493        self.unify_or_type_mismatch(&block_type, &Type::Unit, block_location);
494
495        self.pop_scope();
496        self.current_loop = old_loop;
497
498        let statement = HirStatement::For(HirForStatement {
499            start_range,
500            end_range,
501            block,
502            identifier,
503            inclusive,
504        });
505
506        (statement, Type::Unit)
507    }

509    #[tracing::instrument(level = "trace", skip_all)]
510    pub(super) fn elaborate_loop(&mut self, loop_: LoopStatement) -> (HirStatement, Type) {
511        let LoopStatement { body: block, loop_keyword_location: location } = loop_;
512        let in_constrained_function = self.in_constrained_function();
513        if in_constrained_function {
514            self.push_err(ResolverError::LoopInConstrainedFn { location });
515        }
516
517        let old_loop = std::mem::take(&mut self.current_loop);
518        self.current_loop = Some(Loop { is_for: false, has_break: false });
519        self.push_scope();
520
521        let block_location = block.type_location();
522        let (block, block_type) = self.elaborate_expression(block);
523
524        self.unify_or_type_mismatch(&block_type, &Type::Unit, block_location);
525
526        self.pop_scope();
527
528        let last_loop =
529            std::mem::replace(&mut self.current_loop, old_loop).expect("Expected a loop");
530        if !last_loop.has_break {
531            self.push_err(ResolverError::LoopWithoutBreak { location });
532        }
533
534        let statement = HirStatement::Loop(block);
535
536        (statement, Type::Unit)
537    }

539    #[tracing::instrument(level = "trace", skip_all)]
540    pub(super) fn elaborate_while(&mut self, while_: WhileStatement) -> (HirStatement, Type) {
541        let in_constrained_function = self.in_constrained_function();
542        if in_constrained_function {
543            self.push_err(ResolverError::WhileInConstrainedFn {
544                location: while_.while_keyword_location,
545            });
546        }
547
548        let old_loop = std::mem::take(&mut self.current_loop);
549        self.current_loop = Some(Loop { is_for: false, has_break: false });
550        self.push_scope();
551
552        let location = while_.condition.type_location();
553        let (condition, cond_type) = self.elaborate_expression(while_.condition);
554
555        self.unify_or_type_mismatch(&cond_type, &Type::Bool, location);
556
557        let block_location = while_.body.type_location();
558        let (block, block_type) = self.elaborate_expression(while_.body);
559
560        self.unify_or_type_mismatch(&block_type, &Type::Unit, block_location);
561
562        self.pop_scope();
563
564        std::mem::replace(&mut self.current_loop, old_loop).expect("Expected a loop");
565
566        let statement = HirStatement::While(condition, block);
567
568        (statement, Type::Unit)
569    }

571    #[tracing::instrument(level = "trace", skip_all)]
572    fn elaborate_jump(&mut self, is_break: bool, location: Location) -> (HirStatement, Type) {
573        let in_constrained_function = self.in_constrained_function();
574
575        if in_constrained_function {
576            self.push_err(ResolverError::JumpInConstrainedFn { is_break, location });
577        }
578
579        if let Some(current_loop) = &mut self.current_loop {
580            if is_break {
581                current_loop.has_break = true;
582            }
583        } else {
584            self.push_err(ResolverError::JumpOutsideLoop { is_break, location });
585        }
586
587        let expr = if is_break { HirStatement::Break } else { HirStatement::Continue };
588        (expr, Type::Unit)
589    }

591    fn get_lvalue_error_info(
592        &self,
593        lvalue: &HirLValue,
594    ) -> (Option<DefinitionId>, String, Location) {
595        match lvalue {
596            HirLValue::Ident(name, _) => {
597                let location = name.location;
598                let definition = self.interner.definition(name.id);
599                (Some(name.id), definition.name.clone(), location)
600            }
601            HirLValue::MemberAccess { object, .. } => self.get_lvalue_error_info(object),
602            HirLValue::Index { array, .. } => self.get_lvalue_error_info(array),
603            HirLValue::Dereference { lvalue, .. } => self.get_lvalue_error_info(lvalue),
604            HirLValue::Error { location, .. } => (None, "(undeclared variable)".into(), *location),
605        }
606    }

608    /// Elaborates an lvalue returning:
609    /// - The HirLValue equivalent of the given `lvalue`
610    /// - The type being assigned to
611    /// - Whether the underlying variable is mutable
612    /// - A vector of new statements which need to prefix the resulting assign statement.
613    ///   This hoists out any sub-expressions to simplify sequencing of side-effects.
614    #[tracing::instrument(level = "trace", skip_all)]
615    fn elaborate_lvalue(&mut self, lvalue: LValue) -> (HirLValue, Type, bool, Vec<StmtId>) {
616        match lvalue {
617            LValue::Path(path) => {
618                let location = path.location;
619                let path = self.validate_path(path);
620                match self.get_ident_from_path_or_error(path.clone()) {
621                    Ok(IdentFromPath::Variable(variable)) => {
622                        self.check_if_variable_is_captured_by_closure(&variable);
623                        self.elaborate_lvalue_ident(variable.ident, location)
624                    }
625                    Ok(IdentFromPath::Definition { id, item: _ }) => {
626                        let ident = HirIdent::non_trait_method(id, location);
627                        self.elaborate_lvalue_ident(ident, location)
628                    }
629                    Ok(IdentFromPath::TypeAlias(type_alias_id)) => {
630                        let type_alias = self.interner.get_type_alias(type_alias_id);
631                        self.push_err(ResolverError::Expected {
632                            location,
633                            expected: "value",
634                            found: format!("type alias `{}`", type_alias.borrow().name),
635                        });
636                        let mutable = true;
637                        (HirLValue::Error { location }, Type::Error, mutable, Vec::new())
638                    }
639                    Err(error) => {
640                        // We couldn't find a variable or global. Let's see if the identifier refers to something
641                        // else, like a module or type.
642                        let result = self.resolve_path_inner(
643                            path,
644                            PathResolutionTarget::Value,
645                            super::PathResolutionMode::MarkAsUsed,
646                        );
647                        if let Ok(result) = result {
648                            self.push_errors(result.errors);
649                            self.push_err(ResolverError::Expected {
650                                location,
651                                expected: "value",
652                                found: result.item.description(self.interner),
653                            });
654                        } else {
655                            self.push_err(error);
656                        }
657
658                        // We se this to mutable to prevent further errors from being reported
659                        let mutable = true;
660                        (HirLValue::Error { location }, Type::Error, mutable, Vec::new())
661                    }
662                }
663            }
664            LValue::MemberAccess { object, field_name, location } => {
665                let (object, lhs_type, mut mutable, statements) = self.elaborate_lvalue(*object);
666                let mut object = Box::new(object);
667
668                let object_ref = &mut object;
669                let mutable_ref = &mut mutable;
670
671                let dereference_lhs = move |_: &mut Self, _, element_type, is_mutable| {
672                    // We must create a temporary value first to move out of object_ref before
673                    // we eventually reassign to it.
674                    let tmp_value = HirLValue::Error { location };
675
676                    let lvalue = std::mem::replace(object_ref, Box::new(tmp_value));
677                    **object_ref = HirLValue::Dereference {
678                        lvalue,
679                        element_type,
680                        location,
681                        implicitly_added: true,
682                    };
683                    *mutable_ref = is_mutable;
684                };
685
686                let name = field_name.as_str();
687                let (object_type, field_index) = self
688                    .check_field_access(
689                        &lhs_type,
690                        name,
691                        field_name.location(),
692                        Some(dereference_lhs),
693                    )
694                    .unwrap_or((Type::Error, 0));
695
696                let field_index = Some(field_index);
697                let typ = object_type.clone();
698                let lvalue =
699                    HirLValue::MemberAccess { object, field_name, field_index, typ, location };
700                (lvalue, object_type, mutable, statements)
701            }
702            LValue::Index { array, index, location } => {
703                let expr_location = index.location;
704                let (mut index, index_type) = self.elaborate_expression(index);
705
706                let expected = Type::u32();
707                self.unify_or_type_mismatch_with_source(
708                    &index_type,
709                    &expected,
710                    Source::ArrayIndex,
711                    expr_location,
712                );
713
714                let (mut lvalue, mut lvalue_type, mut mutable, mut statements) =
715                    self.elaborate_lvalue(*array);
716
717                // Push the index expression to the end of the new statements list, referring to it
718                // afterward with a let binding. Note that since we recur first then push to the
719                // end of the list we're evaluating side-effects such that in `a[i][j]`, `i` will
720                // be evaluated first, followed by `j`.
721                if let Some((index_definition, new_index)) =
722                    self.fresh_definition_for_lvalue_index(index, index_type, expr_location)
723                {
724                    index = new_index;
725                    statements.push(index_definition);
726                }
727
728                // Before we check that the lvalue is an array, try to dereference it as many times
729                // as needed to unwrap any `&` or `&mut` wrappers.
730                while let Type::Reference(element, is_mutable) = lvalue_type.follow_bindings() {
731                    let element_type = element.as_ref().clone();
732                    lvalue = HirLValue::Dereference {
733                        lvalue: Box::new(lvalue),
734                        element_type,
735                        location,
736                        implicitly_added: true,
737                    };
738                    lvalue_type = *element;
739                    mutable = is_mutable;
740                }
741
742                let typ = match lvalue_type.follow_bindings() {
743                    Type::Array(ref size, ref elem_type) => {
744                        self.check_array_index_out_of_bounds(size, &index, location);
745                        *elem_type.clone()
746                    }
747                    Type::Vector(elem_type) => *elem_type,
748                    Type::Error => Type::Error,
749                    Type::String(_) => {
750                        let (_id, _lvalue_name, lvalue_location) =
751                            self.get_lvalue_error_info(&lvalue);
752                        self.push_err(TypeCheckError::StringIndexAssign {
753                            location: lvalue_location,
754                        });
755                        Type::Error
756                    }
757                    Type::TypeVariable(_) => {
758                        self.push_err(TypeCheckError::TypeAnnotationsNeededForIndex { location });
759                        Type::Error
760                    }
761                    other => {
762                        self.push_err(TypeCheckError::TypeMismatch {
763                            expected_typ: "array".to_string(),
764                            expr_typ: other.to_string(),
765                            expr_location: location,
766                            similarly_named_types: Vec::new(),
767                        });
768                        Type::Error
769                    }
770                };
771
772                let array = Box::new(lvalue);
773                let array_type = typ.clone();
774                (HirLValue::Index { array, index, typ, location }, array_type, mutable, statements)
775            }
776            LValue::Dereference(lvalue, location) => {
777                let (lvalue, reference_type, mut mutable, statements) =
778                    self.elaborate_lvalue(*lvalue);
779                let lvalue = Box::new(lvalue);
780
781                let element_type = Type::type_variable(self.interner.next_type_variable_id());
782
783                // Always expect a mutable reference here since we're storing to it
784                let expected_type = Type::Reference(Box::new(element_type.clone()), true);
785
786                self.unify_or_type_mismatch(&reference_type, &expected_type, location);
787
788                // Check mutability after unification so that type variables are resolved first
789                if let Type::Reference(_, is_mutable) =
790                    reference_type.follow_bindings_shallow().as_ref()
791                {
792                    mutable = *is_mutable;
793                }
794
795                let typ = element_type.clone();
796                let lvalue = HirLValue::Dereference {
797                    lvalue,
798                    element_type,
799                    location,
800                    implicitly_added: false,
801                };
802                (lvalue, typ, mutable, statements)
803            }
804            LValue::Interned(id, location) => {
805                let lvalue = self.interner.get_lvalue(id, location);
806                self.elaborate_lvalue(lvalue)
807            }
808        }
809    }

811    #[tracing::instrument(level = "trace", skip_all)]
812    fn elaborate_lvalue_ident(
813        &mut self,
814        ident: HirIdent,
815        location: Location,
816    ) -> (HirLValue, Type, bool, Vec<StmtId>) {
817        let definition = self.interner.definition(ident.id);
818        let mutable = definition.mutable;
819
820        if definition.comptime && !self.in_comptime_context() {
821            self.push_err(ResolverError::MutatingComptimeInNonComptimeContext {
822                name: definition.name.clone(),
823                location: ident.location,
824            });
825        }
826
827        let typ = self.interner.definition_type(ident.id).instantiate(self.interner).0;
828        let typ = typ.follow_bindings();
829
830        self.interner.add_local_reference(ident.id, location);
831
832        (HirLValue::Ident(ident, typ.clone()), typ, mutable, Vec::new())
833    }

835    #[tracing::instrument(level = "trace", skip_all)]
836    fn fresh_definition_for_lvalue_index(
837        &mut self,
838        expr: ExprId,
839        typ: Type,
840        location: Location,
841    ) -> Option<(StmtId, ExprId)> {
842        // If the original expression trivially cannot have side-effects, don't bother cluttering
843        // the output with a let binding. Note that array literals can have side-effects but these
844        // would produce type errors anyway.
845        if !self.index_could_have_side_effects(expr) {
846            return None;
847        }
848
849        let lvalue_index_counter = self.next_lvalue_index_counter();
850        let id = self.interner.push_definition(
851            format!("i_{lvalue_index_counter}"),
852            false,
853            false,
854            DefinitionKind::Local(None),
855            location,
856        );
857        let ident = HirIdent::non_trait_method(id, location);
858        let ident_expr = HirExpression::Ident(ident.clone(), None);
859
860        let ident_id = self.interner.push_expr_full(ident_expr, location, typ.clone());
861
862        let pattern = HirPattern::Identifier(ident);
863        let let_ = HirStatement::Let(HirLetStatement::basic(pattern, typ, expr));
864        let let_ = self.interner.push_stmt_full(let_, location);
865        Some((let_, ident_id))
866    }

868    /// Returns `true` if the given index expression could potentially have side-effects.
869    /// Returns `false` if the expression is guaranteed to not have side-effects.
870    fn index_could_have_side_effects(&self, expr: ExprId) -> bool {
871        match self.interner.expression_ref(&expr) {
872            HirExpression::Ident(..) => false,
873            HirExpression::Literal(hir_literal) => match hir_literal {
874                HirLiteral::Bool(_)
875                | HirLiteral::Integer(..)
876                | HirLiteral::Str(_)
877                | HirLiteral::Unit => false,
878                HirLiteral::Array(..) | HirLiteral::Vector(..) | HirLiteral::FmtStr(..) => true,
879            },
880            HirExpression::Prefix(hir_prefix_expression) => {
881                hir_prefix_expression.trait_method_id.is_some()
882                    || self.index_could_have_side_effects(hir_prefix_expression.rhs)
883            }
884            HirExpression::Infix(hir_infix_expression) => {
885                hir_infix_expression.trait_method_id.is_some()
886                    || self.index_could_have_side_effects(hir_infix_expression.lhs)
887                    || self.index_could_have_side_effects(hir_infix_expression.rhs)
888            }
889            HirExpression::Index(hir_index_expression) => {
890                self.index_could_have_side_effects(hir_index_expression.collection)
891                    || self.index_could_have_side_effects(hir_index_expression.index)
892            }
893            HirExpression::Constructor(hir_constructor_expression) => {
894                !hir_constructor_expression.fields.is_empty()
895            }
896            HirExpression::EnumConstructor(hir_enum_constructor_expression) => {
897                !hir_enum_constructor_expression.arguments.is_empty()
898            }
899            HirExpression::MemberAccess(hir_member_access) => {
900                self.index_could_have_side_effects(hir_member_access.lhs)
901            }
902            HirExpression::Call(..) => true,
903            HirExpression::Constrain(..) => true,
904            HirExpression::Cast(hir_cast_expression) => {
905                self.index_could_have_side_effects(hir_cast_expression.lhs)
906            }
907            HirExpression::If(..)
908            | HirExpression::Match(..)
909            | HirExpression::Tuple(..)
910            | HirExpression::Lambda(..)
911            | HirExpression::Quote(..)
912            | HirExpression::Unquote(..)
913            | HirExpression::Unsafe(..)
914            | HirExpression::Block(..)
915            | HirExpression::Error => true,
916        }
917    }

919    #[tracing::instrument(level = "trace", skip_all)]
920    fn elaborate_comptime_statement(
921        &mut self,
922        statement: Statement,
923        target_type: Option<&Type>,
924    ) -> (HirStatement, Type) {
925        let location = statement.location;
926        let hir_statement = self.elaborate_in_comptime_context(|this| {
927            let (hir_statement, typ) = this.elaborate_statement(statement);
928
929            // If the comptime statement is expected to return a specific type, unify their types.
930            // This for example allows this code to compile:
931            //
932            // ```
933            // fn foo() -> u8 {
934            //   comptime { 1 }
935            // }
936            // ```
937            //
938            // If we don't do this, "1" will end up with the default integer or field type,
939            // which is Field.
940            if let Some(target_type) = target_type {
941                this.unify_or_type_mismatch(&typ, target_type, location);
942            }
943
944            hir_statement
945        });
946
947        // Run the interpreter - it will check if execution has been halted
948        let mut interpreter = self.setup_interpreter();
949        let value = interpreter.evaluate_statement(hir_statement);
950
951        let (expr, typ) = self.inline_comptime_value(value, location);
952
953        let location = self.interner.id_location(hir_statement);
954        self.debug_comptime(location, |interner| expr.to_display_ast(interner).kind);
955
956        (HirStatement::Expression(expr), typ)
957    }
