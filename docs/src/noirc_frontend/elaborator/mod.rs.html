1//! # The Elaborator
2//!
3//! The elaborator is the core semantic analysis phase of the compiler. It transforms collected,
4//! unresolved AST items into fully resolved and type-checked HIR through simultaneous and intertwined
5//! name resolution, type checking, trait resolution, and macro expansion. It also handles pattern
6//! elaboration, generic type processing, comptime interpretation, reference validation, and scope management.
7//!
8//! ## Architecture Overview
9//!
10//! The elaborator operates in several distinct phases, processing items in a specific order to handle
11//! dependencies correctly:
12//!
13//! ### Early Resolution
14//! 1. Globals - Set up their dependency ordering. Deferred for elaboration later after type resolution.
15//!    Globals will be lazily elaborated when other types or expressions bring them into scope.
16//! 2. Type aliases - Defined to allow their use in subsequent type definitions
17//!
18//! ### Type Collection
19//! 1. Struct definitions - Collected so their types are available for use
20//! 2. Enum definitions - Collected so their types are available for use
21//! 3. Trait definitions - Collected so trait bounds can be resolved
22//!
23//! ### Function metadata and Implementations
24//! 1. Function metadata - Signatures collected before bodies are elaborated
25//! 2. Trait methods - Method signatures collected from trait definitions
26//! 3. Impl blocks - Methods organized into their proper modules based on the impl's type
27//! 4. Trait impls - Linked to their corresponding traits and validated
28//!
29//! ### Global Elaboration
30//! - Elaborate any remaining globals which were not brought into scope.
31//!   Elaborated after type resolution since they may use struct types which need global type information
32//!
33//! ### Attribute Processing
34//! - Comptime attributes - Executed before function body elaboration, as generated items may change what is in scope or modify functions
35//!
36//! ### Full Item Elaboration
37//! 1. Functions - Function bodies elaborated (resolved & type-checked)
38//! 2. Traits - Trait default method implementations elaborated
39//! 3. Impls - Implementation method bodies elaborated
40//! 4. Trait impls - Trait implementation method bodies elaborated and validated against trait signatures
41//!
42//! ### Dependency Analysis
43//! Detect and report dependency cycles to prevent infinite elaboration loops
44//!
45//! ## Error Handling
46//!
47//! The elaborator accumulates errors rather than failing fast, allowing it to report multiple
48//! errors in a single compilation pass. Errors are collected throughout elaboration and may be
49//! wrapped with additional context when elaborating generated code (e.g., from attributes or
50//! comptime calls).

127/// Maximum number of recursive calls allowed at comptime.
128///
129/// Ideally we would like this to be 1000 to match what happens in ACIR,
130/// however due to the overhead of the `Interpreter` itself Rust itself
131/// would exhaust the stack earlier (or later, because `nargo` increases
132/// the stack size for parsing for example).
133///
134/// Potentially we could increase this if the `Interpreter` used an iterative
135/// strategy instead of recursion.
136///
137/// Note that if we increase this, currently we would hit the `MAX_EVALUATION_DEPTH`.
138const MAX_INTERPRETER_CALL_STACK_SIZE: usize = 100;
139
140/// Maximum depth of macro expansion (attribute execution).
141///
142/// This prevents infinite recursion when an attribute generates code that
143/// triggers further attribute expansion, which would otherwise cause a
144/// Rust stack overflow.
145///
146/// This limit is lower than the [interpreter call stack limit][MAX_INTERPRETER_CALL_STACK_SIZE] because macro
147/// expansion involves more stack frames per level (elaboration, comptime evaluation, etc.).
148pub(crate) const MAX_MACRO_EXPANSION_DEPTH: usize = 32;

296#[derive(Copy, Clone)]
297pub enum ElaborateReason {
298    /// A function attribute generated an item that's being elaborated.
299    RunningAttribute(Location),
300    /// Evaluating a comptime call like `Module::add_item`
301    EvaluatingComptimeCall(&'static str, Location),
302}
303
304impl ElaborateReason {
305    fn to_macro_error(self, error: CompilationError) -> ComptimeError {
306        match self {
307            ElaborateReason::RunningAttribute(location) => {
308                ComptimeError::ErrorRunningAttribute { error: Box::new(error), location }
309            }
310            ElaborateReason::EvaluatingComptimeCall(method_name, location) => {
311                let error = Box::new(error);
312                ComptimeError::ErrorEvaluatingComptimeCall { method_name, error, location }
313            }
314        }
315    }
316}

421    pub(crate) fn elaborate_items(&mut self, mut items: CollectedItems) {
422        self.set_unresolved_globals_ordering(items.globals);
423
424        for (alias_id, alias) in items.type_aliases {
425            self.define_type_alias(alias_id, alias);
426        }
427
428        // Must resolve types before we resolve globals.
429        self.collect_struct_definitions(&items.structs);
430        self.collect_enum_definitions(&items.enums);
431        self.collect_traits(&mut items.traits);
432
433        self.define_function_metas(&mut items.functions, &mut items.impls, &mut items.trait_impls);
434
435        // Before we resolve any function symbols we must go through our impls and
436        // re-collect the methods within into their proper module. This cannot be
437        // done during def collection since we need to be able to resolve the type of
438        // the impl since that determines the module we should collect into.
439        for ((self_type, module), impls) in &mut items.impls {
440            self.collect_impls(*module, impls, self_type);
441        }
442
443        self.collect_trait_methods(&mut items.traits);
444
445        // Bind trait impls to their trait. Collect trait functions, that have a
446        // default implementation, which hasn't been overridden.
447        for trait_impl in &mut items.trait_impls {
448            self.collect_trait_impl(trait_impl);
449        }
450
451        // We must wait to resolve non-literal globals until after we resolve structs since struct
452        // globals will need to reference the struct type they're initialized to ensure they are valid.
453        self.elaborate_remaining_globals();
454
455        // We have to run any comptime attributes on functions before the function is elaborated
456        // since the generated items are checked beforehand as well.
457        self.run_attributes(
458            &items.traits,
459            &items.structs,
460            &items.functions,
461            &items.module_attributes,
462        );
463
464        for functions in items.functions {
465            self.elaborate_functions(functions);
466        }
467
468        self.elaborate_traits(items.traits);
469
470        for impls in items.impls.into_values() {
471            self.elaborate_impls(impls);
472        }
473
474        for trait_impl in items.trait_impls {
475            self.elaborate_trait_impl(trait_impl);
476        }
477
478        self.push_errors(self.interner.check_for_dependency_cycles());
479    }

490    pub(crate) fn push_err(&mut self, error: impl Into<CompilationError>) {
491        let error: CompilationError = error.into();
492        // Filter out internal control flow errors that should not be displayed
493        if !error.should_be_filtered() {
494            self.errors.push(error);
495        }
496    }

499        &mut self,
500        errors: impl IntoIterator<Item = E>,

502        for error in errors {

504        }

507    /// Run a given function while also tracking whether any new errors were generated as a result.
508    pub(crate) fn with_error_guard<T>(&mut self, f: impl FnOnce(&mut Self) -> T) -> (T, bool) {
509        // Count actual errors (ignore warnings)
510        let initial_error_count = self.errors.len();
511        let result = f(self);
512        let has_new_errors = self.errors[initial_error_count..].iter().any(|e| e.is_error());
513        (result, has_new_errors)
514    }

534            Ok(PathResolution { item: PathResolutionItem::Trait(trait_id), errors }) => {

549    fn mark_type_as_used_helper(
550        &mut self,
551        typ: &Type,
552        mut type_recursion_context: TypeRecursionContext,
553    ) {
554        match typ {
555            Type::Array(_n, typ) => {
556                self.mark_type_as_used_helper(typ, type_recursion_context.recur());
557            }
558            Type::Vector(typ) => self.mark_type_as_used_helper(typ, type_recursion_context.recur()),
559            Type::Tuple(types) => {
560                for typ in types {
561                    self.mark_type_as_used_helper(typ, type_recursion_context.clone().recur());
562                }
563            }
564            Type::DataType(datatype, generics) => {
565                if type_recursion_context.insert_data_type(datatype.borrow().id, generics.clone()) {
566                    self.mark_struct_as_constructed(datatype.clone());
567                    for generic in generics {
568                        self.mark_type_as_used_helper(
569                            generic,
570                            type_recursion_context.clone().recur(),
571                        );
572                    }
573                    if let Some(fields) = datatype.borrow().get_fields(generics) {
574                        for (_, typ, _) in fields {
575                            self.mark_type_as_used_helper(
576                                &typ,
577                                type_recursion_context.clone().recur(),
578                            );
579                        }
580                    } else if let Some(variants) = datatype.borrow().get_variants(generics) {
581                        for (_, variant_types) in variants {
582                            for typ in variant_types {
583                                self.mark_type_as_used_helper(
584                                    &typ,
585                                    type_recursion_context.clone().recur(),
586                                );
587                            }
588                        }
589                    }
590                }
591            }
592            Type::Alias(alias_type, generics) => {
593                if type_recursion_context.insert_alias(alias_type.borrow().id, generics.clone()) {
594                    self.mark_type_as_used_helper(
595                        &alias_type.borrow().get_type(generics),
596                        type_recursion_context.recur(),
597                    );
598                }
599            }
600            Type::CheckedCast { from, to } => {
601                self.mark_type_as_used_helper(from, type_recursion_context.clone().recur());
602                self.mark_type_as_used_helper(to, type_recursion_context.recur());
603            }
604            Type::Reference(typ, _) => {
605                self.mark_type_as_used_helper(typ, type_recursion_context.recur());
606            }
607            Type::InfixExpr(left, _op, right, _) => {
608                self.mark_type_as_used_helper(left, type_recursion_context.clone().recur());
609                self.mark_type_as_used_helper(right, type_recursion_context.recur());
610            }
611            Type::FieldElement
612            | Type::Integer(..)
613            | Type::Bool
614            | Type::String(_)
615            | Type::FmtString(_, _)
616            | Type::Unit
617            | Type::Quoted(..)
618            | Type::Constant(..)
619            | Type::TraitAsType(..)
620            | Type::TypeVariable(..)
621            | Type::NamedGeneric(..)
622            | Type::Function(..)
623            | Type::Forall(..)
624            | Type::Error => (),
625        }
626    }

643    fn elaborate_traits(&mut self, traits: BTreeMap<TraitId, UnresolvedTrait>) {
644        for (trait_id, unresolved_trait) in traits {
645            self.current_trait = Some(trait_id);
646            self.elaborate_functions(unresolved_trait.fns_with_default_impl);
647        }
648        self.current_trait = None;
649    }

651    fn elaborate_impls(&mut self, impls: Vec<(UnresolvedGenerics, Location, UnresolvedFunctions)>) {
652        for (_, _, functions) in impls {
653            self.recover_generics(|this| this.elaborate_functions(functions));
654        }
655    }

657    fn elaborate_trait_impl(&mut self, trait_impl: UnresolvedTraitImpl) {
658        self.local_module = Some(trait_impl.module_id);
659
660        self.generics = trait_impl.resolved_generics.clone();
661        self.current_trait_impl = trait_impl.impl_id;
662
663        self.add_trait_impl_assumed_trait_implementations(trait_impl.impl_id);
664        self.check_trait_impl_where_clause_matches_trait_where_clause(&trait_impl);
665        self.check_parent_traits_are_implemented(&trait_impl);
666        self.remove_trait_impl_assumed_trait_implementations(trait_impl.impl_id);
667
668        for (module, function, noir_function) in &trait_impl.methods.functions {
669            self.local_module = Some(*module);
670            let errors = check_trait_impl_method_matches_declaration(
671                self.interner,
672                *function,
673                noir_function,
674            );
675            self.push_errors(errors);
676        }
677
678        self.elaborate_functions(trait_impl.methods);
679
680        self.self_type = None;
681        self.current_trait_impl = None;
682        self.generics.clear();
683    }

695    fn define_type_alias(&mut self, alias_id: TypeAliasId, alias: UnresolvedTypeAlias) {
696        self.local_module = Some(alias.module_id);
697
698        let name = &alias.type_alias_def.name;
699        let visibility = alias.type_alias_def.visibility;
700        let location = alias.type_alias_def.location;
701
702        let generics = self.add_generics(&alias.type_alias_def.generics);
703        self.current_item = Some(DependencyId::Alias(alias_id));
704        let wildcard_allowed = types::WildcardAllowed::No(WildcardDisallowedContext::TypeAlias);
705        let (typ, num_expr) = if let Some(num_type) = alias.type_alias_def.numeric_type {
706            let num_type = self.resolve_type(num_type, wildcard_allowed);
707            let kind = Kind::numeric(num_type);
708            let num_expr = alias.type_alias_def.typ.typ.try_into_expression();
709
710            if let Some(num_expr) = num_expr {
711                // Checks that the expression only references generics and constants
712                if !num_expr.is_valid_expression() {
713                    self.errors.push(CompilationError::ResolverError(
714                        ResolverError::RecursiveTypeAlias {
715                            location: alias.type_alias_def.numeric_location,
716                        },
717                    ));
718                    (Type::Error, None)
719                } else {
720                    (
721                        self.resolve_type_with_kind(
722                            alias.type_alias_def.typ,
723                            &kind,
724                            wildcard_allowed,
725                        ),
726                        Some(num_expr),
727                    )
728                }
729            } else {
730                self.errors.push(CompilationError::ResolverError(
731                    ResolverError::ExpectedNumericExpression {
732                        typ: alias.type_alias_def.typ.typ.to_string(),
733                        location,
734                    },
735                ));
736                (Type::Error, None)
737            }
738        } else {
739            (self.use_type(alias.type_alias_def.typ, wildcard_allowed), None)
740        };
741
742        if !visibility.is_private() {
743            self.check_type_is_not_more_private_then_item(name, visibility, &typ, location);
744        }
745        self.interner.set_type_alias(alias_id, typ, generics, num_expr);
746        self.generics.clear();
747    }

749    /// True if we're currently within a constrained function or lambda.
750    /// Defaults to `true` if the current function is unknown.
751    fn in_constrained_function(&self) -> bool {
752        if self.in_comptime_context() {
753            return false;
754        }
755
756        let in_unconstrained_function = self.current_item.is_some_and(|id| {
757            if let DependencyId::Function(id) = id {
758                self.interner.function_meta(&id).is_unconstrained()
759            } else {
760                false
761            }
762        });
763
764        let in_unconstrained_lambda = self.lambda_stack.last().is_some_and(|ctx| ctx.unconstrained);
765
766        !in_unconstrained_function && !in_unconstrained_lambda
767    }

769    /// Register a use of the given unstable feature. Errors if the feature has not
770    /// been explicitly enabled in this package.
771    pub fn use_unstable_feature(&mut self, feature: UnstableFeature, location: Location) {
772        // Is the feature globally enabled via CLI options?
773        if self.options.enabled_unstable_features.contains(&feature) {
774            return;
775        }
776
777        // Can crates require unstable features in their manifest?
778        let enable_required_unstable_features = self.options.enabled_unstable_features.is_empty()
779            && !self.options.disable_required_unstable_features;
780
781        // Is it required by the current crate?
782        if enable_required_unstable_features
783            && self
784                .required_unstable_features
785                .get(&self.crate_id)
786                .is_some_and(|fs| fs.contains(&feature))
787        {
788            return;
789        }
790
791        let reason = ParserErrorReason::ExperimentalFeature(feature);
792        self.push_err(ParserError::with_reason(reason, location));
793    }

795    /// Run the given function using the resolver and return true if any errors (not warnings)
796    /// occurred while running it.
797    pub fn errors_occurred_in<T>(&mut self, f: impl FnOnce(&mut Self) -> T) -> (bool, T) {
798        let previous_errors = self.errors.len();
799        let ret = f(self);
800        let errored = self.errors.iter().skip(previous_errors).any(|error| error.is_error());
801        (errored, ret)
802    }

804    /// Push a new location to the interpreter call stack.
805    ///
806    /// Return [InterpreterError::StackOverflow] if the stack size exceeds `MAX_INTERPRETER_CALL_STACK_SIZE`.
807    pub(crate) fn push_interpreter_call_stack(
808        &mut self,
809        location: Location,
810    ) -> Result<(), InterpreterError> {
811        if self.interpreter_call_stack.len() >= MAX_INTERPRETER_CALL_STACK_SIZE {
812            return Err(InterpreterError::StackOverflow {
813                location,
814                call_stack: self.interpreter_call_stack.clone(),
815            });
816        }
817        self.interpreter_call_stack.push_back(location);
818        Ok(())
819    }

821    /// Pops the last item from the interpreter call stack.
822    ///
823    /// Panics if the call stack is empty.
824    pub(crate) fn pop_interpreter_call_stack(&mut self) {
825        self.interpreter_call_stack
826            .pop_back()
827            .expect("call stack pushes and pops should be balanced");
828    }

836#[cfg(feature = "test_utils")]
837pub mod test_utils {
838    use std::cell::RefCell;
839    use std::io::Write;
840    use std::rc::Rc;
841
842    use crate::hir::comptime::InterpreterError;
843    use crate::{hir::def_collector::dc_crate::CompilationError, parser::ParserError};
844
845    /// The possible errors of interpreting given code
846    /// into a monomorphized AST expression.
847    #[derive(Debug, Clone, PartialEq, Eq)]
848    pub enum ElaboratorError {
849        Parse(Vec<ParserError>),
850        Compile(Vec<CompilationError>),
851        Interpret(InterpreterError),
852        HIRConvert(InterpreterError),
853    }
854
855    /// Interpret source code using the elaborator, without
856    /// parsing and compiling it with nargo, converting
857    /// the result into a monomorphized AST expression.
858    ///
859    /// The source is treated as root and stdlib, so stdlib snippets are allowed.
860    pub fn interpret<W: Write + 'static>(
861        src: &str,
862        output: Rc<RefCell<W>>,
863    ) -> Result<crate::monomorphization::ast::Expression, ElaboratorError> {
864        use crate::elaborator::ElaboratorOptions;
865        use crate::monomorphization::{Monomorphizer, debug_types::DebugTypeTracker};
866        use crate::parse_program;
867        use crate::{
868            elaborator::Elaborator,
869            hir::{
870                Context, ParsedFiles,
871                def_collector::{dc_crate::DefCollector, dc_mod::collect_defs},
872                def_map::{CrateDefMap, ModuleData},
873            },
874        };
875        use fm::{FileId, FileManager};
876        use noirc_errors::Location;
877        use std::path::PathBuf;
878
879        let file = FileId::default();
880
881        let location = Location::new(Default::default(), file);
882        let root_module = ModuleData::new(
883            None,
884            None,
885            location,
886            Vec::new(),
887            Vec::new(),
888            false, // is contract
889            false, // is struct
890        );
891
892        let file_manager = FileManager::new(&PathBuf::new());
893        let parsed_files = ParsedFiles::new();
894        let mut context = Context::new(file_manager, parsed_files);
895        context.def_interner.populate_dummy_operator_traits();
896        context.set_comptime_printing(output);
897
898        let krate = context.crate_graph.add_crate_root_and_stdlib(FileId::dummy());
899
900        let (module, errors) = parse_program(src, file);
901        // Skip parser warnings
902        let errors: Vec<_> = errors.iter().filter(|e| !e.is_warning()).cloned().collect();
903        if !errors.is_empty() {
904            return Err(ElaboratorError::Parse(errors));
905        }
906
907        let ast = module.into_sorted();
908
909        let def_map = CrateDefMap::new(krate, root_module);
910        let root_module_id = def_map.root();
911        let mut collector = DefCollector::new(def_map);
912        let reuse_existing_module_declarations = false;
913
914        collect_defs(
915            &mut collector,
916            ast,
917            FileId::dummy(),
918            root_module_id,
919            krate,
920            &mut context,
921            reuse_existing_module_declarations,
922        );
923        context.def_maps.insert(krate, collector.def_map);
924
925        let main = context.get_main_function(&krate).expect("Expected 'main' function");
926
927        let mut elaborator = Elaborator::elaborate_and_return_self(
928            &mut context,
929            krate,
930            collector.items,
931            ElaboratorOptions::test_default(),
932        );
933
934        // Skip the elaborator's compilation warnings
935        let errors: Vec<_> = elaborator.errors.iter().filter(|&e| e.is_error()).cloned().collect();
936        if !errors.is_empty() {
937            return Err(ElaboratorError::Compile(errors));
938        }
939
940        let mut interpreter = elaborator.setup_interpreter();
941
942        // The most straightforward way to convert the interpreter result into
943        // an acceptable monomorphized AST expression seems to be converting it
944        // into HIR first and then processing it with the monomorphizer
945        let expr_id = match interpreter.call_function(
946            main,
947            Vec::new(),
948            Default::default(),
949            Location::dummy(),
950        ) {
951            Err(e) => return Err(ElaboratorError::Interpret(e)),
952            Ok(value) => {
953                match value.into_runtime_hir_expression(elaborator.interner, Location::dummy()) {
954                    Err(e) => return Err(ElaboratorError::HIRConvert(e)),
955                    Ok(expr_id) => expr_id,
956                }
957            }
958        };
959
960        let mut monomorphizer =
961            Monomorphizer::new(elaborator.interner, DebugTypeTracker::default(), false);
962        Ok(monomorphizer.expr(expr_id).expect("monomorphization error while converting interpreter execution result, should not be possible"))
963    }
