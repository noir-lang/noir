1//! # The Elaborator
2//!
3//! The elaborator is the core semantic analysis phase of the compiler. It transforms collected,
4//! unresolved AST items into fully resolved and type-checked HIR through simultaneous and intertwined
5//! name resolution, type checking, trait resolution, and macro expansion. It also handles pattern
6//! elaboration, generic type processing, comptime interpretation, reference validation, and scope management.
7//!
8//! ## Architecture Overview
9//!
10//! The elaborator operates in several distinct phases, processing items in a specific order to handle
11//! dependencies correctly:
12//!
13//! ### Early Resolution
14//! 1. Globals - Set up their dependency ordering. Deferred for elaboration later after type resolution.
15//!    Globals will be lazily elaborated when other types or expressions bring them into scope.
16//! 2. Type aliases - Defined to allow their use in subsequent type definitions
17//!
18//! ### Type Collection
19//! 1. Struct definitions - Collected so their types are available for use
20//! 2. Enum definitions - Collected so their types are available for use
21//! 3. Trait definitions - Collected so trait bounds can be resolved
22//!
23//! ### Function metadata and Implementations
24//! 1. Function metadata - Signatures collected before bodies are elaborated
25//! 2. Trait methods - Method signatures collected from trait definitions
26//! 3. Impl blocks - Methods organized into their proper modules based on the impl's type
27//! 4. Trait impls - Linked to their corresponding traits and validated
28//!
29//! ### Global Elaboration
30//! - Elaborate any remaining globals which were not brought into scope.
31//!   Elaborated after type resolution since they may use struct types which need global type information
32//!
33//! ### Attribute Processing
34//! - Comptime attributes - Executed before function body elaboration, as generated items may change what is in scope or modify functions
35//!
36//! ### Full Item Elaboration
37//! 1. Functions - Function bodies elaborated (resolved & type-checked)
38//! 2. Traits - Trait default method implementations elaborated
39//! 3. Impls - Implementation method bodies elaborated
40//! 4. Trait impls - Trait implementation method bodies elaborated and validated against trait signatures
41//!
42//! ### Dependency Analysis
43//! Detect and report dependency cycles to prevent infinite elaboration loops
44//!
45//! ## Error Handling
46//!
47//! The elaborator accumulates errors rather than failing fast, allowing it to report multiple
48//! errors in a single compilation pass. Errors are collected throughout elaboration and may be
49//! wrapped with additional context when elaborating generated code (e.g., from attributes or
50//! comptime calls).

126/// Maximum number of recursive calls allowed at comptime.
127///
128/// Ideally we would like this to be 1000 to match what happens in ACIR,
129/// however due to the overhead of the `Interpreter` itself Rust itself
130/// would exhaust the stack earlier (or later, because `nargo` increases
131/// the stack size for parsing for example).
132///
133/// Potentially we could increase this if the `Interpreter` used an iterative
134/// strategy instead of recursion.
135///
136/// Note that if we increase this, currently we would hit the `MAX_EVALUATION_DEPTH`.
137const MAX_INTERPRETER_CALL_STACK_SIZE: usize = 100;
138
139/// Maximum depth of macro expansion (attribute execution).
140///
141/// This prevents infinite recursion when an attribute generates code that
142/// triggers further attribute expansion, which would otherwise cause a
143/// Rust stack overflow.
144///
145/// This limit is lower than the [interpreter call stack limit][MAX_INTERPRETER_CALL_STACK_SIZE] because macro
146/// expansion involves more stack frames per level (elaboration, comptime evaluation, etc.).
147pub(crate) const MAX_MACRO_EXPANSION_DEPTH: usize = 32;

295#[derive(Copy, Clone)]
296pub enum ElaborateReason {
297    /// A function attribute generated an item that's being elaborated.
298    RunningAttribute(Location),
299    /// Evaluating a comptime call like `Module::add_item`
300    EvaluatingComptimeCall(&'static str, Location),
301}
302
303impl ElaborateReason {
304    fn to_macro_error(self, error: CompilationError) -> ComptimeError {
305        match self {
306            ElaborateReason::RunningAttribute(location) => {
307                ComptimeError::ErrorRunningAttribute { error: Box::new(error), location }
308            }
309            ElaborateReason::EvaluatingComptimeCall(method_name, location) => {
310                let error = Box::new(error);
311                ComptimeError::ErrorEvaluatingComptimeCall { method_name, error, location }
312            }
313        }
314    }
315}

420    pub(crate) fn elaborate_items(&mut self, mut items: CollectedItems) {
421        self.set_unresolved_globals_ordering(items.globals);
422
423        for (alias_id, alias) in items.type_aliases {
424            self.define_type_alias(alias_id, alias);
425        }
426
427        // Must resolve types before we resolve globals.
428        self.collect_struct_definitions(&items.structs);
429        self.collect_enum_definitions(&items.enums);
430        self.collect_traits(&mut items.traits);
431
432        self.define_function_metas(&mut items.functions, &mut items.impls, &mut items.trait_impls);
433
434        // Before we resolve any function symbols we must go through our impls and
435        // re-collect the methods within into their proper module. This cannot be
436        // done during def collection since we need to be able to resolve the type of
437        // the impl since that determines the module we should collect into.
438        for ((self_type, module), impls) in &mut items.impls {
439            self.collect_impls(*module, impls, self_type);
440        }
441
442        self.collect_trait_methods(&mut items.traits);
443
444        // Bind trait impls to their trait. Collect trait functions, that have a
445        // default implementation, which hasn't been overridden.
446        for trait_impl in &mut items.trait_impls {
447            self.collect_trait_impl(trait_impl);
448        }
449
450        // We must wait to resolve non-literal globals until after we resolve structs since struct
451        // globals will need to reference the struct type they're initialized to ensure they are valid.
452        self.elaborate_remaining_globals();
453
454        // We have to run any comptime attributes on functions before the function is elaborated
455        // since the generated items are checked beforehand as well.
456        self.run_attributes(
457            &items.traits,
458            &items.structs,
459            &items.functions,
460            &items.module_attributes,
461        );
462
463        for functions in items.functions {
464            self.elaborate_functions(functions);
465        }
466
467        self.elaborate_traits(items.traits);
468
469        for impls in items.impls.into_values() {
470            self.elaborate_impls(impls);
471        }
472
473        for trait_impl in items.trait_impls {
474            self.elaborate_trait_impl(trait_impl);
475        }
476
477        self.push_errors(self.interner.check_for_dependency_cycles());
478    }

489    pub(crate) fn push_err(&mut self, error: impl Into<CompilationError>) {
490        let error: CompilationError = error.into();
491        // Filter out internal control flow errors that should not be displayed
492        if !error.should_be_filtered() {
493            self.errors.push(error);
494        }
495    }

497    pub(crate) fn push_errors<E: Into<CompilationError>>(
498        &mut self,
499        errors: impl IntoIterator<Item = E>,
500    ) {
501        for error in errors {
502            self.push_err(error);
503        }
504    }

506    /// Run a given function while also tracking whether any new errors were generated as a result.
507    pub(crate) fn with_error_guard<T>(&mut self, f: impl FnOnce(&mut Self) -> T) -> (T, bool) {
508        // Count actual errors (ignore warnings)
509        let initial_error_count = self.errors.len();
510        let result = f(self);
511        let has_new_errors = self.errors[initial_error_count..].iter().any(|e| e.is_error());
512        (result, has_new_errors)
513    }

531    fn resolve_trait_by_path(&mut self, path: TypedPath) -> Option<TraitId> {
532        let error = match self.resolve_path_as_type(path.clone()) {
533            Ok(PathResolution { item: PathResolutionItem::Trait(trait_id), errors }) => {
534                self.push_errors(errors);
535                return Some(trait_id);
536            }
537            Ok(_) => DefCollectorErrorKind::NotATrait { not_a_trait_name: path },
538            Err(_) => DefCollectorErrorKind::TraitNotFound { trait_path: path },
539        };
540        self.push_err(error);
541        None

549    fn mark_type_as_used_helper(&mut self, typ: &Type, seen_data_types: &mut SeenDataTypes) {
550        match typ {
551            Type::Array(_n, typ) => self.mark_type_as_used_helper(typ, seen_data_types),
552            Type::Vector(typ) => self.mark_type_as_used_helper(typ, seen_data_types),
553            Type::Tuple(types) => {
554                for typ in types {
555                    self.mark_type_as_used_helper(typ, seen_data_types);
556                }
557            }
558            Type::DataType(datatype, generics) => {
559                let key = (datatype.borrow().id, generics.clone());
560                if seen_data_types.insert(key) {
561                    self.mark_struct_as_constructed(datatype.clone());
562                    for generic in generics {
563                        self.mark_type_as_used_helper(generic, seen_data_types);
564                    }
565                    if let Some(fields) = datatype.borrow().get_fields(generics) {
566                        for (_, typ, _) in fields {
567                            self.mark_type_as_used_helper(&typ, seen_data_types);
568                        }
569                    } else if let Some(variants) = datatype.borrow().get_variants(generics) {
570                        for (_, variant_types) in variants {
571                            for typ in variant_types {
572                                self.mark_type_as_used_helper(&typ, seen_data_types);
573                            }
574                        }
575                    }
576                }
577            }
578            Type::Alias(alias_type, generics) => {
579                self.mark_type_as_used_helper(
580                    &alias_type.borrow().get_type(generics),
581                    seen_data_types,
582                );
583            }
584            Type::CheckedCast { from, to } => {
585                self.mark_type_as_used_helper(from, seen_data_types);
586                self.mark_type_as_used_helper(to, seen_data_types);
587            }
588            Type::Reference(typ, _) => {
589                self.mark_type_as_used_helper(typ, seen_data_types);
590            }
591            Type::InfixExpr(left, _op, right, _) => {
592                self.mark_type_as_used_helper(left, seen_data_types);
593                self.mark_type_as_used_helper(right, seen_data_types);
594            }
595            Type::FieldElement
596            | Type::Integer(..)
597            | Type::Bool
598            | Type::String(_)
599            | Type::FmtString(_, _)
600            | Type::Unit
601            | Type::Quoted(..)
602            | Type::Constant(..)
603            | Type::TraitAsType(..)
604            | Type::TypeVariable(..)
605            | Type::NamedGeneric(..)
606            | Type::Function(..)
607            | Type::Forall(..)
608            | Type::Error => (),
609        }
610    }

613    ///
614    /// This is usually determined by `self.module_id()`, but it can
615    /// be overridden for impls. Impls are an odd case since the methods within resolve
616    /// as if they're in the parent module, but should be placed in a child module.

641    fn elaborate_trait_impl(&mut self, trait_impl: UnresolvedTraitImpl) {
642        self.local_module = Some(trait_impl.module_id);
643
644        self.generics = trait_impl.resolved_generics.clone();
645        self.current_trait_impl = trait_impl.impl_id;
646
647        self.add_trait_impl_assumed_trait_implementations(trait_impl.impl_id);
648        self.check_trait_impl_where_clause_matches_trait_where_clause(&trait_impl);
649        self.check_parent_traits_are_implemented(&trait_impl);
650        self.remove_trait_impl_assumed_trait_implementations(trait_impl.impl_id);
651
652        for (module, function, noir_function) in &trait_impl.methods.functions {
653            self.local_module = Some(*module);
654            let errors = check_trait_impl_method_matches_declaration(
655                self.interner,
656                *function,
657                noir_function,
658            );
659            self.push_errors(errors);
660        }
661
662        self.elaborate_functions(trait_impl.methods);
663
664        self.self_type = None;
665        self.current_trait_impl = None;
666        self.generics.clear();
667    }

679    fn define_type_alias(&mut self, alias_id: TypeAliasId, alias: UnresolvedTypeAlias) {
680        self.local_module = Some(alias.module_id);
681
682        let name = &alias.type_alias_def.name;
683        let visibility = alias.type_alias_def.visibility;
684        let location = alias.type_alias_def.location;
685
686        let generics = self.add_generics(&alias.type_alias_def.generics);
687        self.current_item = Some(DependencyId::Alias(alias_id));
688        let wildcard_allowed = types::WildcardAllowed::No(WildcardDisallowedContext::TypeAlias);
689        let (typ, num_expr) = if let Some(num_type) = alias.type_alias_def.numeric_type {
690            let num_type = self.resolve_type(num_type, wildcard_allowed);
691            let kind = Kind::numeric(num_type);
692            let num_expr = alias.type_alias_def.typ.typ.try_into_expression();
693
694            if let Some(num_expr) = num_expr {
695                // Checks that the expression only references generics and constants
696                if !num_expr.is_valid_expression() {
697                    self.errors.push(CompilationError::ResolverError(
698                        ResolverError::RecursiveTypeAlias {
699                            location: alias.type_alias_def.numeric_location,
700                        },
701                    ));
702                    (Type::Error, None)
703                } else {
704                    (
705                        self.resolve_type_with_kind(
706                            alias.type_alias_def.typ,
707                            &kind,
708                            wildcard_allowed,
709                        ),
710                        Some(num_expr),
711                    )
712                }
713            } else {
714                self.errors.push(CompilationError::ResolverError(
715                    ResolverError::ExpectedNumericExpression {
716                        typ: alias.type_alias_def.typ.typ.to_string(),
717                        location,
718                    },
719                ));
720                (Type::Error, None)
721            }
722        } else {
723            (self.use_type(alias.type_alias_def.typ, wildcard_allowed), None)
724        };
725
726        if !visibility.is_private() {
727            self.check_type_is_not_more_private_then_item(name, visibility, &typ, location);
728        }
729        self.interner.set_type_alias(alias_id, typ, generics, num_expr);
730        self.generics.clear();
731    }

733    /// True if we're currently within a constrained function or lambda.
734    /// Defaults to `true` if the current function is unknown.
735    fn in_constrained_function(&self) -> bool {
736        if self.in_comptime_context() {
737            return false;
738        }
739
740        let in_unconstrained_function = self.current_item.is_some_and(|id| {
741            if let DependencyId::Function(id) = id {
742                self.interner.function_modifiers(&id).is_unconstrained
743            } else {
744                false
745            }
746        });
747
748        let in_unconstrained_lambda = self.lambda_stack.last().is_some_and(|ctx| ctx.unconstrained);
749
750        !in_unconstrained_function && !in_unconstrained_lambda
751    }

753    /// Register a use of the given unstable feature. Errors if the feature has not
754    /// been explicitly enabled in this package.
755    pub fn use_unstable_feature(&mut self, feature: UnstableFeature, location: Location) {
756        // Is the feature globally enabled via CLI options?
757        if self.options.enabled_unstable_features.contains(&feature) {
758            return;
759        }
760
761        // Can crates require unstable features in their manifest?
762        let enable_required_unstable_features = self.options.enabled_unstable_features.is_empty()
763            && !self.options.disable_required_unstable_features;
764
765        // Is it required by the current crate?
766        if enable_required_unstable_features
767            && self
768                .required_unstable_features
769                .get(&self.crate_id)
770                .is_some_and(|fs| fs.contains(&feature))
771        {
772            return;
773        }
774
775        let reason = ParserErrorReason::ExperimentalFeature(feature);
776        self.push_err(ParserError::with_reason(reason, location));
777    }

779    /// Run the given function using the resolver and return true if any errors (not warnings)
780    /// occurred while running it.
781    pub fn errors_occurred_in<T>(&mut self, f: impl FnOnce(&mut Self) -> T) -> (bool, T) {
782        let previous_errors = self.errors.len();
783        let ret = f(self);
784        let errored = self.errors.iter().skip(previous_errors).any(|error| error.is_error());
785        (errored, ret)
786    }

788    /// Push a new location to the interpreter call stack.
789    ///
790    /// Return [InterpreterError::StackOverflow] if the stack size exceeds `MAX_INTERPRETER_CALL_STACK_SIZE`.
791    pub(crate) fn push_interpreter_call_stack(
792        &mut self,
793        location: Location,
794    ) -> Result<(), InterpreterError> {
795        if self.interpreter_call_stack.len() >= MAX_INTERPRETER_CALL_STACK_SIZE {
796            return Err(InterpreterError::StackOverflow {
797                location,
798                call_stack: self.interpreter_call_stack.clone(),
799            });
800        }
801        self.interpreter_call_stack.push_back(location);
802        Ok(())
803    }

805    /// Pops the last item from the interpreter call stack.
806    ///
807    /// Panics if the call stack is empty.
808    pub(crate) fn pop_interpreter_call_stack(&mut self) {
809        self.interpreter_call_stack
810            .pop_back()
811            .expect("call stack pushes and pops should be balanced");
812    }

822    use std::cell::RefCell;
823    use std::io::Write;

829    /// The possible errors of interpreting given code
830    /// into a monomorphized AST expression.
831    #[derive(Debug, Clone, PartialEq, Eq)]
832    pub enum ElaboratorError {
833        Parse(Vec<ParserError>),
834        Compile(Vec<CompilationError>),
835        Interpret(InterpreterError),
836        HIRConvert(InterpreterError),
837    }
838
839    /// Interpret source code using the elaborator, without
840    /// parsing and compiling it with nargo, converting
841    /// the result into a monomorphized AST expression.
842    ///
843    /// The source is treated as root and stdlib, so stdlib snippets are allowed.
844    pub fn interpret<W: Write + 'static>(
845        src: &str,
846        output: Rc<RefCell<W>>,
847    ) -> Result<crate::monomorphization::ast::Expression, ElaboratorError> {
848        use crate::elaborator::ElaboratorOptions;
849        use crate::monomorphization::{Monomorphizer, debug_types::DebugTypeTracker};
850        use crate::parse_program;
851        use crate::{
852            elaborator::Elaborator,
853            hir::{
854                Context, ParsedFiles,
855                def_collector::{dc_crate::DefCollector, dc_mod::collect_defs},
856                def_map::{CrateDefMap, ModuleData},
857            },
858        };
859        use fm::{FileId, FileManager};
860        use noirc_errors::Location;
861        use std::path::PathBuf;
862
863        let file = FileId::default();
864
865        let location = Location::new(Default::default(), file);
866        let root_module = ModuleData::new(
867            None,
868            None,
869            location,
870            Vec::new(),
871            Vec::new(),
872            false, // is contract
873            false, // is struct
874        );
875
876        let file_manager = FileManager::new(&PathBuf::new());
877        let parsed_files = ParsedFiles::new();
878        let mut context = Context::new(file_manager, parsed_files);
879        context.def_interner.populate_dummy_operator_traits();
880        context.set_comptime_printing(output);
881
882        let krate = context.crate_graph.add_crate_root_and_stdlib(FileId::dummy());
883
884        let (module, errors) = parse_program(src, file);
885        // Skip parser warnings
886        let errors: Vec<_> = errors.iter().filter(|e| !e.is_warning()).cloned().collect();
887        if !errors.is_empty() {
888            return Err(ElaboratorError::Parse(errors));
889        }
890
891        let ast = module.into_sorted();
892
893        let def_map = CrateDefMap::new(krate, root_module);
894        let root_module_id = def_map.root();
895        let mut collector = DefCollector::new(def_map);
896        let reuse_existing_module_declarations = false;
897
898        collect_defs(
899            &mut collector,
900            ast,
901            FileId::dummy(),
902            root_module_id,
903            krate,
904            &mut context,
905            reuse_existing_module_declarations,
906        );
907        context.def_maps.insert(krate, collector.def_map);
908
909        let main = context.get_main_function(&krate).expect("Expected 'main' function");
910
911        let mut elaborator = Elaborator::elaborate_and_return_self(
912            &mut context,
913            krate,
914            collector.items,
915            ElaboratorOptions::test_default(),
916        );
917
918        // Skip the elaborator's compilation warnings
919        let errors: Vec<_> = elaborator.errors.iter().filter(|&e| e.is_error()).cloned().collect();
920        if !errors.is_empty() {
921            return Err(ElaboratorError::Compile(errors));
922        }
923
924        let mut interpreter = elaborator.setup_interpreter();
925
926        // The most straightforward way to convert the interpreter result into
927        // an acceptable monomorphized AST expression seems to be converting it
928        // into HIR first and then processing it with the monomorphizer
929        let expr_id = match interpreter.call_function(
930            main,
931            Vec::new(),
932            Default::default(),
933            Location::dummy(),
934        ) {
935            Err(e) => return Err(ElaboratorError::Interpret(e)),
936            Ok(value) => {
937                match value.into_runtime_hir_expression(elaborator.interner, Location::dummy()) {
938                    Err(e) => return Err(ElaboratorError::HIRConvert(e)),
939                    Ok(expr_id) => expr_id,
940                }
941            }
942        };
943
944        let mut monomorphizer =
945            Monomorphizer::new(elaborator.interner, DebugTypeTracker::default(), false);
946        Ok(monomorphizer.expr(expr_id).expect("monomorphization error while converting interpreter execution result, should not be possible"))
947    }
