30/// The result of [`Elaborator::resolve_variable`].
31#[allow(clippy::large_enum_variant)]
32pub(crate) enum VariableResolution {
33    /// The path resolved to a variable or a definition.
34    Ident(HirIdent, Option<PathResolutionItem>),
35    /// The path resolved to a type alias that is numeric, infinitely recursive or one that errored.
36    TypeAlias(TypeAliasId),
37}

39impl Elaborator<'_> {
40    #[tracing::instrument(level = "trace", skip_all)]
41    pub(super) fn elaborate_variable(&mut self, variable: Path) -> (ExprId, Type) {
42        let (id, typ, is_comptime_local, location) = self.elaborate_variable_inner(variable);
43
44        // Only check has_errors when we need to call the interpreter
45        // If this variable is a comptime local variable, use its current value as the final expression
46        if is_comptime_local {
47            let mut interpreter = self.setup_interpreter();
48            let value = interpreter.evaluate(id);
49            // If the value is an error it means the variable already had an error, so don't report it here again
50            // (the error will make no sense, it will say that a non-comptime variable was referenced at runtime
51            // but that's not true)
52            if value.is_ok() {
53                let (id, typ) = self.inline_comptime_value(value, location);
54                self.debug_comptime(location, |interner| id.to_display_ast(interner).kind);
55                (id, typ)
56            } else {
57                (id, typ)
58            }
59        } else {
60            (id, typ)
61        }
62    }

64    /// Helper function containing the elaboration logic for a variable.
65    /// Returns the expression ID, type, whether it's a comptime local, and location.
66    #[tracing::instrument(level = "trace", skip_all)]
67    fn elaborate_variable_inner(&mut self, variable: Path) -> (ExprId, Type, bool, Location) {
68        let variable = self.validate_path(variable);
69        if let Some((expr_id, typ)) =
70            self.elaborate_variable_as_self_method_or_associated_constant(&variable)
71        {
72            return (expr_id, typ, false, variable.location);
73        }
74
75        let resolved_turbofish = variable.segments.last().unwrap().generics.clone();
76
77        let location = variable.location;
78        let variable_resolution = self.resolve_variable(variable);
79
80        let (hir_ident, item) = match variable_resolution {
81            Some(VariableResolution::TypeAlias(type_alias_id)) => {
82                // A type alias to a numeric generics is considered like a variable,
83                // but it is not a real variable so it does not resolve to a valid Identifier.
84                // In order to handle this, we retrieve the numeric generics expression that the type aliases to.
85                let type_alias = self.interner.get_type_alias(type_alias_id);
86                if let Some(expr) = &type_alias.borrow().numeric_expr {
87                    // Extract the declared numeric type from the type alias's kind.
88                    let declared_type = match type_alias.borrow().typ.kind() {
89                        Kind::Numeric(declared_type) => declared_type,
90                        _ => Box::new(Type::Error),
91                    };
92                    let declared_type = *declared_type;
93                    let expr_location = type_alias.borrow().location;
94                    let expr = UnresolvedTypeExpression::to_expression_kind(expr);
95                    let expr = Expression::new(expr, expr_location);
96
97                    // Resolve turbofish generics for the type alias.
98                    // `resolved_turbofish` contains already-resolved types from
99                    // validate_path, so we use resolve_alias_turbofish_generics
100                    // directly which accepts resolved types.
101                    let alias_generics = &type_alias.borrow().generics;
102                    let alias_generic_types = vecmap(alias_generics, |generic| {
103                        self.interner.next_type_variable_with_kind(generic.kind())
104                    });
105                    let mut errors = Vec::new();
106                    let type_alias_ref = self.interner.get_type_alias(type_alias_id);
107                    let type_alias_ref = type_alias_ref.borrow();
108                    let resolved_generics = self.resolve_alias_turbofish_generics(
109                        &type_alias_ref,
110                        alias_generic_types,
111                        resolved_turbofish,
112                        location,
113                        &mut errors,
114                    );
115                    self.push_errors(errors);
116
117                    // Introduce alias generics into scope so the numeric expression
118                    // resolves them correctly (not to globals or other variables
119                    // that happen to share the same name). Bind each generic's type
120                    // variable to the turbofish-resolved type.
121                    self.push_scope();
122                    for (generic, resolved_type) in
123                        alias_generics.iter().zip_eq(resolved_generics.iter())
124                    {
125                        if let Kind::Numeric(numeric_type) = &generic.kind() {
126                            let id = self.interner.next_type_variable_id();
127                            let type_var = TypeVariable::unbound(id, generic.kind());
128                            type_var.bind(resolved_type.clone());
129                            let definition =
130                                DefinitionKind::NumericGeneric(type_var, numeric_type.clone());
131                            let ident = Ident::new(generic.name.to_string(), generic.location);
132                            let hir_ident = self.add_variable_decl(
133                                ident, false, // mutable
134                                true,  // allow_shadowing
135                                false, // warn_if_unused
136                                false, // warn_if_not_mutated
137                                definition,
138                            );
139                            self.interner.push_definition_type(hir_ident.id, *numeric_type.clone());
140                        }
141                    }
142
143                    let (id, typ) = self.elaborate_expression(expr);
144                    self.pop_scope();
145
146                    // Unify the expression's type with the declared type from the type alias
147                    // to ensure proper type checking.
148                    self.unify(&typ, &declared_type, || TypeCheckError::TypeMismatch {
149                        expected_typ: declared_type.to_string(),
150                        expr_typ: typ.to_string(),
151                        expr_location,
152                    });
153                    return (id, declared_type, false, location);
154                }
155                (None, None)
156            }
157            Some(VariableResolution::Ident(ident, item)) => (Some(ident), item),
158            None => (None, None),
159        };
160
161        let definition_id = hir_ident.as_ref().map(|ident| ident.id);
162
163        let (type_generics, self_generic) = if let Some(item) = item {
164            self.resolve_item_turbofish_and_self_type(item)
165        } else {
166            (Vec::new(), None)
167        };
168
169        let definition = definition_id.map(|id| self.interner.definition(id));
170        let is_comptime_local = !self.in_comptime_context()
171            && definition.is_some_and(DefinitionInfo::is_comptime_local);
172        let definition_kind = definition.as_ref().map(|definition| definition.kind.clone());
173
174        let mut bindings = TypeBindings::default();
175        let generics = if let Some(DefinitionKind::Function(func_id)) = &definition_kind {
176            // If there's a self type, bind it to the self type generic
177            if let Some(self_generic) = self_generic {
178                let func_generics = &self.interner.function_meta(func_id).all_generics;
179                let self_resolved_generic =
180                    func_generics.iter().find(|generic| generic.name.as_str() == SELF_TYPE_NAME);
181                if let Some(self_resolved_generic) = self_resolved_generic {
182                    let type_var = &self_resolved_generic.type_var;
183                    bindings
184                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), self_generic));
185                }
186            }
187
188            // If this is a function call on a type that has generics, we need to bind those generic types.
189            if !type_generics.is_empty() {
190                // `all_generics` has the enclosing type generics first, followed by `direct_generics`
191                // (the method's own generics). We must only bind the type-level portion here;
192                // method generics are handled separately by the method turbofish.
193                let func_meta = self.interner.function_meta(func_id);
194                let impl_generic_count =
195                    func_meta.all_generics.len() - func_meta.direct_generics.len();
196                let impl_generics = &func_meta.all_generics[..impl_generic_count];
197
198                // For partially concrete impls (e.g. `impl<B> S<u32, B>`), the number of
199                // impl generics differs from the number of struct generics. The turbofish
200                // `S::<u32, bool>` provides type_generics aligned with the struct's params
201                // [A, B], not the impl's generics [B]. Use the impl's self_type to find
202                // the correct positional mapping from struct params to impl generics.
203                if let Some(Type::DataType(_, self_type_args)) = &func_meta.self_type {
204                    assert_eq!(
205                        type_generics.len(),
206                        self_type_args.len(),
207                        "ICE: turbofish type_generics count ({}) should match self_type_args count ({})",
208                        type_generics.len(),
209                        self_type_args.len(),
210                    );
211                    let mut concrete_mismatches = Vec::new();
212                    for (type_generic, self_type_arg) in
213                        type_generics.into_iter().zip_eq(self_type_args)
214                    {
215                        let type_var = match self_type_arg {
216                            Type::NamedGeneric(named) => Some(&named.type_var),
217                            Type::TypeVariable(tv) => Some(tv),
218                            _ => None,
219                        };
220                        if let Some(type_var) = type_var {
221                            if impl_generics.iter().any(|g| g.type_var.id() == type_var.id()) {
222                                bindings.insert(
223                                    type_var.id(),
224                                    (type_var.clone(), type_var.kind(), type_generic),
225                                );
226                            }
227                        } else {
228                            // Concrete position: collect for verification after releasing borrow
229                            concrete_mismatches.push((type_generic, self_type_arg.clone()));
230                        }
231                    }
232                    // Verify turbofish types match the impl's concrete types
233                    for (turbofish_type, concrete_type) in concrete_mismatches {
234                        self.unify(&turbofish_type, &concrete_type, || {
235                            TypeCheckError::TypeMismatch {
236                                expected_typ: concrete_type.to_string(),
237                                expr_typ: turbofish_type.to_string(),
238                                expr_location: location,
239                            }
240                        });
241                    }
242                } else if type_generics.len() <= impl_generics.len() {
243                    // For trait function paths, impl_generics may include Self and associated
244                    // type generics after the trait's declared generics. The turbofish only
245                    // provides types for the declared generics, which are always the first
246                    // elements. Slice to match.
247                    let impl_generics = &impl_generics[..type_generics.len()];
248                    for (type_generic, func_generic) in
249                        type_generics.into_iter().zip_eq(impl_generics)
250                    {
251                        let type_var = &func_generic.type_var;
252                        bindings.insert(
253                            type_var.id(),
254                            (type_var.clone(), type_var.kind(), type_generic),
255                        );
256                    }
257                } else {
258                    unreachable!(
259                        "type_generics.len() ({}) > impl_generics.len() ({}): \
260                        turbofish resolution should normalize the count",
261                        type_generics.len(),
262                        impl_generics.len()
263                    );
264                }
265            }
266
267            // Resolve any generics if the variable we have resolved is a function
268            // and if the turbofish operator was used.
269            self.resolve_function_turbofish_generics(func_id, resolved_turbofish, location)
270        } else {
271            if let Some(unused_resolved_turbofish) = resolved_turbofish {
272                let message = format!(
273                    "elaborate_variable_inner: unused resolved_turbofish: {unused_resolved_turbofish:?}"
274                );
275                self.push_err(TypeCheckError::expecting_other_error(message, location));
276            }
277
278            None
279        };
280
281        let (id, typ) = if let Some(hir_ident) = hir_ident {
282            let id = self
283                .intern_expr(HirExpression::Ident(hir_ident.clone(), generics.clone()), location);
284
285            // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
286            let push_required_type_variables = self.current_trait.is_none();
287            let typ = self.type_check_variable_with_bindings(
288                hir_ident,
289                &id,
290                generics,
291                bindings,
292                push_required_type_variables,
293            );
294            let id = self.intern_expr_type(id, typ.clone());
295            (id, typ)
296        } else {
297            let expr = HirExpression::Error;
298            let id = self.intern_expr(expr, location);
299            let typ = Type::Error;
300            let id = self.intern_expr_type(id, typ.clone());
301            (id, typ)
302        };
303
304        (id, typ, is_comptime_local, location)
305    }

307    /// Checks whether `variable` is `Self::method_name`, `Self::AssociatedConstant`, or
308    /// `Self::AssociatedType::method_name` when we are inside a trait impl and `Self`
309    /// resolves to a primitive type.
310    ///
311    /// In the first case we elaborate this as if it were a [TypePath]
312    /// (for example, if `Self` is `u32` then we consider this the same as `u32::method_name`).
313    /// A regular path lookup won't work here for the same reason [TypePath] exists.
314    ///
315    /// In the second case we solve the associated constant by looking up its value, later
316    /// turning it into a literal.
317    ///
318    /// In the third case, we resolve the associated type first, then elaborate the method
319    /// call on that resolved type.
320    #[tracing::instrument(level = "trace", skip_all)]
321    fn elaborate_variable_as_self_method_or_associated_constant(
322        &mut self,
323        variable: &TypedPath,
324    ) -> Option<(ExprId, Type)> {
325        // We need at least 2 segments and the first must be `Self`
326        if variable.segments.len() < 2 || !variable.segments[0].ident.is_self_type_name() {
327            return None;
328        }
329
330        let location = variable.location;
331        let name = variable.segments[1].ident.as_str();
332        let self_type = self.self_type.as_ref()?;
333        let trait_impl_id = &self.current_trait_impl?;
334
335        // Check for `Self::AssociatedType::method_name` (exactly 3 segments).
336        // Longer paths like `Self::AssocType::foo::bar` are not valid since associated types
337        // resolve to concrete types and you cannot chain further path segments after a method name.
338        if variable.segments.len() == 3 {
339            // Try to resolve the second segment as an associated type
340            if let Some(assoc_type) =
341                self.interner.find_associated_type_for_impl(*trait_impl_id, name).cloned()
342            {
343                let method_ident = variable.segments[2].ident.clone();
344                let typ_location = variable.segments[1].location;
345                // Extract already-resolved turbofish generics from the path segment
346                let resolved_generics = variable.segments[2].generics.as_ref().map(|generics| {
347                    generics.iter().map(|located| located.contents.clone()).collect()
348                });
349                return Some(self.elaborate_type_path_impl_with_resolved_generics(
350                    assoc_type,
351                    method_ident,
352                    resolved_generics,
353                    typ_location,
354                ));
355            }
356            // If it's not an associated type, fall through to let regular path resolution handle it
357            return None;
358        }
359
360        // Check the `Self::AssociatedConstant` case when inside a trait impl (2 segments)
361        if let Some((definition_id, numeric_type)) =
362            self.interner.get_trait_impl_associated_constant(*trait_impl_id, name).cloned()
363        {
364            let hir_ident = HirIdent::non_trait_method(definition_id, location);
365            let hir_expr = HirExpression::Ident(hir_ident, None);
366            let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
367            return Some((id, numeric_type));
368        }
369
370        // Check if the constant exists in the trait definition (even if impl is missing it).
371        // This prevents spurious "Could not resolve" errors inside trait methods when the impl is missing the constant,
372        // since the "missing associated constant" error is reported elsewhere.
373        if let Some(trait_impl) = self.interner.try_get_trait_implementation(*trait_impl_id) {
374            let trait_id = trait_impl.borrow().trait_id;
375            let trait_ = self.interner.get_trait(trait_id);
376            if let Some(definition_id) = trait_.associated_constant_ids.get(name).copied() {
377                let numeric_type = self.interner.definition_type(definition_id);
378                let hir_ident = HirIdent::non_trait_method(definition_id, location);
379                let hir_expr = HirExpression::Ident(hir_ident, None);
380                let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
381                return Some((id, numeric_type));
382            }
383        }
384
385        // Check the `Self::method_name` case when `Self` is a primitive type (2 segments)
386        if matches!(self.self_type, Some(Type::DataType(..))) {
387            return None;
388        }
389
390        let ident = variable.segments[1].ident.clone();
391        let typ_location = variable.segments[0].location;
392        Some(self.elaborate_type_path_impl(self_type.clone(), ident, None, typ_location))
393    }

395    /// Resolve a [TypedPath] to a [HirIdent] of either some trait method, or a local or global variable.
396    #[tracing::instrument(level = "trace", skip_all)]
397    fn resolve_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
398        if let Some(trait_path_resolution) = self.resolve_trait_generic_path(&path) {
399            self.push_errors(trait_path_resolution.errors);
400
401            return match trait_path_resolution.method {
402                TraitPathResolutionMethod::NotATraitMethod(func_id) => {
403                    let ident = HirIdent {
404                        location: path.location,
405                        id: self.interner.function_definition_id(func_id),
406                        impl_kind: ImplKind::NotATraitMethod,
407                    };
408                    Some(VariableResolution::Ident(ident, trait_path_resolution.item))
409                }
410
411                TraitPathResolutionMethod::TraitItem(item) => {
412                    let ident = HirIdent {
413                        location: path.location,
414                        id: item.definition,
415                        impl_kind: ImplKind::TraitItem(item),
416                    };
417                    Some(VariableResolution::Ident(ident, trait_path_resolution.item))
418                }
419
420                TraitPathResolutionMethod::MultipleTraitsInScope => {
421                    // An error has already been pushed, don't return an identifier
422                    None
423                }
424            };
425        }
426
427        // The location of variables or definitions we register (for LSP) must be that of the
428        // path's last segment, as intermediate segments solve to other definitions.
429        let location = path.last_ident().location();
430
431        // If the Path is being used as an Expression, then it is referring to a global from a separate module
432        // Otherwise, then it is referring to an Identifier
433        // This lookup allows support of such statements: let x = foo::bar::SOME_GLOBAL + 10;
434        // If the expression is a singular indent, we search the resolver's current scope as normal.
435        let ident_from_path = self.get_ident_from_path(path);
436        ident_from_path.map(|ident_from_path| match ident_from_path {
437            IdentFromPath::Variable(variable) => {
438                self.handle_local_variable(&variable);
439                let hir_ident = HirIdent::non_trait_method(variable.ident.id, location);
440                VariableResolution::Ident(hir_ident, None)
441            }
442            IdentFromPath::Definition { id, item } => {
443                self.handle_definition_id(id, location);
444                let hir_ident = HirIdent::non_trait_method(id, location);
445                VariableResolution::Ident(hir_ident, Some(item))
446            }
447            IdentFromPath::TypeAlias(type_alias_id) => VariableResolution::TypeAlias(type_alias_id),
448        })
449    }

451    /// Solve any generics that are part of the path before the function, for example:
452    ///
453    /// ```noir
454    /// foo::Bar::<i32>::baz
455    /// ```
456    /// Solve `<i32>` above
457    #[tracing::instrument(level = "trace", skip_all)]
458    fn resolve_item_turbofish_and_self_type(
459        &mut self,
460        item: PathResolutionItem,
461    ) -> (Vec<Type>, Option<Type>) {
462        let mut errors = Vec::new();
463        let result = match item {
464            PathResolutionItem::Method(struct_id, Some(generics), _func_id) => {
465                let generics = self.resolve_struct_id_turbofish_generics(
466                    struct_id,
467                    Some(generics),
468                    &mut errors,
469                );
470                (generics, None)
471            }
472            PathResolutionItem::SelfMethod(_) => {
473                let generics = if let Some(Type::DataType(_, generics)) = &self.self_type {
474                    generics.clone()
475                } else {
476                    Vec::new()
477                };
478                (generics, None)
479            }
480            PathResolutionItem::TypeAliasFunction(type_alias_id, generics, _func_id) => {
481                let type_alias = self.interner.get_type_alias(type_alias_id);
482                let type_alias = type_alias.borrow();
483                let generics = self.resolve_type_alias_id_turbofish_generics(
484                    type_alias_id,
485                    generics,
486                    &mut errors,
487                );
488
489                // Now instantiate the underlying struct or alias with those generics, the struct might
490                // have more generics than those in the alias, like in this example:
491                //
492                // type Alias<T> = Struct<T, i32>;
493                let generics = get_type_alias_generics(&type_alias, &generics);
494                (generics, None)
495            }
496            PathResolutionItem::TraitFunction(trait_id, Some(generics), _func_id) => {
497                let trait_ = self.interner.get_trait(trait_id);
498                let kinds = vecmap(&trait_.generics, |generic| generic.kind());
499                let trait_generics =
500                    vecmap(&kinds, |kind| self.interner.next_type_variable_with_kind(kind.clone()));
501
502                let generics = self.resolve_trait_turbofish_generics(
503                    &trait_.name.to_string(),
504                    kinds,
505                    trait_generics,
506                    Some(generics.generics),
507                    generics.location,
508                    &mut errors,
509                );
510                (generics, None)
511            }
512            PathResolutionItem::TypeTraitFunction(self_type, _trait_id, _func_id) => {
513                (Vec::new(), Some(self_type))
514            }
515            PathResolutionItem::PrimitiveFunction(primitive_type, turbofish, _func_id) => {
516                let (typ, has_generics) = self.instantiate_primitive_type_with_turbofish(
517                    primitive_type,
518                    turbofish,
519                    &mut errors,
520                );
521                let generics = if has_generics {
522                    match typ {
523                        Type::String(length) => vec![*length],
524                        Type::FmtString(length, element) => vec![*length, *element],
525                        _ => {
526                            unreachable!("ICE: Primitive type has been specified to have generics")
527                        }
528                    }
529                } else {
530                    Vec::new()
531                };
532                (generics, None)
533            }
534            PathResolutionItem::Method(_, None, _)
535            | PathResolutionItem::TraitFunction(_, None, _)
536            | PathResolutionItem::Module(..)
537            | PathResolutionItem::Type(..)
538            | PathResolutionItem::TypeAlias(..)
539            | PathResolutionItem::PrimitiveType(..)
540            | PathResolutionItem::Trait(..)
541            | PathResolutionItem::TraitAssociatedType(..)
542            | PathResolutionItem::Global(..)
543            | PathResolutionItem::ModuleFunction(..)
544            | PathResolutionItem::TraitConstant(..) => (Vec::new(), None),
545        };
546        self.push_errors(errors);
547        result
548    }

550    /// Elaborates a type path used in an expression, e.g. `Type::method::<Args>`
551    #[tracing::instrument(level = "trace", skip_all)]
552    pub(super) fn elaborate_type_path(&mut self, path: TypePath) -> (ExprId, Type) {
553        let typ_location = path.typ.location;
554        let turbofish = path.turbofish;
555        let wildcard_allowed = WildcardAllowed::Yes;
556        let typ = self.use_type(path.typ, wildcard_allowed);
557        self.elaborate_type_path_impl(typ, path.item, turbofish, typ_location)
558    }

560    /// Variant of [Self::elaborate_type_path_impl_inner] that accepts unresolved generics.
561    #[tracing::instrument(level = "trace", skip_all)]
562    fn elaborate_type_path_impl(
563        &mut self,
564        typ: Type,
565        ident: Ident,
566        turbofish: Option<GenericTypeArgs>,
567        typ_location: Location,
568    ) -> (ExprId, Type) {
569        let ident_location = ident.location();
570        let check_self_param = false;
571
572        self.interner.push_type_ref_location(&typ, typ_location);
573
574        let Some(method) = self.lookup_method(
575            &typ,
576            ident.as_str(),
577            ident_location,
578            typ_location,
579            check_self_param,
580        ) else {
581            let error = Expression::new(ExpressionKind::Error, ident_location);
582            return self.elaborate_expression(error);
583        };
584
585        let func_id = method
586            .func_id(self.interner)
587            .expect("Expected trait function to be a DefinitionKind::Function");
588
589        let generics =
590            turbofish.map(|turbofish| self.use_type_args(turbofish, func_id, ident_location).0);
591
592        self.elaborate_type_path_impl_inner(&typ, typ_location, ident_location, method, generics)
593    }

630    /// Common implementation for type path impl variants.
631    #[tracing::instrument(level = "trace", skip_all)]
632    fn elaborate_type_path_impl_inner(
633        &mut self,
634        typ: &Type,
635        _typ_location: Location,
636        ident_location: Location,
637        method: HirMethodReference,
638        generics: Option<Vec<Type>>,
639    ) -> (ExprId, Type) {
640        let func_id = method
641            .func_id(self.interner)
642            .expect("Expected trait function to be a DefinitionKind::Function");
643
644        let id = self.interner.function_definition_id(func_id);
645
646        let impl_kind = match method {
647            HirMethodReference::FuncId(_) => ImplKind::NotATraitMethod,
648            HirMethodReference::TraitItemId(HirTraitMethodReference {
649                definition,
650                trait_id,
651                trait_generics,
652                assumed: _,
653            }) => {
654                let mut constraint =
655                    self.interner.get_trait(trait_id).as_constraint(ident_location);
656                constraint.trait_bound.trait_generics = trait_generics;
657                ImplKind::TraitItem(TraitItem { definition, constraint, assumed: false })
658            }
659        };
660
661        let ident = HirIdent { location: ident_location, id, impl_kind };
662        let id =
663            self.intern_expr(HirExpression::Ident(ident.clone(), generics.clone()), ident_location);
664
665        // If the method has a self type (it's an impl or trait impl), bind `typ` to the instantiated self type.
666        let function_meta = self.interner.function_meta(&func_id);
667        let self_type_generics_count =
668            function_meta.all_generics.len() - function_meta.direct_generics.len();
669        let bindings = if self_type_generics_count > 0 {
670            if let Type::Forall(type_vars, _) = &function_meta.typ
671                && let Some(self_type) = &function_meta.self_type
672            {
673                // Only instantiate type vars corresponding to the `self` type, not to function direct generics
674                let type_vars =
675                    type_vars.iter().take(self_type_generics_count).cloned().collect::<Vec<_>>();
676                let (self_type, instantiation_bindings) =
677                    self_type.substitute_type_vars_with_fresh_type_vars(&type_vars, self.interner);
678                let _ = typ.unify(&self_type);
679                instantiation_bindings
680            } else {
681                TypeBindings::default()
682            }
683        } else {
684            TypeBindings::default()
685        };
686
687        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
688        let push_required_type_variables = self.current_trait.is_none();
689        let typ = self.type_check_variable_with_bindings(
690            ident,
691            &id,
692            generics,
693            bindings,
694            push_required_type_variables,
695        );
696        let id = self.intern_expr_type(id, typ.clone());
697
698        (id, typ)
699    }

701    /// Given an [HirIdent], look up its definition, and:
702    /// * mark it as referenced at the ident [Location] (LSP mode only)
703    /// * mark the item currently being elaborated as a dependency of it
704    /// * elaborate a global definition, if needed
705    /// * add local identifiers to lambda captures
706    #[tracing::instrument(level = "trace", skip_all)]
707    pub(crate) fn handle_definition_id(&mut self, definition_id: DefinitionId, location: Location) {
708        match self.interner.definition(definition_id).kind {
709            DefinitionKind::Function(func_id) => {
710                if let Some(current_item) = self.current_item {
711                    self.interner.add_function_dependency(current_item, func_id);
712                }
713
714                self.interner.add_function_reference(func_id, location);
715            }
716            DefinitionKind::Global(global_id) => {
717                self.elaborate_global_if_unresolved(&global_id);
718                if let Some(current_item) = self.current_item {
719                    self.interner.add_global_dependency(current_item, global_id);
720                }
721
722                let global = self.interner.get_global_definition(global_id);
723                if global.comptime && !self.in_comptime_context() {
724                    self.push_err(ResolverError::ComptimeGlobalInNonComptimeCode {
725                        location,
726                        name: global.name.clone(),
727                    });
728                }
729
730                self.interner.add_global_reference(global_id, location);
731            }
732            DefinitionKind::NumericGeneric(_, ref numeric_typ) => {
733                // Initialize numeric generics to a polymorphic integer type in case
734                // they're used in expressions. We must do this here since type_check_variable
735                // does not check definition kinds and otherwise expects parameters to
736                // already be typed.
737                if self.interner.definition_type(definition_id) == Type::Error {
738                    let type_var_kind = Kind::Numeric(numeric_typ.clone());
739                    let typ = self.type_variable_with_kind(type_var_kind);
740                    self.interner.push_definition_type(definition_id, typ);
741                }
742            }
743            DefinitionKind::Local(_) => {
744                // Handled separately in `handle_local_variable`
745            }
746            DefinitionKind::AssociatedConstant(..) => {
747                // Nothing to do here
748            }
749        }
750    }

753    pub(crate) fn handle_local_variable(&mut self, variable: &Variable) {
754        self.check_if_variable_is_captured_by_closure(variable);
755        let hir_ident = &variable.ident;
756        self.interner.add_local_reference(hir_ident.id, hir_ident.location);
757    }

759    /// Starting with empty bindings, perform the type checking of an interned expression
760    /// and a corresponding identifier, returning the instantiated [Type].
761    #[tracing::instrument(level = "trace", skip_all)]
762    pub(crate) fn type_check_variable(
763        &mut self,
764        ident: HirIdent,
765        expr_id: &PushedExpr<HasLocation>,
766        generics: Option<Vec<Type>>,
767    ) -> Type {
768        let bindings = TypeBindings::default();
769        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
770        let push_required_type_variables = self.current_trait.is_none();
771        self.type_check_variable_with_bindings(
772            ident,
773            expr_id,
774            generics,
775            bindings,
776            push_required_type_variables,
777        )
778    }

780    /// Perform the type checking of an interned expression and a corresponding identifier,
781    /// returning the instantiated [Type].
782    ///
783    /// If `push_required_type_variables`, the bindings are added to the function context,
784    /// to be checked before it's finished.
785    #[tracing::instrument(level = "trace", skip_all)]
786    pub(crate) fn type_check_variable_with_bindings(
787        &mut self,
788        ident: HirIdent,
789        expr_id: &PushedExpr<HasLocation>,
790        generics: Option<Vec<Type>>,
791        mut bindings: TypeBindings,
792        push_required_type_variables: bool,
793    ) -> Type {
794        // Add type bindings from any constraints that were used.
795        // We need to do this first since otherwise instantiating the type below
796        // will replace each trait generic with a fresh type variable, rather than
797        // the type used in the trait constraint (if it exists). See #4088.
798        if let ImplKind::TraitItem(method) = &ident.impl_kind {
799            self.bind_generics_from_trait_constraint(
800                &method.constraint,
801                method.assumed,
802                &mut bindings,
803            );
804        }
805
806        // If a global variable hasn't been defined yet, then we are most likely dealing with a self-dependency-cycle.
807        let definition = self.interner.definition(ident.id);
808        // Some associated constants also have Global as Kind, and they are not defined when look them up here; want to restrict to global `let` statements.
809        if self.in_comptime_context()
810            && definition.kind.is_global()
811            && self.interner.try_definition_type(ident.id).is_none()
812        {
813            self.push_err(ResolverError::DependencyCycle {
814                location: ident.location,
815                item: definition.name.clone(),
816                cycle: "the variable definition type hasn't been resolved yet".to_string(),
817            });
818            return Type::Error;
819        }
820
821        // An identifiers type may be forall-quantified in the case of generic functions.
822        // E.g. `fn foo<T>(t: T, field: Field) -> T` has type `forall T. fn(T, Field) -> T`.
823        // We must instantiate identifiers at every call site to replace this T with a new type
824        // variable to handle generic functions.
825        let t = self.type_substitute_trait_as_type(&ident);
826
827        let definition = self.interner.definition(ident.id);
828        let function_generic_count = match definition.kind {
829            DefinitionKind::Function(function) => {
830                self.interner.function_modifiers(&function).generic_count
831            }
832            _ => 0,
833        };
834
835        let location = self.interner.expr_location(expr_id);
836
837        // This instantiates a trait's generics as well which need to be set
838        // when the constraint below is later solved for when the function is
839        // finished. How to link the two?
840        let (typ, bindings) =
841            self.instantiate(t, bindings, generics, function_generic_count, location);
842
843        if let ImplKind::TraitItem(mut method) = ident.impl_kind {
844            method.constraint.apply_bindings(&bindings);
845            if method.assumed {
846                let trait_generics = method.constraint.trait_bound.trait_generics.clone();
847                let object_type = method.constraint.typ;
848                let trait_impl = TraitImplKind::Assumed { object_type, trait_generics };
849                self.interner.select_impl_for_expression(**expr_id, trait_impl);
850            } else {
851                // this constraint should lead to choosing a trait impl method
852                self.push_trait_constraint(method.constraint, **expr_id, true);
853            }
854        }
855
856        // Push any trait constraints required by this definition to the context
857        // to be checked later when the type of this variable is further constrained.
858        //
859        // This must be done before the above trait constraint in case the above one further
860        // restricts types.
861        //
862        // For example, in this code:
863        //
864        // ```noir
865        // trait One {}
866        //
867        // trait Two<O: One> {
868        //     fn new() -> Self;
869        // }
870        //
871        // fn foo<X: One, T: Two<X>>() {
872        //     let _: T = Two::new();
873        // }
874        // ```
875        //
876        // when type-checking `Two::new` we'll have a return type `'2` which is constrained by `'2: Two<'1>`.
877        // Then the definition for `new` has a constraint on it, `O: One`, which translates to `'1: One`.
878        //
879        // Because of the explicit type in the `let`, `'2` will be unified with `T`.
880        // Then we must first verify the constraint `'2: Two<'1>`, which is now `T: Two<'1>`, to find
881        // that the implementation is the assumed one `T: Two<X>` so that `'1` is bound to `X`.
882        // Then we can successfully verify the constraint `'1: One` which now became `X: One` which holds
883        // because of the assumed constraint.
884        //
885        // If we try to find a trait implementation for `'1` before finding one for `'2` we'll never find it.
886        let definition = self.interner.definition(ident.id);
887        if let DefinitionKind::Function(function) = definition.kind {
888            let function = self.interner.function_meta(&function);
889            for mut constraint in function.all_trait_constraints().cloned().collect::<Vec<_>>() {
890                constraint.apply_bindings(&bindings);
891
892                // This constraint shouldn't lead to choosing a trait impl method
893                self.push_trait_constraint(constraint, **expr_id, false);
894            }
895        }
896
897        if push_required_type_variables {
898            // Record required type variables in a predictable order to avoid nondeterminism in error messages.
899            let required_type_variables =
900                btree_map(bindings.values(), |(type_variable, _, typ)| {
901                    (type_variable.id(), typ.clone())
902                });
903
904            for (type_variable_id, typ) in required_type_variables {
905                self.push_required_type_variable(
906                    type_variable_id,
907                    typ,
908                    BindableTypeVariableKind::Ident(ident.id),
909                    ident.location,
910                );
911            }
912        }
913
914        self.interner.store_instantiation_bindings(**expr_id, bindings);
915        typ
916    }

918    /// If the type of the [HirIdent] is a function that returns an `impl Trait`,
919    /// then it might need elaboration before it can be substituted to a [Type].
920    /// Try to elaborate it now.
921    ///
922    /// Returns a type error if the callee cannot be resolved on a second try,
923    /// which indicates a dependency cycle.
924    #[tracing::instrument(level = "trace", skip_all)]
925    fn type_substitute_trait_as_type(&mut self, ident: &HirIdent) -> Type {
926        let func_id = match self.interner.id_type_substitute_trait_as_type(ident.id) {
927            Ok(typ) => return typ,
928            Err(func_id) => func_id,
929        };
930
931        // Try to elaborate, so we get an expression for the body.
932        self.elaborate_function(func_id);
933
934        // Now try again. If it's still not working, give up.
935        match self.interner.id_type_substitute_trait_as_type(ident.id) {
936            Ok(typ) => typ,
937            Err(_) => {
938                let def = self.interner.definition(ident.id);
939                self.push_err(ResolverError::DependencyCycle {
940                    location: ident.location,
941                    item: def.name.clone(),
942                    cycle: "'impl Trait' could not be resolved to the type of the function body"
943                        .to_string(),
944                });
945                Type::Error
946            }
947        }
948    }

950    /// Instantiate a [Type] with the given [TypeBindings], returning the bindings potentially
951    /// extended from any turbofish generics.
952    ///
953    /// If there are turbofish generics and their number matches the expectations of the function,
954    /// those are used as well, otherwise they are ignored and an error is pushed.
955    #[tracing::instrument(level = "trace", skip_all)]
956    fn instantiate(
957        &mut self,
958        typ: Type,
959        bindings: TypeBindings,
960        turbofish_generics: Option<Vec<Type>>,
961        function_generic_count: usize,
962        location: Location,
963    ) -> (Type, TypeBindings) {
964        match turbofish_generics {
965            Some(turbofish_generics) => {
966                let forall_generic_count =
967                    if let Type::Forall(generics, _) = &typ { generics.len() } else { 0 };
968
969                if turbofish_generics.len() != function_generic_count {
970                    let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
971                        expected_count: function_generic_count,
972                        actual_count: turbofish_generics.len(),
973                        location,
974                    };
975                    self.push_err(CompilationError::TypeError(type_check_err));
976                    typ.instantiate_with_bindings(bindings, self.interner)
977                } else if forall_generic_count < function_generic_count {
978                    // In the next branch, calling instantiate_with_bindings_and_turbofish asserts that
979                    // turbofish_generics.len() + implicit_generic_count == forall_generic_count,
980                    // but if we have less generics in forall than the the turbo fish than this will never hold.
981                    // This is the case when the function generics have duplicates, which are filtered out.
982                    // This means some duplicate error has already been reported, so there is no point trying.
983                    self.push_err(TypeCheckError::expecting_other_error(
984                        "forall has fewer generics than function",
985                        location,
986                    ));
987                    (Type::Error, bindings)
988                } else {
989                    // Fetch the count of any implicit generics on the function, such as
990                    // for a method within a generic impl.
991                    let implicit_generic_count = forall_generic_count
992                        .checked_sub(function_generic_count)
993                        .expect("forall should have at least as many generics as the function");
994
995                    typ.instantiate_with_bindings_and_turbofish(
996                        bindings,
997                        turbofish_generics,
998                        self.interner,
999                        implicit_generic_count,
1000                    )
1001                }
1002            }
1003            None => typ.instantiate_with_bindings(bindings, self.interner),
1004        }
1005    }
1006}

1008/// Bind the generics of the [Type] aliased by the [TypeAlias] to a list of generic arguments,
1009/// recursively expanding the generics aliased aliases, finally returning the generics of the
1010/// innermost aliased struct.
1011///
1012/// Panics if it encounters a type other than alias or struct.
1013fn get_type_alias_generics(type_alias: &TypeAlias, generics: &[Type]) -> Vec<Type> {
1014    let typ = type_alias.get_type(generics);
1015    match typ {
1016        Type::DataType(_, generics) => generics,
1017        Type::Alias(type_alias, generics) => {
1018            get_type_alias_generics(&type_alias.borrow(), &generics)
1019        }
1020        // Primitive types have no generics
1021        _ => Vec::new(),
1022    }
