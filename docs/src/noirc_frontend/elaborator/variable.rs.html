37/// The result of [`Elaborator::resolve_variable`]: what a path used as an expression resolves
38/// to. See [`Elaborator::resolve_variable`] for the full, ordered list of the path forms each
39/// variant covers.
40#[allow(clippy::large_enum_variant)]
41pub(crate) enum VariableResolution {
42    /// The path was elaborated straight to an expression. This happens for the `Self::*` forms
43    /// handled inside a trait impl: an associated constant, an associated-type method, or a
44    /// method when `Self` is a primitive type.
45    Expression(ExprId, Type),
46    /// The path resolved to a local variable, a definition (global, function, enum-variant
47    /// global), or a trait item (method or associated constant).
48    Ident(HirIdent, Option<PathResolutionItem>),
49    /// The path resolved to a type alias that is numeric, infinitely recursive or one that errored.
50    TypeAlias(TypeAliasId),
51}
52
53/// Describes how the type-level generics of a path item (everything before the final method)
54/// should be bound when the item resolves to a function.
55enum ItemTypeBinding {
56    /// Nothing to bind.
57    None,
58    /// A concrete self type to unify against the impl's declared self type, binding the impl's
59    /// generics. Used whenever a method is reached through a concrete type — a (possibly
60    /// partially concrete) data type `S::<u32, bool>::foo`, a primitive `str::<3>::foo`, a type
61    /// alias `type Pair = (u8, u32); Pair::foo`, or `Self::foo` inside an impl.
62    ConcreteSelfType(Type),
63    /// A concrete type to bind to the function's `Self` generic, leaving impl selection to
64    /// trait constraint solving. Used for `<Type as Trait>::foo`.
65    SelfGeneric(Type),
66    /// The trait's declared generic arguments to bind, in order. Used for a static method reached
67    /// through a trait, e.g. `Trait::<u32>::foo`, where the turbofish provides the trait's
68    /// generics rather than a self type.
69    TraitGenerics(Vec<Type>),
70    /// An enum whose variant constructor is being called (e.g. `Foo::<u32>::Eggs`), carrying the
71    /// enum's generic arguments. A constructor has no impl and no self type, so the enum's
72    /// generics are bound directly rather than by unifying a self type against an impl.
73    EnumVariantConstructor(Vec<Type>),
74}

76/// A path's prefix resolved into [its kind](PathPrefixKind) plus the "rest" — the last segment
77/// (the item accessed on the prefix) and the prefix's own turbofish. Produced by
78/// [`Elaborator::resolve_path_prefix`]; [`Elaborator::resolve_prefixed_variable`] resolves the last
79/// segment against the kind without re-deriving anything from the original path.
80#[derive(Debug)]
81struct ResolvedPrefix {
82    /// The path's last segment: the item being accessed on the prefix.
83    last_segment: TypedPathSegment,
84    /// Turbofish on the prefix's own last segment (e.g. the `<T>` in `Foo::<T>::bar`).
85    turbofish: Option<Turbofish>,
86    kind: PathPrefixKind,
87}

89/// What the prefix of a path (every segment but the last) is, when the path is used as an
90/// expression. This describes *what the prefix is*, not the already-resolved item; the shared
91/// "rest" (last segment, turbofish) lives on [`ResolvedPrefix`].
92#[derive(Debug)]
93enum PathPrefixKind {
94    /// `Self` inside a trait *impl*, where `Self` is the impl's concrete type (carried here, along
95    /// with the impl) since a trait impl always has both. The last segment is an associated-type
96    /// method, an associated constant, a primitive-`Self` method, or a plain method on the self
97    /// type ([`Elaborator::resolve_self_in_trait_impl`]).
98    SelfInTraitImpl { self_type: Type, trait_impl_id: TraitImplId },
99    /// `Self` inside a trait *definition* (carried here): an assumed constraint on the current
100    /// trait (or a supertrait reached through it) ([`Elaborator::resolve_self_in_trait`]).
101    SelfInTrait { trait_id: TraitId },
102    /// `Self` as a plain concrete type (an inherent impl, or any other context where `Self` names
103    /// a data type): the last segment resolves like `Type::method`
104    /// ([`Elaborator::resolve_self_as_concrete_type`]).
105    SelfInImpl,
106    /// A generic parameter with `: Trait` bound(s) in scope (e.g. `T` in `T::method`); the carried
107    /// constraints are the matched bounds. The last segment is a method or associated constant
108    /// reached through one of them.
109    BoundedGeneric(Vec<TraitConstraint>),
110    /// A trait (e.g. `Trait` in `Trait::method` / `Trait::CONST`). The last segment is a trait
111    /// static method or an associated constant.
112    Trait { trait_id: TraitId, resolution: PathResolution },
113    /// A concrete type, type alias, or primitive type (e.g. `Type` in `Type::method`). The last
114    /// segment is an inherent or qualified trait method.
115    Type { resolution: PathResolution },
116    /// A module (e.g. `foo::bar` in `foo::bar::GLOBAL`). The last segment is an ordinary value
117    /// item, resolved as a value directly in `module_id`. `errors` are the prefix's own resolution
118    /// errors (e.g. an intermediate segment's visibility), reported when the last segment resolves.
119    Module { module_id: ModuleId, errors: Vec<PathResolutionError> },
120    /// The prefix is `Self` but there is no self type in scope (e.g. in a free function), so `Self`
121    /// names nothing.
122    SelfNotInScope,
123    /// The prefix is not something that can carry an associated item: it is not a type, trait, or
124    /// module (it is a value, or it failed to resolve), so the path names nothing. The carried error
125    /// (either the prefix's own resolution failure, or that it is a value rather than a namespace) is
126    /// reported as-is.
127    InvalidPrefix { error: PathResolutionError },
128}

131    #[tracing::instrument(level = "trace", skip_all)]
132    pub(super) fn elaborate_variable(&mut self, variable: Path) -> (ExprId, Type) {
133        let (id, typ, is_comptime_local, location) = self.elaborate_variable_inner(variable);
134
135        // Only check has_errors when we need to call the interpreter
136        // If this variable is a comptime local variable, use its current value as the final expression
137        if is_comptime_local {
138            let mut interpreter = self.setup_interpreter();
139            let value = interpreter.evaluate(id);
140            // If the value is an error it means the variable already had an error, so don't report it here again
141            // (the error will make no sense, it will say that a non-comptime variable was referenced at runtime
142            // but that's not true)
143            if value.is_ok() {
144                let from_macro_call = false;
145                let (id, typ) = self.inline_comptime_value(value, location, from_macro_call);
146                self.debug_comptime(location, |interner, _| id.to_display_ast(interner).kind);
147                (id, typ)
148            } else {
149                (id, typ)
150            }
151        } else {
152            (id, typ)
153        }
154    }

156    /// Helper function containing the elaboration logic for a variable.
157    /// Returns the expression ID, type, whether it's a comptime local, and location.
158    #[tracing::instrument(level = "trace", skip_all)]
159    fn elaborate_variable_inner(&mut self, variable: Path) -> (ExprId, Type, bool, Location) {
160        let variable = self.validate_path(variable);
161
162        let resolved_turbofish = variable.segments.last().unwrap().generics.clone();
163
164        // A turbofish on the segment *before* the last one (e.g. `Foo::<u32>::Spam`) provides
165        // type generics for the type the last segment is resolved within. This is needed for
166        // fieldless enum variants, which resolve to a global rather than a function.
167        let type_segment_turbofish = (variable.segments.len() >= 2)
168            .then(|| variable.segments[variable.segments.len() - 2].generics.clone())
169            .flatten();
170
171        let location = variable.location;
172        let variable_resolution = self.resolve_variable(variable);
173
174        let (hir_ident, item) = match variable_resolution {
175            Some(VariableResolution::Expression(id, typ)) => return (id, typ, false, location),
176            Some(VariableResolution::TypeAlias(type_alias_id)) => {
177                // A type alias to a numeric generics is considered like a variable,
178                // but it is not a real variable so it does not resolve to a valid Identifier.
179                // In order to handle this, we retrieve the numeric generics expression that the type aliases to.
180                let type_alias = self.interner.get_type_alias(type_alias_id);
181                let alias_module_id = type_alias.borrow().module_id;
182                if let Some(type_alias_expr) = &type_alias.borrow().numeric_expr {
183                    // Extract the declared numeric type from the type alias's kind.
184                    let declared_type = match type_alias.borrow().typ.kind() {
185                        Kind::Numeric(declared_type) => declared_type,
186                        _ => Box::new(Type::Error),
187                    };
188                    let declared_type = *declared_type;
189                    let var_expr = UnresolvedTypeExpression::to_expression_kind(type_alias_expr);
190
191                    // The expression we create for this particular instantiation of the numeric type alias
192                    // must have the same location as the path that refers to it.
193                    let var_expr = Expression::new(var_expr, location);
194
195                    // Resolve turbofish generics for the type alias.
196                    // `resolved_turbofish` contains already-resolved types from
197                    // validate_path, so we use resolve_alias_turbofish_generics
198                    // directly which accepts resolved types.
199                    let alias_generics = &type_alias.borrow().generics;
200                    let alias_generic_types = vecmap(alias_generics, |generic| {
201                        self.interner.next_type_variable_with_kind(generic.kind())
202                    });
203                    let mut errors = Vec::new();
204                    let type_alias_ref = type_alias.borrow();
205                    let resolved_generics = self.resolve_alias_turbofish_generics(
206                        &type_alias_ref,
207                        alias_generic_types,
208                        resolved_turbofish,
209                        location,
210                        &mut errors,
211                    );
212                    self.push_errors(errors);
213
214                    // Introduce alias generics into scope so the numeric expression
215                    // resolves them correctly (not to globals or other variables
216                    // that happen to share the same name). Bind each generic's type
217                    // variable to the turbofish-resolved type.
218                    self.push_scope();
219                    for (generic, resolved_type) in
220                        alias_generics.iter().zip_eq(resolved_generics.iter())
221                    {
222                        if let Kind::Numeric(numeric_type) = &generic.kind() {
223                            let id = self.interner.next_type_variable_id();
224                            let type_var = TypeVariable::unbound(id, generic.kind());
225                            type_var.bind(resolved_type.clone());
226                            let definition =
227                                DefinitionKind::NumericGeneric(type_var, numeric_type.clone());
228                            let ident = Ident::new(generic.name.to_string(), generic.location);
229                            let hir_ident = self.add_variable_decl(
230                                ident, false, // mutable
231                                true,  // allow_shadowing
232                                false, // warn_if_unused
233                                false, // warn_if_not_mutated
234                                definition,
235                            );
236                            self.interner.push_definition_type(hir_ident.id, *numeric_type.clone());
237                        }
238                    }
239
240                    // The alias's numeric expression has already been kind-checked at
241                    // alias-definition time (see `convert_expression_type`), which is
242                    // where any "value does not fit" diagnostic is emitted. Drop any
243                    // literals queued for the function-context fit check during
244                    // re-elaboration so the same overflow is not reported twice.
245                    let literals_before = self.integer_literal_expr_ids_len();
246                    // Re-elaborate the alias body in the alias's defining module
247                    // so unqualified names resolve against the alias's scope,
248                    // not the caller's. Mirrors `define_type_alias` in mod.rs.
249                    let previous_module = self.replace_module(alias_module_id);
250                    let (id, typ) = self.elaborate_expression(var_expr);
251                    self.restore_module(previous_module);
252                    self.truncate_integer_literal_expr_ids(literals_before);
253                    self.pop_scope();
254
255                    // Unify the expression's type with the declared type from the type alias
256                    // to ensure proper type checking.
257                    self.unify_or_type_mismatch(&typ, &declared_type, type_alias_expr.location());
258
259                    return (id, declared_type, false, location);
260                }
261                (None, None)
262            }
263            Some(VariableResolution::Ident(ident, item)) => (Some(ident), item),
264            None => (None, None),
265        };
266
267        let definition_id = hir_ident.as_ref().map(|ident| ident.id);
268
269        let item_type_binding = if let Some(item) = item {
270            self.resolve_item_type_binding(item)
271        } else {
272            ItemTypeBinding::None
273        };
274
275        let definition = definition_id.map(|id| self.interner.definition(id));
276        let is_comptime_local = !self.in_comptime_context()
277            && definition.is_some_and(DefinitionInfo::is_comptime_local);
278        let definition_kind = definition.as_ref().map(|definition| definition.kind.clone());
279
280        let mut bindings = TypeBindings::default();
281        let generics = if let Some(DefinitionKind::Function(func_id)) = &definition_kind {
282            self.usage_tracker.mark_impl_function_as_used(func_id);
283
284            match item_type_binding {
285                ItemTypeBinding::None => {}
286                ItemTypeBinding::SelfGeneric(self_generic) => {
287                    // Bind the concrete self type to the function's `Self` generic, leaving impl
288                    // selection to trait constraint solving.
289                    let func_generics = &self.function_meta(*func_id).all_generics;
290                    let self_resolved_generic = func_generics
291                        .iter()
292                        .find(|generic| generic.name.as_str() == SELF_TYPE_NAME);
293                    if let Some(self_resolved_generic) = self_resolved_generic {
294                        let type_var = &self_resolved_generic.type_var;
295                        bindings.insert(
296                            type_var.id(),
297                            (type_var.clone(), type_var.kind(), self_generic),
298                        );
299                    }
300                }
301                ItemTypeBinding::ConcreteSelfType(self_type) => {
302                    // The method was reached through a concrete type. Unify that type against the
303                    // impl's self type to bind the impl's generics, regardless of which type former
304                    // it is (data type, array, slice, tuple, primitive, function type, ...).
305                    self.bind_impl_generics_from_self_type(
306                        *func_id,
307                        &self_type,
308                        &mut bindings,
309                        location,
310                    );
311                }
312                ItemTypeBinding::EnumVariantConstructor(type_generics) => {
313                    self.bind_enum_variant_constructor_generics(
314                        *func_id,
315                        type_generics,
316                        &mut bindings,
317                    );
318                }
319                ItemTypeBinding::TraitGenerics(trait_generics) => {
320                    self.bind_trait_function_generics(*func_id, trait_generics, &mut bindings);
321                }
322            }
323
324            // Resolve any generics if the variable we have resolved is a function
325            // and if the turbofish operator was used.
326            self.resolve_function_turbofish_generics(func_id, resolved_turbofish, location)
327        } else {
328            // A fieldless enum variant resolves to a global. A turbofish on the variant path
329            // binds the enum's generics, which global path resolution does not carry on its
330            // own. The turbofish may be on the type segment (`Foo::<u32>::Spam`) or the
331            // variant segment (`Foo::Spam::<u32>`); both denote the same enum generics.
332            let is_enum_variant_global =
333                if let Some(DefinitionKind::Global(global_id)) = &definition_kind {
334                    self.interner.is_enum_variant_global(*global_id)
335                } else {
336                    false
337                };
338
339            if is_enum_variant_global {
340                if let Some(turbofish) = resolved_turbofish.or(type_segment_turbofish) {
341                    self.bind_enum_variant_global_turbofish(
342                        definition_id.unwrap(),
343                        &turbofish,
344                        location,
345                        &mut bindings,
346                    );
347                }
348            } else if let Some(unused_resolved_turbofish) = resolved_turbofish {
349                let message = format!(
350                    "elaborate_variable_inner: unused resolved_turbofish: {unused_resolved_turbofish:?}"
351                );
352                self.push_err(TypeCheckError::expecting_other_error(message, location));
353            }
354
355            None
356        };
357
358        let (id, typ) = if let Some(hir_ident) = hir_ident {
359            let id = self
360                .intern_expr(HirExpression::Ident(hir_ident.clone(), generics.clone()), location);
361
362            let typ = self.type_check_variable_with_bindings(hir_ident, &id, generics, bindings);
363            let id = self.intern_expr_type(id, typ.clone());
364            (id, typ)
365        } else {
366            let expr = HirExpression::Error;
367            let id = self.intern_expr(expr, location);
368            let typ = Type::Error;
369            let id = self.intern_expr_type(id, typ.clone());
370            (id, typ)
371        };
372
373        (id, typ, is_comptime_local, location)
374    }

376    /// Bind the type-segment turbofish of a fieldless enum variant path (e.g. `Foo::<u32>::Spam`)
377    /// to the variant global's `Forall` generics (the enum's generics), validating the count the
378    /// same way [`Self::resolve_item_turbofish_generics`] does. A non-generic enum has no `Forall`,
379    /// so a turbofish on it produces a count-mismatch error.
380    pub(super) fn bind_enum_variant_global_turbofish(
381        &mut self,
382        definition_id: DefinitionId,
383        turbofish: &[Located<Type>],
384        location: Location,
385        bindings: &mut TypeBindings,
386    ) {
387        let global_type = self.interner.definition_type(definition_id);
388        let (typevars, enum_name) = match &global_type {
389            Type::Forall(typevars, body) => (typevars.clone(), data_type_name(body)),
390            other => (Vec::new(), data_type_name(other)),
391        };
392
393        let mut turbofish = turbofish.to_vec();
394        if turbofish.len() != typevars.len() {
395            self.push_err(TypeCheckError::GenericCountMismatch {
396                item: format!("enum `{}`", enum_name.unwrap_or_default()),
397                expected: typevars.len(),
398                found: turbofish.len(),
399                location,
400            });
401            // Pad/truncate to the expected length so every generic is still determined,
402            // matching `resolve_function_turbofish_generics` and avoiding a cascade of
403            // "type annotation needed" errors.
404            turbofish.resize(typevars.len(), Located::from(location, Type::Error));
405        }
406
407        for (located_type, type_var) in turbofish.into_iter().zip(&typevars) {
408            let type_location = located_type.location();
409            let typ = self.check_type_kind(located_type.contents, &type_var.kind(), type_location);
410            bindings.insert(type_var.id(), (type_var.clone(), type_var.kind(), typ));
411        }
412    }

414    /// Resolves the `Self::…` forms that only make sense inside a trait impl, where `Self` is a
415    /// concrete type with associated items. The segments *after* `Self` (the "rest") select the
416    /// form:
417    ///
418    /// - `[AssociatedType, method]` — resolve the associated type, then elaborate the method on it.
419    /// - `[item]` — an associated constant (looked up for its value, later a literal), or, when
420    ///   `Self` is a primitive type, a method elaborated as if it were a [TypePath]
421    ///   (`u32::method_name`); a regular path lookup won't work, for the same reason [TypePath]
422    ///   exists.
423    ///
424    /// Returns `None` for any other shape (including a data-type `Self`, handled as a plain type
425    /// prefix), so the caller falls back to resolving `Self` as a type.
426    #[tracing::instrument(level = "trace", skip_all)]
427    pub(super) fn resolve_variable_as_self_method_or_associated_constant(
428        &mut self,
429        variable: &TypedPath,
430        self_type: Type,
431        trait_impl_id: TraitImplId,
432    ) -> Option<(ExprId, Type)> {
433        match &variable.segments[1..] {
434            [associated_type, method] => {
435                let associated_type = self
436                    .interner
437                    .find_associated_type_for_impl(trait_impl_id, associated_type.ident.as_str())
438                    .cloned()?;
439                // Extract already-resolved turbofish generics from the method segment.
440                let resolved_generics = method.generics.as_ref().map(|generics| {
441                    generics.iter().map(|located| located.contents.clone()).collect()
442                });
443                Some(self.elaborate_type_path_impl_with_resolved_generics(
444                    associated_type,
445                    &method.ident,
446                    resolved_generics,
447                    variable.segments[1].location,
448                ))
449            }
450            [item] => self.elaborate_self_associated_constant_or_primitive_method(
451                &self_type,
452                trait_impl_id,
453                item,
454                variable.location,
455                variable.segments[0].location,
456            ),
457            _ => None,
458        }
459    }

461    /// The `Self::item` (single segment after `Self`) case of
462    /// [`Self::resolve_variable_as_self_method_or_associated_constant`]: an associated constant
463    /// (from the impl, or from the trait when the impl is missing it), or a method when `Self` is a
464    /// primitive type. A data-type `Self` returns `None` so it is resolved as a plain type prefix.
465    fn elaborate_self_associated_constant_or_primitive_method(
466        &mut self,
467        self_type: &Type,
468        trait_impl_id: TraitImplId,
469        item: &TypedPathSegment,
470        location: Location,
471        self_location: Location,
472    ) -> Option<(ExprId, Type)> {
473        let name = item.ident.as_str();
474
475        // The associated constant declared on the impl.
476        if let Some((definition_id, numeric_type)) =
477            self.interner.get_trait_impl_associated_constant(trait_impl_id, name).cloned()
478        {
479            return Some(self.intern_associated_constant(definition_id, numeric_type, location));
480        }
481
482        // The constant declared on the trait, even if the impl is missing it. This prevents a
483        // spurious "Could not resolve" inside trait methods; the "missing associated constant"
484        // error is reported elsewhere.
485        if let Some(trait_impl) = self.interner.try_get_trait_implementation(trait_impl_id) {
486            let trait_id = trait_impl.borrow().trait_id;
487            let trait_ = self.interner.get_trait(trait_id);
488            if let Some(definition_id) = trait_.associated_constant_ids.get(name).copied() {
489                let numeric_type = self.interner.definition_type(definition_id);
490                return Some(self.intern_associated_constant(
491                    definition_id,
492                    numeric_type,
493                    location,
494                ));
495            }
496        }
497
498        // A data-type `Self::method` is resolved as a plain type prefix, not here.
499        if matches!(self_type, Type::DataType(..)) {
500            return None;
501        }
502
503        Some(self.elaborate_type_path_impl(self_type.clone(), &item.ident, None, self_location))
504    }

519    /// Resolve a [`TypedPath`] used as an expression to the item it names.
520    ///
521    /// A path with a prefix (more than one segment) names an item accessed *through* that prefix,
522    /// fully handled by [`Self::resolve_prefixed_variable`]. A single-segment path resolves in the
523    /// current scope (a local variable, or a value item — global, function, enum-variant global,
524    /// numeric type alias) via [`Self::resolve_unprefixed_variable`].
525    #[tracing::instrument(level = "trace", skip_all)]
526    fn resolve_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
527        if path.segments.len() > 1 {
528            self.resolve_prefixed_variable(path)
529        } else {
530            self.resolve_unprefixed_variable(path)
531        }
532    }

534    /// Resolve a [`TypedPath`] that has a prefix (more than one segment) to the item it names —
535    /// fully: it always resolves the path, reports an error, or falls back to a value lookup, so
536    /// the caller never needs a further fallback. [`Self::resolve_path_prefix`] classifies the
537    /// prefix once; the last segment is resolved against that classification directly in the
538    /// already-resolved prefix (a module value via [`Self::resolve_value_in_module`], an enum
539    /// variant or associated constant on a type via [`Self::resolve_value_in_type`]).
540    ///
541    /// Returns `None` only when an error has already been reported (an ambiguous trait method, or
542    /// an unresolved name), so the caller should produce an error expression rather than retry.
543    #[tracing::instrument(level = "trace", skip_all)]
544    fn resolve_prefixed_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
545        let ResolvedPrefix { last_segment, turbofish, kind } = self.resolve_path_prefix(&path);
546
547        match kind {
548            // `Self` is contextual; each context resolves the last segment its own way.
549            PathPrefixKind::SelfInTraitImpl { self_type, trait_impl_id } => self
550                .resolve_self_in_trait_impl(
551                    path,
552                    self_type,
553                    trait_impl_id,
554                    last_segment,
555                    turbofish,
556                ),
557            PathPrefixKind::SelfInTrait { trait_id } => {
558                self.resolve_self_in_trait(path, trait_id, last_segment, turbofish)
559            }
560            // `Self` is a concrete type here (classification guarantees `self_type` exists), so it
561            // resolves exactly like `Type::method`.
562            PathPrefixKind::SelfInImpl => {
563                self.resolve_self_as_concrete_type(path, last_segment, turbofish)
564            }
565            PathPrefixKind::BoundedGeneric(bounds) => {
566                self.resolve_bounded_generic_item(bounds, &last_segment, turbofish, path.location)
567            }
568            PathPrefixKind::Type { resolution } => {
569                let is_self_prefix = false;
570                self.resolve_method_on_type_prefix(
571                    last_segment,
572                    turbofish,
573                    is_self_prefix,
574                    resolution,
575                    path.location,
576                )
577            }
578            // A trait prefix: the last segment is either a trait static method (`Trait::method`)
579            // or an associated constant (`Trait::CONST`).
580            PathPrefixKind::Trait { trait_id, resolution } => self.resolve_trait_item_on_prefix(
581                trait_id,
582                turbofish,
583                &last_segment,
584                resolution,
585                path.location,
586            ),
587            // A module prefix: the last segment is an ordinary value item, resolved as a value
588            // directly in the already-resolved module.
589            PathPrefixKind::Module { module_id, errors } => {
590                self.push_errors(errors);
591                self.resolve_value_in_module(module_id, last_segment)
592            }
593            // No usable prefix: the path names nothing, and a value lookup would only rediscover
594            // the resolution failure already carried here, so report it directly.
595            PathPrefixKind::InvalidPrefix { error } => {
596                self.push_err(error);
597                None
598            }
599            // `Self` with no self type in scope: report it directly.
600            PathPrefixKind::SelfNotInScope => {
601                self.push_err(PathResolutionError::Unresolved(path.segments[0].ident.clone()));
602                None
603            }
604        }
605    }

607    /// Classify the prefix (every segment but the last) of a path used as an expression. This does
608    /// not resolve the last segment — it only answers "what is the prefix?", which the caller uses
609    /// to decide how to resolve the last segment. The caller only invokes this for a path that has
610    /// a prefix (more than one segment).
611    ///
612    /// The classification order is significant: it decides which interpretation wins when a path
613    /// is ambiguous, and mirrors the historical probe order. `Self` and a bounded generic are
614    /// recognized from context (`Self` is contextual; a generic is not in the module namespace)
615    /// before the prefix is resolved as a type to tell trait-, type-, and module-prefixed forms
616    /// apart.
617    #[tracing::instrument(level = "trace", skip_all)]
618    fn resolve_path_prefix(&mut self, path: &TypedPath) -> ResolvedPrefix {
619        // The caller (`resolve_variable`) only takes this path for a multi-segment path, so popping
620        // the last segment is always valid.
621        debug_assert!(path.segments.len() >= 2);
622
623        let mut prefix = path.clone();
624        let last_segment = prefix.pop();
625        let turbofish = prefix.last_segment().turbofish();
626
627        // `Self` and a generic parameter are only meaningful as a plain path: `crate::Self` /
628        // `super::T` name an item in another module, not the contextual `Self` or an in-scope
629        // generic, so they must not be classified as such.
630        let kind = if path.kind == PathKind::Plain && path.segments[0].ident.is_self_type_name() {
631            // `Self` is contextual; which kind it is depends only on where we are (a trait impl, a
632            // trait definition, somewhere `Self` is a plain type, or nowhere at all), so it is
633            // classified here and the matching arm resolves the last segment.
634            if let Some(trait_impl_id) = self.current_trait_impl {
635                let self_type =
636                    self.self_type.clone().expect("a trait impl always has a self type");
637                PathPrefixKind::SelfInTraitImpl { self_type, trait_impl_id }
638            } else if let Some(trait_id) = self.current_trait {
639                PathPrefixKind::SelfInTrait { trait_id }
640            } else if self.self_type.is_some() {
641                PathPrefixKind::SelfInImpl
642            } else {
643                PathPrefixKind::SelfNotInScope
644            }
645        } else if path.kind == PathKind::Plain
646            && let Some(bounds) = self.matching_generic_bounds(path)
647        {
648            // A generic parameter (e.g. `T`) with a `T: Trait` bound in scope. A generic is not in
649            // the module namespace, so this is recognized from the in-scope bounds, not resolution.
650            PathPrefixKind::BoundedGeneric(bounds)
651        } else {
652            // Otherwise the prefix is resolved as a type to distinguish trait/type/module.
653            let prefix_last_ident = prefix.last_segment().ident;
654            match self.use_path_as_type(prefix) {
655                Ok(resolution) => match &resolution.item {
656                    PathResolutionItem::Trait(trait_id) => {
657                        PathPrefixKind::Trait { trait_id: *trait_id, resolution }
658                    }
659                    PathResolutionItem::Type(..)
660                    | PathResolutionItem::TypeAlias(..)
661                    | PathResolutionItem::PrimitiveType(..) => PathPrefixKind::Type { resolution },
662                    PathResolutionItem::Module(module_id) => PathPrefixKind::Module {
663                        module_id: *module_id,
664                        errors: resolution.errors.clone(),
665                    },
666                    // Resolving a type path falls back to the value namespace, so the prefix's last
667                    // segment can also resolve to a value item; that (and an associated type) can't
668                    // carry the last segment of the path as an associated item, so the path names
669                    // nothing. Listed explicitly (rather than `_`) so a new `PathResolutionItem`
670                    // must be classified here.
671                    PathResolutionItem::TraitAssociatedType(..)
672                    | PathResolutionItem::Global(..)
673                    | PathResolutionItem::EnumVariant(..)
674                    | PathResolutionItem::ModuleFunction(..)
675                    | PathResolutionItem::Method(..)
676                    | PathResolutionItem::SelfMethod(..)
677                    | PathResolutionItem::TypeAliasFunction(..)
678                    | PathResolutionItem::TraitFunction(..)
679                    | PathResolutionItem::TypeTraitFunction(..)
680                    | PathResolutionItem::PrimitiveFunction(..)
681                    | PathResolutionItem::TraitConstant(..) => PathPrefixKind::InvalidPrefix {
682                        error: PathResolutionError::Unresolved(prefix_last_ident),
683                    },
684                },
685                Err(error) => PathPrefixKind::InvalidPrefix { error },
686            }
687        };
688
689        ResolvedPrefix { last_segment, turbofish, kind }
690    }

692    /// If the path's first segment names a generic parameter with `: Trait` bound(s) in scope,
693    /// return those matching bounds (so `Head::item` can be resolved through them). Only meaningful
694    /// for a two-segment path; a generic is not in the module namespace, so this scans the in-scope
695    /// bounds rather than resolving.
696    pub(super) fn matching_generic_bounds(&self, path: &TypedPath) -> Option<Vec<TraitConstraint>> {
697        if path.segments.len() != 2 {
698            return None;
699        }
700        let head = path.segments[0].ident.as_str();
701        let bounds: Vec<_> = self
702            .trait_bounds
703            .iter()
704            .filter(|constraint| {
705                matches!(&constraint.typ, Type::NamedGeneric(generic) if generic.name.as_str() == head)
706            })
707            .cloned()
708            .collect();
709        (!bounds.is_empty()).then_some(bounds)
710    }

712    /// Resolve an unprefixed (single-segment) path to a local variable, or to a value item it
713    /// names (global, function, enum-variant global, numeric type alias). The counterpart to
714    /// [`Self::resolve_prefixed_variable`], which resolves a prefixed path's last segment directly
715    /// in the already-resolved prefix ([`Self::resolve_value_in_module`] /
716    /// [`Self::resolve_value_in_type`]) and never looks for a local variable.
717    fn resolve_unprefixed_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
718        // The location of variables or definitions we register (for LSP) must be that of the
719        // path's last segment, as intermediate segments solve to other definitions.
720        let location = path.last_ident().location();
721
722        // If the Path is being used as an Expression, then it is referring to a global from a separate module
723        // Otherwise, then it is referring to an Identifier
724        // This lookup allows support of such statements: let x = foo::bar::SOME_GLOBAL + 10;
725        // If the expression is a singular indent, we search the resolver's current scope as normal.
726        let ident_from_path = self.resolve_path_as_value(path)?;
727        Some(self.variable_resolution_from_path_value(ident_from_path, location))
728    }

775    /// Build the [`VariableResolution`] an already-resolved [`PathValue`] denotes, registering
776    /// the reference (for LSP) along the way.
777    fn variable_resolution_from_path_value(
778        &mut self,
779        ident_from_path: PathValue,
780        location: Location,
781    ) -> VariableResolution {
782        match ident_from_path {
783            PathValue::Variable(variable) => {
784                self.handle_local_variable(&variable);
785                let hir_ident = HirIdent::non_trait_method(variable.ident.id, location);
786                VariableResolution::Ident(hir_ident, None)
787            }
788            PathValue::Definition { id, item } => {
789                self.handle_definition_id(id, location);
790                let hir_ident = HirIdent::non_trait_method(id, location);
791                VariableResolution::Ident(hir_ident, Some(item))
792            }
793            PathValue::TypeAlias(type_alias_id) => VariableResolution::TypeAlias(type_alias_id),
794        }
795    }

797    /// Bind an impl's generics by unifying a concrete `self_type` against the impl's declared
798    /// self type (with the impl's own generics first replaced by fresh type variables). The
799    /// resulting bindings are recorded in `bindings`.
800    ///
801    /// This works for any built-in type former the self type can be (data type, array, slice,
802    /// tuple, string, ...): unifying `(u8, u32)` against `(A, B)`, or `[u8; 4]` against
803    /// `[T; N]`, recovers the impl generics structurally, and is correct regardless of the order
804    /// the impl declares them in, e.g. `impl<let N: u32, T> Trait for [T; N]`.
805    fn bind_impl_generics_from_self_type(
806        &mut self,
807        func_id: FuncId,
808        self_type: &Type,
809        bindings: &mut TypeBindings,
810        location: Location,
811    ) {
812        let func_meta = self.function_meta(func_id);
813        let impl_generics = vecmap(func_meta.impl_generics(), |g| g.type_var.clone());
814        let Some(impl_self_type) =
815            func_meta.self_type.as_ref().map(|t| t.follow_bindings_shallow().into_owned())
816        else {
817            return;
818        };
819
820        let impl_replacements: TypeBindings = impl_generics
821            .iter()
822            .map(|type_var| {
823                let kind = type_var.kind();
824                let fresh = self.interner.next_type_variable_with_kind(kind.clone());
825                (type_var.id(), (type_var.clone(), kind, fresh))
826            })
827            .collect();
828        let substituted = impl_self_type.substitute(&impl_replacements);
829        self.unify_or_type_mismatch(self_type, &substituted, location);
830        bindings.extend(impl_replacements);
831    }

833    /// Bind a trait's declared generics from the turbofish solved for the trait before the method,
834    /// e.g. the `<u32>` in `Trait::<u32>::foo`. We only bind the trait-level portion here; method
835    /// generics are handled separately by the method turbofish.
836    fn bind_trait_function_generics(
837        &mut self,
838        func_id: FuncId,
839        trait_generics: Vec<Type>,
840        bindings: &mut TypeBindings,
841    ) {
842        if trait_generics.is_empty() {
843            return;
844        }
845
846        let func_meta = self.function_meta(func_id);
847        let impl_generics = vecmap(func_meta.impl_generics(), |g| g.type_var.clone());
848
849        // `impl_generics` may include `Self` and associated type generics after the trait's
850        // declared generics. The turbofish only provides types for the declared generics, which
851        // are always the first elements, so slice to match.
852        if trait_generics.len() <= impl_generics.len() {
853            let impl_generics = &impl_generics[..trait_generics.len()];
854            for (trait_generic, type_var) in trait_generics.into_iter().zip_eq(impl_generics) {
855                bindings.insert(type_var.id(), (type_var.clone(), type_var.kind(), trait_generic));
856            }
857        } else {
858            unreachable!(
859                "trait_generics.len() ({}) > impl_generics.len() ({}): \
860                turbofish resolution should normalize the count",
861                trait_generics.len(),
862                impl_generics.len()
863            );
864        }
865    }

867    /// Bind an enum's generics for a call to one of its variant constructors, e.g. the `<u32>` in
868    /// `Foo::<u32>::Eggs`. A constructor has no impl and no self type; the enum's generics are its
869    /// `direct_generics`, bound directly from `type_generics` (which was resolved against the
870    /// enum's generics, so its length matches).
871    fn bind_enum_variant_constructor_generics(
872        &mut self,
873        func_id: FuncId,
874        type_generics: Vec<Type>,
875        bindings: &mut TypeBindings,
876    ) {
877        let func_meta = self.function_meta(func_id);
878        let direct_generics = vecmap(&func_meta.direct_generics, |g| g.type_var.clone());
879        for (type_generic, type_var) in type_generics.into_iter().zip_eq(direct_generics) {
880            bindings.insert(type_var.id(), (type_var.clone(), type_var.kind(), type_generic));
881        }
882    }

884    /// Determine how the type-level generics of a resolved path item — everything in the path
885    /// before the final method — should be bound. For example, given `foo::Bar::<i32>::baz`, this
886    /// resolves the `Bar::<i32>` portion into the generics (`i32`) that the `baz` call must bind.
887    ///
888    /// The returned [`ItemTypeBinding`] describes which binding strategy applies; see its variants.
889    #[tracing::instrument(level = "trace", skip_all)]
890    fn resolve_item_type_binding(&mut self, item: PathResolutionItem) -> ItemTypeBinding {
891        let mut errors = Vec::new();
892        let result = match item {
893            PathResolutionItem::Method(struct_id, Some(generics), func_id) => {
894                let generics = self.resolve_struct_id_turbofish_generics(
895                    struct_id,
896                    Some(generics),
897                    &mut errors,
898                );
899                // An enum variant constructor (e.g. `Foo::<u32>::Eggs`) has no impl and no self
900                // type; its enum generics are bound directly rather than by unifying a self type.
901                if self.function_meta(func_id).enum_variant_index.is_some() {
902                    ItemTypeBinding::EnumVariantConstructor(generics)
903                } else {
904                    // Reconstruct the concrete self type `S<generics>` to unify against the impl.
905                    let data_type = self.interner.get_type(struct_id);
906                    ItemTypeBinding::ConcreteSelfType(Type::DataType(data_type, generics))
907                }
908            }
909            PathResolutionItem::SelfMethod(_) => match &self.self_type {
910                Some(self_type) => ItemTypeBinding::ConcreteSelfType(self_type.clone()),
911                None => ItemTypeBinding::None,
912            },
913            PathResolutionItem::TypeAliasFunction(type_alias_id, generics, _func_id) => {
914                let type_alias = self.interner.get_type_alias(type_alias_id);
915                let type_alias = type_alias.borrow();
916                let generics = self.resolve_type_alias_id_turbofish_generics(
917                    type_alias_id,
918                    generics,
919                    &mut errors,
920                );
921
922                // Expand the alias (recursively, through aliases-of-aliases) to its concrete
923                // underlying type, e.g. `type Pair = (u8, u32)` becomes `(u8, u32)`. We later
924                // unify this against the impl's self type to bind the impl's generics.
925                ItemTypeBinding::ConcreteSelfType(alias_concrete_type(&type_alias, &generics))
926            }
927            PathResolutionItem::TraitFunction(trait_id, Some(generics), _func_id) => {
928                let trait_ = self.interner.get_trait(trait_id);
929                let kinds = vecmap(&trait_.generics, |generic| generic.kind());
930                let trait_generics =
931                    vecmap(&kinds, |kind| self.interner.next_type_variable_with_kind(kind.clone()));
932
933                let generics = self.resolve_trait_turbofish_generics(
934                    &trait_.name.to_string(),
935                    kinds,
936                    trait_generics,
937                    Some(generics.generics),
938                    generics.location,
939                    &mut errors,
940                );
941                ItemTypeBinding::TraitGenerics(generics)
942            }
943            PathResolutionItem::TypeTraitFunction(self_type, _trait_id, _func_id) => {
944                ItemTypeBinding::SelfGeneric(self_type)
945            }
946            PathResolutionItem::PrimitiveFunction(primitive_type, turbofish, _func_id) => {
947                // Build the concrete primitive self type (e.g. `str<3>`) and unify it against the
948                // impl, just like any other concrete type.
949                let (typ, _has_generics) = self.instantiate_primitive_type_with_turbofish(
950                    primitive_type,
951                    turbofish,
952                    &mut errors,
953                );
954                ItemTypeBinding::ConcreteSelfType(typ)
955            }
956            PathResolutionItem::Method(_, None, _)
957            | PathResolutionItem::TraitFunction(_, None, _)
958            | PathResolutionItem::Module(..)
959            | PathResolutionItem::Type(..)
960            | PathResolutionItem::TypeAlias(..)
961            | PathResolutionItem::PrimitiveType(..)
962            | PathResolutionItem::Trait(..)
963            | PathResolutionItem::TraitAssociatedType(..)
964            | PathResolutionItem::Global(..)
965            | PathResolutionItem::EnumVariant(..)
966            | PathResolutionItem::ModuleFunction(..)
967            | PathResolutionItem::TraitConstant(..) => ItemTypeBinding::None,
968        };
969        self.push_errors(errors);
970        result
971    }

984    #[tracing::instrument(level = "trace", skip_all)]
985    fn elaborate_type_path_impl(
986        &mut self,

988        ident: &Ident,
989        turbofish: Option<GenericTypeArgs>,

1016    }

1044        self.elaborate_type_path_impl_inner(
1045            &typ,
1046            typ_location,
1047            ident_location,
1048            method,
1049            resolved_generics,
1050        )
1051    }

1053    /// Common implementation for type path impl variants.
1054    #[tracing::instrument(level = "trace", skip_all)]
1055    fn elaborate_type_path_impl_inner(
1056        &mut self,
1057        typ: &Type,
1058        _typ_location: Location,
1059        ident_location: Location,
1060        method: HirMethodReference,
1061        generics: Option<Vec<Type>>,
1062    ) -> (ExprId, Type) {
1063        let func_id = method
1064            .func_id(self.interner)
1065            .expect("Expected trait function to be a DefinitionKind::Function");
1066
1067        let id = self.interner.function_definition_id(func_id);
1068
1069        let impl_kind = match method {
1070            HirMethodReference::FuncId(_) => ImplKind::NotATraitMethod,
1071            HirMethodReference::TraitItemId(HirTraitMethodReference {
1072                definition,
1073                trait_id,
1074                trait_generics,
1075                assumed: _,
1076            }) => {
1077                let mut constraint =
1078                    self.interner.get_trait(trait_id).as_constraint(ident_location);
1079                constraint.trait_bound.trait_generics = trait_generics;
1080                ImplKind::TraitItem(TraitItem { definition, constraint, assumed: false })
1081            }
1082        };
1083
1084        let ident = HirIdent { location: ident_location, id, impl_kind };
1085        let id =
1086            self.intern_expr(HirExpression::Ident(ident.clone(), generics.clone()), ident_location);
1087
1088        // If the method has a self type (it's an impl or trait impl), bind `typ` to the instantiated self type.
1089        let (self_type_generics_count, function_typ, function_self_type) = self
1090            .with_function_meta(func_id, |meta| {
1091                (meta.impl_generics_count(), meta.typ.clone(), meta.self_type.clone())
1092            });
1093        let bindings = if self_type_generics_count > 0 {
1094            if let Type::Forall(type_vars, _) = &function_typ
1095                && let Some(self_type) = &function_self_type
1096            {
1097                // Only instantiate type vars corresponding to the `self` type, not to function direct generics
1098                let type_vars =
1099                    type_vars.iter().take(self_type_generics_count).cloned().collect::<Vec<_>>();
1100                let (self_type, instantiation_bindings) =
1101                    self_type.substitute_type_vars_with_fresh_type_vars(&type_vars, self.interner);
1102                let _ = typ.unify(&self_type);
1103                instantiation_bindings
1104            } else {
1105                TypeBindings::default()
1106            }
1107        } else {
1108            TypeBindings::default()
1109        };
1110
1111        let typ = self.type_check_variable_with_bindings(ident, &id, generics, bindings);
1112        let id = self.intern_expr_type(id, typ.clone());
1113
1114        (id, typ)
1115    }

1117    /// Given an [`HirIdent`], look up its definition, and:
1118    /// * mark it as referenced at the ident [Location] (LSP mode only)
1119    /// * mark the item currently being elaborated as a dependency of it
1120    /// * elaborate a global definition, if needed
1121    /// * add local identifiers to lambda captures
1122    #[tracing::instrument(level = "trace", skip_all)]
1123    pub(crate) fn handle_definition_id(&mut self, definition_id: DefinitionId, location: Location) {
1124        match self.interner.definition(definition_id).kind {
1125            DefinitionKind::Function(func_id) => {
1126                if let Some(current_item) = self.current_item {
1127                    self.interner.add_function_dependency(current_item, func_id);
1128                }
1129
1130                self.interner.add_function_reference(func_id, location);
1131            }
1132            DefinitionKind::Global(global_id) => {
1133                self.elaborate_global_if_unresolved(&global_id);
1134                if let Some(current_item) = self.current_item {
1135                    self.interner.add_global_dependency(current_item, global_id);
1136                }
1137
1138                let global = self.interner.get_global_definition(global_id);
1139                if global.comptime && !self.in_comptime_context() {
1140                    self.push_err(ResolverError::ComptimeGlobalInNonComptimeCode {
1141                        location,
1142                        name: global.name.clone(),
1143                    });
1144                }
1145
1146                self.interner.add_global_reference(global_id, location);
1147            }
1148            DefinitionKind::NumericGeneric(_, ref numeric_typ) => {
1149                // Initialize numeric generics to a polymorphic integer type in case
1150                // they're used in expressions. We must do this here since type_check_variable
1151                // does not check definition kinds and otherwise expects parameters to
1152                // already be typed.
1153                if self.interner.definition_type(definition_id) == Type::Error {
1154                    let type_var_kind = Kind::Numeric(numeric_typ.clone());
1155                    let typ = self.type_variable_with_kind(type_var_kind);
1156                    self.interner.push_definition_type(definition_id, typ);
1157                }
1158            }
1159            DefinitionKind::Local(_) => {
1160                // Handled separately in `handle_local_variable`
1161            }
1162            DefinitionKind::AssociatedConstant(..) => {
1163                // Nothing to do here
1164            }
1165        }
1166    }

1188    /// Perform the type checking of an interned expression and a corresponding identifier,
1189    /// returning the instantiated [Type].
1190    ///
1191    /// The instantiation bindings are pushed as required type variables on the current
1192    /// function context, to be checked at end-of-function. `push_required_type_variable`
1193    /// already skips this in a comptime context, where unbound generics on quoted typed
1194    /// expressions are expected.
1195    #[tracing::instrument(level = "trace", skip_all)]
1196    pub(crate) fn type_check_variable_with_bindings(
1197        &mut self,
1198        ident: HirIdent,
1199        expr_id: &PushedExpr<HasLocation>,
1200        generics: Option<Vec<Type>>,
1201        mut bindings: TypeBindings,
1202    ) -> Type {
1203        // Add type bindings from any constraints that were used.
1204        // We need to do this first since otherwise instantiating the type below
1205        // will replace each trait generic with a fresh type variable, rather than
1206        // the type used in the trait constraint (if it exists). See #4088.
1207        if let ImplKind::TraitItem(method) = &ident.impl_kind {
1208            self.bind_generics_from_trait_constraint(
1209                &method.constraint,
1210                method.assumed,
1211                &mut bindings,
1212            );
1213        }
1214
1215        // If a global variable hasn't been defined yet, then we are most likely dealing with a self-dependency-cycle.
1216        let definition = self.interner.definition(ident.id);
1217        // Some associated constants also have Global as Kind, and they are not defined when look them up here; want to restrict to global `let` statements.
1218        if self.in_comptime_context()
1219            && definition.kind.is_global()
1220            && self.interner.try_definition_type(ident.id).is_none()
1221        {
1222            self.push_err(ResolverError::DependencyCycle {
1223                location: ident.location,
1224                item: definition.name.clone(),
1225                cycle: "the variable definition type hasn't been resolved yet".to_string(),
1226            });
1227            return Type::Error;
1228        }
1229
1230        let func_id = match definition.kind {
1231            DefinitionKind::Function(func_id) => Some(func_id),
1232            _ => None,
1233        };
1234
1235        // If the variable is a function whose meta hasn't been resolved yet, resolve
1236        // it now. This handles forward references — a global's RHS may name a
1237        // function whose meta would otherwise only be drained at end-of-elaboration.
1238        if let Some(func_id) = func_id {
1239            let item_name = definition.name.clone();
1240            self.define_function_meta_if_undefined(func_id);
1241
1242            // If lazy resolution leaves the meta still unset, the function is currently
1243            // mid-resolution and we have a dependency cycle (e.g. `global F = f();`
1244            // combined with `fn f(_: [u8; F]) {}`).
1245            if self.interner.try_function_meta(&func_id).is_none() {
1246                self.push_err(ResolverError::DependencyCycle {
1247                    location: ident.location,
1248                    item: item_name,
1249                    cycle: "the function signature hasn't been resolved yet".to_string(),
1250                });
1251                return Type::Error;
1252            }
1253        }
1254
1255        // An identifiers type may be forall-quantified in the case of generic functions.
1256        // E.g. `fn foo<T>(t: T, field: Field) -> T` has type `forall T. fn(T, Field) -> T`.
1257        // We must instantiate identifiers at every call site to replace this T with a new type
1258        // variable to handle generic functions.
1259        let t = self.type_substitute_trait_as_type(&ident);
1260
1261        let direct_generic_ids = match func_id {
1262            Some(function) => vecmap(&self.function_meta(function).direct_generics, |generic| {
1263                generic.type_var.id()
1264            }),
1265            None => Vec::new(),
1266        };
1267
1268        let location = self.interner.expr_location(expr_id);
1269
1270        // This instantiates a trait's generics as well which need to be set
1271        // when the constraint below is later solved for when the function is
1272        // finished. How to link the two?
1273        let (typ, bindings) =
1274            self.instantiate(t, bindings, generics, &direct_generic_ids, location);
1275
1276        if let ImplKind::TraitItem(mut method) = ident.impl_kind {
1277            method.constraint.apply_bindings(&bindings);
1278            if method.assumed {
1279                let trait_generics = method.constraint.trait_bound.trait_generics.clone();
1280                let object_type = method.constraint.typ;
1281                let trait_impl = TraitImplKind::Assumed { object_type, trait_generics };
1282                self.interner.select_impl_for_expression(**expr_id, trait_impl);
1283            } else {
1284                // this constraint should lead to choosing a trait impl method
1285                self.push_trait_constraint(method.constraint, **expr_id, true);
1286            }
1287        }
1288
1289        // Push any trait constraints required by this definition to the context
1290        // to be checked later when the type of this variable is further constrained.
1291        //
1292        // This must be done before the above trait constraint in case the above one further
1293        // restricts types.
1294        //
1295        // For example, in this code:
1296        //
1297        // ```noir
1298        // trait One {}
1299        //
1300        // trait Two<O: One> {
1301        //     fn new() -> Self;
1302        // }
1303        //
1304        // fn foo<X: One, T: Two<X>>() {
1305        //     let _: T = Two::new();
1306        // }
1307        // ```
1308        //
1309        // when type-checking `Two::new` we'll have a return type `'2` which is constrained by `'2: Two<'1>`.
1310        // Then the definition for `new` has a constraint on it, `O: One`, which translates to `'1: One`.
1311        //
1312        // Because of the explicit type in the `let`, `'2` will be unified with `T`.
1313        // Then we must first verify the constraint `'2: Two<'1>`, which is now `T: Two<'1>`, to find
1314        // that the implementation is the assumed one `T: Two<X>` so that `'1` is bound to `X`.
1315        // Then we can successfully verify the constraint `'1: One` which now became `X: One` which holds
1316        // because of the assumed constraint.
1317        //
1318        // If we try to find a trait implementation for `'1` before finding one for `'2` we'll never find it.
1319        if let Some(function) = func_id {
1320            let function = self.function_meta(function);
1321            for mut constraint in function.all_trait_constraints().cloned().collect::<Vec<_>>() {
1322                constraint.apply_bindings(&bindings);
1323
1324                // This constraint shouldn't lead to choosing a trait impl method
1325                self.push_trait_constraint(constraint, **expr_id, false);
1326            }
1327        }
1328
1329        // Record required type variables in a predictable order to avoid nondeterminism in error messages.
1330        let required_type_variables = btree_map(bindings.values(), |(type_variable, _, typ)| {
1331            (type_variable.id(), typ.clone())
1332        });
1333
1334        for (type_variable_id, typ) in required_type_variables {
1335            self.push_required_type_variable(
1336                type_variable_id,
1337                typ,
1338                BindableTypeVariableKind::Ident(ident.id),
1339                ident.location,
1340            );
1341        }
1342
1343        self.interner.store_instantiation_bindings(**expr_id, bindings);
1344        typ
1345    }
1346
1347    /// If the type of the [`HirIdent`] is a function that returns an `impl Trait`,
1348    /// then it might need elaboration before it can be substituted to a [Type].
1349    /// Try to elaborate it now.
1350    ///
1351    /// Returns a type error if the callee cannot be resolved on a second try,
1352    /// which indicates a dependency cycle.
1353    #[tracing::instrument(level = "trace", skip_all)]
1354    fn type_substitute_trait_as_type(&mut self, ident: &HirIdent) -> Type {
1355        let func_id = match self.interner.id_type_substitute_trait_as_type(ident.id) {
1356            Ok(typ) => return typ,
1357            Err(func_id) => func_id,
1358        };
1359
1360        // Try to elaborate, so we get an expression for the body.
1361        self.elaborate_function(func_id);
1362
1363        // Now try again. If it's still not working, give up.
1364        match self.interner.id_type_substitute_trait_as_type(ident.id) {
1365            Ok(typ) => typ,
1366            Err(_) => {
1367                let def = self.interner.definition(ident.id);
1368                self.push_err(ResolverError::DependencyCycle {
1369                    location: ident.location,
1370                    item: def.name.clone(),
1371                    cycle: "'impl Trait' could not be resolved to the type of the function body"
1372                        .to_string(),
1373                });
1374                Type::Error
1375            }
1376        }
1377    }

1379    /// Instantiate a [Type] with the given [`TypeBindings`], returning the bindings potentially
1380    /// extended from any turbofish generics.
1381    ///
1382    /// If there are turbofish generics and their number matches the expectations of the function,
1383    /// those are used as well, otherwise they are ignored and an error is pushed.
1384    #[tracing::instrument(level = "trace", skip_all)]
1385    fn instantiate(
1386        &mut self,
1387        typ: Type,
1388        bindings: TypeBindings,
1389        turbofish_generics: Option<Vec<Type>>,
1390        direct_generic_ids: &[TypeVariableId],
1391        location: Location,
1392    ) -> (Type, TypeBindings) {
1393        match turbofish_generics {
1394            Some(turbofish_generics) => {
1395                let function_generic_count = direct_generic_ids.len();
1396                let forall_generic_count =
1397                    if let Type::Forall(generics, _) = &typ { generics.len() } else { 0 };
1398
1399                if turbofish_generics.len() != function_generic_count {
1400                    let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
1401                        expected_count: function_generic_count,
1402                        actual_count: turbofish_generics.len(),
1403                        location,
1404                    };
1405                    self.push_err(CompilationError::TypeError(type_check_err));
1406                    typ.instantiate_with_bindings(bindings, self.interner)
1407                } else if forall_generic_count < function_generic_count {
1408                    // The next branch asserts that the number of turbofish-bound typevars matches
1409                    // `direct_generic_ids`, but if the forall has fewer generics than the function
1410                    // (e.g. when the function's generics have duplicates that were filtered out)
1411                    // this can never hold. A duplicate error has already been reported elsewhere,
1412                    // so bail out silently.
1413                    self.push_err(TypeCheckError::expecting_other_error(
1414                        "forall has fewer generics than function",
1415                        location,
1416                    ));
1417                    (Type::Error, bindings)
1418                } else {
1419                    typ.instantiate_with_bindings_and_turbofish(
1420                        bindings,
1421                        turbofish_generics,
1422                        self.interner,
1423                        direct_generic_ids,
1424                    )
1425                }
1426            }
1427            None => typ.instantiate_with_bindings(bindings, self.interner),
1428        }
1429    }
1430}
