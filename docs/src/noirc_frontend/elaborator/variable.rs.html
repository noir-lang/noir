28/// The result of [`Elaborator::resolve_variable`].
29#[allow(clippy::large_enum_variant)]
30pub(crate) enum VariableResolution {
31    /// The path resolved to a variable or a definition.
32    Ident(HirIdent, Option<PathResolutionItem>),
33    /// The path resolved to a type alias that is numeric, infinitely recursive or one that errored.
34    TypeAlias(TypeAliasId),
35}

37impl Elaborator<'_> {
38    pub(super) fn elaborate_variable(&mut self, variable: Path) -> (ExprId, Type) {
39        let (id, typ, is_comptime_local, location) = self.elaborate_variable_inner(variable);
40
41        // Only check has_errors when we need to call the interpreter
42        // If this variable is a comptime local variable, use its current value as the final expression
43        if is_comptime_local {
44            let mut interpreter = self.setup_interpreter();
45            let value = interpreter.evaluate(id);
46            // If the value is an error it means the variable already had an error, so don't report it here again
47            // (the error will make no sense, it will say that a non-comptime variable was referenced at runtime
48            // but that's not true)
49            if value.is_ok() {
50                let (id, typ) = self.inline_comptime_value(value, location);
51                self.debug_comptime(location, |interner| id.to_display_ast(interner).kind);
52                (id, typ)
53            } else {
54                (id, typ)
55            }
56        } else {
57            (id, typ)
58        }
59    }

61    /// Helper function containing the elaboration logic for a variable.
62    /// Returns the expression ID, type, whether it's a comptime local, and location.
63    fn elaborate_variable_inner(&mut self, variable: Path) -> (ExprId, Type, bool, Location) {
64        let variable = self.validate_path(variable);
65        if let Some((expr_id, typ)) =
66            self.elaborate_variable_as_self_method_or_associated_constant(&variable)
67        {
68            return (expr_id, typ, false, variable.location);
69        }
70
71        let resolved_turbofish = variable.segments.last().unwrap().generics.clone();
72
73        let location = variable.location;
74        let variable_resolution = self.resolve_variable(variable);
75
76        let (hir_ident, item) = match variable_resolution {
77            Some(VariableResolution::TypeAlias(type_alias_id)) => {
78                // A type alias to a numeric generics is considered like a variable,
79                // but it is not a real variable so it does not resolve to a valid Identifier.
80                // In order to handle this, we retrieve the numeric generics expression that the type aliases to.
81                let type_alias = self.interner.get_type_alias(type_alias_id);
82                if let Some(expr) = &type_alias.borrow().numeric_expr {
83                    // Extract the declared numeric type from the type alias's kind.
84                    let declared_type = match type_alias.borrow().typ.kind() {
85                        Kind::Numeric(declared_type) => declared_type,
86                        _ => Box::new(Type::Error),
87                    };
88                    let declared_type = *declared_type;
89                    let expr_location = type_alias.borrow().location;
90                    let expr = UnresolvedTypeExpression::to_expression_kind(expr);
91                    let expr = Expression::new(expr, expr_location);
92                    let (id, typ) = self.elaborate_expression(expr);
93                    // Unify the expression's type with the declared type from the type alias
94                    // to ensure proper type checking.
95                    self.unify(&typ, &declared_type, || TypeCheckError::TypeMismatch {
96                        expected_typ: declared_type.to_string(),
97                        expr_typ: typ.to_string(),
98                        expr_location,
99                    });
100                    return (id, declared_type, false, location);
101                }
102                (None, None)
103            }
104            Some(VariableResolution::Ident(ident, item)) => (Some(ident), item),
105            None => (None, None),
106        };
107
108        let definition_id = hir_ident.as_ref().map(|ident| ident.id);
109
110        let (type_generics, self_generic) = if let Some(item) = item {
111            self.resolve_item_turbofish_and_self_type(item)
112        } else {
113            (Vec::new(), None)
114        };
115
116        let definition = definition_id.map(|id| self.interner.definition(id));
117        let is_comptime_local = !self.in_comptime_context()
118            && definition.is_some_and(DefinitionInfo::is_comptime_local);
119        let definition_kind = definition.as_ref().map(|definition| definition.kind.clone());
120
121        let mut bindings = TypeBindings::default();
122        let generics = if let Some(DefinitionKind::Function(func_id)) = &definition_kind {
123            // If there's a self type, bind it to the self type generic
124            if let Some(self_generic) = self_generic {
125                let func_generics = &self.interner.function_meta(func_id).all_generics;
126                let self_resolved_generic =
127                    func_generics.iter().find(|generic| generic.name.as_str() == SELF_TYPE_NAME);
128                if let Some(self_resolved_generic) = self_resolved_generic {
129                    let type_var = &self_resolved_generic.type_var;
130                    bindings
131                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), self_generic));
132                }
133            }
134
135            // If this is a function call on a type that has generics, we need to bind those generic types.
136            if !type_generics.is_empty() {
137                // `all_generics` will always have the enclosing type generics first, so we need to bind those
138                let func_generics = &self.interner.function_meta(func_id).all_generics;
139                for (type_generic, func_generic) in type_generics.into_iter().zip(func_generics) {
140                    let type_var = &func_generic.type_var;
141                    bindings
142                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), type_generic));
143                }
144            }
145
146            // Resolve any generics if the variable we have resolved is a function
147            // and if the turbofish operator was used.
148            self.resolve_function_turbofish_generics(func_id, resolved_turbofish, location)
149        } else {
150            if let Some(unused_resolved_turbofish) = resolved_turbofish {
151                let message = format!(
152                    "elaborate_variable_inner: unused resolved_turbofish: {unused_resolved_turbofish:?}"
153                );
154                self.push_err(TypeCheckError::ExpectingOtherError { message, location });
155            }
156
157            None
158        };
159
160        let (id, typ) = if let Some(hir_ident) = hir_ident {
161            let id = self
162                .intern_expr(HirExpression::Ident(hir_ident.clone(), generics.clone()), location);
163
164            // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
165            let push_required_type_variables = self.current_trait.is_none();
166            let typ = self.type_check_variable_with_bindings(
167                hir_ident,
168                &id,
169                generics,
170                bindings,
171                push_required_type_variables,
172            );
173            let id = self.intern_expr_type(id, typ.clone());
174            (id, typ)
175        } else {
176            let expr = HirExpression::Error;
177            let id = self.intern_expr(expr, location);
178            let typ = Type::Error;
179            let id = self.intern_expr_type(id, typ.clone());
180            (id, typ)
181        };
182
183        (id, typ, is_comptime_local, location)
184    }

186    /// Checks whether `variable` is `Self::method_name`, `Self::AssociatedConstant`, or
187    /// `Self::AssociatedType::method_name` when we are inside a trait impl and `Self`
188    /// resolves to a primitive type.
189    ///
190    /// In the first case we elaborate this as if it were a [TypePath]
191    /// (for example, if `Self` is `u32` then we consider this the same as `u32::method_name`).
192    /// A regular path lookup won't work here for the same reason [TypePath] exists.
193    ///
194    /// In the second case we solve the associated constant by looking up its value, later
195    /// turning it into a literal.
196    ///
197    /// In the third case, we resolve the associated type first, then elaborate the method
198    /// call on that resolved type.
199    fn elaborate_variable_as_self_method_or_associated_constant(
200        &mut self,
201        variable: &TypedPath,
202    ) -> Option<(ExprId, Type)> {
203        // We need at least 2 segments and the first must be `Self`
204        if variable.segments.len() < 2 || !variable.segments[0].ident.is_self_type_name() {
205            return None;
206        }
207
208        let location = variable.location;
209        let name = variable.segments[1].ident.as_str();
210        let self_type = self.self_type.as_ref()?;
211        let trait_impl_id = &self.current_trait_impl?;
212
213        // Check for `Self::AssociatedType::method_name` (exactly 3 segments).
214        // Longer paths like `Self::AssocType::foo::bar` are not valid since associated types
215        // resolve to concrete types and you cannot chain further path segments after a method name.
216        if variable.segments.len() == 3 {
217            // Try to resolve the second segment as an associated type
218            if let Some(assoc_type) =
219                self.interner.find_associated_type_for_impl(*trait_impl_id, name).cloned()
220            {
221                let method_ident = variable.segments[2].ident.clone();
222                let typ_location = variable.segments[1].location;
223                // Extract already-resolved turbofish generics from the path segment
224                let resolved_generics = variable.segments[2].generics.as_ref().map(|generics| {
225                    generics.iter().map(|located| located.contents.clone()).collect()
226                });
227                return Some(self.elaborate_type_path_impl_with_resolved_generics(
228                    assoc_type,
229                    method_ident,
230                    resolved_generics,
231                    typ_location,
232                ));
233            }
234            // If it's not an associated type, fall through to let regular path resolution handle it
235            return None;
236        }
237
238        // Check the `Self::AssociatedConstant` case when inside a trait impl (2 segments)
239        if let Some((definition_id, numeric_type)) =
240            self.interner.get_trait_impl_associated_constant(*trait_impl_id, name).cloned()
241        {
242            let hir_ident = HirIdent::non_trait_method(definition_id, location);
243            let hir_expr = HirExpression::Ident(hir_ident, None);
244            let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
245            return Some((id, numeric_type));
246        }
247
248        // Check if the constant exists in the trait definition (even if impl is missing it).
249        // This prevents spurious "Could not resolve" errors inside trait methods when the impl is missing the constant,
250        // since the "missing associated constant" error is reported elsewhere.
251        if let Some(trait_impl) = self.interner.try_get_trait_implementation(*trait_impl_id) {
252            let trait_id = trait_impl.borrow().trait_id;
253            let trait_ = self.interner.get_trait(trait_id);
254            if let Some(definition_id) = trait_.associated_constant_ids.get(name).copied() {
255                let numeric_type = self.interner.definition_type(definition_id);
256                let hir_ident = HirIdent::non_trait_method(definition_id, location);
257                let hir_expr = HirExpression::Ident(hir_ident, None);
258                let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
259                return Some((id, numeric_type));
260            }
261        }
262
263        // Check the `Self::method_name` case when `Self` is a primitive type (2 segments)
264        if matches!(self.self_type, Some(Type::DataType(..))) {
265            return None;
266        }
267
268        let ident = variable.segments[1].ident.clone();
269        let typ_location = variable.segments[0].location;
270        Some(self.elaborate_type_path_impl(self_type.clone(), ident, None, typ_location))
271    }

273    /// Resolve a [TypedPath] to a [HirIdent] of either some trait method, or a local or global variable.
274    fn resolve_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
275        if let Some(trait_path_resolution) = self.resolve_trait_generic_path(&path) {
276            self.push_errors(trait_path_resolution.errors);
277
278            return match trait_path_resolution.method {
279                TraitPathResolutionMethod::NotATraitMethod(func_id) => {
280                    let ident = HirIdent {
281                        location: path.location,
282                        id: self.interner.function_definition_id(func_id),
283                        impl_kind: ImplKind::NotATraitMethod,
284                    };
285                    Some(VariableResolution::Ident(ident, trait_path_resolution.item))
286                }
287
288                TraitPathResolutionMethod::TraitItem(item) => {
289                    let ident = HirIdent {
290                        location: path.location,
291                        id: item.definition,
292                        impl_kind: ImplKind::TraitItem(item),
293                    };
294                    Some(VariableResolution::Ident(ident, trait_path_resolution.item))
295                }
296
297                TraitPathResolutionMethod::MultipleTraitsInScope => {
298                    // An error has already been pushed, don't return an identifier
299                    None
300                }
301            };
302        }
303
304        // The location of variables or definitions we register (for LSP) must be that of the
305        // path's last segment, as intermediate segments solve to other definitions.
306        let location = path.last_ident().location();
307
308        // If the Path is being used as an Expression, then it is referring to a global from a separate module
309        // Otherwise, then it is referring to an Identifier
310        // This lookup allows support of such statements: let x = foo::bar::SOME_GLOBAL + 10;
311        // If the expression is a singular indent, we search the resolver's current scope as normal.
312        let ident_from_path = self.get_ident_from_path(path);
313        ident_from_path.map(|ident_from_path| match ident_from_path {
314            IdentFromPath::Variable(variable) => {
315                self.handle_local_variable(&variable);
316                let hir_ident = HirIdent::non_trait_method(variable.ident.id, location);
317                VariableResolution::Ident(hir_ident, None)
318            }
319            IdentFromPath::Definition { id, item } => {
320                self.handle_definition_id(id, location);
321                let hir_ident = HirIdent::non_trait_method(id, location);
322                VariableResolution::Ident(hir_ident, Some(item))
323            }
324            IdentFromPath::TypeAlias(type_alias_id) => VariableResolution::TypeAlias(type_alias_id),
325        })
326    }

328    /// Solve any generics that are part of the path before the function, for example:
329    ///
330    /// ```noir
331    /// foo::Bar::<i32>::baz
332    /// ```
333    /// Solve `<i32>` above
334    fn resolve_item_turbofish_and_self_type(
335        &mut self,
336        item: PathResolutionItem,
337    ) -> (Vec<Type>, Option<Type>) {
338        match item {
339            PathResolutionItem::Method(struct_id, Some(generics), _func_id) => {
340                let generics = self.resolve_struct_id_turbofish_generics(struct_id, Some(generics));
341                (generics, None)
342            }
343            PathResolutionItem::SelfMethod(_) => {
344                let generics = if let Some(Type::DataType(_, generics)) = &self.self_type {
345                    generics.clone()
346                } else {
347                    Vec::new()
348                };
349                (generics, None)
350            }
351            PathResolutionItem::TypeAliasFunction(type_alias_id, generics, _func_id) => {
352                let type_alias = self.interner.get_type_alias(type_alias_id);
353                let type_alias = type_alias.borrow();
354                let generics =
355                    self.resolve_type_alias_id_turbofish_generics(type_alias_id, generics);
356
357                // Now instantiate the underlying struct or alias with those generics, the struct might
358                // have more generics than those in the alias, like in this example:
359                //
360                // type Alias<T> = Struct<T, i32>;
361                let generics = get_type_alias_generics(&type_alias, &generics);
362                (generics, None)
363            }
364            PathResolutionItem::TraitFunction(trait_id, Some(generics), _func_id) => {
365                let trait_ = self.interner.get_trait(trait_id);
366                let kinds = vecmap(&trait_.generics, |generic| generic.kind());
367                let trait_generics =
368                    vecmap(&kinds, |kind| self.interner.next_type_variable_with_kind(kind.clone()));
369
370                let generics = self.resolve_trait_turbofish_generics(
371                    &trait_.name.to_string(),
372                    kinds,
373                    trait_generics,
374                    Some(generics.generics),
375                    generics.location,
376                );
377                (generics, None)
378            }
379            PathResolutionItem::TypeTraitFunction(self_type, _trait_id, _func_id) => {
380                (Vec::new(), Some(self_type))
381            }
382            PathResolutionItem::PrimitiveFunction(primitive_type, turbofish, _func_id) => {
383                let (typ, has_generics) =
384                    self.instantiate_primitive_type_with_turbofish(primitive_type, turbofish);
385                let generics = if has_generics {
386                    match typ {
387                        Type::String(length) => vec![*length],
388                        Type::FmtString(length, element) => vec![*length, *element],
389                        _ => {
390                            unreachable!("ICE: Primitive type has been specified to have generics")
391                        }
392                    }
393                } else {
394                    Vec::new()
395                };
396                (generics, None)
397            }
398            PathResolutionItem::Method(_, None, _)
399            | PathResolutionItem::TraitFunction(_, None, _)
400            | PathResolutionItem::Module(..)
401            | PathResolutionItem::Type(..)
402            | PathResolutionItem::TypeAlias(..)
403            | PathResolutionItem::PrimitiveType(..)
404            | PathResolutionItem::Trait(..)
405            | PathResolutionItem::TraitAssociatedType(..)
406            | PathResolutionItem::Global(..)
407            | PathResolutionItem::ModuleFunction(..)
408            | PathResolutionItem::TraitConstant(..) => (Vec::new(), None),
409        }
410    }

412    /// Elaborates a type path used in an expression, e.g. `Type::method::<Args>`
413    pub(super) fn elaborate_type_path(&mut self, path: TypePath) -> (ExprId, Type) {
414        let typ_location = path.typ.location;
415        let turbofish = path.turbofish;
416        let wildcard_allowed = WildcardAllowed::Yes;
417        let typ = self.use_type(path.typ, wildcard_allowed);
418        self.elaborate_type_path_impl(typ, path.item, turbofish, typ_location)
419    }

421    /// Variant of [Self::elaborate_type_path_impl_inner] that accepts unresolved generics.
422    fn elaborate_type_path_impl(
423        &mut self,
424        typ: Type,
425        ident: Ident,
426        turbofish: Option<GenericTypeArgs>,
427        typ_location: Location,
428    ) -> (ExprId, Type) {
429        let ident_location = ident.location();
430        let check_self_param = false;
431
432        self.interner.push_type_ref_location(&typ, typ_location);
433
434        let Some(method) = self.lookup_method(
435            &typ,
436            ident.as_str(),
437            ident_location,
438            typ_location,
439            check_self_param,
440        ) else {
441            let error = Expression::new(ExpressionKind::Error, ident_location);
442            return self.elaborate_expression(error);
443        };
444
445        let func_id = method
446            .func_id(self.interner)
447            .expect("Expected trait function to be a DefinitionKind::Function");
448
449        let generics =
450            turbofish.map(|turbofish| self.use_type_args(turbofish, func_id, ident_location).0);
451
452        self.elaborate_type_path_impl_inner(typ_location, ident_location, method, generics)
453    }

455    /// Variant of [Self::elaborate_type_path_impl_inner] that accepts already resolved generics.
456    /// Used when the turbofish generics have already been resolved.
457    fn elaborate_type_path_impl_with_resolved_generics(
458        &mut self,
459        typ: Type,
460        ident: Ident,
461        resolved_generics: Option<Vec<Type>>,
462        typ_location: Location,
463    ) -> (ExprId, Type) {
464        let ident_location = ident.location();
465        let check_self_param = false;
466
467        self.interner.push_type_ref_location(&typ, typ_location);
468
469        let Some(method) = self.lookup_method(
470            &typ,
471            ident.as_str(),
472            ident_location,
473            typ_location,
474            check_self_param,
475        ) else {
476            let error = Expression::new(ExpressionKind::Error, ident_location);
477            return self.elaborate_expression(error);
478        };
479
480        self.elaborate_type_path_impl_inner(typ_location, ident_location, method, resolved_generics)
481    }

483    /// Common implementation for type path impl variants.
484    fn elaborate_type_path_impl_inner(
485        &mut self,
486        _typ_location: Location,
487        ident_location: Location,
488        method: HirMethodReference,
489        generics: Option<Vec<Type>>,
490    ) -> (ExprId, Type) {
491        let func_id = method
492            .func_id(self.interner)
493            .expect("Expected trait function to be a DefinitionKind::Function");
494
495        let id = self.interner.function_definition_id(func_id);
496
497        let impl_kind = match method {
498            HirMethodReference::FuncId(_) => ImplKind::NotATraitMethod,
499            HirMethodReference::TraitItemId(HirTraitMethodReference {
500                definition,
501                trait_id,
502                trait_generics,
503                assumed: _,
504            }) => {
505                let mut constraint =
506                    self.interner.get_trait(trait_id).as_constraint(ident_location);
507                constraint.trait_bound.trait_generics = trait_generics;
508                ImplKind::TraitItem(TraitItem { definition, constraint, assumed: false })
509            }
510        };
511
512        let ident = HirIdent { location: ident_location, id, impl_kind };
513        let id =
514            self.intern_expr(HirExpression::Ident(ident.clone(), generics.clone()), ident_location);
515
516        let typ = self.type_check_variable(ident, &id, generics);
517        let id = self.intern_expr_type(id, typ.clone());
518
519        (id, typ)
520    }

522    /// Given an [HirIdent], look up its definition, and:
523    /// * mark it as referenced at the ident [Location] (LSP mode only)
524    /// * mark the item currently being elaborated as a dependency of it
525    /// * elaborate a global definition, if needed
526    /// * add local identifiers to lambda captures
527    pub(crate) fn handle_definition_id(&mut self, definition_id: DefinitionId, location: Location) {
528        match self.interner.definition(definition_id).kind {
529            DefinitionKind::Function(func_id) => {
530                if let Some(current_item) = self.current_item {
531                    self.interner.add_function_dependency(current_item, func_id);
532                }
533
534                self.interner.add_function_reference(func_id, location);
535            }
536            DefinitionKind::Global(global_id) => {
537                self.elaborate_global_if_unresolved(&global_id);
538                if let Some(current_item) = self.current_item {
539                    self.interner.add_global_dependency(current_item, global_id);
540                }
541
542                let global = self.interner.get_global_definition(global_id);
543                if global.comptime && !self.in_comptime_context() {
544                    self.push_err(ResolverError::ComptimeGlobalInNonComptimeCode {
545                        location,
546                        name: global.name.clone(),
547                    });
548                }
549
550                self.interner.add_global_reference(global_id, location);
551            }
552            DefinitionKind::NumericGeneric(_, ref numeric_typ) => {
553                // Initialize numeric generics to a polymorphic integer type in case
554                // they're used in expressions. We must do this here since type_check_variable
555                // does not check definition kinds and otherwise expects parameters to
556                // already be typed.
557                if self.interner.definition_type(definition_id) == Type::Error {
558                    let type_var_kind = Kind::Numeric(numeric_typ.clone());
559                    let typ = self.type_variable_with_kind(type_var_kind);
560                    self.interner.push_definition_type(definition_id, typ);
561                }
562            }
563            DefinitionKind::Local(_) => {
564                // Handled separately in `handle_local_variable`
565            }
566            DefinitionKind::AssociatedConstant(..) => {
567                // Nothing to do here
568            }
569        }
570    }

574        let hir_ident = &variable.ident;

579    /// and a corresponding identifier, returning the instantiated [Type].
580    pub(crate) fn type_check_variable(
581        &mut self,
582        ident: HirIdent,
583        expr_id: &PushedExpr<HasLocation>,
584        generics: Option<Vec<Type>>,
585    ) -> Type {
586        let bindings = TypeBindings::default();
587        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
588        let push_required_type_variables = self.current_trait.is_none();

590            ident,
591            expr_id,
592            generics,
593            bindings,
594            push_required_type_variables,
595        )
596    }

598    /// Perform the type checking of an interned expression and a corresponding identifier,
599    /// returning the instantiated [Type].
600    ///
601    /// If `push_required_type_variables`, the bindings are added to the function context,
602    /// to be checked before it's finished.
603    pub(crate) fn type_check_variable_with_bindings(
604        &mut self,
605        ident: HirIdent,
606        expr_id: &PushedExpr<HasLocation>,
607        generics: Option<Vec<Type>>,
608        mut bindings: TypeBindings,
609        push_required_type_variables: bool,
610    ) -> Type {
611        // Add type bindings from any constraints that were used.
612        // We need to do this first since otherwise instantiating the type below
613        // will replace each trait generic with a fresh type variable, rather than
614        // the type used in the trait constraint (if it exists). See #4088.
615        if let ImplKind::TraitItem(method) = &ident.impl_kind {
616            self.bind_generics_from_trait_constraint(
617                &method.constraint,
618                method.assumed,
619                &mut bindings,
620            );
621        }
622
623        // An identifiers type may be forall-quantified in the case of generic functions.
624        // E.g. `fn foo<T>(t: T, field: Field) -> T` has type `forall T. fn(T, Field) -> T`.
625        // We must instantiate identifiers at every call site to replace this T with a new type
626        // variable to handle generic functions.
627        let t = self.type_substitute_trait_as_type(&ident);
628
629        let definition = self.interner.definition(ident.id);
630        let function_generic_count = match definition.kind {
631            DefinitionKind::Function(function) => {
632                self.interner.function_modifiers(&function).generic_count
633            }
634            _ => 0,
635        };
636
637        let location = self.interner.expr_location(expr_id);
638
639        // This instantiates a trait's generics as well which need to be set
640        // when the constraint below is later solved for when the function is
641        // finished. How to link the two?
642        let (typ, bindings) =
643            self.instantiate(t, bindings, generics, function_generic_count, location);
644
645        if let ImplKind::TraitItem(mut method) = ident.impl_kind {
646            method.constraint.apply_bindings(&bindings);
647            if method.assumed {
648                let trait_generics = method.constraint.trait_bound.trait_generics.clone();
649                let object_type = method.constraint.typ;
650                let trait_impl = TraitImplKind::Assumed { object_type, trait_generics };
651                self.interner.select_impl_for_expression(**expr_id, trait_impl);
652            } else {
653                // this constraint should lead to choosing a trait impl method
654                self.push_trait_constraint(method.constraint, **expr_id, true);
655            }
656        }
657
658        // Push any trait constraints required by this definition to the context
659        // to be checked later when the type of this variable is further constrained.
660        //
661        // This must be done before the above trait constraint in case the above one further
662        // restricts types.
663        //
664        // For example, in this code:
665        //
666        // ```noir
667        // trait One {}
668        //
669        // trait Two<O: One> {
670        //     fn new() -> Self;
671        // }
672        //
673        // fn foo<X: One, T: Two<X>>() {
674        //     let _: T = Two::new();
675        // }
676        // ```
677        //
678        // when type-checking `Two::new` we'll have a return type `'2` which is constrained by `'2: Two<'1>`.
679        // Then the definition for `new` has a constraint on it, `O: One`, which translates to `'1: One`.
680        //
681        // Because of the explicit type in the `let`, `'2` will be unified with `T`.
682        // Then we must first verify the constraint `'2: Two<'1>`, which is now `T: Two<'1>`, to find
683        // that the implementation is the assumed one `T: Two<X>` so that `'1` is bound to `X`.
684        // Then we can successfully verify the constraint `'1: One` which now became `X: One` which holds
685        // because of the assumed constraint.
686        //
687        // If we try to find a trait implementation for `'1` before finding one for `'2` we'll never find it.
688        let definition = self.interner.definition(ident.id);
689        if let DefinitionKind::Function(function) = definition.kind {
690            let function = self.interner.function_meta(&function);
691            for mut constraint in function.all_trait_constraints().cloned().collect::<Vec<_>>() {
692                constraint.apply_bindings(&bindings);
693
694                // This constraint shouldn't lead to choosing a trait impl method
695                self.push_trait_constraint(constraint, **expr_id, false);
696            }
697        }
698
699        if push_required_type_variables {
700            for (type_variable, _kind, typ) in bindings.values() {
701                self.push_required_type_variable(
702                    type_variable.id(),
703                    typ.clone(),
704                    BindableTypeVariableKind::Ident(ident.id),
705                    ident.location,
706                );
707            }
708        }
709
710        self.interner.store_instantiation_bindings(**expr_id, bindings);
711        typ
712    }
713
714    /// If the type of the [HirIdent] is a function that returns an `impl Trait`,
715    /// then it might need elaboration before it can be substituted to a [Type].
716    /// Try to elaborate it now.
717    ///
718    /// Returns a type error if the callee cannot be resolved on a second try,
719    /// which indicates a dependency cycle.
720    fn type_substitute_trait_as_type(&mut self, ident: &HirIdent) -> Type {
721        let func_id = match self.interner.id_type_substitute_trait_as_type(ident.id) {
722            Ok(typ) => return typ,
723            Err(func_id) => func_id,
724        };
725
726        // Try to elaborate, so we get an expression for the body.
727        self.elaborate_function(func_id);
728
729        // Now try again. If it's still not working, give up.
730        match self.interner.id_type_substitute_trait_as_type(ident.id) {
731            Ok(typ) => typ,
732            Err(_) => {
733                let def = self.interner.definition(ident.id);
734                self.push_err(ResolverError::DependencyCycle {
735                    location: ident.location,
736                    item: def.name.clone(),
737                    cycle: "'impl Trait' could not be resolved to the type of the function body"
738                        .to_string(),
739                });
740                Type::Error
741            }
742        }
743    }

745    /// Instantiate a [Type] with the given [TypeBindings], returning the bindings potentially
746    /// extended from any turbofish generics.
747    ///
748    /// If there are turbofish generics and their number matches the expectations of the function,
749    /// those are used as well, otherwise they are ignored and an error is pushed.
750    fn instantiate(
751        &mut self,
752        typ: Type,
753        bindings: TypeBindings,
754        turbofish_generics: Option<Vec<Type>>,
755        function_generic_count: usize,
756        location: Location,
757    ) -> (Type, TypeBindings) {
758        match turbofish_generics {
759            Some(turbofish_generics) => {
760                if turbofish_generics.len() != function_generic_count {
761                    let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
762                        expected_count: function_generic_count,
763                        actual_count: turbofish_generics.len(),
764                        location,
765                    };
766                    self.push_err(CompilationError::TypeError(type_check_err));
767                    typ.instantiate_with_bindings(bindings, self.interner)
768                } else {
769                    // Fetch the count of any implicit generics on the function, such as
770                    // for a method within a generic impl.
771                    let implicit_generic_count = match &typ {
772                        Type::Forall(generics, _) => generics.len() - function_generic_count,
773                        _ => 0,
774                    };
775                    typ.instantiate_with_bindings_and_turbofish(
776                        bindings,
777                        turbofish_generics,
778                        self.interner,
779                        implicit_generic_count,
780                    )
781                }
782            }
783            None => typ.instantiate_with_bindings(bindings, self.interner),
784        }
785    }
786}

788/// Bind the generics of the [Type] aliased by the [TypeAlias] to a list of generic arguments,
789/// recursively expanding the generics aliased aliases, finally returning the generics of the
790/// innermost aliased struct.
791///
792/// Panics if it encounters a type other than alias or struct.
793fn get_type_alias_generics(type_alias: &TypeAlias, generics: &[Type]) -> Vec<Type> {
794    let typ = type_alias.get_type(generics);
795    match typ {
796        Type::DataType(_, generics) => generics,
797        Type::Alias(type_alias, generics) => {
798            get_type_alias_generics(&type_alias.borrow(), &generics)
799        }
800        _ => panic!("Expected type alias to point to struct or alias"),
801    }
