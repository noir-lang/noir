26    pub(super) fn elaborate_variable(&mut self, variable: Path) -> (ExprId, Type) {
27        let (id, typ, is_comptime_local, location) = self.elaborate_variable_inner(variable);
28
29        // Only check has_errors when we need to call the interpreter
30        // If this variable is a comptime local variable, use its current value as the final expression
31        if is_comptime_local {
32            let mut interpreter = self.setup_interpreter();
33            let value = interpreter.evaluate(id);
34            // If the value is an error it means the variable already had an error, so don't report it here again
35            // (the error will make no sense, it will say that a non-comptime variable was referenced at runtime
36            // but that's not true)
37            if value.is_ok() {
38                let (id, typ) = self.inline_comptime_value(value, location);
39                self.debug_comptime(location, |interner| id.to_display_ast(interner).kind);
40                (id, typ)
41            } else {
42                (id, typ)
43            }
44        } else {
45            (id, typ)
46        }
47    }

49    /// Helper function containing the elaboration logic for a variable.
50    /// Returns the expression ID, type, whether it's a comptime local, and location.
51    fn elaborate_variable_inner(&mut self, variable: Path) -> (ExprId, Type, bool, Location) {
52        let variable = self.validate_path(variable);
53        if let Some((expr_id, typ)) =
54            self.elaborate_variable_as_self_method_or_associated_constant(&variable)
55        {
56            return (expr_id, typ, false, variable.location);
57        }
58
59        let resolved_turbofish = variable.segments.last().unwrap().generics.clone();
60
61        let location = variable.location;
62        let (expr, item) = self.resolve_variable(variable);
63        let definition_id = expr.id;
64
65        if let Some(PathResolutionItem::TypeAlias(alias)) = item {
66            // A type alias to a numeric generics is considered like a variable,
67            // but it is not a real variable so it does not resolve to a valid Identifier.
68            // In order to handle this, we retrieve the numeric generics expression that the type aliases to.
69            let type_alias = self.interner.get_type_alias(alias);
70            if let Some(expr) = &type_alias.borrow().numeric_expr {
71                // Extract the declared numeric type from the type alias's kind.
72                let declared_type = match type_alias.borrow().typ.kind() {
73                    Kind::Numeric(declared_type) => declared_type,
74                    _ => Box::new(Type::Error),
75                };
76                let declared_type = *declared_type;
77                let expr_location = type_alias.borrow().location;
78                let expr = UnresolvedTypeExpression::to_expression_kind(expr);
79                let expr = Expression::new(expr, expr_location);
80                let (id, typ) = self.elaborate_expression(expr);
81                // Unify the expression's type with the declared type from the type alias
82                // to ensure proper type checking.
83                self.unify(&typ, &declared_type, || TypeCheckError::TypeMismatch {
84                    expected_typ: declared_type.to_string(),
85                    expr_typ: typ.to_string(),
86                    expr_location,
87                });
88                return (id, declared_type, false, location);
89            }
90        }
91
92        let (type_generics, self_generic) = if let Some(item) = item {
93            self.resolve_item_turbofish_and_self_type(item)
94        } else {
95            (Vec::new(), None)
96        };
97
98        let definition = self.interner.try_definition(definition_id);
99        let is_comptime_local = !self.in_comptime_context()
100            && definition.is_some_and(DefinitionInfo::is_comptime_local);
101        let definition_kind = definition.as_ref().map(|definition| definition.kind.clone());
102
103        let mut bindings = TypeBindings::default();
104        let generics = if let Some(DefinitionKind::Function(func_id)) = &definition_kind {
105            // If there's a self type, bind it to the self type generic
106            if let Some(self_generic) = self_generic {
107                let func_generics = &self.interner.function_meta(func_id).all_generics;
108                let self_resolved_generic =
109                    func_generics.iter().find(|generic| generic.name.as_str() == SELF_TYPE_NAME);
110                if let Some(self_resolved_generic) = self_resolved_generic {
111                    let type_var = &self_resolved_generic.type_var;
112                    bindings
113                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), self_generic));
114                }
115            }
116
117            // If this is a function call on a type that has generics, we need to bind those generic types.
118            if !type_generics.is_empty() {
119                // `all_generics` will always have the enclosing type generics first, so we need to bind those
120                let func_generics = &self.interner.function_meta(func_id).all_generics;
121                for (type_generic, func_generic) in type_generics.into_iter().zip(func_generics) {
122                    let type_var = &func_generic.type_var;
123                    bindings
124                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), type_generic));
125                }
126            }
127
128            // Resolve any generics if the variable we have resolved is a function
129            // and if the turbofish operator was used.
130            self.resolve_function_turbofish_generics(func_id, resolved_turbofish, location)
131        } else {
132            if let Some(unused_resolved_turbofish) = resolved_turbofish {
133                let message = format!(
134                    "elaborate_variable_inner: unused resolved_turbofish: {unused_resolved_turbofish:?}"
135                );
136                self.push_err(TypeCheckError::ExpectingOtherError { message, location });
137            }
138
139            None
140        };
141
142        let id = self.intern_expr(HirExpression::Ident(expr.clone(), generics.clone()), location);
143
144        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
145        let push_required_type_variables = self.current_trait.is_none();
146        let typ = self.type_check_variable_with_bindings(
147            expr,
148            &id,
149            generics,
150            bindings,
151            push_required_type_variables,
152        );
153        let id = self.intern_expr_type(id, typ.clone());
154
155        (id, typ, is_comptime_local, location)
156    }

158    /// Checks whether `variable` is `Self::method_name`, `Self::AssociatedConstant`, or
159    /// `Self::AssociatedType::method_name` when we are inside a trait impl and `Self`
160    /// resolves to a primitive type.
161    ///
162    /// In the first case we elaborate this as if it were a [TypePath]
163    /// (for example, if `Self` is `u32` then we consider this the same as `u32::method_name`).
164    /// A regular path lookup won't work here for the same reason [TypePath] exists.
165    ///
166    /// In the second case we solve the associated constant by looking up its value, later
167    /// turning it into a literal.
168    ///
169    /// In the third case, we resolve the associated type first, then elaborate the method
170    /// call on that resolved type.
171    fn elaborate_variable_as_self_method_or_associated_constant(
172        &mut self,
173        variable: &TypedPath,
174    ) -> Option<(ExprId, Type)> {
175        // We need at least 2 segments and the first must be `Self`
176        if variable.segments.len() < 2 || !variable.segments[0].ident.is_self_type_name() {
177            return None;
178        }
179
180        let location = variable.location;
181        let name = variable.segments[1].ident.as_str();
182        let self_type = self.self_type.as_ref()?;
183        let trait_impl_id = &self.current_trait_impl?;
184
185        // Check for `Self::AssociatedType::method_name` (exactly 3 segments).
186        // Longer paths like `Self::AssocType::foo::bar` are not valid since associated types
187        // resolve to concrete types and you cannot chain further path segments after a method name.
188        if variable.segments.len() == 3 {
189            // Try to resolve the second segment as an associated type
190            if let Some(assoc_type) =
191                self.interner.find_associated_type_for_impl(*trait_impl_id, name).cloned()
192            {
193                let method_ident = variable.segments[2].ident.clone();
194                let typ_location = variable.segments[1].location;
195                // Extract already-resolved turbofish generics from the path segment
196                let resolved_generics = variable.segments[2].generics.as_ref().map(|generics| {
197                    generics.iter().map(|located| located.contents.clone()).collect()
198                });
199                return Some(self.elaborate_type_path_impl_with_resolved_generics(
200                    assoc_type,
201                    method_ident,
202                    resolved_generics,
203                    typ_location,
204                ));
205            }
206            // If it's not an associated type, fall through to let regular path resolution handle it
207            return None;
208        }
209
210        // Check the `Self::AssociatedConstant` case when inside a trait impl (2 segments)
211        if let Some((definition_id, numeric_type)) =
212            self.interner.get_trait_impl_associated_constant(*trait_impl_id, name).cloned()
213        {
214            let hir_ident = HirIdent::non_trait_method(definition_id, location);
215            let hir_expr = HirExpression::Ident(hir_ident, None);
216            let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
217            return Some((id, numeric_type));
218        }
219
220        // Check if the constant exists in the trait definition (even if impl is missing it).
221        // This prevents spurious "Could not resolve" errors inside trait methods when the impl is missing the constant,
222        // since the "missing associated constant" error is reported elsewhere.
223        let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
224        let trait_id = trait_impl.borrow().trait_id;
225        let trait_ = self.interner.get_trait(trait_id);
226        if let Some(definition_id) = trait_.associated_constant_ids.get(name).copied() {
227            let numeric_type = self.interner.definition_type(definition_id);
228            let hir_ident = HirIdent::non_trait_method(definition_id, location);
229            let hir_expr = HirExpression::Ident(hir_ident, None);
230            let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
231            return Some((id, numeric_type));
232        }
233
234        // Check the `Self::method_name` case when `Self` is a primitive type (2 segments)
235        if matches!(self.self_type, Some(Type::DataType(..))) {
236            return None;
237        }
238
239        let ident = variable.segments[1].ident.clone();
240        let typ_location = variable.segments[0].location;
241        Some(self.elaborate_type_path_impl(self_type.clone(), ident, None, typ_location))
242    }

244    /// Resolve a [TypedPath] to a [HirIdent] of either some trait method, or a local or global variable.
245    fn resolve_variable(&mut self, path: TypedPath) -> (HirIdent, Option<PathResolutionItem>) {
246        if let Some(trait_path_resolution) = self.resolve_trait_generic_path(&path) {
247            self.push_errors(trait_path_resolution.errors);
248
249            return match trait_path_resolution.method {
250                TraitPathResolutionMethod::NotATraitMethod(func_id) => (
251                    HirIdent {
252                        location: path.location,
253                        id: self.interner.function_definition_id(func_id),
254                        impl_kind: ImplKind::NotATraitMethod,
255                    },
256                    trait_path_resolution.item,
257                ),
258
259                TraitPathResolutionMethod::TraitItem(item) => (
260                    HirIdent {
261                        location: path.location,
262                        id: item.definition,
263                        impl_kind: ImplKind::TraitItem(item),
264                    },
265                    trait_path_resolution.item,
266                ),
267
268                TraitPathResolutionMethod::MultipleTraitsInScope => {
269                    // An error has already been pushed, return a dummy identifier
270                    let hir_ident = HirIdent {
271                        location: path.location,
272                        id: DefinitionId::dummy_id(),
273                        impl_kind: ImplKind::NotATraitMethod,
274                    };
275                    (hir_ident, trait_path_resolution.item)
276                }
277            };
278        }
279
280        // If the Path is being used as an Expression, then it is referring to a global from a separate module
281        // Otherwise, then it is referring to an Identifier
282        // This lookup allows support of such statements: let x = foo::bar::SOME_GLOBAL + 10;
283        // If the expression is a singular indent, we search the resolver's current scope as normal.
284        let location = path.location;
285        let ((hir_ident, var_scope_index), item) = self.get_ident_from_path(path);
286
287        self.handle_hir_ident(&hir_ident, var_scope_index, location);
288
289        (hir_ident, item)
290    }

292    /// Solve any generics that are part of the path before the function, for example:
293    ///
294    /// ```noir
295    /// foo::Bar::<i32>::baz
296    /// ```
297    /// Solve `<i32>` above
298    fn resolve_item_turbofish_and_self_type(
299        &mut self,
300        item: PathResolutionItem,
301    ) -> (Vec<Type>, Option<Type>) {
302        match item {
303            PathResolutionItem::Method(struct_id, Some(generics), _func_id) => {
304                let generics = self.resolve_struct_id_turbofish_generics(struct_id, Some(generics));
305                (generics, None)
306            }
307            PathResolutionItem::SelfMethod(_) => {
308                let generics = if let Some(Type::DataType(_, generics)) = &self.self_type {
309                    generics.clone()
310                } else {
311                    Vec::new()
312                };
313                (generics, None)
314            }
315            PathResolutionItem::TypeAliasFunction(type_alias_id, generics, _func_id) => {
316                let type_alias = self.interner.get_type_alias(type_alias_id);
317                let type_alias = type_alias.borrow();
318                let generics =
319                    self.resolve_type_alias_id_turbofish_generics(type_alias_id, generics);
320
321                // Now instantiate the underlying struct or alias with those generics, the struct might
322                // have more generics than those in the alias, like in this example:
323                //
324                // type Alias<T> = Struct<T, i32>;
325                let generics = get_type_alias_generics(&type_alias, &generics);
326                (generics, None)
327            }
328            PathResolutionItem::TraitFunction(trait_id, Some(generics), _func_id) => {
329                let trait_ = self.interner.get_trait(trait_id);
330                let kinds = vecmap(&trait_.generics, |generic| generic.kind());
331                let trait_generics =
332                    vecmap(&kinds, |kind| self.interner.next_type_variable_with_kind(kind.clone()));
333
334                let generics = self.resolve_trait_turbofish_generics(
335                    &trait_.name.to_string(),
336                    kinds,
337                    trait_generics,
338                    Some(generics.generics),
339                    generics.location,
340                );
341                (generics, None)
342            }
343            PathResolutionItem::TypeTraitFunction(self_type, _trait_id, _func_id) => {
344                (Vec::new(), Some(self_type))
345            }
346            PathResolutionItem::PrimitiveFunction(primitive_type, turbofish, _func_id) => {
347                let (typ, has_generics) =
348                    self.instantiate_primitive_type_with_turbofish(primitive_type, turbofish);
349                let generics = if has_generics {
350                    match typ {
351                        Type::String(length) => vec![*length],
352                        Type::FmtString(length, element) => vec![*length, *element],
353                        _ => {
354                            unreachable!("ICE: Primitive type has been specified to have generics")
355                        }
356                    }
357                } else {
358                    Vec::new()
359                };
360                (generics, None)
361            }
362            PathResolutionItem::Method(_, None, _)
363            | PathResolutionItem::TraitFunction(_, None, _)
364            | PathResolutionItem::Module(..)
365            | PathResolutionItem::Type(..)
366            | PathResolutionItem::TypeAlias(..)
367            | PathResolutionItem::PrimitiveType(..)
368            | PathResolutionItem::Trait(..)
369            | PathResolutionItem::TraitAssociatedType(..)
370            | PathResolutionItem::Global(..)
371            | PathResolutionItem::ModuleFunction(..)
372            | PathResolutionItem::TraitConstant(..) => (Vec::new(), None),
373        }
374    }

376    /// Elaborates a type path used in an expression, e.g. `Type::method::<Args>`
377    pub(super) fn elaborate_type_path(&mut self, path: TypePath) -> (ExprId, Type) {
378        let typ_location = path.typ.location;
379        let turbofish = path.turbofish;
380        let wildcard_allowed = WildcardAllowed::Yes;
381        let typ = self.use_type(path.typ, wildcard_allowed);
382        self.elaborate_type_path_impl(typ, path.item, turbofish, typ_location)
383    }

385    /// Variant of [Self::elaborate_type_path_impl_inner] that accepts unresolved generics.
386    fn elaborate_type_path_impl(
387        &mut self,
388        typ: Type,
389        ident: Ident,
390        turbofish: Option<GenericTypeArgs>,
391        typ_location: Location,
392    ) -> (ExprId, Type) {
393        let ident_location = ident.location();
394        let check_self_param = false;
395
396        self.interner.push_type_ref_location(&typ, typ_location);
397
398        let Some(method) = self.lookup_method(
399            &typ,
400            ident.as_str(),
401            ident_location,
402            typ_location,
403            check_self_param,
404        ) else {
405            let error = Expression::new(ExpressionKind::Error, ident_location);
406            return self.elaborate_expression(error);
407        };
408
409        let func_id = method
410            .func_id(self.interner)
411            .expect("Expected trait function to be a DefinitionKind::Function");
412
413        let generics =
414            turbofish.map(|turbofish| self.use_type_args(turbofish, func_id, ident_location).0);
415
416        self.elaborate_type_path_impl_inner(typ_location, ident_location, method, generics)
417    }

419    /// Variant of [Self::elaborate_type_path_impl_inner] that accepts already resolved generics.
420    /// Used when the turbofish generics have already been resolved.
421    fn elaborate_type_path_impl_with_resolved_generics(
422        &mut self,
423        typ: Type,
424        ident: Ident,
425        resolved_generics: Option<Vec<Type>>,
426        typ_location: Location,
427    ) -> (ExprId, Type) {
428        let ident_location = ident.location();
429        let check_self_param = false;
430
431        self.interner.push_type_ref_location(&typ, typ_location);
432
433        let Some(method) = self.lookup_method(
434            &typ,
435            ident.as_str(),
436            ident_location,
437            typ_location,
438            check_self_param,
439        ) else {
440            let error = Expression::new(ExpressionKind::Error, ident_location);
441            return self.elaborate_expression(error);
442        };
443
444        self.elaborate_type_path_impl_inner(typ_location, ident_location, method, resolved_generics)
445    }

447    /// Common implementation for type path impl variants.
448    fn elaborate_type_path_impl_inner(
449        &mut self,
450        _typ_location: Location,
451        ident_location: Location,
452        method: HirMethodReference,
453        generics: Option<Vec<Type>>,
454    ) -> (ExprId, Type) {
455        let func_id = method
456            .func_id(self.interner)
457            .expect("Expected trait function to be a DefinitionKind::Function");
458
459        let id = self.interner.function_definition_id(func_id);
460
461        let impl_kind = match method {
462            HirMethodReference::FuncId(_) => ImplKind::NotATraitMethod,
463            HirMethodReference::TraitItemId(HirTraitMethodReference {
464                definition,
465                trait_id,
466                trait_generics,
467                assumed: _,
468            }) => {
469                let mut constraint =
470                    self.interner.get_trait(trait_id).as_constraint(ident_location);
471                constraint.trait_bound.trait_generics = trait_generics;
472                ImplKind::TraitItem(TraitItem { definition, constraint, assumed: false })
473            }
474        };
475
476        let ident = HirIdent { location: ident_location, id, impl_kind };
477        let id =
478            self.intern_expr(HirExpression::Ident(ident.clone(), generics.clone()), ident_location);
479
480        let typ = self.type_check_variable(ident, &id, generics);
481        let id = self.intern_expr_type(id, typ.clone());
482
483        (id, typ)
484    }

486    /// Given an [HirIdent], look up its definition, and:
487    /// * mark it as referenced at the ident [Location] (LSP mode only)
488    /// * mark the item currently being elaborated as a dependency of it
489    /// * elaborate a global definition, if needed
490    /// * add local identifiers to lambda captures
491    pub(crate) fn handle_hir_ident(
492        &mut self,
493        hir_ident: &HirIdent,
494        var_scope_index: usize,
495        location: Location,
496    ) {
497        if hir_ident.id == DefinitionId::dummy_id() {
498            return;
499        }
500
501        match self.interner.definition(hir_ident.id).kind {
502            DefinitionKind::Function(func_id) => {
503                if let Some(current_item) = self.current_item {
504                    self.interner.add_function_dependency(current_item, func_id);
505                }
506
507                self.interner.add_function_reference(func_id, hir_ident.location);
508            }
509            DefinitionKind::Global(global_id) => {
510                self.elaborate_global_if_unresolved(&global_id);
511                if let Some(current_item) = self.current_item {
512                    self.interner.add_global_dependency(current_item, global_id);
513                }
514
515                self.interner.add_global_reference(global_id, hir_ident.location);
516            }
517            DefinitionKind::NumericGeneric(_, ref numeric_typ) => {
518                // Initialize numeric generics to a polymorphic integer type in case
519                // they're used in expressions. We must do this here since type_check_variable
520                // does not check definition kinds and otherwise expects parameters to
521                // already be typed.
522                if self.interner.definition_type(hir_ident.id) == Type::Error {
523                    let type_var_kind = Kind::Numeric(numeric_typ.clone());
524                    let typ = self.type_variable_with_kind(type_var_kind);
525                    self.interner.push_definition_type(hir_ident.id, typ);
526                }
527            }
528            DefinitionKind::Local(_) => {
529                // only local variables can be captured by closures.
530                self.resolve_local_variable(hir_ident.clone(), var_scope_index);
531
532                self.interner.add_local_reference(hir_ident.id, location);
533            }
534            DefinitionKind::AssociatedConstant(..) => {
535                // Nothing to do here
536            }
537        }
538    }

540    /// Starting with empty bindings, perform the type checking of an interned expression
541    /// and a corresponding identifier, returning the instantiated [Type].
542    pub(crate) fn type_check_variable(
543        &mut self,
544        ident: HirIdent,
545        expr_id: &PushedExpr<HasLocation>,
546        generics: Option<Vec<Type>>,
547    ) -> Type {
548        let bindings = TypeBindings::default();
549        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
550        let push_required_type_variables = self.current_trait.is_none();
551        self.type_check_variable_with_bindings(
552            ident,
553            expr_id,
554            generics,
555            bindings,
556            push_required_type_variables,
557        )
558    }

560    /// Perform the type checking of an interned expression and a corresponding identifier,
561    /// returning the instantiated [Type].
562    ///
563    /// If `push_required_type_variables`, the bindings are added to the function context,
564    /// to be checked before it's finished.
565    pub(crate) fn type_check_variable_with_bindings(
566        &mut self,
567        ident: HirIdent,
568        expr_id: &PushedExpr<HasLocation>,
569        generics: Option<Vec<Type>>,
570        mut bindings: TypeBindings,
571        push_required_type_variables: bool,
572    ) -> Type {
573        // Add type bindings from any constraints that were used.
574        // We need to do this first since otherwise instantiating the type below
575        // will replace each trait generic with a fresh type variable, rather than
576        // the type used in the trait constraint (if it exists). See #4088.
577        if let ImplKind::TraitItem(method) = &ident.impl_kind {
578            self.bind_generics_from_trait_constraint(
579                &method.constraint,
580                method.assumed,
581                &mut bindings,
582            );
583        }
584
585        // An identifiers type may be forall-quantified in the case of generic functions.
586        // E.g. `fn foo<T>(t: T, field: Field) -> T` has type `forall T. fn(T, Field) -> T`.
587        // We must instantiate identifiers at every call site to replace this T with a new type
588        // variable to handle generic functions.
589        let t = self.type_substitute_trait_as_type(&ident);
590
591        let definition = self.interner.try_definition(ident.id);
592        let function_generic_count = definition.map_or(0, |definition| match &definition.kind {
593            DefinitionKind::Function(function) => {
594                self.interner.function_modifiers(function).generic_count
595            }
596            _ => 0,
597        });
598
599        let location = self.interner.expr_location(expr_id);
600
601        // This instantiates a trait's generics as well which need to be set
602        // when the constraint below is later solved for when the function is
603        // finished. How to link the two?
604        let (typ, bindings) =
605            self.instantiate(t, bindings, generics, function_generic_count, location);
606
607        if let ImplKind::TraitItem(mut method) = ident.impl_kind {
608            method.constraint.apply_bindings(&bindings);
609            if method.assumed {
610                let trait_generics = method.constraint.trait_bound.trait_generics.clone();
611                let object_type = method.constraint.typ;
612                let trait_impl = TraitImplKind::Assumed { object_type, trait_generics };
613                self.interner.select_impl_for_expression(**expr_id, trait_impl);
614            } else {
615                // this constraint should lead to choosing a trait impl method
616                self.push_trait_constraint(method.constraint, **expr_id, true);
617            }
618        }
619
620        // Push any trait constraints required by this definition to the context
621        // to be checked later when the type of this variable is further constrained.
622        //
623        // This must be done before the above trait constraint in case the above one further
624        // restricts types.
625        //
626        // For example, in this code:
627        //
628        // ```noir
629        // trait One {}
630        //
631        // trait Two<O: One> {
632        //     fn new() -> Self;
633        // }
634        //
635        // fn foo<X: One, T: Two<X>>() {
636        //     let _: T = Two::new();
637        // }
638        // ```
639        //
640        // when type-checking `Two::new` we'll have a return type `'2` which is constrained by `'2: Two<'1>`.
641        // Then the definition for `new` has a constraint on it, `O: One`, which translates to `'1: One`.
642        //
643        // Because of the explicit type in the `let`, `'2` will be unified with `T`.
644        // Then we must first verify the constraint `'2: Two<'1>`, which is now `T: Two<'1>`, to find
645        // that the implementation is the assumed one `T: Two<X>` so that `'1` is bound to `X`.
646        // Then we can successfully verify the constraint `'1: One` which now became `X: One` which holds
647        // because of the assumed constraint.
648        //
649        // If we try to find a trait implementation for `'1` before finding one for `'2` we'll never find it.
650        if let Some(definition) = self.interner.try_definition(ident.id) {
651            if let DefinitionKind::Function(function) = definition.kind {
652                let function = self.interner.function_meta(&function);
653                for mut constraint in function.all_trait_constraints().cloned().collect::<Vec<_>>()
654                {
655                    constraint.apply_bindings(&bindings);
656
657                    // This constraint shouldn't lead to choosing a trait impl method
658                    self.push_trait_constraint(constraint, **expr_id, false);
659                }
660            }
661        }
662
663        if push_required_type_variables {
664            for (type_variable, _kind, typ) in bindings.values() {
665                self.push_required_type_variable(
666                    type_variable.id(),
667                    typ.clone(),
668                    BindableTypeVariableKind::Ident(ident.id),
669                    ident.location,
670                );
671            }
672        }
673
674        self.interner.store_instantiation_bindings(**expr_id, bindings);
675        typ
676    }
677
678    /// If the type of the [HirIdent] is a function that returns an `impl Trait`,
679    /// then it might need elaboration before it can be substituted to a [Type].
680    /// Try to elaborate it now.
681    ///
682    /// Returns a type error if the callee cannot be resolved on a second try,
683    /// which indicates a dependency cycle.
684    fn type_substitute_trait_as_type(&mut self, ident: &HirIdent) -> Type {
685        let func_id = match self.interner.id_type_substitute_trait_as_type(ident.id) {
686            Ok(typ) => return typ,
687            Err(func_id) => func_id,
688        };
689
690        // Try to elaborate, so we get an expression for the body.
691        self.elaborate_function(func_id);
692
693        // Now try again. If it's still not working, give up.
694        match self.interner.id_type_substitute_trait_as_type(ident.id) {
695            Ok(typ) => typ,
696            Err(_) => {
697                let def = self.interner.definition(ident.id);
698                self.push_err(ResolverError::DependencyCycle {
699                    location: ident.location,
700                    item: def.name.clone(),
701                    cycle: "'impl Trait' could not be resolved to the type of the function body"
702                        .to_string(),
703                });
704                Type::Error
705            }
706        }
707    }
708
709    /// Instantiate a [Type] with the given [TypeBindings], returning the bindings potentially
710    /// extended from any turbofish generics.
711    ///
712    /// If there are turbofish generics and their number matches the expectations of the function,
713    /// those are used as well, otherwise they are ignored and an error is pushed.
714    fn instantiate(
715        &mut self,
716        typ: Type,
717        bindings: TypeBindings,
718        turbofish_generics: Option<Vec<Type>>,
719        function_generic_count: usize,
720        location: Location,
721    ) -> (Type, TypeBindings) {
722        match turbofish_generics {
723            Some(turbofish_generics) => {
724                if turbofish_generics.len() != function_generic_count {
725                    let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
726                        expected_count: function_generic_count,
727                        actual_count: turbofish_generics.len(),
728                        location,
729                    };
730                    self.push_err(CompilationError::TypeError(type_check_err));
731                    typ.instantiate_with_bindings(bindings, self.interner)
732                } else {
733                    // Fetch the count of any implicit generics on the function, such as
734                    // for a method within a generic impl.
735                    let implicit_generic_count = match &typ {
736                        Type::Forall(generics, _) => generics.len() - function_generic_count,
737                        _ => 0,
738                    };
739                    typ.instantiate_with_bindings_and_turbofish(
740                        bindings,
741                        turbofish_generics,
742                        self.interner,
743                        implicit_generic_count,
744                    )
745                }
746            }
747            None => typ.instantiate_with_bindings(bindings, self.interner),
748        }
749    }
750}
751
752/// Bind the generics of the [Type] aliased by the [TypeAlias] to a list of generic arguments,
753/// recursively expanding the generics aliased aliases, finally returning the generics of the
754/// innermost aliased struct.
755///
756/// Panics if it encounters a type other than alias or struct.
757fn get_type_alias_generics(type_alias: &TypeAlias, generics: &[Type]) -> Vec<Type> {
758    let typ = type_alias.get_type(generics);
759    match typ {
760        Type::DataType(_, generics) => generics,
761        Type::Alias(type_alias, generics) => {
762            get_type_alias_generics(&type_alias.borrow(), &generics)
763        }
764        _ => panic!("Expected type alias to point to struct or alias"),
765    }
766}
