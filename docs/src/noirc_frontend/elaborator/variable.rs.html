31/// The result of [`Elaborator::resolve_variable`]: what a path used as an expression resolves
32/// to. See [`Elaborator::resolve_variable`] for the full, ordered list of the path forms each
33/// variant covers.
34#[allow(clippy::large_enum_variant)]
35pub(crate) enum VariableResolution {
36    /// The path was elaborated straight to an expression. This happens for the `Self::*` forms
37    /// handled inside a trait impl: an associated constant, an associated-type method, or a
38    /// method when `Self` is a primitive type.
39    Expression(ExprId, Type),
40    /// The path resolved to a local variable, a definition (global, function, enum-variant
41    /// global), or a trait item (method or associated constant).
42    Ident(HirIdent, Option<PathResolutionItem>),
43    /// The path resolved to a type alias that is numeric, infinitely recursive or one that errored.
44    TypeAlias(TypeAliasId),
45}
46
47impl Elaborator<'_> {
48    #[tracing::instrument(level = "trace", skip_all)]
49    pub(super) fn elaborate_variable(&mut self, variable: Path) -> (ExprId, Type) {
50        let (id, typ, is_comptime_local, location) = self.elaborate_variable_inner(variable);
51
52        // Only check has_errors when we need to call the interpreter
53        // If this variable is a comptime local variable, use its current value as the final expression
54        if is_comptime_local {
55            let mut interpreter = self.setup_interpreter();
56            let value = interpreter.evaluate(id);
57            // If the value is an error it means the variable already had an error, so don't report it here again
58            // (the error will make no sense, it will say that a non-comptime variable was referenced at runtime
59            // but that's not true)
60            if value.is_ok() {
61                let from_macro_call = false;
62                let (id, typ) = self.inline_comptime_value(value, location, from_macro_call);
63                self.debug_comptime(location, |interner, _| id.to_display_ast(interner).kind);
64                (id, typ)
65            } else {
66                (id, typ)
67            }
68        } else {
69            (id, typ)
70        }
71    }
72
73    /// Helper function containing the elaboration logic for a variable.
74    /// Returns the expression ID, type, whether it's a comptime local, and location.
75    #[tracing::instrument(level = "trace", skip_all)]
76    fn elaborate_variable_inner(&mut self, variable: Path) -> (ExprId, Type, bool, Location) {
77        let variable = self.validate_path(variable);
78
79        let resolved_turbofish = variable.segments.last().unwrap().generics.clone();
80
81        // A turbofish on the segment *before* the last one (e.g. `Foo::<u32>::Spam`) provides
82        // type generics for the type the last segment is resolved within. This is needed for
83        // fieldless enum variants, which resolve to a global rather than a function.
84        let type_segment_turbofish = (variable.segments.len() >= 2)
85            .then(|| variable.segments[variable.segments.len() - 2].generics.clone())
86            .flatten();
87
88        let location = variable.location;
89        let variable_resolution = self.resolve_variable(variable);
90
91        let (hir_ident, item) = match variable_resolution {
92            Some(VariableResolution::Expression(id, typ)) => return (id, typ, false, location),
93            Some(VariableResolution::TypeAlias(type_alias_id)) => {
94                // A type alias to a numeric generics is considered like a variable,
95                // but it is not a real variable so it does not resolve to a valid Identifier.
96                // In order to handle this, we retrieve the numeric generics expression that the type aliases to.
97                let type_alias = self.interner.get_type_alias(type_alias_id);
98                let alias_module_id = type_alias.borrow().module_id;
99                if let Some(type_alias_expr) = &type_alias.borrow().numeric_expr {
100                    // Extract the declared numeric type from the type alias's kind.
101                    let declared_type = match type_alias.borrow().typ.kind() {
102                        Kind::Numeric(declared_type) => declared_type,
103                        _ => Box::new(Type::Error),
104                    };
105                    let declared_type = *declared_type;
106                    let var_expr = UnresolvedTypeExpression::to_expression_kind(type_alias_expr);
107
108                    // The expression we create for this particular instantiation of the numeric type alias
109                    // must have the same location as the path that refers to it.
110                    let var_expr = Expression::new(var_expr, location);
111
112                    // Resolve turbofish generics for the type alias.
113                    // `resolved_turbofish` contains already-resolved types from
114                    // validate_path, so we use resolve_alias_turbofish_generics
115                    // directly which accepts resolved types.
116                    let alias_generics = &type_alias.borrow().generics;
117                    let alias_generic_types = vecmap(alias_generics, |generic| {
118                        self.interner.next_type_variable_with_kind(generic.kind())
119                    });
120                    let mut errors = Vec::new();
121                    let type_alias_ref = type_alias.borrow();
122                    let resolved_generics = self.resolve_alias_turbofish_generics(
123                        &type_alias_ref,
124                        alias_generic_types,
125                        resolved_turbofish,
126                        location,
127                        &mut errors,
128                    );
129                    self.push_errors(errors);
130
131                    // Introduce alias generics into scope so the numeric expression
132                    // resolves them correctly (not to globals or other variables
133                    // that happen to share the same name). Bind each generic's type
134                    // variable to the turbofish-resolved type.
135                    self.push_scope();
136                    for (generic, resolved_type) in
137                        alias_generics.iter().zip_eq(resolved_generics.iter())
138                    {
139                        if let Kind::Numeric(numeric_type) = &generic.kind() {
140                            let id = self.interner.next_type_variable_id();
141                            let type_var = TypeVariable::unbound(id, generic.kind());
142                            type_var.bind(resolved_type.clone());
143                            let definition =
144                                DefinitionKind::NumericGeneric(type_var, numeric_type.clone());
145                            let ident = Ident::new(generic.name.to_string(), generic.location);
146                            let hir_ident = self.add_variable_decl(
147                                ident, false, // mutable
148                                true,  // allow_shadowing
149                                false, // warn_if_unused
150                                false, // warn_if_not_mutated
151                                definition,
152                            );
153                            self.interner.push_definition_type(hir_ident.id, *numeric_type.clone());
154                        }
155                    }
156
157                    // The alias's numeric expression has already been kind-checked at
158                    // alias-definition time (see `convert_expression_type`), which is
159                    // where any "value does not fit" diagnostic is emitted. Drop any
160                    // literals queued for the function-context fit check during
161                    // re-elaboration so the same overflow is not reported twice.
162                    let literals_before = self.integer_literal_expr_ids_len();
163                    // Re-elaborate the alias body in the alias's defining module
164                    // so unqualified names resolve against the alias's scope,
165                    // not the caller's. Mirrors `define_type_alias` in mod.rs.
166                    let previous_module = self.replace_module(alias_module_id);
167                    let (id, typ) = self.elaborate_expression(var_expr);
168                    self.restore_module(previous_module);
169                    self.truncate_integer_literal_expr_ids(literals_before);
170                    self.pop_scope();
171
172                    // Unify the expression's type with the declared type from the type alias
173                    // to ensure proper type checking.
174                    self.unify_or_type_mismatch(&typ, &declared_type, type_alias_expr.location());
175
176                    return (id, declared_type, false, location);
177                }
178                (None, None)
179            }
180            Some(VariableResolution::Ident(ident, item)) => (Some(ident), item),
181            None => (None, None),
182        };
183
184        let definition_id = hir_ident.as_ref().map(|ident| ident.id);
185
186        let (type_generics, self_generic) = if let Some(item) = item {
187            self.resolve_item_turbofish_and_self_type(item)
188        } else {
189            (Vec::new(), None)
190        };
191
192        let definition = definition_id.map(|id| self.interner.definition(id));
193        let is_comptime_local = !self.in_comptime_context()
194            && definition.is_some_and(DefinitionInfo::is_comptime_local);
195        let definition_kind = definition.as_ref().map(|definition| definition.kind.clone());
196
197        let mut bindings = TypeBindings::default();
198        let generics = if let Some(DefinitionKind::Function(func_id)) = &definition_kind {
199            self.usage_tracker.mark_impl_function_as_used(func_id);
200
201            // If there's a self type, bind it to the self type generic
202            if let Some(self_generic) = self_generic {
203                let func_generics = &self.function_meta(*func_id).all_generics;
204                let self_resolved_generic =
205                    func_generics.iter().find(|generic| generic.name.as_str() == SELF_TYPE_NAME);
206                if let Some(self_resolved_generic) = self_resolved_generic {
207                    let type_var = &self_resolved_generic.type_var;
208                    bindings
209                        .insert(type_var.id(), (type_var.clone(), type_var.kind(), self_generic));
210                }
211            }
212
213            // If this is a function call on a type that has generics, we need to bind those generic types.
214            if !type_generics.is_empty() {
215                // We must only bind the type-level portion here; method generics are handled
216                // separately by the method turbofish.
217                let func_meta = self.function_meta(*func_id);
218                let impl_generics = vecmap(func_meta.impl_generics(), |g| g.type_var.clone());
219                let self_type =
220                    func_meta.self_type.as_ref().map(|t| t.follow_bindings_shallow().into_owned());
221
222                // An enum variant constructor (e.g. `Foo::<u32>::Eggs`) has no `impl_generics`
223                // and no `self_type`; the enum's generics are its `direct_generics`. The
224                // type-segment turbofish provides types for exactly those generics, so bind
225                // them directly. `type_generics` was resolved against the enum's generics, so
226                // its length matches `direct_generics`.
227                if func_meta.enum_variant_index.is_some() {
228                    let direct_generics =
229                        vecmap(&func_meta.direct_generics, |g| g.type_var.clone());
230                    for (type_generic, type_var) in
231                        type_generics.into_iter().zip_eq(direct_generics)
232                    {
233                        bindings.insert(
234                            type_var.id(),
235                            (type_var.clone(), type_var.kind(), type_generic),
236                        );
237                    }
238                }
239                // For partially concrete impls (e.g. `impl<B> S<u32, B>`), the number of
240                // impl generics differs from the number of struct generics. The turbofish
241                // `S::<u32, bool>` provides type_generics aligned with the struct's params
242                // [A, B], not the impl's generics [B]. Replace each impl generic in
243                // `self_type`'s args with a fresh type variable and unify those with the
244                // turbofish-provided type generics. The fresh type variables get bound by
245                // unification, and we record those bindings for the impl generics.
246                else if let Some(Type::DataType(_, self_type_args)) = self_type {
247                    assert_eq!(
248                        type_generics.len(),
249                        self_type_args.len(),
250                        "ICE: turbofish type_generics count ({}) should match self_type_args count ({})",
251                        type_generics.len(),
252                        self_type_args.len(),
253                    );
254                    let impl_replacements: TypeBindings = impl_generics
255                        .iter()
256                        .map(|type_var| {
257                            let kind = type_var.kind();
258                            let fresh = self.interner.next_type_variable_with_kind(kind.clone());
259                            (type_var.id(), (type_var.clone(), kind, fresh))
260                        })
261                        .collect();
262                    for (type_generic, self_type_arg) in
263                        type_generics.into_iter().zip_eq(self_type_args)
264                    {
265                        let substituted = self_type_arg.substitute(&impl_replacements);
266                        self.unify_or_type_mismatch(&type_generic, &substituted, location);
267                    }
268                    bindings.extend(impl_replacements);
269                } else if type_generics.len() <= impl_generics.len() {
270                    // For trait function paths, impl_generics may include Self and associated
271                    // type generics after the trait's declared generics. The turbofish only
272                    // provides types for the declared generics, which are always the first
273                    // elements. Slice to match.
274                    let impl_generics = &impl_generics[..type_generics.len()];
275                    for (type_generic, type_var) in type_generics.into_iter().zip_eq(impl_generics)
276                    {
277                        bindings.insert(
278                            type_var.id(),
279                            (type_var.clone(), type_var.kind(), type_generic),
280                        );
281                    }
282                } else {
283                    unreachable!(
284                        "type_generics.len() ({}) > impl_generics.len() ({}): \
285                        turbofish resolution should normalize the count",
286                        type_generics.len(),
287                        impl_generics.len()
288                    );
289                }
290            }
291
292            // Resolve any generics if the variable we have resolved is a function
293            // and if the turbofish operator was used.
294            self.resolve_function_turbofish_generics(func_id, resolved_turbofish, location)
295        } else {
296            // A fieldless enum variant resolves to a global. A turbofish on the variant path
297            // binds the enum's generics, which global path resolution does not carry on its
298            // own. The turbofish may be on the type segment (`Foo::<u32>::Spam`) or the
299            // variant segment (`Foo::Spam::<u32>`); both denote the same enum generics.
300            let is_enum_variant_global =
301                if let Some(DefinitionKind::Global(global_id)) = &definition_kind {
302                    self.interner.is_enum_variant_global(*global_id)
303                } else {
304                    false
305                };
306
307            if is_enum_variant_global {
308                if let Some(turbofish) = resolved_turbofish.or(type_segment_turbofish) {
309                    self.bind_enum_variant_global_turbofish(
310                        definition_id.unwrap(),
311                        &turbofish,
312                        location,
313                        &mut bindings,
314                    );
315                }
316            } else if let Some(unused_resolved_turbofish) = resolved_turbofish {
317                let message = format!(
318                    "elaborate_variable_inner: unused resolved_turbofish: {unused_resolved_turbofish:?}"
319                );
320                self.push_err(TypeCheckError::expecting_other_error(message, location));
321            }
322
323            None
324        };
325
326        let (id, typ) = if let Some(hir_ident) = hir_ident {
327            let id = self
328                .intern_expr(HirExpression::Ident(hir_ident.clone(), generics.clone()), location);
329
330            let typ = self.type_check_variable_with_bindings(hir_ident, &id, generics, bindings);
331            let id = self.intern_expr_type(id, typ.clone());
332            (id, typ)
333        } else {
334            let expr = HirExpression::Error;
335            let id = self.intern_expr(expr, location);
336            let typ = Type::Error;
337            let id = self.intern_expr_type(id, typ.clone());
338            (id, typ)
339        };
340
341        (id, typ, is_comptime_local, location)
342    }

344    /// Bind the type-segment turbofish of a fieldless enum variant path (e.g. `Foo::<u32>::Spam`)
345    /// to the variant global's `Forall` generics (the enum's generics), validating the count the
346    /// same way [`Self::resolve_item_turbofish_generics`] does. A non-generic enum has no `Forall`,
347    /// so a turbofish on it produces a count-mismatch error.
348    fn bind_enum_variant_global_turbofish(
349        &mut self,
350        definition_id: DefinitionId,
351        turbofish: &[Located<Type>],
352        location: Location,
353        bindings: &mut TypeBindings,
354    ) {
355        let global_type = self.interner.definition_type(definition_id);
356        let (typevars, enum_name) = match &global_type {
357            Type::Forall(typevars, body) => (typevars.clone(), data_type_name(body)),
358            other => (Vec::new(), data_type_name(other)),
359        };
360
361        let mut turbofish = turbofish.to_vec();
362        if turbofish.len() != typevars.len() {
363            self.push_err(TypeCheckError::GenericCountMismatch {
364                item: format!("enum `{}`", enum_name.unwrap_or_default()),
365                expected: typevars.len(),
366                found: turbofish.len(),
367                location,
368            });
369            // Pad/truncate to the expected length so every generic is still determined,
370            // matching `resolve_function_turbofish_generics` and avoiding a cascade of
371            // "type annotation needed" errors.
372            turbofish.resize(typevars.len(), Located::from(location, Type::Error));
373        }
374
375        for (located_type, type_var) in turbofish.into_iter().zip(&typevars) {
376            let type_location = located_type.location();
377            let typ = self.check_type_kind(located_type.contents, &type_var.kind(), type_location);
378            bindings.insert(type_var.id(), (type_var.clone(), type_var.kind(), typ));
379        }
380    }

382    /// Resolves the `Self::…` forms that only make sense inside a trait impl, where `Self` is a
383    /// concrete type with associated items. The segments *after* `Self` (the "rest") select the
384    /// form:
385    ///
386    /// - `[AssociatedType, method]` — resolve the associated type, then elaborate the method on it.
387    /// - `[item]` — an associated constant (looked up for its value, later a literal), or, when
388    ///   `Self` is a primitive type, a method elaborated as if it were a [TypePath]
389    ///   (`u32::method_name`); a regular path lookup won't work, for the same reason [TypePath]
390    ///   exists.
391    ///
392    /// Returns `None` for any other shape (including a data-type `Self`, handled as a plain type
393    /// prefix), so the caller falls back to resolving `Self` as a type.
394    #[tracing::instrument(level = "trace", skip_all)]
395    pub(super) fn resolve_variable_as_self_method_or_associated_constant(
396        &mut self,
397        variable: &TypedPath,
398        self_type: Type,
399        trait_impl_id: TraitImplId,
400    ) -> Option<(ExprId, Type)> {
401        match &variable.segments[1..] {
402            [associated_type, method] => {
403                let associated_type = self
404                    .interner
405                    .find_associated_type_for_impl(trait_impl_id, associated_type.ident.as_str())
406                    .cloned()?;
407                // Extract already-resolved turbofish generics from the method segment.
408                let resolved_generics = method.generics.as_ref().map(|generics| {
409                    generics.iter().map(|located| located.contents.clone()).collect()
410                });
411                Some(self.elaborate_type_path_impl_with_resolved_generics(
412                    associated_type,
413                    method.ident.clone(),
414                    resolved_generics,
415                    variable.segments[1].location,
416                ))
417            }
418            [item] => self.elaborate_self_associated_constant_or_primitive_method(
419                &self_type,
420                trait_impl_id,
421                item,
422                variable.location,
423                variable.segments[0].location,
424            ),
425            _ => None,
426        }
427    }

429    /// The `Self::item` (single segment after `Self`) case of
430    /// [`Self::resolve_variable_as_self_method_or_associated_constant`]: an associated constant
431    /// (from the impl, or from the trait when the impl is missing it), or a method when `Self` is a
432    /// primitive type. A data-type `Self` returns `None` so it is resolved as a plain type prefix.
433    fn elaborate_self_associated_constant_or_primitive_method(
434        &mut self,
435        self_type: &Type,
436        trait_impl_id: TraitImplId,
437        item: &TypedPathSegment,
438        location: Location,
439        self_location: Location,
440    ) -> Option<(ExprId, Type)> {
441        let name = item.ident.as_str();
442
443        // The associated constant declared on the impl.
444        if let Some((definition_id, numeric_type)) =
445            self.interner.get_trait_impl_associated_constant(trait_impl_id, name).cloned()
446        {
447            return Some(self.intern_associated_constant(definition_id, numeric_type, location));
448        }
449
450        // The constant declared on the trait, even if the impl is missing it. This prevents a
451        // spurious "Could not resolve" inside trait methods; the "missing associated constant"
452        // error is reported elsewhere.
453        if let Some(trait_impl) = self.interner.try_get_trait_implementation(trait_impl_id) {
454            let trait_id = trait_impl.borrow().trait_id;
455            let trait_ = self.interner.get_trait(trait_id);
456            if let Some(definition_id) = trait_.associated_constant_ids.get(name).copied() {
457                let numeric_type = self.interner.definition_type(definition_id);
458                return Some(self.intern_associated_constant(
459                    definition_id,
460                    numeric_type,
461                    location,
462                ));
463            }
464        }
465
466        // A data-type `Self::method` is resolved as a plain type prefix, not here.
467        if matches!(self_type, Type::DataType(..)) {
468            return None;
469        }
470
471        Some(self.elaborate_type_path_impl(
472            self_type.clone(),
473            item.ident.clone(),
474            None,
475            self_location,
476        ))
477    }

479    /// Intern an identifier expression referring to an associated constant of the given type.
480    fn intern_associated_constant(
481        &mut self,
482        definition_id: DefinitionId,
483        numeric_type: Type,
484        location: Location,
485    ) -> (ExprId, Type) {
486        let hir_ident = HirIdent::non_trait_method(definition_id, location);
487        let hir_expr = HirExpression::Ident(hir_ident, None);
488        let id = self.interner.push_expr_full(hir_expr, location, numeric_type.clone());
489        (id, numeric_type)
490    }

492    /// Resolve a [`TypedPath`] used as an expression to the item it names.
493    ///
494    /// A path with a prefix (more than one segment) names an item accessed *through* that prefix,
495    /// fully handled by [`Self::resolve_prefixed_variable`]. A single-segment path resolves in the
496    /// current scope (a local variable, or a value item — global, function, enum-variant global,
497    /// numeric type alias) via [`Self::resolve_unprefixed_variable`].
498    #[tracing::instrument(level = "trace", skip_all)]
499    fn resolve_variable(&mut self, path: TypedPath) -> Option<VariableResolution> {
500        if path.segments.len() > 1 {
501            self.resolve_prefixed_variable(path)
502        } else {
503            self.resolve_unprefixed_variable(path)
504        }
505    }

507    /// Resolve an unprefixed (single-segment) path to a local variable, or to a value item it
508    /// names (global, function, enum-variant global, numeric type alias). The counterpart to
509    /// [`Self::resolve_prefixed_variable`]; a prefixed path's value fallback uses
510    /// [`Self::resolve_value_item`], which skips the local-variable lookup.
511    pub(super) fn resolve_unprefixed_variable(
512        &mut self,
513        path: TypedPath,
514    ) -> Option<VariableResolution> {
515        // The location of variables or definitions we register (for LSP) must be that of the
516        // path's last segment, as intermediate segments solve to other definitions.
517        let location = path.last_ident().location();
518
519        // If the Path is being used as an Expression, then it is referring to a global from a separate module
520        // Otherwise, then it is referring to an Identifier
521        // This lookup allows support of such statements: let x = foo::bar::SOME_GLOBAL + 10;
522        // If the expression is a singular indent, we search the resolver's current scope as normal.
523        let ident_from_path = self.get_ident_from_path(path)?;
524        Some(self.variable_resolution_from_ident_from_path(ident_from_path, location))
525    }

527    /// Resolve a path's last segment as a value item (a global, function, enum-variant global,
528    /// trait constant, or numeric type alias). Unlike [`Self::resolve_unprefixed_variable`] this does
529    /// not look for a local variable, so it is the value fallback for a prefixed path: its last
530    /// segment is not a method or trait item, and a prefixed path can never name a local variable.
531    pub(super) fn resolve_value_item(&mut self, path: TypedPath) -> Option<VariableResolution> {
532        let location = path.last_ident().location();
533        match self.ident_from_value_item(path) {
534            Ok(ident_from_path) => {
535                Some(self.variable_resolution_from_ident_from_path(ident_from_path, location))
536            }
537            Err(error) => {
538                self.push_err(error);
539                None
540            }
541        }
542    }

544    /// Build the [`VariableResolution`] an already-resolved [`IdentFromPath`] denotes, registering
545    /// the reference (for LSP) along the way.
546    fn variable_resolution_from_ident_from_path(
547        &mut self,
548        ident_from_path: IdentFromPath,
549        location: Location,
550    ) -> VariableResolution {
551        match ident_from_path {
552            IdentFromPath::Variable(variable) => {
553                self.handle_local_variable(&variable);
554                let hir_ident = HirIdent::non_trait_method(variable.ident.id, location);
555                VariableResolution::Ident(hir_ident, None)
556            }
557            IdentFromPath::Definition { id, item } => {
558                self.handle_definition_id(id, location);
559                let hir_ident = HirIdent::non_trait_method(id, location);
560                VariableResolution::Ident(hir_ident, Some(item))
561            }
562            IdentFromPath::TypeAlias(type_alias_id) => VariableResolution::TypeAlias(type_alias_id),
563        }
564    }

566    /// Solve any generics that are part of the path before the function, for example:
567    ///
568    /// ```noir
569    /// foo::Bar::<i32>::baz
570    /// ```
571    /// Solve `<i32>` above
572    #[tracing::instrument(level = "trace", skip_all)]
573    fn resolve_item_turbofish_and_self_type(
574        &mut self,
575        item: PathResolutionItem,
576    ) -> (Vec<Type>, Option<Type>) {
577        let mut errors = Vec::new();
578        let result = match item {
579            PathResolutionItem::Method(struct_id, Some(generics), _func_id) => {
580                let generics = self.resolve_struct_id_turbofish_generics(
581                    struct_id,
582                    Some(generics),
583                    &mut errors,
584                );
585                (generics, None)
586            }
587            PathResolutionItem::SelfMethod(_) => {
588                let generics = if let Some(Type::DataType(_, generics)) = &self.self_type {
589                    generics.clone()
590                } else {
591                    Vec::new()
592                };
593                (generics, None)
594            }
595            PathResolutionItem::TypeAliasFunction(type_alias_id, generics, _func_id) => {
596                let type_alias = self.interner.get_type_alias(type_alias_id);
597                let type_alias = type_alias.borrow();
598                let generics = self.resolve_type_alias_id_turbofish_generics(
599                    type_alias_id,
600                    generics,
601                    &mut errors,
602                );
603
604                // Now instantiate the underlying struct or alias with those generics, the struct might
605                // have more generics than those in the alias, like in this example:
606                //
607                // type Alias<T> = Struct<T, i32>;
608                let generics = get_type_alias_generics(&type_alias, &generics);
609                (generics, None)
610            }
611            PathResolutionItem::TraitFunction(trait_id, Some(generics), _func_id) => {
612                let trait_ = self.interner.get_trait(trait_id);
613                let kinds = vecmap(&trait_.generics, |generic| generic.kind());
614                let trait_generics =
615                    vecmap(&kinds, |kind| self.interner.next_type_variable_with_kind(kind.clone()));
616
617                let generics = self.resolve_trait_turbofish_generics(
618                    &trait_.name.to_string(),
619                    kinds,
620                    trait_generics,
621                    Some(generics.generics),
622                    generics.location,
623                    &mut errors,
624                );
625                (generics, None)
626            }
627            PathResolutionItem::TypeTraitFunction(self_type, _trait_id, _func_id) => {
628                (Vec::new(), Some(self_type))
629            }
630            PathResolutionItem::PrimitiveFunction(primitive_type, turbofish, _func_id) => {
631                let (typ, has_generics) = self.instantiate_primitive_type_with_turbofish(
632                    primitive_type,
633                    turbofish,
634                    &mut errors,
635                );
636                let generics = if has_generics {
637                    match typ {
638                        Type::String(length) => vec![*length],
639                        Type::FmtString(length, element) => vec![*length, *element],
640                        _ => {
641                            unreachable!("ICE: Primitive type has been specified to have generics")
642                        }
643                    }
644                } else {
645                    Vec::new()
646                };
647                (generics, None)
648            }
649            PathResolutionItem::Method(_, None, _)
650            | PathResolutionItem::TraitFunction(_, None, _)
651            | PathResolutionItem::Module(..)
652            | PathResolutionItem::Type(..)
653            | PathResolutionItem::TypeAlias(..)
654            | PathResolutionItem::PrimitiveType(..)
655            | PathResolutionItem::Trait(..)
656            | PathResolutionItem::TraitAssociatedType(..)
657            | PathResolutionItem::Global(..)
658            | PathResolutionItem::EnumVariant(..)
659            | PathResolutionItem::ModuleFunction(..)
660            | PathResolutionItem::TraitConstant(..) => (Vec::new(), None),
661        };
662        self.push_errors(errors);
663        result
664    }

667    #[tracing::instrument(level = "trace", skip_all)]

671        let wildcard_allowed = WildcardAllowed::Yes;

676    /// Variant of [`Self::elaborate_type_path_impl_inner`] that accepts unresolved generics.
677    #[tracing::instrument(level = "trace", skip_all)]
678    fn elaborate_type_path_impl(
679        &mut self,
680        typ: Type,
681        ident: Ident,
682        turbofish: Option<GenericTypeArgs>,
683        typ_location: Location,
684    ) -> (ExprId, Type) {
685        let ident_location = ident.location();
686        let check_self_param = false;
687
688        self.interner.push_type_ref_location(&typ, typ_location);
689
690        let Some(method) = self.lookup_method(
691            &typ,
692            ident.as_str(),
693            ident_location,
694            typ_location,
695            check_self_param,
696        ) else {
697            let error = Expression::new(ExpressionKind::Error, ident_location);
698            return self.elaborate_expression(error);
699        };
700
701        let func_id = method
702            .func_id(self.interner)
703            .expect("Expected trait function to be a DefinitionKind::Function");
704
705        let generics =
706            turbofish.map(|turbofish| self.use_type_args(turbofish, func_id, ident_location).0);
707
708        self.elaborate_type_path_impl_inner(&typ, typ_location, ident_location, method, generics)
709    }

746    /// Common implementation for type path impl variants.
747    #[tracing::instrument(level = "trace", skip_all)]
748    fn elaborate_type_path_impl_inner(
749        &mut self,
750        typ: &Type,
751        _typ_location: Location,
752        ident_location: Location,
753        method: HirMethodReference,
754        generics: Option<Vec<Type>>,
755    ) -> (ExprId, Type) {
756        let func_id = method
757            .func_id(self.interner)
758            .expect("Expected trait function to be a DefinitionKind::Function");
759
760        let id = self.interner.function_definition_id(func_id);
761
762        let impl_kind = match method {
763            HirMethodReference::FuncId(_) => ImplKind::NotATraitMethod,
764            HirMethodReference::TraitItemId(HirTraitMethodReference {
765                definition,
766                trait_id,
767                trait_generics,
768                assumed: _,
769            }) => {
770                let mut constraint =
771                    self.interner.get_trait(trait_id).as_constraint(ident_location);
772                constraint.trait_bound.trait_generics = trait_generics;
773                ImplKind::TraitItem(TraitItem { definition, constraint, assumed: false })
774            }
775        };
776
777        let ident = HirIdent { location: ident_location, id, impl_kind };
778        let id =
779            self.intern_expr(HirExpression::Ident(ident.clone(), generics.clone()), ident_location);
780
781        // If the method has a self type (it's an impl or trait impl), bind `typ` to the instantiated self type.
782        let (self_type_generics_count, function_typ, function_self_type) = self
783            .with_function_meta(func_id, |meta| {
784                (meta.impl_generics_count(), meta.typ.clone(), meta.self_type.clone())
785            });
786        let bindings = if self_type_generics_count > 0 {
787            if let Type::Forall(type_vars, _) = &function_typ
788                && let Some(self_type) = &function_self_type
789            {
790                // Only instantiate type vars corresponding to the `self` type, not to function direct generics
791                let type_vars =
792                    type_vars.iter().take(self_type_generics_count).cloned().collect::<Vec<_>>();
793                let (self_type, instantiation_bindings) =
794                    self_type.substitute_type_vars_with_fresh_type_vars(&type_vars, self.interner);
795                let _ = typ.unify(&self_type);
796                instantiation_bindings
797            } else {
798                TypeBindings::default()
799            }
800        } else {
801            TypeBindings::default()
802        };
803
804        let typ = self.type_check_variable_with_bindings(ident, &id, generics, bindings);
805        let id = self.intern_expr_type(id, typ.clone());
806
807        (id, typ)
808    }

810    /// Given an [`HirIdent`], look up its definition, and:
811    /// * mark it as referenced at the ident [Location] (LSP mode only)
812    /// * mark the item currently being elaborated as a dependency of it
813    /// * elaborate a global definition, if needed
814    /// * add local identifiers to lambda captures
815    #[tracing::instrument(level = "trace", skip_all)]
816    pub(crate) fn handle_definition_id(&mut self, definition_id: DefinitionId, location: Location) {
817        match self.interner.definition(definition_id).kind {
818            DefinitionKind::Function(func_id) => {
819                if let Some(current_item) = self.current_item {
820                    self.interner.add_function_dependency(current_item, func_id);
821                }
822
823                self.interner.add_function_reference(func_id, location);
824            }
825            DefinitionKind::Global(global_id) => {
826                self.elaborate_global_if_unresolved(&global_id);
827                if let Some(current_item) = self.current_item {
828                    self.interner.add_global_dependency(current_item, global_id);
829                }
830
831                let global = self.interner.get_global_definition(global_id);
832                if global.comptime && !self.in_comptime_context() {
833                    self.push_err(ResolverError::ComptimeGlobalInNonComptimeCode {
834                        location,
835                        name: global.name.clone(),
836                    });
837                }
838
839                self.interner.add_global_reference(global_id, location);
840            }
841            DefinitionKind::NumericGeneric(_, ref numeric_typ) => {
842                // Initialize numeric generics to a polymorphic integer type in case
843                // they're used in expressions. We must do this here since type_check_variable
844                // does not check definition kinds and otherwise expects parameters to
845                // already be typed.
846                if self.interner.definition_type(definition_id) == Type::Error {
847                    let type_var_kind = Kind::Numeric(numeric_typ.clone());
848                    let typ = self.type_variable_with_kind(type_var_kind);
849                    self.interner.push_definition_type(definition_id, typ);
850                }
851            }
852            DefinitionKind::Local(_) => {
853                // Handled separately in `handle_local_variable`
854            }
855            DefinitionKind::AssociatedConstant(..) => {
856                // Nothing to do here
857            }
858        }
859    }

869    /// and a corresponding identifier, returning the instantiated [Type].
870    #[tracing::instrument(level = "trace", skip_all)]

872        &mut self,

874        expr_id: &PushedExpr<HasLocation>,

877        let bindings = TypeBindings::default();

881    /// Perform the type checking of an interned expression and a corresponding identifier,
882    /// returning the instantiated [Type].
883    ///
884    /// The instantiation bindings are pushed as required type variables on the current
885    /// function context, to be checked at end-of-function. `push_required_type_variable`
886    /// already skips this in a comptime context, where unbound generics on quoted typed
887    /// expressions are expected.
888    #[tracing::instrument(level = "trace", skip_all)]
889    pub(crate) fn type_check_variable_with_bindings(
890        &mut self,
891        ident: HirIdent,
892        expr_id: &PushedExpr<HasLocation>,
893        generics: Option<Vec<Type>>,
894        mut bindings: TypeBindings,
895    ) -> Type {
896        // Add type bindings from any constraints that were used.
897        // We need to do this first since otherwise instantiating the type below
898        // will replace each trait generic with a fresh type variable, rather than
899        // the type used in the trait constraint (if it exists). See #4088.
900        if let ImplKind::TraitItem(method) = &ident.impl_kind {
901            self.bind_generics_from_trait_constraint(
902                &method.constraint,
903                method.assumed,
904                &mut bindings,
905            );
906        }
907
908        // If a global variable hasn't been defined yet, then we are most likely dealing with a self-dependency-cycle.
909        let definition = self.interner.definition(ident.id);
910        // Some associated constants also have Global as Kind, and they are not defined when look them up here; want to restrict to global `let` statements.
911        if self.in_comptime_context()
912            && definition.kind.is_global()
913            && self.interner.try_definition_type(ident.id).is_none()
914        {
915            self.push_err(ResolverError::DependencyCycle {
916                location: ident.location,
917                item: definition.name.clone(),
918                cycle: "the variable definition type hasn't been resolved yet".to_string(),
919            });
920            return Type::Error;
921        }
922
923        let func_id = match definition.kind {
924            DefinitionKind::Function(func_id) => Some(func_id),
925            _ => None,
926        };
927
928        // If the variable is a function whose meta hasn't been resolved yet, resolve
929        // it now. This handles forward references — a global's RHS may name a
930        // function whose meta would otherwise only be drained at end-of-elaboration.
931        if let Some(func_id) = func_id {
932            let item_name = definition.name.clone();
933            self.define_function_meta_if_undefined(func_id);
934
935            // If lazy resolution leaves the meta still unset, the function is currently
936            // mid-resolution and we have a dependency cycle (e.g. `global F = f();`
937            // combined with `fn f(_: [u8; F]) {}`).
938            if self.interner.try_function_meta(&func_id).is_none() {
939                self.push_err(ResolverError::DependencyCycle {
940                    location: ident.location,
941                    item: item_name,
942                    cycle: "the function signature hasn't been resolved yet".to_string(),
943                });
944                return Type::Error;
945            }
946        }
947
948        // An identifiers type may be forall-quantified in the case of generic functions.
949        // E.g. `fn foo<T>(t: T, field: Field) -> T` has type `forall T. fn(T, Field) -> T`.
950        // We must instantiate identifiers at every call site to replace this T with a new type
951        // variable to handle generic functions.
952        let t = self.type_substitute_trait_as_type(&ident);
953
954        let direct_generic_ids = match func_id {
955            Some(function) => vecmap(&self.function_meta(function).direct_generics, |generic| {
956                generic.type_var.id()
957            }),
958            None => Vec::new(),
959        };
960
961        let location = self.interner.expr_location(expr_id);
962
963        // This instantiates a trait's generics as well which need to be set
964        // when the constraint below is later solved for when the function is
965        // finished. How to link the two?
966        let (typ, bindings) =
967            self.instantiate(t, bindings, generics, &direct_generic_ids, location);
968
969        if let ImplKind::TraitItem(mut method) = ident.impl_kind {
970            method.constraint.apply_bindings(&bindings);
971            if method.assumed {
972                let trait_generics = method.constraint.trait_bound.trait_generics.clone();
973                let object_type = method.constraint.typ;
974                let trait_impl = TraitImplKind::Assumed { object_type, trait_generics };
975                self.interner.select_impl_for_expression(**expr_id, trait_impl);
976            } else {
977                // this constraint should lead to choosing a trait impl method
978                self.push_trait_constraint(method.constraint, **expr_id, true);
979            }
980        }
981
982        // Push any trait constraints required by this definition to the context
983        // to be checked later when the type of this variable is further constrained.
984        //
985        // This must be done before the above trait constraint in case the above one further
986        // restricts types.
987        //
988        // For example, in this code:
989        //
990        // ```noir
991        // trait One {}
992        //
993        // trait Two<O: One> {
994        //     fn new() -> Self;
995        // }
996        //
997        // fn foo<X: One, T: Two<X>>() {
998        //     let _: T = Two::new();
999        // }
1000        // ```
1001        //
1002        // when type-checking `Two::new` we'll have a return type `'2` which is constrained by `'2: Two<'1>`.
1003        // Then the definition for `new` has a constraint on it, `O: One`, which translates to `'1: One`.
1004        //
1005        // Because of the explicit type in the `let`, `'2` will be unified with `T`.
1006        // Then we must first verify the constraint `'2: Two<'1>`, which is now `T: Two<'1>`, to find
1007        // that the implementation is the assumed one `T: Two<X>` so that `'1` is bound to `X`.
1008        // Then we can successfully verify the constraint `'1: One` which now became `X: One` which holds
1009        // because of the assumed constraint.
1010        //
1011        // If we try to find a trait implementation for `'1` before finding one for `'2` we'll never find it.
1012        if let Some(function) = func_id {
1013            let function = self.function_meta(function);
1014            for mut constraint in function.all_trait_constraints().cloned().collect::<Vec<_>>() {
1015                constraint.apply_bindings(&bindings);
1016
1017                // This constraint shouldn't lead to choosing a trait impl method
1018                self.push_trait_constraint(constraint, **expr_id, false);
1019            }
1020        }
1021
1022        // Record required type variables in a predictable order to avoid nondeterminism in error messages.
1023        let required_type_variables = btree_map(bindings.values(), |(type_variable, _, typ)| {
1024            (type_variable.id(), typ.clone())
1025        });
1026
1027        for (type_variable_id, typ) in required_type_variables {
1028            self.push_required_type_variable(
1029                type_variable_id,
1030                typ,
1031                BindableTypeVariableKind::Ident(ident.id),
1032                ident.location,
1033            );
1034        }
1035
1036        self.interner.store_instantiation_bindings(**expr_id, bindings);
1037        typ
1038    }
1039
1040    /// If the type of the [`HirIdent`] is a function that returns an `impl Trait`,
1041    /// then it might need elaboration before it can be substituted to a [Type].
1042    /// Try to elaborate it now.
1043    ///
1044    /// Returns a type error if the callee cannot be resolved on a second try,
1045    /// which indicates a dependency cycle.
1046    #[tracing::instrument(level = "trace", skip_all)]
1047    fn type_substitute_trait_as_type(&mut self, ident: &HirIdent) -> Type {
1048        let func_id = match self.interner.id_type_substitute_trait_as_type(ident.id) {
1049            Ok(typ) => return typ,
1050            Err(func_id) => func_id,
1051        };
1052
1053        // Try to elaborate, so we get an expression for the body.
1054        self.elaborate_function(func_id);
1055
1056        // Now try again. If it's still not working, give up.
1057        match self.interner.id_type_substitute_trait_as_type(ident.id) {
1058            Ok(typ) => typ,
1059            Err(_) => {
1060                let def = self.interner.definition(ident.id);
1061                self.push_err(ResolverError::DependencyCycle {
1062                    location: ident.location,
1063                    item: def.name.clone(),
1064                    cycle: "'impl Trait' could not be resolved to the type of the function body"
1065                        .to_string(),
1066                });
1067                Type::Error
1068            }
1069        }
1070    }

1072    /// Instantiate a [Type] with the given [`TypeBindings`], returning the bindings potentially
1073    /// extended from any turbofish generics.
1074    ///
1075    /// If there are turbofish generics and their number matches the expectations of the function,
1076    /// those are used as well, otherwise they are ignored and an error is pushed.
1077    #[tracing::instrument(level = "trace", skip_all)]
1078    fn instantiate(
1079        &mut self,
1080        typ: Type,
1081        bindings: TypeBindings,
1082        turbofish_generics: Option<Vec<Type>>,
1083        direct_generic_ids: &[TypeVariableId],
1084        location: Location,
1085    ) -> (Type, TypeBindings) {
1086        match turbofish_generics {
1087            Some(turbofish_generics) => {
1088                let function_generic_count = direct_generic_ids.len();
1089                let forall_generic_count =
1090                    if let Type::Forall(generics, _) = &typ { generics.len() } else { 0 };
1091
1092                if turbofish_generics.len() != function_generic_count {
1093                    let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
1094                        expected_count: function_generic_count,
1095                        actual_count: turbofish_generics.len(),
1096                        location,
1097                    };
1098                    self.push_err(CompilationError::TypeError(type_check_err));
1099                    typ.instantiate_with_bindings(bindings, self.interner)
1100                } else if forall_generic_count < function_generic_count {
1101                    // The next branch asserts that the number of turbofish-bound typevars matches
1102                    // `direct_generic_ids`, but if the forall has fewer generics than the function
1103                    // (e.g. when the function's generics have duplicates that were filtered out)
1104                    // this can never hold. A duplicate error has already been reported elsewhere,
1105                    // so bail out silently.
1106                    self.push_err(TypeCheckError::expecting_other_error(
1107                        "forall has fewer generics than function",
1108                        location,
1109                    ));
1110                    (Type::Error, bindings)
1111                } else {
1112                    typ.instantiate_with_bindings_and_turbofish(
1113                        bindings,
1114                        turbofish_generics,
1115                        self.interner,
1116                        direct_generic_ids,
1117                    )
1118                }
1119            }
1120            None => typ.instantiate_with_bindings(bindings, self.interner),
1121        }
1122    }
1123}

1125/// Returns the name of the data type a [Type] resolves to, looking through a `Forall` quantifier.
1126fn data_type_name(typ: &Type) -> Option<String> {
1127    match typ {
1128        Type::Forall(_, body) => data_type_name(body),
1129        Type::DataType(datatype, _) => Some(datatype.borrow().name.to_string()),
1130        _ => None,
1131    }
1132}

1134/// Bind the generics of the [Type] aliased by the [`TypeAlias`] to a list of generic arguments,
1135/// recursively expanding the generics aliased aliases, finally returning the generics of the
1136/// innermost aliased struct.
1137///
1138/// Panics if it encounters a type other than alias or struct.
1139fn get_type_alias_generics(type_alias: &TypeAlias, generics: &[Type]) -> Vec<Type> {
1140    let typ = type_alias.get_type(generics);
1141    match typ {
1142        Type::DataType(_, generics) => generics,
1143        Type::Alias(type_alias, generics) => {
1144            get_type_alias_generics(&type_alias.borrow(), &generics)
1145        }
1146        // Primitive types have no generics
1147        _ => Vec::new(),
1148    }
1149}
