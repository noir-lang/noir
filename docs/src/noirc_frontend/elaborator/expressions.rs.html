57    pub(crate) fn elaborate_expression_with_target_type(
58        &mut self,
59        expr: Expression,
60        target_type: Option<&Type>,
61    ) -> (ExprId, Type) {
62        let ((id, typ), has_errors) =
63            self.with_error_guard(|this| this.elaborate_expression_inner(expr, target_type));
64
65        if has_errors {
66            self.interner.exprs_with_errors.insert(id);
67        }
68
69        (id, typ)
70    }

72    fn elaborate_expression_inner(
73        &mut self,
74        expr: Expression,
75        target_type: Option<&Type>,
76    ) -> (ExprId, Type) {
77        let is_integer_literal = matches!(expr.kind, ExpressionKind::Literal(Literal::Integer(..)));
78
79        let (hir_expr, typ) = match expr.kind {
80            ExpressionKind::Literal(literal) => self.elaborate_literal(literal, expr.location),
81            ExpressionKind::Block(block) => self.elaborate_block(block, target_type),
82            ExpressionKind::Prefix(prefix) => return self.elaborate_prefix(*prefix, expr.location),
83            ExpressionKind::Index(index) => self.elaborate_index(*index),
84            ExpressionKind::Call(call) => self.elaborate_call(*call, expr.location),
85            ExpressionKind::MethodCall(call) => self.elaborate_method_call(*call, expr.location),
86            ExpressionKind::Constrain(constrain) => self.elaborate_constrain(constrain),
87            ExpressionKind::Constructor(constructor) => self.elaborate_constructor(*constructor),
88            ExpressionKind::MemberAccess(access) => {
89                let (expr_id, typ, _) = self.elaborate_member_access(*access, expr.location, false);
90                return (expr_id, typ);
91            }
92            ExpressionKind::Cast(cast) => self.elaborate_cast(*cast, expr.location),
93            ExpressionKind::Infix(infix) => return self.elaborate_infix(*infix, expr.location),
94            ExpressionKind::If(if_) => self.elaborate_if(*if_, target_type),
95            ExpressionKind::Match(match_) => self.elaborate_match(*match_, expr.location),
96            ExpressionKind::Variable(variable) => return self.elaborate_variable(variable),
97            ExpressionKind::Tuple(tuple) => self.elaborate_tuple(tuple, target_type),
98            ExpressionKind::Lambda(lambda) => {
99                self.elaborate_lambda_with_target_type(*lambda, target_type)
100            }
101            ExpressionKind::Parenthesized(expr) => {
102                return self.elaborate_expression_with_target_type(*expr, target_type);
103            }
104            ExpressionKind::Quote(quote) => self.elaborate_quote(quote, expr.location),
105            ExpressionKind::Comptime(block, _) => {
106                if self.in_comptime_context {
107                    // Treat a nested comptime block as a regular block. Nested comptime blocks
108                    // can happen as a result of macro expansion so it wouldn't be good to produce
109                    // a warning in that case.
110                    self.elaborate_block(block, target_type)
111                } else {
112                    return self.elaborate_comptime_block(block, expr.location, target_type);
113                }
114            }
115            ExpressionKind::Unsafe(unsafe_expression) => {
116                self.elaborate_unsafe_block(unsafe_expression, target_type)
117            }
118            ExpressionKind::Resolved(id) => return (id, self.interner.id_type(id)),
119            ExpressionKind::Interned(id) => {
120                let expr_kind = self.interner.get_expression_kind(id);
121                let expr = Expression::new(expr_kind.clone(), expr.location);
122                return self.elaborate_expression(expr);
123            }
124            ExpressionKind::InternedStatement(id) => {
125                return self.elaborate_interned_statement_as_expr(id, expr.location);
126            }
127            ExpressionKind::Error => (HirExpression::Error, Type::Error),
128            ExpressionKind::Unquote(_) => {
129                self.push_err(ResolverError::UnquoteUsedOutsideQuote { location: expr.location });
130                (HirExpression::Error, Type::Error)
131            }
132            ExpressionKind::AsTraitPath(path) => {
133                return self.elaborate_as_trait_path(*path);
134            }
135            ExpressionKind::TypePath(path) => return self.elaborate_type_path(*path),
136        };
137        let id = self.interner.push_expr_full(hir_expr, expr.location, typ.clone());
138
139        if is_integer_literal {
140            self.push_integer_literal_expr_id(id);
141        }
142
143        (id, typ)
144    }

146    /// Elaborate a member access expression without adding the automatic dereferencing operations
147    /// to it, treating it as an offset instead. This also returns a boolean indicating whether the
148    /// result skipped any required auto-dereferences (and thus needs dereferencing to be used as a value
149    /// instead of a reference). This flag is used when `&mut foo.bar.baz` is used to cancel out
150    /// the `&mut`.
151    fn elaborate_reference_expression(&mut self, expr: Expression) -> (ExprId, Type, bool) {
152        match expr.kind {
153            ExpressionKind::MemberAccess(access) => {
154                self.elaborate_member_access(*access, expr.location, true)
155            }
156            _ => {
157                let (expr_id, typ) = self.elaborate_expression(expr);
158                (expr_id, typ, false)
159            }
160        }
161    }

163    /// Given its ID, retrieve and elaborate an interned [StatementKind].
164    fn elaborate_interned_statement_as_expr(
165        &mut self,
166        id: InternedStatementKind,
167        location: Location,
168    ) -> (ExprId, Type) {
169        match self.interner.get_statement_kind(id) {
170            StatementKind::Expression(expr) | StatementKind::Semi(expr) => {
171                self.elaborate_expression(expr.clone())
172            }
173            StatementKind::Interned(id) => self.elaborate_interned_statement_as_expr(*id, location),
174            StatementKind::Error => {
175                let expr = Expression::new(ExpressionKind::Error, location);
176                self.elaborate_expression(expr)
177            }
178            other => {
179                let statement = other.to_string();
180                self.push_err(ResolverError::InvalidInternedStatementInExpr {
181                    statement,
182                    location,
183                });
184                let expr = Expression::new(ExpressionKind::Error, location);
185                self.elaborate_expression(expr)
186            }
187        }
188    }

199    fn elaborate_block_expression(
200        &mut self,
201        block: BlockExpression,
202        target_type: Option<&Type>,
203    ) -> (HirBlockExpression, Type) {
204        self.push_scope();
205        let mut block_type = Type::Unit;
206        let statements_len = block.statements.len();
207        let mut statements = Vec::with_capacity(statements_len);
208
209        // If we found a break or continue statement, this holds its location (only for the first one)
210        let mut break_or_continue_location = None;
211        // When encountering a statement after a break or continue we'll error saying it's unreachable,
212        // but we only want to error for the first statement.
213        let mut errored_unreachable = false;
214
215        for (i, statement) in block.statements.into_iter().enumerate() {
216            let location = statement.location;
217            let statement_target_type = if i == statements_len - 1 { target_type } else { None };
218            let (id, stmt_type) =
219                self.elaborate_statement_with_target_type(statement, statement_target_type);
220
221            statements.push(id);
222
223            let stmt = self.interner.statement(&id);
224
225            if let HirStatement::Semi(expr) = stmt {
226                let inner_expr_type = self.interner.id_type(expr);
227                let location = self.interner.expr_location(&expr);
228
229                self.unify(&inner_expr_type, &Type::Unit, || {
230                    let expr_type = inner_expr_type.clone();
231                    let expr_location = location;
232
233                    if let MustUse::MustUse(message) = Self::type_is_must_use(&expr_type) {
234                        TypeCheckError::UnusedResultError { expr_type, expr_location, message }
235                    } else {
236                        TypeCheckError::UnusedResultWarning { expr_type, expr_location }
237                    }
238                });
239            }
240
241            let is_break_or_continue = matches!(stmt, HirStatement::Break | HirStatement::Continue);
242
243            if let Some(break_or_continue_location) = break_or_continue_location {
244                if !errored_unreachable {
245                    self.push_err(ResolverError::UnreachableStatement {
246                        location,
247                        break_or_continue_location,
248                    });
249                    errored_unreachable = true;
250                }
251            } else if is_break_or_continue {
252                break_or_continue_location = Some(location);
253            }
254
255            if i + 1 == statements.len() {
256                block_type = stmt_type;
257            }
258        }
259
260        self.pop_scope();
261        (HirBlockExpression { statements }, block_type)
262    }

264    /// If the given type was declared as:
265    /// - `#[must_use = "message"]`, return [MustUse::MustUse(Some("message"))]
266    /// - `#[must_use]`, return [MustUse::MustUse(None)]
267    /// - otherwise, return `MustUse::NoMustUse`
268    fn type_is_must_use(typ: &Type) -> MustUse {
269        /// Helper function to avoid infinite recursion for infinitely recursive types
270        fn helper(typ: &Type, fuel: u32) -> MustUse {
271            if fuel == 0 {
272                return MustUse::NoMustUse;
273            }
274            let fuel = fuel - 1;
275            match typ.follow_bindings_shallow().as_ref() {
276                Type::DataType(data_type, _generics) => data_type.borrow().must_use.clone(),
277                // If any element in the tuple is `#[must_use]`, the whole tuple is
278                Type::Tuple(elements) => {
279                    for element in elements {
280                        if let MustUse::MustUse(message) = helper(element, fuel) {
281                            return MustUse::MustUse(message);
282                        }
283                    }
284                    MustUse::NoMustUse
285                }
286                Type::Alias(alias, generics) => helper(&alias.borrow().get_type(generics), fuel),
287                Type::CheckedCast { to, .. } => helper(to.as_ref(), fuel),
288                Type::Reference(element, _) => helper(element.as_ref(), fuel),
289                _ => MustUse::NoMustUse,
290            }
291        }
292
293        // 10 is an arbitrary maximum bound on recursion through `Type`s here
294        // in case an infinitely recursive type is used. In practice most types should
295        // require just 1 iteration, or up to 3 for a reference to an aliased type.
296        helper(typ, 10)
297    }

299    fn elaborate_unsafe_block(
300        &mut self,
301        unsafe_expression: UnsafeExpression,
302        target_type: Option<&Type>,
303    ) -> (HirExpression, Type) {
304        use UnsafeBlockStatus::*;
305        // Before entering the block we cache the old value of the unsafe block status, so it can be restored.
306        let old_in_unsafe_block = self.unsafe_block_status;
307        let is_nested_unsafe_block = !matches!(old_in_unsafe_block, NotInUnsafeBlock);
308
309        if is_nested_unsafe_block {
310            self.push_err(TypeCheckError::NestedUnsafeBlock {
311                location: unsafe_expression.unsafe_keyword_location,
312            });
313        }
314
315        self.unsafe_block_status = InUnsafeBlockWithoutUnconstrainedCalls;
316
317        let (hir_block_expression, typ) =
318            self.elaborate_block_expression(unsafe_expression.block, target_type);
319
320        let has_unconstrained_call =
321            matches!(self.unsafe_block_status, InUnsafeBlockWithUnconstrainedCalls);
322
323        if !has_unconstrained_call {
324            self.push_err(TypeCheckError::UnnecessaryUnsafeBlock {
325                location: unsafe_expression.unsafe_keyword_location,
326            });
327        }
328
329        // Finally, we restore the original value of the unsafe block status,
330        // unless we are in a nested block and we have found an unconstrained call,
331        // in which case we should consider the outer block as having that call as well.
332        if !is_nested_unsafe_block || !has_unconstrained_call {
333            self.unsafe_block_status = old_in_unsafe_block;
334        }
335
336        (HirExpression::Unsafe(hir_block_expression), typ)
337    }

339    fn elaborate_literal(&mut self, literal: Literal, location: Location) -> (HirExpression, Type) {
340        use HirExpression::Literal as Lit;
341        match literal {
342            Literal::Unit => (Lit(HirLiteral::Unit), Type::Unit),
343            Literal::Bool(b) => (Lit(HirLiteral::Bool(b)), Type::Bool),
344            Literal::Integer(integer, suffix) => {
345                (Lit(HirLiteral::Integer(integer)), self.integer_suffix_type(suffix))
346            }
347            Literal::Str(str) | Literal::RawStr(str, _) => {
348                let len: u32 = str.len().try_into().expect(
349                    "ICE: Elaborator::elaborate_literal: str.len() is expected to fit into a u32",
350                );
351                let len = len.into();
352                (Lit(HirLiteral::Str(str)), Type::String(Box::new(len)))
353            }
354            Literal::FmtStr(fragments, length) => self.elaborate_fmt_string(fragments, length),
355            Literal::Array(array_literal) => {
356                self.elaborate_array_literal(array_literal, location, true)
357            }
358            Literal::Vector(array_literal) => {
359                self.elaborate_array_literal(array_literal, location, false)
360            }
361        }
362    }

382    fn elaborate_array_literal(
383        &mut self,
384        array_literal: ArrayLiteral,
385        location: Location,
386        is_array: bool,
387    ) -> (HirExpression, Type) {
388        let (expr, elem_type, length) = match array_literal {
389            ArrayLiteral::Standard(elements) => {
390                let type_variable_id = self.interner.next_type_variable_id();
391                let type_variable = TypeVariable::unbound(type_variable_id, Kind::Any);
392                self.push_required_type_variable(
393                    type_variable.id(),
394                    Type::TypeVariable(type_variable.clone()),
395                    BindableTypeVariableKind::ArrayLiteral { is_array },
396                    location,
397                );
398
399                let first_elem_type = Type::TypeVariable(type_variable);
400                let first_location = elements.first().map(|elem| elem.location).unwrap_or(location);
401
402                let elements = vecmap(elements.into_iter().enumerate(), |(i, elem)| {
403                    let location = elem.location;
404                    let (elem_id, elem_type) = self.elaborate_expression(elem);
405
406                    self.unify(&elem_type, &first_elem_type, || {
407                        TypeCheckError::NonHomogeneousArray {
408                            first_location,
409                            first_type: first_elem_type.to_string(),
410                            first_index: 0,
411                            second_location: location,
412                            second_type: elem_type.to_string(),
413                            second_index: i,
414                        }
415                        .add_context("elements in an array must have the same type")
416                    });
417                    elem_id
418                });
419
420                let length: u32 = elements.len().try_into().expect("ICE: Elaborator::elaborate_array_literal: elements.len() is expected to fit into a u32");
421                let length = length.into();
422                (HirArrayLiteral::Standard(elements), first_elem_type, length)
423            }
424            ArrayLiteral::Repeated { repeated_element, length } => {
425                let location = length.location;
426                let length = UnresolvedTypeExpression::from_expr(*length, location).unwrap_or_else(
427                    |error| {
428                        self.push_err(ResolverError::ParserError(Box::new(error)));
429                        UnresolvedTypeExpression::Constant(SignedField::zero(), None, location)
430                    },
431                );
432
433                let wildcard_allowed = WildcardAllowed::Yes;
434                let length =
435                    self.convert_expression_type(length, &Kind::u32(), location, wildcard_allowed);
436                let (repeated_element, elem_type) = self.elaborate_expression(*repeated_element);
437
438                let length_clone = length.clone();
439                (HirArrayLiteral::Repeated { repeated_element, length }, elem_type, length_clone)
440            }
441        };
442        let constructor = if is_array { HirLiteral::Array } else { HirLiteral::Vector };
443        let elem_type = Box::new(elem_type);
444        let typ = if is_array {
445            Type::Array(Box::new(length), elem_type)
446        } else {
447            Type::Vector(elem_type)
448        };
449        (HirExpression::Literal(constructor(expr)), typ)
450    }

452    fn elaborate_fmt_string(
453        &mut self,
454        fragments: Vec<FmtStrFragment>,
455        length: u32,
456    ) -> (HirExpression, Type) {
457        let mut fmt_str_idents = Vec::new();
458        let mut capture_types = Vec::new();
459
460        for fragment in &fragments {
461            if let FmtStrFragment::Interpolation(ident_name, location) = fragment {
462                let ((hir_ident, var_scope_index), _) = self
463                    .get_ident_from_path(TypedPath::from_single(ident_name.to_string(), *location));
464                self.handle_hir_ident(&hir_ident, var_scope_index, *location);
465
466                let hir_expr = HirExpression::Ident(hir_ident.clone(), None);
467                let expr_id = self.intern_expr(hir_expr, *location);
468                let typ = self.type_check_variable(hir_ident, &expr_id, None);
469                let expr_id = self.intern_expr_type(expr_id, typ.clone());
470
471                capture_types.push(typ);
472                fmt_str_idents.push(expr_id);
473            }
474        }
475
476        let len = length.into();
477        let fmtstr_type =
478            if capture_types.is_empty() { Type::Unit } else { Type::Tuple(capture_types) };
479        let typ = Type::FmtString(Box::new(len), Box::new(fmtstr_type));
480        (HirExpression::Literal(HirLiteral::FmtStr(fragments, fmt_str_idents, length)), typ)
481    }

483    fn elaborate_prefix(&mut self, prefix: PrefixExpression, location: Location) -> (ExprId, Type) {
484        let rhs_location = prefix.rhs.location;
485        let operator = prefix.operator;
486
487        let (rhs, rhs_type, skip_op) = if matches!(operator, UnaryOp::Reference { .. }) {
488            let (rhs, rhs_type, needs_deref) = self.elaborate_reference_expression(prefix.rhs);
489
490            // If the reference expression delayed a needed deref, we can skip the `&mut _`
491            // operation since the expression is already a reference.
492            (rhs, rhs_type, needs_deref)
493        } else {
494            let (rhs, rhs_type) = self.elaborate_expression(prefix.rhs);
495            (rhs, rhs_type, false)
496        };
497
498        let trait_method_id = self.interner.get_prefix_operator_trait_method(&operator);
499
500        if let UnaryOp::Reference { mutable } = operator {
501            if mutable {
502                // If skip_op is set we already know we have a mutable reference
503                if !skip_op {
504                    self.check_can_mutate(rhs, rhs_location);
505                }
506            } else {
507                self.use_unstable_feature(UnstableFeature::Ownership, location);
508            }
509        }
510
511        let expr = HirExpression::Prefix(HirPrefixExpression {
512            operator,
513            rhs,
514            trait_method_id,
515            skip: skip_op,
516        });
517        let expr_id = self.intern_expr(expr, location);
518
519        // If `skip_op` is set we already know we have a mutable reference due to a member access on a mutable reference.
520        // The prefix operand type rules will return the result of a prefix operation.
521        // We do not want to check the prefix operand type rules as we will then get a type mismatch.
522        let typ = if skip_op {
523            rhs_type
524        } else {
525            let result = self.prefix_operand_type_rules(&operator, &rhs_type, location);
526            let not_ord = false;
527            self.handle_operand_type_rules_result(
528                result,
529                &rhs_type,
530                trait_method_id,
531                *expr_id,
532                location,
533                not_ord,
534            )
535        };
536
537        let expr_id = self.intern_expr_type(expr_id, typ.clone());
538        (expr_id, typ)
539    }

541    /// Check whether we can create a mutable reference over an expression.
542    ///
543    /// Pushes an error if it cannot be done.
544    pub(super) fn check_can_mutate(&mut self, expr_id: ExprId, location: Location) {
545        match self.interner.expression(&expr_id) {
546            HirExpression::Ident(hir_ident, _) => {
547                if let Some(definition) = self.interner.try_definition(hir_ident.id) {
548                    let name = definition.name.clone();
549                    if !definition.mutable {
550                        self.push_err(TypeCheckError::CannotMutateImmutableVariable {
551                            name,
552                            location,
553                        });
554                    } else {
555                        self.check_can_mutate_lambda_capture(hir_ident.id, name, location);
556                    }
557                }
558            }
559            HirExpression::Index(_) => {
560                self.push_err(TypeCheckError::MutableReferenceToArrayElement { location });
561            }
562            HirExpression::MemberAccess(member_access) => {
563                self.check_can_mutate(member_access.lhs, location);
564            }
565            _ => (),
566        }
567    }

569    /// We must check whether the mutable variable we are attempting to mutate
570    /// comes from a lambda capture. All captures are immutable so we want to error
571    /// if the user attempts to mutate a captured variable inside of a lambda without
572    /// having captured a mutable reference.
573    ///
574    /// Pushes an error if the mutation is illegal.
575    pub(super) fn check_can_mutate_lambda_capture(
576        &mut self,
577        id: DefinitionId,
578        name: String,
579        location: Location,
580    ) {
581        if let Some(lambda_context) = self.lambda_stack.last() {
582            let typ = self.interner.definition_type(id);
583            if !typ.is_mutable_ref() && lambda_context.captures.iter().any(|var| var.ident.id == id)
584            {
585                self.push_err(TypeCheckError::MutableCaptureWithoutRef { name, location });
586            }
587        }
588    }

590    fn elaborate_index(&mut self, index_expr: IndexExpression) -> (HirExpression, Type) {
591        let location = index_expr.index.location;
592
593        let (index, index_type) = self.elaborate_expression(index_expr.index);
594
595        let expected = Type::u32();
596        self.unify(&index_type, &expected, || TypeCheckError::TypeMismatchWithSource {
597            expected: expected.clone(),
598            actual: index_type.clone(),
599            location,
600            source: Source::ArrayIndex,
601        });
602
603        // When writing `a[i]`, if `a : &mut ...` then automatically dereference `a` as many
604        // times as needed to get the underlying array.
605        let lhs_location = index_expr.collection.location;
606        let (lhs, lhs_type) = self.elaborate_expression(index_expr.collection);
607        let (collection, lhs_type) = self.insert_auto_dereferences(lhs, lhs_type);
608
609        let typ = match lhs_type.follow_bindings() {
610            // XXX: We can check the array bounds here also, but it may be better to constant fold first
611            // and have ConstId instead of ExprId for constants
612            Type::Array(_, base_type) => *base_type,
613            Type::Vector(base_type) => *base_type,
614            Type::Error => Type::Error,
615            Type::TypeVariable(_) => {
616                self.push_err(TypeCheckError::TypeAnnotationsNeededForIndex {
617                    location: lhs_location,
618                });
619                Type::Error
620            }
621            typ => {
622                self.push_err(TypeCheckError::TypeMismatch {
623                    expected_typ: "Array".to_owned(),
624                    expr_typ: typ.to_string(),
625                    expr_location: lhs_location,
626                });
627                Type::Error
628            }
629        };
630
631        let expr = HirExpression::Index(HirIndexExpression { collection, index });
632        (expr, typ)
633    }

659    /// Helper function containing the elaboration logic for a call expression.
660    /// Returns the HIR call and its type.
661    fn elaborate_call_inner(
662        &mut self,
663        call: CallExpression,
664        location: Location,
665        is_macro_call: bool,
666    ) -> (HirCallExpression, Type) {
667        let (func, func_type) = self.elaborate_expression(*call.func);
668        let func_type = func_type.follow_bindings();
669
670        // Even if the function type is a Type::Error, we still want to elaborate the call's function arguments.
671        // Thus, we simply return None here for the argument types rather than returning early.
672        let (func_arg_types, unconstrained) =
673            if let Type::Function(args, _, _, unconstrained) = &func_type {
674                (Some(args), *unconstrained)
675            } else {
676                (None, false)
677            };
678
679        // When calling an unconstrained function, we can elaborate lambda arguments to be unconstrained.
680        let was_in_unconstrained_args =
681            std::mem::replace(&mut self.in_unconstrained_args, unconstrained);
682
683        let mut arguments = Vec::with_capacity(call.arguments.len());
684        let args = vecmap(call.arguments.into_iter().enumerate(), |(arg_index, arg)| {
685            let location = arg.location;
686            let expected_type = func_arg_types.and_then(|args| args.get(arg_index));
687
688            let (arg, typ) = if is_macro_call {
689                self.elaborate_in_comptime_context(|this| {
690                    this.elaborate_expression_with_target_type(arg, expected_type)
691                })
692            } else {
693                self.elaborate_expression_with_target_type(arg, expected_type)
694            };
695
696            // Try to unify this argument type against the function's argument type
697            // so that a potential lambda following this argument can have more concrete types.
698            if let Some(expected_type) = expected_type {
699                let _ = typ.unify(expected_type);
700            }
701
702            arguments.push(arg);
703            (typ, arg, location)
704        });
705
706        let hir_call = HirCallExpression { func, arguments, location, is_macro_call };
707        let typ = self.type_check_call(&hir_call, func_type, args, location);
708
709        // Restore the old one after type checking.
710        self.in_unconstrained_args = was_in_unconstrained_args;
711
712        (hir_call, typ)
713    }

741    /// Helper function containing the elaboration logic for a method call.
742    /// Returns the desugared function call and its type.
743    fn elaborate_method_call_inner(
744        &mut self,
745        method_call: MethodCallExpression,
746        location: Location,
747    ) -> (HirCallExpression, Type) {
748        let object_location = method_call.object.location;
749        let (mut object, mut object_type) = self.elaborate_expression(method_call.object);
750        object_type = object_type.follow_bindings();
751
752        let method_name_location = method_call.method_name.location();
753        let method_name = method_call.method_name.as_str();
754        let check_self_param = true;
755
756        let method_ref = self.lookup_method(
757            &object_type,
758            method_name,
759            location,
760            object_location,
761            check_self_param,
762        );
763        let Some(method_ref) = method_ref else {
764            // Return a dummy call expression with Error type
765            let error_func =
766                self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
767            let error_call = HirCallExpression {
768                func: error_func,
769                arguments: Vec::new(),
770                location,
771                is_macro_call: method_call.is_macro_call,
772            };
773            return (error_call, Type::Error);
774        };
775
776        // Automatically add `&mut` if the method expects a mutable reference and
777        // the object is not already one.
778        let func_id = method_ref
779            .func_id(self.interner)
780            .expect("Expected trait function to be a DefinitionKind::Function");
781
782        let function_type = self.interner.function_meta(&func_id).typ.clone();
783        self.try_add_mutable_reference_to_object(&function_type, &mut object_type, &mut object);
784
785        let generics = method_call.generics;
786        let generics = generics.map(|generics| {
787            vecmap(generics, |generic| {
788                let location = generic.location;
789                let wildcard_allowed = WildcardAllowed::Yes;
790                let typ = self.use_type_with_kind(generic, &Kind::Any, wildcard_allowed);
791                Located::from(location, typ)
792            })
793        });
794        let generics = self.resolve_function_turbofish_generics(&func_id, generics, location);
795
796        let location = object_location.merge(method_name_location);
797
798        let (function_id, function_name) = method_ref.clone().into_function_id_and_name(
799            object_type.clone(),
800            generics.clone(),
801            location,
802            self.interner,
803        );
804
805        let func_type =
806            self.type_check_variable(function_name.clone(), &function_id, generics.clone());
807
808        let function_id = self.intern_expr_type(function_id, func_type.clone());
809
810        let func_arg_types =
811            if let Type::Function(args, _, _, _) = &func_type { Some(args) } else { None };
812
813        // Try to unify the object type with the first argument of the function.
814        // The reason to do this is that many methods that take a lambda will yield `self` or part of `self`
815        // as a parameter. By unifying `self` with the first argument we'll potentially get more
816        // concrete types in the arguments that are function types, which will later be passed as
817        // lambda parameter hints.
818        if let Some(first_arg_type) = func_arg_types.and_then(|args| args.first()) {
819            let _ = first_arg_type.unify(&object_type);
820        }
821
822        // These arguments will be given to the desugared function call.
823        // Compared to the method arguments, they also contain the object.
824        let mut function_args = Vec::with_capacity(method_call.arguments.len() + 1);
825        let mut arguments = Vec::with_capacity(method_call.arguments.len());
826
827        function_args.push((object_type.clone(), object, object_location));
828
829        for (arg_index, arg) in method_call.arguments.into_iter().enumerate() {
830            let location = arg.location;
831            // The argument types also contain the object type as the first argument.
832            // Thus, we need to add one when indexing the argument types to match them up with method arguments.
833            let expected_type = func_arg_types.and_then(|args| args.get(arg_index + 1));
834            let (arg, typ) = self.elaborate_expression_with_target_type(arg, expected_type);
835
836            // Try to unify this argument type against the function's argument type
837            // so that a potential lambda following this argument can have more concrete types.
838            if let Some(expected_type) = expected_type {
839                let _ = expected_type.unify(&typ);
840            }
841
842            arguments.push(arg);
843            function_args.push((typ, arg, location));
844        }
845
846        let method = method_call.method_name;
847        let is_macro_call = method_call.is_macro_call;
848        let method_call = HirMethodCallExpression { method, object, arguments, location, generics };
849
850        self.check_method_call_visibility(func_id, &object_type, &method_call.method);
851
852        // Desugar the method call into a normal, resolved function call
853        // so that the backend doesn't need to worry about methods
854        let function_call = method_call.into_function_call(function_id, is_macro_call, location);
855
856        self.interner.add_function_reference(func_id, method_name_location);
857
858        // Type check the new call now that it has been changed from a method call
859        // to a function call. This way we avoid duplicating code.
860        let typ = self.type_check_call(&function_call, func_type, function_args, location);
861
862        (function_call, typ)
863    }

865    pub(super) fn elaborate_constrain(
866        &mut self,
867        mut expr: ConstrainExpression,
868    ) -> (HirExpression, Type) {
869        let location = expr.location;
870        let min_args_count = expr.kind.required_arguments_count();
871        let actual_args_count = expr.arguments.len();
872        let has_optional_msg = actual_args_count == min_args_count + 1;
873
874        let (message, expr) = if actual_args_count != min_args_count && !has_optional_msg {
875            self.push_err(TypeCheckError::AssertionParameterCountMismatch {
876                kind: expr.kind,
877                found: actual_args_count,
878                location,
879            });
880
881            // Given that we already produced an error, let's make this an `assert(true)` so
882            // we don't get further errors.
883            let message = None;
884            let kind = ExpressionKind::Literal(Literal::Bool(true));
885            let expr = Expression { kind, location };
886            (message, expr)
887        } else {
888            let message = has_optional_msg.then(|| expr.arguments.pop().unwrap());
889            let expr = match expr.kind {
890                ConstrainKind::Assert | ConstrainKind::Constrain => expr.arguments.pop().unwrap(),
891                ConstrainKind::AssertEq => {
892                    let rhs = expr.arguments.pop().unwrap();
893                    let lhs = expr.arguments.pop().unwrap();
894                    let location = lhs.location.merge(rhs.location);
895                    let operator = Located::from(location, BinaryOpKind::Equal);
896                    let kind =
897                        ExpressionKind::Infix(Box::new(InfixExpression { lhs, operator, rhs }));
898                    Expression { kind, location }
899                }
900            };
901            (message, expr)
902        };
903
904        let expr_location = expr.location;
905        let (expr_id, expr_type) = self.elaborate_expression(expr);
906
907        // Must type check the assertion message expression so that we instantiate bindings
908        let msg = message.map(|assert_msg_expr| {
909            let (msg, typ) = self.elaborate_expression(assert_msg_expr);
910            // If the error message contains a format string, those types need to appear in the ABI,
911            // except if we are in a meta-programming context, in which case the comptime interpreter
912            // handles a wider variety of types, e.g. quoted types.
913            if !self.in_comptime_context() {
914                let location = self.interner.expr_location(&msg);
915                let typ = typ.follow_bindings();
916                let mut check_msg_compat = |typ: &Type| {
917                    if typ.is_message_compatible(false) || matches!(typ, Type::Error) {
918                        return;
919                    }
920                    let error = TypeCheckError::TypeCannotBeUsed {
921                        typ: typ.clone(),
922                        place: "message",
923                        location,
924                    };
925                    self.push_err(CompilationError::TypeError(error));
926                };
927                if let Type::FmtString(_, item_types) = typ {
928                    if let Type::Tuple(item_types) = item_types.as_ref() {
929                        item_types.iter().for_each(check_msg_compat);
930                    }
931                } else {
932                    check_msg_compat(&typ);
933                }
934            }
935            msg
936        });
937
938        self.unify(&expr_type, &Type::Bool, || TypeCheckError::TypeMismatch {
939            expr_typ: expr_type.to_string(),
940            expected_typ: Type::Bool.to_string(),
941            expr_location,
942        });
943
944        (HirExpression::Constrain(HirConstrainExpression(expr_id, location.file, msg)), Type::Unit)
945    }

947    /// Elaborate a struct constructor.
948    ///
949    /// This method resolves the [UnresolvedType][crate::ast::UnresolvedType] into the [Type] being constructed,
950    /// then delegates to [Elaborator::elaborate_constructor_with_type] to handle the fields.
951    fn elaborate_constructor(
952        &mut self,
953        constructor: ConstructorExpression,
954    ) -> (HirExpression, Type) {
955        let location = constructor.typ.location;
956
957        // A constructor type can either be a Path or an interned UnresolvedType.
958        // We represent both as UnresolvedType (with Path being a Named UnresolvedType)
959        // and error if we don't get a Named path.
960        let mut typ = constructor.typ.typ;
961        if let UnresolvedTypeData::Interned(id) = typ {
962            typ = self.interner.get_unresolved_type_data(id).clone();
963        }
964        if let UnresolvedTypeData::Resolved(id) = typ {
965            // If this type is already resolved we can skip the rest of this function
966            // which just resolves the type, and go straight to resolving the fields.
967            let resolved = self.interner.get_quoted_type(id).clone();
968            return self.elaborate_constructor_with_type(
969                resolved,
970                constructor.fields,
971                location,
972                None,
973            );
974        }
975        let UnresolvedTypeData::Named(mut path, generics, _) = typ else {
976            self.push_err(ResolverError::NonStructUsedInConstructor {
977                typ: typ.to_string(),
978                location,
979            });
980            return (HirExpression::Error, Type::Error);
981        };
982
983        // When instantiating a generic struct, treat any generics in the type
984        // as if they were part of the turbofish, so they can be validated with the path.
985        if !generics.ordered_args.is_empty() {
986            let last_segment = path.segments.last_mut().unwrap();
987            last_segment.generics = Some(generics.ordered_args);
988        }
989
990        let path = self.validate_path(path);
991        let last_segment = path.last_segment();
992
993        let Some(typ) = self.lookup_type_or_error(path) else {
994            return (HirExpression::Error, Type::Error);
995        };
996
997        self.elaborate_constructor_with_type(typ, constructor.fields, location, Some(last_segment))
998    }

1000    /// Knowing the [Type] being constructed, elaborate all field expressions.
1001    fn elaborate_constructor_with_type(
1002        &mut self,
1003        typ: Type,
1004        fields: Vec<(Ident, Expression)>,
1005        location: Location,
1006        last_segment: Option<TypedPathSegment>,
1007    ) -> (HirExpression, Type) {
1008        let typ = typ.follow_bindings_shallow();
1009        let (struct_type, generics) = match typ.as_ref() {
1010            Type::DataType(struct_type, struct_generics) if struct_type.borrow().is_struct() => {
1011                (struct_type.clone(), struct_generics)
1012            }
1013            typ => {
1014                self.push_err(ResolverError::NonStructUsedInConstructor {
1015                    typ: typ.to_string(),
1016                    location,
1017                });
1018                return (HirExpression::Error, Type::Error);
1019            }
1020        };
1021        let struct_id = struct_type.borrow().id;
1022
1023        self.mark_struct_as_constructed(struct_type.clone());
1024
1025        // `last_segment` is optional if this constructor was resolved from a quoted type
1026        let mut generics = generics.clone();
1027        let mut is_self_type = false;
1028        let mut constructor_type_location = location;
1029
1030        if let Some(last_segment) = last_segment {
1031            let turbofish_location = last_segment.turbofish_location();
1032            is_self_type = last_segment.ident.is_self_type_name();
1033            constructor_type_location = last_segment.ident.location();
1034
1035            generics = self.resolve_struct_turbofish_generics(
1036                &struct_type.borrow(),
1037                generics,
1038                last_segment.generics,
1039                turbofish_location,
1040            );
1041        }
1042
1043        // Each of the struct generics must be bound at the end of the function
1044        for (index, generic) in generics.iter().enumerate() {
1045            if let Type::TypeVariable(type_variable) = generic {
1046                self.push_required_type_variable(
1047                    type_variable.id(),
1048                    Type::TypeVariable(type_variable.clone()),
1049                    BindableTypeVariableKind::StructGeneric { struct_id, index },
1050                    location,
1051                );
1052            }
1053        }
1054
1055        let field_types = struct_type
1056            .borrow()
1057            .get_fields_with_visibility(&generics)
1058            .expect("This type should already be validated to be a struct");
1059
1060        let fields = self.resolve_constructor_expr_fields(
1061            struct_type.clone(),
1062            field_types,
1063            fields,
1064            location,
1065        );
1066        let expr = HirExpression::Constructor(HirConstructorExpression {
1067            fields,
1068            r#type: struct_type.clone(),
1069            struct_generics: generics.clone(),
1070        });
1071
1072        self.interner.add_type_reference(struct_id, constructor_type_location, is_self_type);
1073
1074        (expr, Type::DataType(struct_type, generics))
1075    }

1084    /// Resolve all the fields of a struct constructor expression.
1085    /// Ensures all fields are present, none are repeated, and all
1086    /// are part of the struct.
1087    fn resolve_constructor_expr_fields(
1088        &mut self,
1089        struct_type: Shared<DataType>,
1090        field_types: Vec<(String, ItemVisibility, Type)>,
1091        fields: Vec<(Ident, Expression)>,
1092        location: Location,
1093    ) -> Vec<(Ident, ExprId)> {
1094        let mut ret = Vec::with_capacity(fields.len());
1095        let mut seen_fields = HashSet::default();
1096        let mut unseen_fields = struct_type
1097            .borrow()
1098            .field_names()
1099            .expect("This type should already be validated to be a struct");
1100
1101        let expected_fields_by_name = field_types
1102            .iter()
1103            .enumerate()
1104            .map(|(i, (name, vis, typ))| (name.as_str(), (i, vis, typ)))
1105            .collect::<HashMap<_, _>>();
1106
1107        for (field_name, field) in fields {
1108            let expected_field = expected_fields_by_name.get(field_name.as_str());
1109
1110            let expected_index_and_visibility =
1111                expected_field.map(|(index, visibility, _)| (index, visibility));
1112            let expected_type = expected_field.map(|(_, _, typ)| typ).unwrap_or(&&Type::Error);
1113
1114            let field_location = field.location;
1115            let (resolved, field_type) = self.elaborate_expression(field);
1116
1117            if unseen_fields.remove(&field_name) {
1118                seen_fields.insert(field_name.clone());
1119
1120                self.unify_with_coercions(
1121                    &field_type,
1122                    expected_type,
1123                    resolved,
1124                    field_location,
1125                    || {
1126                        CompilationError::TypeError(TypeCheckError::TypeMismatch {
1127                            expected_typ: expected_type.to_string(),
1128                            expr_typ: field_type.to_string(),
1129                            expr_location: field_location,
1130                        })
1131                    },
1132                );
1133            } else if seen_fields.contains(&field_name) {
1134                // duplicate field
1135                self.push_err(ResolverError::DuplicateField { field: field_name.clone() });
1136            } else {
1137                // field not required by struct
1138                self.push_err(ResolverError::NoSuchField {
1139                    field: field_name.clone(),
1140                    struct_definition: struct_type.borrow().name.clone(),
1141                });
1142            }
1143
1144            if let Some((index, visibility)) = expected_index_and_visibility {
1145                let struct_type = struct_type.borrow();
1146                let field_location = field_name.location();
1147                let field_name = field_name.as_str();
1148                self.check_struct_field_visibility(
1149                    &struct_type,
1150                    field_name,
1151                    **visibility,
1152                    field_location,
1153                );
1154
1155                self.interner.add_struct_member_reference(struct_type.id, *index, field_location);
1156            }
1157
1158            ret.push((field_name, resolved));
1159        }
1160
1161        if !unseen_fields.is_empty() {
1162            self.push_err(ResolverError::MissingFields {
1163                location,
1164                missing_fields: unseen_fields.into_iter().map(|field| field.to_string()).collect(),
1165                struct_definition: struct_type.borrow().name.clone(),
1166            });
1167        }
1168
1169        ret
1170    }

1173    /// `is_offset`:
1174    /// - `is_offset = false`: Auto-dereferencing will occur, and this will always return false
1175    /// - `is_offset = true`: Auto-dereferencing is disabled, and this will return true if the lhs
1176    ///   is a reference.
1177    fn elaborate_member_access(
1178        &mut self,
1179        access: MemberAccessExpression,
1180        location: Location,
1181        is_offset: bool,
1182    ) -> (ExprId, Type, bool) {
1183        // We don't need the boolean 'skipped auto-dereferences' from elaborate_reference_expression
1184        // since if we have skipped any then `lhs_type` will be a reference and we will need to
1185        // skip the deref (if is_offset is true) here anyway to extract the field out of the reference.
1186        // This is more reliable than using the boolean return value here since the return value
1187        // doesn't account for reference variables which we need to account for.
1188        let (lhs, lhs_type, _) = self.elaborate_reference_expression(access.lhs);
1189        let is_reference = lhs_type.is_ref();
1190
1191        let rhs = access.rhs;
1192        let rhs_location = rhs.location();
1193        // `is_offset` is only used when lhs is a reference and we want to return a reference to rhs
1194        let access = HirMemberAccess { lhs, rhs, is_offset };
1195        let expr_id = self.intern_expr(HirExpression::MemberAccess(access.clone()), location);
1196        let typ = self.type_check_member_access(access, *expr_id, lhs_type, rhs_location);
1197        let expr_id = self.intern_expr_type(expr_id, typ.clone());
1198        (expr_id, typ, is_offset && is_reference)
1199    }

1261    /// Handles the results of [Self::prefix_operand_type_rules] and [Self::infix_operand_type_rules].
1262    /// * if the rules returned an `Err`, it returns [Type::Error]
1263    /// * if the results indicate that a trait method should be used,
1264    ///   it pushes a trait constraint and checks that the expression type is compatible with the trait method
1265    fn handle_operand_type_rules_result(
1266        &mut self,
1267        result: Result<(Type, bool), TypeCheckError>,
1268        operand_type: &Type,
1269        trait_method_id: Option<TraitItemId>,
1270        expr_id: ExprId,
1271        location: Location,
1272        is_ord: bool,
1273    ) -> Type {
1274        match result {
1275            Ok((typ, use_impl)) => {
1276                if use_impl {
1277                    let trait_method_id = trait_method_id
1278                        .expect("ICE: expected some trait_method_id when use_impl is true");
1279
1280                    // Delay checking the trait constraint until the end of the function.
1281                    // Checking it now could bind an unbound type variable to any type
1282                    // that implements the trait.
1283                    let trait_id = trait_method_id.trait_id;
1284                    let trait_generics = TraitGenerics::default();
1285                    let trait_bound = ResolvedTraitBound { trait_id, trait_generics, location };
1286                    let constraint = TraitConstraint { typ: operand_type.clone(), trait_bound };
1287                    let select_impl = true; // this constraint should lead to choosing a trait impl
1288                    self.push_trait_constraint(constraint, expr_id, select_impl);
1289
1290                    self.type_check_operator_method(
1291                        expr_id,
1292                        trait_method_id,
1293                        operand_type,
1294                        &typ,
1295                        location,
1296                        is_ord,
1297                    );
1298                }
1299                typ
1300            }
1301            Err(error) => {
1302                self.push_err(error);
1303                Type::Error
1304            }
1305        }
1306    }

1308    fn elaborate_if(
1309        &mut self,
1310        if_expr: IfExpression,
1311        target_type: Option<&Type>,
1312    ) -> (HirExpression, Type) {
1313        let expr_location = if_expr.condition.type_location();
1314        let consequence_location = if_expr.consequence.type_location();
1315        let (condition, cond_type) = self.elaborate_expression(if_expr.condition);
1316        let (consequence, mut ret_type) =
1317            self.elaborate_expression_with_target_type(if_expr.consequence, target_type);
1318
1319        self.unify(&cond_type, &Type::Bool, || TypeCheckError::TypeMismatch {
1320            expected_typ: Type::Bool.to_string(),
1321            expr_typ: cond_type.to_string(),
1322            expr_location,
1323        });
1324
1325        let (alternative, else_type, error_location) =
1326            if let Some(alternative) = if_expr.alternative {
1327                let alternative_location = alternative.type_location();
1328                let (else_, else_type) =
1329                    self.elaborate_expression_with_target_type(alternative, target_type);
1330                (Some(else_), else_type, alternative_location)
1331            } else {
1332                (None, Type::Unit, consequence_location)
1333            };
1334
1335        self.unify(&ret_type, &else_type, || {
1336            let err = TypeCheckError::TypeMismatch {
1337                expected_typ: ret_type.to_string(),
1338                expr_typ: else_type.to_string(),
1339                expr_location: error_location,
1340            };
1341
1342            let context = if ret_type == Type::Unit {
1343                "Are you missing a semicolon at the end of your 'else' branch?"
1344            } else if else_type == Type::Unit {
1345                "Are you missing a semicolon at the end of the first block of this 'if'?"
1346            } else {
1347                "Expected the types of both if branches to be equal"
1348            };
1349
1350            err.add_context(context)
1351        });
1352
1353        if alternative.is_none() {
1354            ret_type = Type::Unit;
1355        }
1356
1357        let if_expr = HirIfExpression { condition, consequence, alternative };
1358        (HirExpression::If(if_expr), ret_type)
1359    }

1361    /// Elaborate a `match <expr> { <rules> }` expression by creating an block such as this:
1362    /// ```text
1363    /// {
1364    ///   let internal variable = <expr>;
1365    ///   match internal variable { <rules> }
1366    /// }
1367    /// ```
1368    fn elaborate_match(
1369        &mut self,
1370        match_expr: MatchExpression,
1371        location: Location,
1372    ) -> (HirExpression, Type) {
1373        // Show error on the `match` keyword
1374        let match_location = Location::new(
1375            Span::from(location.span.start()..location.span.start() + 5),
1376            location.file,
1377        );
1378        self.use_unstable_feature(UnstableFeature::Enums, match_location);
1379
1380        let expr_location = match_expr.expression.location;
1381        let (expression, typ) = self.elaborate_expression(match_expr.expression);
1382        let (let_, variable) = self.wrap_in_let(expression, typ.clone());
1383
1384        let (errored, (rows, result_type)) =
1385            self.errors_occurred_in(|this| this.elaborate_match_rules(variable, match_expr.rules));
1386
1387        // Avoid calling `elaborate_match_rows` if there were errors while constructing
1388        // the match rows - it'll just lead to extra errors like `unreachable pattern`
1389        // warnings on branches which previously had type errors.
1390        let tree = HirExpression::Match(if !errored {
1391            self.elaborate_match_rows(rows, &typ, expr_location)
1392        } else {
1393            HirMatch::Failure { missing_case: false }
1394        });
1395
1396        let tree = self.interner.push_expr_full(tree, location, result_type.clone());
1397
1398        let tree = self.interner.push_stmt_full(HirStatement::Expression(tree), location);
1399
1400        let block = HirExpression::Block(HirBlockExpression { statements: vec![let_, tree] });
1401        (block, result_type)
1402    }

1456    /// For elaborating a lambda we might get `parameters_type_hints`. These come from a potential
1457    /// call that has this lambda as the argument. The parameter type hints will be the types of
1458    /// the function type corresponding to the lambda argument.
1459    ///
1460    /// The `unconstrained` parameter is set based on whether the lambda is expected to be unconstrained
1461    /// by the function we are passing it to. If we just assign the lambda to a variable, then it's `false`.
1462    fn elaborate_lambda_with_parameter_type_hints(
1463        &mut self,
1464        lambda: Lambda,
1465        parameters_type_hints: Option<&Vec<Type>>,
1466        unconstrained: bool,
1467    ) -> (HirExpression, Type) {
1468        self.push_scope();
1469        let scope_index = self.scopes.current_scope_index();
1470
1471        self.lambda_stack.push(LambdaContext { captures: Vec::new(), scope_index, unconstrained });
1472
1473        let mut arg_types = Vec::with_capacity(lambda.parameters.len());
1474        let mut parameter_names_in_list = HashMap::default();
1475        let parameters =
1476            vecmap(lambda.parameters.into_iter().enumerate(), |(index, (pattern, typ))| {
1477                let parameter = DefinitionKind::Local(None);
1478                let typ = match typ {
1479                    Some(typ) => {
1480                        let wildcard_allowed = WildcardAllowed::Yes;
1481                        self.resolve_type(typ, wildcard_allowed)
1482                    }
1483                    None => {
1484                        if let Some(parameter_type_hint) =
1485                            parameters_type_hints.and_then(|hints| hints.get(index))
1486                        {
1487                            parameter_type_hint.clone()
1488                        } else {
1489                            self.interner.next_type_variable_with_kind(Kind::Any)
1490                        }
1491                    }
1492                };
1493
1494                arg_types.push(typ.clone());
1495                (
1496                    self.elaborate_pattern(
1497                        pattern,
1498                        typ.clone(),
1499                        parameter,
1500                        true,
1501                        &mut parameter_names_in_list,
1502                    ),
1503                    typ,
1504                )
1505            });
1506
1507        let wildcard_allowed = WildcardAllowed::Yes;
1508        let return_type = self.resolve_inferred_type(lambda.return_type, wildcard_allowed);
1509        let body_location = lambda.body.location;
1510        let (body, body_type) = self.elaborate_expression(lambda.body);
1511
1512        let lambda_context = self.lambda_stack.pop().unwrap();
1513        self.pop_scope();
1514
1515        self.unify(&body_type, &return_type, || TypeCheckError::TypeMismatch {
1516            expected_typ: return_type.to_string(),
1517            expr_typ: body_type.to_string(),
1518            expr_location: body_location,
1519        });
1520
1521        let captured_vars = vecmap(&lambda_context.captures, |capture| {
1522            self.interner.definition_type(capture.ident.id)
1523        });
1524
1525        let env_type =
1526            if captured_vars.is_empty() { Type::Unit } else { Type::Tuple(captured_vars) };
1527
1528        let captures = lambda_context.captures;
1529        let expr = HirExpression::Lambda(HirLambda {
1530            parameters,
1531            return_type,
1532            body,
1533            captures,
1534            unconstrained,
1535        });
1536        (expr, Type::Function(arg_types, Box::new(body_type), Box::new(env_type), unconstrained))
1537    }

1550    fn elaborate_comptime_block(
1551        &mut self,
1552        block: BlockExpression,
1553        location: Location,
1554        target_type: Option<&Type>,
1555    ) -> (ExprId, Type) {
1556        let block = self.elaborate_in_comptime_context(|this| {
1557            let (block, block_type) = this.elaborate_block_expression(block, target_type);
1558

1560            // This for example allows this code to compile: `let x: u8 = comptime { 1 }`.

1562            // which is Field.
1563            if let Some(target_type) = target_type {
1564                this.unify(&block_type, target_type, || TypeCheckError::TypeMismatch {
1565                    expected_typ: target_type.to_string(),
1566                    expr_typ: block_type.to_string(),
1567                    expr_location: location,
1568                });
1569            }

1571            block

1574        let mut interpreter = self.setup_interpreter();
1575        let value = interpreter.evaluate_block(block);
1576
1577        let (id, typ) = self.inline_comptime_value(value, location);
1578
1579        let location = self.interner.id_location(id);
1580        self.debug_comptime(location, |interner| {
1581            interner.expression(&id).to_display_ast(interner, location).kind

1587    pub fn inline_comptime_value(
1588        &mut self,
1589        value: Result<comptime::Value, InterpreterError>,
1590        location: Location,
1591    ) -> (ExprId, Type) {
1592        let make_error = |this: &mut Self, error: InterpreterError| {
1593            let error: CompilationError = error.into();
1594            this.push_err(error);
1595            let typ = Type::Error;
1596            let error = this.interner.push_expr_full(HirExpression::Error, location, typ.clone());
1597            (error, typ)
1598        };
1599
1600        let value = match value {
1601            Ok(value) => value,
1602            Err(error) => return make_error(self, error),
1603        };
1604
1605        match value.into_expression(self, location) {
1606            Ok(new_expr) => {
1607                // At this point the Expression was already elaborated and we got a Value.
1608                // We'll elaborate this value turned into Expression to inline it and get
1609                // an ExprId and Type, but we don't want any visibility errors to happen
1610                // here (they could if we have `Foo { inner: 5 }` and `inner` is not
1611                // accessible from where this expression is being elaborated).
1612                self.silence_field_visibility_errors += 1;
1613                let value = self.elaborate_expression(new_expr);
1614                self.silence_field_visibility_errors -= 1;
1615                value
1616            }
1617            Err(error) => make_error(self, error),
1618        }
1619    }

1648    /// Call a macro function and inlines its code at the call site.
1649    /// This will also perform a type check to ensure that the return type is an `Expr` value.
1650    fn call_macro(
1651        &mut self,
1652        func: ExprId,
1653        arguments: Vec<ExprId>,
1654        location: Location,
1655        return_type: Type,
1656    ) -> Option<(HirExpression, Type)> {
1657        self.unify(&return_type, &Type::Quoted(QuotedType::Quoted), || {
1658            TypeCheckError::MacroReturningNonExpr { typ: return_type.clone(), location }
1659        });
1660
1661        let function = match self.try_get_comptime_function(func, location) {
1662            Ok(function) => function?,
1663            Err(error) => {
1664                self.push_err(error);
1665                return None;
1666            }
1667        };
1668
1669        let mut interpreter = self.setup_interpreter();
1670        let mut comptime_args = Vec::new();
1671        let mut errors: Vec<CompilationError> = Vec::new();
1672
1673        for argument in arguments {
1674            match interpreter.evaluate(argument) {
1675                Ok(arg) => {
1676                    let location = interpreter.elaborator.interner.expr_location(&argument);
1677                    comptime_args.push((arg, location));
1678                }
1679                Err(error) => errors.push(error.into()),
1680            }
1681        }
1682
1683        let bindings = interpreter.elaborator.interner.get_instantiation_bindings(func).clone();
1684        let result = interpreter.call_function(function, comptime_args, bindings, location);
1685
1686        if !errors.is_empty() {
1687            self.push_errors(errors);
1688            return None;
1689        }
1690
1691        let (expr_id, typ) = self.inline_comptime_value(result, location);
1692        Some((self.interner.expression(&expr_id), typ))
1693    }

1695    fn elaborate_as_trait_path(&mut self, path: AsTraitPath) -> (ExprId, Type) {
1696        let location = path.typ.location.merge(path.trait_path.location);
1697
1698        let constraint = UnresolvedTraitConstraint {
1699            typ: path.typ,
1700            trait_bound: TraitBound {
1701                trait_path: path.trait_path,
1702                trait_generics: path.trait_generics,
1703            },
1704        };
1705
1706        let wildcard_allowed = WildcardAllowed::Yes;
1707        let typ = self.use_type(constraint.typ.clone(), wildcard_allowed);
1708        let Some(trait_bound) = self.use_trait_bound(&constraint.trait_bound) else {
1709            // resolve_trait_bound only returns None if it has already issued an error, so don't
1710            // issue another here.
1711            let error = self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
1712            return (error, Type::Error);
1713        };
1714
1715        let constraint = TraitConstraint { typ, trait_bound };
1716
1717        let the_trait = self.interner.get_trait(constraint.trait_bound.trait_id);
1718        let self_type = the_trait.self_type_typevar.clone();
1719        let kind = the_trait.self_type_typevar.kind();
1720
1721        let Some(definition) =
1722            the_trait.find_method_or_constant(path.impl_item.as_str(), self.interner)
1723        else {
1724            let trait_name = the_trait.name.to_string();
1725            let method_name = path.impl_item.to_string();
1726            let location = path.impl_item.location();
1727            self.push_err(ResolverError::NoSuchMethodInTrait { trait_name, method_name, location });
1728            let error = self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
1729            return (error, Type::Error);
1730        };
1731
1732        let trait_item = TraitItem { definition, constraint: constraint.clone(), assumed: false };
1733
1734        let ident = HirIdent {
1735            location: path.impl_item.location(),
1736            id: definition,
1737            impl_kind: ImplKind::TraitItem(trait_item),
1738        };
1739
1740        let id = self.intern_expr(HirExpression::Ident(ident.clone(), None), location);
1741
1742        let mut bindings = TypeBindings::default();
1743
1744        // In `<Type as Trait>::method` we know `Self` is `Type` so we bind that now
1745        bindings.insert(self_type.id(), (self_type, kind, constraint.typ.clone()));
