63    #[tracing::instrument(level = "trace", skip_all)]
64    pub(crate) fn elaborate_expression_with_target_type(
65        &mut self,
66        expr: Expression,
67        target_type: Option<&Type>,
68    ) -> (ExprId, Type) {
69        if !self.inc_recursion_depth(expr.location) {
70            let id = self.interner.push_expr_full(HirExpression::Error, expr.location, Type::Error);
71            return (id, Type::Error);
72        }
73
74        let ((id, typ), has_errors) =
75            self.with_error_guard(|this| this.elaborate_expression_inner(expr, target_type));
76
77        self.dec_recursion_depth();
78
79        if has_errors {
80            self.interner.exprs_with_errors.insert(id);
81        }
82
83        (id, typ)
84    }

86    #[tracing::instrument(level = "trace", skip_all)]
87    fn elaborate_expression_inner(
88        &mut self,
89        expr: Expression,
90        target_type: Option<&Type>,
91    ) -> (ExprId, Type) {
92        let is_integer_literal = matches!(expr.kind, ExpressionKind::Literal(Literal::Integer(..)));
93
94        let (hir_expr, typ) = match expr.kind {
95            ExpressionKind::Literal(literal) => self.elaborate_literal(literal, expr.location),
96            ExpressionKind::Block(block) => self.elaborate_block(block, target_type),
97            ExpressionKind::Prefix(prefix) => return self.elaborate_prefix(*prefix, expr.location),
98            ExpressionKind::Index(index) => self.elaborate_index(*index),
99            ExpressionKind::Call(call) => self.elaborate_call(*call, expr.location),
100            ExpressionKind::MethodCall(call) => self.elaborate_method_call(*call, expr.location),
101            ExpressionKind::Constrain(constrain) => self.elaborate_constrain(constrain),
102            ExpressionKind::Constructor(constructor) => self.elaborate_constructor(*constructor),
103            ExpressionKind::MemberAccess(access) => {
104                let (expr_id, typ, _) = self.elaborate_member_access(*access, expr.location, false);
105                return (expr_id, typ);
106            }
107            ExpressionKind::Cast(cast) => self.elaborate_cast(*cast, expr.location),
108            ExpressionKind::Infix(infix) => return self.elaborate_infix(*infix, expr.location),
109            ExpressionKind::If(if_) => self.elaborate_if(*if_, target_type),
110            ExpressionKind::Match(match_) => self.elaborate_match(*match_, expr.location),
111            ExpressionKind::Variable(variable) => return self.elaborate_variable(variable),
112            ExpressionKind::Tuple(tuple) => self.elaborate_tuple(tuple, target_type),
113            ExpressionKind::Lambda(lambda) => {
114                self.elaborate_lambda_with_target_type(*lambda, target_type)
115            }
116            ExpressionKind::Parenthesized(expr) => {
117                return self.elaborate_expression_with_target_type(*expr, target_type);
118            }
119            ExpressionKind::Quote(quote) => self.elaborate_quote(quote, expr.location),
120            ExpressionKind::Comptime(block, _) => {
121                if self.in_comptime_context {
122                    // Treat a nested comptime block as a regular block. Nested comptime blocks
123                    // can happen as a result of macro expansion so it wouldn't be good to produce
124                    // a warning in that case.
125                    self.elaborate_block(block, target_type)
126                } else {
127                    return self.elaborate_comptime_block(block, expr.location, target_type);
128                }
129            }
130            ExpressionKind::Unsafe(unsafe_expression) => {
131                self.elaborate_unsafe_block(unsafe_expression, target_type)
132            }
133            ExpressionKind::Resolved(id) => return (id, self.interner.id_type(id)),
134            ExpressionKind::Interned(id) => {
135                let expr_kind = self.interner.get_expression_kind(id);
136                let expr = Expression::new(expr_kind.clone(), expr.location);
137                return self.elaborate_expression(expr);
138            }
139            ExpressionKind::InternedStatement(id) => {
140                return self.elaborate_interned_statement_as_expr(id, expr.location);
141            }
142            ExpressionKind::Error => (HirExpression::Error, Type::Error),
143            ExpressionKind::Unquote(_) => {
144                self.push_err(ResolverError::UnquoteUsedOutsideQuote { location: expr.location });
145                (HirExpression::Error, Type::Error)
146            }
147            ExpressionKind::AsTraitPath(path) => {
148                return self.elaborate_as_trait_path(*path);
149            }
150            ExpressionKind::TypePath(path) => return self.elaborate_type_path(*path),
151        };
152        let id = self.interner.push_expr_full(hir_expr, expr.location, typ.clone());
153
154        if is_integer_literal {
155            self.push_integer_literal_expr_id(id);
156        }
157
158        (id, typ)
159    }

161    /// Elaborate a member access expression without adding the automatic dereferencing operations
162    /// to it, treating it as an offset instead. This also returns a boolean indicating whether the
163    /// result skipped any required auto-dereferences (and thus needs dereferencing to be used as a value
164    /// instead of a reference). This flag is used when `&mut foo.bar.baz` is used to cancel out
165    /// the `&mut`.
166    #[tracing::instrument(level = "trace", skip_all)]
167    fn elaborate_reference_expression(&mut self, expr: Expression) -> (ExprId, Type, bool) {
168        match expr.kind {
169            ExpressionKind::MemberAccess(access) => {
170                self.elaborate_member_access(*access, expr.location, true)
171            }
172            _ => {
173                let (expr_id, typ) = self.elaborate_expression(expr);
174                (expr_id, typ, false)
175            }
176        }
177    }

179    /// Given its ID, retrieve and elaborate an interned [StatementKind].
180    #[tracing::instrument(level = "trace", skip_all)]
181    fn elaborate_interned_statement_as_expr(
182        &mut self,
183        id: InternedStatementKind,
184        location: Location,
185    ) -> (ExprId, Type) {
186        match self.interner.get_statement_kind(id) {
187            StatementKind::Expression(expr) | StatementKind::Semi(expr) => {
188                self.elaborate_expression(expr.clone())
189            }
190            StatementKind::Interned(id) => self.elaborate_interned_statement_as_expr(*id, location),
191            StatementKind::Error => {
192                let expr = Expression::new(ExpressionKind::Error, location);
193                self.elaborate_expression(expr)
194            }
195            other => {
196                let statement = other.to_string();
197                self.push_err(ResolverError::InvalidInternedStatementInExpr {
198                    statement,
199                    location,
200                });
201                let expr = Expression::new(ExpressionKind::Error, location);
202                self.elaborate_expression(expr)
203            }
204        }

217    #[tracing::instrument(level = "trace", skip_all)]
218    fn elaborate_block_expression(
219        &mut self,
220        block: BlockExpression,
221        target_type: Option<&Type>,
222    ) -> (HirBlockExpression, Type) {
223        self.push_scope();
224        let mut block_type = Type::Unit;
225        let statements_len = block.statements.len();
226        let mut statements = Vec::with_capacity(statements_len);
227
228        // If we found a break or continue statement, this holds its location (only for the first one)
229        let mut break_or_continue_location = None;
230        // When encountering a statement after a break or continue we'll error saying it's unreachable,
231        // but we only want to error for the first statement.
232        let mut errored_unreachable = false;
233
234        for (i, statement) in block.statements.into_iter().enumerate() {
235            let location = statement.location;
236            let statement_target_type = if i == statements_len - 1 { target_type } else { None };
237            let (id, stmt_type) =
238                self.elaborate_statement_with_target_type(statement, statement_target_type);
239
240            statements.push(id);
241
242            let stmt = self.interner.statement(&id);
243
244            if let HirStatement::Semi(expr) = stmt {
245                let inner_expr_type = self.interner.id_type(expr);
246                let location = self.interner.expr_location(&expr);
247
248                self.unify(&inner_expr_type, &Type::Unit, |_| {
249                    let expr_type = inner_expr_type.clone();
250                    let expr_location = location;
251
252                    if let MustUse::MustUse(message) = Self::type_is_must_use(&expr_type) {
253                        TypeCheckError::UnusedResultError { expr_type, expr_location, message }
254                    } else {
255                        TypeCheckError::UnusedResultWarning { expr_type, expr_location }
256                    }
257                });
258            }
259
260            let is_break_or_continue = matches!(stmt, HirStatement::Break | HirStatement::Continue);
261
262            if let Some(break_or_continue_location) = break_or_continue_location {
263                if !errored_unreachable {
264                    self.push_err(ResolverError::UnreachableStatement {
265                        location,
266                        break_or_continue_location,
267                    });
268                    errored_unreachable = true;
269                }
270            } else if is_break_or_continue {
271                break_or_continue_location = Some(location);
272            }
273
274            if i + 1 == statements.len() {
275                block_type = stmt_type;
276            }
277        }
278
279        self.pop_scope();
280        (HirBlockExpression { statements }, block_type)
281    }

283    /// If the given type was declared as:
284    /// - `#[must_use = "message"]`, return [MustUse::MustUse(Some("message"))]
285    /// - `#[must_use]`, return [MustUse::MustUse(None)]
286    /// - otherwise, return `MustUse::NoMustUse`
287    fn type_is_must_use(typ: &Type) -> MustUse {
288        /// Helper function to avoid infinite recursion for infinitely recursive types
289        fn helper(typ: &Type, fuel: u32) -> MustUse {
290            if fuel == 0 {
291                return MustUse::NoMustUse;
292            }
293            let fuel = fuel - 1;
294            match typ.follow_bindings_shallow().as_ref() {
295                Type::DataType(data_type, _generics) => data_type.borrow().must_use.clone(),
296                // If any element in the tuple is `#[must_use]`, the whole tuple is
297                Type::Tuple(elements) => {
298                    for element in elements {
299                        if let MustUse::MustUse(message) = helper(element, fuel) {
300                            return MustUse::MustUse(message);
301                        }
302                    }
303                    MustUse::NoMustUse
304                }
305                Type::Alias(alias, generics) => helper(&alias.borrow().get_type(generics), fuel),
306                Type::CheckedCast { to, .. } => helper(to.as_ref(), fuel),
307                Type::Reference(element, _) => helper(element.as_ref(), fuel),
308                _ => MustUse::NoMustUse,
309            }
310        }
311
312        // 10 is an arbitrary maximum bound on recursion through `Type`s here
313        // in case an infinitely recursive type is used. In practice most types should
314        // require just 1 iteration, or up to 3 for a reference to an aliased type.
315        helper(typ, 10)
316    }

318    #[tracing::instrument(level = "trace", skip_all)]
319    fn elaborate_unsafe_block(
320        &mut self,
321        unsafe_expression: UnsafeExpression,
322        target_type: Option<&Type>,
323    ) -> (HirExpression, Type) {
324        use UnsafeBlockStatus::*;
325        // Before entering the block we cache the old value of the unsafe block status, so it can be restored.
326        let old_in_unsafe_block = self.unsafe_block_status;
327        let is_nested_unsafe_block = !matches!(old_in_unsafe_block, NotInUnsafeBlock);
328
329        if is_nested_unsafe_block {
330            self.push_err(TypeCheckError::NestedUnsafeBlock {
331                location: unsafe_expression.unsafe_keyword_location,
332            });
333        }
334
335        self.unsafe_block_status = InUnsafeBlockWithoutUnconstrainedCalls;
336
337        let (hir_block_expression, typ) =
338            self.elaborate_block_expression(unsafe_expression.block, target_type);
339
340        let has_unconstrained_call =
341            matches!(self.unsafe_block_status, InUnsafeBlockWithUnconstrainedCalls);
342
343        if !has_unconstrained_call {
344            self.push_err(TypeCheckError::UnnecessaryUnsafeBlock {
345                location: unsafe_expression.unsafe_keyword_location,
346            });
347        }
348
349        // Finally, we restore the original value of the unsafe block status,
350        // unless we are in a nested block and we have found an unconstrained call,
351        // in which case we should consider the outer block as having that call as well.
352        if !is_nested_unsafe_block || !has_unconstrained_call {
353            self.unsafe_block_status = old_in_unsafe_block;
354        }
355
356        (HirExpression::Unsafe(hir_block_expression), typ)
357    }

359    #[tracing::instrument(level = "trace", skip_all)]
360    fn elaborate_literal(&mut self, literal: Literal, location: Location) -> (HirExpression, Type) {
361        use HirExpression::Literal as Lit;
362        match literal {
363            Literal::Unit => (Lit(HirLiteral::Unit), Type::Unit),
364            Literal::Bool(b) => (Lit(HirLiteral::Bool(b)), Type::Bool),
365            Literal::Integer(integer, suffix) => {
366                (Lit(HirLiteral::Integer(integer)), self.integer_suffix_type(suffix))
367            }
368            Literal::Str(str) | Literal::RawStr(str, _) => {
369                let len: u32 = str.len().try_into().expect(
370                    "ICE: Elaborator::elaborate_literal: str.len() is expected to fit into a u32",
371                );
372                let len = Type::constant_u32(len);
373                (Lit(HirLiteral::Str(str)), Type::String(Box::new(len)))
374            }
375            Literal::FmtStr(fragments, length) => self.elaborate_fmt_string(fragments, length),
376            Literal::Array(array_literal) => {
377                self.elaborate_array_literal(array_literal, location, true)
378            }
379            Literal::Vector(array_literal) => {
380                self.elaborate_array_literal(array_literal, location, false)
381            }
382        }
383    }

404    #[tracing::instrument(level = "trace", skip_all)]
405    fn elaborate_array_literal(
406        &mut self,
407        array_literal: ArrayLiteral,
408        location: Location,
409        is_array: bool,
410    ) -> (HirExpression, Type) {
411        let (expr, elem_type, length) = match array_literal {
412            ArrayLiteral::Standard(elements) => {
413                let type_variable_id = self.interner.next_type_variable_id();
414                let type_variable = TypeVariable::unbound(type_variable_id, Kind::Any);
415                self.push_required_type_variable(
416                    type_variable.id(),
417                    Type::TypeVariable(type_variable.clone()),
418                    BindableTypeVariableKind::ArrayLiteral { is_array },
419                    location,
420                );
421
422                let first_elem_type = Type::TypeVariable(type_variable);
423                let first_location = elements.first().map_or(location, |elem| elem.location);
424
425                let elements = vecmap(elements.into_iter().enumerate(), |(i, elem)| {
426                    let location = elem.location;
427                    let (elem_id, elem_type) = self.elaborate_expression(elem);
428
429                    self.unify(&elem_type, &first_elem_type, |_| {
430                        TypeCheckError::NonHomogeneousArray {
431                            first_location,
432                            first_type: first_elem_type.to_string(),
433                            first_index: 0,
434                            second_location: location,
435                            second_type: elem_type.to_string(),
436                            second_index: i,
437                        }
438                        .add_context("elements in an array must have the same type")
439                    });
440                    elem_id
441                });
442
443                let length: u32 = elements.len().try_into().expect("ICE: Elaborator::elaborate_array_literal: elements.len() is expected to fit into a u32");
444                let length = Type::constant_u32(length);
445                (HirArrayLiteral::Standard(elements), first_elem_type, length)
446            }
447            ArrayLiteral::Repeated { repeated_element, length } => {
448                let location = length.location;
449                let length = UnresolvedTypeExpression::from_expr(*length, location).unwrap_or_else(
450                    |error| {
451                        self.push_err(ResolverError::ParserError(Box::new(error)));
452                        UnresolvedTypeExpression::Constant(FieldElement::zero(), None, location)
453                    },
454                );
455
456                let wildcard_allowed = WildcardAllowed::Yes;
457                let length =
458                    self.convert_expression_type(length, &Kind::u32(), location, wildcard_allowed);
459                let (repeated_element, elem_type) = self.elaborate_expression(*repeated_element);
460
461                let length_clone = length.clone();
462                (HirArrayLiteral::Repeated { repeated_element, length }, elem_type, length_clone)
463            }
464        };
465        let constructor = if is_array { HirLiteral::Array } else { HirLiteral::Vector };
466        let elem_type = Box::new(elem_type);
467        let typ = if is_array {
468            Type::Array(Box::new(length), elem_type)
469        } else {
470            Type::Vector(elem_type)
471        };
472        (HirExpression::Literal(constructor(expr)), typ)
473    }

475    #[tracing::instrument(level = "trace", skip_all)]
476    fn elaborate_fmt_string(
477        &mut self,
478        fragments: Vec<FmtStrFragment>,
479        length: u32,
480    ) -> (HirExpression, Type) {
481        let mut fmt_str_idents = Vec::new();
482        let mut capture_types = Vec::new();
483
484        for fragment in &fragments {
485            if let FmtStrFragment::Interpolation(ident_name, location) = fragment {
486                let (typ, expr_id) = match self
487                    .get_ident_from_path(TypedPath::from_single(ident_name.clone(), *location))
488                {
489                    Some(IdentFromPath::Variable(variable)) => {
490                        self.handle_local_variable(&variable);
491                        self.elaborate_fmt_string_ident(variable.ident, *location)
492                    }
493                    Some(IdentFromPath::Definition { id, item: _ }) => {
494                        self.handle_definition_id(id, *location);
495                        let hir_ident = HirIdent::non_trait_method(id, *location);
496                        self.elaborate_fmt_string_ident(hir_ident, *location)
497                    }
498                    Some(IdentFromPath::TypeAlias(_)) | None => {
499                        let hir_expr = HirExpression::Error;
500                        let expr_id = self.intern_expr(hir_expr, *location);
501                        let typ = Type::Error;
502                        let expr_id = self.intern_expr_type(expr_id, typ.clone());
503                        (typ, expr_id)
504                    }
505                };
506                capture_types.push(typ);
507                fmt_str_idents.push(expr_id);
508            }
509        }
510
511        let len = Type::constant_u32(length);
512        let fmtstr_type =
513            if capture_types.is_empty() { Type::Unit } else { Type::Tuple(capture_types) };
514        let typ = Type::FmtString(Box::new(len), Box::new(fmtstr_type));
515        (HirExpression::Literal(HirLiteral::FmtStr(fragments, fmt_str_idents, length)), typ)
516    }

531    #[tracing::instrument(level = "trace", skip_all)]
532    fn elaborate_prefix(&mut self, prefix: PrefixExpression, location: Location) -> (ExprId, Type) {
533        // Simplify `*&x` and `*&mut x` to just `x`
534        if let UnaryOp::Dereference { .. } = prefix.operator
535            && let ExpressionKind::Prefix(ref inner) = prefix.rhs.kind
536            && let UnaryOp::Reference { mutable } = inner.operator
537        {
538            let ExpressionKind::Prefix(inner) = prefix.rhs.kind else { unreachable!() };
539            let rhs_location = inner.rhs.location;
540            let (rhs, typ) = self.elaborate_expression(inner.rhs);
541            if mutable {
542                self.check_can_mutate(rhs, rhs_location);
543            }
544            return (rhs, typ);
545        }
546
547        let rhs_location = prefix.rhs.location;
548        let operator = prefix.operator;
549
550        let (rhs, rhs_type, skip_op) = if matches!(operator, UnaryOp::Reference { .. }) {
551            let (rhs, rhs_type, needs_deref) = self.elaborate_reference_expression(prefix.rhs);
552
553            // If the reference expression delayed a needed deref, we can skip the `&mut _`
554            // operation since the expression is already a reference.
555            (rhs, rhs_type, needs_deref)
556        } else {
557            let (rhs, rhs_type) = self.elaborate_expression(prefix.rhs);
558            (rhs, rhs_type, false)
559        };
560
561        let trait_method_id = self.interner.get_prefix_operator_trait_method(&operator);
562
563        if let UnaryOp::Reference { mutable } = operator
564            && mutable
565        {
566            // If skip_op is set we already know we have a mutable reference
567            if !skip_op {
568                self.check_can_mutate(rhs, rhs_location);
569            }
570        }
571
572        let expr = HirExpression::Prefix(HirPrefixExpression {
573            operator,
574            rhs,
575            trait_method_id,
576            skip: skip_op,
577        });
578        let expr_id = self.intern_expr(expr, location);
579
580        // If `skip_op` is set we already know we have a mutable reference due to a member access on a mutable reference.
581        // The prefix operand type rules will return the result of a prefix operation.
582        // We do not want to check the prefix operand type rules as we will then get a type mismatch.
583        let typ = if skip_op {
584            rhs_type
585        } else {
586            let result = self.prefix_operand_type_rules(&operator, &rhs_type, location);
587            let not_ord = false;
588            self.handle_operand_type_rules_result(
589                result,
590                &rhs_type,
591                trait_method_id,
592                *expr_id,
593                location,
594                not_ord,
595            )
596        };
597
598        let expr_id = self.intern_expr_type(expr_id, typ.clone());
599        (expr_id, typ)
600    }

602    /// Check whether we can create a mutable reference over an expression.
603    /// Pushes an error if it cannot be done.
604    ///
605    /// Also, if the expression is a local variable, marks it as mutated.
606    #[tracing::instrument(level = "trace", skip_all)]
607    pub(super) fn check_can_mutate(&mut self, expr_id: ExprId, location: Location) {
608        match self.interner.expression(&expr_id) {
609            HirExpression::Ident(hir_ident, _) => {
610                let definition = self.interner.definition(hir_ident.id);
611                let name = definition.name.clone();
612                Self::mark_local_variable_as_mutated(definition, &mut self.scopes);
613
614                if !definition.mutable {
615                    self.push_err(TypeCheckError::CannotMutateImmutableVariable { name, location });
616                } else {
617                    self.check_can_mutate_lambda_capture(hir_ident.id, name, location);
618                }
619            }
620            HirExpression::Index(_) => {
621                self.push_err(TypeCheckError::MutableReferenceToArrayElement { location });
622            }
623            HirExpression::MemberAccess(member_access) => {
624                self.check_can_mutate(member_access.lhs, location);
625            }
626            _ => (),
627        }
628    }

687    fn elaborate_index(&mut self, index_expr: IndexExpression) -> (HirExpression, Type) {
688        let location = index_expr.index.location;
689
690        let (index, index_type) = self.elaborate_expression(index_expr.index);
691
692        let expected = Type::u32();
693        self.unify_or_type_mismatch_with_source(
694            &index_type,
695            &expected,
696            Source::ArrayIndex,
697            location,
698        );
699
700        // When writing `a[i]`, if `a : &mut ...` then automatically dereference `a` as many
701        // times as needed to get the underlying array.
702        let lhs_location = index_expr.collection.location;
703        let (lhs, lhs_type) = self.elaborate_expression(index_expr.collection);
704        let (collection, lhs_type) = self.insert_auto_dereferences(lhs, lhs_type);
705
706        let typ = match lhs_type.follow_bindings() {
707            Type::Array(ref size, ref base_type) => {
708                self.check_array_index_out_of_bounds(size, &index, location);
709                *base_type.clone()
710            }
711            Type::Vector(base_type) => *base_type,
712            Type::Error => Type::Error,
713            Type::TypeVariable(_) => {
714                self.push_err(TypeCheckError::TypeAnnotationsNeededForIndex {
715                    location: lhs_location,
716                });
717                Type::Error
718            }
719            typ => {
720                self.push_err(TypeCheckError::TypeMismatch {
721                    expected_typ: "Array".to_owned(),
722                    expr_typ: typ.to_string(),
723                    expr_location: lhs_location,
724                    similarly_named_types: Vec::new(),
725                });
726                Type::Error
727            }
728        };
729
730        let expr = HirExpression::Index(HirIndexExpression { collection, index });
731        (expr, typ)
732    }

784    /// Helper function containing the elaboration logic for a call expression.
785    /// Returns the HIR call and its type.
786    #[tracing::instrument(level = "trace", skip_all)]
787    fn elaborate_call_inner(
788        &mut self,
789        call: CallExpression,
790        location: Location,
791        is_macro_call: bool,
792    ) -> (HirCallExpression, Type) {
793        let (func, func_type) = self.elaborate_expression(*call.func);
794        let func_type = func_type.follow_bindings();
795
796        // Even if the function type is a Type::Error, we still want to elaborate the call's function arguments.
797        // Thus, we simply return None here for the argument types rather than returning early.
798        let (func_arg_types, unconstrained) =
799            if let Type::Function(args, _, _, unconstrained) = &func_type {
800                (Some(args), *unconstrained)
801            } else {
802                (None, false)
803            };
804
805        // When calling an unconstrained function, we can elaborate lambda arguments to be unconstrained.
806        let was_in_unconstrained_args =
807            std::mem::replace(&mut self.in_unconstrained_args, unconstrained);
808
809        let mut arguments = Vec::with_capacity(call.arguments.len());
810        let args = vecmap(call.arguments.into_iter().enumerate(), |(arg_index, arg)| {
811            let location = arg.location;
812            let expected_type = func_arg_types.and_then(|args| args.get(arg_index));
813
814            let (arg, typ) = if is_macro_call {
815                self.elaborate_in_comptime_context(|this| {
816                    this.elaborate_expression_with_target_type(arg, expected_type)
817                })
818            } else {
819                self.elaborate_expression_with_target_type(arg, expected_type)
820            };
821
822            // Try to unify this argument type against the function's argument type
823            // so that a potential lambda following this argument can have more concrete types.
824            if let Some(expected_type) = expected_type {
825                let _ = typ.unify(expected_type);
826            }
827
828            arguments.push(arg);
829            (typ, arg, location)
830        });
831
832        let hir_call = HirCallExpression { func, arguments, location, is_macro_call };
833        let typ = self.type_check_call(&hir_call, func_type, args, location);
834
835        // Restore the old one after type checking.
836        self.in_unconstrained_args = was_in_unconstrained_args;
837
838        (hir_call, typ)
839    }

870    /// Helper function containing the elaboration logic for a method call.
871    /// Returns the desugared function call and its type.
872    #[tracing::instrument(level = "trace", skip_all)]
873    fn elaborate_method_call_inner(
874        &mut self,
875        method_call: MethodCallExpression,
876        location: Location,
877    ) -> (HirCallExpression, Type) {
878        let object_location = method_call.object.location;
879        let (mut object, mut object_type) = self.elaborate_expression(method_call.object);
880        object_type = object_type.follow_bindings();
881
882        let method_name_location = method_call.method_name.location();
883        let method_name = method_call.method_name.as_str();
884        let check_self_param = true;
885
886        let method_ref = self.lookup_method(
887            &object_type,
888            method_name,
889            location,
890            object_location,
891            check_self_param,
892        );
893        let Some(method_ref) = method_ref else {
894            // Return a dummy call expression with Error type
895            let error_func =
896                self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
897            let error_call = HirCallExpression {
898                func: error_func,
899                arguments: Vec::new(),
900                location,
901                is_macro_call: method_call.is_macro_call,
902            };
903            return (error_call, Type::Error);
904        };
905
906        // Automatically add `&mut` if the method expects a mutable reference and
907        // the object is not already one.
908        let func_id = method_ref
909            .func_id(self.interner)
910            .expect("Expected trait function to be a DefinitionKind::Function");
911
912        let function_type = self.interner.function_meta(&func_id).typ.clone();
913        self.try_add_mutable_reference_to_object(&function_type, &mut object_type, &mut object);
914
915        let generics = method_call.generics;
916        let generics = generics.map(|generics| {
917            vecmap(generics, |generic| {
918                let location = generic.location;
919                let wildcard_allowed = WildcardAllowed::Yes;
920                let typ = self.use_type_with_kind(generic, &Kind::Any, wildcard_allowed);
921                Located::from(location, typ)
922            })
923        });
924        let generics = self.resolve_function_turbofish_generics(&func_id, generics, location);
925
926        let location = object_location.merge(method_name_location);
927
928        let (function_id, function_name) = method_ref.into_function_id_and_name(
929            object_type.clone(),
930            generics.clone(),
931            location,
932            self.interner,
933        );
934
935        let func_type = self.type_check_variable(function_name, &function_id, generics.clone());
936
937        let function_id = self.intern_expr_type(function_id, func_type.clone());
938
939        let func_arg_types =
940            if let Type::Function(args, _, _, _) = &func_type { Some(args) } else { None };
941
942        // Try to unify the object type with the first argument of the function.
943        // The reason to do this is that many methods that take a lambda will yield `self` or part of `self`
944        // as a parameter. By unifying `self` with the first argument we'll potentially get more
945        // concrete types in the arguments that are function types, which will later be passed as
946        // lambda parameter hints.
947        if let Some(first_arg_type) = func_arg_types.and_then(|args| args.first()) {
948            let _ = first_arg_type.unify(&object_type);
949        }
950
951        // These arguments will be given to the desugared function call.
952        // Compared to the method arguments, they also contain the object.
953        let mut function_args = Vec::with_capacity(method_call.arguments.len() + 1);
954        let mut arguments = Vec::with_capacity(method_call.arguments.len());
955
956        function_args.push((object_type.clone(), object, object_location));
957
958        for (arg_index, arg) in method_call.arguments.into_iter().enumerate() {
959            let location = arg.location;
960            // The argument types also contain the object type as the first argument.
961            // Thus, we need to add one when indexing the argument types to match them up with method arguments.
962            let expected_type = func_arg_types.and_then(|args| args.get(arg_index + 1));
963            let (arg, typ) = self.elaborate_expression_with_target_type(arg, expected_type);
964
965            // Try to unify this argument type against the function's argument type
966            // so that a potential lambda following this argument can have more concrete types.
967            if let Some(expected_type) = expected_type {
968                let _ = expected_type.unify(&typ);
969            }
970
971            arguments.push(arg);
972            function_args.push((typ, arg, location));
973        }
974
975        let method = method_call.method_name;
976        let is_macro_call = method_call.is_macro_call;
977        let method_call = HirMethodCallExpression { method, object, arguments, location, generics };
978
979        self.check_method_call_visibility(func_id, &object_type, &method_call.method);
980
981        // Desugar the method call into a normal, resolved function call
982        // so that the backend doesn't need to worry about methods
983        let function_call = method_call.into_function_call(function_id, is_macro_call, location);
984
985        self.interner.add_function_reference(func_id, method_name_location);
986
987        // Type check the new call now that it has been changed from a method call
988        // to a function call. This way we avoid duplicating code.
989        let typ = self.type_check_call(&function_call, func_type, function_args, location);
990
991        (function_call, typ)
992    }

994    #[tracing::instrument(level = "trace", skip_all)]
995    pub(super) fn elaborate_constrain(
996        &mut self,
997        mut expr: ConstrainExpression,
998    ) -> (HirExpression, Type) {
999        let location = expr.location;
1000        let min_args_count = expr.kind.required_arguments_count();
1001        let actual_args_count = expr.arguments.len();
1002        let has_optional_msg = actual_args_count == min_args_count + 1;
1003
1004        let (message, expr) = if actual_args_count != min_args_count && !has_optional_msg {
1005            self.push_err(TypeCheckError::AssertionParameterCountMismatch {
1006                kind: expr.kind,
1007                found: actual_args_count,
1008                location,
1009            });
1010
1011            // Given that we already produced an error, let's make this an `assert(true)` so
1012            // we don't get further errors.
1013            let message = None;
1014            let kind = ExpressionKind::Literal(Literal::Bool(true));
1015            let expr = Expression { kind, location };
1016            (message, expr)
1017        } else {
1018            let message = has_optional_msg.then(|| expr.arguments.pop().unwrap());
1019            let expr = match expr.kind {
1020                ConstrainKind::Assert | ConstrainKind::Constrain => expr.arguments.pop().unwrap(),
1021                ConstrainKind::AssertEq => {
1022                    let rhs = expr.arguments.pop().unwrap();
1023                    let lhs = expr.arguments.pop().unwrap();
1024                    let location = lhs.location.merge(rhs.location);
1025                    let operator = Located::from(location, BinaryOpKind::Equal);
1026                    let kind =
1027                        ExpressionKind::Infix(Box::new(InfixExpression { lhs, operator, rhs }));
1028                    Expression { kind, location }
1029                }
1030            };
1031            (message, expr)
1032        };
1033
1034        let expr_location = expr.location;
1035        let (expr_id, expr_type) = self.elaborate_expression(expr);
1036
1037        // Must type check the assertion message expression so that we instantiate bindings
1038        let msg = message.map(|assert_msg_expr| {
1039            let (msg, typ) = self.elaborate_expression(assert_msg_expr);
1040            // If the error message contains a format string, those types need to appear in the ABI,
1041            // except if we are in a meta-programming context, in which case the comptime interpreter
1042            // handles a wider variety of types, e.g. quoted types.
1043            if !self.in_comptime_context() {
1044                let location = self.interner.expr_location(&msg);
1045                let typ = typ.follow_bindings();
1046                let mut check_msg_compat = |typ: &Type| {
1047                    if typ.is_message_compatible(false) || matches!(typ, Type::Error) {
1048                        return;
1049                    }
1050                    let error = TypeCheckError::TypeCannotBeUsed {
1051                        typ: typ.clone(),
1052                        place: "message",
1053                        location,
1054                    };
1055                    self.push_err(CompilationError::TypeError(error));
1056                };
1057                if let Type::FmtString(_, item_types) = typ {
1058                    if let Type::Tuple(item_types) = item_types.as_ref() {
1059                        item_types.iter().for_each(check_msg_compat);
1060                    }
1061                } else {
1062                    check_msg_compat(&typ);
1063                }
1064            }
1065            msg
1066        });
1067
1068        self.unify_or_type_mismatch(&expr_type, &Type::Bool, expr_location);
1069
1070        (HirExpression::Constrain(HirConstrainExpression(expr_id, location.file, msg)), Type::Unit)
1071    }

1073    /// Elaborate a struct constructor.
1074    ///
1075    /// This method resolves the [UnresolvedType][crate::ast::UnresolvedType] into the [Type] being constructed,
1076    /// then delegates to [Elaborator::elaborate_constructor_with_type] to handle the fields.
1077    #[tracing::instrument(level = "trace", skip_all)]
1078    fn elaborate_constructor(
1079        &mut self,
1080        constructor: ConstructorExpression,
1081    ) -> (HirExpression, Type) {
1082        let location = constructor.typ.location;
1083
1084        // A constructor type can either be a Path or an interned UnresolvedType.
1085        // We represent both as UnresolvedType (with Path being a Named UnresolvedType)
1086        // and error if we don't get a Named path.
1087        let mut typ = constructor.typ.typ;
1088        if let UnresolvedTypeData::Interned(id) = typ {
1089            typ = self.interner.get_unresolved_type_data(id).clone();
1090        }
1091        if let UnresolvedTypeData::Resolved(id) = typ {
1092            // If this type is already resolved we can skip the rest of this function
1093            // which just resolves the type, and go straight to resolving the fields.
1094            let resolved = self.interner.get_quoted_type(id).clone();
1095            return self.elaborate_constructor_with_type(
1096                resolved,
1097                constructor.fields,
1098                location,
1099                None,
1100            );
1101        }
1102        let UnresolvedTypeData::Named(mut path, generics, _) = typ else {
1103            self.push_err(ResolverError::NonStructUsedInConstructor {
1104                typ: typ.to_string(),
1105                location,
1106            });
1107            return (HirExpression::Error, Type::Error);
1108        };
1109
1110        // When instantiating a generic struct, treat any generics in the type
1111        // as if they were part of the turbofish, so they can be validated with the path.
1112        if !generics.ordered_args.is_empty() {
1113            let last_segment = path.segments.last_mut().unwrap();
1114            last_segment.generics = Some(generics.ordered_args);
1115        }
1116
1117        let path = self.validate_path(path);
1118        let last_segment = path.last_segment();
1119
1120        let Some(typ) = self.lookup_type_or_error(path) else {
1121            return (HirExpression::Error, Type::Error);
1122        };
1123
1124        self.elaborate_constructor_with_type(typ, constructor.fields, location, Some(last_segment))
1125    }

1127    /// Knowing the [Type] being constructed, elaborate all field expressions.
1128    #[tracing::instrument(level = "trace", skip_all)]
1129    fn elaborate_constructor_with_type(
1130        &mut self,
1131        typ: Type,
1132        fields: Vec<(Ident, Expression)>,
1133        location: Location,
1134        last_segment: Option<TypedPathSegment>,
1135    ) -> (HirExpression, Type) {
1136        let typ = typ.follow_bindings_shallow();
1137        let (struct_type, generics) = match typ.as_ref() {
1138            Type::DataType(struct_type, struct_generics) if struct_type.borrow().is_struct() => {
1139                (struct_type.clone(), struct_generics)
1140            }
1141            typ => {
1142                self.push_err(ResolverError::NonStructUsedInConstructor {
1143                    typ: typ.to_string(),
1144                    location,
1145                });
1146                return (HirExpression::Error, Type::Error);
1147            }
1148        };
1149        let struct_id = struct_type.borrow().id;
1150
1151        self.mark_struct_as_constructed(struct_type.clone());
1152
1153        // `last_segment` is optional if this constructor was resolved from a quoted type
1154        let mut generics = generics.clone();
1155        let mut is_self_type = false;
1156        let mut constructor_type_location = location;
1157
1158        if let Some(last_segment) = last_segment {
1159            let turbofish_location = last_segment.turbofish_location();
1160            is_self_type = last_segment.ident.is_self_type_name();
1161            constructor_type_location = last_segment.ident.location();
1162
1163            let mut errors = Vec::new();
1164            generics = self.resolve_struct_turbofish_generics(
1165                &struct_type.borrow(),
1166                generics,
1167                last_segment.generics,
1168                turbofish_location,
1169                &mut errors,
1170            );
1171            self.push_errors(errors);
1172        }
1173
1174        // Each of the struct generics must be bound at the end of the function
1175        for (index, generic) in generics.iter().enumerate() {
1176            if let Type::TypeVariable(type_variable) = generic {
1177                self.push_required_type_variable(
1178                    type_variable.id(),
1179                    Type::TypeVariable(type_variable.clone()),
1180                    BindableTypeVariableKind::StructGeneric { struct_id, index },
1181                    location,
1182                );
1183            }
1184        }
1185
1186        let field_types = struct_type
1187            .borrow()
1188            .get_fields_with_visibility(&generics)
1189            .expect("This type should already be validated to be a struct");
1190
1191        let fields = self.resolve_constructor_expr_fields(
1192            struct_type.clone(),
1193            field_types,
1194            fields,
1195            location,
1196        );
1197        let expr = HirExpression::Constructor(HirConstructorExpression {
1198            fields,
1199            r#type: struct_type.clone(),
1200            struct_generics: generics.clone(),
1201        });
1202
1203        self.interner.add_type_reference(struct_id, constructor_type_location, is_self_type);
1204
1205        (expr, Type::DataType(struct_type, generics))
1206    }

1216    /// Resolve all the fields of a struct constructor expression.
1217    /// Ensures all fields are present, none are repeated, and all
1218    /// are part of the struct.
1219    #[tracing::instrument(level = "trace", skip_all)]
1220    fn resolve_constructor_expr_fields(
1221        &mut self,
1222        struct_type: Shared<DataType>,
1223        field_types: Vec<(String, ItemVisibility, Type)>,
1224        fields: Vec<(Ident, Expression)>,
1225        location: Location,
1226    ) -> Vec<(Ident, ExprId)> {
1227        let mut ret = Vec::with_capacity(fields.len());
1228        let mut seen_fields = HashSet::default();
1229        let mut unseen_fields = struct_type
1230            .borrow()
1231            .field_names()
1232            .expect("This type should already be validated to be a struct");
1233
1234        let expected_fields_by_name = field_types
1235            .iter()
1236            .enumerate()
1237            .map(|(i, (name, vis, typ))| (name.as_str(), (i, vis, typ)))
1238            .collect::<HashMap<_, _>>();
1239
1240        for (field_name, field) in fields {
1241            let expected_field = expected_fields_by_name.get(field_name.as_str());
1242
1243            let expected_index_and_visibility =
1244                expected_field.map(|(index, visibility, _)| (index, visibility));
1245            let expected_type = expected_field.map_or(&&Type::Error, |(_, _, typ)| typ);
1246
1247            let field_location = field.location;
1248            let (resolved, field_type) = self.elaborate_expression(field);
1249
1250            if unseen_fields.remove(&field_name) {
1251                seen_fields.insert(field_name.clone());
1252
1253                self.unify_with_coercions(
1254                    &field_type,
1255                    expected_type,
1256                    resolved,
1257                    field_location,
1258                    |elaborator| {
1259                        CompilationError::TypeError(elaborator.new_type_mismatch_error(
1260                            &field_type,
1261                            expected_type,
1262                            field_location,
1263                        ))
1264                    },
1265                );
1266            } else if seen_fields.contains(&field_name) {
1267                // duplicate field
1268                self.push_err(ResolverError::DuplicateField { field: field_name.clone() });
1269            } else {
1270                // field not required by struct
1271                self.push_err(ResolverError::NoSuchField {
1272                    field: field_name.clone(),
1273                    struct_definition: struct_type.borrow().name.clone(),
1274                });
1275            }
1276
1277            if let Some((index, visibility)) = expected_index_and_visibility {
1278                let struct_type = struct_type.borrow();
1279                let field_location = field_name.location();
1280                let field_name = field_name.as_str();
1281                self.check_struct_field_visibility(
1282                    &struct_type,
1283                    field_name,
1284                    **visibility,
1285                    field_location,
1286                );
1287
1288                self.interner.add_struct_member_reference(struct_type.id, *index, field_location);
1289            }
1290
1291            ret.push((field_name, resolved));
1292        }
1293
1294        if !unseen_fields.is_empty() {
1295            self.push_err(ResolverError::MissingFields {
1296                location,
1297                missing_fields: unseen_fields.into_iter().map(|field| field.to_string()).collect(),
1298                struct_definition: struct_type.borrow().name.clone(),
1299            });
1300        }
1301
1302        ret
1303    }

1306    /// `is_offset`:
1307    /// - `is_offset = false`: Auto-dereferencing will occur, and this will always return false
1308    /// - `is_offset = true`: Auto-dereferencing is disabled, and this will return true if the lhs
1309    ///   is a reference.
1310    #[tracing::instrument(level = "trace", skip_all)]
1311    fn elaborate_member_access(
1312        &mut self,
1313        access: MemberAccessExpression,
1314        location: Location,
1315        is_offset: bool,
1316    ) -> (ExprId, Type, bool) {
1317        // We don't need the boolean 'skipped auto-dereferences' from elaborate_reference_expression
1318        // since if we have skipped any then `lhs_type` will be a reference and we will need to
1319        // skip the deref (if is_offset is true) here anyway to extract the field out of the reference.
1320        // This is more reliable than using the boolean return value here since the return value
1321        // doesn't account for reference variables which we need to account for.
1322        let (lhs, lhs_type, _) = self.elaborate_reference_expression(access.lhs);
1323        let is_reference = lhs_type.is_ref();
1324
1325        let rhs = access.rhs;
1326        let rhs_location = rhs.location();
1327        // `is_offset` is only used when lhs is a reference and we want to return a reference to rhs
1328        let access = HirMemberAccess { lhs, rhs, is_offset };
1329        let expr_id = self.intern_expr(HirExpression::MemberAccess(access.clone()), location);
1330        let typ = self.type_check_member_access(access, *expr_id, lhs_type, rhs_location);
1331        let expr_id = self.intern_expr_type(expr_id, typ.clone());
1332        (expr_id, typ, is_offset && is_reference)
1333    }

1351    /// Elaborate the expression, resolve the target type, then type check that they are compatible.
1352    #[tracing::instrument(level = "trace", skip_all)]
1353    fn elaborate_cast(
1354        &mut self,
1355        cast: CastExpression,
1356        location: Location,
1357    ) -> (HirExpression, Type) {
1358        let (lhs, lhs_type) = self.elaborate_expression(cast.lhs);
1359        let wildcard_allowed = WildcardAllowed::No(WildcardDisallowedContext::Cast);
1360        let r#type = self.resolve_type(cast.r#type, wildcard_allowed);
1361        let result = self.check_cast(&lhs, &lhs_type, &r#type, location);
1362
1363        // `Field as u1` is not supported directly by the backend. Insert an intermediate
1364        // cast to u8 to transform it into: `(Field as u8) as u1`.
1365        let lhs_could_be_field = match lhs_type.follow_bindings() {
1366            Type::FieldElement => true,
1367            Type::TypeVariable(ref var) => var.is_integer_or_field(),
1368            _ => false,
1369        };
1370        let lhs = if lhs_could_be_field
1371            && matches!(r#type, Type::Integer(Signedness::Unsigned, IntegerBitSize::One))
1372        {
1373            let u8_type = Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight);
1374            let cast_to_u8 =
1375                HirExpression::Cast(HirCastExpression { lhs, r#type: u8_type.clone() });
1376            self.interner.push_expr_full(cast_to_u8, location, u8_type)
1377        } else {
1378            lhs
1379        };
1380
1381        let expr = HirExpression::Cast(HirCastExpression { lhs, r#type });
1382        (expr, result)
1383    }

1418    /// Handles the results of [Self::prefix_operand_type_rules] and [Self::infix_operand_type_rules].
1419    /// * if the rules returned an `Err`, it returns [Type::Error]
1420    /// * if the results indicate that a trait method should be used,
1421    ///   it pushes a trait constraint and checks that the expression type is compatible with the trait method
1422    #[tracing::instrument(level = "trace", skip_all)]
1423    fn handle_operand_type_rules_result(
1424        &mut self,
1425        result: Result<(Type, bool), TypeCheckError>,
1426        operand_type: &Type,
1427        trait_method_id: Option<TraitItemId>,
1428        expr_id: ExprId,
1429        location: Location,
1430        is_ord: bool,
1431    ) -> Type {
1432        match result {
1433            Ok((typ, use_impl)) => {
1434                if use_impl {
1435                    let trait_method_id = trait_method_id
1436                        .expect("ICE: expected some trait_method_id when use_impl is true");
1437
1438                    // Delay checking the trait constraint until the end of the function.
1439                    // Checking it now could bind an unbound type variable to any type
1440                    // that implements the trait.
1441                    let trait_id = trait_method_id.trait_id;
1442                    let trait_generics = TraitGenerics::default();
1443                    let trait_bound = ResolvedTraitBound { trait_id, trait_generics, location };
1444                    let constraint = TraitConstraint { typ: operand_type.clone(), trait_bound };
1445                    let select_impl = true; // this constraint should lead to choosing a trait impl
1446                    self.push_trait_constraint(constraint, expr_id, select_impl);
1447
1448                    self.type_check_operator_method(
1449                        expr_id,
1450                        trait_method_id,
1451                        operand_type,
1452                        &typ,
1453                        location,
1454                        is_ord,
1455                    );
1456                }
1457                typ
1458            }
1459            Err(error) => {
1460                self.push_err(error);
1461                Type::Error
1462            }
1463        }
1464    }

1466    #[tracing::instrument(level = "trace", skip_all)]
1467    fn elaborate_if(
1468        &mut self,
1469        if_expr: IfExpression,
1470        target_type: Option<&Type>,
1471    ) -> (HirExpression, Type) {
1472        let expr_location = if_expr.condition.type_location();
1473        let consequence_location = if_expr.consequence.type_location();
1474        let (condition, cond_type) = self.elaborate_expression(if_expr.condition);
1475        let (consequence, mut ret_type) =
1476            self.elaborate_expression_with_target_type(if_expr.consequence, target_type);
1477
1478        self.unify_or_type_mismatch(&cond_type, &Type::Bool, expr_location);
1479
1480        let (alternative, else_type, error_location) =
1481            if let Some(alternative) = if_expr.alternative {
1482                let alternative_location = alternative.type_location();
1483                let (else_, else_type) =
1484                    self.elaborate_expression_with_target_type(alternative, target_type);
1485                (Some(else_), else_type, alternative_location)
1486            } else {
1487                (None, Type::Unit, consequence_location)
1488            };
1489
1490        self.unify(&else_type, &ret_type, |elaborator| {
1491            let err = elaborator.new_type_mismatch_error(&else_type, &ret_type, error_location);
1492
1493            let context = if ret_type == Type::Unit {
1494                "Are you missing a semicolon at the end of your 'else' branch?"
1495            } else if else_type == Type::Unit {
1496                "Are you missing a semicolon at the end of the first block of this 'if'?"
1497            } else {
1498                "Expected the types of both if branches to be equal"
1499            };
1500
1501            err.add_context(context)
1502        });
1503
1504        if alternative.is_none() {
1505            ret_type = Type::Unit;
1506        }
1507
1508        let if_expr = HirIfExpression { condition, consequence, alternative };
1509        (HirExpression::If(if_expr), ret_type)
1510    }

1512    /// Elaborate a `match <expr> { <rules> }` expression by creating an block such as this:
1513    /// ```text
1514    /// {
1515    ///   let internal variable = <expr>;
1516    ///   match internal variable { <rules> }
1517    /// }
1518    /// ```
1519    #[tracing::instrument(level = "trace", skip_all)]
1520    fn elaborate_match(
1521        &mut self,
1522        match_expr: MatchExpression,
1523        location: Location,
1524    ) -> (HirExpression, Type) {
1525        // Show error on the `match` keyword
1526        let match_location = Location::new(
1527            Span::from(location.span.start()..location.span.start() + 5),
1528            location.file,
1529        );
1530        self.use_unstable_feature(UnstableFeature::Enums, match_location);
1531
1532        let expr_location = match_expr.expression.location;
1533        let (expression, typ) = self.elaborate_expression(match_expr.expression);
1534        let (let_, variable) = self.wrap_in_let(expression, typ.clone());
1535
1536        let (errored, (rows, result_type)) =
1537            self.errors_occurred_in(|this| this.elaborate_match_rules(variable, match_expr.rules));
1538
1539        // Avoid calling `elaborate_match_rows` if there were errors while constructing
1540        // the match rows - it'll just lead to extra errors like `unreachable pattern`
1541        // warnings on branches which previously had type errors.
1542        let tree = HirExpression::Match(if !errored {
1543            self.elaborate_match_rows(rows, &typ, expr_location)
1544        } else {
1545            HirMatch::Failure { missing_case: false }
1546        });
1547
1548        let tree = self.interner.push_expr_full(tree, location, result_type.clone());
1549
1550        let tree = self.interner.push_stmt_full(HirStatement::Expression(tree), location);
1551
1552        let block = HirExpression::Block(HirBlockExpression { statements: vec![let_, tree] });
1553        (block, result_type)
1554    }

1611    /// For elaborating a lambda we might get `parameters_type_hints`. These come from a potential
1612    /// call that has this lambda as the argument. The parameter type hints will be the types of
1613    /// the function type corresponding to the lambda argument.
1614    ///
1615    /// The `unconstrained` parameter is set based on whether the lambda is expected to be unconstrained
1616    /// by the function we are passing it to. If we just assign the lambda to a variable, then it's `false`.
1617    #[tracing::instrument(level = "trace", skip_all)]
1618    fn elaborate_lambda_with_parameter_type_hints(
1619        &mut self,
1620        lambda: Lambda,
1621        parameters_type_hints: Option<&Vec<Type>>,
1622        unconstrained: bool,
1623    ) -> (HirExpression, Type) {
1624        self.push_scope();
1625        let scope_index = self.scopes.current_scope_index();
1626
1627        self.lambda_stack.push(LambdaContext { captures: Vec::new(), scope_index, unconstrained });
1628
1629        let mut arg_types = Vec::with_capacity(lambda.parameters.len());
1630        let mut parameter_names_in_list = HashMap::default();
1631        let parameters =
1632            vecmap(lambda.parameters.into_iter().enumerate(), |(index, (pattern, typ))| {
1633                let parameter = DefinitionKind::Local(None);
1634                let typ = match typ {
1635                    Some(typ) => {
1636                        let wildcard_allowed = WildcardAllowed::Yes;
1637                        self.resolve_type(typ, wildcard_allowed)
1638                    }
1639                    None => {
1640                        if let Some(parameter_type_hint) =
1641                            parameters_type_hints.and_then(|hints| hints.get(index))
1642                        {
1643                            parameter_type_hint.clone()
1644                        } else {
1645                            self.interner.next_type_variable_with_kind(Kind::Any)
1646                        }
1647                    }
1648                };
1649
1650                arg_types.push(typ.clone());
1651                (
1652                    self.elaborate_pattern(
1653                        pattern,
1654                        typ.clone(),
1655                        parameter,
1656                        true, // warn_if_unused
1657                        true, // warn_if_not_mutated
1658                        &mut parameter_names_in_list,
1659                    ),
1660                    typ,
1661                )
1662            });
1663
1664        let wildcard_allowed = WildcardAllowed::Yes;
1665        let return_type = self.resolve_inferred_type(lambda.return_type, wildcard_allowed);
1666        let body_location = lambda.body.location;
1667        let (body, body_type) = self.elaborate_expression(lambda.body);
1668
1669        let lambda_context = self.lambda_stack.pop().unwrap();
1670        self.pop_scope();
1671
1672        self.unify_or_type_mismatch(&body_type, &return_type, body_location);
1673
1674        let captured_vars = vecmap(&lambda_context.captures, |capture| {
1675            self.interner.definition_type(capture.ident.id)
1676        });
1677
1678        let env_type =
1679            if captured_vars.is_empty() { Type::Unit } else { Type::Tuple(captured_vars) };
1680
1681        let captures = lambda_context.captures;
1682        let expr = HirExpression::Lambda(HirLambda {
1683            parameters,
1684            return_type,
1685            body,
1686            captures,
1687            unconstrained,
1688        });
1689        (expr, Type::Function(arg_types, Box::new(body_type), Box::new(env_type), unconstrained))
1690    }

1712            let (block, block_type) = this.elaborate_block_expression(block, target_type);
1713
1714            // If the comptime block is expected to return a specific type, unify their types.
1715            // This for example allows this code to compile: `let x: u8 = comptime { 1 }`.
1716            // If we don't do this, "1" will end up with the default integer or field type,
1717            // which is Field.
1718            if let Some(target_type) = target_type {
1719                this.unify_or_type_mismatch(&block_type, target_type, location);
1720            }
1721
1722            block
1723        });
1724
1725        let mut interpreter = self.setup_interpreter();
1726        let value = interpreter.evaluate_block(block);
1727
1728        let (id, typ) = self.inline_comptime_value(value, location);
1729
1730        let location = self.interner.id_location(id);
1731        self.debug_comptime(location, |interner| {
1732            interner.expression(&id).to_display_ast(interner, location).kind
1733        });
1734
1735        (id, typ)
1736    }

1738    #[tracing::instrument(level = "trace", skip_all)]
1739    pub fn inline_comptime_value(
1740        &mut self,
1741        value: Result<comptime::Value, InterpreterError>,
1742        location: Location,
1743    ) -> (ExprId, Type) {
1744        let make_error = |this: &mut Self, error: InterpreterError| {
1745            let error: CompilationError = error.into();
1746            this.push_err(error);
1747            let typ = Type::Error;
1748            let error = this.interner.push_expr_full(HirExpression::Error, location, typ.clone());
1749            (error, typ)
1750        };
1751
1752        let value = match value {
1753            Ok(value) => value,
1754            Err(error) => return make_error(self, error),
1755        };
1756
1757        match value.into_expression(self, location) {
1758            Ok(new_expr) => {
1759                // At this point the Expression was already elaborated and we got a Value.
1760                // We'll elaborate this value turned into Expression to inline it and get
1761                // an ExprId and Type, but we don't want any visibility errors to happen
1762                // here (they could if we have `Foo { inner: 5 }` and `inner` is not
1763                // accessible from where this expression is being elaborated).
1764                self.silence_field_visibility_errors += 1;
1765                let value = self.elaborate_expression(new_expr);
1766                self.silence_field_visibility_errors -= 1;
1767                value
1768            }
1769            Err(error) => make_error(self, error),
1770        }
1771    }

1773    fn try_get_comptime_function(
1774        &self,
1775        func: ExprId,
1776        location: Location,
1777    ) -> Result<FuncId, ResolverError> {
1778        match self.interner.expression(&func) {
1779            HirExpression::Ident(ident, _generics) => {
1780                let definition = self.interner.definition(ident.id);
1781                if let DefinitionKind::Function(function) = definition.kind {
1782                    let meta = self.interner.function_modifiers(&function);
1783                    if meta.is_comptime {
1784                        Ok(function)
1785                    } else {
1786                        Err(ResolverError::MacroIsNotComptime { location })
1787                    }
1788                } else {
1789                    Err(ResolverError::InvalidSyntaxInMacroCall { location })
1790                }
1791            }
1792            _ => Err(ResolverError::InvalidSyntaxInMacroCall { location }),
1793        }
1794    }

1796    /// Validate a macro call without executing it.
1797    /// Checks that the callee is a comptime function and the return type is `Quoted`.
1798    #[tracing::instrument(level = "trace", skip_all)]
1799    fn validate_macro_call(
1800        &mut self,
1801        func: ExprId,
1802        return_type: &Type,
1803        location: Location,
1804    ) -> Option<FuncId> {
1805        self.unify(return_type, &Type::Quoted(QuotedType::Quoted), |_| {
1806            TypeCheckError::MacroReturningNonExpr { typ: return_type.clone(), location }
1807        });
1808
1809        match self.try_get_comptime_function(func, location) {
1810            Ok(function) => Some(function),
1811            Err(error) => {
1812                self.push_err(error);
1813                None
1814            }
1815        }
1816    }

1818    /// Call a macro function and inlines its code at the call site.
1819    /// This will also perform a type check to ensure that the return type is an `Expr` value.
1820    #[tracing::instrument(level = "trace", skip_all)]
1821    fn call_macro(
1822        &mut self,
1823        func: ExprId,
1824        arguments: Vec<ExprId>,
1825        location: Location,
1826        return_type: Type,
1827    ) -> Option<(HirExpression, Type)> {
1828        let function = self.validate_macro_call(func, &return_type, location)?;
1829
1830        let mut interpreter = self.setup_interpreter();
1831        let mut comptime_args = Vec::new();
1832        let mut errors: Vec<CompilationError> = Vec::new();
1833
1834        for argument in arguments {
1835            match interpreter.evaluate(argument) {
1836                Ok(arg) => {
1837                    let location = interpreter.elaborator.interner.expr_location(&argument);
1838                    comptime_args.push((arg, location));
1839                }
1840                Err(error) => errors.push(error.into()),
1841            }
1842        }
1843
1844        let bindings = interpreter.elaborator.interner.get_instantiation_bindings(func).clone();
1845        let result = interpreter.call_function(function, comptime_args, bindings, location);
1846
1847        if !errors.is_empty() {
1848            self.push_errors(errors);
1849            return None;
1850        }
1851
1852        let (expr_id, typ) = self.inline_comptime_value(result, location);
1853        Some((self.interner.expression(&expr_id), typ))
1854    }

1856    #[tracing::instrument(level = "trace", skip_all)]
1857    fn elaborate_as_trait_path(&mut self, path: AsTraitPath) -> (ExprId, Type) {
1858        let location = path.typ.location.merge(path.trait_path.location);
1859
1860        let constraint = UnresolvedTraitConstraint {
1861            typ: path.typ,
1862            trait_bound: TraitBound {
1863                trait_path: path.trait_path,
1864                trait_generics: path.trait_generics,
1865            },
1866        };
1867
1868        let wildcard_allowed = WildcardAllowed::Yes;
1869        let typ = self.use_type(constraint.typ.clone(), wildcard_allowed);
1870        let Some(trait_bound) = self.use_trait_bound(&constraint.trait_bound) else {
1871            // resolve_trait_bound only returns None if it has already issued an error, so don't
1872            // issue another here.
1873            let error = self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
1874            return (error, Type::Error);
1875        };
1876
1877        let constraint = TraitConstraint { typ, trait_bound };
1878
1879        let the_trait = self.interner.get_trait(constraint.trait_bound.trait_id);
1880        let self_type = the_trait.self_type_typevar.clone();
1881        let kind = the_trait.self_type_typevar.kind();
1882
1883        let Some(definition) =
1884            the_trait.find_method_or_constant(path.impl_item.as_str(), self.interner)
1885        else {
1886            let trait_name = the_trait.name.to_string();
1887            let method_name = path.impl_item.to_string();
1888            let location = path.impl_item.location();
1889            self.push_err(ResolverError::NoSuchMethodInTrait { trait_name, method_name, location });
1890            let error = self.interner.push_expr_full(HirExpression::Error, location, Type::Error);
1891            return (error, Type::Error);
1892        };
1893
1894        let trait_item = TraitItem { definition, constraint: constraint.clone(), assumed: false };
1895
1896        let ident = HirIdent {
1897            location: path.impl_item.location(),
1898            id: definition,
1899            impl_kind: ImplKind::TraitItem(trait_item),
1900        };
1901
1902        let id = self.intern_expr(HirExpression::Ident(ident.clone(), None), location);
1903
1904        let mut bindings = TypeBindings::default();
1905
1906        // In `<Type as Trait>::method` we know `Self` is `Type` so we bind that now
1907        bindings.insert(self_type.id(), (self_type, kind, constraint.typ));
1908
1909        // TODO: set this to `true`. See https://github.com/noir-lang/noir/issues/8687
1910        let push_required_type_variables = self.current_trait.is_none();
1911
1912        let typ = self.type_check_variable_with_bindings(
1913            ident,
1914            &id,
1915            None,
1916            bindings,
1917            push_required_type_variables,
1918        );
1919        let id = self.intern_expr_type(id, typ.clone());
1920        (id, typ)
1921    }
