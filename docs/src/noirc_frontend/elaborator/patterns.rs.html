28/// Represents a variable in the source code.

32    /// The scope index where the variable is declared.

47    /// Elaborate a pattern, which can appear in a `let <pattern> = <expr>`, or a `match` statement.
48    ///
49    /// The `definition_kind` specifies the kind of variables the pattern will create, e.g. a local or global variable.
50    ///
51    /// The `expected_type` is always known, because we can first infer the type of the `<expr>` and try to match it to
52    /// the pattern.
53    ///
54    /// `parameter_names_in_list` keeps track of parameter names, and their location, across multiple
55    /// patterns in a list. If a name is found multiple times, an error is captured.
56    #[tracing::instrument(level = "trace", skip_all)]
57    pub(super) fn elaborate_pattern(
58        &mut self,
59        pattern: Pattern,
60        expected_type: Type,
61        definition_kind: DefinitionKind,
62        warn_if_unused: bool,
63        warn_if_not_mutated: bool,
64        parameter_names_in_list: &mut HashMap<String, Location>,
65    ) -> HirPattern {
66        self.elaborate_pattern_mut(
67            pattern,
68            expected_type,
69            definition_kind,
70            None,
71            &mut Vec::new(),
72            warn_if_unused,
73            warn_if_not_mutated,
74            &mut HashSet::default(),
75            parameter_names_in_list,
76        )
77    }

109    /// Elaborate the (potentially mutable) pattern and add the variables
110    /// it created to the scope if necessary.
111    ///
112    /// - `pattern_names` keeps track of parameter names within this single pattern (or an outer
113    ///   one, when called recursively). If a name is found multiple times, an error is captured.
114    /// - `parameter_names_in_list` keeps track of parameter names, and their location, across multiple
115    ///   patterns in a list. If a name is found multiple times, an error is captured.
116    #[allow(clippy::too_many_arguments)]
117    #[tracing::instrument(level = "trace", skip_all)]
118    fn elaborate_pattern_mut(
119        &mut self,
120        pattern: Pattern,
121        expected_type: Type,
122        definition: DefinitionKind,
123        // Location of the `mut` keyword.
124        mutable: Option<Location>,
125        new_definitions: &mut Vec<HirIdent>,
126        warn_if_unused: bool,
127        warn_if_not_mutated: bool,
128        pattern_names: &mut HashSet<String>,
129        parameter_names_in_list: &mut HashMap<String, Location>,
130    ) -> HirPattern {
131        match pattern {
132            // e.g. let <ident> = ...;
133            Pattern::Identifier(name) => {
134                // If this definition is mutable, do not store the RHS because it will
135                // not always refer to the correct value of the variable.
136                let definition = match (mutable, definition) {
137                    (Some(_), DefinitionKind::Local(_)) => DefinitionKind::Local(None),
138                    (_, other) => other,
139                };
140                let ident = if let DefinitionKind::Global(global_id) = definition {
141                    // Globals don't need to be added to scope, they're already in the def_maps
142                    let id = self.interner.get_global(global_id).definition_id;
143                    let location = name.location();
144                    HirIdent::non_trait_method(id, location)
145                } else {
146                    if name.as_str() != "_" {
147                        if !pattern_names.insert(name.to_string()) {
148                            self.push_err(ResolverError::PatternBoundMoreThanOnce {
149                                ident: name.clone(),
150                            });
151                        } else if let Some(first_location) =
152                            parameter_names_in_list.insert(name.to_string(), name.location())
153                        {
154                            self.push_err(ResolverError::DuplicateDefinition {
155                                name: name.to_string(),
156                                first_location,
157                                second_location: name.location(),
158                            });
159                        }
160                    }
161
162                    self.add_variable_decl(
163                        name,
164                        mutable.is_some(),
165                        true, // allow_shadowing
166                        warn_if_unused,
167                        warn_if_not_mutated,
168                        definition,
169                    )
170                };
171                self.interner.push_definition_type(ident.id, expected_type);
172                new_definitions.push(ident.clone());
173                HirPattern::Identifier(ident)
174            }
175            // e.g. let mut <pattern> = ...;
176            Pattern::Mutable(pattern, location, _) => {
177                if let Some(first_mut) = mutable {
178                    self.push_err(ResolverError::UnnecessaryMut {
179                        first_mut,
180                        second_mut: location,
181                    });
182                }
183
184                let pattern = self.elaborate_pattern_mut(
185                    *pattern,
186                    expected_type,
187                    definition,
188                    Some(location),
189                    new_definitions,
190                    warn_if_unused,
191                    warn_if_not_mutated,
192                    pattern_names,
193                    parameter_names_in_list,
194                );
195                HirPattern::Mutable(Box::new(pattern), location)
196            }
197            // e.g. let (<pattern 0>, <pattern 1>, ...) = ...;
198            Pattern::Tuple(fields, location) => {
199                // Returns Some for valid tuple types (where arity checking makes sense),
200                // None when we've already issued an error or have an invalid type.
201                let field_types = match expected_type.follow_bindings() {
202                    Type::Tuple(fields) => Some(fields),
203                    Type::Error => None,
204                    expected_type => {
205                        let tuple =
206                            Type::Tuple(vecmap(&fields, |_| self.interner.next_type_variable()));
207
208                        self.push_err(TypeCheckError::TypeMismatchWithSource {
209                            expected: expected_type,
210                            actual: tuple,
211                            location,
212                            source: Source::Assignment,
213                        });
214                        None
215                    }
216                };
217
218                // Only check tuple arity if the expected type was actually a tuple.
219                // If it wasn't, we've already issued a type mismatch error above.
220                if let Some(field_types) = &field_types
221                    && fields.len() != field_types.len()
222                {
223                    self.push_err(TypeCheckError::TupleMismatch {
224                        tuple_types: field_types.clone(),
225                        actual_count: fields.len(),
226                        location,
227                    });
228                }
229
230                let fields = vecmap(fields.into_iter().enumerate(), |(i, field)| {
231                    let field_type = field_types
232                        .as_ref()
233                        .and_then(|types| types.get(i).cloned())
234                        .unwrap_or(Type::Error);
235                    self.elaborate_pattern_mut(
236                        field,
237                        field_type,
238                        definition.clone(),
239                        mutable,
240                        new_definitions,
241                        warn_if_unused,
242                        warn_if_not_mutated,
243                        pattern_names,
244                        parameter_names_in_list,
245                    )
246                });
247                HirPattern::Tuple(fields, location)
248            }
249            // e.g. let <name> { <field 0>: <pattern 0>, <field 1>: <pattern 0>, ... } = ...'
250            Pattern::Struct(name, fields, location) => {
251                let name = self.validate_path(name);
252                self.elaborate_struct_pattern(
253                    name,
254                    fields,
255                    location,
256                    expected_type,
257                    definition,
258                    mutable,
259                    new_definitions,
260                    pattern_names,
261                    parameter_names_in_list,
262                )
263            }
264            // e.g. let (<pattern>) = ...;
265            Pattern::Parenthesized(pattern, _) => self.elaborate_pattern_mut(
266                *pattern,
267                expected_type,
268                definition,
269                mutable,
270                new_definitions,
271                warn_if_unused,
272                warn_if_not_mutated,
273                pattern_names,
274                parameter_names_in_list,
275            ),
276            Pattern::Interned(id, _) => {
277                let pattern = self.interner.get_pattern(id).clone();
278                self.elaborate_pattern_mut(
279                    pattern,
280                    expected_type,
281                    definition,
282                    mutable,
283                    new_definitions,
284                    warn_if_unused,
285                    warn_if_not_mutated,
286                    pattern_names,
287                    parameter_names_in_list,
288                )
289            }
290        }
291    }

293    #[allow(clippy::too_many_arguments)]
294    #[tracing::instrument(level = "trace", skip_all)]
295    fn elaborate_struct_pattern(
296        &mut self,
297        name: TypedPath,
298        fields: Vec<(Ident, Pattern)>,
299        location: Location,
300        expected_type: Type,
301        definition: DefinitionKind,
302        mutable: Option<Location>,
303        new_definitions: &mut Vec<HirIdent>,
304        pattern_names: &mut HashSet<String>,
305        parameter_names_in_list: &mut HashMap<String, Location>,
306    ) -> HirPattern {
307        let last_segment = name.last_segment();
308        let name_location = last_segment.ident.location();
309        let is_self_type = last_segment.ident.is_self_type_name();
310
311        let error_identifier = |this: &mut Self| {
312            // Must create a name here to return a HirPattern::Identifier. Allowing
313            // shadowing here lets us avoid further errors if we define ERROR_IDENT
314            // multiple times.
315            let name = ERROR_IDENT.into();
316            let identifier =
317                this.add_variable_decl(name, false, true, true, true, definition.clone());
318            HirPattern::Identifier(identifier)
319        };
320
321        let (struct_type, generics) = match self.lookup_type_or_error(name) {
322            Some(Type::DataType(struct_type, struct_generics))
323                if struct_type.borrow().is_struct() =>
324            {
325                (struct_type, struct_generics)
326            }
327            None => return error_identifier(self),
328            Some(typ) => {
329                let typ = typ.to_string();
330                self.push_err(ResolverError::NonStructUsedInConstructor { typ, location });
331                return error_identifier(self);
332            }
333        };
334
335        let turbofish_location = last_segment.turbofish_location();
336
337        let mut errors = Vec::new();
338        let generics = self.resolve_struct_turbofish_generics(
339            &struct_type.borrow(),
340            generics,
341            last_segment.generics,
342            turbofish_location,
343            &mut errors,
344        );
345        self.push_errors(errors);
346
347        let actual_type = Type::DataType(struct_type.clone(), generics);
348
349        self.unify(&actual_type, &expected_type, || TypeCheckError::TypeMismatchWithSource {
350            expected: expected_type.clone(),
351            actual: actual_type.clone(),
352            location,
353            source: Source::Assignment,
354        });
355
356        let fields = self.resolve_constructor_pattern_fields(
357            fields,
358            location,
359            actual_type.clone(),
360            definition,
361            mutable,
362            new_definitions,
363            pattern_names,
364            parameter_names_in_list,
365        );
366
367        let struct_id = struct_type.borrow().id;
368
369        self.interner.add_type_reference(struct_id, name_location, is_self_type);
370
371        for (field_index, field) in fields.iter().enumerate() {
372            let reference_location = field.0.location();
373            self.interner.add_struct_member_reference(struct_id, field_index, reference_location);
374        }
375
376        HirPattern::Struct(actual_type, fields, location)
377    }

379    /// Resolve all the fields of a struct constructor expression.
380    /// Ensures all fields are present, none are repeated, and all
381    /// are part of the struct.
382    #[allow(clippy::too_many_arguments)]
383    #[tracing::instrument(level = "trace", skip_all)]
384    fn resolve_constructor_pattern_fields(
385        &mut self,
386        fields: Vec<(Ident, Pattern)>,
387        location: Location,
388        typ: Type,
389        definition: DefinitionKind,
390        mutable: Option<Location>,
391        new_definitions: &mut Vec<HirIdent>,
392        pattern_names: &mut HashSet<String>,
393        parameter_names_in_list: &mut HashMap<String, Location>,
394    ) -> Vec<(Ident, HirPattern)> {
395        let mut ret = Vec::with_capacity(fields.len());
396        let mut seen_fields = HashSet::default();
397        let Type::DataType(struct_type, _) = &typ else {
398            unreachable!("Should be validated as struct before getting here")
399        };
400        let mut unseen_fields = struct_type
401            .borrow()
402            .field_names()
403            .expect("This type should already be validated to be a struct");
404
405        for (field, pattern) in fields {
406            let (field_type, visibility) = typ
407                .get_field_type_and_visibility(field.as_str())
408                .unwrap_or((Type::Error, ItemVisibility::Public));
409            let resolved = self.elaborate_pattern_mut(
410                pattern,
411                field_type,
412                definition.clone(),
413                mutable,
414                new_definitions,
415                true, // warn_if_unused
416                true, // warn_if_not_mutated
417                pattern_names,
418                parameter_names_in_list,
419            );
420
421            if unseen_fields.contains(&field) {
422                unseen_fields.remove(&field);
423                seen_fields.insert(field.clone());
424
425                self.check_struct_field_visibility(
426                    &struct_type.borrow(),
427                    field.as_str(),
428                    visibility,
429                    field.location(),
430                );
431            } else if seen_fields.contains(&field) {
432                // duplicate field
433                self.push_err(ResolverError::DuplicateField { field: field.clone() });
434            } else {
435                // field not required by struct
436                self.push_err(ResolverError::NoSuchField {
437                    field: field.clone(),
438                    struct_definition: struct_type.borrow().name.clone(),
439                });
440            }
441
442            ret.push((field, resolved));
443        }
444
445        if !unseen_fields.is_empty() {
446            self.push_err(ResolverError::MissingFields {
447                location,
448                missing_fields: unseen_fields.into_iter().map(|field| field.to_string()).collect(),
449                struct_definition: struct_type.borrow().name.clone(),
450            });
451        }
452
453        ret
454    }

456    /// Add a local or const numeric variable declaration to the scope,
457    /// unless the name is `"_"`.
458    ///
459    /// Returns the created identifier.
460    ///
461    /// Panics if the `definition` is [DefinitionKind::Global].
462    #[tracing::instrument(level = "trace", skip_all)]
463    pub(super) fn add_variable_decl(
464        &mut self,
465        name: Ident,
466        mutable: bool,
467        allow_shadowing: bool,
468        warn_if_unused: bool,
469        warn_if_not_mutated: bool,
470        definition: DefinitionKind,
471    ) -> HirIdent {
472        if let DefinitionKind::Global(_) = definition {
473            unreachable!("ICE: globals don't need to be added to the scope");
474        }
475
476        let location = name.location();
477        let name = name.into_string();
478        let comptime = self.in_comptime_context();
479        let id =
480            self.interner.push_definition(name.clone(), mutable, comptime, definition, location);
481        let ident = HirIdent::non_trait_method(id, location);
482
483        self.add_existing_variable_to_scope(
484            name,
485            ident.clone(),
486            warn_if_unused,
487            warn_if_not_mutated,
488            allow_shadowing,
489        );
490
491        ident
492    }

494    /// Add a [ResolverMeta] to the last scope for a given [HirIdent], which already has its definition interned,
495    /// unless its name is `"_"`.
496    #[tracing::instrument(level = "trace", skip_all)]
497    pub fn add_existing_variable_to_scope(
498        &mut self,
499        name: String,
500        ident: HirIdent,
501        warn_if_unused: bool,
502        warn_if_not_mutated: bool,
503        allow_shadowing: bool,
504    ) {
505        if name == "_" {
506            return;
507        }
508
509        let second_location = ident.location;
510        let resolver_meta = ResolverMeta {
511            used: false,
512            mutated: false,
513            ident,
514            warn_if_unused,
515            warn_if_not_mutated,
516        };
517
518        let old_value = self.scopes.get_mut_scope().add_key_value(name.clone(), resolver_meta);
519
520        if !allow_shadowing && let Some(old_value) = old_value {
521            self.push_err(ResolverError::DuplicateDefinition {
522                name,
523                first_location: old_value.ident.location,
524                second_location,
525            });
526        }
527    }

529    /// Lookup and use the specified local variable.
530    /// This will increment its use counter by one and return the variable if found.
531    /// If the variable is not found, an error is returned.
532    ///
533    /// This method is private and is expected to be called through [Self::get_ident_from_path_or_error].
534    #[tracing::instrument(level = "trace", skip_all)]
535    fn use_variable(&mut self, name: &Ident) -> Result<Variable, ResolverError> {
536        // Find the definition for this Ident
537        let scope_tree = self.scopes.current_scope_tree();
538        let variable = scope_tree.find(name.as_str());
539
540        let location = name.location();
541        if let Some((variable_found, scope)) = variable {
542            variable_found.used = true;
543            let id = variable_found.ident.id;
544            let ident = HirIdent::non_trait_method(id, location);
545            let variable = Variable { ident, scope };
546            Ok(variable)
547        } else if self.parent_runtime_variables.contains(name.as_str()) {
548            Err(ResolverError::RuntimeVarReferencedInComptime {
549                name: name.to_string(),
550                location: name.location(),
551            })
552        } else {
553            Err(ResolverError::VariableNotDeclared {
554                name: name.to_string(),
555                location: name.location(),
556            })
557        }
558    }

560    /// Resolve generics using the expected kinds of the function we are calling.
561    ///
562    /// Looks up the generics of the function in [FuncMeta][crate::hir_def::function::FuncMeta].
563    ///
564    /// If there is no turbofish, it returns `None`.
565    #[tracing::instrument(level = "trace", skip_all)]
566    pub(super) fn resolve_function_turbofish_generics(
567        &mut self,
568        func_id: &FuncId,
569        resolved_turbofish: Option<Vec<Located<Type>>>,
570        location: Location,
571    ) -> Option<Vec<Type>> {
572        resolved_turbofish.map(|mut resolved_turbofish| {
573            let direct_generic_kinds =
574                vecmap(&self.interner.function_meta(func_id).direct_generics, |generic| {
575                    generic.kind()
576                });
577            let expected = direct_generic_kinds.len();
578            let actual = resolved_turbofish.len();
579
580            if actual != expected {
581                self.push_err(TypeCheckError::IncorrectTurbofishGenericCount {
582                    expected_count: expected,
583                    actual_count: actual,
584                    location,
585                });
586
587                // Pad so the result has the expected length for future checks
588                resolved_turbofish.resize(expected, Located::from(location, Type::Error));
589            }
590
591            self.resolve_turbofish_generics(direct_generic_kinds, resolved_turbofish)
592        })
593    }

675        item_kind: &'static str,
676        item_name: &str,
677        item_generic_kinds: Vec<Kind>,
678        generics: Vec<Type>,
679        resolved_turbofish: Option<Vec<Located<Type>>>,
680        location: Location,
681        errors: &mut Vec<CompilationError>,
682    ) -> Vec<Type> {
683        assert_eq!(
684            generics.len(),
685            item_generic_kinds.len(),
686            "ICE: generics count should match the expected kinds"
687        );
688
689        let Some(turbofish_generics) = resolved_turbofish else {
690            return generics;
691        };
692
693        if turbofish_generics.len() != generics.len() {
694            errors.push(
695                TypeCheckError::GenericCountMismatch {
696                    item: format!("{item_kind} {item_name}"),
697                    expected: generics.len(),
698                    found: turbofish_generics.len(),
699                    location,
700                }
701                .into(),
702            );
703            return generics;
704        }
705
706        self.resolve_turbofish_generics(item_generic_kinds, turbofish_generics)
707    }

709    /// Given the generic [Kind]s of a type, and the list of generic types in a non-empty turbofish,
710    /// which have already been verified to match the expected number of generics, run type checking
711    /// to ensure each turbofish generic matches the expected kind, and return the unified types.
712    #[tracing::instrument(level = "trace", skip_all)]
713    fn resolve_turbofish_generics(
714        &mut self,
715        kinds: Vec<Kind>,
716        turbofish_generics: Vec<Located<Type>>,
717    ) -> Vec<Type> {
718        // Use zip (not zip_eq) since callers like resolve_function_turbofish_generics
719        // may push an error for mismatched counts but still call this function.
720        let kinds_with_types = kinds.into_iter().zip(turbofish_generics);
721
722        vecmap(kinds_with_types, |(kind, located_type)| {
723            let location = located_type.location();
724            let typ = located_type.contents;
725            self.check_type_kind(typ, &kind, location)
726        })
727    }

729    /// Create a validated [TypedPath] from a [Path] by resolving all generics in every [PathSegment] in it.
730    ///
731    /// Pushes an error if the first segment is `Self` and it has turbofish generics.
732    #[tracing::instrument(level = "trace", skip_all)]
733    pub(crate) fn validate_path(&mut self, path: Path) -> TypedPath {
734        let mut segments = vecmap(path.segments, |segment| self.validate_path_segment(segment));
735
736        if let Some(first_segment) = segments.first_mut()
737            && first_segment.generics.is_some()
738            && first_segment.ident.is_self_type_name()
739        {
740            self.push_err(PathResolutionError::TurbofishNotAllowedOnItem {
741                item: "self type".to_string(),
742                location: first_segment.turbofish_location(),
743            });
744            first_segment.generics = None;
745        }
746
747        TypedPath {
748            segments,
749            kind: path.kind,
750            location: path.location,
751            kind_location: path.kind_location,
752        }
753    }
754
755    /// Create a validated [TypedPathSegment] from a [PathSegment] by resolving all turbofish generics
756    /// in it with [Kind::Any], allowing wildcards, and marking them as _used_.
757    #[tracing::instrument(level = "trace", skip_all)]
758    fn validate_path_segment(&mut self, segment: PathSegment) -> TypedPathSegment {
759        let generics = segment.generics.map(|generics| {
760            vecmap(generics, |generic| {
761                let location = generic.location;
762                let wildcard_allowed = WildcardAllowed::Yes;
763                let typ = self.use_type_with_kind(generic, &Kind::Any, wildcard_allowed);
764                Located::from(location, typ)
765            })
766        });
767        TypedPathSegment { ident: segment.ident, generics, location: segment.location }
768    }

770    /// Get the [DataType] of a [TypeId] and call [Elaborator::resolve_struct_turbofish_generics].
771    #[tracing::instrument(level = "trace", skip_all)]
772    pub(super) fn resolve_struct_id_turbofish_generics(
773        &mut self,
774        struct_id: TypeId,
775        mut turbofish: Option<Turbofish>,
776        errors: &mut Vec<CompilationError>,
777    ) -> Vec<Type> {
778        let struct_type = self.interner.get_type(struct_id);
779        let struct_type = struct_type.borrow();
780        let struct_generics = struct_type.instantiate(self.interner);
781        if let Some(turbofish) = turbofish.take() {
782            self.resolve_struct_turbofish_generics(
783                &struct_type,
784                struct_generics,
785                Some(turbofish.generics),
786                turbofish.location,
787                errors,
788            )
789        } else {
790            struct_generics
791        }
792    }

794    /// Get the [TypeAlias] of a [TypeAliasId] and call [Elaborator::resolve_alias_turbofish_generics].
795    #[tracing::instrument(level = "trace", skip_all)]
796    pub(super) fn resolve_type_alias_id_turbofish_generics(
797        &mut self,
798        type_alias_id: TypeAliasId,
799        generics: Option<Turbofish>,
800        errors: &mut Vec<CompilationError>,
801    ) -> Vec<Type> {
802        let type_alias = self.interner.get_type_alias(type_alias_id);
803        let type_alias = type_alias.borrow();
804        let alias_generics = vecmap(&type_alias.generics, |generic| {
805            self.interner.next_type_variable_with_kind(generic.kind())
806        });
807
808        if let Some(generics) = generics {
809            self.resolve_alias_turbofish_generics(
810                &type_alias,
811                alias_generics,
812                Some(generics.generics),
813                generics.location,
814                errors,
815            )
816        } else {
817            alias_generics
818        }
819    }

821    /// Resolve a [TypedPath] into a local or global [HirIdent].
822    ///
823    /// If it cannot be found, then it pushes the error and returns [None].
824    #[tracing::instrument(level = "trace", skip_all)]
825    pub(crate) fn get_ident_from_path(&mut self, path: TypedPath) -> Option<IdentFromPath> {
826        match self.get_ident_from_path_or_error(path) {
827            Ok(value) => Some(value),
828            Err(error) => {
829                self.push_err(error);
830                None
831            }
832        }
833    }

835    /// Resolve a [TypedPath] into a local or global [HirIdent], or return `Err` if it could not be found.
836    #[tracing::instrument(level = "trace", skip_all)]
837    pub(crate) fn get_ident_from_path_or_error(
838        &mut self,
839        path: TypedPath,
840    ) -> Result<IdentFromPath, ResolverError> {
841        // If the path is a single segment, try to resolve it as a local variable first
842        let use_variable_error = match path.as_single_segment() {
843            Some(segment) => match self.use_variable(&segment.ident) {
844                Ok(variable) => {
845                    // Succeed even if the variable has turbofish on it, but report an error for that
846                    if segment.generics.is_some() {
847                        let item = "local variables".to_string();
848                        let location = segment.turbofish_location();
849                        let error =
850                            PathResolutionError::TurbofishNotAllowedOnItem { item, location };
851                        self.push_err(error);
852                    }
853                    return Ok(IdentFromPath::Variable(variable));
854                }
855                Err(error) => Some(error),
856            },
857            None => None,
858        };
859
860        match self.lookup_item_as_value(path) {
861            Ok(ItemAsValue::Definition { id, item }) => Ok(IdentFromPath::Definition { id, item }),
862            Ok(ItemAsValue::TypeAlias(type_alias_id)) => {
863                Ok(IdentFromPath::TypeAlias(type_alias_id))
864            }
865            Err(ResolverError::PathResolutionError(PathResolutionError::Unresolved(ident))) => {
866                // If we can't resolve a path, but we have an error from trying to resolve a variable
867                // (in which case the path was a single segment), prefer saying "variable not found"
868                // instead of "Cannot resolve '...' in path".
869                match use_variable_error {
870                    Some(error) => Err(error),
871                    None => Err(ResolverError::PathResolutionError(
872                        PathResolutionError::Unresolved(ident),
873                    )),
874                }
875            }
876            Err(item_as_value_error) => {
877                match use_variable_error {
878                    // If the path was "_" then we want to preserve that error as it's clearer
879                    // than the error of an item not being found (it will mention that "_" is
880                    // not valid as an expression).
881                    Some(ResolverError::VariableNotDeclared { name, .. }) if name != "_" => {
882                        Err(item_as_value_error)
883                    }
884                    Some(use_variable_error) => Err(use_variable_error),
885                    None => Err(item_as_value_error),
886                }
887            }
888        }
889    }
