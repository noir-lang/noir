46    /// Elaborate a pattern, which can appear in a `let <pattern> = <expr>`, or a `match` statement.
47    ///
48    /// The `definition_kind` specifies the kind of variables the pattern will create, e.g. a local or global variable.
49    ///
50    /// The `expected_type` is always known, because we can first infer the type of the `<expr>` and try to match it to
51    /// the pattern.
52    ///
53    /// `parameter_names_in_list` keeps track of parameter names, and their location, across multiple
54    /// patterns in a list. If a name is found multiple times, an error is captured.
55    pub(super) fn elaborate_pattern(
56        &mut self,
57        pattern: Pattern,
58        expected_type: Type,
59        definition_kind: DefinitionKind,
60        warn_if_unused: bool,
61        parameter_names_in_list: &mut HashMap<String, Location>,
62    ) -> HirPattern {
63        self.elaborate_pattern_mut(
64            pattern,
65            expected_type,
66            definition_kind,
67            None,
68            &mut Vec::new(),
69            warn_if_unused,
70            &mut HashSet::default(),
71            parameter_names_in_list,
72        )
73    }

101    /// Elaborate the (potentially mutable) pattern and add the variables
102    /// it created to the scope if necessary.
103    ///
104    /// - `pattern_names` keeps track of parameter names within this single pattern (or an outer
105    ///   one, when called recursively). If a name is found multiple times, an error is captured.
106    /// - `parameter_names_in_list` keeps track of parameter names, and their location, across multiple
107    ///   patterns in a list. If a name is found multiple times, an error is captured.
108    #[allow(clippy::too_many_arguments)]
109    fn elaborate_pattern_mut(
110        &mut self,
111        pattern: Pattern,
112        expected_type: Type,
113        definition: DefinitionKind,
114        // Location of the `mut` keyword.
115        mutable: Option<Location>,
116        new_definitions: &mut Vec<HirIdent>,
117        warn_if_unused: bool,
118        pattern_names: &mut HashSet<String>,
119        parameter_names_in_list: &mut HashMap<String, Location>,
120    ) -> HirPattern {
121        match pattern {
122            // e.g. let <ident> = ...;
123            Pattern::Identifier(name) => {
124                // If this definition is mutable, do not store the RHS because it will
125                // not always refer to the correct value of the variable.
126                let definition = match (mutable, definition) {
127                    (Some(_), DefinitionKind::Local(_)) => DefinitionKind::Local(None),
128                    (_, other) => other,
129                };
130                let ident = if let DefinitionKind::Global(global_id) = definition {
131                    // Globals don't need to be added to scope, they're already in the def_maps
132                    let id = self.interner.get_global(global_id).definition_id;
133                    let location = name.location();
134                    HirIdent::non_trait_method(id, location)
135                } else {
136                    if name.as_str() != "_" {
137                        if !pattern_names.insert(name.to_string()) {
138                            self.push_err(ResolverError::PatternBoundMoreThanOnce {
139                                ident: name.clone(),
140                            });
141                        } else if let Some(first_location) =
142                            parameter_names_in_list.insert(name.to_string(), name.location())
143                        {
144                            self.push_err(ResolverError::DuplicateDefinition {
145                                name: name.to_string(),
146                                first_location,
147                                second_location: name.location(),
148                            });
149                        }
150                    }
151
152                    self.add_variable_decl(
153                        name,
154                        mutable.is_some(),
155                        true, // allow_shadowing
156                        warn_if_unused,
157                        definition,
158                    )
159                };
160                self.interner.push_definition_type(ident.id, expected_type);
161                new_definitions.push(ident.clone());
162                HirPattern::Identifier(ident)
163            }
164            // e.g. let mut <pattern> = ...;
165            Pattern::Mutable(pattern, location, _) => {
166                if let Some(first_mut) = mutable {
167                    self.push_err(ResolverError::UnnecessaryMut {
168                        first_mut,
169                        second_mut: location,
170                    });
171                }
172
173                let pattern = self.elaborate_pattern_mut(
174                    *pattern,
175                    expected_type,
176                    definition,
177                    Some(location),
178                    new_definitions,
179                    warn_if_unused,
180                    pattern_names,
181                    parameter_names_in_list,
182                );
183                HirPattern::Mutable(Box::new(pattern), location)
184            }
185            // e.g. let (<pattern 0>, <pattern 1>, ...) = ...;
186            Pattern::Tuple(fields, location) => {
187                // Returns Some for valid tuple types (where arity checking makes sense),
188                // None when we've already issued an error or have an invalid type.
189                let field_types = match expected_type.follow_bindings() {
190                    Type::Tuple(fields) => Some(fields),
191                    Type::Error => None,
192                    expected_type => {
193                        let tuple =
194                            Type::Tuple(vecmap(&fields, |_| self.interner.next_type_variable()));
195
196                        self.push_err(TypeCheckError::TypeMismatchWithSource {
197                            expected: expected_type,
198                            actual: tuple,
199                            location,
200                            source: Source::Assignment,
201                        });
202                        None
203                    }
204                };
205
206                // Only check tuple arity if the expected type was actually a tuple.
207                // If it wasn't, we've already issued a type mismatch error above.
208                if let Some(field_types) = &field_types
209                    && fields.len() != field_types.len()
210                {
211                    self.push_err(TypeCheckError::TupleMismatch {
212                        tuple_types: field_types.clone(),
213                        actual_count: fields.len(),
214                        location,
215                    });
216                }
217
218                let fields = vecmap(fields.into_iter().enumerate(), |(i, field)| {
219                    let field_type = field_types
220                        .as_ref()
221                        .and_then(|types| types.get(i).cloned())
222                        .unwrap_or(Type::Error);
223                    self.elaborate_pattern_mut(
224                        field,
225                        field_type,
226                        definition.clone(),
227                        mutable,
228                        new_definitions,
229                        warn_if_unused,
230                        pattern_names,
231                        parameter_names_in_list,
232                    )
233                });
234                HirPattern::Tuple(fields, location)
235            }
236            // e.g. let <name> { <field 0>: <pattern 0>, <field 1>: <pattern 0>, ... } = ...'
237            Pattern::Struct(name, fields, location) => {
238                let name = self.validate_path(name);
239                self.elaborate_struct_pattern(
240                    name,
241                    fields,
242                    location,
243                    expected_type,
244                    definition,
245                    mutable,
246                    new_definitions,
247                    pattern_names,
248                    parameter_names_in_list,
249                )
250            }
251            // e.g. let (<pattern>) = ...;
252            Pattern::Parenthesized(pattern, _) => self.elaborate_pattern_mut(
253                *pattern,
254                expected_type,
255                definition,
256                mutable,
257                new_definitions,
258                warn_if_unused,
259                pattern_names,
260                parameter_names_in_list,
261            ),
262            Pattern::Interned(id, _) => {
263                let pattern = self.interner.get_pattern(id).clone();
264                self.elaborate_pattern_mut(
265                    pattern,
266                    expected_type,
267                    definition,
268                    mutable,
269                    new_definitions,
270                    warn_if_unused,
271                    pattern_names,
272                    parameter_names_in_list,
273                )
274            }
275        }
276    }

278    #[allow(clippy::too_many_arguments)]
279    fn elaborate_struct_pattern(
280        &mut self,
281        name: TypedPath,
282        fields: Vec<(Ident, Pattern)>,
283        location: Location,
284        expected_type: Type,
285        definition: DefinitionKind,
286        mutable: Option<Location>,
287        new_definitions: &mut Vec<HirIdent>,
288        pattern_names: &mut HashSet<String>,
289        parameter_names_in_list: &mut HashMap<String, Location>,
290    ) -> HirPattern {
291        let last_segment = name.last_segment();
292        let name_location = last_segment.ident.location();
293        let is_self_type = last_segment.ident.is_self_type_name();
294
295        let error_identifier = |this: &mut Self| {
296            // Must create a name here to return a HirPattern::Identifier. Allowing
297            // shadowing here lets us avoid further errors if we define ERROR_IDENT
298            // multiple times.
299            let name = ERROR_IDENT.into();
300            let identifier = this.add_variable_decl(name, false, true, true, definition.clone());
301            HirPattern::Identifier(identifier)
302        };
303
304        let (struct_type, generics) = match self.lookup_type_or_error(name) {
305            Some(Type::DataType(struct_type, struct_generics))
306                if struct_type.borrow().is_struct() =>
307            {
308                (struct_type, struct_generics)
309            }
310            None => return error_identifier(self),
311            Some(typ) => {
312                let typ = typ.to_string();
313                self.push_err(ResolverError::NonStructUsedInConstructor { typ, location });
314                return error_identifier(self);
315            }
316        };
317
318        let turbofish_location = last_segment.turbofish_location();
319
320        let generics = self.resolve_struct_turbofish_generics(
321            &struct_type.borrow(),
322            generics,
323            last_segment.generics,
324            turbofish_location,
325        );
326
327        let actual_type = Type::DataType(struct_type.clone(), generics);
328
329        self.unify(&actual_type, &expected_type, || TypeCheckError::TypeMismatchWithSource {
330            expected: expected_type.clone(),
331            actual: actual_type.clone(),
332            location,
333            source: Source::Assignment,
334        });
335
336        let fields = self.resolve_constructor_pattern_fields(
337            fields,
338            location,
339            actual_type.clone(),
340            definition,
341            mutable,
342            new_definitions,
343            pattern_names,
344            parameter_names_in_list,
345        );
346
347        let struct_id = struct_type.borrow().id;
348
349        self.interner.add_type_reference(struct_id, name_location, is_self_type);
350
351        for (field_index, field) in fields.iter().enumerate() {
352            let reference_location = field.0.location();
353            self.interner.add_struct_member_reference(struct_id, field_index, reference_location);
354        }
355
356        HirPattern::Struct(actual_type, fields, location)
357    }

359    /// Resolve all the fields of a struct constructor expression.
360    /// Ensures all fields are present, none are repeated, and all
361    /// are part of the struct.
362    #[allow(clippy::too_many_arguments)]
363    fn resolve_constructor_pattern_fields(
364        &mut self,
365        fields: Vec<(Ident, Pattern)>,
366        location: Location,
367        typ: Type,
368        definition: DefinitionKind,
369        mutable: Option<Location>,
370        new_definitions: &mut Vec<HirIdent>,
371        pattern_names: &mut HashSet<String>,
372        parameter_names_in_list: &mut HashMap<String, Location>,
373    ) -> Vec<(Ident, HirPattern)> {
374        let mut ret = Vec::with_capacity(fields.len());
375        let mut seen_fields = HashSet::default();
376        let Type::DataType(struct_type, _) = &typ else {
377            unreachable!("Should be validated as struct before getting here")
378        };
379        let mut unseen_fields = struct_type
380            .borrow()
381            .field_names()
382            .expect("This type should already be validated to be a struct");
383
384        for (field, pattern) in fields {
385            let (field_type, visibility) = typ
386                .get_field_type_and_visibility(field.as_str())
387                .unwrap_or((Type::Error, ItemVisibility::Public));
388            let resolved = self.elaborate_pattern_mut(
389                pattern,
390                field_type,
391                definition.clone(),
392                mutable,
393                new_definitions,
394                true, // warn_if_unused
395                pattern_names,
396                parameter_names_in_list,
397            );
398
399            if unseen_fields.contains(&field) {
400                unseen_fields.remove(&field);
401                seen_fields.insert(field.clone());
402
403                self.check_struct_field_visibility(
404                    &struct_type.borrow(),
405                    field.as_str(),
406                    visibility,
407                    field.location(),
408                );
409            } else if seen_fields.contains(&field) {
410                // duplicate field
411                self.push_err(ResolverError::DuplicateField { field: field.clone() });
412            } else {
413                // field not required by struct
414                self.push_err(ResolverError::NoSuchField {
415                    field: field.clone(),
416                    struct_definition: struct_type.borrow().name.clone(),
417                });
418            }
419
420            ret.push((field, resolved));
421        }
422
423        if !unseen_fields.is_empty() {
424            self.push_err(ResolverError::MissingFields {
425                location,
426                missing_fields: unseen_fields.into_iter().map(|field| field.to_string()).collect(),
427                struct_definition: struct_type.borrow().name.clone(),
428            });
429        }
430
431        ret
432    }

434    /// Add a local or const numeric variable declaration to the scope,
435    /// unless the name is `"_"`.
436    ///
437    /// Returns the created identifier.
438    ///
439    /// Panics if the `definition` is [DefinitionKind::Global].
440    pub(super) fn add_variable_decl(
441        &mut self,
442        name: Ident,
443        mutable: bool,
444        allow_shadowing: bool,
445        warn_if_unused: bool,
446        definition: DefinitionKind,
447    ) -> HirIdent {
448        if let DefinitionKind::Global(_) = definition {
449            unreachable!("ICE: globals don't need to be added to the scope");
450        }
451
452        let location = name.location();
453        let name = name.into_string();
454        let comptime = self.in_comptime_context();
455        let id =
456            self.interner.push_definition(name.clone(), mutable, comptime, definition, location);
457        let ident = HirIdent::non_trait_method(id, location);
458
459        self.add_existing_variable_to_scope(name, ident.clone(), warn_if_unused, allow_shadowing);
460
461        ident
462    }

464    /// Add a [ResolverMeta] to the last scope for a given [HirIdent], which already has its definition interned,
465    /// unless its name is `"_"`.
466    pub fn add_existing_variable_to_scope(
467        &mut self,
468        name: String,
469        ident: HirIdent,
470        warn_if_unused: bool,
471        allow_shadowing: bool,
472    ) {
473        if name == "_" {
474            return;
475        }
476
477        let second_location = ident.location;
478        let resolver_meta = ResolverMeta { num_times_used: 0, ident, warn_if_unused };
479
480        let old_value = self.scopes.get_mut_scope().add_key_value(name.clone(), resolver_meta);
481
482        if !allow_shadowing && let Some(old_value) = old_value {
483            self.push_err(ResolverError::DuplicateDefinition {
484                name,
485                first_location: old_value.ident.location,
486                second_location,
487            });
488        }
489    }

491    /// Lookup and use the specified local variable.
492    /// This will increment its use counter by one and return the variable if found.
493    /// If the variable is not found, an error is returned.
494    ///
495    /// This method is private and is expected to be called through [Self::get_ident_from_path_or_error].
496    fn use_variable(&mut self, name: &Ident) -> Result<Variable, ResolverError> {
497        // Find the definition for this Ident
498        let scope_tree = self.scopes.current_scope_tree();
499        let variable = scope_tree.find(name.as_str());
500
501        let location = name.location();
502        if let Some((variable_found, scope)) = variable {
503            variable_found.num_times_used += 1;
504            let id = variable_found.ident.id;
505            let ident = HirIdent::non_trait_method(id, location);
506            let variable = Variable { ident, scope };
507            Ok(variable)
508        } else {
509            Err(ResolverError::VariableNotDeclared {
510                name: name.to_string(),
511                location: name.location(),
512            })
513        }
514    }

516    /// Resolve generics using the expected kinds of the function we are calling.
517    ///
518    /// Looks up the generics of the function in [FuncMeta][crate::hir_def::function::FuncMeta].
519    ///
520    /// If there is no turbofish, it returns `None`.
521    pub(super) fn resolve_function_turbofish_generics(
522        &mut self,
523        func_id: &FuncId,
524        resolved_turbofish: Option<Vec<Located<Type>>>,
525        location: Location,
526    ) -> Option<Vec<Type>> {
527        resolved_turbofish.map(|resolved_turbofish| {
528            let direct_generic_kinds =
529                vecmap(&self.interner.function_meta(func_id).direct_generics, |generic| {
530                    generic.kind()
531                });
532
533            if resolved_turbofish.len() != direct_generic_kinds.len() {
534                let type_check_err = TypeCheckError::IncorrectTurbofishGenericCount {
535                    expected_count: direct_generic_kinds.len(),
536                    actual_count: resolved_turbofish.len(),
537                    location,
538                };
539                self.push_err(type_check_err);
540            }
541
542            self.resolve_turbofish_generics(direct_generic_kinds, resolved_turbofish)
543        })
544    }

609    /// Given the generic [Kind]s of a type and its own declared generic [Type]s,
610    /// check if we have a non-empty turbofish with the expected number of generics,
611    /// and if so try unify them with the expected kinds, otherwise return the default
612    /// generics of the type.
613    pub(super) fn resolve_item_turbofish_generics(
614        &mut self,
615        item_kind: &'static str,
616        item_name: &str,
617        item_generic_kinds: Vec<Kind>,
618        generics: Vec<Type>,
619        resolved_turbofish: Option<Vec<Located<Type>>>,
620        location: Location,
621    ) -> Vec<Type> {
622        assert_eq!(
623            generics.len(),
624            item_generic_kinds.len(),
625            "ICE: generics count should match the expected kinds"
626        );
627
628        let Some(turbofish_generics) = resolved_turbofish else {
629            return generics;
630        };
631
632        if turbofish_generics.len() != generics.len() {
633            self.push_err(TypeCheckError::GenericCountMismatch {
634                item: format!("{item_kind} {item_name}"),
635                expected: generics.len(),
636                found: turbofish_generics.len(),
637                location,
638            });
639            return generics;
640        }
641
642        self.resolve_turbofish_generics(item_generic_kinds, turbofish_generics)
643    }

645    /// Given the generic [Kind]s of a type, and the list of generic types in a non-empty turbofish,
646    /// which have already been verified to match the expected number of generics, run type checking
647    /// to ensure each turbofish generic matches the expected kind, and return the unified types.
648    fn resolve_turbofish_generics(
649        &mut self,
650        kinds: Vec<Kind>,
651        turbofish_generics: Vec<Located<Type>>,
652    ) -> Vec<Type> {
653        let kinds_with_types = kinds.into_iter().zip(turbofish_generics);
654
655        vecmap(kinds_with_types, |(kind, located_type)| {
656            let location = located_type.location();
657            let typ = located_type.contents;
658            let typ = typ.substitute_kind_any_with_kind(&kind);
659            self.check_type_kind(typ, &kind, location)
660        })
661    }

663    /// Create a validated [TypedPath] from a [Path] by resolving all generics in every [PathSegment] in it.
664    ///
665    /// Pushes an error if the first segment is `Self` and it has turbofish generics.
666    pub(crate) fn validate_path(&mut self, path: Path) -> TypedPath {
667        let mut segments = vecmap(path.segments, |segment| self.validate_path_segment(segment));
668
669        if let Some(first_segment) = segments.first_mut()
670            && first_segment.generics.is_some()
671            && first_segment.ident.is_self_type_name()
672        {
673            self.push_err(PathResolutionError::TurbofishNotAllowedOnItem {
674                item: "self type".to_string(),
675                location: first_segment.turbofish_location(),
676            });
677            first_segment.generics = None;
678        }
679
680        TypedPath {
681            segments,
682            kind: path.kind,
683            location: path.location,
684            kind_location: path.kind_location,
685        }
686    }

688    /// Create a validated [TypedPathSegment] from a [PathSegment] by resolving all turbofish generics
689    /// in it with [Kind::Any], allowing wildcards, and marking them as _used_.
690    fn validate_path_segment(&mut self, segment: PathSegment) -> TypedPathSegment {
691        let generics = segment.generics.map(|generics| {
692            vecmap(generics, |generic| {
693                let location = generic.location;
694                let wildcard_allowed = WildcardAllowed::Yes;
695                let typ = self.use_type_with_kind(generic, &Kind::Any, wildcard_allowed);
696                Located::from(location, typ)
697            })
698        });
699        TypedPathSegment { ident: segment.ident, generics, location: segment.location }
700    }

702    /// Get the [DataType] of a [TypeId] and call [Elaborator::resolve_struct_turbofish_generics].
703    pub(super) fn resolve_struct_id_turbofish_generics(
704        &mut self,
705        struct_id: TypeId,
706        mut turbofish: Option<Turbofish>,
707    ) -> Vec<Type> {
708        let struct_type = self.interner.get_type(struct_id);
709        let struct_type = struct_type.borrow();
710        let struct_generics = struct_type.instantiate(self.interner);
711        if let Some(turbofish) = turbofish.take() {
712            self.resolve_struct_turbofish_generics(
713                &struct_type,
714                struct_generics,
715                Some(turbofish.generics),
716                turbofish.location,
717            )
718        } else {
719            struct_generics
720        }
721    }

723    /// Get the [TypeAlias] of a [TypeAliasId] and call [Elaborator::resolve_alias_turbofish_generics].
724    pub(super) fn resolve_type_alias_id_turbofish_generics(
725        &mut self,
726        type_alias_id: TypeAliasId,
727        generics: Option<Turbofish>,
728    ) -> Vec<Type> {
729        let type_alias = self.interner.get_type_alias(type_alias_id);
730        let type_alias = type_alias.borrow();
731        let alias_generics = vecmap(&type_alias.generics, |generic| {
732            self.interner.next_type_variable_with_kind(generic.kind())
733        });
734
735        if let Some(generics) = generics {
736            self.resolve_alias_turbofish_generics(
737                &type_alias,
738                alias_generics,
739                Some(generics.generics),
740                generics.location,
741            )
742        } else {
743            alias_generics
744        }
745    }

747    /// Resolve a [TypedPath] into a local or global [HirIdent].
748    ///
749    /// If it cannot be found, then it pushes the error and returns [None].
750    pub(crate) fn get_ident_from_path(&mut self, path: TypedPath) -> Option<IdentFromPath> {
751        match self.get_ident_from_path_or_error(path) {
752            Ok(value) => Some(value),
753            Err(error) => {
754                self.push_err(error);
755                None
756            }
757        }
758    }
759
760    /// Resolve a [TypedPath] into a local or global [HirIdent], or return `Err` if it could not be found.
761    pub(crate) fn get_ident_from_path_or_error(
762        &mut self,
763        path: TypedPath,
764    ) -> Result<IdentFromPath, ResolverError> {
765        // If the path is a single segment, try to resolve it as a local variable first
766        let use_variable_error = match path.as_single_segment() {
767            Some(segment) => match self.use_variable(&segment.ident) {
768                Ok(variable) => {
769                    // Succeed even if the variable has turbofish on it, but report an error for that
770                    if segment.generics.is_some() {
771                        let item = "local variables".to_string();
772                        let location = segment.turbofish_location();
773                        let error =
774                            PathResolutionError::TurbofishNotAllowedOnItem { item, location };
775                        self.push_err(error);
776                    }
777                    return Ok(IdentFromPath::Variable(variable));
778                }
779                Err(error) => Some(error),
780            },
781            None => None,
782        };
783
784        match self.lookup_item_as_value(path) {
785            Ok(ItemAsValue::Definition { id, item }) => Ok(IdentFromPath::Definition { id, item }),
786            Ok(ItemAsValue::TypeAlias(type_alias_id)) => {
787                Ok(IdentFromPath::TypeAlias(type_alias_id))
788            }
789            Err(ResolverError::PathResolutionError(PathResolutionError::Unresolved(ident))) => {
790                // If we can't resolve a path, but we have an error from trying to resolve a variable
791                // (in which case the path was a single segment), prefer saying "variable not found"
792                // instead of "Cannot resolve '...' in path".
793                match use_variable_error {
794                    Some(error) => Err(error),
795                    None => Err(ResolverError::PathResolutionError(
796                        PathResolutionError::Unresolved(ident),
797                    )),
798                }
799            }
800            Err(item_as_value_error) => {
801                match use_variable_error {
802                    // If the path was "_" then we want to preserve that error as it's clearer
803                    // than the error of an item not being found (it will mention that "_" is
804                    // not valid as an expression).
805                    Some(ResolverError::VariableNotDeclared { name, .. }) if name != "_" => {
806                        Err(item_as_value_error)
807                    }
808                    Some(use_variable_error) => Err(use_variable_error),
809                    None => Err(item_as_value_error),
810                }
811            }
812        }
813    }
