85    pub fn to_type(self) -> Type {
86        match self {
87            Self::Bool => Type::Bool,
88            Self::CtString => Type::Quoted(QuotedType::CtString),
89            Self::Expr => Type::Quoted(QuotedType::Expr),
90            Self::Fmtstr => Type::FmtString(Box::new(Type::Error), Box::new(Type::Error)),
91            Self::Field => Type::FieldElement,
92            Self::FunctionDefinition => Type::Quoted(QuotedType::FunctionDefinition),
93            Self::I8 => Type::Integer(Signedness::Signed, IntegerBitSize::Eight),
94            Self::I16 => Type::Integer(Signedness::Signed, IntegerBitSize::Sixteen),
95            Self::I32 => Type::Integer(Signedness::Signed, IntegerBitSize::ThirtyTwo),
96            Self::I64 => Type::Integer(Signedness::Signed, IntegerBitSize::SixtyFour),
97            Self::U1 => Type::Integer(Signedness::Unsigned, IntegerBitSize::One),
98            Self::U8 => Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight),
99            Self::U16 => Type::Integer(Signedness::Unsigned, IntegerBitSize::Sixteen),
100            Self::U32 => Type::Integer(Signedness::Unsigned, IntegerBitSize::ThirtyTwo),
101            Self::U64 => Type::Integer(Signedness::Unsigned, IntegerBitSize::SixtyFour),
102            Self::U128 => Type::Integer(Signedness::Unsigned, IntegerBitSize::HundredTwentyEight),
103            Self::Module => Type::Quoted(QuotedType::Module),
104            Self::Quoted => Type::Quoted(QuotedType::Quoted),
105            Self::Str => Type::String(Box::new(Type::Error)),
106            Self::TraitConstraint => Type::Quoted(QuotedType::TraitConstraint),
107            Self::TraitDefinition => Type::Quoted(QuotedType::TraitDefinition),
108            Self::TraitImpl => Type::Quoted(QuotedType::TraitImpl),
109            Self::StructDefinition | Self::TypeDefinition => {
110                Type::Quoted(QuotedType::TypeDefinition)
111            }
112            Self::TypedExpr => Type::Quoted(QuotedType::TypedExpr),
113            Self::Type => Type::Quoted(QuotedType::Type),
114            Self::UnresolvedType => Type::Quoted(QuotedType::UnresolvedType),
115        }
116    }

185impl Elaborator<'_> {
186    pub(crate) fn instantiate_primitive_type(
187        &mut self,
188        primitive_type: PrimitiveType,
189        args: GenericTypeArgs,
190        location: Location,
191        wildcard_allowed: WildcardAllowed,
192    ) -> Type {
193        match primitive_type {
194            PrimitiveType::Bool
195            | PrimitiveType::CtString
196            | PrimitiveType::Expr
197            | PrimitiveType::Field
198            | PrimitiveType::FunctionDefinition
199            | PrimitiveType::I8
200            | PrimitiveType::I16
201            | PrimitiveType::I32
202            | PrimitiveType::I64
203            | PrimitiveType::U1
204            | PrimitiveType::U8
205            | PrimitiveType::U16
206            | PrimitiveType::U32
207            | PrimitiveType::U64
208            | PrimitiveType::U128
209            | PrimitiveType::Module
210            | PrimitiveType::Quoted
211            | PrimitiveType::StructDefinition
212            | PrimitiveType::TraitConstraint
213            | PrimitiveType::TraitDefinition
214            | PrimitiveType::TraitImpl
215            | PrimitiveType::TypeDefinition
216            | PrimitiveType::TypedExpr
217            | PrimitiveType::Type
218            | PrimitiveType::UnresolvedType => {
219                if !args.is_empty() {
220                    let found = args.ordered_args.len() + args.named_args.len();
221                    self.push_err(CompilationError::TypeError(
222                        TypeCheckError::GenericCountMismatch {
223                            item: primitive_type.name().to_string(),
224                            expected: 0,
225                            found,
226                            location,
227                        },
228                    ));
229                }
230            }
231            PrimitiveType::Str => {
232                let item = StrPrimitiveType;
233                let (mut args, _) = self.resolve_type_args_inner(
234                    args,
235                    item,
236                    location,
237                    PathResolutionMode::MarkAsReferenced,
238                    wildcard_allowed,
239                );
240                assert_eq!(args.len(), 1, "str generics should be: [length]");
241                let length = args.pop().unwrap();
242                return Type::String(Box::new(length));
243            }
244            PrimitiveType::Fmtstr => {
245                let item = FmtstrPrimitiveType;
246                let (mut args, _) = self.resolve_type_args_inner(
247                    args,
248                    item,
249                    location,
250                    PathResolutionMode::MarkAsReferenced,
251                    wildcard_allowed,
252                );
253                assert_eq!(args.len(), 2, "fmtstr generics should be: [length, element]");
254                let element = args.pop().unwrap();
255                let length = args.pop().unwrap();
256                return Type::FmtString(Box::new(length), Box::new(element));
257            }
258        }
259
260        primitive_type.to_type()
261    }

263    /// Instantiates a primitive type with turbofish generics.
264    ///
265    /// # Returns
266    /// A tuple of:
267    /// - The instantiated [Type]
268    /// - A boolean indicating whether this primitive type has generics
269    pub(crate) fn instantiate_primitive_type_with_turbofish(
270        &mut self,
271        primitive_type: PrimitiveType,
272        turbofish: Option<Turbofish>,
273    ) -> (Type, bool) {
274        match primitive_type {
275            PrimitiveType::Bool
276            | PrimitiveType::CtString
277            | PrimitiveType::Expr
278            | PrimitiveType::Field
279            | PrimitiveType::FunctionDefinition
280            | PrimitiveType::I8
281            | PrimitiveType::I16
282            | PrimitiveType::I32
283            | PrimitiveType::I64
284            | PrimitiveType::U1
285            | PrimitiveType::U8
286            | PrimitiveType::U16
287            | PrimitiveType::U32
288            | PrimitiveType::U64
289            | PrimitiveType::U128
290            | PrimitiveType::Module
291            | PrimitiveType::Quoted
292            | PrimitiveType::StructDefinition
293            | PrimitiveType::TraitConstraint
294            | PrimitiveType::TraitDefinition
295            | PrimitiveType::TraitImpl
296            | PrimitiveType::TypeDefinition
297            | PrimitiveType::TypedExpr
298            | PrimitiveType::Type
299            | PrimitiveType::UnresolvedType => {
300                if let Some(turbofish) = turbofish {
301                    self.push_err(CompilationError::TypeError(
302                        TypeCheckError::GenericCountMismatch {
303                            item: primitive_type.name().to_string(),
304                            expected: 0,
305                            found: turbofish.generics.len(),
306                            location: turbofish.location,
307                        },
308                    ));
309                }
310                (primitive_type.to_type(), false)
311            }
312            PrimitiveType::Str => {
313                let item = StrPrimitiveType;
314                let item_generic_kinds = item.generic_kinds(self.interner);
315                let generics = vecmap(&item_generic_kinds, |kind| {
316                    self.interner.next_type_variable_with_kind(kind.clone())
317                });
318                let mut args = if let Some(turbofish) = turbofish {
319                    self.resolve_item_turbofish_generics(
320                        item.item_kind(),
321                        &item.item_name(self.interner),
322                        item_generic_kinds,
323                        generics,
324                        Some(turbofish.generics),
325                        turbofish.location,
326                    )
327                } else {
328                    generics
329                };
330                assert_eq!(args.len(), 1, "str generics should be: [length]");
331                let length = args.pop().unwrap();
332                (Type::String(Box::new(length)), true)
333            }
334            PrimitiveType::Fmtstr => {
335                let item = FmtstrPrimitiveType;
336                let item_generic_kinds = item.generic_kinds(self.interner);
337                let generics = vecmap(&item_generic_kinds, |kind| {
338                    self.interner.next_type_variable_with_kind(kind.clone())
339                });
340                let mut args = if let Some(turbofish) = turbofish {
341                    self.resolve_item_turbofish_generics(
342                        FmtstrPrimitiveType.item_kind(),
343                        &item.item_name(self.interner),
344                        item_generic_kinds,
345                        generics,
346                        Some(turbofish.generics),
347                        turbofish.location,
348                    )
349                } else {
350                    generics
351                };
352                assert_eq!(args.len(), 2, "fmtstr generics should be: [length, element]");
353                let element = args.pop().unwrap();
354                let length = args.pop().unwrap();
355                (Type::FmtString(Box::new(length), Box::new(element)), true)
356            }
357        }
358    }
