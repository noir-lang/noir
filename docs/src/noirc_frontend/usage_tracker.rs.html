11#[derive(Debug, Copy, Clone, PartialEq, Eq)]
12pub enum UnusedItem {
13    Import,
14    Function(FuncId),
15    Struct(TypeId),
16    Enum(TypeId),
17    Trait(TraitId),
18    TypeAlias(TypeAliasId),
19    Global(GlobalId),
20}
21
22impl UnusedItem {
23    pub fn item_type(&self) -> &'static str {
24        match self {
25            UnusedItem::Import => "import",
26            UnusedItem::Function(_) => "function",
27            UnusedItem::Struct(_) => "struct",
28            UnusedItem::Enum(_) => "enum",
29            UnusedItem::Trait(_) => "trait",
30            UnusedItem::TypeAlias(_) => "type alias",
31            UnusedItem::Global(_) => "global",
32        }
33    }
34
35    /// The namespace a definition occupies. Imports are tracked separately (by name only) since
36    /// a `use` is one syntactic unit regardless of which namespaces the name resolves into.
37    fn namespace(&self) -> Namespace {
38        match self {
39            UnusedItem::Function(_) | UnusedItem::Global(_) => Namespace::Value,
40            UnusedItem::Struct(_)
41            | UnusedItem::Enum(_)
42            | UnusedItem::Trait(_)
43            | UnusedItem::TypeAlias(_) => Namespace::Type,
44            UnusedItem::Import => {
45                unreachable!("imports are tracked by name only, not by namespace")
46            }
47        }
48    }
49}
50
51#[derive(Debug, Default)]
52pub struct UsageTracker {
53    /// Unused definitions per module, keyed by `(namespace, name)`. Noir's type and value
54    /// namespaces are separate, so a type-namespace item and a value-namespace item can legally
55    /// share a name within a module (e.g. `struct N` and `fn N`) without a duplicate-definition
56    /// error. Keying by namespace as well as name keeps the two in separate slots, so marking one
57    /// as used (e.g. constructing the struct) leaves the genuinely unused sibling tracked.
58    unused_items: HashMap<ModuleId, HashMap<(Namespace, Ident), UnusedItem>>,
59    /// Unused imports per module, keyed by the imported name *and its location*, recording which
60    /// namespace(s) that one `use` brought the name into. `Ident`'s `Eq`/`Hash` ignore the location,
61    /// so keying by name alone would merge two distinct `use`s of the same name (`use foo::N` and
62    /// `use bar::N`) into one entry; pairing the name with its location keeps each `use` separate,
63    /// both so the warning points at the right line and so using one doesn't silence the other. A
64    /// `use` is a single syntactic unit, so using the name in any namespace it brought in uses that
65    /// whole import (the entry is dropped, warning at most once); but using a same-named item in a
66    /// namespace the import does not occupy leaves it untouched — e.g. constructing a local
67    /// `struct N` must not silence an unused `use foo::N` that only imported a value `fn N`.
68    unused_imports: HashMap<ModuleId, HashMap<(Ident, Location), HashSet<Namespace>>>,
69    /// Unused methods from inherent impls (`impl Foo { fn bar() {} }`), keyed by `FuncId`. Unlike
70    /// free functions, these don't occupy a module's `(namespace, name)` slot — they live in a
71    /// type's method set — so they're tracked by id and removed when the method is called or
72    /// otherwise referenced. The `Ident` is the method name, used to locate the unused warning.
73    unused_impl_functions: HashMap<FuncId, Ident>,
74}
75
76impl UsageTracker {
77    /// Register a definition as unused, waiting to be marked as used later.
78    /// Things that should not emit warnings should not be added at all.
79    pub(crate) fn add_unused_item(
80        &mut self,
81        module_id: ModuleId,
82        name: Ident,
83        item: UnusedItem,
84        visibility: ItemVisibility,
85    ) {
86        // Empty spans could come from implicitly injected imports, and we don't want to track those
87        if visibility != ItemVisibility::Public && !name.span().is_empty() {
88            // A genuine same-namespace clash is a duplicate-definition error reported elsewhere;
89            // `entry(..).or_insert` keeps the first item we saw so the unused warning's recorded
90            // location and item kind stay self-consistent.
91            self.unused_items
92                .entry(module_id)
93                .or_default()
94                .entry((item.namespace(), name))
95                .or_insert(item);
96        }
97    }
98
99    /// Register an unused import. Imports are keyed by name and location; the imported name may
100    /// occupy both namespaces (e.g. a re-exported `struct N` and `fn N`), in which case this is
101    /// called once per namespace with the same name and location, accumulating into the one
102    /// `use`'s entry.
103    pub(crate) fn add_unused_import(
104        &mut self,
105        module_id: ModuleId,
106        name: Ident,
107        namespace: Namespace,
108        visibility: ItemVisibility,
109    ) {
110        if visibility != ItemVisibility::Public && !name.span().is_empty() {
111            self.unused_imports
112                .entry(module_id)
113                .or_default()
114                .entry((name.clone(), name.location()))
115                .or_default()
116                .insert(namespace);
117        }
118    }
119
120    /// Marks the import that brought `name` into `namespace` as used. A `use` is a single
121    /// syntactic unit, so using the name in any namespace it occupies uses that whole import (its
122    /// entry is dropped). Each entry is keyed by the importing `use`'s location, so a same-named
123    /// item imported by a *different* `use` (a different location) into another namespace is left
124    /// alone.
125    fn mark_import_as_used(
126        &mut self,
127        current_mod_id: ModuleId,
128        name: &Ident,
129        namespace: Namespace,
130    ) {
131        let Some(imports) = self.unused_imports.get_mut(&current_mod_id) else {
132            return;
133        };
134        imports.retain(|(import_name, _location), namespaces| {
135            !(import_name == name && namespaces.contains(&namespace))
136        });
137    }
138
139    /// Marks an item as being referenced. This doesn't always makes the item as used. For example
140    /// if a struct is referenced it will still be considered unused unless it's constructed somewhere.
141    pub(crate) fn mark_as_referenced(
142        &mut self,
143        current_mod_id: ModuleId,
144        name: &Ident,
145        namespace: Namespace,
146    ) {
147        self.mark_import_as_used(current_mod_id, name, namespace);
148
149        let key = (namespace, name.clone());
150        if let Some(UnusedItem::Struct(_)) =
151            self.unused_items.get(&current_mod_id).and_then(|items| items.get(&key))
152        {
153            return;
154        }
155
156        if let Some(items) = self.unused_items.get_mut(&current_mod_id) {
157            items.remove(&key);
158        }
159    }
160
161    /// Marks an item as being used.
162    pub(crate) fn mark_as_used(
163        &mut self,
164        current_mod_id: ModuleId,
165        name: &Ident,
166        namespace: Namespace,
167    ) {
168        self.mark_import_as_used(current_mod_id, name, namespace);
169
170        let key = (namespace, name.clone());
171        if let Some(items) = self.unused_items.get_mut(&current_mod_id) {
172            items.remove(&key);
173        }
174    }
175
176    /// Register an inherent impl function as unused.
177    pub(crate) fn add_unused_impl_function(
178        &mut self,
179        func_id: FuncId,
180        name: Ident,
181        visibility: ItemVisibility,
182    ) {
183        if visibility != ItemVisibility::Public && !name.span().is_empty() {
184            self.unused_impl_functions.insert(func_id, name);
185        }
186    }
187
188    /// Marks an inherent impl function as used.
189    pub(crate) fn mark_impl_function_as_used(&mut self, func_id: &FuncId) {
190        self.unused_impl_functions.remove(func_id);
191    }
192
193    /// Get all the unused impl functions.
194    pub fn unused_impl_functions(&self) -> &HashMap<FuncId, Ident> {
195        &self.unused_impl_functions
196    }
197
198    /// Get all the unused definitions per module, keyed by `(namespace, name)`.
199    pub fn unused_items(&self) -> &HashMap<ModuleId, HashMap<(Namespace, Ident), UnusedItem>> {
200        &self.unused_items
201    }
202
203    /// Get all the unused imports per module, keyed by the imported name and its location; the
204    /// value is the set of namespaces that one `use` brought the name into.
205    pub fn unused_imports(
206        &self,
207    ) -> &HashMap<ModuleId, HashMap<(Ident, Location), HashSet<Namespace>>> {
208        &self.unused_imports
209    }
210}
