11#[derive(Debug, Copy, Clone, PartialEq, Eq)]
12pub enum UnusedItem {
13    Import,
14    Function(FuncId),
15    Struct(TypeId),
16    Enum(TypeId),
17    Trait(TraitId),
18    TypeAlias(TypeAliasId),
19    Global(GlobalId),
20}
21
22impl UnusedItem {
23    pub fn item_type(&self) -> &'static str {
24        match self {
25            UnusedItem::Import => "import",
26            UnusedItem::Function(_) => "function",
27            UnusedItem::Struct(_) => "struct",
28            UnusedItem::Enum(_) => "enum",
29            UnusedItem::Trait(_) => "trait",
30            UnusedItem::TypeAlias(_) => "type alias",
31            UnusedItem::Global(_) => "global",
32        }
33    }
34
35    /// The namespace a definition occupies. Imports are tracked separately (by name only) since
36    /// a `use` is one syntactic unit regardless of which namespaces the name resolves into.
37    fn namespace(&self) -> Namespace {
38        match self {
39            UnusedItem::Function(_) | UnusedItem::Global(_) => Namespace::Value,
40            UnusedItem::Struct(_)
41            | UnusedItem::Enum(_)
42            | UnusedItem::Trait(_)
43            | UnusedItem::TypeAlias(_) => Namespace::Type,
44            UnusedItem::Import => {
45                unreachable!("imports are tracked by name only, not by namespace")
46            }
47        }
48    }
49}

51#[derive(Debug, Default)]
52pub struct UsageTracker {
53    /// Unused definitions per module, keyed by `(namespace, name)`. Noir's type and value
54    /// namespaces are separate, so a type-namespace item and a value-namespace item can legally
55    /// share a name within a module (e.g. `struct N` and `fn N`) without a duplicate-definition
56    /// error. Keying by namespace as well as name keeps the two in separate slots, so marking one
57    /// as used (e.g. constructing the struct) leaves the genuinely unused sibling tracked.
58    unused_items: HashMap<ModuleId, HashMap<(Namespace, Ident), UnusedItem>>,
59    /// Unused imports per module, keyed by the imported name *and its location*, recording which
60    /// namespace(s) that one `use` brought the name into. `Ident`'s `Eq`/`Hash` ignore the location,
61    /// so keying by name alone would merge two distinct `use`s of the same name (`use foo::N` and
62    /// `use bar::N`) into one entry; pairing the name with its location keeps each `use` separate,
63    /// both so the warning points at the right line and so using one doesn't silence the other. A
64    /// `use` is a single syntactic unit, so using the name in any namespace it brought in uses that
65    /// whole import (the entry is dropped, warning at most once); but using a same-named item in a
66    /// namespace the import does not occupy leaves it untouched — e.g. constructing a local
67    /// `struct N` must not silence an unused `use foo::N` that only imported a value `fn N`.
68    unused_imports: HashMap<ModuleId, HashMap<(Ident, Location), HashSet<Namespace>>>,
69    /// Unused methods from inherent impls (`impl Foo { fn bar() {} }`), keyed by `FuncId`. Unlike
70    /// free functions, these don't occupy a module's `(namespace, name)` slot — they live in a
71    /// type's method set — so they're tracked by id and removed when the method is called or
72    /// otherwise referenced. The `Ident` is the method name, used to locate the unused warning.
73    unused_impl_functions: HashMap<FuncId, Ident>,
74    /// The undo log of the speculative transaction in progress, if any. `None` means no
75    /// transaction is open; `Some` records every entry removed from `unused_items`/`unused_imports`
76    /// while it is open, so they can be restored on rollback. This lets a speculative
77    /// path-resolution probe (which marks segments as used/referenced while resolving) undo those
78    /// marks when it turns out the path wasn't what it was probing for. Only one transaction may be
79    /// open at a time — see [`begin_speculative`](Self::begin_speculative).
80    speculative_undo: Option<Vec<SpeculativeUndo>>,
81}
82
83/// A single removal recorded during a speculative transaction, kept so it can be re-inserted on
84/// rollback.
85#[derive(Debug)]
86enum SpeculativeUndo {
87    Item(ModuleId, (Namespace, Ident), UnusedItem),
88    Import(ModuleId, (Ident, Location), HashSet<Namespace>),
89}
90
91/// Proof that a speculative transaction is open, returned by [`UsageTracker::begin_speculative`]
92/// and consumed by [`UsageTracker::commit_speculative`] / [`UsageTracker::rollback_speculative`].
93/// It can only be constructed here, so a transaction cannot be started without going through
94/// `begin_speculative`, and `#[must_use]` nudges callers to finish it. Prefer driving this through
95/// [`Elaborator::speculatively`](crate::elaborator::Elaborator) rather than by hand.
96#[must_use]
97pub(crate) struct SpeculativeTx {
98    _private: (),
99}
100
101impl UsageTracker {
102    /// Register a definition as unused, waiting to be marked as used later.
103    /// Things that should not emit warnings should not be added at all.
104    pub(crate) fn add_unused_item(
105        &mut self,
106        module_id: ModuleId,
107        name: Ident,
108        item: UnusedItem,
109        visibility: ItemVisibility,
110    ) {
111        // Empty spans could come from implicitly injected imports, and we don't want to track those
112        if visibility != ItemVisibility::Public && !name.span().is_empty() {
113            // A genuine same-namespace clash is a duplicate-definition error reported elsewhere;
114            // `entry(..).or_insert` keeps the first item we saw so the unused warning's recorded
115            // location and item kind stay self-consistent.
116            self.unused_items
117                .entry(module_id)
118                .or_default()
119                .entry((item.namespace(), name))
120                .or_insert(item);
121        }
122    }
123
124    /// Register an unused import. Imports are keyed by name and location; the imported name may
125    /// occupy both namespaces (e.g. a re-exported `struct N` and `fn N`), in which case this is
126    /// called once per namespace with the same name and location, accumulating into the one
127    /// `use`'s entry.
128    pub(crate) fn add_unused_import(
129        &mut self,
130        module_id: ModuleId,
131        name: Ident,
132        namespace: Namespace,
133        visibility: ItemVisibility,
134    ) {
135        if visibility != ItemVisibility::Public && !name.span().is_empty() {
136            self.unused_imports
137                .entry(module_id)
138                .or_default()
139                .entry((name.clone(), name.location()))
140                .or_default()
141                .insert(namespace);
142        }
143    }
144
145    /// Marks the import that brought `name` into `namespace` as used. A `use` is a single
146    /// syntactic unit, so using the name in any namespace it occupies uses that whole import (its
147    /// entry is dropped). Each entry is keyed by the importing `use`'s location, so a same-named
148    /// item imported by a *different* `use` (a different location) into another namespace is left
149    /// alone.
150    fn mark_import_as_used(
151        &mut self,
152        current_mod_id: ModuleId,
153        name: &Ident,
154        namespace: Namespace,
155    ) {
156        let recording = self.speculative_undo.is_some();
157        let Some(imports) = self.unused_imports.get_mut(&current_mod_id) else {
158            return;
159        };
160        let mut removed = Vec::new();
161        imports.retain(|(import_name, location), namespaces| {
162            let remove = import_name == name && namespaces.contains(&namespace);
163            if remove && recording {
164                removed.push(((import_name.clone(), *location), namespaces.clone()));
165            }
166            !remove
167        });
168        if let Some(undo) = &mut self.speculative_undo {
169            for (key, namespaces) in removed {
170                undo.push(SpeculativeUndo::Import(current_mod_id, key, namespaces));
171            }
172        }
173    }
174
175    /// Marks an item as being referenced. This doesn't always makes the item as used. For example
176    /// if a struct is referenced it will still be considered unused unless it's constructed somewhere.
177    pub(crate) fn mark_as_referenced(
178        &mut self,
179        current_mod_id: ModuleId,
180        name: &Ident,
181        namespace: Namespace,
182    ) {
183        self.mark_import_as_used(current_mod_id, name, namespace);
184
185        let key = (namespace, name.clone());
186        if let Some(UnusedItem::Struct(_)) =
187            self.unused_items.get(&current_mod_id).and_then(|items| items.get(&key))
188        {
189            return;
190        }
191
192        let removed =
193            self.unused_items.get_mut(&current_mod_id).and_then(|items| items.remove_entry(&key));
194        if let Some((stored_key, item)) = removed {
195            self.record_item_removal(current_mod_id, stored_key, item);
196        }
197    }
198
199    /// Marks an item as being used.
200    pub(crate) fn mark_as_used(
201        &mut self,
202        current_mod_id: ModuleId,
203        name: &Ident,
204        namespace: Namespace,
205    ) {
206        self.mark_import_as_used(current_mod_id, name, namespace);
207
208        let key = (namespace, name.clone());
209        let removed =
210            self.unused_items.get_mut(&current_mod_id).and_then(|items| items.remove_entry(&key));
211        if let Some((stored_key, item)) = removed {
212            self.record_item_removal(current_mod_id, stored_key, item);
213        }
214    }
215
216    /// While a speculative transaction is open, record an `unused_items` removal so it can be
217    /// restored on rollback. A no-op when no transaction is in progress.
218    ///
219    /// `key` must be the *stored* key returned by `remove_entry`, not the lookup key built from the
220    /// caller's ident. `Ident`'s `Eq`/`Hash` ignore location, so a lookup ident carries the
221    /// reference's location while the stored ident carries the definition's. Recording the stored
222    /// key keeps the restored entry pointing at the definition, so a later unused warning lands on
223    /// the definition rather than the (rolled-back) reference site.
224    fn record_item_removal(
225        &mut self,
226        module_id: ModuleId,
227        key: (Namespace, Ident),
228        item: UnusedItem,
229    ) {
230        if let Some(undo) = &mut self.speculative_undo {
231            undo.push(SpeculativeUndo::Item(module_id, key, item));
232        }
233    }
234
235    /// Begin a speculative transaction: until it is committed or rolled back, every removal from
236    /// `unused_items`/`unused_imports` is recorded so it can be undone. Panics if one is already
237    /// open — transactions don't nest. Prefer [`Elaborator::speculatively`] over calling this
238    /// directly.
239    ///
240    /// [`Elaborator::speculatively`]: crate::elaborator::Elaborator
241    pub(crate) fn begin_speculative(&mut self) -> SpeculativeTx {
242        assert!(
243            self.speculative_undo.is_none(),
244            "a speculative transaction is already in progress; they do not nest"
245        );
246        self.speculative_undo = Some(Vec::new());
247        SpeculativeTx { _private: () }
248    }
249
250    /// Commit a speculative transaction, keeping every change made while it was open.
251    pub(crate) fn commit_speculative(&mut self, _tx: SpeculativeTx) {
252        self.speculative_undo = None;
253    }
254
255    /// Roll back a speculative transaction, restoring every entry removed while it was open.
256    pub(crate) fn rollback_speculative(&mut self, _tx: SpeculativeTx) {
257        let undo =
258            self.speculative_undo.take().expect("speculative transaction must be in progress");
259        for entry in undo.into_iter().rev() {
260            match entry {
261                SpeculativeUndo::Item(module_id, key, item) => {
262                    self.unused_items.entry(module_id).or_default().insert(key, item);
263                }
264                SpeculativeUndo::Import(module_id, key, namespaces) => {
265                    self.unused_imports.entry(module_id).or_default().insert(key, namespaces);
266                }
267            }
268        }
269    }
270
271    /// Register an inherent impl function as unused.
272    pub(crate) fn add_unused_impl_function(
273        &mut self,
274        func_id: FuncId,
275        name: Ident,
276        visibility: ItemVisibility,
277    ) {
278        if visibility != ItemVisibility::Public && !name.span().is_empty() {
279            self.unused_impl_functions.insert(func_id, name);
280        }
281    }
282
283    /// Marks an inherent impl function as used.
284    pub(crate) fn mark_impl_function_as_used(&mut self, func_id: &FuncId) {
285        self.unused_impl_functions.remove(func_id);
286    }
287
288    /// Get all the unused impl functions.
289    pub fn unused_impl_functions(&self) -> &HashMap<FuncId, Ident> {
290        &self.unused_impl_functions
291    }
292
293    /// Get all the unused definitions per module, keyed by `(namespace, name)`.
294    pub fn unused_items(&self) -> &HashMap<ModuleId, HashMap<(Namespace, Ident), UnusedItem>> {
295        &self.unused_items
296    }
297
298    /// Get all the unused imports per module, keyed by the imported name and its location; the
299    /// value is the set of namespaces that one `use` brought the name into.
300    pub fn unused_imports(
301        &self,
302    ) -> &HashMap<ModuleId, HashMap<(Ident, Location), HashSet<Namespace>>> {
303        &self.unused_imports
304    }
