1use std::collections::HashMap;

3use crate::{
4    ast::{Ident, ItemVisibility},
5    hir::def_map::ModuleId,
6    node_interner::{FuncId, GlobalId, TraitId, TypeAliasId, TypeId},
7};

9#[derive(Debug, Copy, Clone, PartialEq, Eq)]
10pub enum UnusedItem {
11    Import,
12    Function(FuncId),
13    Struct(TypeId),
14    Enum(TypeId),
15    Trait(TraitId),
16    TypeAlias(TypeAliasId),
17    Global(GlobalId),
18}
19
20impl UnusedItem {
21    pub fn item_type(&self) -> &'static str {
22        match self {
23            UnusedItem::Import => "import",
24            UnusedItem::Function(_) => "function",
25            UnusedItem::Struct(_) => "struct",
26            UnusedItem::Enum(_) => "enum",
27            UnusedItem::Trait(_) => "trait",
28            UnusedItem::TypeAlias(_) => "type alias",
29            UnusedItem::Global(_) => "global",
30        }
31    }
32}
33
34#[derive(Debug, Default)]
35pub struct UsageTracker {
36    unused_items: HashMap<ModuleId, HashMap<Ident, UnusedItem>>,
37}
38
39impl UsageTracker {
40    /// Register an item as unused, waiting to be marked as used later.
41    /// Things that should not emit warnings should not be added at all.
42    pub(crate) fn add_unused_item(
43        &mut self,
44        module_id: ModuleId,
45        name: Ident,
46        item: UnusedItem,
47        visibility: ItemVisibility,
48    ) {
49        // Empty spans could come from implicitly injected imports, and we don't want to track those
50        if visibility != ItemVisibility::Public && name.span().start() < name.span().end() {
51            self.unused_items.entry(module_id).or_default().insert(name, item);
52        }
53    }
54
55    /// Marks an item as being referenced. This doesn't always makes the item as used. For example
56    /// if a struct is referenced it will still be considered unused unless it's constructed somewhere.
57    pub(crate) fn mark_as_referenced(&mut self, current_mod_id: ModuleId, name: &Ident) {
58        let Some(items) = self.unused_items.get_mut(&current_mod_id) else {
59            return;
60        };
61
62        let Some(unused_item) = items.get(name) else {
63            return;
64        };
65
66        if let UnusedItem::Struct(_) = unused_item {
67            return;
68        }
69
70        items.remove(name);
71    }
72
73    /// Marks an item as being used.
74    pub(crate) fn mark_as_used(&mut self, current_mod_id: ModuleId, name: &Ident) {
75        if let Some(items) = self.unused_items.get_mut(&current_mod_id) {
76            items.remove(name);
77        }
78    }
79
80    /// Get all the unused items per module.
81    pub fn unused_items(&self) -> &HashMap<ModuleId, HashMap<Ident, UnusedItem>> {
82        &self.unused_items
83    }
84}
