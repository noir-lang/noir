22/// This structure is used to collect information about variables to track
23/// for debugging during the instrumentation injection phase.
24#[derive(Debug, Clone)]
25pub struct DebugInstrumenter {
26    // all collected variable names while instrumenting the source for variable tracking
27    pub variables: HashMap<SourceVarId, String>,
28
29    // all field names referenced when assigning to a member of a variable
30    pub field_names: HashMap<SourceFieldId, String>,
31
32    // all collected function metadata (name + argument names)
33    pub functions: HashMap<DebugFnId, DebugFunction>,
34
35    next_var_id: u32,
36    next_field_name_id: u32,
37    next_fn_id: u32,
38
39    // last seen variable names and their IDs grouped by scope
40    scope: Vec<HashMap<String, SourceVarId>>,
41}
42
43impl Default for DebugInstrumenter {
44    fn default() -> Self {
45        Self {
46            variables: HashMap::default(),
47            field_names: HashMap::default(),
48            functions: HashMap::default(),
49            scope: vec![],
50            next_var_id: 0,
51            next_field_name_id: 1,
52            next_fn_id: 0,
53        }
54    }
55}

57impl DebugInstrumenter {
58    pub fn instrument_module(&mut self, module: &mut ParsedModule, file: FileId) {
59        module.items.iter_mut().for_each(|item| {
60            match &mut item.kind {
61                // Instrument top-level functions of a module
62                ItemKind::Function(f) => self.walk_fn(&mut f.def),
63                // Instrument contract module
64                ItemKind::Submodules(ParsedSubModule {
65                    is_contract: true,
66                    contents: contract_module,
67                    ..
68                }) => {
69                    self.instrument_module(contract_module, file);
70                }
71                _ => (),
72            }
73        });
74
75        // this part absolutely must happen after ast traversal above
76        // so that oracle functions don't get wrapped, resulting in infinite recursion:
77        self.insert_state_set_oracle(module, file);
78    }

80    fn insert_var(&mut self, var_name: &str) -> Option<SourceVarId> {
81        if var_name == "_" {
82            return None;
83        }
84
85        let var_id = SourceVarId(self.next_var_id);
86        self.next_var_id += 1;
87        self.variables.insert(var_id, var_name.to_string());
88        self.scope.last_mut().unwrap().insert(var_name.to_string(), var_id);
89        Some(var_id)
90    }
91
92    fn lookup_var(&self, var_name: &str) -> Option<SourceVarId> {
93        self.scope.iter().rev().find_map(|vars| vars.get(var_name).copied())
94    }
95
96    fn insert_field_name(&mut self, field_name: &str) -> SourceFieldId {
97        let field_name_id = SourceFieldId(self.next_field_name_id);
98        self.next_field_name_id += 1;
99        self.field_names.insert(field_name_id, field_name.to_string());
100        field_name_id
101    }

103    fn insert_function(&mut self, fn_name: String, arguments: Vec<String>) -> DebugFnId {
104        let fn_id = DebugFnId(self.next_fn_id);
105        self.next_fn_id += 1;
106        self.functions.insert(fn_id, DebugFunction { name: fn_name, arg_names: arguments });
107        fn_id
108    }

110    fn walk_fn(&mut self, func: &mut ast::FunctionDefinition) {
111        // Don't instrument functions that are not supposed to have a body
112        if let Some((func, _)) = &func.attributes.function {
113            match func.kind {
114                FunctionAttributeKind::Foreign(_)
115                | FunctionAttributeKind::Builtin(_)
116                | FunctionAttributeKind::Oracle(_) => return,
117                FunctionAttributeKind::Test(..)
118                | FunctionAttributeKind::Fold
119                | FunctionAttributeKind::NoPredicates
120                | FunctionAttributeKind::InlineAlways
121                | FunctionAttributeKind::InlineNever
122                | FunctionAttributeKind::FuzzingHarness(..) => (),
123            }
124        }
125
126        let func_name = func.name.to_string();
127        let func_args =
128            func.parameters.iter().map(|param| pattern_to_string(&param.pattern)).collect();
129        let fn_id = self.insert_function(func_name, func_args);
130        let enter_stmt = build_debug_call_stmt("enter", fn_id, func.location);
131        self.scope.push(HashMap::default());
132
133        let set_fn_params: Vec<_> = func
134            .parameters
135            .iter()
136            .flat_map(|param| {
137                pattern_vars(&param.pattern)
138                    .iter()
139                    .filter_map(|(id, _is_mut)| {
140                        let var_id = self.insert_var(id.as_str())?;
141                        Some(build_assign_var_stmt(var_id, id_expr(id)))
142                    })
143                    .collect::<Vec<_>>()
144            })
145            .collect();
146
147        let func_body = &mut func.body.statements;
148        let mut statements = take(func_body);
149
150        self.walk_scope(&mut statements, func.location);
151
152        // walk_scope ensures that the last statement is the return value of the function
153        let last_stmt = statements.pop().expect("at least one statement after walk_scope");
154        let exit_stmt = build_debug_call_stmt("exit", fn_id, last_stmt.location);
155
156        // rebuild function body
157        func_body.push(enter_stmt);
158        func_body.extend(set_fn_params);
159        func_body.extend(statements);
160        func_body.push(exit_stmt);
161        func_body.push(last_stmt);
162    }

164    // Modify a vector of statements in-place, adding instrumentation for sets and drops.
165    // This function will consume a scope level.
166    fn walk_scope(&mut self, statements: &mut Vec<ast::Statement>, location: Location) {
167        statements.iter_mut().for_each(|stmt| self.walk_statement(stmt));
168
169        // extract and save the return value from the scope if there is one
170        let ret_stmt = statements.pop();
171        let has_ret_expr = match ret_stmt {
172            None => false,
173            Some(ast::Statement { kind: ast::StatementKind::Expression(ret_expr), .. }) => {
174                let save_ret_expr = ast::Statement {
175                    kind: ast::StatementKind::new_let(
176                        ast::Pattern::Identifier(ident("__debug_expr", ret_expr.location)),
177                        None,
178                        ret_expr.clone(),
179                        vec![],
180                    ),
181                    location: ret_expr.location,
182                };
183                statements.push(save_ret_expr);
184                true
185            }
186            Some(ret_stmt) => {
187                // not an expression, so leave it untouched
188                statements.push(ret_stmt);
189                false
190            }
191        };
192
193        let span = Span::empty(location.span.end());
194        let location = Location::new(span, location.file);
195
196        // drop scope variables
197        let scope_vars = self.scope.pop().unwrap_or_default();
198        let drop_vars_stmts =
199            scope_vars.values().map(|var_id| build_drop_var_stmt(*var_id, location));
200        statements.extend(drop_vars_stmts);
201
202        // return the saved value in __debug_expr, or unit otherwise
203        let last_stmt = if has_ret_expr {
204            ast::Statement {
205                kind: ast::StatementKind::Expression(ast::Expression {
206                    kind: ast::ExpressionKind::Variable(Path::plain(
207                        vec![PathSegment::from(ident("__debug_expr", location))],
208                        location,
209                    )),
210                    location,
211                }),
212                location,
213            }
214        } else {
215            ast::Statement {
216                kind: ast::StatementKind::Expression(ast::Expression {
217                    kind: ast::ExpressionKind::Literal(ast::Literal::Unit),
218                    location,
219                }),
220                location,
221            }
222        };
223        statements.push(last_stmt);
224    }

226    fn walk_let_statement(
227        &mut self,
228        let_stmt: &ast::LetStatement,
229        location: Location,
230    ) -> ast::Statement {
231        // rewrites let statements written like this:
232        //   let (((a,b,c),D { d }),e,f) = x;
233        //
234        // into statements like this:
235        //
236        //   let (a,b,c,d,e,f,g) = {
237        //     let (((a,b,c),D { d }),e,f) = x;
238        //     wrap(1, a);
239        //     wrap(2, b);
240        //     ...
241        //     wrap(6, f);
242        //     (a,b,c,d,e,f,g)
243        //   };
244
245        // a.b.c[3].x[i*4+1].z
246
247        let vars = pattern_vars(&let_stmt.pattern);
248        let vars_pattern: Vec<ast::Pattern> = vars
249            .iter()
250            .map(|(id, is_mut)| {
251                if *is_mut {
252                    ast::Pattern::Mutable(
253                        Box::new(ast::Pattern::Identifier(id.clone())),
254                        id.location(),
255                        true,
256                    )
257                } else {
258                    ast::Pattern::Identifier(id.clone())
259                }
260            })
261            .collect();
262        let vars_exprs: Vec<ast::Expression> = vars
263            .iter()
264            .map(|(id, _)| {
265                // We don't want to generate an expression to read from "_".
266                // And since this expression is going to be assigned to "_" so it doesn't matter
267                // what it is, we can use `()` for it.
268                if id.as_str() == "_" {
269                    ast::Expression {
270                        kind: ast::ExpressionKind::Literal(ast::Literal::Unit),
271                        location: id.location(),
272                    }
273                } else {
274                    id_expr(id)
275                }
276            })
277            .collect();
278
279        let mut block_stmts =
280            vec![ast::Statement { kind: ast::StatementKind::Let(let_stmt.clone()), location }];
281        block_stmts.extend(vars.iter().filter_map(|(id, _)| {
282            let var_id = self.insert_var(id.as_str())?;
283            Some(build_assign_var_stmt(var_id, id_expr(id)))
284        }));
285        block_stmts.push(ast::Statement {
286            kind: ast::StatementKind::Expression(ast::Expression {
287                kind: ast::ExpressionKind::Tuple(vars_exprs),
288                location: let_stmt.pattern.location(),
289            }),
290            location: let_stmt.pattern.location(),
291        });
292
293        ast::Statement {
294            kind: ast::StatementKind::new_let(
295                ast::Pattern::Tuple(vars_pattern, let_stmt.pattern.location()),
296                None,
297                ast::Expression {
298                    kind: ast::ExpressionKind::Block(ast::BlockExpression {
299                        statements: block_stmts,
300                    }),
301                    location: let_stmt.expression.location,
302                },
303                vec![],
304            ),
305            location,
306        }
307    }

309    fn walk_assign_statement(
310        &mut self,
311        assign_stmt: &ast::AssignStatement,
312        location: Location,
313    ) -> ast::Statement {
314        // X = Y becomes:
315        // X = {
316        //   let __debug_expr = Y;
317        //
318        //   __debug_var_assign(17, __debug_expr);
319        //   // or:
320        //   __debug_member_assign_{arity}(17, __debug_expr, _v0, _v1..., _v{arity});
321        //
322        //   __debug_expr
323        // };
324
325        let let_kind = ast::StatementKind::new_let(
326            ast::Pattern::Identifier(ident("__debug_expr", assign_stmt.expression.location)),
327            None,
328            assign_stmt.expression.clone(),
329            vec![],
330        );
331        let expression_location = assign_stmt.expression.location;
332        let new_assign_stmt = match &assign_stmt.lvalue {
333            ast::LValue::Path(id) => {
334                let Some(id) = id.as_ident() else {
335                    panic!("var lookup failed for var_name={id}");
336                };
337                let var_id = self
338                    .lookup_var(id.as_str())
339                    .unwrap_or_else(|| panic!("var lookup failed for var_name={id}"));
340                build_assign_var_stmt(var_id, id_expr(&ident("__debug_expr", id.location())))
341            }
342            ast::LValue::Dereference(_lv, location) => {
343                // TODO: this is a dummy statement for now, but we should
344                // somehow track the dereference and update the pointed to
345                // variable
346                ast::Statement {
347                    kind: ast::StatementKind::Expression(uint_expr(0, *location)),
348                    location: *location,
349                }
350            }
351            _ => {
352                let mut indexes = vec![];
353                let mut cursor = &assign_stmt.lvalue;
354                let var_id;
355                loop {
356                    match cursor {
357                        ast::LValue::Path(id) => {
358                            let Some(id) = id.as_ident() else {
359                                panic!("var lookup failed for var_name={id}");
360                            };
361
362                            var_id = self
363                                .lookup_var(id.as_str())
364                                .unwrap_or_else(|| panic!("var lookup failed for var_name={id}"));
365                            break;
366                        }
367                        ast::LValue::MemberAccess { object, field_name, location } => {
368                            cursor = object;
369                            let field_name_id = self.insert_field_name(field_name.as_str());
370                            indexes.push(sint_expr(-i128::from(field_name_id.0), *location));
371                        }
372                        ast::LValue::Index { index, array, location: _ } => {
373                            cursor = array;
374                            indexes.push(index.clone());
375                        }
376                        ast::LValue::Dereference(_ref, _span) => {
377                            unimplemented![]
378                        }
379                        ast::LValue::Interned(..) => {
380                            unimplemented![]
381                        }
382                    }
383                }
384                build_assign_member_stmt(
385                    var_id,
386                    &indexes,
387                    &id_expr(&ident("__debug_expr", expression_location)),
388                )
389            }
390        };
391
392        let ret_kind =
393            ast::StatementKind::Expression(id_expr(&ident("__debug_expr", expression_location)));
394
395        ast::Statement {
396            kind: ast::StatementKind::Assign(ast::AssignStatement {
397                lvalue: assign_stmt.lvalue.clone(),
398                expression: ast::Expression {
399                    kind: ast::ExpressionKind::Block(ast::BlockExpression {
400                        statements: vec![
401                            ast::Statement { kind: let_kind, location: expression_location },
402                            new_assign_stmt,
403                            ast::Statement { kind: ret_kind, location: expression_location },
404                        ],
405                    }),
406                    location: expression_location,
407                },
408            }),
409            location,
410        }
411    }

413    fn walk_expr(&mut self, expr: &mut ast::Expression) {
414        match &mut expr.kind {
415            ast::ExpressionKind::Block(ast::BlockExpression { statements, .. }) => {
416                self.scope.push(HashMap::default());
417                self.walk_scope(statements, expr.location);
418            }
419            ast::ExpressionKind::Prefix(prefix_expr) => {
420                self.walk_expr(&mut prefix_expr.rhs);
421            }
422            ast::ExpressionKind::Index(index_expr) => {
423                self.walk_expr(&mut index_expr.collection);
424                self.walk_expr(&mut index_expr.index);
425            }
426            ast::ExpressionKind::Call(call_expr) => {
427                // TODO: push a stack frame or something here?
428                self.walk_expr(&mut call_expr.func);
429                call_expr.arguments.iter_mut().for_each(|expr| {
430                    self.walk_expr(expr);
431                });
432            }
433            ast::ExpressionKind::MethodCall(mc_expr) => {
434                // TODO: also push a stack frame here
435                self.walk_expr(&mut mc_expr.object);
436                mc_expr.arguments.iter_mut().for_each(|expr| {
437                    self.walk_expr(expr);
438                });
439            }
440            ast::ExpressionKind::Constructor(c_expr) => {
441                c_expr.fields.iter_mut().for_each(|(_id, expr)| {
442                    self.walk_expr(expr);
443                });
444            }
445            ast::ExpressionKind::MemberAccess(ma_expr) => {
446                self.walk_expr(&mut ma_expr.lhs);
447            }
448            ast::ExpressionKind::Cast(cast_expr) => {
449                self.walk_expr(&mut cast_expr.lhs);
450            }
451            ast::ExpressionKind::Infix(infix_expr) => {
452                self.walk_expr(&mut infix_expr.lhs);
453                self.walk_expr(&mut infix_expr.rhs);
454            }
455            ast::ExpressionKind::If(if_expr) => {
456                self.walk_expr(&mut if_expr.condition);
457                self.walk_expr(&mut if_expr.consequence);
458                if let Some(ref mut alt) = if_expr.alternative {
459                    self.walk_expr(alt);
460                }
461            }
462            ast::ExpressionKind::Tuple(exprs) => {
463                exprs.iter_mut().for_each(|ref mut expr| {
464                    self.walk_expr(expr);
465                });
466            }
467            ast::ExpressionKind::Lambda(lambda) => {
468                self.walk_expr(&mut lambda.body);
469            }
470            ast::ExpressionKind::Parenthesized(expr) => {
471                self.walk_expr(expr);
472            }
473            _ => {}
474        }
475    }

477    fn walk_for(&mut self, for_stmt: &mut ast::ForLoopStatement) {
478        let var_name = for_stmt.identifier.as_str();
479        let var_id = self.insert_var(var_name);
480
481        let set_and_drop_stmt = var_id.map(|var_id| {
482            let span = Span::empty(for_stmt.location.span.end());
483            (
484                build_assign_var_stmt(var_id, id_expr(&for_stmt.identifier)),
485                build_drop_var_stmt(var_id, Location::new(span, for_stmt.location.file)),
486            )
487        });
488
489        self.walk_expr(&mut for_stmt.block);
490
491        let mut statements = Vec::new();
492        let block_statement = ast::Statement {
493            kind: ast::StatementKind::Semi(for_stmt.block.clone()),
494            location: for_stmt.block.location,
495        };
496
497        if let Some((set_stmt, drop_stmt)) = set_and_drop_stmt {
498            statements.push(set_stmt);
499            statements.push(block_statement);
500            statements.push(drop_stmt);
501        } else {
502            statements.push(block_statement);
503        }
504
505        for_stmt.block = ast::Expression {
506            kind: ast::ExpressionKind::Block(ast::BlockExpression { statements }),
507            location: for_stmt.location,
508        };
509    }
510
511    fn walk_statement(&mut self, stmt: &mut ast::Statement) {
512        match &mut stmt.kind {
513            ast::StatementKind::Let(let_stmt) => {
514                *stmt = self.walk_let_statement(let_stmt, stmt.location);
515            }
516            ast::StatementKind::Assign(assign_stmt) => {
517                *stmt = self.walk_assign_statement(assign_stmt, stmt.location);
518            }
519            ast::StatementKind::Expression(expr) => {
520                self.walk_expr(expr);
521            }
522            ast::StatementKind::Semi(expr) => {
523                self.walk_expr(expr);
524            }
525            ast::StatementKind::For(for_stmt) => {
526                self.walk_for(for_stmt);
527            }
528            _ => {} // Constrain, Error
529        }
530    }

532    fn insert_state_set_oracle(&self, module: &mut ParsedModule, file: FileId) {
533        let member_assigns = (1..=MAX_MEMBER_ASSIGN_DEPTH)
534            .map(|i| format!["__debug_member_assign_{i}"])
535            .collect::<Vec<String>>()
536            .join(",\n");
537        let (program, errors) = parse_program(
538            &format!(
539                r#"
540            use __debug::{{
541                __debug_var_assign,
542                __debug_var_drop,
543                __debug_fn_enter,
544                __debug_fn_exit,
545                __debug_dereference_assign,
546                {member_assigns},
547            }};"#
548            ),
549            file,
550        );
551        if !errors.is_empty() {
552            panic!("errors parsing internal oracle definitions: {errors:?}")
553        }
554        module.items.extend(program.items);
555    }
556}

558pub fn build_debug_crate_file() -> String {
559    [
560        r#"
561            #[oracle(__debug_var_assign)]
562            unconstrained fn __debug_var_assign_oracle<T>(_var_id: u32, _value: T) {}
563            unconstrained fn __debug_var_assign_inner<T>(var_id: u32, value: T) {
564                __debug_var_assign_oracle(var_id, value);
565            }
566            pub fn __debug_var_assign<T>(var_id: u32, value: T) {
567                // Safety: debug context
568                unsafe {
569                {
570                    __debug_var_assign_inner(var_id, value);
571                }}
572            }
573
574            #[oracle(__debug_var_drop)]
575            unconstrained fn __debug_var_drop_oracle(_var_id: u32) {}
576            unconstrained fn __debug_var_drop_inner(var_id: u32) {
577                __debug_var_drop_oracle(var_id);
578            }
579            pub fn __debug_var_drop(var_id: u32) {
580                // Safety: debug context
581                unsafe {
582                {
583                    __debug_var_drop_inner(var_id);
584                }}
585            }
586
587            #[oracle(__debug_fn_enter)]
588            unconstrained fn __debug_fn_enter_oracle(_fn_id: u32) {}
589            unconstrained fn __debug_fn_enter_inner(fn_id: u32) {
590                __debug_fn_enter_oracle(fn_id);
591            }
592            pub fn __debug_fn_enter(fn_id: u32) {
593                // Safety: debug context
594                unsafe {
595                {
596                    __debug_fn_enter_inner(fn_id);
597                }}
598            }
599
600            #[oracle(__debug_fn_exit)]
601            unconstrained fn __debug_fn_exit_oracle(_fn_id: u32) {}
602            unconstrained fn __debug_fn_exit_inner(fn_id: u32) {
603                __debug_fn_exit_oracle(fn_id);
604            }
605            pub fn __debug_fn_exit(fn_id: u32) {
606                // Safety: debug context
607                unsafe {
608                {
609                    __debug_fn_exit_inner(fn_id);
610                }}
611            }
612
613            #[oracle(__debug_dereference_assign)]
614            unconstrained fn __debug_dereference_assign_oracle<T>(_var_id: u32, _value: T) {}
615            unconstrained fn __debug_dereference_assign_inner<T>(var_id: u32, value: T) {
616                __debug_dereference_assign_oracle(var_id, value);
617            }
618            pub fn __debug_dereference_assign<T>(var_id: u32, value: T) {
619                // Safety: debug context
620                unsafe {
621                {
622                    __debug_dereference_assign_inner(var_id, value);
623                }}
624            }
625        "#
626        .to_string(),
627        (1..=MAX_MEMBER_ASSIGN_DEPTH)
628            .map(|n| {
629                // The variable signature has to be generic as Noir supports using any polymorphic integer as an index.
630                // If we were to set a specific type for index signatures here, such as `Field`, we will error in
631                // type checking if we attempt to index with a different type such as `u8`.
632                let indices =
633                    (0..n).map(|i| format!["Index{i}"]).collect::<Vec<String>>().join(", ");
634                let var_sig =
635                    (0..n).map(|i| format!["_v{i}: Index{i}"]).collect::<Vec<String>>().join(", ");
636                let vars = (0..n).map(|i| format!["_v{i}"]).collect::<Vec<String>>().join(", ");
637                format!(
638                    r#"
639                #[oracle(__debug_member_assign_{n})]
640                unconstrained fn __debug_oracle_member_assign_{n}<T, {indices}>(
641                    _var_id: u32, _value: T, {var_sig}
642                ) {{}}
643                unconstrained fn __debug_inner_member_assign_{n}<T, {indices}>(
644                    var_id: u32, value: T, {var_sig}
645                ) {{
646                    __debug_oracle_member_assign_{n}(var_id, value, {vars});
647                }}
648                pub fn __debug_member_assign_{n}<T, {indices}>(var_id: u32, value: T, {var_sig}) {{
649                    /// Safety: debug context
650                    unsafe {{
651                        __debug_inner_member_assign_{n}(var_id, value, {vars});
652                    }}
653                }}
654
655            "#
656                )
657            })
658            .collect::<Vec<String>>()
659            .join("\n"),
660    ]
661    .join("\n")
662}

664fn build_assign_var_stmt(var_id: SourceVarId, expr: ast::Expression) -> ast::Statement {
665    let location = expr.location;
666    let kind = ast::ExpressionKind::Call(Box::new(ast::CallExpression {
667        func: Box::new(ast::Expression {
668            kind: ast::ExpressionKind::Variable(Path::plain(
669                vec![PathSegment::from(ident("__debug_var_assign", location))],
670                location,
671            )),
672            location,
673        }),
674        is_macro_call: false,
675        arguments: vec![uint_expr(u128::from(var_id.0), location), expr],
676    }));
677    ast::Statement { kind: ast::StatementKind::Semi(ast::Expression { kind, location }), location }
678}

739fn pattern_vars(pattern: &ast::Pattern) -> Vec<(ast::Ident, bool)> {
740    let mut vars = vec![];
741    let mut stack = VecDeque::from([(pattern, false)]);
742    while stack.front().is_some() {
743        let (pattern, is_mut) = stack.pop_front().unwrap();
744        match pattern {
745            ast::Pattern::Identifier(id) => {
746                if id.as_str() != "_" {
747                    vars.push((id.clone(), is_mut));
748                }
749            }
750            ast::Pattern::Mutable(pattern, _, _) => {
751                stack.push_back((pattern, true));
752            }
753            ast::Pattern::Tuple(patterns, _) => {
754                stack.extend(patterns.iter().map(|pattern| (pattern, false)));
755            }
756            ast::Pattern::Struct(_, fields, _) => {
757                stack.extend(fields.iter().map(|(_, pattern)| (pattern, is_mut)));
758            }
759            ast::Pattern::Parenthesized(pattern, _) => {
760                stack.push_back((pattern, false));
761            }
762            ast::Pattern::Interned(_, _) => (),
763        }
764    }
765    vars
766}
767
768fn pattern_to_string(pattern: &ast::Pattern) -> String {
769    match pattern {
770        ast::Pattern::Identifier(id) => id.to_string(),
771        ast::Pattern::Mutable(pattern, _, _) => {
772            format!("mut {}", pattern_to_string(pattern.as_ref()))
773        }
774        ast::Pattern::Tuple(elements, _) => format!(
775            "({})",
776            elements.iter().map(pattern_to_string).collect::<Vec<String>>().join(", ")
777        ),
778        ast::Pattern::Struct(name, fields, _) => {
779            format!(
780                "{} {{ {} }}",
781                name,
782                fields
783                    .iter()
784                    .map(|(field_ident, field_pattern)| {
785                        format!("{}: {}", field_ident, pattern_to_string(field_pattern))
786                    })
787                    .collect::<Vec<_>>()
788                    .join(", "),
789            )
790        }
791        ast::Pattern::Parenthesized(pattern, _) => {
792            format!("({})", pattern_to_string(pattern.as_ref()))
793        }
794        ast::Pattern::Interned(_, _) => "?Interned".to_string(),
795    }
796}
