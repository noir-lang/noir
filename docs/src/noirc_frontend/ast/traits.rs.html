17/// AST node for trait definitions:
18/// `trait name<generics> { ... items ... }`
19#[derive(Clone, Debug)]
20pub struct NoirTrait {
21    pub name: Ident,
22    pub generics: UnresolvedGenerics,
23    pub bounds: Vec<TraitBound>,
24    pub where_clause: Vec<UnresolvedTraitConstraint>,
25    pub location: Location,
26    pub items: Vec<Documented<TraitItem>>,
27    pub attributes: Vec<SecondaryAttribute>,
28    pub visibility: ItemVisibility,
29    pub is_alias: bool,
30}

32/// Any declaration inside the body of a trait that a user is required to
33/// specify when implementing the trait.
34#[derive(Clone, Debug)]
35pub enum TraitItem {
36    Function {
37        is_unconstrained: bool,
38        visibility: ItemVisibility,
39        is_comptime: bool,
40        name: Ident,
41        generics: UnresolvedGenerics,
42        parameters: Vec<(Ident, UnresolvedType)>,
43        return_type: FunctionReturnType,
44        where_clause: Vec<UnresolvedTraitConstraint>,
45        body: Option<BlockExpression>,
46    },
47    Constant {
48        name: Ident,
49        typ: Option<UnresolvedType>,
50    },
51    Type {
52        name: Ident,
53        bounds: Vec<TraitBound>,
54    },
55}

86/// Represents a simple trait constraint such as `where Foo: TraitY<U, V>`
87/// Complex trait constraints such as `where Foo: Display + TraitX + TraitY<U, V>` are converted
88/// in the parser to a series of simple constraints:
89///   `Foo: Display`
90///   `Foo: TraitX`
91///   `Foo: TraitY<U, V>`
92#[derive(Clone, Debug, PartialEq, Eq)]
93pub struct UnresolvedTraitConstraint {
94    pub typ: UnresolvedType,
95    pub trait_bound: TraitBound,
96}

118impl Display for TypeImpl {
119    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
120        let generics = vecmap(&self.generics, |generic| generic.to_string());
121        let generics =
122            if generics.is_empty() { "".into() } else { format!("<{}>", generics.join(", ")) };
123
124        writeln!(f, "impl{} {} {{", generics, self.object_type)?;
125
126        for (method, _) in self.methods.iter() {
127            let method = method.to_string();
128            for line in method.lines() {
129                writeln!(f, "    {line}")?;
130            }
131        }
132
133        write!(f, "}}")
134    }
135}

138impl Display for NoirTrait {
139    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
140        let generics = vecmap(&self.generics, |generic| generic.to_string());
141        let generics =
142            if generics.is_empty() { "".into() } else { format!("<{}>", generics.join(", ")) };
143
144        write!(f, "trait {}{}", self.name, generics)?;
145
146        if self.is_alias {
147            let bounds = vecmap(&self.bounds, |bound| bound.to_string()).join(" + ");
148            return write!(f, " = {bounds};");
149        }
150
151        if !self.bounds.is_empty() {
152            let bounds = vecmap(&self.bounds, |bound| bound.to_string()).join(" + ");
153            write!(f, ": {bounds}")?;
154        }
155
156        let where_clause = vecmap(&self.where_clause, ToString::to_string);
157        if !where_clause.is_empty() {
158            write!(f, " where {}", where_clause.join(", "))?;
159        };
160
161        writeln!(f, " {{")?;
162
163        for item in self.items.iter() {
164            let item = item.to_string();
165            for line in item.lines() {
166                writeln!(f, "    {line}")?;
167            }
168        }
169
170        write!(f, "}}")
171    }
172}

174impl Display for TraitItem {
175    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
176        match self {
177            TraitItem::Function {
178                name,
179                generics,
180                parameters,
181                return_type,
182                where_clause,
183                body,
184                is_unconstrained,
185                visibility,
186                is_comptime,
187            } => {
188                let generics = vecmap(generics, |generic| generic.to_string());
189                let parameters = vecmap(parameters, |(name, typ)| format!("{name}: {typ}"));
190                let where_clause = vecmap(where_clause, ToString::to_string);
191
192                let generics = generics.join(", ");
193                let parameters = parameters.join(", ");
194                let where_clause = where_clause.join(", ");
195
196                let unconstrained = if *is_unconstrained { "unconstrained " } else { "" };
197                let visibility = if *visibility == ItemVisibility::Private {
198                    "".to_string()
199                } else {
200                    format!("{visibility} ")
201                };
202                let is_comptime = if *is_comptime { "comptime " } else { "" };
203
204                write!(
205                    f,
206                    "{unconstrained}{visibility}{is_comptime}fn {name}<{generics}>({parameters}) -> {return_type} where {where_clause}"
207                )?;
208
209                if let Some(body) = body { write!(f, "{body}") } else { write!(f, ";") }
210            }
211            TraitItem::Constant { name, typ } => {
212                if let Some(typ) = typ {
213                    write!(f, "let {name}: {typ};")
214                } else {
215                    write!(f, "let {name};")
216                }
217            }
218            TraitItem::Type { name, bounds } => {
219                if bounds.is_empty() {
220                    write!(f, "type {name};")
221                } else {
222                    let bounds = vecmap(bounds, |bound| bound.to_string()).join(" + ");
223                    write!(f, "type {name}: {bounds};")
224                }
225            }
226        }
227    }
228}

242impl Display for NoirTraitImpl {
243    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
244        // Synthetic NoirTraitImpl's don't get printed
245        if self.is_synthetic {
246            return Ok(());
247        }
248
249        write!(f, "impl")?;
250        if !self.impl_generics.is_empty() {
251            write!(
252                f,
253                "<{}>",
254                self.impl_generics.iter().map(ToString::to_string).collect::<Vec<_>>().join(", ")
255            )?;
256        }
257
258        write!(f, " {} for {}", self.r#trait, self.object_type)?;
259        if !self.where_clause.is_empty() {
260            write!(
261                f,
262                " where {}",
263                self.where_clause.iter().map(ToString::to_string).collect::<Vec<_>>().join(", ")
264            )?;
265        }
266        writeln!(f, "{{")?;
267
268        for item in self.items.iter() {
269            let item = item.to_string();
270            for line in item.lines() {
271                writeln!(f, "    {line}")?;
272            }
273        }
274
275        write!(f, "}}")
276    }
277}

286    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
287        match self {
288            TraitImplItemKind::Function(function) => function.fmt(f),
289            TraitImplItemKind::Type { name, alias } => {

292                } else {
293                    write!(f, "type {name};")
294                }
295            }
296            TraitImplItemKind::Constant(name, typ, value) => {
297                if let Some(typ) = typ {
298                    write!(f, "let {name}: {typ} = {value};")
299                } else {
300                    write!(f, "let {name} = {value};")
301                }
302            }
303        }
304    }

307/// Does both `desugar_generic_trait_bounds` and `reorder_where_clause`.
308pub(crate) fn desugar_generic_trait_bounds_and_reorder_where_clause(
309    generics: &mut Vec<UnresolvedGeneric>,
310    where_clause: &mut Vec<UnresolvedTraitConstraint>,
311) {
312    desugar_generic_trait_bounds(generics, where_clause);
313    reorder_where_clause(where_clause);
314}

316/// Moves trait bounds from generics into where clauses. For example:
317///
318/// ```noir
319/// fn foo<T: Trait>(x: T) -> T {}
320/// ```
321///
322/// becomes:
323///
324/// ```noir
325/// fn foo<T>(x: T) -> T where T: Trait {}
326/// ```
327fn desugar_generic_trait_bounds(
328    generics: &mut Vec<UnresolvedGeneric>,
329    where_clause: &mut Vec<UnresolvedTraitConstraint>,
330) {
331    for generic in generics {
332        let UnresolvedGeneric::Variable(IdentOrQuotedType::Ident(ident), trait_bounds) = generic
333        else {
334            continue;
335        };
336
337        if trait_bounds.is_empty() {
338            continue;
339        }
340
341        for trait_bound in std::mem::take(trait_bounds) {
342            let path = Path::from_ident(ident.clone());
343            let typ = UnresolvedTypeData::Named(path, GenericTypeArgs::default(), true);
344            let typ = UnresolvedType { typ, location: ident.location() };
345            let trait_constraint = UnresolvedTraitConstraint { typ, trait_bound };
346            where_clause.push(trait_constraint);
347        }
348    }
349}

351/// Reorders a where clause in-place so that simpler constraints come before more complex ones.
352/// The resulting where clause will have constraints in this order:
353/// 1. Paths without generics
354/// 2. Paths with generics
355/// 3. Everything else
356fn reorder_where_clause(where_clause: &mut Vec<UnresolvedTraitConstraint>) {
357    let mut paths_without_generics = Vec::new();
358    let mut paths_with_generics = Vec::new();
359    let mut others = Vec::new();
360
361    for clause in std::mem::take(where_clause) {
362        if let UnresolvedTypeData::Named(_, generics, _) = &clause.typ.typ {
363            if generics.is_empty() {
364                paths_without_generics.push(clause);
365            } else {
366                paths_with_generics.push(clause);
367            }
368        } else {
369            others.push(clause);
370        }
371    }
372    where_clause.extend(paths_without_generics);
373    where_clause.extend(paths_with_generics);
374    where_clause.extend(others);
