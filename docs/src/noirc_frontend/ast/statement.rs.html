43/// Ast node for statements in noir. Statements are always within a block { }
44/// of some kind and are terminated via a Semicolon, except if the statement
45/// ends in a block, such as a Statement::Expression containing an if expression.
46#[derive(Debug, PartialEq, Eq, Clone)]
47pub enum StatementKind {
48    Let(LetStatement),
49    Expression(Expression),
50    Assign(AssignStatement),
51    AssignOp(AssignOpStatement),
52    For(ForLoopStatement),
53    Loop(LoopStatement),
54    While(WhileStatement),
55    Break,
56    Continue,
57    /// This statement should be executed at compile-time
58    Comptime(Box<Statement>),
59    /// This is an expression with a trailing semi-colon
60    Semi(Expression),
61    /// This is an interned StatementKind during comptime code.
62    /// The actual StatementKind can be retrieved with a NodeInterner.
63    Interned(InternedStatementKind),
64    /// This statement is the result of a recovered parse error.
65    /// To avoid issuing multiple errors in later steps, it should
66    /// be skipped in any future analysis if possible.
67    Error,
68}

70impl Statement {
71    pub fn add_semicolon(
72        mut self,
73        semi: Option<Token>,
74        location: Location,
75        last_statement_in_block: bool,
76        emit_error: &mut dyn FnMut(ParserError),
77    ) -> Self {
78        self.kind = self.kind.add_semicolon(semi, location, last_statement_in_block, emit_error);
79        self
80    }
81
82    /// Returns the innermost location that gives this statement its type.
83    pub fn type_location(&self) -> Location {
84        match &self.kind {
85            StatementKind::Expression(expression) => expression.type_location(),
86            StatementKind::Comptime(statement) => statement.type_location(),
87            StatementKind::Let(..)
88            | StatementKind::Assign(..)
89            | StatementKind::AssignOp(..)
90            | StatementKind::For(..)
91            | StatementKind::Loop(..)
92            | StatementKind::While(..)
93            | StatementKind::Break
94            | StatementKind::Continue
95            | StatementKind::Semi(..)
96            | StatementKind::Interned(..)
97            | StatementKind::Error => self.location,
98        }
99    }
100}

102impl StatementKind {
103    pub fn add_semicolon(
104        self,
105        semi: Option<Token>,
106        location: Location,
107        last_statement_in_block: bool,
108        emit_error: &mut dyn FnMut(ParserError),
109    ) -> Self {
110        let missing_semicolon =
111            ParserError::with_reason(ParserErrorReason::MissingSeparatingSemi, location);
112
113        match self {
114            StatementKind::Let(_) => {
115                // To match rust, a let statement always requires a semicolon, even at the end of a block
116                if semi.is_none() {
117                    let reason = ParserErrorReason::MissingSemicolonAfterLet;
118                    emit_error(ParserError::with_reason(reason, location));
119                }
120                self
121            }
122            StatementKind::Assign(_)
123            | StatementKind::AssignOp(_)
124            | StatementKind::Semi(_)
125            | StatementKind::Break
126            | StatementKind::Continue
127            | StatementKind::Error => {
128                // These statements can omit the semicolon if they are the last statement in a block
129                if !last_statement_in_block && semi.is_none() {
130                    emit_error(missing_semicolon);
131                }
132                self
133            }
134            StatementKind::Comptime(mut statement) => {
135                // `comptime { expr };` => `Semi(comptime { expr })`
136                if semi.is_some()
137                    && let StatementKind::Expression(Expression {
138                        kind: ExpressionKind::Block(block),
139                        ..
140                    }) = statement.kind
141                {
142                    let comptime_expr = Expression {
143                        kind: ExpressionKind::Comptime(block, statement.location),
144                        location,
145                    };
146                    return StatementKind::Semi(comptime_expr);
147                }
148
149                *statement =
150                    statement.add_semicolon(semi, location, last_statement_in_block, emit_error);
151                StatementKind::Comptime(statement)
152            }
153            // A semicolon on a for loop, loop or while is optional and does nothing
154            StatementKind::For(_) | StatementKind::Loop(..) | StatementKind::While(..) => self,
155
156            // No semicolon needed for a resolved statement
157            StatementKind::Interned(_) => self,
158
159            StatementKind::Expression(expr) => {
160                match (&expr.kind, semi, last_statement_in_block) {
161                    // Semicolons are optional for these expressions
162                    (ExpressionKind::Block(_), semi, _)
163                    | (ExpressionKind::Unsafe(..), semi, _)
164                    | (ExpressionKind::Interned(..), semi, _)
165                    | (ExpressionKind::InternedStatement(..), semi, _)
166                    | (ExpressionKind::Match(..), semi, _)
167                    | (ExpressionKind::If(_), semi, _) => {
168                        if semi.is_some() {
169                            StatementKind::Semi(expr)
170                        } else {
171                            StatementKind::Expression(expr)
172                        }
173                    }
174                    (_, None, false) => {
175                        emit_error(missing_semicolon);
176                        StatementKind::Expression(expr)
177                    }
178                    (_, Some(_), _) => StatementKind::Semi(expr),
179                    (_, None, true) => StatementKind::Expression(expr),
180                }
181            }
182        }
183    }
184}

348#[derive(Debug, PartialEq, Eq, Clone)]
349pub struct UseTree {
350    pub prefix: Path,
351    pub kind: UseTreeKind,
352    pub location: Location,
353}
354
355impl Display for UseTree {
356    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
357        write!(f, "{}", self.prefix)?;
358
359        if !self.prefix.segments.is_empty() {
360            write!(f, "::")?;
361        }
362
363        match &self.kind {
364            UseTreeKind::Path(name, alias) => {
365                write!(f, "{name}")?;
366
367                if let Some(alias) = alias {
368                    write!(f, " as {alias}")?;
369                }
370
371                Ok(())
372            }
373            UseTreeKind::List(trees) => {
374                write!(f, "{{")?;
375                let tree = vecmap(trees, ToString::to_string).join(", ");
376                write!(f, "{tree}}}")
377            }
378        }
379    }
380}
381
382#[derive(Debug, PartialEq, Eq, Clone)]
383pub enum UseTreeKind {
384    Path(Ident, Option<Ident>),
385    List(Vec<UseTree>),
386}
387
388impl UseTree {
389    pub fn desugar(self, root: Option<Path>, visibility: ItemVisibility) -> Vec<ImportStatement> {
390        let prefix = if let Some(mut root) = root {
391            root.segments.extend(self.prefix.segments);
392            root
393        } else {
394            self.prefix
395        };
396
397        match self.kind {
398            UseTreeKind::Path(name, alias) => {
399                // Desugar `use foo::{self}` to `use foo`
400                let path = if name.as_str() == "self" { prefix } else { prefix.join(name) };
401                vec![ImportStatement { visibility, path, alias }]
402            }
403            UseTreeKind::List(trees) => {
404                let trees = trees.into_iter();
405                trees.flat_map(|tree| tree.desugar(Some(prefix.clone()), visibility)).collect()
406            }
407        }
408    }
409}

447pub struct Path {
448    pub segments: Vec<PathSegment>,
449    pub kind: PathKind,
450    pub location: Location,
451    /// The location of `kind` (this is the same as `location` for plain kinds)
452    pub kind_location: Location,

455impl Path {
456    pub fn plain(segments: Vec<PathSegment>, location: Location) -> Self {
457        Self { segments, location, kind: PathKind::Plain, kind_location: location }
458    }
459
460    pub fn pop(&mut self) -> PathSegment {
461        self.segments.pop().unwrap()
462    }
463
464    pub fn join(mut self, ident: Ident) -> Path {
465        self.segments.push(PathSegment::from(ident));
466        self
467    }

469    /// Construct a [PathKind::Plain] from a single identifier name.

472        Path::from_ident(segment)
473    }
474
475    /// Construct a [PathKind::Plain] from a single [Ident].
476    pub fn from_ident(name: Ident) -> Path {
477        let location = name.location();
478        Path::plain(vec![PathSegment::from(name)], location)
479    }
480
481    pub fn span(&self) -> Span {
482        self.location.span
483    }
484
485    pub fn last_segment(&self) -> PathSegment {
486        assert!(!self.segments.is_empty());
487        self.segments.last().unwrap().clone()
488    }
489
490    pub fn last_ident(&self) -> Ident {
491        self.last_segment().ident
492    }
493
494    pub fn first_name(&self) -> Option<&str> {
495        self.segments.first().map(|segment| segment.ident.as_str())
496    }
497
498    pub fn last_name(&self) -> &str {
499        assert!(!self.segments.is_empty());
500        self.segments.last().unwrap().ident.as_str()
501    }
502
503    pub fn is_ident(&self) -> bool {
504        self.kind == PathKind::Plain
505            && self.segments.len() == 1
506            && self.segments.first().unwrap().generics.is_none()
507    }
508
509    pub fn no_generic(&self) -> bool {
510        for segment in &self.segments {
511            if segment.generics.is_some() {
512                return false;
513            }
514        }
515
516        true
517    }
518
519    pub fn as_ident(&self) -> Option<&Ident> {
520        if !self.is_ident() {
521            return None;
522        }
523        self.segments.first().map(|segment| &segment.ident)
524    }
525
526    pub fn is_empty(&self) -> bool {
527        self.segments.is_empty() && self.kind == PathKind::Plain
528    }
529}

534    pub generics: Option<Vec<UnresolvedType>>,

538impl PathSegment {
539    /// Returns the span where turbofish happen. For example:
540    ///
541    /// ```noir
542    ///    foo::<T>
543    ///       ~^^^^
544    /// ```
545    ///
546    /// Returns an empty span at the end of `foo` if there's no turbofish.
547    pub fn turbofish_span(&self) -> Span {
548        if self.ident.location().file == self.location.file {
549            Span::from(self.ident.span().end()..self.location.span.end())
550        } else {
551            self.location.span
552        }
553    }
554
555    pub fn turbofish_location(&self) -> Location {
556        Location::new(self.turbofish_span(), self.location.file)
557    }
558}

648#[derive(Debug, PartialEq, Eq, Clone)]
649pub enum Pattern {
650    Identifier(Ident),
651    Mutable(Box<Pattern>, Location, /*is_synthesized*/ bool),
652    Tuple(Vec<Pattern>, Location),
653    Struct(Path, Vec<(Ident, Pattern)>, Location),
654    Parenthesized(Box<Pattern>, Location),
655    Interned(InternedPattern, Location),
656}
657
658impl Pattern {
659    pub fn location(&self) -> Location {
660        match self {
661            Pattern::Identifier(ident) => ident.location(),
662            Pattern::Mutable(_, location, _)
663            | Pattern::Tuple(_, location)
664            | Pattern::Struct(_, _, location)
665            | Pattern::Parenthesized(_, location)
666            | Pattern::Interned(_, location) => *location,
667        }
668    }
669
670    pub fn span(&self) -> Span {
671        self.location().span
672    }
673
674    pub fn name_ident(&self) -> &Ident {
675        match self {
676            Pattern::Identifier(name_ident) => name_ident,
677            Pattern::Mutable(pattern, ..) => pattern.name_ident(),
678            _ => panic!("Only the Identifier or Mutable patterns can return a name"),
679        }
680    }
681
682    pub(crate) fn try_as_expression(&self, interner: &NodeInterner) -> Option<Expression> {
683        match self {
684            Pattern::Identifier(ident) => Some(Expression {
685                kind: ExpressionKind::Variable(Path::from_ident(ident.clone())),
686                location: ident.location(),
687            }),
688            Pattern::Mutable(_, _, _) => None,
689            Pattern::Tuple(patterns, location) => {
690                let mut expressions = Vec::new();
691                for pattern in patterns {
692                    expressions.push(pattern.try_as_expression(interner)?);
693                }
694                Some(Expression { kind: ExpressionKind::Tuple(expressions), location: *location })
695            }
696            Pattern::Struct(path, patterns, location) => {
697                let mut fields = Vec::new();
698                for (field, pattern) in patterns {
699                    let expression = pattern.try_as_expression(interner)?;
700                    fields.push((field.clone(), expression));
701                }
702                Some(Expression {
703                    kind: ExpressionKind::Constructor(Box::new(ConstructorExpression {
704                        typ: UnresolvedType::from_path(path.clone()),
705                        fields,
706                    })),
707                    location: *location,
708                })
709            }
710            Pattern::Parenthesized(pattern, _) => pattern.try_as_expression(interner),
711            Pattern::Interned(id, _) => interner.get_pattern(*id).try_as_expression(interner),
712        }
713    }
714}

716impl LValue {
717    pub fn as_expression(&self) -> Expression {
718        let kind = match self {
719            LValue::Path(path) => ExpressionKind::Variable(path.clone()),
720            LValue::MemberAccess { object, field_name, location: _ } => {
721                ExpressionKind::MemberAccess(Box::new(MemberAccessExpression {
722                    lhs: object.as_expression(),
723                    rhs: field_name.clone(),
724                }))
725            }
726            LValue::Index { array, index, location: _ } => {
727                ExpressionKind::Index(Box::new(IndexExpression {
728                    collection: array.as_expression(),
729                    index: index.clone(),
730                }))
731            }
732            LValue::Dereference(lvalue, _span) => {
733                ExpressionKind::Prefix(Box::new(crate::ast::PrefixExpression {
734                    operator: crate::ast::UnaryOp::Dereference { implicitly_added: false },
735                    rhs: lvalue.as_expression(),
736                }))
737            }
738            LValue::Interned(id, _) => ExpressionKind::Interned(*id),
739        };
740        Expression::new(kind, self.location())
741    }
742
743    pub fn from_expression(expr: Expression) -> Option<LValue> {
744        LValue::from_expression_kind(expr.kind, expr.location)
745    }
746
747    pub fn from_expression_kind(expr: ExpressionKind, location: Location) -> Option<LValue> {
748        match expr {
749            ExpressionKind::Variable(path) => Some(LValue::Path(path)),
750            ExpressionKind::MemberAccess(member_access) => Some(LValue::MemberAccess {
751                object: Box::new(LValue::from_expression(member_access.lhs)?),
752                field_name: member_access.rhs,
753                location,
754            }),
755            ExpressionKind::Index(index) => Some(LValue::Index {
756                array: Box::new(LValue::from_expression(index.collection)?),
757                index: index.index,
758                location,
759            }),
760            ExpressionKind::Prefix(prefix) => {
761                if matches!(
762                    prefix.operator,
763                    crate::ast::UnaryOp::Dereference { implicitly_added: false }
764                ) {
765                    Some(LValue::Dereference(
766                        Box::new(LValue::from_expression(prefix.rhs)?),
767                        location,
768                    ))
769                } else {
770                    None
771                }
772            }
773            ExpressionKind::Parenthesized(expr) => LValue::from_expression(*expr),
774            ExpressionKind::Interned(id) => Some(LValue::Interned(id, location)),
775            _ => None,
776        }
777    }
778
779    pub fn location(&self) -> Location {
780        match self {
781            LValue::Path(path) => path.location,
782            LValue::MemberAccess { location, .. }
783            | LValue::Index { location, .. }
784            | LValue::Dereference(_, location)
785            | LValue::Interned(_, location) => *location,
786        }
787    }
788
789    pub fn span(&self) -> Span {
790        self.location().span
791    }
792}

801#[derive(Debug, PartialEq, Eq, Clone)]
802pub enum ForRange {
803    Range(ForBounds),
804    Array(Expression),
805}
806
807impl ForRange {
808    /// Create a half-open range, bounded inclusively below and exclusively above.
809    pub fn range(start: Expression, end: Expression, inclusive: bool) -> Self {
810        Self::Range(ForBounds { start, end, inclusive })
811    }
812
813    /// Create a 'for' expression taking care of desugaring a 'for e in array' loop
814    /// into the following if needed:
815    ///
816    /// ```text
817    /// {
818    ///     let fresh1 = array;
819    ///     for fresh2 in 0 .. std::array::len(fresh1) {
820    ///         let elem = fresh1[fresh2];
821    ///         ...
822    ///     }
823    /// }
824    /// ````
825    pub(crate) fn into_for(
826        self,
827        identifier: Ident,
828        block: Expression,
829        for_loop_location: Location,
830    ) -> Statement {
831        // Counter used to generate unique names when desugaring
832        // code in the parser requires the creation of fresh variables.
833        let mut unique_name_counter: u32 = 0;
834
835        match self {
836            ForRange::Range(..) => {
837                unreachable!()
838            }
839            ForRange::Array(array) => {
840                let array_location = array.location;
841                let start_range = ExpressionKind::integer(FieldElement::zero(), None);
842                let start_range = Expression::new(start_range, array_location);
843
844                let next_unique_id = unique_name_counter;
845                unique_name_counter += 1;
846                let array_name = format!("$i{next_unique_id}");
847                let array_location = array.location;
848                let array_ident = Ident::new(array_name, array_location);
849
850                // let fresh1 = array;
851                let let_array = Statement {
852                    kind: StatementKind::new_let(
853                        Pattern::Identifier(array_ident.clone()),
854                        None,
855                        array,
856                        vec![],
857                    ),
858                    location: array_location,
859                };
860
861                // array.len()
862                let segments = vec![PathSegment::from(array_ident)];
863                let array_ident = ExpressionKind::Variable(Path::plain(segments, array_location));
864
865                let end_range = ExpressionKind::MethodCall(Box::new(MethodCallExpression {
866                    object: Expression::new(array_ident.clone(), array_location),
867                    method_name: Ident::new("len".to_string(), array_location),
868                    generics: None,
869                    is_macro_call: false,
870                    arguments: vec![],
871                }));
872                let end_range = Expression::new(end_range, array_location);
873
874                let next_unique_id = unique_name_counter;
875                let index_name = format!("$i{next_unique_id}");
876                let fresh_identifier = Ident::new(index_name.clone(), array_location);
877
878                // array[i]
879                let segments = vec![PathSegment::from(Ident::new(index_name, array_location))];
880                let index_ident = ExpressionKind::Variable(Path::plain(segments, array_location));
881
882                let loop_element = ExpressionKind::Index(Box::new(IndexExpression {
883                    collection: Expression::new(array_ident, array_location),
884                    index: Expression::new(index_ident, array_location),
885                }));
886
887                // let elem = array[i];
888                let let_elem = Statement {
889                    kind: StatementKind::new_let(
890                        Pattern::Identifier(identifier),
891                        None,
892                        Expression::new(loop_element, array_location),
893                        vec![],
894                    ),
895                    location: array_location,
896                };
897
898                let block_location = block.location;
899                let new_block = BlockExpression {
900                    statements: vec![
901                        let_elem,
902                        Statement {
903                            kind: StatementKind::Expression(block),
904                            location: block_location,
905                        },
906                    ],
907                };
908                let new_block = Expression::new(ExpressionKind::Block(new_block), block_location);
909                let for_loop = Statement {
910                    kind: StatementKind::For(ForLoopStatement {
911                        identifier: fresh_identifier,
912                        range: ForRange::range(start_range, end_range, false),
913                        block: new_block,
914                        location: for_loop_location,
915                    }),
916                    location: for_loop_location,
917                };
918
919                let block = ExpressionKind::Block(BlockExpression {
920                    statements: vec![let_array, for_loop],
921                });
922                Statement {
923                    kind: StatementKind::Expression(Expression::new(block, for_loop_location)),
924                    location: for_loop_location,
925                }
926            }
927        }
928    }
929}

952impl Display for StatementKind {
953    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
954        match self {
955            StatementKind::Let(let_statement) => let_statement.fmt(f),
956            StatementKind::Expression(expression) => expression.fmt(f),
957            StatementKind::Assign(assign) => assign.fmt(f),
958            StatementKind::AssignOp(assign_op) => assign_op.fmt(f),
959            StatementKind::For(for_loop) => for_loop.fmt(f),
960            StatementKind::Loop(loop_) => write!(f, "loop {}", loop_.body),
961            StatementKind::While(while_) => {
962                write!(f, "while {} {}", while_.condition, while_.body)
963            }
964            StatementKind::Break => write!(f, "break"),
965            StatementKind::Continue => write!(f, "continue"),
966            StatementKind::Comptime(statement) => write!(f, "comptime {}", statement.kind),
967            StatementKind::Semi(semi) => write!(f, "{semi};"),
968            StatementKind::Interned(_) => write!(f, "(resolved);"),
969            StatementKind::Error => write!(f, "Error"),
970        }
971    }
972}

1060impl Display for Pattern {
1061    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1062        match self {
1063            Pattern::Identifier(name) => name.fmt(f),
1064            Pattern::Mutable(name, _, _) => write!(f, "mut {name}"),
1065            Pattern::Tuple(fields, _) => {
1066                let fields = vecmap(fields, ToString::to_string);
1067                if fields.len() == 1 {
1068                    write!(f, "({},)", fields[0])
1069                } else {
1070                    write!(f, "({})", fields.join(", "))
1071                }
1072            }
1073            Pattern::Struct(typename, fields, _) => {
1074                let fields = vecmap(fields, |(name, pattern)| format!("{name}: {pattern}"));
1075                write!(f, "{} {{ {} }}", typename, fields.join(", "))
1076            }
1077            Pattern::Parenthesized(pattern, _) => {
1078                write!(f, "({pattern})")
1079            }
1080            Pattern::Interned(_, _) => {
1081                write!(f, "?Interned")
1082            }
1083        }
1084    }

1087impl Display for ForLoopStatement {
1088    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1089        let range = match &self.range {
1090            ForRange::Range(bounds) => {
1091                format!(
1092                    "{}{}{}",
1093                    bounds.start,
1094                    if bounds.inclusive { "..=" } else { ".." },
1095                    bounds.end
1096                )
1097            }
1098            ForRange::Array(expr) => expr.to_string(),
1099        };
1100
1101        write!(f, "for {} in {range} {}", self.identifier, self.block)
1102    }
