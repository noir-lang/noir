20/// The monomorphized AST is expression-based, all statements are also
21/// folded into this expression enum. Compared to the HIR, the monomorphized
22/// AST has several differences:
23/// - It is self-contained and does not require referencing an external interner
24/// - All Types used within are monomorphized and no longer contain any generic types
25/// - All Patterns are expanded into multiple variables. This means each definition now
26///   defines only 1 variable `let a = 1;`, and any that previously defined multiple,
27///   e.g. `let (a, b) = (1, 2)` have been split up: `let tmp = (1, 2); let a = tmp.0; let b = tmp.1;`.
28///   This also affects function parameters: `fn foo((a, b): (i32, i32)` => `fn foo(a: i32, b: i32)`.
29/// - All structs are replaced with tuples
30#[derive(Debug, Clone, Hash)]
31pub enum Expression {
32    Ident(Ident),
33    Literal(Literal),
34    Block(Vec<Expression>),
35    Unary(Unary),
36    Binary(Binary),
37    Index(Index),
38    Cast(Cast),
39    For(For),
40    Loop(Box<Expression>),
41    While(While),
42    If(If),
43    Match(Match),
44    Tuple(Vec<Expression>),
45    ExtractTupleField(Box<Expression>, usize),
46    Call(Call),
47    Let(Let),
48    Constrain(Box<Expression>, Location, Option<Box<(Expression, HirType)>>),
49    Assign(Assign),
50    Semi(Box<Expression>),
51    Clone(Box<Expression>),
52    Drop(Box<Expression>),
53    Break,
54    Continue,
55}

57impl Expression {
58    pub fn is_array_or_vector_literal(&self) -> bool {
59        matches!(self, Expression::Literal(Literal::Array(_) | Literal::Vector(_)))
60    }

62    /// The return type of an expression, if it has an obvious one.
63    pub fn return_type(&self) -> Option<Cow<Type>> {
64        fn borrowed(typ: &Type) -> Option<Cow<Type>> {
65            Some(Cow::Borrowed(typ))
66        }
67        let owned = |typ: Type| Some(Cow::Owned(typ));
68
69        match self {
70            Expression::Ident(ident) => borrowed(&ident.typ),
71            Expression::Literal(literal) => match literal {
72                Literal::Array(literal) | Literal::Vector(literal) => borrowed(&literal.typ),
73                Literal::Repeated { typ, .. } => borrowed(typ),
74                Literal::Integer(_, typ, _) => borrowed(typ),
75                Literal::Bool(_) => borrowed(&Type::Bool),
76                Literal::Unit => borrowed(&Type::Unit),
77                Literal::Str(s) => owned(Type::String(s.len() as u32)),
78                Literal::FmtStr(_, size, expr) => {
79                    let typ = expr.return_type()?;
80                    owned(Type::FmtString(*size as u32, Box::new(typ.into_owned())))
81                }
82            },
83            Expression::Block(xs) => {
84                let x = xs.last()?;
85                x.return_type()
86            }
87            Expression::Unary(unary) => borrowed(&unary.result_type),
88            Expression::Binary(binary) => {
89                if binary.operator.is_comparator() {
90                    borrowed(&Type::Bool)
91                } else {
92                    binary.lhs.return_type()
93                }
94            }
95            Expression::Index(index) => borrowed(&index.element_type),
96            Expression::Cast(cast) => borrowed(&cast.r#type),
97            Expression::If(if_) => borrowed(&if_.typ),
98            Expression::ExtractTupleField(x, idx) => match x.as_ref() {
99                Expression::Tuple(xs) => {
100                    assert!(xs.len() > *idx, "index out of bounds in tuple return type");
101                    xs[*idx].return_type()
102                }
103                x => {
104                    let typ = x.return_type()?;
105                    let Type::Tuple(types) = typ.as_ref() else {
106                        unreachable!("unexpected type for tuple field extraction: {typ}");
107                    };
108                    assert!(types.len() > *idx, "index out of bounds in tuple return type");
109                    owned(types[*idx].clone())
110                }
111            },
112            Expression::Clone(x) => x.return_type(),
113            Expression::Call(call) => borrowed(&call.return_type),
114            Expression::Match(m) => borrowed(&m.typ),
115
116            Expression::Tuple(xs) => {
117                let types = xs
118                    .iter()
119                    .filter_map(|x| x.return_type())
120                    .map(|t| t.into_owned())
121                    .collect::<Vec<_>>();
122                if types.len() != xs.len() {
123                    return None;
124                }
125                owned(Type::Tuple(types))
126            }
127
128            Expression::For(_)
129            | Expression::Loop(_)
130            | Expression::While(_)
131            | Expression::Let(_)
132            | Expression::Constrain(_, _, _)
133            | Expression::Assign(_)
134            | Expression::Semi(_)
135            | Expression::Drop(_)
136            | Expression::Break
137            | Expression::Continue => None,
138        }
139    }
140
141    /// Check if the expression will need to have its type deduced from a literal,
142    /// which could be ambiguous.
143    ///
144    /// For example:
145    /// ```ignore
146    /// let a = 1;
147    /// let b = if (a > 0) { 2 } else { 3 };
148    /// ```
149    pub fn needs_type_inference_from_literal(&self) -> bool {
150        match self {
151            Expression::Literal(_) => true,
152
153            Expression::Block(expressions) => expressions
154                .last()
155                .map(|x| x.needs_type_inference_from_literal())
156                .unwrap_or_default(),
157
158            Expression::Unary(unary) => unary.rhs.needs_type_inference_from_literal(),
159
160            Expression::Binary(binary) => {
161                if binary.operator.is_comparator() {
162                    false
163                } else {
164                    binary.lhs.needs_type_inference_from_literal()
165                }
166            }
167            Expression::If(if_) => {
168                if_.consequence.needs_type_inference_from_literal()
169                    && if_
170                        .alternative
171                        .as_ref()
172                        .map(|x| x.needs_type_inference_from_literal())
173                        .unwrap_or_default()
174            }
175            Expression::Match(m) => {
176                m.cases.iter().all(|c| c.branch.needs_type_inference_from_literal())
177                    && m.default_case.as_ref().is_none_or(|x| x.needs_type_inference_from_literal())
178            }
179
180            Expression::Tuple(xs) => xs.iter().any(|x| x.needs_type_inference_from_literal()),
181
182            Expression::ExtractTupleField(x, _) => x.needs_type_inference_from_literal(),
183
184            // The following expressions either carry an obvious type, or return nothing.
185            Expression::Ident(_)
186            | Expression::Call(_)
187            | Expression::Index(_)
188            | Expression::Cast(_)
189            | Expression::For(_)
190            | Expression::Loop(_)
191            | Expression::While(_)
192            | Expression::Let(_)
193            | Expression::Constrain(_, _, _)
194            | Expression::Assign(_)
195            | Expression::Semi(_)
196            | Expression::Clone(_)
197            | Expression::Drop(_)
198            | Expression::Break
199            | Expression::Continue => false,
200        }
201    }
202}

429/// Represents how an Acir function should be inlined.
430/// This type is only relevant for ACIR functions as we do not inline any Brillig functions
431#[derive(
432    Default, Clone, Copy, PartialEq, Eq, Debug, Hash, Serialize, Deserialize, PartialOrd, Ord,
433)]
434pub enum InlineType {
435    /// The most basic entry point can expect all its functions to be inlined.
436    /// All function calls are expected to be inlined into a single ACIR.
437    #[default]
438    Inline,
439    /// Functions marked as inline always will always be inlined, even in brillig contexts.
440    InlineAlways,
441    /// Functions marked as inline never will never be inlined
442    InlineNever,
443    /// Functions marked as foldable will not be inlined and compiled separately into ACIR
444    Fold,
445    /// Functions marked to have no predicates will not be inlined in the default inlining pass
446    /// and will be separately inlined after the flattening pass.
447    ///
448    /// Flattening and inlining are necessary compiler passes in the ACIR runtime. More specifically,
449    /// flattening is the removal of control flow through predicating side-effectual instructions.
450    /// In some cases, a user may only want predicates applied to the result of a function call rather
451    /// than all of a function's internal execution. To allow this behavior, we can simply inline a function
452    /// after performing flattening (as ultimately in ACIR a non-entry point function will have to be inlined).
453    /// These functions are different from `Fold` as they are expected to be inlined into the program
454    /// entry point before being used in the backend.
455    ///
456    /// This attribute is unsafe and can cause a function whose logic relies on predicates from
457    /// the flattening pass to fail.
458    NoPredicates,
459}
460
461impl From<&Attributes> for InlineType {
462    fn from(attributes: &Attributes) -> Self {
463        attributes.function().map_or(InlineType::default(), |func_attribute| match &func_attribute
464            .kind
465        {
466            FunctionAttributeKind::Fold => InlineType::Fold,
467            FunctionAttributeKind::NoPredicates => InlineType::NoPredicates,
468            FunctionAttributeKind::InlineAlways => InlineType::InlineAlways,
469            FunctionAttributeKind::InlineNever => InlineType::InlineNever,
470            _ => InlineType::default(),
471        })
472    }
473}
474
475impl InlineType {
476    pub fn is_entry_point(&self) -> bool {
477        match self {
478            InlineType::Inline => false,
479            InlineType::InlineAlways => false,
480            InlineType::InlineNever => false,
481            InlineType::Fold => true,
482            InlineType::NoPredicates => false,
483        }
484    }
485
486    /// Produce an `InlineType` which we can use with an unconstrained version of a function.
487    pub fn into_unconstrained(self) -> Self {
488        match self {
489            InlineType::Inline | InlineType::InlineAlways | InlineType::InlineNever => self,
490            InlineType::Fold => {
491                // The #[fold] attribute is about creating separate ACIR circuits for proving,
492                // not relevant in Brillig. Leaving it violates some expectations that each
493                // will become its own entry point.
494                Self::default()
495            }
496            InlineType::NoPredicates => {
497                // The #[no_predicates] are guaranteed to be inlined after flattening,
498                // which is needed for some of the programs even in Brillig, otherwise
499                // some intrinsics can survive until Brillig-gen that weren't supposed to.
500                // We can keep these, or try inlining more aggressively, since we don't
501                // have to wait until after flattening in Brillig, but InlineAlways
502                // resulted in some Brillig bytecode size regressions.
503                self
504            }
505        }
506    }
507}

536/// Compared to hir_def::types::Type, this monomorphized Type has:
537/// - All type variables and generics removed
538/// - Concrete lengths for each array and string
539/// - Several other variants removed (such as Type::Constant)
540/// - All structs replaced with tuples
541#[derive(Debug, PartialEq, Eq, Clone, Hash, PartialOrd, Ord)]
542pub enum Type {
543    Field,
544    Array(/*len:*/ u32, Box<Type>), // Array(4, Field) = [Field; 4]
545    Integer(Signedness, /*bits:*/ IntegerBitSize), // u32 = Integer(unsigned, ThirtyTwo)
546    Bool,
547    String(/*len:*/ u32), // String(4) = str[4]
548    FmtString(/*len:*/ u32, Box<Type>),
549    Unit,
550    Tuple(Vec<Type>),
551    Vector(Box<Type>),
552    Reference(Box<Type>, /*mutable:*/ bool),
553    /// `(args, ret, env, unconstrained)`
554    Function(
555        /*args:*/ Vec<Type>,
556        /*ret:*/ Box<Type>,
557        /*env:*/ Box<Type>,
558        /*unconstrained:*/ bool,
559    ),
560}

562impl Type {
563    pub fn flatten(&self) -> Vec<Type> {
564        match self {
565            Type::Tuple(fields) => fields.iter().flat_map(|field| field.flatten()).collect(),
566            _ => vec![self.clone()],
567        }
568    }
569
570    /// Returns the element type of this array or vector
571    pub fn array_element_type(&self) -> Option<&Type> {
572        match self {
573            Type::Array(_, elem) | Type::Vector(elem) => Some(elem),
574            _ => None,
575        }
576    }
577
578    /// Returns the number of field elements required to represent the type once encoded
579    /// as a parameter to an entry point function.
580    ///
581    /// Panics if the type is not valid as a parameter to main.
582    pub fn entry_point_field_count(&self) -> u32 {
583        match self {
584            Type::Field | Type::Integer(..) | Type::Bool => 1,
585            Type::Array(length, typ) => {
586                let typ = typ.as_ref();
587                length.checked_mul(typ.entry_point_field_count()).expect("Array length overflow")
588            }
589            Type::String(length) => *length,
590            Type::Tuple(fields) => fields.iter().fold(0, |acc, field_typ| {
591                acc.checked_add(field_typ.entry_point_field_count()).expect("Tuple size overflow")
592            }),
593            Type::Function(..)
594            | Type::FmtString(..)
595            | Type::Unit
596            | Type::Vector(_)
597            | Type::Reference(_, _) => {
598                unreachable!("This type cannot exist as a parameter to main: {self:?}")
599            }
600        }
601    }
602}

706impl Display for Type {
707    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
708        match self {
709            Type::Field => write!(f, "Field"),
710            Type::Array(len, elements) => write!(f, "[{elements}; {len}]"),
711            Type::Integer(sign, bits) => match sign {
712                Signedness::Unsigned => write!(f, "u{bits}"),
713                Signedness::Signed => write!(f, "i{bits}"),
714            },
715            Type::Bool => write!(f, "bool"),
716            Type::String(len) => write!(f, "str<{len}>"),
717            Type::FmtString(len, elements) => {
718                write!(f, "fmtstr<{len}, {elements}>")
719            }
720            Type::Unit => write!(f, "()"),
721            Type::Tuple(elements) => {
722                let elements = vecmap(elements, ToString::to_string);
723                if elements.len() == 1 {
724                    write!(f, "({},)", elements[0])
725                } else {
726                    write!(f, "({})", elements.join(", "))
727                }
728            }
729            Type::Function(args, ret, env, unconstrained) => {
730                if *unconstrained {
731                    write!(f, "unconstrained ")?;
732                }
733
734                let args = vecmap(args, ToString::to_string);
735                let closure_env_text = match **env {
736                    Type::Unit => "".to_string(),
737                    _ => format!(" with closure environment {env}"),
738                };
739                write!(f, "fn({}) -> {}{}", args.join(", "), ret, closure_env_text)
740            }
741            Type::Vector(element) => write!(f, "[{element}]"),
742            Type::Reference(element, mutable) if *mutable => write!(f, "&mut {element}"),
743            Type::Reference(element, _mutable) => write!(f, "&{element}"),
744        }
745    }
746}
