10    /// Scans the interner for the item which is located at that [Location]
11    ///
12    /// The [Location] may not necessarily point to the beginning of the item
13    /// so we check if the location's span is contained within the start or end
14    /// of each items [Location]
15    pub fn find_location_index(&self, location: Location) -> Option<impl Into<Index>> {
16        let mut location_candidate: Option<(&Index, &Location)> = None;
17
18        // Note: we can modify this in the future to not do a linear
19        // scan by storing a separate map of the spans or by sorting the locations.
20        for (index, interned_location) in self.id_to_location.iter() {
21            if interned_location.contains(&location) {
22                if let Some(current_location) = location_candidate {
23                    if interned_location.span.is_smaller(&current_location.1.span) {
24                        location_candidate = Some((index, interned_location));
25                    }
26                } else {
27                    location_candidate = Some((index, interned_location));
28                }
29            }
30        }
31        location_candidate.map(|(index, _location)| *index)
32    }

34    /// Returns the [Type] of the expression that exists at the given location.
35    pub fn type_at_location(&self, location: Location) -> Option<&Type> {
36        // This is similar to `find_location_index` except that we skip indexes for which there is no type
37        let mut location_candidate: Option<(&Index, &Location, &Type)> = None;
38
39        for (index, interned_location) in self.id_to_location.iter() {
40            if interned_location.contains(&location)
41                && let Some(typ) = self.try_id_type(*index)
42            {
43                if let Some(current_location) = location_candidate {
44                    if interned_location.span.is_smaller(&current_location.1.span) {
45                        location_candidate = Some((index, interned_location, typ));
46                    }
47                } else {
48                    location_candidate = Some((index, interned_location, typ));
49                }
50            }
51        }
52        location_candidate.map(|(_index, _location, typ)| typ)
53    }

55    /// Returns the [Location] of the definition of the given Ident found at `location`.
56    /// Returns [None] when definition is not found.
57    pub fn get_definition_location_from(
58        &self,
59        location: Location,
60        return_type_location_instead: bool,
61    ) -> Option<Location> {
62        // First try to find the location in the reference graph
63        self.find_referenced_location(location).or_else(|| {
64            // Otherwise fallback to the location indices
65            self.find_location_index(location)
66                .and_then(|index| self.resolve_location(index, return_type_location_instead))
67                .or_else(|| self.try_resolve_trait_impl_location(location))
68                .or_else(|| self.try_resolve_trait_method_declaration(location))
69                .or_else(|| self.try_resolve_type_ref(location))
70                .or_else(|| self.try_resolve_type_alias(location))
71        })
72    }

74    pub fn get_declaration_location_from(&self, location: Location) -> Option<Location> {
75        self.try_resolve_trait_method_declaration(location).or_else(|| {
76            let found_impl_location = self
77                .find_location_index(location)
78                .and_then(|index| self.resolve_location(index, false))?;
79            self.try_resolve_trait_method_declaration(found_impl_location)
80        })
81    }

83    /// For a given [Index] we return [Location] to which we resolved to
84    /// We currently return None for features not yet implemented
85    /// TODO(#3659): LSP goto def should error when Ident at Location could not resolve
86    fn resolve_location(
87        &self,
88        index: impl Into<Index>,
89        return_type_location_instead: bool,
90    ) -> Option<Location> {
91        if return_type_location_instead {
92            return self.get_type_location_from_index(index);
93        }
94
95        let node = self.nodes.get(index.into())?;
96
97        match node {
98            Node::Function(func) => {
99                self.resolve_location(func.as_expr(), return_type_location_instead)
100            }
101            Node::Expression(expression) => {
102                self.resolve_expression_location(expression, return_type_location_instead)
103            }
104            _ => None,
105        }
106    }

108    fn get_type_location_from_index(&self, index: impl Into<Index>) -> Option<Location> {
109        match self.id_type(index.into()) {
110            Type::DataType(struct_type, _) => Some(struct_type.borrow().location),
111            _ => None,
112        }
113    }

115    /// Resolves the [Location] of the definition for a given [HirExpression]
116    ///
117    /// Note: current the code returns None because some expressions are not yet implemented.
118    fn resolve_expression_location(
119        &self,
120        expression: &HirExpression,
121        return_type_location_instead: bool,
122    ) -> Option<Location> {
123        match expression {
124            HirExpression::Ident(ident, _) => {
125                let definition_info = self.definition(ident.id);
126                match definition_info.kind {
127                    DefinitionKind::Function(func_id) => {
128                        Some(self.function_meta(&func_id).location)
129                    }
130                    DefinitionKind::Local(_local_id) => Some(definition_info.location),
131                    DefinitionKind::Global(_global_id) => Some(definition_info.location),
132                    _ => None,
133                }
134            }
135            HirExpression::Constructor(expr) => {
136                let struct_type = &expr.r#type.borrow();
137                Some(struct_type.location)
138            }
139            HirExpression::MemberAccess(expr_member_access) => {
140                self.resolve_struct_member_access(expr_member_access)
141            }
142            HirExpression::Call(expr_call) => {
143                let func = expr_call.func;
144                self.resolve_location(func, return_type_location_instead)
145            }
146
147            _ => None,
148        }
149    }

151    /// Resolves the [Location] of the definition for a given [crate::hir_def::expr::HirMemberAccess]
152    /// This is used to resolve the location of a struct member access.
153    /// For example, in the expression `foo.bar` we want to resolve the location of `bar`
154    /// to the location of the definition of `bar` in the struct `foo`.
155    fn resolve_struct_member_access(
156        &self,
157        expr_member_access: &crate::hir_def::expr::HirMemberAccess,
158    ) -> Option<Location> {
159        let expr_lhs = &expr_member_access.lhs;
160        let expr_rhs = &expr_member_access.rhs;
161
162        let lhs_self_struct = match self.id_type(expr_lhs) {
163            Type::DataType(struct_type, _) => struct_type,
164            _ => return None,
165        };
166
167        let struct_type = lhs_self_struct.borrow();
168        let field_names = struct_type.field_names()?;
169
170        field_names.iter().find(|field_name| field_name.as_str() == expr_rhs.as_str()).map(
171            |found_field_name| Location::new(found_field_name.span(), struct_type.location.file),
172        )
173    }

175    /// Attempts to resolve [Location] of [Trait][crate::hir_def::traits::Trait] based on [Location] of [TraitImpl][crate::hir_def::traits::TraitImpl]
176    /// This is used by LSP to resolve the location of a trait based on the location of a trait impl.
177    ///
178    /// Example:
179    /// impl Foo for Bar { ... } -> trait Foo { ... }
180    fn try_resolve_trait_impl_location(&self, location: Location) -> Option<Location> {
181        let shared_trait_impl = self.trait_implementations.iter().find(|shared_trait_impl| {
182            let trait_impl = shared_trait_impl.1.borrow();
183            trait_impl.file == location.file && trait_impl.ident.span().contains(&location.span)
184        })?;
185        let trait_impl = shared_trait_impl.1.borrow();
186        self.traits.get(&trait_impl.trait_id).map(|trait_| trait_.location)
187    }

189    /// Attempts to resolve [Location] of [Trait][crate::hir_def::traits::Trait]'s [TraitFunction][crate::hir_def::traits::TraitFunction] declaration based on the [Location] of a [TraitFunction][crate::hir_def::traits::TraitFunction] call.
190    ///
191    /// This is used by LSP to resolve the location.
192    ///
193    /// ### Example:
194    /// ```nr
195    /// trait HasToField {
196    ///     fn to_field(self) -> Field;
197    ///        ^------------------------------\
198    /// }                                     |    
199    ///                                       |
200    /// fn main_func(x: u32) {                |
201    ///     assert(x.to_field() == 15);       |
202    ///               \......................./
203    /// }
204    /// ```
205    ///
206    fn try_resolve_trait_method_declaration(&self, location: Location) -> Option<Location> {
207        let (func_id, _func_meta) =
208            self.func_meta.iter().find(|(_, func_meta)| func_meta.location.contains(&location))?;
209        let (_, trait_id) = self.get_function_trait(func_id)?;
210
211        let mut methods = self.traits.get(&trait_id)?.methods.iter();
212        let method = methods.find(|method| method.name.as_str() == self.function_name(func_id));
213        method.map(|method| method.location)
214    }

216    /// Attempts to resolve [Location] of [Type] based on [Location] of reference in code
217    pub(crate) fn try_resolve_type_ref(&self, location: Location) -> Option<Location> {
218        let typ = self.try_type_ref_at_location(location)?;
219        match typ {
220            Type::DataType(struct_typ, _) => Some(struct_typ.borrow().location),
221            _ => None,
222        }
223    }

225    pub fn try_type_ref_at_location(&self, location: Location) -> Option<Type> {
226        self.type_ref_locations
227            .iter()
228            .find(|(_typ, type_ref_location)| type_ref_location.contains(&location))
229            .map(|(typ, _type_ref_location)| typ.clone())
230    }

232    fn try_resolve_type_alias(&self, location: Location) -> Option<Location> {
233        self.type_alias_ref
234            .iter()
235            .find(|(_, named_type_location)| named_type_location.span.contains(&location.span))
236            .map(|(type_alias_id, _found_location)| {
237                self.get_type_alias(*type_alias_id).borrow().location
238            })
