11impl Parser<'_> {
12    /// ModOrContract
13    ///     = ( 'mod' | 'contract' ) identifier ( '{' Module '}' | ';' )
14    pub(super) fn parse_mod_or_contract(
15        &mut self,
16        ident: Ident,
17        attributes: Vec<(Attribute, Location)>,
18        is_contract: bool,
19        visibility: ItemVisibility,
20    ) -> ItemKind {
21        let outer_attributes = self.validate_secondary_attributes(attributes);
22
23        if self.eat_left_brace() {
24            let contents = self.parse_module(
25                true, // nested
26            );
27            self.eat_or_error(Token::RightBrace);
28            ItemKind::Submodules(ParsedSubModule {
29                visibility,
30                name: ident,
31                contents,
32                outer_attributes,
33                is_contract,
34            })
35        } else {
36            let has_semicolon = self.eat_semicolons();
37            if !has_semicolon {
38                self.expected_token(Token::Semicolon);
39            }
40            ItemKind::ModuleDecl(ModuleDeclaration {
41                visibility,
42                ident,
43                outer_attributes,
44                has_semicolon,
45            })
46        }
47    }
48}

62        expect_no_errors(&errors);

136    #[test]
137    fn parse_program_rejects_utf8_in_identifier() {
138        use crate::lexer::errors::LexerErrorKind;
139        use crate::parser::ParserErrorReason;
140
141        // cSpell:disable-next-line
142        let src = "fn schön() {}";
143        let (module, errors) = parse_program_with_dummy_file(src);
144
145        let non_warning_errors: Vec<_> =
146            errors.iter().filter(|error| !error.is_warning()).collect();
147        assert_eq!(
148            non_warning_errors.len(),
149            1,
150            "Expected exactly one error, got: {non_warning_errors:?}",
151        );
152        match non_warning_errors[0].reason() {
153            Some(ParserErrorReason::Lexer(LexerErrorKind::NonAsciiIdentifier {
154                found, ..
155            })) => {
156                assert_eq!(found, "schön");
157            }
158            other => panic!("Expected NonAsciiIdentifier, got {other:?}"),
159        }
160
161        // The parser still recognised the function declaration, with the original name.
162        assert_eq!(module.items.len(), 1);
163        let ItemKind::Function(func) = &module.items[0].kind else {
164            panic!("Expected function declaration");
165        };
166        assert_eq!(func.def.name.to_string(), "schön");
167    }
