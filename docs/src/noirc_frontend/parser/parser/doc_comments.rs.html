9impl Parser<'_> {
10    /// InnerDocComments = inner_doc_comment*
11    pub(super) fn parse_inner_doc_comments(&mut self) -> Vec<DocComment> {
12        self.parse_many("inner doc comments", without_separator(), Self::parse_inner_doc_comment)
13    }
14
15    fn parse_inner_doc_comment(&mut self) -> Option<DocComment> {
16        self.eat_kind(TokenKind::InnerDocComment).map(|token| {
17            let location = token.location();
18            match token.into_token() {
19                Token::LineComment(comment, Some(DocStyle::Inner)) => {
20                    let comment = fix_line_comment(comment);
21                    DocComment::from(location, comment)
22                }
23                Token::BlockComment(comment, Some(DocStyle::Inner)) => {
24                    let comment = fix_block_comment(comment);
25                    DocComment::from(location, comment)
26                }
27                _ => unreachable!(),
28            }
29        })
30    }
31
32    /// OuterDocComments = OuterDocComment*
33    pub(super) fn parse_outer_doc_comments(&mut self) -> Vec<DocComment> {
34        self.parse_many("outer doc comments", without_separator(), Self::parse_outer_doc_comment)
35    }
36
37    /// OuterDocComment = outer_doc_comment
38    pub(super) fn parse_outer_doc_comment(&mut self) -> Option<DocComment> {
39        self.eat_kind(TokenKind::OuterDocComment).map(|token| {
40            let location = token.location();
41            match token.into_token() {
42                Token::LineComment(comment, Some(DocStyle::Outer)) => {
43                    let comment = fix_line_comment(comment);
44                    DocComment::from(location, comment)
45                }
46                Token::BlockComment(comment, Some(DocStyle::Outer)) => {
47                    let comment = fix_block_comment(comment);
48                    DocComment::from(location, comment)
49                }
50                _ => unreachable!(),
51            }
52        })
53    }
54
55    /// Skips any outer doc comments but produces a warning saying that they don't document anything.
56    pub(super) fn warn_on_outer_doc_comments(&mut self) {
57        self.skip_doc_comments_with_reason(ParserErrorReason::DocCommentDoesNotDocumentAnything);
58    }
59
60    /// Skips any outer doc comments but produces an error saying that they can't be applied to parameters
61    pub(super) fn error_on_outer_doc_comments_on_parameter(&mut self) {
62        let reason = ParserErrorReason::DocCommentCannotBeAppliedToFunctionParameters;
63        self.skip_doc_comments_with_reason(reason);
64    }
65
66    fn skip_doc_comments_with_reason(&mut self, reason: ParserErrorReason) {
67        let location_before_doc_comments = self.current_token_location;
68        let doc_comments = self.parse_outer_doc_comments();
69        if !doc_comments.is_empty() {
70            self.push_error(reason, self.location_since(location_before_doc_comments));
71        }
72    }
73}

75/// Strips leading ' ' from a line comment.
76fn fix_line_comment(comment: String) -> String {
77    if let Some(comment) = comment.strip_prefix(' ') { comment.to_string() } else { comment }
78}
79
80/// Strips leading '*' from a block comment if all non-empty lines have it.
81fn fix_block_comment(comment: String) -> String {
82    let all_stars = block_comment_has_all_leading_stars(&comment);
83
84    let mut fixed_comment = String::new();
85    for (index, line) in comment.lines().enumerate() {
86        if index > 0 {
87            fixed_comment.push('\n');
88        }
89
90        if all_stars {
91            if let Some(line) = line.trim_start().strip_prefix("*") {
92                fixed_comment.push_str(line.strip_prefix(' ').unwrap_or(line));
93                continue;
94            }
95        }
96
97        if let Some(line) = line.strip_prefix(' ') {
98            fixed_comment.push_str(line);
99            continue;
100        }
101
102        fixed_comment.push_str(line);
103    }
104    fixed_comment.trim().to_string()
105}
106
107/// Returns true if a block comment has a '*' at the start of every non-empty line.
108pub fn block_comment_has_all_leading_stars(comment: &str) -> bool {
109    comment.lines().enumerate().all(|(index, line)| {
110        if index == 0 || line.trim().is_empty() {
111            // The first line never has a star. Then we ignore empty lines.
112            true
113        } else {
114            line.trim_start().starts_with('*')
115        }
116    })
117}

124    fn parses_inner_doc_comments() {
125        let src = "//! Hello\n//! World";
126        let mut parser = Parser::for_str_with_dummy_file(src);
127        let comments = parser.parse_inner_doc_comments();
128        expect_no_errors(&parser.errors);
129        assert_eq!(comments.len(), 2);
