12impl Parser<'_> {
13    /// WhereClause = 'where' WhereClauseItems?
14    ///
15    /// WhereClauseItems = WhereClauseItem ( ',' WhereClauseItem )* ','?
16    ///
17    /// WhereClauseItem = Type ':' TraitBounds
18    pub(super) fn parse_where_clause(&mut self) -> Vec<UnresolvedTraitConstraint> {
19        if !self.eat_keyword(Keyword::Where) {
20            return Vec::new();
21        }
22
23        // Constraints might end up being empty, but that's accepted as valid syntax
24        let constraints =
25            self.parse_many("where clauses", separated_by_comma(), Self::parse_single_where_clause);
26
27        constraints
28            .into_iter()
29            .flat_map(|(typ, trait_bounds)| {
30                trait_bounds.into_iter().map(move |trait_bound| UnresolvedTraitConstraint {
31                    typ: typ.clone(),
32                    trait_bound,
33                })
34            })
35            .collect()
36    }
37
38    fn parse_single_where_clause(&mut self) -> Option<(UnresolvedType, Vec<TraitBound>)> {
39        let typ = self.parse_type()?;
40
41        self.eat_or_error(Token::Colon);
42
43        let trait_bounds = self.parse_trait_bounds();
44
45        Some((typ, trait_bounds))
46    }
47
48    /// TraitBounds = TraitBound ( '+' TraitBound )? '+'?
49    pub(super) fn parse_trait_bounds(&mut self) -> Vec<TraitBound> {
50        self.parse_many(
51            "trait bounds",
52            separated_by(Token::Plus).stop_if_separator_is_missing(),
53            Self::parse_trait_bound_in_list,
54        )
55    }
56
57    fn parse_trait_bound_in_list(&mut self) -> Option<TraitBound> {
58        if let Some(trait_bound) = self.parse_trait_bound() {
59            Some(trait_bound)
60        } else {
61            self.expected_label(ParsingRuleLabel::TraitBound);
62            None
63        }
64    }
65
66    pub(crate) fn parse_trait_bound_or_error(&mut self) -> TraitBound {
67        if let Some(trait_bound) = self.parse_trait_bound() {
68            return trait_bound;
69        }
70
71        self.expected_label(ParsingRuleLabel::TraitBound);
72        TraitBound {
73            trait_path: Path::plain(Vec::new(), self.location_at_previous_token_end()),
74            trait_generics: GenericTypeArgs::default(),
75        }
76    }
77
78    /// TraitBound = PathNoTurbofish GenericTypeArgs
79    pub(crate) fn parse_trait_bound(&mut self) -> Option<TraitBound> {
80        let trait_path = self.parse_path_no_turbofish()?;
81        let trait_generics = self.parse_generic_type_args();
82        Some(TraitBound { trait_path, trait_generics })
83    }
84}

99    fn parse_where_clause_no_errors(src: &str) -> Vec<UnresolvedTraitConstraint> {
100        let mut parser = Parser::for_str_with_dummy_file(src);
101        let constraints = parser.parse_where_clause();
102        expect_no_errors(&parser.errors);
103        constraints
104    }
105
106    #[test]
107    fn parses_no_where_clause() {
108        let src = "{";
109        let constraints = parse_where_clause_no_errors(src);
110        assert!(constraints.is_empty());
111    }
112
113    #[test]
114    fn parses_one_where_clause_with_two_constraints() {
115        let src = "where Foo: Bar<T> + Baz";
116        let mut constraints = parse_where_clause_no_errors(src);
117        assert_eq!(constraints.len(), 2);
118
119        let constraint = constraints.remove(0);
120        assert_eq!(constraint.typ.to_string(), "Foo");
121        assert_eq!(constraint.trait_bound.trait_path.to_string(), "Bar");
122        assert_eq!(constraint.trait_bound.trait_generics.ordered_args[0].to_string(), "T");
123
124        let constraint = constraints.remove(0);
125        assert_eq!(constraint.typ.to_string(), "Foo");
126        assert_eq!(constraint.trait_bound.trait_path.to_string(), "Baz");
127    }
