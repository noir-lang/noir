5pub(crate) struct CallArguments {
6    pub(crate) arguments: Vec<Expression>,
7    pub(crate) is_macro_call: bool,
8}
9
10impl Parser<'_> {
11    /// Arguments = '(' ArgumentsList? ')'
12    ///
13    /// ArgumentsList = Expression ( ',' Expression )? ','?
14    pub(crate) fn parse_arguments(&mut self) -> Option<Vec<Expression>> {
15        if !self.eat_left_paren() {
16            return None;
17        }
18
19        let arguments = self.parse_many(
20            "arguments",
21            separated_by_comma_until_right_paren(),
22            Self::parse_expression_in_list,
23        );
24
25        Some(arguments)
26    }
27
28    /// CallArguments = '!'? Arguments
29    pub(super) fn parse_call_arguments(&mut self) -> Option<CallArguments> {
30        let is_macro_call = self.at(Token::Bang) && self.next_is(Token::LeftParen);
31
32        if is_macro_call {
33            // Given that we expected '!' '(', it's safe to skip the '!' because the next
34            // `self.parse_arguments()` will always return `Some`.
35            self.bump();
36        }
37
38        self.parse_arguments().map(|arguments| CallArguments { arguments, is_macro_call })
39    }
40}
