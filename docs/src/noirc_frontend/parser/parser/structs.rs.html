11impl Parser<'_> {
12    /// Struct = 'struct' identifier Generics '{' StructField* '}'
13    ///
14    /// StructField = OuterDocComments identifier ':' Type
15    pub(crate) fn parse_struct(
16        &mut self,
17        attributes: Vec<(Attribute, Location)>,
18        visibility: ItemVisibility,
19        start_location: Location,
20    ) -> NoirStruct {
21        let attributes = self.validate_secondary_attributes(attributes);
22
23        let Some(name) = self.eat_non_underscore_ident() else {
24            self.expected_identifier();
25            return self.empty_struct(
26                self.empty_ident_at_previous_token_end(),
27                attributes,
28                visibility,
29                Vec::new(),
30                start_location,
31            );
32        };
33
34        let generics = self.parse_generics_disallowing_trait_bounds();
35
36        if self.eat_semicolons() {
37            return self.empty_struct(name, attributes, visibility, generics, start_location);
38        }
39
40        if !self.eat_left_brace() {
41            self.expected_token(Token::LeftBrace);
42            return self.empty_struct(name, attributes, visibility, generics, start_location);
43        }
44
45        let fields = self.parse_many(
46            "struct fields",
47            separated_by_comma_until_right_brace(),
48            Self::parse_struct_field,
49        );
50
51        NoirStruct {
52            name,
53            attributes,
54            visibility,
55            generics,
56            fields,
57            location: self.location_since(start_location),
58        }
59    }
60
61    fn parse_struct_field(&mut self) -> Option<Documented<StructField>> {
62        let mut doc_comments;
63        let name;
64        let mut visibility;
65
66        // Loop until we find an identifier, skipping anything that's not one
67        loop {
68            let doc_comments_start_location = self.current_token_location;
69            doc_comments = self.parse_outer_doc_comments();
70
71            visibility = self.parse_item_visibility();
72
73            if let Some(ident) = self.eat_non_underscore_ident() {
74                name = ident;
75                break;
76            }
77
78            if visibility != ItemVisibility::Private {
79                self.expected_identifier();
80            }
81
82            if !doc_comments.is_empty() {
83                self.push_error(
84                    ParserErrorReason::DocCommentDoesNotDocumentAnything,
85                    self.location_since(doc_comments_start_location),
86                );
87            }
88
89            // Though we do have to stop at EOF
90            if self.at_eof() {
91                self.expected_token(Token::RightBrace);
92                return None;
93            }
94
95            // Or if we find a right brace
96            if self.at(Token::RightBrace) {
97                return None;
98            }
99
100            self.expected_identifier();
101            self.bump();
102        }
103
104        self.eat_or_error(Token::Colon);
105
106        let typ = self.parse_type_or_error();
107        Some(Documented::new(StructField { visibility, name, typ }, doc_comments))
108    }
109
110    fn empty_struct(
111        &self,
112        name: Ident,
113        attributes: Vec<SecondaryAttribute>,
114        visibility: ItemVisibility,
115        generics: UnresolvedGenerics,
116        start_location: Location,
117    ) -> NoirStruct {
118        NoirStruct {
119            name,
120            attributes,
121            visibility,
122            generics,
123            fields: Vec::new(),
124            location: self.location_since(start_location),
125        }
126    }

146        let (mut module, errors) = parse_program_with_dummy_file(src);
147        expect_no_errors(&errors);

174    #[test]
175    fn parse_empty_struct_with_generics() {
176        let src = "struct Foo<A, let B: u32> {}";
177        let mut noir_struct = parse_struct_no_errors(src);
178        assert_eq!("Foo", noir_struct.name.to_string());
179        assert!(noir_struct.fields.is_empty());
180        assert_eq!(noir_struct.generics.len(), 2);
181
182        let generic = noir_struct.generics.remove(0);
183        let UnresolvedGeneric::Variable(ident, trait_bounds) = generic else {
184            panic!("Expected generic variable");
185        };
186        assert_eq!("A", ident.to_string());
187        assert!(trait_bounds.is_empty());
188
189        let generic = noir_struct.generics.remove(0);
190        let UnresolvedGeneric::Numeric { ident, typ } = generic else {
191            panic!("Expected generic numeric");
192        };
193        assert_eq!("B", ident.to_string());
194        assert_eq!(typ.typ.to_string(), "u32");
195    }
