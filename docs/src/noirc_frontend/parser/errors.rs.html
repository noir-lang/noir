136/// Represents a parsing error, or a parsing error in the making.
137///
138/// `ParserError` is used extensively by the parser, as it not only used to report badly formed
139/// token streams, but also as a general intermediate that accumulates information as various
140/// parsing rules are tried. This struct is constructed and destructed with a very high frequency
141/// and as such, the time taken to do so significantly impacts parsing performance. For this
142/// reason we use `SmallOrdSet` to avoid heap allocations for as long as possible - this greatly
143/// inflates the size of the error, but this is justified by a resulting increase in parsing
144/// speeds of approximately 40% in release mode.
145///
146/// Both `expected_tokens` and `expected_labels` use `SmallOrdSet` sized 1. In the of labels this
147/// is optimal. In the of tokens we stop here due to fast diminishing returns.
148#[derive(Debug, Clone, PartialEq, Eq)]
149pub struct ParserError {
150    expected_tokens: SmallOrdSet<[Token; 1]>,
151    expected_labels: SmallOrdSet<[ParsingRuleLabel; 1]>,
152    found: Token,
153    reason: Option<ParserErrorReason>,
154    location: Location,
155}
156
157impl ParserError {
158    pub fn empty(found: Token, location: Location) -> ParserError {
159        ParserError {
160            expected_tokens: SmallOrdSet::new(),
161            expected_labels: SmallOrdSet::new(),
162            found,
163            reason: None,
164            location,
165        }
166    }
167
168    pub fn expected_token(token: Token, found: Token, location: Location) -> ParserError {
169        let mut error = ParserError::empty(found, location);
170        error.expected_tokens.insert(token);
171        error
172    }
173
174    pub fn expected_one_of_tokens(
175        tokens: &[Token],
176        found: Token,
177        location: Location,
178    ) -> ParserError {
179        let mut error = ParserError::empty(found, location);
180        for token in tokens {
181            error.expected_tokens.insert(token.clone());
182        }
183        error
184    }
185
186    pub fn expected_label(
187        label: ParsingRuleLabel,
188        found: Token,
189        location: Location,
190    ) -> ParserError {
191        let mut error = ParserError::empty(found, location);
192        error.expected_labels.insert(label);
193        error
194    }
195
196    pub fn with_reason(reason: ParserErrorReason, location: Location) -> ParserError {
197        let mut error = ParserError::empty(Token::EOF, location);
198        error.reason = Some(reason);
199        error
200    }
201
202    pub fn found(&self) -> &Token {
203        &self.found
204    }
205
206    pub fn span(&self) -> Span {
207        self.location.span
208    }
209
210    pub fn location(&self) -> Location {
211        self.location
212    }
213
214    pub fn reason(&self) -> Option<&ParserErrorReason> {
215        self.reason.as_ref()
216    }
217
218    pub fn is_warning(&self) -> bool {
219        let diagnostic: Diagnostic = self.into();
220        diagnostic.is_warning()
221    }
222}

224impl std::fmt::Display for ParserError {
225    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
226        let token_to_string = |token: &Token| match token {
227            Token::EOF => token.to_string(),
228            _ => format!("'{token}'"),
229        };
230
231        let reason_str: String = if self.reason.is_none() {
232            "".to_string()
233        } else {
234            format!("\nreason: {}", Diagnostic::from(self))
235        };
236        let mut expected = vecmap(&self.expected_tokens, token_to_string);
237        expected.append(&mut vecmap(&self.expected_labels, |label| format!("{label}")));
238
239        if expected.is_empty() {
240            write!(f, "Unexpected {} in input{}", self.found, reason_str)
241        } else if expected.len() == 1 {
242            let first = expected.first().unwrap();
243            let vowel = "aeiou".contains(first.chars().next().unwrap());
244            write!(
245                f,
246                "Expected a{} {} but found {}{}",
247                if vowel { "n" } else { "" },
248                first,
249                token_to_string(&self.found),
250                reason_str
251            )
252        } else {
253            let expected = expected.iter().map(ToString::to_string).collect::<Vec<_>>().join(", ");
254
255            write!(f, "Unexpected '{}', expected one of {}{}", self.found, expected, reason_str)
256        }
257    }
258}

260impl<'a> From<&'a ParserError> for Diagnostic {
261    fn from(error: &'a ParserError) -> Diagnostic {
262        match &error.reason {
263            Some(reason) => match reason {
264                ParserErrorReason::ConstrainDeprecated => {
265                    let mut diagnostic = Diagnostic::simple_error(
266                        "Use of deprecated keyword 'constrain'".into(),
267                        "The 'constrain' keyword is deprecated. Please use the 'assert' function instead.".into(),
268                        error.location(),
269                    );
270                    diagnostic.deprecated = true;
271                    diagnostic
272                }
273                ParserErrorReason::ExperimentalFeature(feature) => {
274                    let secondary = format!(
275                        "Pass -Z{feature} to nargo to enable this feature at your own risk."
276                    );
277                    match feature {
278                        UnstableFeature::TraitAsType => {
279                            let primary = "`impl Trait` as a type is experimental".to_string();
280                            Diagnostic::simple_warning(primary, secondary, error.location())
281                        }
282                        _ => Diagnostic::simple_error(
283                            reason.to_string(),
284                            secondary,
285                            error.location(),
286                        ),
287                    }
288                }
289                ParserErrorReason::TraitVisibilityIgnored => {
290                    Diagnostic::simple_warning(reason.to_string(), "".into(), error.location())
291                }
292                ParserErrorReason::TraitImplVisibilityIgnored => {
293                    Diagnostic::simple_warning(reason.to_string(), "".into(), error.location())
294                }
295                ParserErrorReason::Lexer(error) => error.into(),
296                ParserErrorReason::ExpectedMutAfterAmpersand { found } => Diagnostic::simple_error(
297                    format!("Expected `mut` after `&`, found `{found}`"),
298                    "Noir doesn't have immutable references, only mutable references".to_string(),
299                    error.location(),
300                ),
301                ParserErrorReason::MissingSafetyComment => Diagnostic::simple_warning(
302                    "Unsafe block must have a safety comment above it".into(),
303                    "The comment must start with the \"Safety: \" word".into(),
304                    error.location(),
305                ),
306                ParserErrorReason::MissingParametersForFunctionDefinition => {
307                    Diagnostic::simple_error(
308                        "Missing parameters for function definition".into(),
309                        "Add a parameter list: `()`".into(),
310                        error.location(),
311                    )
312                }
313                ParserErrorReason::DocCommentDoesNotDocumentAnything => {
314                    let primary = "This doc comment doesn't document anything".to_string();
315                    let secondary = "Consider changing it to a regular `//` comment".to_string();
316                    Diagnostic::simple_warning(primary, secondary, error.location())
317                }
318                ParserErrorReason::MissingAngleBrackets => {
319                    let secondary = "Types that don't start with an identifier need to be surrounded with angle brackets: `<`, `>`".to_string();
320                    Diagnostic::simple_error(format!("{reason}"), secondary, error.location())
321                }
322                ParserErrorReason::LogicalAnd => {
323                    let primary = "Noir has no logical-and (&&) operator since short-circuiting is much less efficient when compiling to circuits".to_string();
324                    let secondary =
325                        "Try `&` instead, or use `if` only if you require short-circuiting"
326                            .to_string();
327                    Diagnostic::simple_error(primary, secondary, error.location)
328                }
329                ParserErrorReason::MissingTypeForAssociatedConstant => Diagnostic::simple_error(
330                    "Missing type for associated constant".to_string(),
331                    "Provide a type for the associated constant: `: u32`".to_string(),
332                    error.location,
333                ),
334                ParserErrorReason::DeprecatedDep(name) => {
335                    let primary = format!("`dep::{name}` path is deprecated");
336                    let secondary = format!("Please use `::{name}` instead");
337                    Diagnostic::simple_warning(primary, secondary, error.location())
338                }
339                ParserErrorReason::StructLiteralInIfCondition => Diagnostic::simple_error(
340                    "Struct literals are not allowed in `if` conditions".to_string(),
341                    "Surround the struct literal with parentheses, for example: `if (MyStruct { field: true }).field { ... }`".to_string(),
342                    error.location(),
343                ),
344                other => {
345                    Diagnostic::simple_error(format!("{other}"), String::new(), error.location())
346                }
347            },
348            None => {
349                if matches!(
350                    error.found.kind(),
351                    TokenKind::InnerDocComment | TokenKind::OuterDocComment
352                ) {
353                    let primary = "This doc comment doesn't document anything".to_string();
354                    let secondary = "Consider changing it to a regular `//` comment".to_string();
355                    Diagnostic::simple_warning(primary, secondary, error.location())
356                } else {
357                    let primary = error.to_string();
358                    Diagnostic::simple_error(primary, String::new(), error.location())
359                }
360            }
361        }
362    }
