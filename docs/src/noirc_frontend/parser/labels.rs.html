1use std::fmt;
2
3use crate::token::TokenKind;
4
5/// Used to annotate parsing rules with extra context that can be presented to the user later in
6/// the case of an error.
7#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord)]
8pub enum ParsingRuleLabel {
9    Atom,
10    BinaryOperator,
11    Cast,
12    Expression,
13    FieldAccess,
14    Function,
15    GenericParameter,
16    Global,
17    Identifier,
18    Integer,
19    IntegerType,
20    Item,
21    LValue,
22    Parameter,
23    Path,
24    Pattern,
25    Statement,
26    String,
27    Term,
28    TraitBound,
29    TraitImplItem,
30    TraitItem,
31    Type,
32    TypeExpression,
33    TypeOrTypeExpression,
34    TokenKind(TokenKind),
35    UseSegment,
36}
37
38impl fmt::Display for ParsingRuleLabel {
39    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
40        match self {
41            ParsingRuleLabel::Atom => write!(f, "atom"),
42            ParsingRuleLabel::BinaryOperator => write!(f, "binary operator"),
43            ParsingRuleLabel::Cast => write!(f, "cast"),
44            ParsingRuleLabel::Expression => write!(f, "expression"),
45            ParsingRuleLabel::FieldAccess => write!(f, "field access"),
46            ParsingRuleLabel::Function => write!(f, "function"),
47            ParsingRuleLabel::GenericParameter => write!(f, "generic parameter"),
48            ParsingRuleLabel::Global => write!(f, "global"),
49            ParsingRuleLabel::Identifier => write!(f, "identifier"),
50            ParsingRuleLabel::Integer => write!(f, "integer"),
51            ParsingRuleLabel::IntegerType => write!(f, "integer type"),
52            ParsingRuleLabel::Item => write!(f, "item"),
53            ParsingRuleLabel::LValue => write!(f, "left-hand side of assignment"),
54            ParsingRuleLabel::Parameter => write!(f, "parameter"),
55            ParsingRuleLabel::Path => write!(f, "path"),
56            ParsingRuleLabel::Pattern => write!(f, "pattern"),
57            ParsingRuleLabel::Statement => write!(f, "statement"),
58            ParsingRuleLabel::String => write!(f, "string"),
59            ParsingRuleLabel::Term => write!(f, "term"),
60            ParsingRuleLabel::TraitBound => write!(f, "trait bound"),
61            ParsingRuleLabel::TraitImplItem => write!(f, "trait impl item"),
62            ParsingRuleLabel::TraitItem => write!(f, "trait item"),
63            ParsingRuleLabel::Type => write!(f, "type"),
64            ParsingRuleLabel::TypeExpression => write!(f, "type expression"),
65            ParsingRuleLabel::TypeOrTypeExpression => write!(f, "type or type expression"),
66            ParsingRuleLabel::TokenKind(token_kind) => write!(f, "{token_kind}"),
67            ParsingRuleLabel::UseSegment => write!(f, "identifier, `crate`, `dep` or `super`"),
68        }
69    }
70}
