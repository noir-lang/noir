31#[derive(Copy, Clone, Debug)]
32pub struct GetProgramOptions<'a> {
33    pub allow_parser_errors: bool,
34    pub allow_elaborator_errors: bool,
35    /// Treats the program snippet as if it was stdlib, which allows the definition of
36    /// low-level (builtin) functions, and standard traits used by prefix/infix operators.
37    pub root_and_stdlib: bool,
38    pub frontend_options: FrontendOptions<'a>,
39}
40
41impl Default for GetProgramOptions<'_> {
42    fn default() -> Self {
43        Self {
44            allow_parser_errors: false,
45            allow_elaborator_errors: false,
46            root_and_stdlib: false,
47            frontend_options: FrontendOptions::test_default(),
48        }
49    }
50}

52/// Compile and monomorphize a program.
53pub fn get_monomorphized(src: &str) -> Result<Program, MonomorphizationError> {
54    get_monomorphized_with_options(src, GetProgramOptions::default())
55}

57/// Helper to monomorphize code which needs some parts of the stdlib repeated for the test.
58pub fn get_monomorphized_with_stdlib(
59    user_src: &str,
60    stdlib_src: &[&str],
61) -> Result<Program, MonomorphizationError> {
62    let mut src = String::new();
63    for s in stdlib_src {
64        src.push_str(s);
65        src.push_str("\n\n");
66    }
67    src.push_str(user_src);
68    get_monomorphized_with_options(
69        &src,
70        GetProgramOptions { root_and_stdlib: true, ..Default::default() },
71    )
72}

74/// Compile and monomorphize a program.
75pub fn get_monomorphized_with_options(
76    src: &str,
77    options: GetProgramOptions,
78) -> Result<Program, MonomorphizationError> {
79    let (_parsed_module, mut context, errors) = get_program_with_options(src, options);
80
81    let only_warnings = errors.iter().all(|err| !err.is_error());
82    let has_defs = !context.def_maps.is_empty();
83    if !options.allow_elaborator_errors && !only_warnings || !has_defs && !errors.is_empty() {
84        let diagnostics = errors.iter().map(CustomDiagnostic::from).collect::<Vec<_>>();
85        report_all(&context.file_manager, &context.parsed_files, &diagnostics, false, false);
86        panic!(
87            "Expected monomorphized program to have no errors before monomorphization, but found: {errors:?}"
88        )
89    }
90
91    let main = context
92        .get_main_function(context.root_crate_id())
93        .unwrap_or_else(|| panic!("get_monomorphized: test program contains no 'main' function"));
94
95    monomorphize(main, &mut context.def_interner, false)
96}

98pub(crate) fn has_parser_error(errors: &[CompilationError]) -> bool {
99    errors.iter().any(|e| matches!(e, CompilationError::ParseError(_)))
100}

102pub(crate) fn remove_experimental_warnings(errors: &mut Vec<CompilationError>) {
103    errors.retain(|error| match error {
104        CompilationError::ParseError(error) => {
105            !matches!(error.reason(), Some(ParserErrorReason::ExperimentalFeature(..)))
106        }
107        _ => true,
108    });
109}

124/// Compile a program.
125///
126/// The stdlib is not available for these snippets, but using the `root_and_stdlib`
127/// option allows the inclusion of trait definitions that are considered for prefix/infix
128/// operators by the compiler, as well as defining low-level builtin functions.
129pub(crate) fn get_program_with_options(
130    src: &str,
131    options: GetProgramOptions,
132) -> (ParsedModule, Context<'static, 'static>, Vec<CompilationError>) {
133    let root = Path::new("/");
134    let mut fm = FileManager::new(root);
135    let root_file_id = fm.add_file_with_source(Path::new("test_file"), src.to_string()).unwrap();
136    let mut context = Context::new(fm, Default::default());
137
138    context.def_interner.populate_dummy_operator_traits();
139    let root_crate_id = if options.root_and_stdlib {
140        context.crate_graph.add_crate_root_and_stdlib(root_file_id)
141    } else {
142        context.crate_graph.add_crate_root(root_file_id)
143    };
144
145    let (program, parser_errors) = parse_program(src, root_file_id);
146    let mut errors = vecmap(parser_errors, |e| e.into());
147    remove_experimental_warnings(&mut errors);
148
149    if options.allow_parser_errors || !has_parser_error(&errors) {
150        let inner_attributes: Vec<SecondaryAttribute> = program
151            .items
152            .iter()
153            .filter_map(|item| {
154                if let ItemKind::InnerAttribute(attribute) = &item.kind {
155                    Some(attribute.clone())
156                } else {
157                    None
158                }
159            })
160            .collect();
161
162        let location = Location::new(Default::default(), root_file_id);
163        let root_module = ModuleData::new(
164            None,
165            None,
166            location,
167            Vec::new(),
168            inner_attributes,
169            false, // is contract
170            false, // is struct
171        );
172
173        let def_map = CrateDefMap::new(root_crate_id, root_module);
174
175        // Now we want to populate the CrateDefMap using the DefCollector
176        errors.extend(DefCollector::collect_crate_and_dependencies(
177            def_map,
178            &mut context,
179            program.clone().into_sorted(),
180            root_file_id,
181            options.frontend_options,
182        ));
183    }
184
185    (program, context, errors)
186}
