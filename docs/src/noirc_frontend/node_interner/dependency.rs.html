16/// A dependency in the dependency graph may be a type or a definition.
17/// Types can depend on definitions too. E.g. `Foo` depends on `COUNT` in:
18///
19/// ```struct
20/// global COUNT = 3;
21///
22/// struct Foo {
23///     array: [Field; COUNT],
24/// }
25/// ```
26#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
27pub enum DependencyId {
28    DataType(TypeId),
29    Global(GlobalId),
30    Function(FuncId),
31    Alias(TypeAliasId),
32    Trait(TraitId),
33    Variable(Location),
34}
35
36impl NodeInterner {
37    /// Gets the dependency graph from the node interner.
38    pub fn dependency_graph(&self) -> &DiGraph<DependencyId, ()> {
39        &self.dependency_graph
40    }

42    /// Register that `dependent` depends on `dependency`.

68    pub fn add_dependency(&mut self, dependent: DependencyId, dependency: DependencyId) {
69        let dependent_index = self.get_or_insert_dependency(dependent);
70        let dependency_index = self.get_or_insert_dependency(dependency);
71        self.dependency_graph.update_edge(dependent_index, dependency_index, ());
72    }
73
74    pub fn get_or_insert_dependency(&mut self, id: DependencyId) -> PetGraphIndex {
75        if let Some(index) = self.dependency_graph_indices.get(&id) {
76            return *index;
77        }
78
79        let index = self.dependency_graph.add_node(id);
80        self.dependency_graph_indices.insert(id, index);
81        index
82    }

84    pub(crate) fn check_for_dependency_cycles(&self) -> Vec<CompilationError> {
85        let mut errors = Vec::new();
86
87        let mut push_error = |item: String, scc: &[_], i, location: Location| {
88            let cycle = self.get_cycle_error_string(scc, i);
89            let error = ResolverError::DependencyCycle { item, cycle, location };
90            errors.push(error.into());
91        };
92
93        let mut push_error_from_index = |scc: &[_], scc_index, node_index: PetGraphIndex| -> bool {
94            match self.dependency_graph[node_index] {
95                DependencyId::DataType(struct_id) => {
96                    let struct_type = self.get_type(struct_id);
97                    let struct_type = struct_type.borrow();
98                    let name = &struct_type.name;
99                    push_error(name.to_string(), scc, scc_index, name.location());
100                    true
101                }
102                DependencyId::Global(global_id) => {
103                    let global = self.get_global(global_id);
104                    let name = global.ident.to_string();
105                    push_error(name, scc, scc_index, global.location);
106                    true
107                }
108                DependencyId::Alias(alias_id) => {
109                    let alias = self.get_type_alias(alias_id);
110                    let alias = alias.borrow();
111                    push_error(alias.name.to_string(), scc, scc_index, alias.name.location());
112                    true
113                }
114                DependencyId::Trait(trait_id) => {
115                    let the_trait = self.get_trait(trait_id);
116                    let name = &the_trait.name;
117                    push_error(name.to_string(), scc, scc_index, name.location());
118                    true
119                }
120                // Mutually recursive functions are allowed
121                DependencyId::Function(_) => false,
122                // Local variables should never be in a dependency cycle, scoping rules
123                // prevents referring to them before they're defined
124                DependencyId::Variable(loc) => {
125                    unreachable!("Variable used at location {loc:?} caught in a dependency cycle")
126                }
127            }
128        };
129
130        // Checking for single-node cycles.
131        //
132        // Enabling this highlights errors such as `type Alias = Alias;`,
133        // however it also emits errors for things like `type Foo = u32; impl Foo {}`,
134        // ie. if there is an impl for something that shouldn't have one, _unless_
135        // there is an `fn main() {}` as well, in which case the error does not appear.
136        // Since all corresponding unit tests seem to carry at least another error,
137        // and this behavior is strange, and the SCC below only looks for more than 1,
138        // for now it's left commented out.
139        //
140        // for edge_reference in self.dependency_graph.edge_references() {
141        //     let source = edge_reference.source();
142        //     let target = edge_reference.target();
143        //     if source == target {
144        //         let scc_index = 0;
145        //         let node_index = source;
146        //         push_error_from_index(&[source], scc_index, node_index);
147        //     }
148        // }
149
150        let strongly_connected_components = tarjan_scc(&self.dependency_graph);
151        for scc in strongly_connected_components {
152            if scc.len() > 1 {
153                // If a SCC contains a type, type alias, or global, it must be the only element in the SCC
154                for (scc_index, node_index) in scc.iter().enumerate() {
155                    if push_error_from_index(&scc, scc_index, *node_index) {
156                        break;
157                    }
158                }
159            }
160        }
161
162        errors
163    }

165    /// Build up a string starting from the given item containing each item in the dependency
166    /// cycle. The final result will resemble `foo -> bar -> baz -> foo`, always going back to the
167    /// element at the given start index.
168    fn get_cycle_error_string(&self, scc: &[PetGraphIndex], start_index: usize) -> String {
169        let index_to_string = |index: PetGraphIndex| match self.dependency_graph[index] {
170            DependencyId::DataType(id) => Cow::Owned(self.get_type(id).borrow().name.to_string()),
171            DependencyId::Function(id) => Cow::Borrowed(self.function_name(&id)),
172            DependencyId::Alias(id) => {
173                Cow::Owned(self.get_type_alias(id).borrow().name.to_string())
174            }
175            DependencyId::Global(id) => Cow::Borrowed(self.get_global(id).ident.as_str()),
176            DependencyId::Trait(id) => Cow::Owned(self.get_trait(id).name.to_string()),
177            DependencyId::Variable(loc) => {
178                unreachable!("Variable used at location {loc:?} caught in a dependency cycle")
179            }
180        };
181
182        let mut cycle = index_to_string(scc[start_index]).to_string();
183
184        // Reversing the dependencies here matches the order users would expect for the error message
185        for i in (0..scc.len()).rev() {
186            cycle += " -> ";
187            cycle += &index_to_string(scc[(start_index + i) % scc.len()]);
188        }
189
190        cycle
191    }
