430#[derive(Debug, Clone, Eq, PartialEq)]
431pub enum DefinitionKind {
432    Function(FuncId),
433
434    Global(GlobalId),
435
436    /// Locals may be defined in let statements or parameters,
437    /// in which case they will not have an associated ExprId.
438    /// For example a mutable variable can change, so it does
439    /// not have a stable defining expression.
440    Local(Option<ExprId>),
441
442    /// Generic types in functions (T, U in `fn foo<T, U>(...)` are declared as variables
443    /// in scope in case they resolve to numeric generics later.
444    NumericGeneric(TypeVariable, Box<Type>),
445
446    AssociatedConstant(TraitImplId, String),
447}

449impl DefinitionKind {
450    /// True if this definition is for a global variable.
451    /// Note that this returns false for top-level functions.
452    pub fn is_global(&self) -> bool {
453        matches!(self, DefinitionKind::Global(..))
454    }
455
456    pub fn get_rhs(&self) -> Option<ExprId> {
457        match self {
458            DefinitionKind::Function(_) => None,
459            DefinitionKind::Global(_) => None,
460            DefinitionKind::Local(id) => *id,
461            DefinitionKind::NumericGeneric(_, _) => None,
462            DefinitionKind::AssociatedConstant(_, _) => None,
463        }
464    }

886    /// Returns the type of the definition, unless it's a function returning an `impl Trait`,
887    /// in which case it looks up the type of its body and returns a new function type with
888    /// the type fo the body substituted to its return type.
889    ///
890    /// Returns:
891    /// * `Ok` if it was able to substitute the type, either because it's not an `impl Trait`
892    ///   or because the type of the body of the function is already known
893    /// * `Err` if the function returning the `impl Trait` needs to be elaborated first
894    pub fn id_type_substitute_trait_as_type(&self, def_id: DefinitionId) -> Result<Type, FuncId> {
895        let typ = self.definition_type(def_id);
896        if let Type::Function(args, ret, env, unconstrained) = &typ {
897            let def = self.definition(def_id);
898            if let Type::TraitAsType(..) = ret.as_ref() {
899                if let DefinitionKind::Function(func_id) = def.kind {
900                    let func = self.function(&func_id);
901                    let Some(func_body) = func.try_as_expr() else {
902                        return Err(func_id);
903                    };
904                    let ret_type = self.id_type(func_body);
905                    let new_type = Type::Function(
906                        args.clone(),
907                        Box::new(ret_type),
908                        env.clone(),
909                        *unconstrained,
910                    );
911                    return Ok(new_type);
912                }
913            }
914        }
915        Ok(typ)
916    }

1012    /// Adds a non-trait method to a type.
1013    ///
1014    /// Returns `Some(duplicate)` if a matching method was already defined.
1015    /// Returns `None` otherwise.
1016    pub fn add_method(
1017        &mut self,
1018        self_type: &Type,
1019        method_name: String,
1020        method_id: FuncId,
1021        trait_id: Option<TraitId>,
1022    ) -> Result<(), CompilationError> {
1023        match self_type {
1024            Type::Error => Ok(()),
1025            Type::Reference(element, _mutable) => {
1026                self.add_method(element, method_name, method_id, trait_id)
1027            }
1028            _ => {
1029                let Some(key) = get_type_method_key(self_type) else {
1030                    let location = self.function_ident(&method_id).location();
1031                    let error = ResolverError::TypeUnsupportedForMethod {
1032                        typ: self_type.clone(),
1033                        location,
1034                    };
1035                    return Err(error.into());
1036                };
1037
1038                if trait_id.is_none() && matches!(self_type, Type::DataType(..)) {
1039                    let check_self_param = false;
1040                    if let Some(existing) =
1041                        self.lookup_direct_method(self_type, &method_name, check_self_param)
1042                    {
1043                        let first_location = self.function_ident(&existing).location();
1044                        let second_location = self.function_ident(&method_id).location();
1045                        let error = ResolverError::DuplicateDefinition {
1046                            name: method_name,
1047                            first_location,
1048                            second_location,
1049                        };
1050                        return Err(error.into());
1051                    }
1052                }
1053
1054                // Only remember the actual type if it's FieldOrInt,
1055                // so later we can disambiguate on calls like `u32::call`.
1056                let typ = self_type.clone();
1057                self.methods
1058                    .entry(key)
1059                    .or_default()
1060                    .entry(method_name)
1061                    .or_default()
1062                    .add_method(method_id, typ, trait_id);
1063                Ok(())
1064            }
1065        }
1066    }

1167    /// Add the given trait as an operator trait if its name matches one of the
1168    /// operator trait names (Add, Sub, ...).
1169    pub fn try_add_infix_operator_trait(&mut self, trait_id: TraitId) {
1170        let the_trait = self.get_trait(trait_id);
1171
1172        let operator = match the_trait.name.as_str() {
1173            "Add" => BinaryOpKind::Add,
1174            "Sub" => BinaryOpKind::Subtract,
1175            "Mul" => BinaryOpKind::Multiply,
1176            "Div" => BinaryOpKind::Divide,
1177            "Rem" => BinaryOpKind::Modulo,
1178            "Eq" => BinaryOpKind::Equal,
1179            "Ord" => BinaryOpKind::Less,
1180            "BitAnd" => BinaryOpKind::And,
1181            "BitOr" => BinaryOpKind::Or,
1182            "BitXor" => BinaryOpKind::Xor,
1183            "Shl" => BinaryOpKind::ShiftLeft,
1184            "Shr" => BinaryOpKind::ShiftRight,
1185            _ => return,
1186        };
1187
1188        self.infix_operator_traits.insert(operator, trait_id);
1189
1190        // Some operators also require we insert a matching entry for related operators
1191        match operator {
1192            BinaryOpKind::Equal => {
1193                self.infix_operator_traits.insert(BinaryOpKind::NotEqual, trait_id);
1194            }
1195            BinaryOpKind::Less => {
1196                self.infix_operator_traits.insert(BinaryOpKind::LessEqual, trait_id);
1197                self.infix_operator_traits.insert(BinaryOpKind::Greater, trait_id);
1198                self.infix_operator_traits.insert(BinaryOpKind::GreaterEqual, trait_id);
1199
1200                let the_trait = self.get_trait(trait_id);
1201                self.ordering_type = match &the_trait.methods[0].typ {
1202                    Type::Forall(_, typ) => match typ.as_ref() {
1203                        Type::Function(_, return_type, _, _) => Some(return_type.as_ref().clone()),
1204                        other => unreachable!("Expected function type for `cmp`, found {}", other),
1205                    },
1206                    other => unreachable!("Expected Forall type for `cmp`, found {}", other),
1207                };
1208            }
1209            _ => (),
1210        }

1360    /// Returns the type of an operator (which is always a function), along with its return type.
1361    pub fn get_infix_operator_type(
1362        &self,
1363        lhs: ExprId,
1364        operator: BinaryOpKind,
1365        operator_expr: ExprId,
1366    ) -> (Type, Type) {
1367        let lhs_type = self.id_type(lhs);
1368        let args = vec![lhs_type.clone(), lhs_type];
1369
1370        // If this is a comparison operator, the result is a boolean but
1371        // the actual method call returns an Ordering
1372        use crate::ast::BinaryOpKind::*;
1373        let ret = if matches!(operator, Less | LessEqual | Greater | GreaterEqual) {
1374            self.ordering_type()
1375        } else {
1376            self.id_type(operator_expr)
1377        };
1378
1379        let env = Box::new(Type::Unit);
1380        (Type::Function(args, Box::new(ret.clone()), env, false), ret)
1381    }

1396    /// Sets the associated types for the given trait impl.
1397    /// Each type in [`NamedType`] will be wrapped in a [`Type::TypeVariable`] if it's of kind [`Kind::Numeric`].
1398    pub(crate) fn set_associated_types_for_impl(
1399        &mut self,
1400        impl_id: TraitImplId,
1401        associated_types: Vec<NamedType>,
1402    ) {
1403        // Wrap the named generics in type variables to be able to refer them as type variables
1404        for associated_type in &associated_types {
1405            let Kind::Numeric(numeric_type) = associated_type.typ.kind() else {
1406                continue;
1407            };
1408
1409            let name = associated_type.name.to_string();
1410            let definition_id = self.push_definition(
1411                associated_type.name.to_string(),
1412                false,
1413                false,
1414                DefinitionKind::AssociatedConstant(impl_id, name.clone()),
1415                associated_type.name.location(),
1416            );
1417            self.trait_impl_associated_constants
1418                .entry(impl_id)
1419                .or_default()
1420                .insert(name, (definition_id, *numeric_type));
1421        }

1453    /// Looks up associated constants for a type by name.
1454    /// Returns all matching constants with their trait info.
1455    pub fn lookup_trait_impl_constants_for_type(
1456        &self,
1457        typ: &Type,
1458        constant_name: &str,
1459    ) -> Vec<(DefinitionId, TraitId, TraitImplId)> {
1460        let mut results = Vec::new();
1461
1462        for (impl_id, constants) in &self.trait_impl_associated_constants {
1463            let Some((def_id, _constant_type)) = constants.get(constant_name) else {
1464                continue;
1465            };
1466            let Some(trait_impl) = self.try_get_trait_implementation(*impl_id) else {
1467                continue;
1468            };
1469
1470            let trait_id = trait_impl.borrow().trait_id;
1471
1472            // Check if typ implements the trait
1473            // This handles instantiation and unification correctly for generic impls
1474            if let Ok((TraitImplKind::Normal(found_impl_id), _, _)) =
1475                self.try_lookup_trait_implementation(typ, trait_id, &[], &[])
1476            {
1477                if found_impl_id == *impl_id {
1478                    results.push((*def_id, trait_id, *impl_id));
1479                }
1480            }
1481        }
1482        results
1483    }

1485    /// Return a set of TypeBindings to bind types from the parent trait to those from the trait impl.
1486    pub fn trait_to_impl_bindings(
1487        &self,
1488        trait_id: TraitId,
1489        impl_id: TraitImplId,
1490        trait_impl_generics: &[Type],
1491        impl_self_type: Type,
1492    ) -> TypeBindings {
1493        let mut bindings = TypeBindings::default();
1494        let the_trait = self.get_trait(trait_id);
1495        let trait_generics = the_trait.generics.clone();
1496
1497        let self_type_var = the_trait.self_type_typevar.clone();
1498        bindings.insert(
1499            self_type_var.id(),
1500            (self_type_var.clone(), self_type_var.kind(), impl_self_type),
1501        );
1502
1503        for (trait_generic, trait_impl_generic) in trait_generics.iter().zip(trait_impl_generics) {
1504            let type_var = trait_generic.type_var.clone();
1505            bindings.insert(
1506                type_var.id(),
1507                (type_var, trait_generic.kind(), trait_impl_generic.clone()),
1508            );
1509        }
1510
1511        // Now that the normal bindings are added, we still need to bind the associated types
1512        let impl_associated_types = self.get_associated_types_for_impl(impl_id);
1513        let trait_associated_types = &the_trait.associated_types;
1514
1515        // `impl_associated_types` may not be in the same order as `trait_associated_types`
1516        let impl_associated_types = impl_associated_types
1517            .iter()
1518            .map(|typ| (typ.name.as_str(), typ))
1519            .collect::<HashMap<_, _>>();
1520
1521        for trait_type in trait_associated_types {
1522            let Some(impl_type) = impl_associated_types.get(trait_type.name.as_str()) else {
1523                // Impl doesn't have the corresponding associated type - an error should already
1524                // have been issued beforehand.
1525                continue;
1526            };
1527
1528            let type_variable = trait_type.type_var.clone();
1529            bindings.insert(
1530                type_variable.id(),
1531                (type_variable, trait_type.kind(), impl_type.typ.clone()),
1532            );
1533        }
1534
1535        bindings
1536    }

1584            }
1585        }
1586
1587        // Clear in auto import names
1588        for (_name, entries) in self.auto_import_names.iter_mut() {
1589            entries.retain(|entry| entry.file != file);

1592        // Clear in reexports
1593        for (_module_def_if, reexports) in self.reexports.iter_mut() {
1594            reexports.retain(|reexport| reexport.name.location().file != file);
1595        }
1596
1597        // Clear in LocationIndices
1598        self.clear_file_locations(file);

1602/// These are the primitive type variants that we support adding methods to
1603#[derive(Copy, Clone, Hash, PartialEq, Eq, Debug)]
1604pub(crate) enum TypeMethodKey {
1605    /// Fields and integers share methods for ease of use. These methods may still
1606    /// accept only fields or integers, it is just that their names may not clash.
1607    FieldOrInt,
1608    Array,
1609    Vector,
1610    Bool,
1611    String,
1612    FmtString,
1613    Unit,
1614    Tuple,
1615    Function,
1616    Generic,
1617    Quoted(QuotedType),
1618    Struct(TypeId),
1619}
1620
1621pub(crate) fn get_type_method_key(typ: &Type) -> Option<TypeMethodKey> {
1622    use TypeMethodKey::*;
1623    let typ = typ.follow_bindings();
1624    match &typ {
1625        Type::FieldElement => Some(FieldOrInt),
1626        Type::Array(_, _) => Some(Array),
1627        Type::Vector(_) => Some(Vector),
1628        Type::Integer(_, _) => Some(FieldOrInt),
1629        Type::TypeVariable(var) => {
1630            if var.is_integer() || var.is_integer_or_field() {
1631                Some(FieldOrInt)
1632            } else {
1633                None
1634            }
1635        }
1636        Type::Bool => Some(Bool),
1637        Type::String(_) => Some(String),
1638        Type::FmtString(_, _) => Some(FmtString),
1639        Type::Unit => Some(Unit),
1640        Type::Tuple(_) => Some(Tuple),
1641        Type::Function(_, _, _, _) => Some(Function),
1642        Type::NamedGeneric(_) => Some(Generic),
1643        Type::Quoted(quoted) => Some(Quoted(*quoted)),
1644        Type::Reference(element, _) => get_type_method_key(element),
1645        Type::Alias(alias, _) => get_type_method_key(&alias.borrow().typ),
1646        Type::DataType(struct_type, _) => Some(Struct(struct_type.borrow().id)),
1647
1648        // We do not support adding methods to these types
1649        Type::Forall(_, _)
1650        | Type::Constant(..)
1651        | Type::Error
1652        | Type::InfixExpr(..)
1653        | Type::CheckedCast { .. }
1654        | Type::TraitAsType(..) => None,
1655    }
