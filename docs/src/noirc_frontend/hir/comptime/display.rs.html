28pub(super) fn display_quoted(
29    tokens: &[LocatedToken],
30    indent: usize,
31    interner: &NodeInterner,
32    f: &mut std::fmt::Formatter<'_>,
33) -> std::fmt::Result {
34    if tokens.is_empty() {
35        write!(f, "quote {{ }}")
36    } else {
37        writeln!(f, "quote {{")?;
38        let indent = indent + 1;
39        write!(f, "{}", " ".repeat(indent * 4))?;
40        TokensPrettyPrinter { tokens, interner, indent, preserve_unquote_markers: false }.fmt(f)?;
41        writeln!(f)?;
42        let indent = indent - 1;
43        write!(f, "{}", " ".repeat(indent * 4))?;
44        write!(f, "}}")
45    }
46}

48struct TokensPrettyPrinter<'tokens, 'interner> {
49    tokens: &'tokens [LocatedToken],
50    interner: &'interner NodeInterner,
51    indent: usize,
52    preserve_unquote_markers: bool,
53}
54
55impl Display for TokensPrettyPrinter<'_, '_> {
56    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
57        let mut token_printer =
58            TokenPrettyPrinter::new(self.interner, self.indent, self.preserve_unquote_markers);
59        for token in self.tokens {
60            token_printer.print(token.token(), f)?;
61        }
62
63        // If the printer refrained from printing a token right away, this will make it do it
64        token_printer.print(&Token::EOF, f)?;
65
66        Ok(())
67    }
68}

83/// Tries to print tokens in a way that it'll be easier for the user to understand a
84/// stream of tokens without having to format it themselves.
85///
86/// The gist is:
87/// - Keep track of the current indent level
88/// - When '{' is found, a newline is inserted and the indent is incremented
89/// - When '}' is found, a newline is inserted and the indent is decremented
90/// - When ';' is found a newline is inserted
91/// - When interned values are encountered, they are turned into strings and indented
92///   according to the current indentation.
93///
94/// There are a few more details that needs to be taken into account:
95/// - two consecutive words shouldn't be glued together (as they are separate tokens)
96/// - inserting spaces when needed
97/// - not inserting extra newlines if possible
98/// - ';' shouldn't always insert newlines (this is when it's something like `[Field; 2]`)
99struct TokenPrettyPrinter<'interner> {
100    interner: &'interner NodeInterner,
101    indent: usize,
102    preserve_unquote_markers: bool,
103    /// Determines whether the last outputted byte was alphanumeric.
104    /// This is used to add a space after the last token and before another token
105    /// that starts with an alphanumeric byte.
106    last_was_alphanumeric: bool,
107    last_was_right_brace: bool,
108    last_was_semicolon: bool,
109    last_was_op: bool,
110}
111
112impl<'interner> TokenPrettyPrinter<'interner> {
113    fn new(
114        interner: &'interner NodeInterner,
115        indent: usize,
116        preserve_unquote_markers: bool,
117    ) -> Self {
118        Self {
119            interner,
120            indent,
121            preserve_unquote_markers,
122            last_was_alphanumeric: false,
123            last_was_right_brace: false,
124            last_was_semicolon: false,
125            last_was_op: false,
126        }
127    }
128
129    fn print(&mut self, token: &Token, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
130        let last_was_alphanumeric = self.last_was_alphanumeric;
131        self.last_was_alphanumeric = false;
132
133        let last_was_op = self.last_was_op;
134        self.last_was_op = false;
135
136        // After `}` we usually want a newline... but not always!
137        if self.last_was_right_brace {
138            self.last_was_right_brace = false;
139
140            match token {
141                Token::Keyword(Keyword::Else) => {
142                    // If we have `} else` we don't want a newline
143                    write!(f, " else")?;
144                    self.last_was_alphanumeric = true;
145                    return Ok(());
146                }
147                Token::RightBrace => {
148                    // Because we insert a newline right before `}`, if we have two
149                    // (or more) in a row we don't want extra newlines.
150                }
151                _ => {
152                    writeln!(f)?;
153                    self.write_indent(f)?;
154                }
155            }
156        }
157
158        // Heuristic: if we have `; 2` then we assume we are inside something like `[Field; 2]`
159        // and don't include a newline.
160        // The only consequence of getting this wrong is that we'll end with two consecutive
161        // statements in a single line (not a big deal).
162        if self.last_was_semicolon {
163            self.last_was_semicolon = false;
164
165            match token {
166                Token::Int(..) => {
167                    write!(f, " ")?;
168                }
169                Token::Ident(ident) => {
170                    if ident.chars().all(|char| char.is_ascii_uppercase()) {
171                        write!(f, " ")?;
172                    } else {
173                        writeln!(f)?;
174                        self.write_indent(f)?;
175                    }
176                }
177                Token::RightBrace => {
178                    // We don't want an extra newline in this case
179                }
180                _ => {
181                    writeln!(f)?;
182                    self.write_indent(f)?;
183                }
184            }
185        }
186
187        // If the last token was one of `+`, `-`, etc. and the current token is not `=`, we want a space
188        // (we avoid outputting a space if the token is `=` a bit below)
189        if last_was_op && !matches!(token, Token::Assign) {
190            write!(f, " ")?;
191        }
192
193        match token {
194            Token::QuotedType(id) => {
195                let value = Value::Type(self.interner.get_quoted_type(*id).clone());
196                self.print_value(&value, last_was_alphanumeric, f)
197            }
198            Token::InternedExpr(id) => {
199                let value = Value::expression(ExpressionKind::Interned(*id));
200                self.print_value(&value, last_was_alphanumeric, f)
201            }
202            Token::InternedStatement(id) => {
203                let value = Value::statement(StatementKind::Interned(*id));
204                self.print_value(&value, last_was_alphanumeric, f)
205            }
206            Token::InternedLValue(id) => {
207                let value = Value::lvalue(LValue::Interned(*id, Location::dummy()));
208                self.print_value(&value, last_was_alphanumeric, f)
209            }
210            Token::InternedUnresolvedTypeData(id) => {
211                let value = Value::UnresolvedType(UnresolvedTypeData::Interned(*id));
212                self.print_value(&value, last_was_alphanumeric, f)
213            }
214            Token::InternedPattern(id) => {
215                let value = Value::pattern(Pattern::Interned(*id, Location::dummy()));
216                self.print_value(&value, last_was_alphanumeric, f)
217            }
218            Token::InternedCrate(_) => write!(f, "$crate"),
219            Token::UnquoteMarker(id) => {
220                let value = Value::TypedExpr(TypedExpr::ExprId(*id));
221                let last_was_alphanumeric = if self.preserve_unquote_markers {
222                    if last_was_alphanumeric {
223                        write!(f, " ")?;
224                    }
225                    write!(f, "$")?;
226                    false
227                } else {
228                    last_was_alphanumeric
229                };
230                self.print_value(&value, last_was_alphanumeric, f)
231            }
232            Token::Keyword(..) | Token::Ident(..) | Token::Int(..) | Token::Bool(..) => {
233                if last_was_alphanumeric {
234                    write!(f, " ")?;
235                }
236                self.last_was_alphanumeric = true;
237                write!(f, "{token}")
238            }
239            Token::Comma => {
240                write!(f, "{token} ")
241            }
242            Token::LeftBrace => {
243                writeln!(f, " {{")?;
244                self.indent += 1;
245                self.write_indent(f)
246            }
247            Token::RightBrace => {
248                self.last_was_right_brace = true;
249                writeln!(f)?;
250                self.indent -= 1;
251                self.write_indent(f)?;
252                write!(f, "}}")
253            }
254            Token::Semicolon => {
255                self.last_was_semicolon = true;
256                write!(f, ";")
257            }
258            Token::Quote(tokens) => {
259                if last_was_alphanumeric {
260                    write!(f, " ")?;
261                }
262                display_quoted(&tokens.0, self.indent, self.interner, f)
263            }
264            Token::Colon => {
265                write!(f, "{token} ")
266            }
267            Token::Less
268            | Token::LessEqual
269            | Token::Greater
270            | Token::GreaterEqual
271            | Token::Equal
272            | Token::NotEqual
273            | Token::FatArrow
274            | Token::Arrow => write!(f, " {token} "),
275            Token::Assign => {
276                if last_was_op {
277                    write!(f, "{token} ")
278                } else {
279                    write!(f, " {token} ")
280                }
281            }
282            Token::Plus
283            | Token::Minus
284            | Token::Star
285            | Token::Slash
286            | Token::Percent
287            | Token::Ampersand
288            | Token::DeprecatedVectorStart
289            | Token::ShiftLeft
290            | Token::ShiftRight
291            | Token::LogicalAnd => {
292                self.last_was_op = true;
293                write!(f, " {token}")
294            }
295            Token::LeftParen
296            | Token::RightParen
297            | Token::LeftBracket
298            | Token::RightBracket
299            | Token::Dot
300            | Token::DoubleColon
301            | Token::DoubleDot
302            | Token::DoubleDotEqual
303            | Token::Caret
304            | Token::Pound
305            | Token::Pipe
306            | Token::Bang
307            | Token::At
308            | Token::Backslash
309            | Token::DollarSign => {
310                write!(f, "{token}")
311            }
312            Token::Str(..)
313            | Token::RawStr(..)
314            | Token::FmtStr(..)
315            | Token::Whitespace(_)
316            | Token::BlockComment(..)
317            | Token::AttributeStart { .. }
318            | Token::Invalid(_) => {
319                if last_was_alphanumeric {
320                    write!(f, " ")?;
321                }
322                write!(f, "{token}")
323            }
324            Token::LineComment(..) => {
325                writeln!(f, "{token}")?;
326                self.write_indent(f)
327            }
328            Token::EOF => Ok(()),
329        }
330    }
331
332    fn print_value(
333        &mut self,
334        value: &Value,
335        last_was_alphanumeric: bool,
336        f: &mut std::fmt::Formatter<'_>,
337    ) -> std::fmt::Result {
338        let string = value.display(self.interner).to_string();
339        if string.is_empty() {
340            return Ok(());
341        }
342
343        if last_was_alphanumeric && string.bytes().next().unwrap().is_ascii_alphanumeric() {
344            write!(f, " ")?;
345        }
346
347        for (index, line) in string.lines().enumerate() {
348            if index > 0 {
349                writeln!(f)?;
350                self.write_indent(f)?;
351            }
352            line.fmt(f)?;
353        }
354
355        self.last_was_alphanumeric = string.bytes().last().unwrap().is_ascii_alphanumeric();
356        self.last_was_right_brace = string.ends_with('}');
357        self.last_was_semicolon = string.ends_with(';');
358
359        Ok(())
360    }
361
362    fn write_indent(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
363        write!(f, "{}", " ".repeat(self.indent * 4))
364    }
365}

381impl Display for ValuePrinter<'_, '_> {
382    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
383        match self.value {
384            Value::Unit => write!(f, "()"),
385            Value::Bool(value) => {
386                let msg = if *value { "true" } else { "false" };
387                write!(f, "{msg}")
388            }
389            Value::Field(value) => {
390                // write!(f, "{value}") // This would display the Field as a number, but it doesn't match the runtime.
391                write!(f, "{}", value.to_field_element().to_short_hex())
392            }
393            Value::I8(value) => write!(f, "{value}"),
394            Value::I16(value) => write!(f, "{value}"),
395            Value::I32(value) => write!(f, "{value}"),
396            Value::I64(value) => write!(f, "{value}"),
397            Value::U1(false) => write!(f, "0"),
398            Value::U1(true) => write!(f, "1"),
399            Value::U8(value) => write!(f, "{value}"),
400            Value::U16(value) => write!(f, "{value}"),
401            Value::U32(value) => write!(f, "{value}"),
402            Value::U64(value) => write!(f, "{value}"),
403            Value::U128(value) => write!(f, "{value}"),
404            Value::String(value) => write!(f, "{value}"),
405            Value::CtString(value) => write!(f, "{value}"),
406            Value::FormatString(fragments, _, _) => {
407                let string = fragments_to_string(fragments, self.interner);
408                write!(f, "{string}")
409            }
410            Value::Function(..) => write!(f, "(function)"),
411            Value::Closure(..) => write!(f, "(closure)"),
412            Value::Tuple(fields) => {
413                let fields =
414                    vecmap(fields, |field| field.borrow().display(self.interner).to_string());
415                if fields.len() == 1 {
416                    write!(f, "({},)", fields[0])
417                } else {
418                    write!(f, "({})", fields.join(", "))
419                }
420            }
421            Value::Struct(fields, typ) => {
422                let data_type = match typ.follow_bindings() {
423                    Type::DataType(def, _) => def,
424                    other => unreachable!("Expected data type, found {other}"),
425                };
426                let data_type = data_type.borrow();
427                let typename = data_type.name.to_string();
428
429                // Display fields in the order they are defined in the struct.
430                // Some fields might not be there if they were missing in the constructor.
431                let fields = data_type
432                    .fields_raw()
433                    .unwrap()
434                    .iter()
435                    .filter_map(|field| {
436                        let name = field.name.as_string();
437                        fields.get(name).map(|value| {
438                            format!("{}: {}", name, value.borrow().display(self.interner))
439                        })
440                    })
441                    .collect::<Vec<_>>();
442                write!(f, "{typename} {{ {} }}", fields.join(", "))
443            }
444            Value::Enum(tag, args, typ) => {
445                let args = vecmap(args, |arg| arg.display(self.interner).to_string()).join(", ");
446
447                match typ.follow_bindings_shallow().as_ref() {
448                    Type::DataType(def, _) => {
449                        let def = def.borrow();
450                        let variant = def.variant_at(*tag);
451                        if variant.is_function {
452                            write!(f, "{}::{}({args})", def.name, variant.name)
453                        } else {
454                            write!(f, "{}::{}", def.name, variant.name)
455                        }
456                    }
457                    other => write!(f, "{other}(args)"),
458                }
459            }
460            Value::Pointer(value, _, mutable) => {
461                if *mutable {
462                    write!(f, "&mut {}", value.borrow().display(self.interner))
463                } else {
464                    write!(f, "&{}", value.borrow().display(self.interner))
465                }
466            }
467            Value::Array(values, _) => {
468                let values = vecmap(values, |value| value.display(self.interner).to_string());
469                write!(f, "[{}]", values.join(", "))
470            }
471            Value::Vector(values, _) => {
472                let values = vecmap(values, |value| value.display(self.interner).to_string());
473                write!(f, "@[{}]", values.join(", "))
474            }
475            Value::Quoted(tokens) => display_quoted(tokens, 0, self.interner, f),
476            Value::TypeDefinition(id) => {
477                let def = self.interner.get_type(*id);
478                let def = def.borrow();
479                write!(f, "{}", def.name)
480            }
481            Value::TraitConstraint(trait_id, generics) => {
482                let trait_ = self.interner.get_trait(*trait_id);
483                write!(f, "{}{generics}", trait_.name)
484            }
485            Value::TraitDefinition(trait_id) => {
486                let trait_ = self.interner.get_trait(*trait_id);
487                write!(f, "{}", trait_.name)
488            }
489            Value::TraitImpl(trait_impl_id) => {
490                let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
491                let trait_impl = trait_impl.borrow();
492
493                let generic_string =
494                    vecmap(&trait_impl.trait_generics, ToString::to_string).join(", ");
495                let generic_string = if generic_string.is_empty() {
496                    generic_string
497                } else {
498                    format!("<{generic_string}>")
499                };
500
501                let where_clause = vecmap(&trait_impl.where_clause, |trait_constraint| {
502                    display_trait_constraint(self.interner, trait_constraint)
503                });
504                let where_clause = where_clause.join(", ");
505                let where_clause = if where_clause.is_empty() {
506                    where_clause
507                } else {
508                    format!(" where {where_clause}")
509                };
510
511                write!(
512                    f,
513                    "impl {}{} for {}{}",
514                    trait_impl.ident, generic_string, trait_impl.typ, where_clause
515                )
516            }
517            Value::FunctionDefinition(function_id) => {
518                write!(f, "{}", self.interner.function_name(function_id))
519            }
520            Value::ModuleDefinition(module_id) => {
521                if let Some(attributes) = self.interner.try_module_attributes(*module_id) {
522                    write!(f, "{}", &attributes.name)
523                } else {
524                    write!(f, "(crate root)")
525                }
526            }
527            Value::Zeroed(typ) => write!(f, "(zeroed {typ})"),
528            Value::Type(typ) => write!(f, "{typ}"),
529            Value::Expr(expr) => match expr.as_ref() {
530                ExprValue::Expression(expr) => {
531                    let expr = remove_interned_in_expression_kind(self.interner, expr.clone());
532                    write!(f, "{expr}")
533                }
534                ExprValue::Statement(statement) => {
535                    write!(
536                        f,
537                        "{}",
538                        remove_interned_in_statement_kind(self.interner, statement.clone())
539                    )
540                }
541                ExprValue::LValue(lvalue) => {
542                    write!(f, "{}", remove_interned_in_lvalue(self.interner, lvalue.clone()))
543                }
544                ExprValue::Pattern(pattern) => {
545                    write!(f, "{}", remove_interned_in_pattern(self.interner, pattern.clone()))
546                }
547            },
548            Value::TypedExpr(TypedExpr::ExprId(id)) => {
549                let hir_expr = self.interner.expression(id);
550                let expr = hir_expr.to_display_ast(self.interner, Location::dummy());
551                write!(f, "{}", expr.kind)
552            }
553            Value::TypedExpr(TypedExpr::StmtId(id)) => {
554                let hir_statement = self.interner.statement(id);
555                let stmt = hir_statement.to_display_ast(self.interner, Location::dummy());
556                write!(f, "{}", stmt.kind)
557            }
558            Value::UnresolvedType(typ) => {
559                write!(f, "{}", remove_interned_in_unresolved_type_data(self.interner, typ.clone()))
560            }
561        }
562    }
563}

579impl Display for TokenPrinter<'_, '_> {
580    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
581        match self.token {
582            Token::QuotedType(id) => {
583                write!(f, "{}", self.interner.get_quoted_type(*id))
584            }
585            Token::InternedExpr(id) => {
586                let value = Value::expression(ExpressionKind::Interned(*id));
587                value.display(self.interner).fmt(f)
588            }
589            Token::InternedStatement(id) => {
590                let value = Value::statement(StatementKind::Interned(*id));
591                value.display(self.interner).fmt(f)
592            }
593            Token::InternedLValue(id) => {
594                let value = Value::lvalue(LValue::Interned(*id, Location::dummy()));
595                value.display(self.interner).fmt(f)
596            }
597            Token::InternedUnresolvedTypeData(id) => {
598                let value = Value::UnresolvedType(UnresolvedTypeData::Interned(*id));
599                value.display(self.interner).fmt(f)
600            }
601            Token::InternedPattern(id) => {
602                let value = Value::pattern(Pattern::Interned(*id, Location::dummy()));
603                value.display(self.interner).fmt(f)
604            }
605            Token::UnquoteMarker(id) => {
606                let value = Value::TypedExpr(TypedExpr::ExprId(*id));
607                value.display(self.interner).fmt(f)
608            }
609            other => write!(f, "{other}"),
610        }
611    }

622// Returns a new Expression where all Interned and Resolved expressions have been turned into non-interned ExpressionKind.
623fn remove_interned_in_expression(interner: &NodeInterner, expr: Expression) -> Expression {
624    Expression {
625        kind: remove_interned_in_expression_kind(interner, expr.kind),
626        location: expr.location,
627    }

630// Returns a new ExpressionKind where all Interned and Resolved expressions have been turned into non-interned ExpressionKind.
631fn remove_interned_in_expression_kind(
632    interner: &NodeInterner,
633    expr: ExpressionKind,
634) -> ExpressionKind {
635    match expr {
636        ExpressionKind::Literal(literal) => {
637            ExpressionKind::Literal(remove_interned_in_literal(interner, literal))
638        }
639        ExpressionKind::Block(block) => {
640            let statements =
641                vecmap(block.statements, |stmt| remove_interned_in_statement(interner, stmt));
642            ExpressionKind::Block(BlockExpression { statements })
643        }
644        ExpressionKind::Prefix(prefix) => ExpressionKind::Prefix(Box::new(PrefixExpression {
645            rhs: remove_interned_in_expression(interner, prefix.rhs),
646            ..*prefix
647        })),
648        ExpressionKind::Index(index) => ExpressionKind::Index(Box::new(IndexExpression {
649            collection: remove_interned_in_expression(interner, index.collection),
650            index: remove_interned_in_expression(interner, index.index),
651        })),
652        ExpressionKind::Call(call) => ExpressionKind::Call(Box::new(CallExpression {
653            func: Box::new(remove_interned_in_expression(interner, *call.func)),
654            arguments: vecmap(call.arguments, |arg| remove_interned_in_expression(interner, arg)),
655            ..*call
656        })),
657        ExpressionKind::MethodCall(call) => {
658            ExpressionKind::MethodCall(Box::new(MethodCallExpression {
659                object: remove_interned_in_expression(interner, call.object),
660                arguments: vecmap(call.arguments, |arg| {
661                    remove_interned_in_expression(interner, arg)
662                }),
663                ..*call
664            }))
665        }
666        ExpressionKind::Constrain(constrain) => ExpressionKind::Constrain(ConstrainExpression {
667            arguments: vecmap(constrain.arguments, |expr| {
668                remove_interned_in_expression(interner, expr)
669            }),
670            ..constrain
671        }),
672        ExpressionKind::Constructor(constructor) => {
673            ExpressionKind::Constructor(Box::new(ConstructorExpression {
674                fields: vecmap(constructor.fields, |(name, expr)| {
675                    (name, remove_interned_in_expression(interner, expr))
676                }),
677                ..*constructor
678            }))
679        }
680        ExpressionKind::MemberAccess(member_access) => {
681            ExpressionKind::MemberAccess(Box::new(MemberAccessExpression {
682                lhs: remove_interned_in_expression(interner, member_access.lhs),
683                ..*member_access
684            }))
685        }
686        ExpressionKind::Cast(cast) => ExpressionKind::Cast(Box::new(CastExpression {
687            lhs: remove_interned_in_expression(interner, cast.lhs),
688            ..*cast
689        })),
690        ExpressionKind::Infix(infix) => ExpressionKind::Infix(Box::new(InfixExpression {
691            lhs: remove_interned_in_expression(interner, infix.lhs),
692            rhs: remove_interned_in_expression(interner, infix.rhs),
693            ..*infix
694        })),
695        ExpressionKind::If(if_expr) => ExpressionKind::If(Box::new(IfExpression {
696            condition: remove_interned_in_expression(interner, if_expr.condition),
697            consequence: remove_interned_in_expression(interner, if_expr.consequence),
698            alternative: if_expr
699                .alternative
700                .map(|alternative| remove_interned_in_expression(interner, alternative)),
701        })),
702        ExpressionKind::Match(match_expr) => ExpressionKind::Match(Box::new(MatchExpression {
703            expression: remove_interned_in_expression(interner, match_expr.expression),
704            rules: vecmap(match_expr.rules, |(pattern, branch)| {
705                let pattern = remove_interned_in_expression(interner, pattern);
706                let branch = remove_interned_in_expression(interner, branch);
707                (pattern, branch)
708            }),
709        })),
710        ExpressionKind::Variable(_) => expr,
711        ExpressionKind::Tuple(expressions) => ExpressionKind::Tuple(vecmap(expressions, |expr| {
712            remove_interned_in_expression(interner, expr)
713        })),
714        ExpressionKind::Lambda(lambda) => ExpressionKind::Lambda(Box::new(Lambda {
715            body: remove_interned_in_expression(interner, lambda.body),
716            ..*lambda
717        })),
718        ExpressionKind::Parenthesized(expr) => {
719            ExpressionKind::Parenthesized(Box::new(remove_interned_in_expression(interner, *expr)))
720        }
721        ExpressionKind::Quote(_) => expr,
722        ExpressionKind::Unquote(expr) => {
723            ExpressionKind::Unquote(Box::new(remove_interned_in_expression(interner, *expr)))
724        }
725        ExpressionKind::Comptime(block, span) => {
726            let statements =
727                vecmap(block.statements, |stmt| remove_interned_in_statement(interner, stmt));
728            ExpressionKind::Comptime(BlockExpression { statements }, span)
729        }
730        ExpressionKind::Unsafe(UnsafeExpression { block, unsafe_keyword_location }) => {
731            let statements =
732                vecmap(block.statements, |stmt| remove_interned_in_statement(interner, stmt));
733            ExpressionKind::Unsafe(UnsafeExpression {
734                block: BlockExpression { statements },
735                unsafe_keyword_location,
736            })
737        }
738        ExpressionKind::AsTraitPath(mut path) => {
739            path.typ = remove_interned_in_unresolved_type(interner, path.typ);
740            path.trait_generics =
741                remove_interned_in_generic_type_args(interner, path.trait_generics);
742            ExpressionKind::AsTraitPath(path)
743        }
744        ExpressionKind::TypePath(mut path) => {
745            path.typ = remove_interned_in_unresolved_type(interner, path.typ);
746            path.turbofish = path
747                .turbofish
748                .map(|turbofish| remove_interned_in_generic_type_args(interner, turbofish));
749            ExpressionKind::TypePath(path)
750        }
751        ExpressionKind::Resolved(id) => {
752            let expr = interner.expression(&id);
753            expr.to_display_ast(interner, Location::dummy()).kind
754        }
755        ExpressionKind::Interned(id) => {
756            let expr = interner.get_expression_kind(id).clone();
757            remove_interned_in_expression_kind(interner, expr)
758        }
759        ExpressionKind::Error => expr,
760        ExpressionKind::InternedStatement(id) => remove_interned_in_statement_expr(interner, id),
761    }
762}
763
764fn remove_interned_in_statement_expr(
765    interner: &NodeInterner,
766    id: InternedStatementKind,
767) -> ExpressionKind {
768    let expr = match interner.get_statement_kind(id).clone() {
769        StatementKind::Expression(expr) | StatementKind::Semi(expr) => expr.kind,
770        StatementKind::Interned(id) => remove_interned_in_statement_expr(interner, id),
771        _ => ExpressionKind::Error,
772    };
773    remove_interned_in_expression_kind(interner, expr)
774}

776fn remove_interned_in_literal(interner: &NodeInterner, literal: Literal) -> Literal {
777    match literal {
778        Literal::Array(array_literal) => {
779            Literal::Array(remove_interned_in_array_literal(interner, array_literal))
780        }
781        Literal::Vector(array_literal) => {
782            Literal::Array(remove_interned_in_array_literal(interner, array_literal))
783        }
784        Literal::Bool(_)
785        | Literal::Integer(..)
786        | Literal::Str(_)
787        | Literal::RawStr(_, _)
788        | Literal::FmtStr(_, _)
789        | Literal::Unit => literal,
790    }
791}
792
793fn remove_interned_in_array_literal(
794    interner: &NodeInterner,
795    literal: ArrayLiteral,
796) -> ArrayLiteral {
797    match literal {
798        ArrayLiteral::Standard(expressions) => {
799            ArrayLiteral::Standard(vecmap(expressions, |expr| {
800                remove_interned_in_expression(interner, expr)
801            }))
802        }
803        ArrayLiteral::Repeated { repeated_element, length } => ArrayLiteral::Repeated {
804            repeated_element: Box::new(remove_interned_in_expression(interner, *repeated_element)),
805            length: Box::new(remove_interned_in_expression(interner, *length)),
806        },
807    }
808}

810// Returns a new Statement where all Interned statements have been turned into non-interned StatementKind.
811fn remove_interned_in_statement(interner: &NodeInterner, statement: Statement) -> Statement {
812    Statement {
813        kind: remove_interned_in_statement_kind(interner, statement.kind),
814        location: statement.location,
815    }
816}
817
818// Returns a new StatementKind where all Interned statements have been turned into non-interned StatementKind.
819fn remove_interned_in_statement_kind(
820    interner: &NodeInterner,
821    statement: StatementKind,
822) -> StatementKind {
823    match statement {
824        StatementKind::Let(let_statement) => StatementKind::Let(LetStatement {
825            pattern: remove_interned_in_pattern(interner, let_statement.pattern),
826            expression: remove_interned_in_expression(interner, let_statement.expression),
827            r#type: remove_interned_in_option_unresolved_type(interner, let_statement.r#type),
828            ..let_statement
829        }),
830        StatementKind::Expression(expr) => {
831            StatementKind::Expression(remove_interned_in_expression(interner, expr))
832        }
833        StatementKind::Assign(assign) => StatementKind::Assign(AssignStatement {
834            lvalue: assign.lvalue,
835            expression: remove_interned_in_expression(interner, assign.expression),
836        }),
837        StatementKind::For(for_loop) => StatementKind::For(ForLoopStatement {
838            range: match for_loop.range {
839                ForRange::Range(ForBounds { start, end, inclusive }) => {
840                    ForRange::Range(ForBounds {
841                        start: remove_interned_in_expression(interner, start),
842                        end: remove_interned_in_expression(interner, end),
843                        inclusive,
844                    })
845                }
846                ForRange::Array(expr) => {
847                    ForRange::Array(remove_interned_in_expression(interner, expr))
848                }
849            },
850            block: remove_interned_in_expression(interner, for_loop.block),
851            ..for_loop
852        }),
853        StatementKind::Loop(loop_) => StatementKind::Loop(LoopStatement {
854            body: remove_interned_in_expression(interner, loop_.body),
855            loop_keyword_location: loop_.loop_keyword_location,
856        }),
857        StatementKind::While(while_) => StatementKind::While(WhileStatement {
858            condition: remove_interned_in_expression(interner, while_.condition),
859            body: remove_interned_in_expression(interner, while_.body),
860            while_keyword_location: while_.while_keyword_location,
861        }),
862        StatementKind::Comptime(statement) => {
863            StatementKind::Comptime(Box::new(remove_interned_in_statement(interner, *statement)))
864        }
865        StatementKind::Semi(expr) => {
866            StatementKind::Semi(remove_interned_in_expression(interner, expr))
867        }
868        StatementKind::Interned(id) => {
869            let statement = interner.get_statement_kind(id).clone();
870            remove_interned_in_statement_kind(interner, statement)
871        }
872        StatementKind::Break | StatementKind::Continue | StatementKind::Error => statement,
873    }
874}

876// Returns a new LValue where all Interned LValues have been turned into LValue.
877fn remove_interned_in_lvalue(interner: &NodeInterner, lvalue: LValue) -> LValue {
878    match lvalue {
879        LValue::Path(_) => lvalue,
880        LValue::MemberAccess { object, field_name, location: span } => LValue::MemberAccess {
881            object: Box::new(remove_interned_in_lvalue(interner, *object)),
882            field_name,
883            location: span,
884        },
885        LValue::Index { array, index, location: span } => LValue::Index {
886            array: Box::new(remove_interned_in_lvalue(interner, *array)),
887            index: remove_interned_in_expression(interner, index),
888            location: span,
889        },
890        LValue::Dereference(lvalue, span) => {
891            LValue::Dereference(Box::new(remove_interned_in_lvalue(interner, *lvalue)), span)
892        }
893        LValue::Interned(id, span) => {
894            let lvalue = interner.get_lvalue(id, span);
895            remove_interned_in_lvalue(interner, lvalue)
896        }
897    }
898}

901    interner: &NodeInterner,

907fn remove_interned_in_unresolved_type(
908    interner: &NodeInterner,
909    typ: UnresolvedType,
910) -> UnresolvedType {
911    UnresolvedType {
912        typ: remove_interned_in_unresolved_type_data(interner, typ.typ),
913        location: typ.location,
914    }
915}

917fn remove_interned_in_unresolved_type_data(
918    interner: &NodeInterner,
919    typ: UnresolvedTypeData,
920) -> UnresolvedTypeData {
921    match typ {
922        UnresolvedTypeData::Array(expr, typ) => UnresolvedTypeData::Array(
923            expr,
924            Box::new(remove_interned_in_unresolved_type(interner, *typ)),
925        ),
926        UnresolvedTypeData::Vector(typ) => {
927            UnresolvedTypeData::Vector(Box::new(remove_interned_in_unresolved_type(interner, *typ)))
928        }
929        UnresolvedTypeData::Parenthesized(typ) => UnresolvedTypeData::Parenthesized(Box::new(
930            remove_interned_in_unresolved_type(interner, *typ),
931        )),
932        UnresolvedTypeData::Named(path, generic_type_args, is_synthesized) => {
933            UnresolvedTypeData::Named(
934                path,
935                remove_interned_in_generic_type_args(interner, generic_type_args),
936                is_synthesized,
937            )
938        }
939        UnresolvedTypeData::TraitAsType(path, generic_type_args) => {
940            UnresolvedTypeData::TraitAsType(
941                path,
942                remove_interned_in_generic_type_args(interner, generic_type_args),
943            )
944        }
945        UnresolvedTypeData::Reference(typ, mutable) => UnresolvedTypeData::Reference(
946            Box::new(remove_interned_in_unresolved_type(interner, *typ)),
947            mutable,
948        ),
949        UnresolvedTypeData::Tuple(types) => UnresolvedTypeData::Tuple(vecmap(types, |typ| {
950            remove_interned_in_unresolved_type(interner, typ)
951        })),
952        UnresolvedTypeData::Function(arg_types, ret_type, env_type, unconstrained) => {
953            UnresolvedTypeData::Function(
954                vecmap(arg_types, |typ| remove_interned_in_unresolved_type(interner, typ)),
955                Box::new(remove_interned_in_unresolved_type(interner, *ret_type)),
956                Box::new(remove_interned_in_unresolved_type(interner, *env_type)),
957                unconstrained,
958            )
959        }
960        UnresolvedTypeData::AsTraitPath(as_trait_path) => {
961            UnresolvedTypeData::AsTraitPath(Box::new(AsTraitPath {
962                typ: remove_interned_in_unresolved_type(interner, as_trait_path.typ),
963                trait_generics: remove_interned_in_generic_type_args(
964                    interner,
965                    as_trait_path.trait_generics,
966                ),
967                ..*as_trait_path
968            }))
969        }
970        UnresolvedTypeData::Interned(id) => interner.get_unresolved_type_data(id).clone(),
971        UnresolvedTypeData::Unit
972        | UnresolvedTypeData::Resolved(_)
973        | UnresolvedTypeData::Expression(_)
974        | UnresolvedTypeData::Error => typ,
975    }
976}

978fn remove_interned_in_generic_type_args(
979    interner: &NodeInterner,
980    args: GenericTypeArgs,
981) -> GenericTypeArgs {
982    GenericTypeArgs {
983        ordered_args: vecmap(args.ordered_args, |typ| {
984            remove_interned_in_unresolved_type(interner, typ)
985        }),
986        named_args: vecmap(args.named_args, |(name, typ)| {
987            (name, remove_interned_in_unresolved_type(interner, typ))
988        }),
989        kinds: args.kinds,
990    }
991}

993// Returns a new Pattern where all Interned Patterns have been turned into Pattern.
994fn remove_interned_in_pattern(interner: &NodeInterner, pattern: Pattern) -> Pattern {
995    match pattern {
996        Pattern::Identifier(_) => pattern,
997        Pattern::Mutable(pattern, location, is_synthesized) => Pattern::Mutable(
998            Box::new(remove_interned_in_pattern(interner, *pattern)),
999            location,
1000            is_synthesized,
1001        ),
1002        Pattern::Tuple(patterns, location) => Pattern::Tuple(
1003            vecmap(patterns, |pattern| remove_interned_in_pattern(interner, pattern)),
1004            location,
1005        ),
1006        Pattern::Struct(path, patterns, location) => {
1007            let patterns = vecmap(patterns, |(name, pattern)| {
1008                (name, remove_interned_in_pattern(interner, pattern))
1009            });
1010            Pattern::Struct(path, patterns, location)
1011        }
1012        Pattern::Parenthesized(pattern, location) => Pattern::Parenthesized(
1013            Box::new(remove_interned_in_pattern(interner, *pattern)),
1014            location,
1015        ),
1016        Pattern::Interned(id, _) => interner.get_pattern(id).clone(),
1017    }
