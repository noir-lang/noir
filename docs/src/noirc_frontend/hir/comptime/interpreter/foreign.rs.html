29        &self,
30        name: &str,
31        arguments: Vec<(Value, Location)>,
32        return_type: Type,

39/// Calls the given foreign function.
40///
41/// Similar to `evaluate_black_box` in `brillig_vm`.
42fn call_foreign(
43    name: &str,
44    args: Vec<(Value, Location)>,
45    return_type: Type,
46    location: Location,
47) -> IResult<Value> {
48    match name {
49        "aes128_encrypt" => aes128_encrypt(args, location),
50        "blake2s" => blake_hash(args, location, acvm::blackbox_solver::blake2s),
51        "blake3" => blake_hash(args, location, acvm::blackbox_solver::blake3),
52        // cSpell:disable-next-line
53        "ecdsa_secp256k1" => {
54            ecdsa_secp256_verify(args, location, acvm::blackbox_solver::ecdsa_secp256k1_verify)
55        }
56        // cSpell:disable-next-line
57        "ecdsa_secp256r1" => {
58            ecdsa_secp256_verify(args, location, acvm::blackbox_solver::ecdsa_secp256r1_verify)
59        }
60        "embedded_curve_add" => embedded_curve_add(args, return_type, location),
61        "multi_scalar_mul" => multi_scalar_mul(args, return_type, location),
62        "poseidon2_permutation" => poseidon2_permutation(args, location),
63        "keccakf1600" => keccakf1600(args, location),
64        "sha256_compression" => sha256_compression(args, location),
65        _ => {
66            let explanation = match name {
67                "and" | "xor" => "It should be turned into a binary operation.".into(),
68                "recursive_aggregation" => "A proof cannot be verified at comptime.".into(),
69                _ => {
70                    let item = format!("Comptime evaluation for foreign function '{name}'");
71                    return Err(InterpreterError::Unimplemented { item, location });
72                }
73            };
74
75            let item = format!("Attempting to evaluate foreign function '{name}'");
76            Err(InterpreterError::InvalidInComptimeContext { item, location, explanation })
77        }
78    }
79}

81/// `pub fn aes128_encrypt<let N: u32>(input: [u8; N], iv: [u8; 16], key: [u8; 16]) -> [u8]`
82fn aes128_encrypt(arguments: Vec<(Value, Location)>, location: Location) -> IResult<Value> {
83    let (inputs, iv, key) = check_three_arguments(arguments, location)?;
84
85    let (inputs, _) = get_array_map(inputs, get_u8)?;
86    let (iv, _) = get_fixed_array_map(iv, get_u8)?;
87    let (key, _) = get_fixed_array_map(key, get_u8)?;
88
89    let output = acvm::blackbox_solver::aes128_encrypt(&inputs, iv, key)
90        .map_err(|e| InterpreterError::BlackBoxError(e, location))?;
91
92    Ok(to_byte_array(&output))
93}

95/// Run one of the Blake hash functions.
96/// ```text
97/// pub fn blake2s<let N: u32>(input: [u8; N]) -> [u8; 32]
98/// pub fn blake3<let N: u32>(input: [u8; N]) -> [u8; 32]
99/// ```
100fn blake_hash(
101    arguments: Vec<(Value, Location)>,
102    location: Location,
103    f: impl Fn(&[u8]) -> Result<[u8; 32], BlackBoxResolutionError>,
104) -> IResult<Value> {
105    let inputs = check_one_argument(arguments, location)?;
106
107    let (inputs, _) = get_array_map(inputs, get_u8)?;
108    let output = f(&inputs).map_err(|e| InterpreterError::BlackBoxError(e, location))?;
109
110    Ok(to_byte_array(&output))
111}

114/// Run one of the Secp256 signature verifications.
115/// ```text
116/// pub fn _verify_signature(
117///     public_key_x: [u8; 32],
118///     public_key_y: [u8; 32],
119///     signature: [u8; 64],
120///     message_hash: [u8; 32],
121///     predicate: bool,
122/// ) -> bool
123// cSpell:disable-next-line
124fn ecdsa_secp256_verify(
125    arguments: Vec<(Value, Location)>,
126    location: Location,
127    f: impl Fn(&[u8; 32], &[u8; 32], &[u8; 32], &[u8; 64]) -> Result<bool, BlackBoxResolutionError>,
128) -> IResult<Value> {
129    let [pub_key_x, pub_key_y, sig, msg_hash, predicate] = check_arguments(arguments, location)?;
130    assert_eq!(predicate.0, Value::Bool(true), "verify_signature predicate should be true");
131
132    let (pub_key_x, _) = get_fixed_array_map(pub_key_x, get_u8)?;
133    let (pub_key_y, _) = get_fixed_array_map(pub_key_y, get_u8)?;
134    let (sig, _) = get_fixed_array_map(sig, get_u8)?;
135    let (msg_hash, _) = get_fixed_array_map(msg_hash.clone(), get_u8)?;
136
137    let is_valid = f(&msg_hash, &pub_key_x, &pub_key_y, &sig)
138        .map_err(|e| InterpreterError::BlackBoxError(e, location))?;
139
140    Ok(Value::Bool(is_valid))
141}

143/// ```text
144/// fn embedded_curve_add(
145///     point1: EmbeddedCurvePoint,
146///     point2: EmbeddedCurvePoint,
147/// ) -> [EmbeddedCurvePoint; 1]
148/// ```
149fn embedded_curve_add(
150    arguments: Vec<(Value, Location)>,
151    return_type: Type,
152    location: Location,
153) -> IResult<Value> {
154    let (point1, point2, predicate) = check_three_arguments(arguments, location)?;
155    assert_eq!(predicate.0, Value::Bool(true), "ec_add predicate should be true");
156
157    let embedded_curve_point_typ = point1.0.get_type().into_owned();
158
159    let (p1x, p1y, p1inf) = get_embedded_curve_point(point1)?;
160    let (p2x, p2y, p2inf) = get_embedded_curve_point(point2)?;
161
162    let (x, y, inf) = Bn254BlackBoxSolver
163        .ec_add(
164            &p1x,
165            &p1y,
166            &p1inf.into(),
167            &p2x,
168            &p2y,
169            &p2inf.into(),
170            true, // Predicate is always true as interpreter has control flow to handle false case
171        )
172        .map_err(|e| InterpreterError::BlackBoxError(e, location))?;
173
174    Ok(Value::Array(
175        vector![to_embedded_curve_point(x, y, inf > 0_usize.into(), embedded_curve_point_typ)],
176        return_type,
177    ))
178}

180/// ```text
181/// pub fn multi_scalar_mul<let N: u32>(
182///     points: [EmbeddedCurvePoint; N],
183///     scalars: [EmbeddedCurveScalar; N],
184///     predicate: bool,
185/// ) -> [EmbeddedCurvePoint; 1]
186/// ```
187fn multi_scalar_mul(
188    arguments: Vec<(Value, Location)>,
189    return_type: Type,
190    location: Location,
191) -> IResult<Value> {
192    let (points, scalars, predicate) = check_three_arguments(arguments, location)?;
193    assert_eq!(predicate.0, Value::Bool(true), "multi_scalar_mul predicate should be true");
194
195    let (points, _) = get_array_map(points, get_embedded_curve_point)?;
196    let (scalars, _) = get_array_map(scalars, get_embedded_curve_scalar)?;
197
198    let points: Vec<_> = points.into_iter().flat_map(|(x, y, inf)| [x, y, inf.into()]).collect();
199    let mut scalars_lo = Vec::new();
200    let mut scalars_hi = Vec::new();
201    for (lo, hi) in scalars {
202        scalars_lo.push(lo);
203        scalars_hi.push(hi);
204    }
205
206    let (x, y, inf) = Bn254BlackBoxSolver
207        .multi_scalar_mul(
208            &points,
209            &scalars_lo,
210            &scalars_hi,
211            true, // Predicate is always true as interpreter has control flow to handle false case
212        )
213        .map_err(|e| InterpreterError::BlackBoxError(e, location))?;
214
215    let embedded_curve_point_typ = match &return_type {
216        Type::Array(_, item_type) => item_type.as_ref().clone(),
217        _ => {
218            return Err(InterpreterError::TypeMismatch {
219                expected: "[EmbeddedCurvePoint; 1]".to_string(),
220                actual: return_type.clone(),
221                location,
222            });
223        }
224    };
225
226    Ok(Value::Array(
227        vector![to_embedded_curve_point(x, y, inf > 0_usize.into(), embedded_curve_point_typ)],
228        return_type,
229    ))
230}

232/// `poseidon2_permutation<let N: u32>(_input: [Field; N], _state_length: u32) -> [Field; N]`
233fn poseidon2_permutation(arguments: Vec<(Value, Location)>, location: Location) -> IResult<Value> {
234    let input = check_one_argument(arguments, location)?;
235
236    let (input, typ) = get_array_map(input, get_field)?;
237    let input = vecmap(input, SignedField::to_field_element);
238
239    let fields = Bn254BlackBoxSolver
240        .poseidon2_permutation(&input)
241        .map_err(|error| InterpreterError::BlackBoxError(error, location))?;
242
243    let array = fields.into_iter().map(|f| Value::Field(SignedField::positive(f))).collect();
244    Ok(Value::Array(array, typ))
245}

247/// `fn keccakf1600(input: [u64; 25]) -> [u64; 25] {}`
248fn keccakf1600(arguments: Vec<(Value, Location)>, location: Location) -> IResult<Value> {
249    let input = check_one_argument(arguments, location)?;
250
251    let (state, typ) = get_fixed_array_map(input, get_u64)?;
252
253    let result_lanes = acvm::blackbox_solver::keccakf1600(state)
254        .map_err(|error| InterpreterError::BlackBoxError(error, location))?;
255
256    let array: Vector<Value> = result_lanes.into_iter().map(Value::U64).collect();
257    Ok(Value::Array(array, typ))
258}

260/// `pub fn sha256_compression(input: [u32; 16], state: [u32; 8]) -> [u32; 8]`
261fn sha256_compression(arguments: Vec<(Value, Location)>, location: Location) -> IResult<Value> {
262    let (input, state) = check_two_arguments(arguments, location)?;
263
264    let (input, _) = get_fixed_array_map(input, get_u32)?;
265    let (mut state, typ) = get_fixed_array_map(state, get_u32)?;
266
267    acvm::blackbox_solver::sha256_compression(&mut state, &input);
268
269    let state = state.into_iter().map(Value::U32).collect();
270    Ok(Value::Array(state, typ))
271}

273/// Decode an `EmbeddedCurvePoint` struct.
274///
275/// Returns `(x, y, is_infinite)`.
276fn get_embedded_curve_point(
277    (value, location): (Value, Location),
278) -> IResult<(FieldElement, FieldElement, bool)> {
279    let (fields, typ) = get_struct_fields("EmbeddedCurvePoint", (value, location))?;
280    let x = get_struct_field("x", &fields, &typ, location, get_field)?;
281    let y = get_struct_field("y", &fields, &typ, location, get_field)?;
282    let is_infinite = get_struct_field("is_infinite", &fields, &typ, location, get_bool)?;
283    Ok((x.to_field_element(), y.to_field_element(), is_infinite))
284}

286/// Decode an `EmbeddedCurveScalar` struct.
287///
288/// Returns `(lo, hi)`.
289fn get_embedded_curve_scalar(
290    (value, location): (Value, Location),
291) -> IResult<(FieldElement, FieldElement)> {
292    let (fields, typ) = get_struct_fields("EmbeddedCurveScalar", (value, location))?;
293    let lo = get_struct_field("lo", &fields, &typ, location, get_field)?;
294    let hi = get_struct_field("hi", &fields, &typ, location, get_field)?;
295    Ok((lo.to_field_element(), hi.to_field_element()))
296}
297
298fn to_embedded_curve_point(
299    x: FieldElement,
300    y: FieldElement,
301    is_infinite: bool,
302    typ: Type,
303) -> Value {
304    to_struct(
305        [
306            ("x", Value::Field(SignedField::positive(x))),
307            ("y", Value::Field(SignedField::positive(y))),
308            ("is_infinite", Value::Bool(is_infinite)),
309        ],
310        typ,
311    )
312}

314#[cfg(test)]
315mod tests {
316    use acvm::acir::BlackBoxFunc;
317    use noirc_errors::Location;
318    use strum::IntoEnumIterator;
319
320    use crate::Type;
321    use crate::hir::comptime::InterpreterError::{
322        ArgumentCountMismatch, InvalidInComptimeContext, Unimplemented,
323    };
324
325    use super::call_foreign;
326
327    /// Check that all `BlackBoxFunc` are covered by `call_foreign`.
328    #[test]
329    fn test_blackbox_implemented() {
330        let no_location = Location::dummy();
331        let mut not_implemented = Vec::new();
332
333        for blackbox in BlackBoxFunc::iter() {
334            // There's no implementation for RANGE as it's actually a builtin function,
335            // and in the comptime interpreter it's not transformed to a foreign function call
336            if blackbox == BlackBoxFunc::RANGE {
337                continue;
338            }
339
340            let name = blackbox.name();
341            match call_foreign(name, Vec::new(), Type::Unit, no_location) {
342                Ok(_) => {
343                    // Exists and works with no args (unlikely)
344                }
345                Err(ArgumentCountMismatch { .. }) => {
346                    // Exists but doesn't work with no args (expected)
347                }
348                Err(InvalidInComptimeContext { .. }) => {}
349                Err(Unimplemented { .. }) => not_implemented.push(name),
350                Err(other) => panic!("unexpected error: {other:?}"),
351            }
352        }
353
354        assert!(
355            not_implemented.is_empty(),
356            "unimplemented blackbox functions: {not_implemented:?}"
357        );
358    }
