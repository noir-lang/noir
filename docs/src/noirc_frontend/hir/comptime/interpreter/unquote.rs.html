10impl Interpreter<'_, '_> {
11    /// Evaluates any expressions within UnquoteMarkers in the given token list
12    /// and replaces the expression held by the marker with the evaluated value
13    /// in expression form.
14    pub(super) fn substitute_unquoted_values_into_tokens(
15        &mut self,
16        tokens: Tokens,
17    ) -> IResult<Vec<LocatedToken>> {
18        let mut new_tokens = Vec::with_capacity(tokens.0.len());
19        let mut tokens = tokens.0.into_iter();
20
21        while let Some(token) = tokens.next() {
22            let location = token.location();
23            match token.into_token() {
24                Token::UnquoteMarker(id) => {
25                    let value = self.evaluate(id)?;
26                    let tokens = value.into_tokens(self.elaborator.interner, location)?;
27                    new_tokens.extend(tokens);
28                }
29                Token::Quote(tokens) => {
30                    // Make sure to substitute in nested `quote { ... }` as well.
31                    let tokens = self.substitute_unquoted_values_into_tokens(tokens)?;
32                    new_tokens.push(LocatedToken::new(Token::Quote(Tokens(tokens)), location));
33                }
34                Token::Backslash => {
35                    Self::escape_token(&mut tokens, &mut new_tokens, location)?;
36                }
37                token => new_tokens.push(LocatedToken::new(token, location)),
38            }
39        }
40
41        Ok(new_tokens)
42    }
43
44    /// Escape the next token in the quoted token stream, issuing an error if it is not `$`.
45    /// If it is `$`, it is appended to the `new_tokens` Vec.
46    pub(crate) fn escape_token(
47        tokens: &mut impl Iterator<Item = LocatedToken>,
48        new_tokens: &mut Vec<LocatedToken>,
49        default_location: Location,
50    ) -> IResult<()> {
51        match tokens.next() {
52            Some(token) if *token.token() == Token::DollarSign => {
53                new_tokens.push(token);
54                Ok(())
55            }
56            Some(token) => {
57                let location = token.location();
58                let token = Some(token.into_token());
59                Err(InterpreterError::UnexpectedEscapedTokenInQuote { token, location })
60            }
61            None => Err(InterpreterError::UnexpectedEscapedTokenInQuote {
62                token: None,
63                location: default_location,
64            }),
65        }
66    }
67}
