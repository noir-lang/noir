48pub(crate) fn check_argument_count(
49    expected: usize,
50    arguments: &[(Value, Location)],
51    location: Location,
52) -> IResult<()> {
53    if arguments.len() == expected {
54        Ok(())
55    } else {
56        let actual = arguments.len();
57        Err(InterpreterError::ArgumentCountMismatch { expected, actual, location })
58    }
59}

125/// Get a specific field of a struct and apply a decoder function on it.
126pub(crate) fn get_struct_field<T>(
127    field_name: &str,
128    struct_fields: &HashMap<Rc<String>, Shared<Value>>,
129    struct_type: &Type,
130    location: Location,
131    f: impl Fn((Value, Location)) -> IResult<T>,
132) -> IResult<T> {
133    let key = field_name.to_string();
134    let Some(value) = struct_fields.get(&key) else {
135        return Err(InterpreterError::ExpectedStructToHaveField {
136            typ: struct_type.clone(),
137            field_name: key,
138            location,
139        });
140    };
141    f((value.borrow().clone(), location))
142}

163/// Interpret the input as an array, then map each element.
164/// Returns the values in the array and the original array type.
165pub(crate) fn get_array_map<T>(
166    (value, location): (Value, Location),
167    f: impl Fn((Value, Location)) -> IResult<T>,
168) -> IResult<(Vec<T>, Type)> {
169    let (values, typ) = get_array((value, location))?;
170    let values = try_vecmap(values, |value| f((value, location)))?;
171    Ok((values, typ))
172}

174/// Get an array and convert it to a fixed size.
175/// Returns the values in the array and the original array type.
176pub(crate) fn get_fixed_array_map<T, const N: usize>(
177    (value, location): (Value, Location),
178    f: impl Fn((Value, Location)) -> IResult<T>,
179) -> IResult<([T; N], Type)> {
180    let (values, typ) = get_array_map((value, location), f)?;
181
182    values.try_into().map(|v| (v, typ.clone())).map_err(|_| {
183        // Assuming that `values.len()` corresponds to `typ`.
184        let Type::Array(_, ref elem) = typ else {
185            unreachable!("get_array_map checked it was an array")
186        };
187        let len: u32 =
188            N.try_into().expect("ICE: get_fixed_array_map: N is expected to fit into a u32");
189        let expected = Type::Array(Box::new(len.into()), elem.clone()).to_string();
190        InterpreterError::TypeMismatch { expected, actual: typ, location }
191    })
192}

259    interner: &NodeInterner,
260    (value, location): (Value, Location),
261) -> IResult<ExprValue> {
262    match value {
263        Value::Expr(expr) => match *expr {
264            ExprValue::Expression(ExpressionKind::Interned(id)) => {
265                Ok(ExprValue::Expression(interner.get_expression_kind(id).clone()))
266            }
267            ExprValue::Statement(StatementKind::Interned(id)) => {
268                Ok(ExprValue::Statement(interner.get_statement_kind(id).clone()))
269            }
270            ExprValue::LValue(LValue::Interned(id, _)) => {
271                Ok(ExprValue::LValue(interner.get_lvalue(id, location).clone()))
272            }
273            ExprValue::Pattern(Pattern::Interned(id, _)) => {
274                Ok(ExprValue::Pattern(interner.get_pattern(id).clone()))
275            }
276            _ => Ok(*expr),
277        },
278        value => type_mismatch(value, Type::Quoted(QuotedType::Expr), location),
279    }

396/// Helper function to report an [InterpreterError::TypeMismatch] where a [Value] could not be unwrapped
397/// into an expected type, which can either be a concrete [Type] or a textual description, e.g. "tuple".
398///
399/// It has special handling for [Value::Zeroed], which wraps a [Type] that might actually be the
400/// one we expected, but we expected a concrete _value_ with that type, not a zeroed placeholder,
401/// which we optimistically created expecting that it will never be used. In such cases
402/// [InterpreterError::UnexpectedZeroedValue] is returned.
403fn type_mismatch<T>(
404    value: Value,
405    expected: impl Into<TypeOrString>,
406    location: Location,
407) -> IResult<T> {
408    let actual = value.get_type().into_owned();
409    let expected = match expected.into() {
410        TypeOrString::Type(t) if t == actual && matches!(value, Value::Zeroed(_)) => {
411            return Err(InterpreterError::UnexpectedZeroedValue {
412                expected: t.to_string(),
413                location,
414            });
415        }
416        TypeOrString::Type(t) => t.to_string(),
417        TypeOrString::String(s) => s,
418    };
419    Err(InterpreterError::TypeMismatch { expected, actual, location })
420}

422pub(crate) fn hir_pattern_to_tokens(
423    interner: &NodeInterner,
424    hir_pattern: &HirPattern,
425) -> Vec<Token> {
426    let mut tokens = Vec::new();
427    gather_hir_pattern_tokens(interner, hir_pattern, &mut tokens);
428    tokens
429}
430
431fn gather_hir_pattern_tokens(
432    interner: &NodeInterner,
433    hir_pattern: &HirPattern,
434    tokens: &mut Vec<Token>,
435) {
436    match hir_pattern {
437        HirPattern::Identifier(hir_ident) => {
438            let name = interner.definition_name(hir_ident.id).to_string();
439            tokens.push(Token::Ident(name));
440        }
441        HirPattern::Mutable(pattern, _) => {
442            tokens.push(Token::Keyword(Keyword::Mut));
443            gather_hir_pattern_tokens(interner, pattern, tokens);
444        }
445        HirPattern::Tuple(patterns, _) => {
446            tokens.push(Token::LeftParen);
447            for (index, pattern) in patterns.iter().enumerate() {
448                if index != 0 {
449                    tokens.push(Token::Comma);
450                }
451                gather_hir_pattern_tokens(interner, pattern, tokens);
452            }
453            tokens.push(Token::RightParen);
454        }
455        HirPattern::Struct(typ, fields, _) => {
456            let Type::DataType(struct_type, _) = typ.follow_bindings() else {
457                panic!("Expected type to be a struct");
458            };
459
460            let name = struct_type.borrow().name.to_string();
461            tokens.push(Token::Ident(name));
462
463            tokens.push(Token::LeftBrace);
464            for (index, (field_name, pattern)) in fields.iter().enumerate() {
465                if index != 0 {
466                    tokens.push(Token::Comma);
467                }
468
469                let field_name = field_name.as_str();
470                tokens.push(Token::Ident(field_name.to_string()));
471
472                // If we have a pattern like `Foo { x }`, that's internally represented as `Foo { x: x }` so
473                // here we check if the field name is the same as the pattern and, if so, omit the `: x` part.
474                let field_name_is_same_as_pattern = if let HirPattern::Identifier(pattern) = pattern
475                {
476                    let pattern_name = interner.definition_name(pattern.id);
477                    field_name == pattern_name
478                } else {
479                    false
480                };
481
482                if !field_name_is_same_as_pattern {
483                    tokens.push(Token::Colon);
484                    gather_hir_pattern_tokens(interner, pattern, tokens);
485                }
486            }
487            tokens.push(Token::RightBrace);
488        }
489    }
490}

492/// If the given `item_module`'s crate does not match the crate the interpreter is in, issue an
493/// error to prevent modifying an item from an external crate.
494pub(super) fn check_item_crate_matches_current_crate(
495    interpreter: &Interpreter,
496    item: &Value,
497    item_module: ModuleId,
498    location: Location,
499) -> IResult<()> {
500    let current_crate = interpreter.elaborator.module_id().krate;
501    if current_crate != item_module.krate {
502        let module = fully_qualified_module_path(
503            interpreter.elaborator.def_maps,
504            interpreter.elaborator.crate_graph,
505            &current_crate,
506            item_module,
507        );
508        let item = item.display(interpreter.elaborator.interner).to_string();
509        Err(InterpreterError::CannotModifyExternalItem { item, module, location })
510    } else {
511        Ok(())
512    }
513}

529pub(super) fn lex(input: &str, location: Location) -> Vec<LocatedToken> {
530    let (tokens, _) = Lexer::lex(input, location.file);
531    let mut tokens: Vec<_> =
532        tokens.0.into_iter().map(|token| LocatedToken::new(token.into_token(), location)).collect();
533    if let Some(Token::EOF) = tokens.last().map(|t| t.token()) {
534        tokens.pop();
535    }
536    tokens
537}

539pub(super) fn parse<'a, T, F>(
540    elaborator: &mut Elaborator,
541    (value, location): (Value, Location),
542    parser: F,
543    rule: &'static str,
544) -> IResult<T>
545where
546    F: FnOnce(&mut Parser<'a>) -> T,
547{
548    let tokens = get_quoted((value, location))?;
549    let quoted = Tokens(unwrap_rc(tokens.clone()));
550    let (result, warnings) =
551        parse_tokens(tokens, quoted, elaborator.interner, location, parser, rule)?;
552    for warning in warnings {
553        let warning: CompilationError = warning.into();
554        elaborator.push_err(warning);
555    }
556    Ok(result)
557}

559pub(super) fn parse_tokens<'a, T, F>(
560    tokens: Rc<Vec<LocatedToken>>,
561    quoted: Tokens,
562    interner: &NodeInterner,
563    location: Location,
564    parsing_function: F,
565    rule: &'static str,
566) -> IResult<(T, Vec<ParserError>)>
567where
568    F: FnOnce(&mut Parser<'a>) -> T,
569{
570    Parser::for_tokens(quoted).parse_result(parsing_function).map_err(|errors| {
571        let error = errors
572            .into_iter()
573            .find(|error| !error.is_warning())
574            .expect("there is at least 1 error");
575        let error = Box::new(error);
576        let tokens = tokens_to_string(&tokens, interner);
577        InterpreterError::FailedToParseMacro { error, tokens, rule, location }
578    })
579}

622pub(super) fn has_named_attribute(

625    interner: &NodeInterner,

627    for attribute in attributes {

638fn secondary_attribute_name(
639    attribute: &SecondaryAttribute,
640    interner: &NodeInterner,
641) -> Option<String> {
642    match &attribute.kind {
643        SecondaryAttributeKind::Deprecated(_) => Some("deprecated".to_string()),
644        SecondaryAttributeKind::ContractLibraryMethod => {
645            Some("contract_library_method".to_string())
646        }
647        SecondaryAttributeKind::Export => Some("export".to_string()),
648        SecondaryAttributeKind::Field(_) => Some("field".to_string()),
649        SecondaryAttributeKind::Tag(contents) => {
650            let mut lexer = Lexer::new_with_dummy_file(contents);
651            let token = lexer.next()?.ok()?;
652            if let Token::Ident(ident) = token.into_token() { Some(ident) } else { None }
653        }
654        SecondaryAttributeKind::Meta(meta) => interner.get_meta_attribute_name(meta),
655        SecondaryAttributeKind::Abi(_) => Some("abi".to_string()),
656        SecondaryAttributeKind::Varargs => Some("varargs".to_string()),
657        SecondaryAttributeKind::UseCallersScope => Some("use_callers_scope".to_string()),
658        SecondaryAttributeKind::Allow(_) => Some("allow".to_string()),
659        SecondaryAttributeKind::MustUse(_) => Some("must_use".to_string()),
660    }
661}

673pub(super) fn hash_item<T: Hash>(
674    arguments: Vec<(Value, Location)>,
675    location: Location,
676    get_item: impl FnOnce((Value, Location)) -> IResult<T>,
677) -> IResult<Value> {
678    let argument = check_one_argument(arguments, location)?;
679    let item = get_item(argument)?;
680
681    let mut hasher = std::collections::hash_map::DefaultHasher::new();
682    item.hash(&mut hasher);
683    let hash = hasher.finish();
684    Ok(Value::Field(SignedField::positive(u128::from(hash))))
685}
686
687pub(super) fn eq_item<T: Eq>(
688    arguments: Vec<(Value, Location)>,
689    location: Location,
690    mut get_item: impl FnMut((Value, Location)) -> IResult<T>,
691) -> IResult<Value> {
692    let (self_arg, other_arg) = check_two_arguments(arguments, location)?;
693    let self_arg = get_item(self_arg)?;
694    let other_arg = get_item(other_arg)?;
695    Ok(Value::Bool(self_arg == other_arg))
696}

698/// Type to be used in `Value::Array(<values>, <array-type>)`.
699pub(crate) fn byte_array_type(len: usize) -> Type {
700    let len: u32 = len.try_into().expect("ICE: byte_array_type: N is expected to fit into a u32");
701    Type::Array(
702        Box::new(len.into()),
703        Box::new(Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight)),
704    )
705}

708pub(crate) fn to_byte_array(values: &[u8]) -> Value {
709    Value::Array(values.iter().copied().map(Value::U8).collect(), byte_array_type(values.len()))

722pub(crate) fn new_unary_op(operator: UnaryOp, typ: Type) -> Option<Value> {
723    // These values should match the values used in noir_stdlib/src/meta/op.nr
724    let unary_op_value: u128 = match operator {
725        UnaryOp::Minus => 0,
726        UnaryOp::Not => 1,
727        UnaryOp::Reference { mutable: true } => 2,
728        UnaryOp::Reference { mutable: false } => {
729            // `&` alone is experimental and currently hidden from the comptime API
730            return None;
731        }
732        UnaryOp::Dereference { .. } => 3,
733    };
734
735    let mut fields = HashMap::default();
736    fields.insert(
737        Rc::new("op".to_string()),
738        Shared::new(Value::Field(SignedField::positive(unary_op_value))),
739    );
740
741    Some(Value::Struct(fields, typ))
742}

744pub(crate) fn new_binary_op(operator: BinaryOp, typ: Type) -> Value {
745    // For the op value we use the enum member index, which should match noir_stdlib/src/meta/op.nr
746    let binary_op_value = operator.contents as u128;
747
748    let mut fields = HashMap::default();
749    fields.insert(
750        Rc::new("op".to_string()),
751        Shared::new(Value::Field(SignedField::positive(binary_op_value))),
752    );
753
754    Value::Struct(fields, typ)
755}

757pub(crate) fn visibility_to_quoted(visibility: ItemVisibility, location: Location) -> Value {
758    let tokens = match visibility {
759        ItemVisibility::Private => vec![],
760        ItemVisibility::PublicCrate => vec![
761            Token::Keyword(Keyword::Pub),
762            Token::LeftParen,
763            Token::Keyword(Keyword::Crate),
764            Token::RightParen,
765        ],
766        ItemVisibility::Public => vec![Token::Keyword(Keyword::Pub)],
767    };
768    let tokens = vecmap(tokens, |token| LocatedToken::new(token, location));
769    Value::Quoted(Rc::new(tokens))
770}

772pub(crate) fn fragments_to_string(
773    fragments: &[FormatStringFragment],
774    interner: &NodeInterner,
775) -> String {
776    let mut result = String::new();
777    for fragment in fragments {
778        match fragment {
779            FormatStringFragment::String(string) => {
780                result.push_str(string);
781            }
782            FormatStringFragment::Value { name: _, value } => {
783                match value {
784                    Value::Quoted(tokens) => {
785                        // When interpolating a quoted value inside a format string, we don't include the
786                        // surrounding `quote {` ... `}` as if we are unquoting the quoted value inside the string.
787                        for (index, token) in tokens.iter().enumerate() {
788                            if index > 0 {
789                                result.push(' ');
790                            }
791                            result.push_str(&token.token().display(interner).to_string());
792                        }
793                    }
794                    Value::FormatString(fragments, _, _) => {
795                        // Nested format strings might have quoted values inside them,
796                        // so we need to recurse here instead of calling `value.display`.
797                        let inner_string = fragments_to_string(fragments, interner);
798                        result.push_str(&inner_string);
799                    }
800                    _ => {
801                        result.push_str(&value.display(interner).to_string());
802                    }
803                }
804            }
805        }
806    }
807    result
808}
