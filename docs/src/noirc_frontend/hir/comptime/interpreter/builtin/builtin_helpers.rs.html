46pub(crate) fn check_argument_count(
47    expected: usize,
48    arguments: &[(Value, Location)],
49    location: Location,
50) -> IResult<()> {
51    if arguments.len() == expected {
52        Ok(())
53    } else {
54        let actual = arguments.len();
55        Err(InterpreterError::ArgumentCountMismatch { expected, actual, location })
56    }
57}

123/// Get a specific field of a struct and apply a decoder function on it.
124pub(crate) fn get_struct_field<T>(
125    field_name: &str,
126    struct_fields: &HashMap<Rc<String>, Shared<Value>>,
127    struct_type: &Type,
128    location: Location,
129    f: impl Fn((Value, Location)) -> IResult<T>,
130) -> IResult<T> {
131    let key = field_name.to_string();
132    let Some(value) = struct_fields.get(&key) else {
133        return Err(InterpreterError::ExpectedStructToHaveField {
134            typ: struct_type.clone(),
135            field_name: key,
136            location,
137        });
138    };
139    f((value.borrow().clone(), location))
140}

161/// Interpret the input as an array, then map each element.
162/// Returns the values in the array and the original array type.
163pub(crate) fn get_array_map<T>(
164    (value, location): (Value, Location),
165    f: impl Fn((Value, Location)) -> IResult<T>,
166) -> IResult<(Vec<T>, Type)> {
167    let (values, typ) = get_array((value, location))?;
168    let values = try_vecmap(values, |value| f((value, location)))?;
169    Ok((values, typ))
170}

172/// Get an array and convert it to a fixed size.
173/// Returns the values in the array and the original array type.
174pub(crate) fn get_fixed_array_map<T, const N: usize>(
175    (value, location): (Value, Location),
176    f: impl Fn((Value, Location)) -> IResult<T>,
177) -> IResult<([T; N], Type)> {
178    let (values, typ) = get_array_map((value, location), f)?;
179
180    values.try_into().map(|v| (v, typ.clone())).map_err(|_| {
181        // Assuming that `values.len()` corresponds to `typ`.
182        let Type::Array(_, ref elem) = typ else {
183            unreachable!("get_array_map checked it was an array")
184        };
185        let len: u32 =
186            N.try_into().expect("ICE: get_fixed_array_map: N is expected to fit into a u32");
187        let expected = Type::Array(Box::new(Type::constant_u32(len)), elem.clone()).to_string();
188        InterpreterError::TypeMismatch { expected, actual: typ, location }
189    })
190}

263            }
264            _ => Ok(*expr),
265        },
266        value => type_mismatch(value, Type::Quoted(QuotedType::Expr), location),

384/// Helper function to report an [InterpreterError::TypeMismatch] where a [Value] could not be unwrapped
385/// into an expected type, which can either be a concrete [Type] or a textual description, e.g. "tuple".
386///
387/// It has special handling for [Value::Zeroed], which wraps a [Type] that might actually be the
388/// one we expected, but we expected a concrete _value_ with that type, not a zeroed placeholder,
389/// which we optimistically created expecting that it will never be used. In such cases
390/// [InterpreterError::UnexpectedZeroedValue] is returned.
391fn type_mismatch<T>(
392    value: Value,
393    expected: impl Into<TypeOrString>,
394    location: Location,
395) -> IResult<T> {
396    let actual = value.get_type().into_owned();
397    let expected = match expected.into() {
398        TypeOrString::Type(t) if t == actual && matches!(value, Value::Zeroed(_)) => {
399            return Err(InterpreterError::UnexpectedZeroedValue {
400                expected: t.to_string(),
401                location,
402            });
403        }
404        TypeOrString::Type(t) => t.to_string(),
405        TypeOrString::String(s) => s,
406    };
407    Err(InterpreterError::TypeMismatch { expected, actual, location })
408}

410pub(crate) fn hir_pattern_to_tokens(
411    interner: &NodeInterner,
412    hir_pattern: &HirPattern,
413) -> Vec<Token> {
414    let mut tokens = Vec::new();
415    gather_hir_pattern_tokens(interner, hir_pattern, &mut tokens);
416    tokens
417}
418
419fn gather_hir_pattern_tokens(
420    interner: &NodeInterner,
421    hir_pattern: &HirPattern,
422    tokens: &mut Vec<Token>,
423) {
424    match hir_pattern {
425        HirPattern::Identifier(hir_ident) => {
426            let name = interner.definition_name(hir_ident.id).to_string();
427            tokens.push(Token::Ident(name));
428        }
429        HirPattern::Mutable(pattern, _) => {
430            tokens.push(Token::Keyword(Keyword::Mut));
431            gather_hir_pattern_tokens(interner, pattern, tokens);
432        }
433        HirPattern::Tuple(patterns, _) => {
434            tokens.push(Token::LeftParen);
435            for (index, pattern) in patterns.iter().enumerate() {
436                if index != 0 {
437                    tokens.push(Token::Comma);
438                }
439                gather_hir_pattern_tokens(interner, pattern, tokens);
440            }
441            tokens.push(Token::RightParen);
442        }
443        HirPattern::Struct(typ, fields, _) => {
444            let Type::DataType(struct_type, _) = typ.follow_bindings() else {
445                panic!("Expected type to be a struct");
446            };
447
448            let name = struct_type.borrow().name.to_string();
449            tokens.push(Token::Ident(name));
450
451            tokens.push(Token::LeftBrace);
452            for (index, (field_name, pattern)) in fields.iter().enumerate() {
453                if index != 0 {
454                    tokens.push(Token::Comma);
455                }
456
457                let field_name = field_name.as_str();
458                tokens.push(Token::Ident(field_name.to_string()));
459
460                // If we have a pattern like `Foo { x }`, that's internally represented as `Foo { x: x }` so
461                // here we check if the field name is the same as the pattern and, if so, omit the `: x` part.
462                let field_name_is_same_as_pattern = if let HirPattern::Identifier(pattern) = pattern
463                {
464                    let pattern_name = interner.definition_name(pattern.id);
465                    field_name == pattern_name
466                } else {
467                    false
468                };
469
470                if !field_name_is_same_as_pattern {
471                    tokens.push(Token::Colon);
472                    gather_hir_pattern_tokens(interner, pattern, tokens);
473                }
474            }
475            tokens.push(Token::RightBrace);
476        }
477    }
478}

480/// If the given `item_module`'s crate does not match the crate the interpreter is in, issue an
481/// error to prevent modifying an item from an external crate.
482pub(super) fn check_item_crate_matches_current_crate(
483    interpreter: &Interpreter,
484    item: &Value,
485    item_module: ModuleId,
486    location: Location,
487) -> IResult<()> {
488    let current_crate = interpreter.elaborator.module_id().krate;
489    if current_crate != item_module.krate {
490        let module = fully_qualified_module_path(
491            interpreter.elaborator.def_maps,
492            interpreter.elaborator.crate_graph,
493            &current_crate,
494            item_module,
495        );
496        let item = item.display(interpreter.elaborator.interner).to_string();
497        Err(InterpreterError::CannotModifyExternalItem { item, module, location })
498    } else {
499        Ok(())
500    }
501}

517pub(super) fn lex(input: &str, location: Location) -> Vec<LocatedToken> {
518    let (tokens, _) = Lexer::lex(input, location.file);
519    let mut tokens: Vec<_> =
520        tokens.0.into_iter().map(|token| LocatedToken::new(token.into_token(), location)).collect();
521    if let Some(Token::EOF) = tokens.last().map(|t| t.token()) {
522        tokens.pop();
523    }
524    tokens
525}

527pub(super) fn parse<'a, T, F>(
528    elaborator: &mut Elaborator,
529    (value, location): (Value, Location),
530    parser: F,
531    rule: &'static str,
532) -> IResult<T>
533where
534    F: FnOnce(&mut Parser<'a>) -> T,
535{
536    let tokens = get_quoted((value, location))?;
537    let quoted = Tokens(unwrap_rc(tokens.clone()));
538    let (result, warnings) =
539        parse_tokens(tokens, quoted, elaborator.interner, location, parser, rule)?;
540    for warning in warnings {
541        let warning: CompilationError = warning.into();
542        elaborator.push_err(warning);
543    }
544    Ok(result)
545}

547pub(super) fn parse_tokens<'a, T, F>(
548    tokens: Rc<Vec<LocatedToken>>,
549    quoted: Tokens,
550    interner: &NodeInterner,
551    location: Location,
552    parsing_function: F,
553    rule: &'static str,
554) -> IResult<(T, Vec<ParserError>)>
555where
556    F: FnOnce(&mut Parser<'a>) -> T,
557{
558    Parser::for_tokens(quoted).parse_result(parsing_function).map_err(|errors| {
559        let error = errors
560            .into_iter()
561            .find(|error| !error.is_warning())
562            .expect("there is at least 1 error");
563        let error = Box::new(error);
564        let tokens = tokens_to_string(&tokens, interner);
565        InterpreterError::FailedToParseMacro { error, tokens, rule, location }
566    })
567}

577pub(super) fn has_named_attribute(
578    name: &str,
579    attributes: &[SecondaryAttribute],
580    interner: &NodeInterner,
581) -> bool {
582    for attribute in attributes {
583        if let Some(attribute_name) = secondary_attribute_name(attribute, interner)
584            && name == attribute_name
585        {
586            return true;
587        }
588    }
589
590    false
591}
592
593fn secondary_attribute_name(
594    attribute: &SecondaryAttribute,
595    interner: &NodeInterner,
596) -> Option<String> {
597    match &attribute.kind {
598        SecondaryAttributeKind::Deprecated(_, _) => Some("deprecated".to_string()),
599        SecondaryAttributeKind::ContractLibraryMethod => {
600            Some("contract_library_method".to_string())
601        }
602        SecondaryAttributeKind::Export => Some("export".to_string()),
603        SecondaryAttributeKind::Field(_) => Some("field".to_string()),
604        SecondaryAttributeKind::Tag(contents) => {
605            let mut lexer = Lexer::new_with_dummy_file(contents);
606            let token = lexer.next()?.ok()?;
607            if let Token::Ident(ident) = token.into_token() { Some(ident) } else { None }
608        }
609        SecondaryAttributeKind::Meta(meta) => interner.get_meta_attribute_name(meta),
610        SecondaryAttributeKind::Abi(_) => Some("abi".to_string()),
611        SecondaryAttributeKind::Varargs => Some("varargs".to_string()),
612        SecondaryAttributeKind::UseCallersScope => Some("use_callers_scope".to_string()),
613        SecondaryAttributeKind::Allow(_) => Some("allow".to_string()),
614        SecondaryAttributeKind::MustUse(_) => Some("must_use".to_string()),
615    }
616}

628pub(super) fn hash_item<T: Hash>(
629    arguments: Vec<(Value, Location)>,
630    location: Location,
631    get_item: impl FnOnce((Value, Location)) -> IResult<T>,
632) -> IResult<Value> {
633    let argument = check_one_argument(arguments, location)?;
634    let item = get_item(argument)?;
635
636    let mut hasher = std::collections::hash_map::DefaultHasher::new();
637    item.hash(&mut hasher);
638    let hash = hasher.finish();
639    Ok(Value::field(u128::from(hash).into()))
640}

642pub(super) fn eq_item<T: Eq>(
643    arguments: Vec<(Value, Location)>,
644    location: Location,
645    mut get_item: impl FnMut((Value, Location)) -> IResult<T>,
646) -> IResult<Value> {
647    let (self_arg, other_arg) = check_two_arguments(arguments, location)?;
648    let self_arg = get_item(self_arg)?;
649    let other_arg = get_item(other_arg)?;
650    Ok(Value::Bool(self_arg == other_arg))
651}

653/// Type to be used in `Value::Array(<values>, <array-type>)`.
654pub(crate) fn byte_array_type(len: usize) -> Type {
655    let len: u32 = len.try_into().expect("ICE: byte_array_type: N is expected to fit into a u32");
656    Type::Array(
657        Box::new(Type::constant_u32(len)),
658        Box::new(Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight)),
659    )
660}
661
662/// Create a `Value::Array` from bytes.
663pub(crate) fn to_byte_array(values: &[u8]) -> Value {
664    Value::Array(values.iter().copied().map(Value::u8).collect(), byte_array_type(values.len()))
665}

667/// Create a `Value::Struct` from fields and the expected return type.
668pub(crate) fn to_struct(
669    fields: impl IntoIterator<Item = (&'static str, Value)>,
670    typ: Type,
671) -> Value {
672    let fields =
673        fields.into_iter().map(|(k, v)| (Rc::new(k.to_string()), Shared::new(v))).collect();
674    Value::Struct(fields, typ)
675}

677pub(crate) fn new_unary_op(operator: UnaryOp, typ: Type) -> Option<Value> {
678    // These values should match the values used in noir_stdlib/src/meta/op.nr
679    let unary_op_value: u128 = match operator {
680        UnaryOp::Minus => 0,
681        UnaryOp::Not => 1,
682        UnaryOp::Reference { mutable: true } => 2,
683        UnaryOp::Reference { mutable: false } => {
684            // `&` alone is experimental and currently hidden from the comptime API
685            return None;
686        }
687        UnaryOp::Dereference { .. } => 3,
688    };
689
690    let mut fields = HashMap::default();
691    fields.insert(Rc::new("op".to_string()), Shared::new(Value::field(unary_op_value.into())));
692
693    Some(Value::Struct(fields, typ))
694}

696pub(crate) fn new_binary_op(operator: BinaryOp, typ: Type) -> Value {
697    // For the op value we use the enum member index, which should match noir_stdlib/src/meta/op.nr
698    let binary_op_value = operator.contents as u128;
699
700    let mut fields = HashMap::default();
701    fields.insert(Rc::new("op".to_string()), Shared::new(Value::field(binary_op_value.into())));
702
703    Value::Struct(fields, typ)
704}

706pub(crate) fn visibility_to_quoted(visibility: ItemVisibility, location: Location) -> Value {
707    let tokens = match visibility {
708        ItemVisibility::Private => vec![],
709        ItemVisibility::PublicCrate => vec![
710            Token::Keyword(Keyword::Pub),
711            Token::LeftParen,
712            Token::Keyword(Keyword::Crate),
713            Token::RightParen,
714        ],
715        ItemVisibility::Public => vec![Token::Keyword(Keyword::Pub)],
716    };
717    let tokens = vecmap(tokens, |token| LocatedToken::new(token, location));
718    Value::Quoted(Rc::new(tokens))
719}

721pub(crate) fn fragments_to_string(
722    fragments: &[FormatStringFragment],
723    interner: &NodeInterner,
724) -> String {
725    let mut result = String::new();
726    for fragment in fragments {
727        match fragment {
728            FormatStringFragment::String(string) => {
729                result.push_str(string);
730            }
731            FormatStringFragment::Value { name: _, value } => {
732                match value {
733                    Value::Quoted(tokens) => {
734                        // When interpolating a quoted value inside a format string, we don't include the
735                        // surrounding `quote {` ... `}` as if we are unquoting the quoted value inside the string.
736                        for (index, token) in tokens.iter().enumerate() {
737                            if index > 0 {
738                                result.push(' ');
739                            }
740                            result.push_str(&token.token().display(interner).to_string());
741                        }
742                    }
743                    Value::FormatString(fragments, _, _) => {
744                        // Nested format strings might have quoted values inside them,
745                        // so we need to recurse here instead of calling `value.display`.
746                        let inner_string = fragments_to_string(fragments, interner);
747                        result.push_str(&inner_string);
748                    }
749                    _ => {
750                        result.push_str(&value.display(interner).to_string());
751                    }
752                }
753            }
754        }
755    }
756    result
757}
