47pub(crate) fn check_argument_count(
48    expected: usize,
49    arguments: &[(Value, Location)],
50    location: Location,
51) -> IResult<()> {
52    if arguments.len() == expected {
53        Ok(())
54    } else {
55        let actual = arguments.len();
56        Err(InterpreterError::ArgumentCountMismatch { expected, actual, location })
57    }
58}

124/// Get a specific field of a struct and apply a decoder function on it.
125pub(crate) fn get_struct_field<T>(
126    field_name: &str,
127    struct_fields: &HashMap<Rc<String>, Shared<Value>>,
128    struct_type: &Type,
129    location: Location,
130    f: impl Fn((Value, Location)) -> IResult<T>,
131) -> IResult<T> {
132    let key = field_name.to_string();
133    let Some(value) = struct_fields.get(&key) else {
134        return Err(InterpreterError::ExpectedStructToHaveField {
135            typ: struct_type.clone(),
136            field_name: key,
137            location,
138        });
139    };
140    f((value.borrow().clone(), location))
141}

162/// Interpret the input as an array, then map each element.
163/// Returns the values in the array and the original array type.
164pub(crate) fn get_array_map<T>(
165    (value, location): (Value, Location),
166    f: impl Fn((Value, Location)) -> IResult<T>,
167) -> IResult<(Vec<T>, Type)> {
168    let (values, typ) = get_array((value, location))?;
169    let values = try_vecmap(values, |value| f((value, location)))?;
170    Ok((values, typ))
171}

173/// Get an array and convert it to a fixed size.
174/// Returns the values in the array and the original array type.
175pub(crate) fn get_fixed_array_map<T, const N: usize>(
176    (value, location): (Value, Location),
177    f: impl Fn((Value, Location)) -> IResult<T>,
178) -> IResult<([T; N], Type)> {
179    let (values, typ) = get_array_map((value, location), f)?;
180
181    values.try_into().map(|v| (v, typ.clone())).map_err(|_| {
182        // Assuming that `values.len()` corresponds to `typ`.
183        let Type::Array(_, ref elem) = typ else {
184            unreachable!("get_array_map checked it was an array")
185        };
186        let len: u32 =
187            N.try_into().expect("ICE: get_fixed_array_map: N is expected to fit into a u32");
188        let expected = Type::Array(Box::new(len.into()), elem.clone()).to_string();
189        InterpreterError::TypeMismatch { expected, actual: typ, location }
190    })
191}

258    interner: &NodeInterner,
259    (value, location): (Value, Location),
260) -> IResult<ExprValue> {
261    match value {
262        Value::Expr(expr) => match *expr {
263            ExprValue::Expression(ExpressionKind::Interned(id)) => {
264                Ok(ExprValue::Expression(interner.get_expression_kind(id).clone()))
265            }
266            ExprValue::Statement(StatementKind::Interned(id)) => {
267                Ok(ExprValue::Statement(interner.get_statement_kind(id).clone()))
268            }
269            ExprValue::LValue(LValue::Interned(id, _)) => {
270                Ok(ExprValue::LValue(interner.get_lvalue(id, location).clone()))
271            }
272            ExprValue::Pattern(Pattern::Interned(id, _)) => {
273                Ok(ExprValue::Pattern(interner.get_pattern(id).clone()))
274            }
275            _ => Ok(*expr),
276        },
277        value => type_mismatch(value, Type::Quoted(QuotedType::Expr), location),
278    }

395/// Helper function to report an [InterpreterError::TypeMismatch] where a [Value] could not be unwrapped
396/// into an expected type, which can either be a concrete [Type] or a textual description, e.g. "tuple".
397///
398/// It has special handling for [Value::Zeroed], which wraps a [Type] that might actually be the
399/// one we expected, but we expected a concrete _value_ with that type, not a zeroed placeholder,
400/// which we optimistically created expecting that it will never be used. In such cases
401/// [InterpreterError::UnexpectedZeroedValue] is returned.
402fn type_mismatch<T>(
403    value: Value,
404    expected: impl Into<TypeOrString>,
405    location: Location,
406) -> IResult<T> {
407    let actual = value.get_type().into_owned();
408    let expected = match expected.into() {
409        TypeOrString::Type(t) if t == actual && matches!(value, Value::Zeroed(_)) => {
410            return Err(InterpreterError::UnexpectedZeroedValue {
411                expected: t.to_string(),
412                location,
413            });
414        }
415        TypeOrString::Type(t) => t.to_string(),
416        TypeOrString::String(s) => s,
417    };
418    Err(InterpreterError::TypeMismatch { expected, actual, location })
419}

421pub(crate) fn hir_pattern_to_tokens(
422    interner: &NodeInterner,
423    hir_pattern: &HirPattern,
424) -> Vec<Token> {
425    let mut tokens = Vec::new();
426    gather_hir_pattern_tokens(interner, hir_pattern, &mut tokens);
427    tokens
428}
429
430fn gather_hir_pattern_tokens(
431    interner: &NodeInterner,
432    hir_pattern: &HirPattern,
433    tokens: &mut Vec<Token>,
434) {
435    match hir_pattern {
436        HirPattern::Identifier(hir_ident) => {
437            let name = interner.definition_name(hir_ident.id).to_string();
438            tokens.push(Token::Ident(name));
439        }
440        HirPattern::Mutable(pattern, _) => {
441            tokens.push(Token::Keyword(Keyword::Mut));
442            gather_hir_pattern_tokens(interner, pattern, tokens);
443        }
444        HirPattern::Tuple(patterns, _) => {
445            tokens.push(Token::LeftParen);
446            for (index, pattern) in patterns.iter().enumerate() {
447                if index != 0 {
448                    tokens.push(Token::Comma);
449                }
450                gather_hir_pattern_tokens(interner, pattern, tokens);
451            }
452            tokens.push(Token::RightParen);
453        }
454        HirPattern::Struct(typ, fields, _) => {
455            let Type::DataType(struct_type, _) = typ.follow_bindings() else {
456                panic!("Expected type to be a struct");
457            };
458
459            let name = struct_type.borrow().name.to_string();
460            tokens.push(Token::Ident(name));
461
462            tokens.push(Token::LeftBrace);
463            for (index, (field_name, pattern)) in fields.iter().enumerate() {
464                if index != 0 {
465                    tokens.push(Token::Comma);
466                }
467
468                let field_name = field_name.as_str();
469                tokens.push(Token::Ident(field_name.to_string()));
470
471                // If we have a pattern like `Foo { x }`, that's internally represented as `Foo { x: x }` so
472                // here we check if the field name is the same as the pattern and, if so, omit the `: x` part.
473                let field_name_is_same_as_pattern = if let HirPattern::Identifier(pattern) = pattern
474                {
475                    let pattern_name = interner.definition_name(pattern.id);
476                    field_name == pattern_name
477                } else {
478                    false
479                };
480
481                if !field_name_is_same_as_pattern {
482                    tokens.push(Token::Colon);
483                    gather_hir_pattern_tokens(interner, pattern, tokens);
484                }
485            }
486            tokens.push(Token::RightBrace);
487        }
488    }
489}

491pub(super) fn check_function_not_yet_resolved(
492    interpreter: &Interpreter,
493    func_id: FuncId,
494    location: Location,
495) -> Result<(), InterpreterError> {
496    let func_meta = interpreter.elaborator.interner.function_meta(&func_id);
497    match func_meta.function_body {
498        FunctionBody::Unresolved(_, _, _) => Ok(()),
499        FunctionBody::Resolving | FunctionBody::Resolved => {
500            Err(InterpreterError::FunctionAlreadyResolved { location })
501        }
502    }
503}

505pub(super) fn lex(input: &str, location: Location) -> Vec<LocatedToken> {
506    let (tokens, _) = Lexer::lex(input, location.file);
507    let mut tokens: Vec<_> =
508        tokens.0.into_iter().map(|token| LocatedToken::new(token.into_token(), location)).collect();
509    if let Some(Token::EOF) = tokens.last().map(|t| t.token()) {
510        tokens.pop();
511    }
512    tokens
513}

515pub(super) fn parse<'a, T, F>(
516    elaborator: &mut Elaborator,
517    (value, location): (Value, Location),
518    parser: F,
519    rule: &'static str,
520) -> IResult<T>
521where
522    F: FnOnce(&mut Parser<'a>) -> T,
523{
524    let tokens = get_quoted((value, location))?;
525    let quoted = Tokens(unwrap_rc(tokens.clone()));
526    let (result, warnings) =
527        parse_tokens(tokens, quoted, elaborator.interner, location, parser, rule)?;
528    for warning in warnings {
529        let warning: CompilationError = warning.into();
530        elaborator.push_err(warning);
531    }
532    Ok(result)
533}

535pub(super) fn parse_tokens<'a, T, F>(
536    tokens: Rc<Vec<LocatedToken>>,
537    quoted: Tokens,
538    interner: &NodeInterner,
539    location: Location,
540    parsing_function: F,
541    rule: &'static str,
542) -> IResult<(T, Vec<ParserError>)>
543where
544    F: FnOnce(&mut Parser<'a>) -> T,
545{
546    Parser::for_tokens(quoted).parse_result(parsing_function).map_err(|errors| {
547        let error = errors
548            .into_iter()
549            .find(|error| !error.is_warning())
550            .expect("there is at least 1 error");
551        let error = Box::new(error);
552        let tokens = tokens_to_string(&tokens, interner);
553        InterpreterError::FailedToParseMacro { error, tokens, rule, location }
554    })
555}

598pub(super) fn has_named_attribute(
599    name: &str,
600    attributes: &[SecondaryAttribute],
601    interner: &NodeInterner,
602) -> bool {
603    for attribute in attributes {
604        if let Some(attribute_name) = secondary_attribute_name(attribute, interner) {
605            if name == attribute_name {
606                return true;
607            }
608        }
609    }
610
611    false
612}

614fn secondary_attribute_name(
615    attribute: &SecondaryAttribute,
616    interner: &NodeInterner,
617) -> Option<String> {
618    match &attribute.kind {
619        SecondaryAttributeKind::Deprecated(_) => Some("deprecated".to_string()),
620        SecondaryAttributeKind::ContractLibraryMethod => {
621            Some("contract_library_method".to_string())
622        }
623        SecondaryAttributeKind::Export => Some("export".to_string()),
624        SecondaryAttributeKind::Field(_) => Some("field".to_string()),
625        SecondaryAttributeKind::Tag(contents) => {
626            let mut lexer = Lexer::new_with_dummy_file(contents);
627            let token = lexer.next()?.ok()?;
628            if let Token::Ident(ident) = token.into_token() { Some(ident) } else { None }
629        }
630        SecondaryAttributeKind::Meta(meta) => interner.get_meta_attribute_name(meta),
631        SecondaryAttributeKind::Abi(_) => Some("abi".to_string()),
632        SecondaryAttributeKind::Varargs => Some("varargs".to_string()),
633        SecondaryAttributeKind::UseCallersScope => Some("use_callers_scope".to_string()),
634        SecondaryAttributeKind::Allow(_) => Some("allow".to_string()),
635        SecondaryAttributeKind::MustUse(_) => Some("must_use".to_string()),
636    }
637}

649pub(super) fn hash_item<T: Hash>(
650    arguments: Vec<(Value, Location)>,
651    location: Location,
652    get_item: impl FnOnce((Value, Location)) -> IResult<T>,
653) -> IResult<Value> {
654    let argument = check_one_argument(arguments, location)?;
655    let item = get_item(argument)?;
656
657    let mut hasher = std::collections::hash_map::DefaultHasher::new();
658    item.hash(&mut hasher);
659    let hash = hasher.finish();
660    Ok(Value::Field(SignedField::positive(u128::from(hash))))
661}
662
663pub(super) fn eq_item<T: Eq>(
664    arguments: Vec<(Value, Location)>,
665    location: Location,
666    mut get_item: impl FnMut((Value, Location)) -> IResult<T>,
667) -> IResult<Value> {
668    let (self_arg, other_arg) = check_two_arguments(arguments, location)?;
669    let self_arg = get_item(self_arg)?;
670    let other_arg = get_item(other_arg)?;
671    Ok(Value::Bool(self_arg == other_arg))
672}

674/// Type to be used in `Value::Array(<values>, <array-type>)`.
675pub(crate) fn byte_array_type(len: usize) -> Type {
676    let len: u32 = len.try_into().expect("ICE: byte_array_type: N is expected to fit into a u32");
677    Type::Array(
678        Box::new(len.into()),
679        Box::new(Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight)),
680    )
681}

683/// Create a `Value::Array` from bytes.
684pub(crate) fn to_byte_array(values: &[u8]) -> Value {
685    Value::Array(values.iter().copied().map(Value::U8).collect(), byte_array_type(values.len()))
686}

689pub(crate) fn to_struct(
690    fields: impl IntoIterator<Item = (&'static str, Value)>,
691    typ: Type,
692) -> Value {

694        fields.into_iter().map(|(k, v)| (Rc::new(k.to_string()), Shared::new(v))).collect();

698pub(crate) fn new_unary_op(operator: UnaryOp, typ: Type) -> Option<Value> {
699    // These values should match the values used in noir_stdlib/src/meta/op.nr
700    let unary_op_value: u128 = match operator {
701        UnaryOp::Minus => 0,
702        UnaryOp::Not => 1,
703        UnaryOp::Reference { mutable: true } => 2,
704        UnaryOp::Reference { mutable: false } => {
705            // `&` alone is experimental and currently hidden from the comptime API
706            return None;
707        }
708        UnaryOp::Dereference { .. } => 3,
709    };
710
711    let mut fields = HashMap::default();
712    fields.insert(
713        Rc::new("op".to_string()),
714        Shared::new(Value::Field(SignedField::positive(unary_op_value))),
715    );
716
717    Some(Value::Struct(fields, typ))
718}

722    let binary_op_value = operator.contents as u128;
723
724    let mut fields = HashMap::default();
725    fields.insert(
726        Rc::new("op".to_string()),

731}

733pub(crate) fn visibility_to_quoted(visibility: ItemVisibility, location: Location) -> Value {
734    let tokens = match visibility {
735        ItemVisibility::Private => vec![],
736        ItemVisibility::PublicCrate => vec![
737            Token::Keyword(Keyword::Pub),
738            Token::LeftParen,
739            Token::Keyword(Keyword::Crate),
740            Token::RightParen,
741        ],
742        ItemVisibility::Public => vec![Token::Keyword(Keyword::Pub)],
743    };
744    let tokens = vecmap(tokens, |token| LocatedToken::new(token, location));
745    Value::Quoted(Rc::new(tokens))
746}

748pub(crate) fn fragments_to_string(
749    fragments: &[FormatStringFragment],
750    interner: &NodeInterner,
751) -> String {
752    let mut result = String::new();
753    for fragment in fragments {
754        match fragment {
755            FormatStringFragment::String(string) => {
756                result.push_str(string);
757            }
758            FormatStringFragment::Value { name: _, value } => {
759                match value {
760                    Value::Quoted(tokens) => {
761                        // When interpolating a quoted value inside a format string, we don't include the
762                        // surrounding `quote {` ... `}` as if we are unquoting the quoted value inside the string.
763                        for (index, token) in tokens.iter().enumerate() {
764                            if index > 0 {
765                                result.push(' ');
766                            }
767                            result.push_str(&token.token().display(interner).to_string());
768                        }
769                    }
770                    Value::FormatString(fragments, _, _) => {
771                        // Nested format strings might have quoted values inside them,
772                        // so we need to recurse here instead of calling `value.display`.
773                        let inner_string = fragments_to_string(fragments, interner);
774                        result.push_str(&inner_string);
775                    }
776                    _ => {
777                        result.push_str(&value.display(interner).to_string());
778                    }
779                }
780            }
781        }
782    }
783    result
784}
