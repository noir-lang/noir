422    pub(crate) fn debug_evaluate_comptime(expr: impl Display, location: Location) -> Self {
423        let mut formatted_result = format!("{expr}");
424        // if multi-line, display on a separate line from the message
425        if formatted_result.contains('\n') {
426            formatted_result.insert(0, '\n');
427        }
428        let diagnostic = CustomDiagnostic::simple_info(
429            "`comptime` expression ran:".to_string(),
430            format!("After evaluation: {formatted_result}"),
431            location,
432        );
433        InterpreterError::DebugEvaluateComptime { diagnostic, location }
434    }

437impl<'a> From<&'a InterpreterError> for CustomDiagnostic {
438    fn from(error: &'a InterpreterError) -> Self {
439        match error {
440            InterpreterError::ArgumentCountMismatch { expected, actual, location } => {
441                let only = if expected > actual { "only " } else { "" };
442                let plural = if *expected == 1 { "" } else { "s" };
443                let was_were = if *actual == 1 { "was" } else { "were" };
444                let msg = format!(
445                    "Expected {expected} argument{plural}, but {only}{actual} {was_were} provided"
446                );
447
448                let few_many = if actual < expected { "few" } else { "many" };
449                let secondary = format!("Too {few_many} arguments");
450                CustomDiagnostic::simple_error(msg, secondary, *location)
451            }
452            InterpreterError::TypeMismatch { expected, actual, location } => {
453                let msg = format!("Expected `{expected}` but a value of type `{actual}` was given");
454                CustomDiagnostic::simple_error(msg, String::new(), *location)
455            }
456            InterpreterError::UnexpectedZeroedValue { expected, location } => {
457                let msg = format!("Expected a concrete `{expected}` but a zeroed value was given");
458                let secondary = format!(
459                    "A zeroed value of `{expected}` may be created to satisfy the type system, but it's not expected to be used"
460                );
461                CustomDiagnostic::simple_error(msg, secondary, *location)
462            }
463            InterpreterError::NonComptimeVarReferenced { name, location } => {
464                let msg = format!("Non-comptime variable `{name}` referenced in comptime code");
465                let secondary = "Non-comptime variables can't be used in comptime code".to_string();
466                CustomDiagnostic::simple_error(msg, secondary, *location)
467            }
468            InterpreterError::VariableNotInScope { location } => {
469                let msg = "Variable not in scope".to_string();
470                let secondary = "Could not find variable".to_string();
471                CustomDiagnostic::simple_error(msg, secondary, *location)
472            }
473            InterpreterError::IntegerOutOfRangeForType { value, typ, location } => {
474                let msg = format!("{value} is outside the range of the {typ} type");
475                CustomDiagnostic::simple_error(msg, String::new(), *location)
476            }
477            InterpreterError::ErrorNodeEncountered { location } => {
478                let msg = "Internal Compiler Error: Error node encountered".to_string();
479                let secondary = "This is a bug, please report this if found!".to_string();
480                CustomDiagnostic::simple_error(msg, secondary, *location)
481            }
482            InterpreterError::NonFunctionCalled { typ, location } => {
483                let msg = "Only functions may be called".to_string();
484                let secondary = format!("Expression has type {typ}");
485                CustomDiagnostic::simple_error(msg, secondary, *location)
486            }
487            InterpreterError::NonBoolUsedInIf { typ, location } => {
488                let msg = format!("Expected a `bool` but found `{typ}`");
489                let secondary = "If conditions must be a boolean value".to_string();
490                CustomDiagnostic::simple_error(msg, secondary, *location)
491            }
492            InterpreterError::NonBoolUsedInWhile { typ, location } => {
493                let msg = format!("Expected a `bool` but found `{typ}`");
494                let secondary = "While conditions must be a boolean value".to_string();
495                CustomDiagnostic::simple_error(msg, secondary, *location)
496            }
497            InterpreterError::NonBoolUsedInConstrain { typ, location } => {
498                let msg = format!("Expected a `bool` but found `{typ}`");
499                CustomDiagnostic::simple_error(msg, String::new(), *location)
500            }
501            InterpreterError::FailingConstraint { message, location, call_stack } => {
502                let (primary, secondary) = match message {
503                    Some(msg) => (msg.clone(), "Assertion failed".into()),
504                    None => ("Assertion failed".into(), String::new()),
505                };
506                let diagnostic = CustomDiagnostic::simple_error(primary, secondary, *location);
507
508                diagnostic.with_call_stack(call_stack.into_iter().copied().collect())
509            }
510            InterpreterError::NonIntegerUsedInLoop { typ, location } => {
511                let msg = format!("Non-integer type `{typ}` used in for loop");
512                let secondary = if matches!(typ, Type::FieldElement) {
513                    "`field` is not an integer type, try `u32` instead".to_string()
514                } else {
515                    String::new()
516                };
517                CustomDiagnostic::simple_error(msg, secondary, *location)
518            }
519            InterpreterError::RangeBoundsTypeMismatch { start_type, end_type, location } => {
520                let msg = format!(
521                    "Range bounds have mismatched types: start is `{start_type}` but end is `{end_type}`"
522                );
523                CustomDiagnostic::simple_error(msg, String::new(), *location)
524            }
525            InterpreterError::NonPointerDereferenced { typ, location } => {
526                let msg = format!("Only references may be dereferenced, but found `{typ}`");
527                CustomDiagnostic::simple_error(msg, String::new(), *location)
528            }
529            InterpreterError::NonTupleOrStructInMemberAccess { typ, location } => {
530                let msg = format!("The type `{typ}` has no fields to access");
531                CustomDiagnostic::simple_error(msg, String::new(), *location)
532            }
533            InterpreterError::NonArrayIndexed { typ, location } => {
534                let msg = format!("Expected an array or vector but found a(n) {typ}");
535                let secondary = "Only arrays or vectors may be indexed".into();
536                CustomDiagnostic::simple_error(msg, secondary, *location)
537            }
538            InterpreterError::NonIntegerUsedAsIndex { typ, location } => {
539                let msg = format!("Expected an integer but found a(n) {typ}");
540                let secondary =
541                    "Only integers may be indexed. Note that this excludes `field`s".into();
542                CustomDiagnostic::simple_error(msg, secondary, *location)
543            }
544            InterpreterError::NonIntegerIntegerLiteral { typ, location } => {
545                let msg = format!("This integer literal somehow has the type `{typ}`");
546                let secondary = "This is likely a bug".into();
547                CustomDiagnostic::simple_error(msg, secondary, *location)
548            }
549            InterpreterError::InvalidArrayLength { err, location } => {
550                let msg = "Invalid array length".to_string();
551                let secondary = err.to_string();
552                CustomDiagnostic::simple_error(msg, secondary, *location)
553            }
554            InterpreterError::InvalidAssociatedConstant { err, location } => {
555                let msg = "Invalid associated constant".to_string();
556                let secondary = err.to_string();
557                CustomDiagnostic::simple_error(msg, secondary, *location)
558            }
559            InterpreterError::InvalidNumericGeneric { err, location } => {
560                let msg = "Invalid numeric generic".to_string();
561                let secondary = err.to_string();
562                CustomDiagnostic::simple_error(msg, secondary, *location)
563            }
564            InterpreterError::NonNumericCasted { typ, location } => {
565                let msg = "Only numeric types may be casted".into();
566                let secondary = format!("`{typ}` is non-numeric");
567                CustomDiagnostic::simple_error(msg, secondary, *location)
568            }
569            InterpreterError::IndexOutOfBounds { index, length, location } => {
570                let msg = format!("{index} is out of bounds for the array of length {length}");
571                CustomDiagnostic::simple_error(msg, String::new(), *location)
572            }
573            InterpreterError::ExpectedStructToHaveField { typ, field_name, location } => {
574                let msg = format!("The type `{typ}` has no field named `{field_name}`");
575                CustomDiagnostic::simple_error(msg, String::new(), *location)
576            }
577            InterpreterError::TypeUnsupported { typ, location } => {
578                let msg =
579                    format!("The type `{typ}` is currently unsupported in comptime expressions");
580                CustomDiagnostic::simple_error(msg, String::new(), *location)
581            }
582            InterpreterError::InvalidValueForUnary { typ, operator, location } => {
583                let msg = format!("`{typ}` cannot be used with unary {operator}");
584                CustomDiagnostic::simple_error(msg, String::new(), *location)
585            }
586            InterpreterError::InvalidValuesForBinary { lhs, rhs, operator, location } => {
587                let msg = if *operator == "/" {
588                    "Attempt to divide by zero".to_string()
589                } else if *operator == "%" {
590                    "Attempt to calculate the remainder with a divisor of zero".to_string()
591                } else {
592                    format!("No implementation for `{lhs}` {operator} `{rhs}`")
593                };
594                CustomDiagnostic::simple_error(msg, String::new(), *location)
595            }
596            InterpreterError::BinaryOperationOverflow { operator, location } => {
597                let operator = match *operator {
598                    "+" => "add",
599                    "-" => "subtract",
600                    "*" => "multiply",
601                    ">>" | "<<" => "bit-shift",
602                    _ => operator,
603                };
604                let msg = format!("Attempt to {operator} with overflow");
605
606                CustomDiagnostic::simple_error(msg, String::new(), *location)
607            }
608            InterpreterError::NegateWithOverflow { location } => {
609                let msg = "Attempt to negate with overflow".to_string();
610                CustomDiagnostic::simple_error(msg, String::new(), *location)
611            }
612            InterpreterError::CannotApplyMinusToType { location, typ } => {
613                let msg = format!("Cannot apply unary operator `-` to type `{typ}`");
614                CustomDiagnostic::simple_error(msg, String::new(), *location)
615            }
616            InterpreterError::CastToNonNumericType { typ, location } => {
617                let msg = format!("Cannot cast to non-numeric type `{typ}`");
618                CustomDiagnostic::simple_error(msg, String::new(), *location)
619            }
620            InterpreterError::NonStructInConstructor { typ, location } => {
621                let msg = format!("`{typ}` is not a struct type");
622                CustomDiagnostic::simple_error(msg, String::new(), *location)
623            }
624            InterpreterError::NonEnumInConstructor { typ, location } => {
625                let msg = format!("`{typ}` is not an enum type");
626                CustomDiagnostic::simple_error(msg, String::new(), *location)
627            }
628            InterpreterError::CannotInlineMacro { value, typ, location } => {
629                let msg = format!("Cannot inline values of type `{typ}` into this position");
630                let secondary = format!("Cannot inline value `{value}`");
631                CustomDiagnostic::simple_error(msg, secondary, *location)
632            }
633            InterpreterError::UnquoteFoundDuringEvaluation { location } => {
634                let msg = "Unquote found during comptime evaluation".into();
635                let secondary = "This is a bug".into();
636                CustomDiagnostic::simple_error(msg, secondary, *location)
637            }
638            InterpreterError::DebugEvaluateComptime { diagnostic, .. } => diagnostic.clone(),
639            InterpreterError::FailedToParseMacro { error, tokens, rule, location } => {
640                // If it's less than 48 chars, the error message fits in a single line (less than 80 chars total)
641                let token_stream = if tokens.len() <= 48 && !tokens.contains('\n') {
642                    format!("The resulting token stream was: {tokens}")
643                } else {
644                    format!(
645                        "The resulting token stream was: (stream starts on next line)\n  {tokens}"
646                    )
647                };
648
649                let push_the_problem_on_the_library_author = "To avoid this error in the future, try adding input validation to your macro. Erroring out early with an `assert` can be a good way to provide a user-friendly error message".into();
650
651                let mut diagnostic = CustomDiagnostic::from(&**error);
652
653                // Given more prominence to where the parser error happened, but still show that it's
654                // because of a failure to parse a macro's token stream, and where that happens.
655                let message = format!("Failed to parse macro's token stream into {rule}");
656                diagnostic.add_secondary(message, *location);
657
658                diagnostic.add_note(token_stream);
659                diagnostic.add_note(push_the_problem_on_the_library_author);
660                diagnostic
661            }
662            InterpreterError::UnsupportedTopLevelItemUnquote { item, location } => {
663                let msg = "Unsupported statement type to unquote".into();
664                let secondary =
665                    "Only functions, structs, globals, and impls can be unquoted here".into();
666                let mut error = CustomDiagnostic::simple_error(msg, secondary, *location);
667                error.add_note(format!("Unquoted item was:\n{item}"));
668                error
669            }
670            InterpreterError::ComptimeDependencyCycle { function, location } => {
671                let msg = format!("Comptime dependency cycle while resolving `{function}`");
672                let secondary =
673                    "This function uses comptime code internally which calls into itself".into();
674                CustomDiagnostic::simple_error(msg, secondary, *location)
675            }
676            InterpreterError::Unimplemented { item, location } => {
677                let msg = format!("{item} is currently unimplemented");
678                CustomDiagnostic::simple_error(msg, String::new(), *location)
679            }
680            InterpreterError::InvalidInComptimeContext { item, location, explanation } => {
681                let msg = format!("{item} is invalid in comptime context");
682                CustomDiagnostic::simple_error(msg, explanation.clone(), *location)
683            }
684            InterpreterError::BreakNotInLoop { location } => {
685                let msg = "There is no loop to break out of!".into();
686                CustomDiagnostic::simple_error(msg, String::new(), *location)
687            }
688            InterpreterError::ContinueNotInLoop { location } => {
689                let msg = "There is no loop to continue!".into();
690                CustomDiagnostic::simple_error(msg, String::new(), *location)
691            }
692            InterpreterError::NoImpl { location } => {
693                let msg = "No impl found due to prior type error".into();
694                CustomDiagnostic::simple_error(msg, String::new(), *location)
695            }
696            InterpreterError::ImplMethodTypeMismatch { expected, actual, location } => {
697                let msg = format!(
698                    "Impl method type {actual} does not unify with trait method type {expected}"
699                );
700                CustomDiagnostic::simple_error(msg, String::new(), *location)
701            }
702            InterpreterError::BlackBoxError(error, location) => {
703                CustomDiagnostic::simple_error(error.to_string(), String::new(), *location)
704            }
705            InterpreterError::FailedToResolveTraitBound { trait_bound, location } => {
706                let msg = format!("Failed to resolve trait bound `{trait_bound}`");
707                CustomDiagnostic::simple_error(msg, String::new(), *location)
708            }
709            InterpreterError::NoMatchingImplFound { error, .. } => error.into(),
710            InterpreterError::Break => unreachable!("Uncaught InterpreterError::Break"),
711            InterpreterError::Continue => unreachable!("Uncaught InterpreterError::Continue"),
712            InterpreterError::TraitDefinitionMustBeAPath { location } => {
713                let msg = "Trait definition arguments must be a variable or path".to_string();
714                CustomDiagnostic::simple_error(msg, String::new(), *location)
715            }
716            InterpreterError::FailedToResolveTraitDefinition { location } => {
717                let msg = "Failed to resolve to a trait definition".to_string();
718                CustomDiagnostic::simple_error(msg, String::new(), *location)
719            }
720            InterpreterError::FunctionAlreadyResolved { location } => {
721                let msg = "Function already resolved".to_string();
722                let secondary =
723                    "The function was previously called at compile-time or is in another crate"
724                        .to_string();
725                CustomDiagnostic::simple_error(msg, secondary, *location)
726            }
727            InterpreterError::MultipleMatchingImpls { object_type, candidates, location } => {
728                let message = format!("Multiple trait impls match the object type `{object_type}`");
729                let secondary = "Ambiguous impl".to_string();
730                let mut error = CustomDiagnostic::simple_error(message, secondary, *location);
731                for (i, candidate) in candidates.iter().enumerate() {
732                    error.add_note(format!("Candidate {}: `{candidate}`", i + 1));
733                }
734                error
735            }
736            InterpreterError::TypeAnnotationsNeededForMethodCall { location } => {
737                let mut error = CustomDiagnostic::simple_error(
738                    "Object type is unknown in method call".to_string(),
739                    "Type must be known by this point to know which method to call".to_string(),
740                    *location,
741                );
742                let message =
743                    "Try adding a type annotation for the object type before this method call";
744                error.add_note(message.to_string());
745                error
746            }
747            InterpreterError::ExpectedIdentForStructField { value, index, location } => {
748                let msg = format!(
749                    "Quoted value in index {index} of this vector is not a valid field name"
750                );
751                let secondary = format!("`{value}` is not a valid field name for `set_fields`");
752                CustomDiagnostic::simple_error(msg, secondary, *location)
753            }
754            InterpreterError::InvalidAttribute { attribute, location } => {
755                let msg = format!("`{attribute}` is not a valid attribute");
756                let secondary = "Note that this method expects attribute contents, without the leading `#[` or trailing `]`".to_string();
757                CustomDiagnostic::simple_error(msg, secondary, *location)
758            }
759            InterpreterError::GenericNameShouldBeAnIdent { name, location } => {
760                let msg =
761                            "Generic name needs to be a valid identifier (one word beginning with a letter)"
762                                .to_string();
763                let secondary = format!("`{name}` is not a valid identifier");
764                CustomDiagnostic::simple_error(msg, secondary, *location)
765            }
766            InterpreterError::DuplicateGeneric {
767                name,
768                struct_name,
769                duplicate_location,
770                existing_location,
771            } => {
772                let msg = format!("`{struct_name}` already has a generic named `{name}`");
773                let secondary = format!("`{name}` added here a second time");
774                let mut error = CustomDiagnostic::simple_error(msg, secondary, *duplicate_location);
775
776                let existing_msg = format!("`{name}` was previously defined here");
777                error.add_secondary(existing_msg, *existing_location);
778                error
779            }
780            InterpreterError::CannotResolveExpression { location, expression } => {
781                let msg = format!("Cannot resolve expression `{expression}`");
782                CustomDiagnostic::simple_error(msg, String::new(), *location)
783            }
784            InterpreterError::CannotSetFunctionBody { location, expression } => {
785                let msg = format!("`{expression}` is not a valid function body");
786                CustomDiagnostic::simple_error(msg, String::new(), *location)
787            }
788            InterpreterError::UnknownArrayLength { length, err, location } => {
789                let msg = format!("Could not determine array length `{length}`");
790                let secondary = format!("Evaluating the length failed with: `{err}`");
791                CustomDiagnostic::simple_error(msg, secondary, *location)
792            }
793            InterpreterError::CannotInterpretFormatStringWithErrors { location } => {
794                let msg = "Cannot interpret format string with errors".to_string();
795                let secondary =
796                    "Some of the variables to interpolate could not be evaluated".to_string();
797                CustomDiagnostic::simple_error(msg, secondary, *location)
798            }
799            InterpreterError::GlobalsDependencyCycle { location } => {
800                let msg = "This global recursively depends on itself".to_string();
801                let secondary = String::new();
802                CustomDiagnostic::simple_error(msg, secondary, *location)
803            }
804            InterpreterError::GlobalCouldNotBeResolved { location } => {
805                let msg = "Failed to resolve this global".to_string();
806                let secondary = String::new();
807                CustomDiagnostic::simple_error(msg, secondary, *location)
808            }
809            InterpreterError::LoopHaltedForUiResponsiveness { location } => {
810                let msg = "This loop took too much time to execute so it was halted for UI responsiveness"
811                            .to_string();
812                let secondary =
813                    "This error doesn't happen in normal executions of `nargo`".to_string();
814                CustomDiagnostic::simple_warning(msg, secondary, *location)
815            }
816            InterpreterError::SkippedDueToEarlierErrors => {
817                unreachable!(
818                    "SkippedDueToTypeErrors should be handled internally like Break/Continue"
819                )
820            }
821            InterpreterError::DuplicateStructFieldInSetFields { name, index, previous_index } => {
822                let msg = "Duplicate field name in call to `set_fields`".to_string();
823                let secondary = format!(
824                    "`{name}` first used as field {} then again as field {}",
825                    previous_index + 1,
826                    index + 1
827                );
828                CustomDiagnostic::simple_error(msg, secondary, name.location())
829            }
830            InterpreterError::CheckedTransmuteFailed { actual, expected, location } => {
831                let msg = format!("Checked transmute failed: `{actual:?}` != `{expected:?}`");
832                let secondary = String::new();
833                CustomDiagnostic::simple_error(msg, secondary, *location)
834            }
835            InterpreterError::StackOverflow { location, call_stack } => {
836                let diagnostic = CustomDiagnostic::simple_error(
837                    "Comptime Stack Overflow".to_string(),
838                    "Exceeded the recursion limit".to_string(),
839                    *location,
840                );
841                diagnostic.with_call_stack(call_stack.into_iter().copied().collect())
842            }
843            InterpreterError::EvaluationDepthOverflow { location, call_stack } => {
844                let diagnostic = CustomDiagnostic::simple_error(
845                    "Comptime Evaluation Depth Overflow".to_string(),
846                    "Exceeded the limit on the combined depth of expressions and recursion"
847                        .to_string(),
848                    *location,
849                );
850                diagnostic.with_call_stack(call_stack.into_iter().copied().collect())
851            }
852            InterpreterError::AttributeRecursionLimitExceeded { location } => {
853                CustomDiagnostic::simple_error(
854                    "Attribute recursion limit exceeded".to_string(),
855                    "This attribute generates code with the same attribute, causing infinite recursion".to_string(),
856                    *location,
857                )
858            }
859            InterpreterError::UnexpectedEscapedTokenInQuote { token, location } => {
860                let primary = match token {
861                    Some(token) => format!("`{token}` cannot be escaped in quoted expressions"),
862                    None => "Unexpected end of input after escape character in quoted expression".to_string(),
863                };
864                let secondary = "Only `$` may be escaped in `quote` expressions".to_string();
865                CustomDiagnostic::simple_error(primary, secondary, *location)
866            }
867            InterpreterError::TraitImplResolutionRecursionLimitReached { location } => {
868                let primary = "Trait impl resolution recursion limit reached".to_string();
869                let secondary = String::new();
870                CustomDiagnostic::simple_warning(primary, secondary, *location)
871            }
872        }
873    }
874}

876/// Comptime errors always wrap another error to show it together with a
877/// comptime call or macro "something" that eventually led to that error.
878#[derive(Debug, Clone, PartialEq, Eq)]
879pub enum ComptimeError {
880    ErrorRunningAttribute {
881        error: Box<CompilationError>,
882        location: Location,
883    },
884    ErrorEvaluatingComptimeCall {
885        method_name: &'static str,
886        error: Box<CompilationError>,
887        location: Location,
888    },
889}
890
891impl ComptimeError {
892    pub fn location(&self) -> Location {
893        match self {
894            ComptimeError::ErrorRunningAttribute { location, .. }
895            | ComptimeError::ErrorEvaluatingComptimeCall { location, .. } => *location,
896        }
897    }
898}
899
900impl<'a> From<&'a ComptimeError> for CustomDiagnostic {
901    fn from(error: &'a ComptimeError) -> Self {
902        match error {
903            ComptimeError::ErrorRunningAttribute { error, location } => {
904                let mut diagnostic = CustomDiagnostic::from(&**error);
905                diagnostic.add_secondary("While running this function attribute".into(), *location);
906                diagnostic
907            }
908            ComptimeError::ErrorEvaluatingComptimeCall { method_name, error, location } => {
909                let mut diagnostic = CustomDiagnostic::from(&**error);
910                diagnostic.add_secondary(format!("While evaluating `{method_name}`"), *location);
911                diagnostic
912            }
913        }
914    }
915}
