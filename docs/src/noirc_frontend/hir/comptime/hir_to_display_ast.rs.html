28impl HirStatement {
29    pub fn to_display_ast(&self, interner: &NodeInterner, location: Location) -> Statement {
30        let kind = match self {
31            HirStatement::Let(let_stmt) => {
32                let pattern = let_stmt.pattern.to_display_ast(interner);
33                let r#type = Some(interner.id_type(let_stmt.expression).to_display_ast());
34                let expression = let_stmt.expression.to_display_ast(interner);
35                StatementKind::new_let(pattern, r#type, expression, let_stmt.attributes.clone())
36            }
37            HirStatement::Assign(assign) => StatementKind::Assign(AssignStatement {
38                lvalue: assign.lvalue.to_display_ast(interner),
39                expression: assign.expression.to_display_ast(interner),
40            }),
41            HirStatement::For(for_stmt) => StatementKind::For(ForLoopStatement {
42                identifier: for_stmt.identifier.to_display_ast(interner),
43                range: ForRange::range(
44                    for_stmt.start_range.to_display_ast(interner),
45                    for_stmt.end_range.to_display_ast(interner),
46                    for_stmt.inclusive,
47                ),
48                block: for_stmt.block.to_display_ast(interner),
49                location,
50            }),
51            HirStatement::Loop(block) => StatementKind::Loop(LoopStatement {
52                body: block.to_display_ast(interner),
53                loop_keyword_location: location,
54            }),
55            HirStatement::While(condition, block) => StatementKind::While(WhileStatement {
56                condition: condition.to_display_ast(interner),
57                body: block.to_display_ast(interner),
58                while_keyword_location: location,
59            }),
60            HirStatement::Break => StatementKind::Break,
61            HirStatement::Continue => StatementKind::Continue,
62            HirStatement::Expression(expr) => {
63                StatementKind::Expression(expr.to_display_ast(interner))
64            }
65            HirStatement::Semi(expr) => StatementKind::Semi(expr.to_display_ast(interner)),
66            HirStatement::Error => StatementKind::Error,
67            HirStatement::Comptime(statement) => {
68                StatementKind::Comptime(Box::new(statement.to_display_ast(interner)))
69            }
70        };
71
72        Statement { kind, location }
73    }
74}

86impl HirExpression {
87    /// Convert to AST for display (some details lost)
88    pub fn to_display_ast(&self, interner: &NodeInterner, location: Location) -> Expression {
89        let kind = match self {
90            HirExpression::Ident(ident, generics) => {
91                ident.to_display_expr(interner, generics, location)
92            }
93            HirExpression::Literal(HirLiteral::Array(array)) => {
94                let array = array.to_display_ast(interner, location);
95                ExpressionKind::Literal(Literal::Array(array))
96            }
97            HirExpression::Literal(HirLiteral::Vector(array)) => {
98                let array = array.to_display_ast(interner, location);
99                ExpressionKind::Literal(Literal::Vector(array))
100            }
101            HirExpression::Literal(HirLiteral::Bool(value)) => {
102                ExpressionKind::Literal(Literal::Bool(*value))
103            }
104            HirExpression::Literal(HirLiteral::Integer(value)) => {
105                // Losing the integer suffix information here, but this should just be for
106                // displaying these values anyway
107                ExpressionKind::Literal(Literal::Integer(*value, None))
108            }
109            HirExpression::Literal(HirLiteral::Str(string)) => {
110                ExpressionKind::Literal(Literal::Str(string.clone()))
111            }
112            HirExpression::Literal(HirLiteral::FmtStr(fragments, _exprs, length)) => {
113                // TODO: Is throwing away the exprs here valid?
114                ExpressionKind::Literal(Literal::FmtStr(fragments.clone(), *length))
115            }
116            HirExpression::Literal(HirLiteral::Unit) => ExpressionKind::Literal(Literal::Unit),
117            HirExpression::Block(expr) => ExpressionKind::Block(expr.to_display_ast(interner)),
118            HirExpression::Prefix(prefix) => ExpressionKind::Prefix(Box::new(PrefixExpression {
119                operator: prefix.operator,
120                rhs: prefix.rhs.to_display_ast(interner),
121            })),
122            HirExpression::Infix(infix) => ExpressionKind::Infix(Box::new(InfixExpression {
123                lhs: infix.lhs.to_display_ast(interner),
124                operator: Located::from(infix.operator.location, infix.operator.kind),
125                rhs: infix.rhs.to_display_ast(interner),
126            })),
127            HirExpression::Index(index) => ExpressionKind::Index(Box::new(IndexExpression {
128                collection: index.collection.to_display_ast(interner),
129                index: index.index.to_display_ast(interner),
130            })),
131            HirExpression::Constructor(constructor) => {
132                let type_name = constructor.r#type.borrow().name.to_string();
133                let type_name = Path::from_single(type_name, location);
134                let fields = vecmap(constructor.fields.clone(), |(name, expr): (Ident, ExprId)| {
135                    (name, expr.to_display_ast(interner))
136                });
137
138                ExpressionKind::Constructor(Box::new(ConstructorExpression {
139                    typ: UnresolvedType::from_path(type_name),
140                    fields,
141                }))
142            }
143            HirExpression::MemberAccess(access) => {
144                ExpressionKind::MemberAccess(Box::new(MemberAccessExpression {
145                    lhs: access.lhs.to_display_ast(interner),
146                    rhs: access.rhs.clone(),
147                }))
148            }
149            HirExpression::Call(call) => {
150                let func = Box::new(call.func.to_display_ast(interner));
151                let arguments = vecmap(call.arguments.clone(), |arg| arg.to_display_ast(interner));
152                let is_macro_call = false;
153                ExpressionKind::Call(Box::new(CallExpression { func, arguments, is_macro_call }))
154            }
155            HirExpression::Constrain(constrain) => {
156                let expr = constrain.0.to_display_ast(interner);
157                let mut arguments = vec![expr];
158                if let Some(message) = constrain.2 {
159                    arguments.push(message.to_display_ast(interner));
160                }
161
162                // TODO: Find difference in usage between Assert & AssertEq
163                ExpressionKind::Constrain(ConstrainExpression {
164                    kind: ConstrainKind::Assert,
165                    arguments,
166                    location,
167                })
168            }
169            HirExpression::Cast(cast) => {
170                let lhs = cast.lhs.to_display_ast(interner);
171                let r#type = cast.r#type.to_display_ast();
172                ExpressionKind::Cast(Box::new(CastExpression { lhs, r#type }))
173            }
174            HirExpression::If(if_expr) => ExpressionKind::If(Box::new(IfExpression {
175                condition: if_expr.condition.to_display_ast(interner),
176                consequence: if_expr.consequence.to_display_ast(interner),
177                alternative: if_expr.alternative.map(|expr| expr.to_display_ast(interner)),
178            })),
179            HirExpression::Match(match_expr) => match_expr.to_display_ast(interner, location),
180            HirExpression::Tuple(fields) => {
181                ExpressionKind::Tuple(vecmap(fields, |field| field.to_display_ast(interner)))
182            }
183            HirExpression::Lambda(lambda) => {
184                let parameters = vecmap(lambda.parameters.clone(), |(pattern, typ)| {
185                    (pattern.to_display_ast(interner), Some(typ.to_display_ast()))
186                });
187                let return_type = Some(lambda.return_type.to_display_ast());
188                let body = lambda.body.to_display_ast(interner);
189                let unconstrained = lambda.unconstrained;
190                ExpressionKind::Lambda(Box::new(Lambda {
191                    parameters,
192                    return_type,
193                    body,
194                    unconstrained,
195                }))
196            }
197            HirExpression::Error => ExpressionKind::Error,
198            HirExpression::Unsafe(block) => ExpressionKind::Unsafe(UnsafeExpression {
199                block: block.to_display_ast(interner),
200                unsafe_keyword_location: location,
201            }),
202            HirExpression::Quote(block) => ExpressionKind::Quote(block.clone()),
203
204            // A macro was evaluated here: return the quoted result
205            HirExpression::Unquote(block) => ExpressionKind::Quote(block.clone()),
206
207            // Convert this back into a function call `Enum::Foo(args)`
208            HirExpression::EnumConstructor(constructor) => {
209                let typ = constructor.r#type.borrow();
210                let variant = &typ.variant_at(constructor.variant_index);
211                let segment1 = PathSegment { ident: typ.name.clone(), location, generics: None };
212                let segment2 =
213                    PathSegment { ident: variant.name.clone(), location, generics: None };
214                let path = Path::plain(vec![segment1, segment2], location);
215                let func = Box::new(Expression::new(ExpressionKind::Variable(path), location));
216                let arguments = vecmap(&constructor.arguments, |arg| arg.to_display_ast(interner));
217                let call = CallExpression { func, arguments, is_macro_call: false };
218                ExpressionKind::Call(Box::new(call))
219            }
220        };
221
222        Expression::new(kind, location)
223    }
224}

226impl HirMatch {
227    fn to_display_ast(&self, interner: &NodeInterner, location: Location) -> ExpressionKind {
228        match self {
229            HirMatch::Success(expr) => expr.to_display_ast(interner).kind,
230            HirMatch::Failure { .. } => ExpressionKind::Error,
231            HirMatch::Guard { cond, body, otherwise } => {
232                let condition = cond.to_display_ast(interner);
233                let consequence = body.to_display_ast(interner);
234                let alternative =
235                    Some(Expression::new(otherwise.to_display_ast(interner, location), location));
236
237                ExpressionKind::If(Box::new(IfExpression { condition, consequence, alternative }))
238            }
239            HirMatch::Switch(variable, cases, default) => {
240                let location = interner.definition(*variable).location;
241                let ident = HirIdent::non_trait_method(*variable, location);
242                let expression = ident.to_display_expr(interner, &None, location);
243                let expression = Expression::new(expression, location);
244
245                let mut rules = vecmap(cases, |case| {
246                    let args = vecmap(&case.arguments, |arg| arg.to_display_ast(interner));
247                    let constructor = case.constructor.to_display_ast(args);
248                    let constructor = Expression::new(constructor, location);
249                    let branch = case.body.to_display_ast(interner, location);
250                    (constructor, Expression::new(branch, location))
251                });
252
253                if let Some(case) = default {
254                    let kind =
255                        ExpressionKind::Variable(Path::from_single("_".to_string(), location));
256                    let pattern = Expression::new(kind, location);
257                    let branch = Expression::new(case.to_display_ast(interner, location), location);
258                    rules.push((pattern, branch));
259                }
260
261                ExpressionKind::Match(Box::new(MatchExpression { expression, rules }))
262            }
263        }
264    }
265}

276impl Constructor {
277    fn to_display_ast(&self, arguments: Vec<Expression>) -> ExpressionKind {
278        match self {
279            Constructor::True => ExpressionKind::Literal(Literal::Bool(true)),
280            Constructor::False => ExpressionKind::Literal(Literal::Bool(false)),
281            Constructor::Unit => ExpressionKind::Literal(Literal::Unit),
282            Constructor::Int(value) => ExpressionKind::Literal(Literal::Integer(*value, None)),
283            Constructor::Tuple(_) => ExpressionKind::Tuple(arguments),
284            Constructor::Variant(typ, index) => {
285                let typ = typ.follow_bindings_shallow();
286                let Type::DataType(def, _) = typ.as_ref() else {
287                    return ExpressionKind::Error;
288                };
289
290                let Some(variants) = def.borrow().get_variants_as_written() else {
291                    return ExpressionKind::Error;
292                };
293
294                let Some(name) = variants.get(*index).map(|variant| variant.name.clone()) else {
295                    return ExpressionKind::Error;
296                };
297
298                let location = name.location();
299                let name = ExpressionKind::Variable(Path::from_ident(name));
300                let func = Box::new(Expression::new(name, location));
301                let is_macro_call = false;
302                ExpressionKind::Call(Box::new(CallExpression { func, arguments, is_macro_call }))
303            }
304            Constructor::Range(_start, _end) => {
305                unreachable!("Range is unimplemented")
306            }
307        }
308    }
309}

321impl HirPattern {
322    /// Convert to AST for display (some details lost)
323    fn to_display_ast(&self, interner: &NodeInterner) -> Pattern {
324        match self {
325            HirPattern::Identifier(ident) => Pattern::Identifier(ident.to_display_ast(interner)),
326            HirPattern::Mutable(pattern, location) => {
327                let pattern = Box::new(pattern.to_display_ast(interner));
328                Pattern::Mutable(pattern, *location, false)
329            }
330            HirPattern::Tuple(patterns, location) => {
331                let patterns = vecmap(patterns, |pattern| pattern.to_display_ast(interner));
332                Pattern::Tuple(patterns, *location)
333            }
334            HirPattern::Struct(typ, patterns, location) => {
335                let patterns = vecmap(patterns, |(name, pattern)| {
336                    (name.clone(), pattern.to_display_ast(interner))
337                });
338                let name = match typ.follow_bindings() {
339                    Type::DataType(struct_def, _) => {
340                        let struct_def = struct_def.borrow();
341                        struct_def.name.to_string()
342                    }
343                    // This pass shouldn't error so if the type isn't a struct we just get a string
344                    // representation of any other type and use that. We're relying on name
345                    // resolution to fail later when this Ast is re-converted to Hir.
346                    other => other.to_string(),
347                };
348                // The name span is lost here
349                let path = Path::from_single(name, *location);
350                Pattern::Struct(path, patterns, *location)
351            }
352        }
353    }

364        &self,
365        interner: &NodeInterner,
366        generics: &Option<Vec<Type>>,
367        location: Location,
368    ) -> ExpressionKind {
369        let ident = self.to_display_ast(interner);
370        let segment = PathSegment {
371            ident,

379
380        ExpressionKind::Variable(path)

384impl Type {
385    /// Convert to AST for display (some details lost)
386    fn to_display_ast(&self) -> UnresolvedType {
387        let typ = match self {
388            Type::FieldElement => UnresolvedTypeData::field(Location::dummy()),
389            Type::Array(length, element) => {
390                let length = length.to_type_expression();
391                let element = Box::new(element.to_display_ast());
392                UnresolvedTypeData::Array(length, element)
393            }
394            Type::Vector(element) => {
395                let element = Box::new(element.to_display_ast());
396                UnresolvedTypeData::Vector(element)
397            }
398            Type::Integer(sign, bit_size) => {
399                UnresolvedTypeData::integer(*sign, *bit_size, Location::dummy())
400            }
401            Type::Bool => UnresolvedTypeData::bool(Location::dummy()),
402            Type::String(length) => {
403                let length = length.to_type_expression();
404                UnresolvedTypeData::str(length, Location::dummy())
405            }
406            Type::FmtString(length, element) => {
407                let length = length.to_type_expression();
408                let element = element.to_display_ast();
409                UnresolvedTypeData::fmtstr(length, element, Location::dummy())
410            }
411            Type::Unit => UnresolvedTypeData::Unit,
412            Type::Tuple(fields) => {
413                let fields = vecmap(fields, |field| field.to_display_ast());
414                UnresolvedTypeData::Tuple(fields)
415            }
416            Type::DataType(def, generics) => {
417                let struct_def = def.borrow();
418                let ordered_args = vecmap(generics, |generic| generic.to_display_ast());
419                let generics =
420                    GenericTypeArgs { ordered_args, named_args: Vec::new(), kinds: Vec::new() };
421                let name = Path::from_ident(struct_def.name.clone());
422                UnresolvedTypeData::Named(name, generics, false)
423            }
424            Type::Alias(type_def, generics) => {
425                // Keep the alias name instead of expanding this in case the
426                // alias' definition was changed
427                let type_def = type_def.borrow();
428                let ordered_args = vecmap(generics, |generic| generic.to_display_ast());
429                let generics =
430                    GenericTypeArgs { ordered_args, named_args: Vec::new(), kinds: Vec::new() };
431                let name = Path::from_ident(type_def.name.clone());
432                UnresolvedTypeData::Named(name, generics, false)
433            }
434            Type::TypeVariable(binding) => match &*binding.borrow() {
435                TypeBinding::Bound(typ) => return typ.to_display_ast(),
436                TypeBinding::Unbound(id, type_var_kind) => {
437                    let name = format!("var_{type_var_kind:?}_{id}");
438                    let path =
439                        Path::from_single(name, Location::new(Span::empty(0), FileId::dummy()));
440                    let expression = UnresolvedTypeExpression::Variable(path);
441                    UnresolvedTypeData::Expression(expression)
442                }
443            },
444            Type::TraitAsType(_, name, generics) => {
445                let ordered_args = vecmap(&generics.ordered, |generic| generic.to_display_ast());
446                let named_args = vecmap(&generics.named, |named_type| {
447                    (named_type.name.clone(), named_type.typ.to_display_ast())
448                });
449                let generics = GenericTypeArgs { ordered_args, named_args, kinds: Vec::new() };
450                let name = Path::from_single(name.as_ref().clone(), Location::dummy());
451                UnresolvedTypeData::TraitAsType(name, generics)
452            }
453            Type::NamedGeneric(NamedGeneric { name, .. }) => {
454                let name = Path::from_single(name.as_ref().clone(), Location::dummy());
455                UnresolvedTypeData::Named(name, GenericTypeArgs::default(), true)
456            }
457            Type::CheckedCast { to, .. } => to.to_display_ast().typ,
458            Type::Function(args, ret, env, unconstrained) => {
459                let args = vecmap(args, |arg| arg.to_display_ast());
460                let ret = Box::new(ret.to_display_ast());
461                let env = Box::new(env.to_display_ast());
462                UnresolvedTypeData::Function(args, ret, env, *unconstrained)
463            }
464            Type::Reference(element, mutable) => {
465                let element = Box::new(element.to_display_ast());
466                UnresolvedTypeData::Reference(element, *mutable)
467            }
468            // Type::Forall is only for generic functions which don't store a type
469            // in their Ast so they don't need to call to_display_ast for their Forall type.
470            // Since there is no UnresolvedTypeData equivalent for Type::Forall, we use
471            // this to ignore this case since it shouldn't be needed anyway.
472            Type::Forall(_, typ) => return typ.to_display_ast(),
473            Type::Constant(value, kind) => {
474                UnresolvedTypeData::Expression(UnresolvedTypeExpression::Constant(
475                    *value,
476                    kind.as_integer_type_suffix(),
477                    Location::dummy(),
478                ))
479            }
480            Type::Quoted(quoted_type) => {
481                UnresolvedTypeData::quoted(*quoted_type, Location::dummy())
482            }
483            Type::Error => UnresolvedTypeData::Error,
484            Type::InfixExpr(lhs, op, rhs, _) => {
485                let lhs = Box::new(lhs.to_type_expression());
486                let rhs = Box::new(rhs.to_type_expression());
487                let location = Location::dummy();
488                let expr = UnresolvedTypeExpression::BinaryOperation(lhs, *op, rhs, location);
489                UnresolvedTypeData::Expression(expr)
490            }
491        };
492
493        UnresolvedType { typ, location: Location::dummy() }
494    }
495
496    /// Convert to AST for display (some details lost)
497    fn to_type_expression(&self) -> UnresolvedTypeExpression {
498        let location = Location::dummy();
499
500        match self.follow_bindings() {
501            Type::Constant(length, kind) => {
502                UnresolvedTypeExpression::Constant(length, kind.as_integer_type_suffix(), location)
503            }
504            Type::NamedGeneric(NamedGeneric { name, .. }) => {
505                let path = Path::from_single(name.as_ref().clone(), location);
506                UnresolvedTypeExpression::Variable(path)
507            }
508            // TODO: This should be turned into a proper error.
509            other => panic!("Cannot represent {other:?} as type expression"),
510        }
511    }
512}

515    /// Convert to AST for display (some details lost)
516    fn to_display_ast(&self, interner: &NodeInterner) -> LValue {

529            }
530            HirLValue::Dereference { lvalue, element_type: _, location, implicitly_added: _ } => {
531                let lvalue = Box::new(lvalue.to_display_ast(interner));
532                LValue::Dereference(lvalue, *location)
533            }
534        }

538impl HirArrayLiteral {
539    /// Convert to AST for display (some details lost)
540    fn to_display_ast(&self, interner: &NodeInterner, location: Location) -> ArrayLiteral {
541        match self {
542            HirArrayLiteral::Standard(elements) => {
543                ArrayLiteral::Standard(vecmap(elements, |element| element.to_display_ast(interner)))
544            }
545            HirArrayLiteral::Repeated { repeated_element, length } => {
546                let repeated_element = Box::new(repeated_element.to_display_ast(interner));
547                let length = match length {
548                    Type::Constant(length, kind) => {
549                        let suffix = kind.as_integer_type_suffix();
550                        let literal = Literal::Integer(*length, suffix);
551                        let expr_kind = ExpressionKind::Literal(literal);
552                        Box::new(Expression::new(expr_kind, location))
553                    }
554                    other => panic!(
555                        "Cannot convert non-constant type for repeated array literal from Hir -> Ast: {other:?}"
556                    ),
557                };
558                ArrayLiteral::Repeated { repeated_element, length }
559            }
560        }
561    }
