25#[derive(Error, Debug, Clone, PartialEq, Eq)]
26pub enum DefCollectorErrorKind {
27    #[error("Duplicate {typ}")]
28    Duplicate { typ: DuplicateType, first_def: Ident, second_def: Ident },
29    #[error("Unresolved import")]
30    UnresolvedModuleDecl { mod_name: Ident, expected_path: String, alternative_path: String },
31    #[error("Overlapping imports")]
32    OverlappingModuleDecls { mod_name: Ident, expected_path: String, alternative_path: String },
33    #[error("Path resolution error")]
34    PathResolutionError(PathResolutionError),
35    #[error("Cannot re-export {item_name} because it has less visibility than this use statement")]
36    CannotReexportItemWithLessVisibility { item_name: Ident, desired_visibility: ItemVisibility },
37    #[error("Non-enum, non-struct type used in impl")]
38    NonEnumNonStructTypeInImpl { location: Location, is_primitive: bool },
39    #[error("Cannot implement trait on a reference type")]
40    ReferenceInTraitImpl { is_alias: bool, location: Location },
41    #[error("Impl for type `{typ}` overlaps with existing impl")]
42    OverlappingImpl { typ: crate::Type, location: Location, prev_location: Location },
43    #[error("Cannot `impl` a type defined outside the current crate")]
44    ForeignImpl { location: Location, type_name: String },
45    #[error("Method is not defined in trait")]
46    MethodNotInTrait { trait_name: Ident, impl_method: Ident },
47    #[error("Only traits can be implemented")]
48    NotATrait { not_a_trait_name: TypedPath },
49    #[error("Trait not found")]
50    TraitNotFound { trait_path: TypedPath },
51    #[error("Missing Trait method implementation")]
52    TraitMissingMethod { trait_name: Ident, method_name: Ident, trait_impl_location: Location },
53    #[error("Module is already part of the crate")]
54    ModuleAlreadyPartOfCrate { mod_name: Ident, location: Location },
55    #[error("Module was originally declared here")]
56    ModuleOriginallyDefined { mod_name: Ident, location: Location },
57    #[error("Either the type or the trait must be from the same crate as the trait implementation")]
58    TraitImplOrphaned { location: Location },
59    #[error("impl has stricter requirements than trait")]
60    ImplIsStricterThanTrait {
61        constraint_typ: crate::Type,
62        constraint_name: String,
63        constraint_generics: TraitGenerics,
64        constraint_location: Location,
65        trait_method_name: String,
66        trait_method_location: Location,
67    },
68    #[error("The `#[test]` attribute may only be used on a non-associated function")]
69    TestOnAssociatedFunction { location: Location },
70    #[error("The `#[export]` attribute may only be used on a non-associated function")]
71    ExportOnAssociatedFunction { location: Location },
72    #[error(
73        "The `#[test(only_fail_with = \"..\")]` attribute may only be used on functions with parameters"
74    )]
75    TestOnlyFailWithWithoutParameters { location: Location },
76    #[error("The `#[fuzz]` attribute may only be used on functions with parameters")]
77    FuzzingHarnessWithoutParameters { location: Location },
78    #[error("`{name}` entry-point function is not allowed to have generic parameters")]
79    EntryPointWithGenerics { name: String, location: Location },
80}

82impl DefCollectorErrorKind {
83    pub fn location(&self) -> Location {
84        match self {
85            DefCollectorErrorKind::Duplicate { second_def: ident, .. }
86            | DefCollectorErrorKind::UnresolvedModuleDecl { mod_name: ident, .. }
87            | DefCollectorErrorKind::CannotReexportItemWithLessVisibility {
88                item_name: ident,
89                ..
90            }
91            | DefCollectorErrorKind::MethodNotInTrait { impl_method: ident, .. }
92            | DefCollectorErrorKind::OverlappingModuleDecls { mod_name: ident, .. } => {
93                ident.location()
94            }
95            DefCollectorErrorKind::PathResolutionError(path_resolution_error) => {
96                path_resolution_error.location()
97            }
98            DefCollectorErrorKind::ImplIsStricterThanTrait {
99                trait_method_location: location,
100                ..
101            }
102            | DefCollectorErrorKind::TestOnAssociatedFunction { location }
103            | DefCollectorErrorKind::ExportOnAssociatedFunction { location }
104            | DefCollectorErrorKind::NonEnumNonStructTypeInImpl { location, .. }
105            | DefCollectorErrorKind::ReferenceInTraitImpl { location, .. }
106            | DefCollectorErrorKind::OverlappingImpl { location, .. }
107            | DefCollectorErrorKind::ModuleAlreadyPartOfCrate { location, .. }
108            | DefCollectorErrorKind::ModuleOriginallyDefined { location, .. }
109            | DefCollectorErrorKind::TraitImplOrphaned { location }
110            | DefCollectorErrorKind::TraitMissingMethod { trait_impl_location: location, .. }
111            | DefCollectorErrorKind::ForeignImpl { location, .. }
112            | DefCollectorErrorKind::TestOnlyFailWithWithoutParameters { location }
113            | DefCollectorErrorKind::FuzzingHarnessWithoutParameters { location }
114            | DefCollectorErrorKind::EntryPointWithGenerics { location, .. } => *location,
115            DefCollectorErrorKind::NotATrait { not_a_trait_name: path }
116            | DefCollectorErrorKind::TraitNotFound { trait_path: path } => path.location,
117        }
118    }
119}

121impl<'a> From<&'a UnsupportedNumericGenericType> for Diagnostic {
122    fn from(error: &'a UnsupportedNumericGenericType) -> Diagnostic {
123        let message = if let Some(name) = &error.name {
124            format!(
125                "{name} has a type of {}. The only supported numeric generic types are `u1`, `u8`, `u16`, and `u32`.",
126                error.typ
127            )
128        } else {
129            error.to_string()
130        };
131
132        Diagnostic::simple_error(
133            message,
134            "Unsupported numeric generic type".to_string(),
135            error.location,
136        )
137    }
138}

140impl fmt::Display for DuplicateType {
141    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
142        match *self {
143            DuplicateType::Function => write!(f, "function"),
144            DuplicateType::Module => write!(f, "module"),
145            DuplicateType::Global => write!(f, "global"),
146            DuplicateType::TypeDefinition => write!(f, "type definition"),
147            DuplicateType::Trait => write!(f, "trait definition"),
148            DuplicateType::TraitImplementation => write!(f, "trait implementation"),
149            DuplicateType::Import => write!(f, "import"),
150            DuplicateType::TraitAssociatedItem => write!(f, "trait associated item"),
151            DuplicateType::StructField => write!(f, "struct field"),
152            DuplicateType::EnumVariant => write!(f, "enum variant"),
153        }
154    }
155}

157impl<'a> From<&'a DefCollectorErrorKind> for Diagnostic {
158    fn from(error: &'a DefCollectorErrorKind) -> Diagnostic {
159        match error {
160            DefCollectorErrorKind::Duplicate { typ, first_def, second_def } => {
161                let primary_message = format!(
162                    "Duplicate definitions of {} with name {} found",
163                    &typ, first_def
164                );
165                {
166                    let mut diag = Diagnostic::simple_error(
167                        primary_message,
168                        format!("Second {} found here", &typ),
169                        second_def.location(),
170                    );
171                    diag.add_secondary(format!("First {} found here", &typ), first_def.location());
172                    diag
173                }
174            }
175            DefCollectorErrorKind::UnresolvedModuleDecl { mod_name, expected_path, alternative_path } => {
176                Diagnostic::simple_error(
177                    format!("No module `{mod_name}` at path `{expected_path}` or `{alternative_path}`"),
178                    String::new(),
179                    mod_name.location(),
180                )
181            }
182            DefCollectorErrorKind::OverlappingModuleDecls { mod_name, expected_path, alternative_path } => {
183                Diagnostic::simple_error(
184                    format!("Overlapping modules `{mod_name}` at  path `{expected_path}` and `{alternative_path}`"),
185                    String::new(),
186                    mod_name.location(),
187                )
188            }
189            DefCollectorErrorKind::PathResolutionError(error) => error.into(),
190            DefCollectorErrorKind::CannotReexportItemWithLessVisibility{item_name, desired_visibility} => {
191                Diagnostic::simple_error(
192                    format!("cannot re-export {item_name} because it has less visibility than this use statement"),
193                    format!("consider marking {item_name} as {desired_visibility}"),
194                    item_name.location())
195            }
196            DefCollectorErrorKind::NonEnumNonStructTypeInImpl { location, is_primitive } =>{
197                if *is_primitive {
198                    Diagnostic::simple_error(
199                        "Cannot define inherent `impl` for primitive types".into(),
200                        "Primitive types can only have implementation methods defined in the standard library".into(),
201                        *location,
202                    )
203                }else{
204                    Diagnostic::simple_error(
205                        "Non-enum, non-struct type used in impl".into(),
206                        "Only enum and struct types may have implementation methods".into(),
207                        *location,
208                    )
209                }
210            }
211            DefCollectorErrorKind::ReferenceInTraitImpl { is_alias, location } => {
212                let alias_str = if *is_alias {
213                    "aliases to "
214                } else {
215                    ""
216                };
217                Diagnostic::simple_error(
218                    format!("Trait impls are not allowed on {alias_str}reference types"),
219                    "Try using a struct or enum type here instead".into(),
220                    *location,
221                )
222            }
223            DefCollectorErrorKind::OverlappingImpl { location, typ, prev_location } => {
224                let mut diagnostic = Diagnostic::simple_error(
225                    format!("Impl for type `{typ}` overlaps with existing impl"),
226                    "Overlapping impl".into(),
227                    *location,
228                );
229                diagnostic.add_secondary("Previous impl defined here".into(), *prev_location);
230                diagnostic
231            }
232            DefCollectorErrorKind::ForeignImpl { location, type_name } => Diagnostic::simple_error(
233                "Cannot `impl` a type that was defined outside the current crate".into(),
234                format!("{type_name} was defined outside the current crate"),
235                *location,
236            ),
237            DefCollectorErrorKind::TraitNotFound { trait_path } => Diagnostic::simple_error(
238                format!("Trait {trait_path} not found"),
239                "".to_string(),
240                trait_path.location,
241            ),
242            DefCollectorErrorKind::MethodNotInTrait { trait_name, impl_method } => {
243                let primary_message = format!("Method with name `{impl_method}` is not part of trait `{trait_name}`, therefore it can't be implemented");
244                Diagnostic::simple_error(primary_message, "".to_owned(), impl_method.location())
245            }
246            DefCollectorErrorKind::TraitMissingMethod {
247                trait_name,
248                method_name,
249                trait_impl_location,
250            } => {
251                let primary_message = format!(
252                    "Method `{method_name}` from trait `{trait_name}` is not implemented"
253                );
254                Diagnostic::simple_error(
255                    primary_message,
256                    format!("Please implement {method_name} here"),
257                    *trait_impl_location,
258                )
259            }
260            DefCollectorErrorKind::NotATrait { not_a_trait_name } => {
261                let location = not_a_trait_name.location;
262                Diagnostic::simple_error(
263                    format!("{not_a_trait_name} is not a trait, therefore it can't be implemented"),
264                    String::new(),
265                    location,
266                )
267            }
268            DefCollectorErrorKind::ModuleAlreadyPartOfCrate { mod_name, location } => {
269                let message = format!("Module '{mod_name}' is already part of the crate");
270                let secondary = String::new();
271                Diagnostic::simple_error(message, secondary, *location)
272            }
273            DefCollectorErrorKind::ModuleOriginallyDefined { mod_name, location } => {
274                let message = format!("Note: {mod_name} was originally declared here");
275                let secondary = String::new();
276                Diagnostic::simple_error(message, secondary, *location)
277            }
278            DefCollectorErrorKind::TraitImplOrphaned { location } => Diagnostic::simple_error(
279                "Orphaned trait implementation".into(),
280                "Either the type or the trait must be from the same crate as the trait implementation".into(),
281                *location,
282            ),
283            DefCollectorErrorKind::ImplIsStricterThanTrait { constraint_typ, constraint_name, constraint_generics, constraint_location, trait_method_name, trait_method_location } => {
284                let constraint = format!("{constraint_name}{constraint_generics}");
285
286                let mut diag = Diagnostic::simple_error(
287                    "impl has stricter requirements than trait".to_string(),
288                    format!("impl has extra requirement `{constraint_typ}: {constraint}`"),
289                    *constraint_location,
290                );
291                diag.add_secondary(format!("definition of `{trait_method_name}` from trait"), *trait_method_location);
292                diag
293            }
294            DefCollectorErrorKind::TestOnAssociatedFunction { location } => Diagnostic::simple_error(
295                "The `#[test]` attribute is disallowed on `impl` methods".into(),
296                String::new(),
297                *location,
298            ),
299            DefCollectorErrorKind::ExportOnAssociatedFunction { location } => Diagnostic::simple_error(
300                "The `#[export]` attribute is disallowed on `impl` methods".into(),
301                String::new(),
302                *location,
303            ),
304            DefCollectorErrorKind::TestOnlyFailWithWithoutParameters { location } => Diagnostic::simple_error(
305                "The `#[test(only_fail_with = \"..\")]` attribute may only be used on functions with parameters".into(),
306                String::new(),
307                *location,
308            ),
309            DefCollectorErrorKind::FuzzingHarnessWithoutParameters { location } => Diagnostic::simple_error(
310                "The `#[fuzz]` attribute may only be used on functions with parameters".into(),
311                String::new(),
312                *location,
313            ),
314            DefCollectorErrorKind::EntryPointWithGenerics { name, location } => Diagnostic::simple_error(
315                format!("`{name}` entry-point function is not allowed to have generic parameters"),
316                String::new(),
317                *location,
318            ),
319        }
320    }
