146/// A batch of imports that resulted from desugaring a single import statement.
147/// This can happen as a result of expanding `use foo::{bar, baz}` into `use foo:bar; use foo:baz;`.
148/// Keeping the desugared imports as a batch is important because all imports in a batch
149/// will first be resolved, and just then those items will be brought in scope.
150/// This lets `use foo::{foo, bar}` work by first resolving `foo::foo`, then resolving `foo::bar`,
151/// then finally adding those items to scope. Compare this with `use foo::foo; use foo::bar;` where
152/// the second import will likely error as `foo` will mean `foo::foo` and not `foo`.
153pub type ImportDirectiveBatch = Vec<ImportDirective>;

253#[derive(Debug, Clone, PartialEq, Eq)]
254pub enum CompilationError {
255    ParseError(ParserError),
256    DefinitionError(DefCollectorErrorKind),
257    ResolverError(ResolverError),
258    TypeError(TypeCheckError),
259    InterpreterError(InterpreterError),
260    ComptimeError(ComptimeError),
261    DebugComptimeScopeNotFound(Vec<PathBuf>, Location),
262}
263
264impl CompilationError {
265    /// Returns the primary location where this error happened.
266    pub fn location(&self) -> Location {
267        match self {
268            CompilationError::ParseError(error) => error.location(),
269            CompilationError::DefinitionError(error) => error.location(),
270            CompilationError::ResolverError(error) => error.location(),
271            CompilationError::TypeError(error) => error.location(),
272            CompilationError::InterpreterError(error) => error.location(),
273            CompilationError::ComptimeError(error) => error.location(),
274            CompilationError::DebugComptimeScopeNotFound(_, location) => *location,
275        }
276    }
277
278    pub(crate) fn is_error(&self) -> bool {
279        // This is a bit expensive but not all error types have a `is_warning` method
280        // and it'd lead to code duplication to add them. `CompilationError::is_error`
281        // also isn't expected to be called too often.
282        CustomDiagnostic::from(self).is_error()
283    }
284
285    pub(crate) fn is_expecting_other_error(&self) -> bool {
286        self.as_expecting_other_error().is_some()
287    }
288
289    pub(crate) fn as_expecting_other_error(&self) -> Option<&ExpectingOtherError> {
290        match self {
291            CompilationError::TypeError(TypeCheckError::ExpectingOtherError(e))
292            | CompilationError::InterpreterError(InterpreterError::ExpectingOtherError(e)) => {
293                Some(e)
294            }
295            _ => None,
296        }
297    }
298
299    pub(crate) fn should_be_filtered(&self) -> bool {
300        let CompilationError::InterpreterError(error) = self else {
301            return false;
302        };
303
304        error.should_be_filtered()
305    }
306}

322impl<'a> From<&'a CompilationError> for CustomDiagnostic {
323    fn from(value: &'a CompilationError) -> Self {
324        match value {
325            CompilationError::ParseError(error) => error.into(),
326            CompilationError::DefinitionError(error) => error.into(),
327            CompilationError::ResolverError(error) => error.into(),
328            CompilationError::TypeError(error) => error.into(),
329            CompilationError::InterpreterError(error) => error.into(),
330            CompilationError::ComptimeError(error) => error.into(),
331            CompilationError::DebugComptimeScopeNotFound(error, _) => {
332                let msg = "multiple files found matching --debug-comptime path".into();
333                let secondary = error.iter().fold(String::new(), |mut output, path| {
334                    let _ = writeln!(output, "    {}", path.display());
335                    output
336                });
337                // NOTE: this location is empty as it is not expected to be displayed
338                let dummy_location = Location::dummy();
339                CustomDiagnostic::simple_error(msg, secondary, dummy_location)
340            }
341        }
342    }

394    /// Collect all of the definitions in a given crate into a CrateDefMap
395    /// Modules which are not a part of the module hierarchy starting with
396    /// the root module, will be ignored.
397    #[allow(clippy::too_many_arguments)]
398    #[tracing::instrument(level = "trace", skip_all)]
399    pub fn collect_crate_and_dependencies(
400        mut def_map: CrateDefMap,
401        context: &mut Context,
402        ast: SortedModule,
403        root_file_id: FileId,
404        options: FrontendOptions,
405    ) -> Vec<CompilationError> {
406        let mut errors = CompilationErrors::default();
407        let crate_id = def_map.krate();
408
409        // Recursively resolve the dependencies
410        //
411        // Dependencies are fetched from the crate graph
412        // Then added these to the context of DefMaps once they are resolved
413        //
414        let crate_graph = &context.crate_graph[crate_id];
415
416        for dep in crate_graph.dependencies.clone() {
417            errors.extend(CrateDefMap::collect_defs(dep.crate_id, context, options));
418
419            let dep_def_map =
420                context.def_map(&dep.crate_id).expect("ICE: def map was just created");
421
422            let dep_def_root = dep_def_map.root();
423            let module_id = ModuleId { krate: dep.crate_id, local_id: dep_def_root };
424            // Add this crate as a dependency by linking it's root module
425            def_map.extern_prelude.insert(dep.as_name(), module_id);
426
427            let location = dep_def_map[dep_def_root].location;
428            let attributes = ModuleAttributes {
429                name: dep.as_name(),
430                location,
431                parent: None,
432                visibility: ItemVisibility::Public,
433            };
434            context.def_interner.add_module_attributes(module_id, attributes);
435        }
436
437        // At this point, all dependencies are resolved and type checked.
438        //
439        // It is now possible to collect all of the definitions of this crate.
440        let crate_root = def_map.root();
441        let def_collector = DefCollector::new(def_map);
442
443        let reuse_existing_module_declarations = false;
444        Self::collect_defs_and_elaborate(
445            ast,
446            root_file_id,
447            crate_root,
448            crate_id,
449            context,
450            def_collector,
451            options,
452            reuse_existing_module_declarations,
453            &mut errors,
454        );
455
456        Self::check_unused_items(context, crate_id, &mut errors);
457
458        filter_expecting_other_errors(errors.into_errors())
459    }

461    /// This method does several things:
462    ///
463    /// 1. Collects definitions in the given module
464    /// 2. Injects the prelude in the collected modules
465    /// 3. Processes all of the imports collected during definition collection
466    /// 4. Elaborates all of the collected items
467    ///
468    /// Any errors are appended to `errors`.
469    ///
470    /// If `reuse_existing_module_declarations` is true, when module declarations like `mod some_module;` are
471    /// encountered they will try to be found in existing modules in the `def_collector.def_map`
472    /// field. This is only `true` when re-checking a file in LSP, where nested modules don't
473    /// need to be re-collected and re-type-checked.
474    #[allow(clippy::too_many_arguments)]
475    #[tracing::instrument(level = "trace", skip_all)]
476    pub fn collect_defs_and_elaborate(
477        ast: SortedModule,
478        file_id: FileId,
479        local_module_id: LocalModuleId,
480        crate_id: CrateId,
481        context: &mut Context,
482        mut def_collector: DefCollector,
483        options: FrontendOptions,
484        reuse_existing_module_declarations: bool,
485        errors: &mut CompilationErrors,
486    ) {
487        let module_id = ModuleId { krate: crate_id, local_id: local_module_id };
488        context
489            .def_interner
490            .set_doc_comments(ReferenceId::Module(module_id), ast.inner_doc_comments.clone());
491
492        // Check how many modules there were before we start collecting items
493        let modules_count_before_collect = def_collector.def_map.modules().iter().count();
494
495        // Collecting module declarations with ModCollector
496        // and lowering the functions
497        // i.e. Use a mod collector to collect the nodes at the root module
498        // and process them
499        errors.extend(collect_defs(
500            &mut def_collector,
501            ast,
502            file_id,
503            local_module_id,
504            crate_id,
505            context,
506            reuse_existing_module_declarations,
507        ));
508
509        let submodules =
510            vecmap(def_collector.def_map.modules().iter(), |(index, _)| LocalModuleId::new(index));
511        // Add the current crate to the collection of DefMaps
512        context.def_maps.insert(crate_id, def_collector.def_map);
513
514        inject_prelude(crate_id, context, local_module_id, &mut def_collector.imports);
515
516        // We only need to inject the prelude for modules that were collected in `collect_defs`.
517        // This isn't important or necessary in the compiler. However, in LSP a file might be
518        // re-checked multiple times, ending in new modules in addition to many already existing
519        // before this function call, and there's no need to inject the prelude in those existing
520        // modules (it's very slow).
521        for submodule in submodules.iter().skip(modules_count_before_collect) {
522            inject_prelude(crate_id, context, *submodule, &mut def_collector.imports);
523        }
524
525        Self::process_imports(def_collector.imports, crate_id, context, errors);
526
527        let debug_comptime_in_file = options.debug_comptime_in_file.and_then(|file_suffix| {
528            let file = context.file_manager.find_by_path_suffix(file_suffix);
529            file.unwrap_or_else(|error| {
530                let location = Location::new(Span::empty(0), file_id);
531                errors.push(CompilationError::DebugComptimeScopeNotFound(error, location));
532                None
533            })
534        });
535
536        let cli_options = crate::elaborator::ElaboratorOptions {
537            debug_comptime_in_file,
538            enabled_unstable_features: options.enabled_unstable_features,
539            disable_required_unstable_features: options.disable_required_unstable_features,
540        };
541
542        let more_errors =
543            Elaborator::elaborate(context, crate_id, def_collector.items, cli_options);
544
545        errors.extend(more_errors);
546    }

548    #[tracing::instrument(level = "trace", skip_all)]
549    fn process_imports(
550        mut collected_imports: Vec<ImportDirectiveBatch>,
551        crate_id: CrateId,
552        context: &mut Context,
553        outer_errors: &mut CompilationErrors,
554    ) {
555        let mut inner_errors = CompilationErrors::default();
556
557        // Resolve unresolved imports collected from the crate, one by one.
558        // Imports that cannot be resolved don't error right away. Instead, they are put in
559        // `import_with_errors`. After trying to resolve all imports, we'll try to resolve, once
560        // more, the imports that previously failed. They can succeed this time if they import
561        // a re-export that is defined later in the crate, as in this case:
562        //
563        // ```noir
564        // mod one {
565        //     // This will fail to resolve the first time, as it refers to a re-export that comes
566        //     // later on. However, after the re-export below is processed, this import will succeed.
567        //     use crate::two::Exported;
568        // }
569        // mod two {
570        //     mod three {
571        //         pub struct Exported {}
572        //     }
573        //     pub use three::Exported;
574        // }
575        // ```
576        //
577        // An alternative way could be to first process pub exports, then pub(crate) exports, then
578        // private exports, but this way is a bit more robust as the above case will work even if
579        // the first `use` is `pub`.
580        loop {
581            let mut imports_with_errors = Vec::new();
582
583            // Keep track of whether we could at least resolve one import. If not, we shouldn't
584            // continue trying to resolve imports indefinitely.
585            let mut resolved_at_least_one_import = false;
586
587            // Resolve unresolved imports collected from the crate, by batch
588            for batch in collected_imports {
589                // First resolve all imports in the batch.
590                // See the doc comments on ImportDirectiveBatch for an explanation of why this is done.
591                let resolved_batch = vecmap(batch, |collected_import| {
592                    let local_module_id = collected_import.module_id;
593                    let module_id = ModuleId { krate: crate_id, local_id: local_module_id };
594
595                    let resolved_import = resolve_import(
596                        collected_import.path.clone(),
597                        module_id,
598                        &context.def_maps,
599                        &mut context.usage_tracker,
600                        Some(ReferencesTracker::new(&mut context.def_interner)),
601                    );
602                    (resolved_import, collected_import)
603                });
604
605                let batch_has_errors = resolved_batch.iter().any(|(result, _)| result.is_err());
606                if batch_has_errors {
607                    // If there are errors with this batch, remember it and proceed with the next batch
608                    imports_with_errors.push(resolved_batch);
609                } else {
610                    // Successful batches are processed right away
611                    resolved_at_least_one_import = true;
612
613                    for (result, collected_import) in resolved_batch {
614                        let resolved_import =
615                            result.expect("We already checked that all results are Ok");
616                        Self::process_resolved_import(
617                            resolved_import,
618                            &collected_import,
619                            crate_id,
620                            context,
621                            &mut inner_errors,
622                        );
623                    }
624                }
625            }
626
627            // If we could process all batches without errors, we are done.
628            if imports_with_errors.is_empty() {
629                break;
630            }
631
632            // If we made some progress, try again with batches that errored.
633            // They might succeed next time.
634            if resolved_at_least_one_import {
635                collected_imports = vecmap(imports_with_errors, |batch| {
636                    vecmap(batch, |(_, collected_import)| collected_import)
637                });
638                continue;
639            }
640
641            // If we didn't make any progress, give up and:
642            // 1. process successful imports
643            // 2. record errors of unsuccessful ones
644            for batch in imports_with_errors {
645                for (result, collected_import) in batch {
646                    match result {
647                        Ok(resolved_import) => {
648                            Self::process_resolved_import(
649                                resolved_import,
650                                &collected_import,
651                                crate_id,
652                                context,
653                                &mut inner_errors,
654                            );
655                        }
656                        Err(error) => {
657                            let error = DefCollectorErrorKind::PathResolutionError(error);
658                            inner_errors.push(error);
659                        }
660                    }
661                }
662            }
663            break;
664        }
665
666        // Because a use tree like `use foo::{bar, baz}` is desugared into `use foo::bar; use foo::baz;`,
667        // if `foo` is not found then the error will happen twice. Given that only one error kind can happen
668        // per path segment (either the segment was not found, or it's private, or it doesn't resolve to a module),
669        // it's safe to only keep one error per location.
670        let inner_errors = inner_errors.into_iter().map(|error| (error.location(), error));
671        let inner_errors = inner_errors.collect::<HashMap<_, _>>().into_values();
672        outer_errors.extend(inner_errors);
673    }
674
675    /// Processes a resolved import by bringing into scope the imported item.
676    fn process_resolved_import(
677        resolved_import: ResolvedImport,
678        collected_import: &ImportDirective,
679        crate_id: CrateId,
680        context: &mut Context,
681        errors: &mut CompilationErrors,
682    ) {
683        let local_module_id = collected_import.module_id;
684        let current_def_map = context.def_maps.get_mut(&crate_id).unwrap();
685        let file_id = current_def_map.file_id(local_module_id);
686
687        let has_path_resolution_error = !resolved_import.errors.is_empty();
688        for error in resolved_import.errors {
689            errors.push(DefCollectorErrorKind::PathResolutionError(error));
690        }
691
692        // Populate module namespaces according to the imports used
693        let name = collected_import.name();
694        let visibility = collected_import.visibility;
695        let is_prelude = collected_import.is_prelude;
696        for (module_def_id, item_visibility, _) in resolved_import.namespace.iter_items() {
697            if item_visibility < visibility {
698                errors.push(DefCollectorErrorKind::CannotReexportItemWithLessVisibility {
699                    item_name: name.clone(),
700                    desired_visibility: visibility,
701                });
702            }
703            let visibility = visibility.min(item_visibility);
704
705            let result = current_def_map.index_mut(local_module_id).import(
706                name.clone(),
707                visibility,
708                module_def_id,
709                is_prelude,
710            );
711
712            // If we error on path resolution don't also say it's unused (in case it ends up being unused)
713            if !has_path_resolution_error {
714                let defining_module = ModuleId { krate: crate_id, local_id: local_module_id };
715
716                context.usage_tracker.add_unused_item(
717                    defining_module,
718                    name.clone(),
719                    UnusedItem::Import,
720                    visibility,
721                );
722
723                if visibility != ItemVisibility::Private {
724                    context.def_interner.register_name_for_auto_import(
725                        name.to_string(),
726                        module_def_id,
727                        file_id,
728                        visibility,
729                        Some(defining_module),
730                    );
731
732                    context.def_interner.add_reexport(
733                        module_def_id,
734                        defining_module,
735                        name.clone(),
736                        visibility,
737                    );
738                }
739            }
740
741            if let Some(ref alias) = collected_import.alias {
742                add_import_reference(module_def_id, alias, &mut context.def_interner, file_id);
743            }
744
745            if let Err((first_def, second_def)) = result {
746                let err = DefCollectorErrorKind::Duplicate {
747                    typ: DuplicateType::Import,
748                    first_def,
749                    second_def,
750                };
751                errors.push(err);
752            }
753        }
754    }
755
756    fn check_unused_items(context: &Context, crate_id: CrateId, errors: &mut CompilationErrors) {
757        let unused_imports = context.usage_tracker.unused_items().iter();
758        let unused_imports = unused_imports.filter(|(module_id, _)| module_id.krate == crate_id);
759        let mut unused_errors = unused_imports
760            .flat_map(|(_, unused_items)| {
761                unused_items.iter().map(|(ident, unused_item)| {
762                    let ident = ident.clone();
763                    CompilationError::ResolverError(ResolverError::UnusedItem {
764                        ident,
765                        item: *unused_item,
766                    })
767                })
768            })
769            .collect::<Vec<_>>();
770
771        // Make sure errors always show up in the same order when compiling the same codebase
772        unused_errors.sort_by_key(|error| error.location());
773
774        errors.extend(unused_errors);
775    }
776}

778fn add_import_reference(
779    def_id: crate::hir::def_map::ModuleDefId,
780    name: &Ident,
781    interner: &mut NodeInterner,
782    file_id: FileId,
783) {
784    if name.span() == Span::empty(0) {
785        // We ignore empty spans at 0 location, this must be Stdlib
786        return;
787    }
788
789    let location = Location::new(name.span(), file_id);
790    interner.add_module_def_id_reference(def_id, location, false);
791}

793#[tracing::instrument(level = "trace", skip_all)]
794fn inject_prelude(
795    crate_id: CrateId,
796    context: &mut Context,
797    crate_root: LocalModuleId,
798    collected_imports: &mut Vec<ImportDirectiveBatch>,
799) {
800    if crate_id.is_stdlib() {
801        return;
802    }
803
804    let location = Location::dummy();
805
806    let segments: Vec<_> = "std::prelude"
807        .split("::")
808        .map(|segment| PathSegment::from(Ident::new(segment.into(), location)))
809        .collect();
810
811    let path = Path {
812        segments: segments.clone(),
813        kind: PathKind::Absolute,
814        location,
815        kind_location: location,
816    };
817
818    let Ok(resolved_import) = resolve_import(
819        path,
820        ModuleId { krate: crate_id, local_id: crate_root },
821        &context.def_maps,
822        &mut context.usage_tracker,
823        None, // references tracker
824    ) else {
825        return;
826    };
827
828    assert!(resolved_import.errors.is_empty(), "Tried to add private item to prelude");
829
830    let (module_def_id, _, _) =
831        resolved_import.namespace.types.expect("couldn't resolve std::prelude");
832    let module_id = module_def_id.as_module().expect("std::prelude should be a module");
833    let directives = vecmap(context.module(module_id).scope().names(), |path| {
834        let mut segments = segments.clone();
835        segments.push(PathSegment::from(Ident::new(path.to_string(), location)));
836
837        let path = Path { segments, kind: PathKind::Absolute, location, kind_location: location };
838
839        ImportDirective {
840            visibility: ItemVisibility::Private,
841            module_id: crate_root,
842            path,
843            alias: None,
844            is_prelude: true,
845        }
846    });
847
848    collected_imports.insert(0, directives);
849}

851/// Filter the errors so that:
852/// * if we have any errors which are _not_ [ExpectingOtherError], then we remove all [ExpectingOtherError] instances
853/// * if there are no other kind of errors, then we leave and deduplicate the [ExpectingOtherError]s
854fn filter_expecting_other_errors(mut errors: Vec<CompilationError>) -> Vec<CompilationError> {
855    let has_expected_errors =
856        errors.iter().any(|error| !error.is_expecting_other_error() && error.is_error());
857
858    if has_expected_errors {
859        errors.retain(|error| !error.is_expecting_other_error());
860        errors
861    } else {
862        dedup_expecting_other_errors(errors)
863    }
864}
865
866/// The same `ExpectingOtherError` can be emitted multiple times.
867/// This function removes duplicates so we don't see the same error
868/// in the output repeatedly.
869fn dedup_expecting_other_errors(mut errors: Vec<CompilationError>) -> Vec<CompilationError> {
870    // Using a HashSet of the inner error, because CompilationError does not implement Ord or Hash.
871    let mut seen: HashSet<ExpectingOtherError> = HashSet::new();
872    errors.retain(|error| match error.as_expecting_other_error() {
873        Some(e) if seen.contains(e) => false,
874        Some(e) => {
875            seen.insert(e.clone());
876            true
877        }
878        None => true,
879    });
880    errors
881}
