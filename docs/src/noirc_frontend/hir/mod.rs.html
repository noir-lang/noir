36/// Helper object which groups together several useful context objects used
37/// during name resolution. Once name resolution is finished, only the
38/// def_interner is required for type inference and monomorphization.
39pub struct Context<'file_manager, 'parsed_files> {
40    pub def_interner: NodeInterner,
41    pub crate_graph: CrateGraph,
42    pub def_maps: DefMaps,
43    pub usage_tracker: UsageTracker,
44    // In the WASM context, we take ownership of the file manager,
45    // which is why this needs to be a Cow. In all use-cases, the file manager
46    // is read-only however, once it has been passed to the Context.
47    pub file_manager: Cow<'file_manager, FileManager>,
48
49    pub debug_instrumenter: DebugInstrumenter,
50
51    /// A map of each file that already has been visited from a prior `mod foo;` declaration.
52    /// This is used to issue an error if a second `mod foo;` is declared to the same file.
53    pub visited_files: BTreeMap<FileId, Location>,
54
55    // A map of all parsed files.
56    // Same as the file manager, we take ownership of the parsed files in the WASM context.
57    // Parsed files is also read only.
58    pub parsed_files: Cow<'parsed_files, ParsedFiles>,
59
60    pub package_build_path: PathBuf,
61
62    /// Writer for comptime prints.
63    pub interpreter_output: Option<Rc<RefCell<dyn std::io::Write>>>,
64
65    /// Any unstable features required by the current package or its dependencies.
66    pub required_unstable_features: BTreeMap<CrateId, Vec<UnstableFeature>>,
67
68    /// Unresolved globals that need to be elaborated.
69    pub unresolved_globals: BTreeMap<GlobalId, UnresolvedGlobal>,
70}

221    /// Returns a list of all functions in the current crate marked with `#[fuzz]`
222    /// whose names contain the given pattern string. An empty pattern string
223    /// will return all functions marked with `#[fuzz]`.
224    pub fn get_all_fuzzing_harnesses_in_crate_matching(
225        &self,
226        crate_id: &CrateId,
227        pattern: &FunctionNameMatch,
228    ) -> Vec<(String, FuzzingHarness)> {
229        let interner = &self.def_interner;
230        let def_map = self.def_map(crate_id).expect("The local crate should be analyzed already");
231
232        def_map
233            .get_all_fuzzing_harnesses(interner)
234            .filter_map(|fuzzing_harness| {
235                let fully_qualified_name =
236                    self.fully_qualified_function_name(crate_id, &fuzzing_harness.id);
237                match &pattern {
238                    FunctionNameMatch::Anything => Some((fully_qualified_name, fuzzing_harness)),
239                    FunctionNameMatch::Exact(patterns) => patterns
240                        .iter()
241                        .any(|pattern| &fully_qualified_name == pattern)
242                        .then_some((fully_qualified_name, fuzzing_harness)),
243                    FunctionNameMatch::Contains(patterns) => patterns
244                        .iter()
245                        .any(|pattern| fully_qualified_name.contains(pattern))
246                        .then_some((fully_qualified_name, fuzzing_harness)),
247                }
248            })
249            .collect()
250    }

269    /// Generics need to be resolved before elaboration to distinguish
270    /// between normal and numeric generics.
271    /// This method is expected to be used during definition collection.
272    /// Each result is returned in a list rather than returned as a single result as to allow
273    /// definition collection to provide an error for each ill-formed numeric generic.
274    pub(crate) fn resolve_generics(
275        interner: &NodeInterner,
276        generics: &UnresolvedGenerics,
277        errors: &mut Vec<CompilationError>,
278    ) -> ResolvedGenerics {
279        vecmap(generics, |generic| {
280            // Map the generic to a fresh type variable
281            let id = interner.next_type_variable_id();
282
283            let type_var_kind = generic.kind().unwrap_or_else(|err| {
284                errors.push(err.into());
285                // When there's an error, unify with any other kinds
286                Kind::Any
287            });
288            let type_var = TypeVariable::unbound(id, type_var_kind);
289            let ident = generic.ident();
290            let location = ident.location();
291
292            if let IdentOrQuotedType::Quoted(quoted_type_id, _) = ident {
293                let typ = interner.get_quoted_type(*quoted_type_id).follow_bindings();
294                if !matches!(typ, Type::NamedGeneric(..)) {
295                    errors.push(
296                        ResolverError::MacroResultInGenericsListNotAGeneric { location, typ }
297                            .into(),
298                    );
299                }
300            }
301
302            // Check for name collisions of this generic
303            let name = Rc::new(ident.to_string());
304
305            ResolvedGeneric { name, type_var, location }
306        })
307    }

342                FunctionNameMatch::Exact(patterns) => patterns
343                    .iter()

345                    .then_some((fully_qualified_name, test_function)),
346                FunctionNameMatch::Contains(patterns) => patterns
347                    .iter()
348                    .any(|pattern| fully_qualified_name.contains(pattern))
349                    .then_some((fully_qualified_name, test_function)),
350            }
351        })
352        .collect()
353}

355fn fully_qualified_function_name(
356    crate_id: CrateId,
357    id: FuncId,
358    interner: &NodeInterner,
359    def_maps: &DefMaps,
360) -> String {
361    let def_map = def_maps.get(&crate_id).expect("The local crate should be analyzed already");
362
363    let name = interner.function_name(&id);
364
365    let module_id = interner.function_module(id);
366    let module = module_id.module(def_maps);
367
368    let parent = def_map.get_module_path_with_separator(module_id.local_id, module.parent, "::");
369
370    if parent.is_empty() { name.into() } else { format!("{parent}::{name}") }
