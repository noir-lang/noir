91    /// Constructs a new `CrateDefMap`, containing only the crate's root module.
92    ///
93    /// # Arguments
94    ///
95    /// - `krate`: The [CrateId] of the crate for which this `CrateDefMap` refers to.
96    /// - `root_module`: The [ModuleData] for the root module of the crate.
97    pub fn new(krate: CrateId, root_module: ModuleData) -> CrateDefMap {
98        let mut modules = Arena::default();
99        let root = LocalModuleId::new(modules.insert(root_module));
100        CrateDefMap { krate, root, modules, extern_prelude: BTreeMap::default() }
101    }

103    /// Collect all definitions in the crate
104    pub fn collect_defs(
105        crate_id: CrateId,
106        context: &mut Context,
107        options: FrontendOptions,
108    ) -> Vec<CompilationError> {
109        // Check if this Crate has already been compiled
110        // XXX: There is probably a better alternative for this.
111        // Without this check, the compiler will panic as it does not
112        // expect the same crate to be processed twice. It would not
113        // make the implementation wrong, if the same crate was processed twice, it just makes it slow.
114        let mut errors: Vec<CompilationError> = vec![];
115        if context.def_map(&crate_id).is_some() {
116            return errors;
117        }
118
119        // First parse the root file.
120        let root_file_id = context.crate_graph[crate_id].root_file_id;
121        let (ast, parsing_errors) = context.parsed_file_results(root_file_id);
122        let ast = ast.into_sorted();
123
124        let location = Location::new(Default::default(), root_file_id);
125
126        let root_module = ModuleData::new(
127            None, // The crate root module has no name
128            None,
129            location,
130            Vec::new(),
131            ast.inner_attributes.clone(),
132            false, // is contract
133            false, // is struct
134        );
135        let def_map = CrateDefMap::new(crate_id, root_module);
136
137        // Now we want to populate the CrateDefMap using the DefCollector
138        errors.extend(DefCollector::collect_crate_and_dependencies(
139            def_map,
140            context,
141            ast,
142            root_file_id,
143            options,
144        ));
145
146        errors.extend(parsing_errors.iter().map(|e| e.clone().into()).collect::<Vec<_>>());
147
148        errors
149    }

193    /// Go through all modules in this crate, and find all functions in
194    /// each module with the #[test] attribute
195    pub fn get_all_test_functions<'a>(
196        &'a self,
197        interner: &'a NodeInterner,
198    ) -> impl Iterator<Item = TestFunction> + 'a {
199        self.modules.iter().flat_map(|(_, module)| {
200            module.value_definitions().filter_map(|id| {
201                if let Some(func_id) = id.as_function() {
202                    let has_arguments = !interner.function_meta(&func_id).parameters.is_empty();
203                    let attributes = interner.function_attributes(&func_id);
204                    match attributes.function().map(|attr| &attr.kind) {
205                        Some(FunctionAttributeKind::Test(scope)) => {
206                            let location = interner.function_meta(&func_id).name.location;
207                            let scope = scope.clone();
208                            Some(TestFunction { id: func_id, scope, location, has_arguments })
209                        }
210                        _ => None,
211                    }
212                } else {
213                    None
214                }
215            })
216        })
217    }

273    /// Find a child module's name by inspecting its parent.
274    /// Currently required as modules do not store their own names.
275    pub fn get_module_path(
276        &self,
277        child_id: LocalModuleId,
278        parent: Option<LocalModuleId>,
279    ) -> String {
280        self.get_module_path_with_separator(child_id, parent, ".")
281    }
282
283    pub fn get_module_path_with_separator(
284        &self,
285        child_id: LocalModuleId,
286        parent: Option<LocalModuleId>,
287        separator: &str,
288    ) -> String {
289        self.get_module_path_with_separator_inner(child_id.0, parent, separator)
290    }
291
292    fn get_module_path_with_separator_inner(
293        &self,
294        child_id: Index,
295        parent: Option<LocalModuleId>,
296        separator: &str,
297    ) -> String {
298        if let Some(id) = parent {
299            let parent = &self.modules[id.0];
300            let name = parent
301                .children
302                .iter()
303                .find(|(_, id)| id.0 == child_id)
304                .map(|(name, _)| name.as_str())
305                .expect("Child module was not a child of the given parent module");
306
307            let parent_name =
308                self.get_module_path_with_separator_inner(id.0, parent.parent, separator);
309            if parent_name.is_empty() {
310                name.to_string()
311            } else {
312                format!("{parent_name}{separator}{name}")
313            }
314        } else {
315            String::new()
316        }
317    }

319    /// Return a topological ordering of each module such that any child modules
320    /// are before their parent modules. Sibling modules will respect the ordering
321    /// declared from their parent module (the `mod foo; mod bar;` declarations).
322    pub fn get_module_topological_order(&self) -> HashMap<LocalModuleId, usize> {
323        let mut ordering = HashMap::default();
324        self.topologically_sort_modules(self.root, &mut 0, &mut ordering);
325        ordering
326    }
327
328    fn topologically_sort_modules(
329        &self,
330        current: LocalModuleId,
331        index: &mut usize,
332        ordering: &mut HashMap<LocalModuleId, usize>,
333    ) {
334        for child in &self.modules[current.0].child_declaration_order {
335            self.topologically_sort_modules(*child, index, ordering);
336        }
337
338        ordering.insert(current, *index);
339        *index += 1;
340    }

343pub fn fully_qualified_module_path(
344    def_maps: &DefMaps,
345    crate_graph: &CrateGraph,
346    crate_id: &CrateId,
347    module_id: ModuleId,
348) -> String {
349    let child_id = module_id.local_id;
350
351    let def_map =
352        def_maps.get(&module_id.krate).expect("The local crate should be analyzed already");
353
354    let module = &def_map.modules()[module_id.local_id.0];
355
356    let module_path = def_map.get_module_path_with_separator(child_id, module.parent, "::");
357
358    if &module_id.krate == crate_id {
359        module_path
360    } else {
361        let crates = crate_graph
362            .find_dependencies(crate_id, &module_id.krate)
363            .expect("The module was supposed to be defined in a dependency");
364        crates.join("::") + "::" + &module_path
365    }
366}

377    pub location: Location,

381impl TestFunction {
382    /// Returns true if the test function has been specified to fail
383    /// This is done by annotating the function with `#[test(should_fail)]`
384    /// or `#[test(should_fail_with = "reason")]`
385    pub fn should_fail(&self) -> bool {
386        match self.scope {
387            TestScope::ShouldFailWith { .. } => true,
388            TestScope::OnlyFailWith { .. } | TestScope::None => false,
389        }
390    }
391
392    /// Returns the reason for the test function to fail if specified
393    /// by the user.
394    pub fn failure_reason(&self) -> Option<&str> {
395        match &self.scope {
396            TestScope::None => None,
397            TestScope::ShouldFailWith { reason } => reason.as_deref(),
398            TestScope::OnlyFailWith { reason } => Some(reason.as_str()),
399        }
400    }
401}

410    /// Returns true if the fuzzing harness has been specified to fail only under specific reason
411    /// This is done by annotating the function with
412    /// `#[fuzz(only_fail_with = "reason")]`
413    pub fn only_fail_enabled(&self) -> bool {
414        match self.scope {
415            FuzzingScope::OnlyFailWith { .. } => true,
416            FuzzingScope::None => false,
417            FuzzingScope::ShouldFailWith { .. } => false,
418        }
419    }
420    /// Returns true if the fuzzing harness has been specified to fail
421    /// This is done by annotating the function with `#[fuzz(should_fail)]`
422    /// or `#[fuzz(should_fail_with = "reason")]`
423    pub fn should_fail_enabled(&self) -> bool {
424        match self.scope {
425            FuzzingScope::OnlyFailWith { .. } => false,
426            FuzzingScope::None => false,
427            FuzzingScope::ShouldFailWith { .. } => true,
428        }
429    }
430
431    /// Returns the reason for the fuzzing harness to fail if specified
432    /// by the user.
433    pub fn failure_reason(&self) -> Option<String> {
434        match &self.scope {
435            FuzzingScope::None => None,
436            FuzzingScope::OnlyFailWith { reason } => Some(reason.clone()),
437            FuzzingScope::ShouldFailWith { reason } => reason.clone(),
438        }
439    }
