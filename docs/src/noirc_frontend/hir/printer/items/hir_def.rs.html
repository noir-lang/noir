26    fn dereference_hir_expression_id(&self, expr_id: ExprId) -> ExprId {
27        let hir_expr = self.interner.expression(&expr_id);
28        let HirExpression::Prefix(prefix) = &hir_expr else {
29            return expr_id;
30        };
31
32        match prefix.operator {
33            UnaryOp::Reference { .. } | UnaryOp::Dereference { implicitly_added: true } => {
34                prefix.rhs
35            }
36            UnaryOp::Minus | UnaryOp::Not | UnaryOp::Dereference { implicitly_added: false } => {
37                expr_id
38            }
39        }
40    }

42    fn show_hir_expression_id_maybe_inside_parens(&mut self, expr_id: ExprId) {
43        let hir_expr = self.interner.expression(&expr_id);
44        let parens = hir_expression_needs_parentheses(&hir_expr);
45        if parens {

66    pub(crate) fn show_hir_expression(&mut self, hir_expr: HirExpression, expr_id: ExprId) {
67        match hir_expr {
68            HirExpression::Ident(hir_ident, generics) => {
69                self.show_hir_ident(hir_ident, Some(expr_id));
70                if let Some(generics) = generics {
71                    let use_colons = true;
72                    self.show_generic_types(&generics, use_colons);
73                }
74            }
75            HirExpression::Literal(hir_literal) => {
76                self.show_hir_literal(hir_literal, expr_id);
77            }
78            HirExpression::Block(hir_block_expression) => {
79                self.show_hir_block_expression(hir_block_expression);
80            }
81            HirExpression::Prefix(hir_prefix_expression) => match hir_prefix_expression.operator {
82                UnaryOp::Minus => {
83                    self.push('-');
84                    self.show_hir_expression_id_maybe_inside_parens(hir_prefix_expression.rhs);
85                }
86                UnaryOp::Not => {
87                    self.push('!');
88                    self.show_hir_expression_id_maybe_inside_parens(hir_prefix_expression.rhs);
89                }
90                UnaryOp::Reference { mutable } => {
91                    if mutable {
92                        self.push_str("&mut ");
93                    } else {
94                        self.push_str("&");
95                    }
96                    self.show_hir_expression_id(hir_prefix_expression.rhs);
97                }
98                UnaryOp::Dereference { implicitly_added } => {
99                    if implicitly_added {
100                        self.show_hir_expression_id(hir_prefix_expression.rhs);
101                    } else {
102                        self.push('*');
103                        self.show_hir_expression_id_maybe_inside_parens(hir_prefix_expression.rhs);
104                    }
105                }
106            },
107            HirExpression::Infix(hir_infix_expression) => {
108                self.show_hir_expression_id_maybe_inside_parens(hir_infix_expression.lhs);
109                self.push(' ');
110                self.push_str(&hir_infix_expression.operator.kind.to_string());
111                self.push(' ');
112                self.show_hir_expression_id_maybe_inside_parens(hir_infix_expression.rhs);
113            }
114            HirExpression::Index(hir_index_expression) => {
115                self.show_hir_expression_id_maybe_inside_parens(hir_index_expression.collection);
116                self.push('[');
117                self.show_hir_expression_id(hir_index_expression.index);
118                self.push(']');
119            }
120            HirExpression::Constructor(hir_constructor_expression) => {
121                // let data_type = hir_constructor_expression.r#type.borrow();
122                let typ = Type::DataType(
123                    hir_constructor_expression.r#type.clone(),
124                    hir_constructor_expression.struct_generics.clone(),
125                );
126                if self.self_type.as_ref() == Some(&typ) {
127                    self.push_str("Self");
128                } else {
129                    let data_type = hir_constructor_expression.r#type.borrow();
130
131                    let use_import = true;
132                    self.show_reference_to_module_def_id(
133                        ModuleDefId::TypeId(data_type.id),
134                        data_type.visibility,
135                        use_import,
136                    );
137
138                    let use_colons = true;
139                    self.show_generic_types(
140                        &hir_constructor_expression.struct_generics,
141                        use_colons,
142                    );
143                }
144
145                self.push_str(" { ");
146                self.show_separated_by_comma(
147                    &hir_constructor_expression.fields,
148                    |this, (name, value)| {
149                        this.push_str(&name.to_string());
150                        this.push_str(": ");
151                        this.show_hir_expression_id(*value);
152                    },
153                );
154                self.push('}');
155            }
156            HirExpression::EnumConstructor(constructor) => {
157                let data_type = constructor.r#type.borrow();
158                let use_import = true;
159                self.show_reference_to_module_def_id(
160                    ModuleDefId::TypeId(data_type.id),
161                    data_type.visibility,
162                    use_import,
163                );
164
165                let variant = data_type.variant_at(constructor.variant_index);
166                self.push_str("::");
167                self.push_str(&variant.name.to_string());
168                if variant.is_function {
169                    self.push('(');
170                    self.show_hir_expression_ids_separated_by_comma(&constructor.arguments);
171                    self.push(')');
172                }
173            }
174            HirExpression::MemberAccess(hir_member_access) => {
175                let lhs_exp = self.interner.expression(&hir_member_access.lhs);
176
177                if let HirExpression::Prefix(HirPrefixExpression {
178                    operator: UnaryOp::Dereference { implicitly_added: false },
179                    ..
180                }) = lhs_exp
181                {
182                    // In general we don't need parentheses around dereferences, but here we do
183                    self.push('(');
184                    self.show_hir_expression(lhs_exp, hir_member_access.lhs);
185                    self.push(')');
186                } else {
187                    self.show_hir_expression_id_maybe_inside_parens(hir_member_access.lhs);
188                }
189                self.push('.');
190                self.push_str(&hir_member_access.rhs.to_string());
191            }
192            HirExpression::Call(hir_call_expression) => {
193                if self.try_show_hir_call_as_method(&hir_call_expression) {
194                    return;
195                }
196
197                let func = self.interner.expression(&hir_call_expression.func);
198
199                // Special case: a call on a member access must have parentheses around it
200                if matches!(func, HirExpression::MemberAccess(..)) {
201                    self.push('(');
202                    self.show_hir_expression_id(hir_call_expression.func);
203                    self.push(')');
204                } else {
205                    self.show_hir_expression_id_maybe_inside_parens(hir_call_expression.func);
206                }
207
208                if hir_call_expression.is_macro_call {
209                    self.push('!');
210                }
211                self.push('(');
212                self.show_hir_expression_ids_separated_by_comma(&hir_call_expression.arguments);
213                self.push(')');
214            }
215            HirExpression::Constrain(hir_constrain_expression) => {
216                self.push_str("assert(");
217                self.show_hir_expression_id(hir_constrain_expression.0);
218                if let Some(message_id) = hir_constrain_expression.2 {
219                    self.push_str(", ");
220                    self.show_hir_expression_id(message_id);
221                }
222                self.push(')');
223            }
224            HirExpression::Cast(hir_cast_expression) => {
225                self.show_hir_expression_id_maybe_inside_parens(hir_cast_expression.lhs);
226                self.push_str(" as ");
227                self.show_type(&hir_cast_expression.r#type);
228            }
229            HirExpression::If(hir_if_expression) => {
230                self.push_str("if ");
231                self.show_hir_expression_id(hir_if_expression.condition);
232                self.push(' ');
233                self.show_hir_expression_id(hir_if_expression.consequence);
234                if let Some(alternative) = hir_if_expression.alternative {
235                    self.push_str(" else ");
236                    self.show_hir_expression_id(alternative);
237                }
238            }
239            HirExpression::Match(hir_match) => self.show_hir_match(hir_match),
240            HirExpression::Tuple(expr_ids) => {
241                let len = expr_ids.len();
242                self.push('(');
243                self.show_hir_expression_ids_separated_by_comma(&expr_ids);
244                if len == 1 {
245                    self.push(',');
246                }
247                self.push(')');
248            }
249            HirExpression::Lambda(hir_lambda) => self.show_hir_lambda(hir_lambda),
250            HirExpression::Quote(tokens) => {
251                self.show_quoted(&tokens.0);
252            }
253            HirExpression::Unsafe(hir_block_expression) => {
254                // The safety comment was already outputted for the enclosing statement
255                self.push_str("unsafe ");
256                self.show_hir_block_expression(hir_block_expression);
257            }
258            HirExpression::Error => unreachable!("error nodes should not happen"),
259            HirExpression::Unquote(_) => unreachable!("unquote should not happen"),
260        }
261    }

263    pub(crate) fn show_hir_lambda(&mut self, hir_lambda: HirLambda) {
264        if hir_lambda.unconstrained {
265            self.push_str("unconstrained ");
266        }
267        self.push('|');
268        self.show_separated_by_comma(&hir_lambda.parameters, |this, (parameter, typ)| {
269            this.show_hir_pattern(parameter.clone());
270            this.push_str(": ");
271            this.show_type(typ);
272        });
273        self.push_str("| ");
274        if hir_lambda.return_type != Type::Unit {
275            self.push_str("-> ");
276            self.show_type(&hir_lambda.return_type);
277            self.push_str(" ");
278        }
279        self.show_hir_expression_id_maybe_inside_curly_braces(hir_lambda.body);
280    }

282    fn show_hir_match(&mut self, hir_match: HirMatch) {
283        match hir_match {
284            HirMatch::Success(expr_id) => self.show_hir_expression_id(expr_id),
285            HirMatch::Failure { .. } => {
286                unreachable!("At this point code should not have errors")
287            }
288            HirMatch::Guard { cond, body, otherwise } => {
289                self.push_str("if ");
290                self.show_hir_expression_id(cond);
291                self.push(' ');
292                self.show_hir_expression_id(body);
293                self.push_str(" else ");
294                self.show_hir_match(*otherwise);
295            }
296            HirMatch::Switch(variable, cases, default) => {
297                self.push_str("match ");
298                self.show_definition_id(variable);
299                self.push_str(" {\n");
300                self.increase_indent();
301                for case in cases {
302                    let typ = self.interner.definition_type(variable).follow_bindings();
303                    self.write_indent();
304
305                    if !matches!(typ, Type::Tuple(..)) {
306                        self.show_constructor(case.constructor);
307                    }
308
309                    if !case.arguments.is_empty() {
310                        if let Some(fields) = get_type_fields(&typ) {
311                            self.push('{');
312                            self.show_separated_by_comma(
313                                &case.arguments.into_iter().zip(fields).collect::<Vec<_>>(),
314                                |this, (argument, (name, _, _))| {
315                                    this.push_str(name);
316                                    this.push_str(": ");
317                                    this.show_definition_id(*argument);
318                                },
319                            );
320                            self.push('}');
321                        } else {
322                            self.push('(');
323                            self.show_separated_by_comma(&case.arguments, |this, argument| {
324                                this.show_definition_id(*argument);
325                            });
326                            self.push(')');
327                        }
328                    }
329                    self.push_str(" => ");
330                    self.show_hir_match(case.body);
331                    self.push(',');
332                    self.push('\n');
333                }
334
335                if let Some(default) = default {
336                    self.write_indent();
337                    self.push_str("_ => ");
338                    self.show_hir_match(*default);
339                    self.push(',');
340                    self.push('\n');
341                }
342
343                self.decrease_indent();
344                self.write_indent();
345                self.push('}');
346            }
347        }
348    }

350    fn show_constructor(&mut self, constructor: Constructor) {
351        match constructor {
352            Constructor::True => self.push_str("true"),
353            Constructor::False => self.push_str("false"),
354            Constructor::Unit => self.push_str("()"),
355            Constructor::Int(signed_field) => self.push_str(&signed_field.to_string()),
356            Constructor::Tuple(items) => {
357                let len = items.len();
358                self.push('(');
359                self.show_types_separated_by_comma(&items);
360                if len == 1 {
361                    self.push(',');
362                }
363                self.push(')');
364            }
365            Constructor::Variant(typ, index) => {
366                self.show_type_name_as_data_type(&typ);
367
368                let Type::DataType(data_type, _) = typ.follow_bindings() else {
369                    panic!("Expected data type")
370                };
371                let data_type = data_type.borrow();
372                if data_type.is_enum() {
373                    let variant = data_type.variant_at(index);
374                    self.push_str("::");
375                    self.push_str(&variant.name.to_string());
376                }
377            }
378            Constructor::Range(from, to) => {
379                self.push_str(&from.to_string());
380                self.push_str("..");
381                self.push_str(&to.to_string());
382            }
383        }
384    }

386    fn try_show_hir_call_as_method(&mut self, hir_call_expression: &HirCallExpression) -> bool {
387        let arguments = &hir_call_expression.arguments;
388
389        // If there are no arguments this is definitely not a method call
390        if arguments.is_empty() {
391            return false;
392        }
393
394        // A method call must have `func` be a HirIdent
395        let HirExpression::Ident(hir_ident, generics) =
396            self.interner.expression(&hir_call_expression.func)
397        else {
398            return false;
399        };
400
401        // That HirIdent must be a function reference
402        let definition = self.interner.definition(hir_ident.id);
403        let DefinitionKind::Function(func_id) = definition.kind else {
404            return false;
405        };
406
407        // Special case: assumed trait method
408        if let ImplKind::TraitItem(trait_method) = &hir_ident.impl_kind {
409            let show_as_trait_as_path = if trait_method.assumed {
410                // Is this `self.foo()` where `self` is currently a trait?
411                // If so, show it as `self.foo()` instead of `Self::foo(self)`.
412                let method_on_trait_self =
413                    if let Type::NamedGeneric(NamedGeneric { name, .. }) =
414                        &trait_method.constraint.typ
415                    {
416                        name.to_string() == "Self"
417                    } else {
418                        false
419                    };
420                !method_on_trait_self
421            } else {
422                let trait_id = trait_method.constraint.trait_bound.trait_id;
423                let module_data = &self.def_maps[&self.module_id.krate][self.module_id.local_id];
424                module_data.find_trait_in_scope(trait_id).is_none()
425            };
426            if show_as_trait_as_path {
427                self.show_hir_call_as_trait_as_path(
428                    hir_call_expression,
429                    arguments,
430                    generics,
431                    func_id,
432                    trait_method,
433                );
434                return true;
435            }
436        }
437
438        // The function must have a self type
439        let func_meta = self.interner.function_meta(&func_id);
440
441        let Some(self_type) = &func_meta.self_type else {
442            return false;
443        };
444
445        // And it must have parameters
446        if func_meta.parameters.is_empty() {
447            return false;
448        }
449
450        let (first_param_patten, first_param_type, _) = &func_meta.parameters.0[0];
451
452        // The first parameter must be `self` or `_self`
453        if !self.pattern_is_self_or_underscore_self(first_param_patten) {
454            return false;
455        }
456
457        // The first parameter must unify with the self type (as-is or after removing `&mut`)
458        let first_param_type = first_param_type.follow_bindings();
459        let first_param_type =
460            if let Type::Reference(typ, ..) = first_param_type { *typ } else { first_param_type };
461
462        let mut bindings = TypeBindings::default();
463        if self_type.try_unify(&first_param_type, &mut bindings).is_err() {
464            return false;
465        }
466
467        let first_argument = self.dereference_hir_expression_id(arguments[0]);
468        self.show_hir_expression_id_maybe_inside_parens(first_argument);
469        self.push('.');
470        self.push_str(self.interner.function_name(&func_id));
471        if let Some(generics) = generics {
472            let use_colons = true;
473            self.show_generic_types(&generics, use_colons);
474        }
475        self.push('(');
476        self.show_hir_expression_ids_separated_by_comma(&arguments[1..]);
477        self.push(')');
478
479        true
480    }

482    fn show_hir_call_as_trait_as_path(
483        &mut self,
484        hir_call_expression: &HirCallExpression,
485        arguments: &[ExprId],
486        generics: Option<Vec<Type>>,
487        func_id: FuncId,
488        trait_method: &TraitItem,
489    ) {
490        let instantiation_bindings =
491            self.interner.get_instantiation_bindings(hir_call_expression.func);
492        let mut constraint = trait_method.constraint.clone();
493        constraint.apply_bindings(instantiation_bindings);
494
495        let trait_id = trait_method.constraint.trait_bound.trait_id;
496        let module_data = &self.def_maps[&self.module_id.krate][self.module_id.local_id];
497        if module_data.find_trait_in_scope(trait_id).is_none() {
498            // It can happen that the trait is not in scope, for example if this call
499            // was generated via macros using `get_trait_impl -> methods`.
500            self.push('<');
501            self.show_type(&constraint.typ);
502            self.push_str(" as ");
503            self.show_trait_bound(&constraint.trait_bound);
504            self.push('>');
505        } else {
506            self.show_type(&constraint.typ);
507        }
508        self.push_str("::");
509        self.push_str(self.interner.function_name(&func_id));
510        if let Some(generics) = generics {
511            let use_colons = true;
512            self.show_generic_types(&generics, use_colons);
513        }
514        self.push('(');
515        self.show_hir_expression_ids_separated_by_comma(arguments);
516        self.push(')');
517    }

519    fn show_hir_block_expression(&mut self, block: HirBlockExpression) {
520        self.push_str("{\n");
521        self.increase_indent();
522        let len = block.statements.len();
523        for (index, statement) in block.statements.into_iter().enumerate() {
524            self.write_indent();
525            self.show_hir_statement_id(statement);
526
527            // For some reason some statements in the middle of a block end up being `Expression`
528            // and not `Semi`, so we add a semicolon, if needed, to produce valid syntax.
529            if index != len - 1 && !self.string.ends_with(';') {
530                self.push(';');
531            }
532
533            self.push_str("\n");
534        }
535        self.decrease_indent();
536        self.write_indent();
537        self.push('}');
538    }

551    fn show_hir_statement(&mut self, statement: HirStatement) {
552        // A safety comment can be put before a statement and it applies to any `unsafe`
553        // expression inside it. Here we check if the statement has `unsafe` in it and
554        // put a safety comment right before it. When printing an `Unsafe` expression
555        // we'll never include a safety comment at that point.
556        let has_unsafe = self.statement_has_unsafe(&statement);
557        if has_unsafe {
558            self.push_str("// Safety: comment added by `nargo expand`\n");
559            self.write_indent();
560        }
561
562        match statement {
563            HirStatement::Let(hir_let_statement) => {
564                self.push_str("let ");
565                self.show_hir_pattern(hir_let_statement.pattern);
566                self.push_str(": ");
567                self.show_type(&hir_let_statement.r#type);
568                self.push_str(" = ");
569                self.show_hir_expression_id(hir_let_statement.expression);
570                self.push(';');
571            }
572            HirStatement::Assign(hir_assign_statement) => {
573                self.show_hir_lvalue(hir_assign_statement.lvalue);
574                self.push_str(" = ");
575                self.show_hir_expression_id(hir_assign_statement.expression);
576                self.push(';');
577            }
578            HirStatement::For(hir_for_statement) => {
579                self.push_str("for ");
580                self.show_hir_ident(hir_for_statement.identifier, None);
581                self.push_str(" in ");
582                self.show_hir_expression_id(hir_for_statement.start_range);
583                if hir_for_statement.inclusive {
584                    self.push_str("..=");
585                } else {
586                    self.push_str("..");
587                }
588                self.show_hir_expression_id(hir_for_statement.end_range);
589                self.push(' ');
590                self.show_hir_expression_id(hir_for_statement.block);
591            }
592            HirStatement::Loop(expr_id) => {
593                self.push_str("loop ");
594                self.show_hir_expression_id(expr_id);
595            }
596            HirStatement::While(condition, body) => {
597                self.push_str("while ");
598                self.show_hir_expression_id(condition);
599                self.push(' ');
600                self.show_hir_expression_id(body);
601            }
602            HirStatement::Break => {
603                self.push_str("break;");
604            }
605            HirStatement::Continue => {
606                self.push_str("continue;");
607            }
608            HirStatement::Expression(expr_id) => {
609                self.show_hir_expression_id(expr_id);
610            }
611            HirStatement::Semi(expr_id) => {
612                self.show_hir_expression_id(expr_id);
613                self.push(';');
614            }
615            HirStatement::Comptime(_) => unreachable!("comptime should not happen"),
616            HirStatement::Error => unreachable!("error should not happen"),
617        }
618    }

620    fn show_hir_literal(&mut self, literal: HirLiteral, expr_id: ExprId) {
621        match literal {
622            HirLiteral::Array(hir_array_literal) => {
623                self.push_str("[");
624                self.show_hir_array_literal(hir_array_literal);
625                self.push(']');
626            }
627            HirLiteral::Vector(hir_array_literal) => {
628                self.push_str("@[");
629                self.show_hir_array_literal(hir_array_literal);
630                self.push(']');
631            }
632            HirLiteral::Bool(value) => {
633                self.push_str(&value.to_string());
634            }
635            HirLiteral::Integer(signed_field) => {
636                self.push_str(&signed_field.to_string());
637                let typ = self.interner.id_type(expr_id);
638                self.push_str("_");
639                self.push_str(&typ.to_string());
640            }
641            HirLiteral::Str(string) => {
642                self.push_str(&format!("{string:?}"));
643            }
644            HirLiteral::FmtStr(fmt_str_fragments, _expr_ids, _) => {
645                self.push_str("f\"");
646                for fragment in fmt_str_fragments {
647                    match fragment {
648                        FmtStrFragment::String(string) => {
649                            let string = string
650                                .replace('\\', "\\\\")
651                                .replace('\n', "\\n")
652                                .replace('\t', "\\t")
653                                .replace('{', "{{")
654                                .replace('}', "}}");
655                            self.push_str(&string);
656                        }
657                        FmtStrFragment::Interpolation(string, _) => {
658                            self.push('{');
659                            self.push_str(&string);
660                            self.push('}');
661                        }
662                    }
663                }
664                self.push('"');
665            }
666            HirLiteral::Unit => {
667                self.push_str("()");
668            }
669        }
670    }
671
672    fn show_hir_array_literal(&mut self, array: HirArrayLiteral) {
673        match array {
674            HirArrayLiteral::Standard(expr_ids) => {
675                self.show_hir_expression_ids_separated_by_comma(&expr_ids);
676            }
677            HirArrayLiteral::Repeated { repeated_element, length } => {
678                self.show_hir_expression_id(repeated_element);
679                self.push_str("; ");
680                self.show_type(&length);
681            }
682        }
683    }

685    fn show_hir_lvalue(&mut self, lvalue: HirLValue) {
686        let lvalue = simplify_hir_lvalue(lvalue);
687        match lvalue {
688            HirLValue::Ident(hir_ident, _) => {
689                self.show_hir_ident(hir_ident, None);
690            }
691            HirLValue::MemberAccess { object, field_name, field_index: _, typ: _, location: _ } => {
692                let object = simplify_hir_lvalue(*object);
693                let object_is_dereference = matches!(object, HirLValue::Dereference { .. });
694                if object_is_dereference {
695                    self.push('(');
696                }
697                self.show_hir_lvalue(object);
698                if object_is_dereference {
699                    self.push(')');
700                }
701                self.push('.');
702                self.push_str(&field_name.to_string());
703            }
704            HirLValue::Index { array, index, typ: _, location: _ } => {
705                let array = simplify_hir_lvalue(*array);
706                self.show_hir_lvalue(array);
707                self.push('[');
708                self.show_hir_expression_id(index);
709                self.push(']');
710            }
711            HirLValue::Dereference {
712                lvalue,
713                implicitly_added: _,
714                element_type: _,
715                location: _,
716            } => {
717                self.push_str("*");
718                self.show_hir_lvalue(*lvalue);
719            }
720            HirLValue::Error { .. } => {
721                unreachable!("error nodes should not happen")
722            }
723        }
724    }

726    fn show_hir_pattern(&mut self, pattern: HirPattern) {
727        match pattern {
728            HirPattern::Identifier(hir_ident) => self.show_hir_ident(hir_ident, None),
729            HirPattern::Mutable(hir_pattern, _) => {
730                self.push_str("mut ");
731                self.show_hir_pattern(*hir_pattern);
732            }
733            HirPattern::Tuple(hir_patterns, _location) => {
734                let len = hir_patterns.len();
735                self.push('(');
736                self.show_separated_by_comma(&hir_patterns, |this, pattern| {
737                    this.show_hir_pattern(pattern.clone());
738                });
739                if len == 1 {
740                    self.push(',');
741                }
742                self.push(')');
743            }
744            HirPattern::Struct(typ, items, _location) => {
745                self.show_type_name_as_data_type(&typ);
746                self.push_str(" {\n");
747                self.increase_indent();
748                self.show_separated_by_comma(&items, |this, (name, pattern)| {
749                    this.push_str(&name.to_string());
750                    this.push_str(": ");
751                    this.show_hir_pattern(pattern.clone());
752                });
753                self.push('\n');
754                self.decrease_indent();
755                self.write_indent();
756                self.push('}');
757            }
758        }
759    }

761    fn show_definition_id(&mut self, definition_id: DefinitionId) {
762        let location = self.interner.definition(definition_id).location;
763        let ident = HirIdent::non_trait_method(definition_id, location);
764        self.show_hir_ident(ident, None);
765    }

767    fn show_hir_ident(&mut self, ident: HirIdent, expr_id: Option<ExprId>) {
768        let instantiation_bindings = if let Some(expr_id) = expr_id {
769            self.interner.try_get_instantiation_bindings(expr_id)
770        } else {
771            None
772        };
773
774        match ident.impl_kind {
775            ImplKind::NotATraitMethod => (),
776            ImplKind::TraitItem(trait_item) => {
777                let mut constraint = trait_item.constraint.clone();
778                constraint.typ = constraint.typ.follow_bindings();
779                if let Some(bindings) = instantiation_bindings {
780                    constraint.typ = constraint.typ.substitute(bindings);
781                    constraint.trait_bound.trait_generics =
782                        constraint.trait_bound.trait_generics.map(|typ| typ.substitute(bindings));
783                }
784
785                if self.trait_constraints.contains(&constraint) {
786                    self.show_type(&constraint.typ);
787                    self.push_str("::");
788                    let name = self.interner.definition_name(trait_item.definition);
789                    self.push_str(name);
790                    return;
791                } else {
792                    match &constraint.typ {
793                        Type::TypeVariable(type_var) if type_var.borrow().is_unbound() => {
794                            // Don't show this as `AsTraitPath`
795                        }
796                        _ => {
797                            self.push('<');
798                            self.show_type(&constraint.typ);
799                            self.push_str(" as ");
800                            let trait_id = constraint.trait_bound.trait_id;
801                            let trait_ = self.interner.get_trait(trait_id);
802                            self.show_reference_to_module_def_id(
803                                ModuleDefId::TraitId(trait_id),
804                                trait_.visibility,
805                                true,
806                            );
807                            self.show_trait_generics(&constraint.trait_bound.trait_generics);
808                            self.push_str(">::");
809                            let name = self.interner.definition_name(trait_item.definition);
810                            self.push_str(name);
811                            return;
812                        }
813                    }
814                }
815            }
816        }
817
818        let definition = self.interner.definition(ident.id);
819        match definition.kind {
820            DefinitionKind::Function(func_id) => {
821                let func_meta = self.interner.function_meta(&func_id);
822                let self_type = &func_meta.self_type;
823
824                if let Some(self_type) = self_type {
825                    // No need to fully-qualify the function name if its self type is the current self type
826                    if Some(self_type) == self.self_type.as_ref() {
827                        let name = self.interner.function_name(&func_id);
828                        self.push_str("Self::");
829                        self.push_str(name);
830                        return;
831                    }
832
833                    // See if we can show this as `Self::method` by substituting instantiation type bindings for self_type
834                    if let Some(instantiation_bindings) = instantiation_bindings {
835                        let self_type = self_type.substitute(instantiation_bindings);
836                        let unbound = if let Type::TypeVariable(type_var) = &self_type {
837                            type_var.borrow().is_unbound()
838                        } else {
839                            false
840                        };
841
842                        if !unbound {
843                            self.show_type_as_expression(&self_type);
844                            self.push_str("::");
845                            let name = self.interner.function_name(&func_id);
846                            self.push_str(name);
847                            return;
848                        }
849                    }
850                }
851
852                let use_import = true;
853                let visibility = self.interner.function_visibility(func_id);
854                self.show_reference_to_module_def_id(
855                    ModuleDefId::FunctionId(func_id),
856                    visibility,
857                    use_import,
858                );
859            }
860            DefinitionKind::Global(global_id) => {
861                let global_info = self.interner.get_global(global_id);
862                let typ = self.interner.definition_type(global_info.definition_id);
863
864                // Special case: the global is an enum value
865                let typ = typ.as_monotype();
866                if let Type::DataType(data_type, _generics) = typ {
867                    let data_type = data_type.borrow();
868                    if data_type.is_enum() {
869                        // The enum expression may have unbound named generics, while the ident itself has them bound.
870                        let typ = match expr_id {
871                            Some(id) => Cow::Owned(self.interner.id_type(id)),
872                            None => Cow::Borrowed(typ),
873                        };
874                        self.show_type_name_as_data_type(typ.as_ref());
875                        self.push_str("::");
876                        self.push_str(global_info.ident.as_str());
877                        return;
878                    }
879                }
880                let use_import = true;
881                self.show_reference_to_module_def_id(
882                    ModuleDefId::GlobalId(global_id),
883                    global_info.visibility,
884                    use_import,
885                );
886            }
887            DefinitionKind::Local(..)
888            | DefinitionKind::NumericGeneric(..)
889            | DefinitionKind::AssociatedConstant(..) => {
890                let name = self.interner.definition_name(ident.id);
891
892                // The compiler uses '$' for some internal identifiers.
893                // We replace them with "___" to make sure they have valid syntax, even though
894                // there's a tiny change they might collide with user code (unlikely, really).
895                //
896                // In other cases these internal names have spaces.
897                let name = name.replace(['$', ' '], "___");
898
899                self.push_str(&name);
900            }
901        }
902    }

904    fn statement_id_has_unsafe(&self, stmt_id: StmtId) -> bool {
905        let statement = self.interner.statement(&stmt_id);
906        self.statement_has_unsafe(&statement)
907    }
908
909    fn statement_has_unsafe(&self, statement: &HirStatement) -> bool {
910        match statement {
911            HirStatement::Let(hir_let_statement) => {
912                self.expression_id_has_unsafe(hir_let_statement.expression)
913            }
914            HirStatement::Assign(hir_assign_statement) => {
915                self.expression_id_has_unsafe(hir_assign_statement.expression)
916            }
917            HirStatement::For(hir_for_statement) => {
918                // We don't check the block, as the block consists of statements and we
919                // can put the safety comment on top of the ones that have unsafe
920                self.expression_id_has_unsafe(hir_for_statement.start_range)
921                    || self.expression_id_has_unsafe(hir_for_statement.end_range)
922            }
923            HirStatement::Loop(expr_id) => self.expression_id_has_unsafe(*expr_id),
924            HirStatement::While(expr_id, expr_id2) => {
925                self.expression_id_has_unsafe(*expr_id) || self.expression_id_has_unsafe(*expr_id2)
926            }
927            HirStatement::Break => false,
928            HirStatement::Continue => false,
929            HirStatement::Expression(expr_id) => self.expression_id_has_unsafe(*expr_id),
930            HirStatement::Semi(expr_id) => self.expression_id_has_unsafe(*expr_id),
931            HirStatement::Comptime(stmt_id) => self.statement_id_has_unsafe(*stmt_id),
932            HirStatement::Error => false,
933        }
934    }
935
936    fn expression_id_has_unsafe(&self, expr_id: ExprId) -> bool {
937        let hir_expr = self.interner.expression(&expr_id);
938        self.expression_has_unsafe(hir_expr)
939    }
940
941    fn expression_has_unsafe(&self, expr: HirExpression) -> bool {
942        match expr {
943            HirExpression::Ident(..) => false,
944            HirExpression::Literal(hir_literal) => match hir_literal {
945                HirLiteral::Array(hir_array_literal) | HirLiteral::Vector(hir_array_literal) => {
946                    match hir_array_literal {
947                        HirArrayLiteral::Standard(expr_ids) => {
948                            expr_ids.iter().any(|expr_id| self.expression_id_has_unsafe(*expr_id))
949                        }
950                        HirArrayLiteral::Repeated { repeated_element, length: _ } => {
951                            self.expression_id_has_unsafe(repeated_element)
952                        }
953                    }
954                }
955                HirLiteral::FmtStr(_, expr_ids, _) => {
956                    expr_ids.iter().any(|expr_id| self.expression_id_has_unsafe(*expr_id))
957                }
958                HirLiteral::Bool(_)
959                | HirLiteral::Integer(..)
960                | HirLiteral::Str(_)
961                | HirLiteral::Unit => false,
962            },
963            HirExpression::Block(_) => {
964                // A block consists of statements so if any of those have `unsafe`, those
965                // should have the safety comment, not this wrapping statement
966                false
967            }
968            HirExpression::Prefix(hir_prefix_expression) => {
969                self.expression_id_has_unsafe(hir_prefix_expression.rhs)
970            }
971            HirExpression::Infix(hir_infix_expression) => {
972                self.expression_id_has_unsafe(hir_infix_expression.lhs)
973                    || self.expression_id_has_unsafe(hir_infix_expression.rhs)
974            }
975            HirExpression::Index(hir_index_expression) => {
976                self.expression_id_has_unsafe(hir_index_expression.collection)
977                    || self.expression_id_has_unsafe(hir_index_expression.index)
978            }
979            HirExpression::Constructor(hir_constructor_expression) => hir_constructor_expression
980                .fields
981                .iter()
982                .any(|(_, expr_id)| self.expression_id_has_unsafe(*expr_id)),
983            HirExpression::EnumConstructor(hir_enum_constructor_expression) => {
984                hir_enum_constructor_expression
985                    .arguments
986                    .iter()
987                    .any(|expr_id| self.expression_id_has_unsafe(*expr_id))
988            }
989            HirExpression::MemberAccess(hir_member_access) => {
990                self.expression_id_has_unsafe(hir_member_access.lhs)
991            }
992            HirExpression::Call(hir_call_expression) => {
993                self.expression_id_has_unsafe(hir_call_expression.func)
994                    || hir_call_expression
995                        .arguments
996                        .iter()
997                        .any(|expr_id| self.expression_id_has_unsafe(*expr_id))
998            }
999            HirExpression::Constrain(hir_constrain_expression) => {
1000                self.expression_id_has_unsafe(hir_constrain_expression.0)
1001                    || hir_constrain_expression
1002                        .2
1003                        .is_some_and(|expr_id| self.expression_id_has_unsafe(expr_id))
1004            }
1005            HirExpression::Cast(hir_cast_expression) => {
1006                self.expression_id_has_unsafe(hir_cast_expression.lhs)
1007            }
1008            HirExpression::If(hir_if_expression) => {
1009                self.expression_id_has_unsafe(hir_if_expression.condition)
1010                    || self.expression_id_has_unsafe(hir_if_expression.consequence)
1011                    || hir_if_expression
1012                        .alternative
1013                        .is_some_and(|expr_id| self.expression_id_has_unsafe(expr_id))
1014            }
1015            HirExpression::Match(hir_match) => self.hir_match_has_unsafe(&hir_match),
1016            HirExpression::Tuple(expr_ids) => {
1017                expr_ids.iter().any(|expr_id| self.expression_id_has_unsafe(*expr_id))
1018            }
1019            HirExpression::Lambda(hir_lambda) => self.expression_id_has_unsafe(hir_lambda.body),
1020            HirExpression::Quote(..) | HirExpression::Unquote(..) => false,
1021            HirExpression::Unsafe(..) => true,
1022            HirExpression::Error => false,
1023        }
1024    }
1025
1026    fn hir_match_has_unsafe(&self, hir_match: &HirMatch) -> bool {
1027        match hir_match {
1028            HirMatch::Success(expr_id) => self.expression_id_has_unsafe(*expr_id),
1029            HirMatch::Failure { .. } => false,
1030            HirMatch::Guard { cond, body, otherwise } => {
1031                self.expression_id_has_unsafe(*cond)
1032                    || self.expression_id_has_unsafe(*body)
1033                    || self.hir_match_has_unsafe(otherwise)
1034            }
1035            HirMatch::Switch(_, cases, hir_match) => {
1036                cases.iter().any(|case| self.hir_match_has_unsafe(&case.body))
1037                    || hir_match
1038                        .as_ref()
1039                        .is_some_and(|hir_match| self.hir_match_has_unsafe(hir_match))
1040            }
1041        }
1042    }
1043}

1045fn hir_expression_needs_parentheses(hir_expr: &HirExpression) -> bool {
1046    match hir_expr {
1047        HirExpression::Infix(..) | HirExpression::Cast(..) | HirExpression::Lambda(..) => true,
1048        HirExpression::Ident(..)
1049        | HirExpression::Literal(..)
1050        | HirExpression::Block(..)
1051        | HirExpression::Prefix(..)
1052        | HirExpression::Index(..)
1053        | HirExpression::Constructor(..)
1054        | HirExpression::EnumConstructor(..)
1055        | HirExpression::MemberAccess(..)
1056        | HirExpression::Call(..)
1057        | HirExpression::Constrain(..)
1058        | HirExpression::If(..)
1059        | HirExpression::Match(..)
1060        | HirExpression::Tuple(..)
1061        | HirExpression::Quote(..)
1062        | HirExpression::Unquote(..)
1063        | HirExpression::Unsafe(..)
1064        | HirExpression::Error => false,
1065    }
1066}

1068fn get_type_fields(typ: &Type) -> Option<Vec<(String, Type, ItemVisibility)>> {
1069    match typ.follow_bindings() {
1070        Type::DataType(data_type, generics) => {
1071            let data_type = data_type.borrow();
1072            data_type.get_fields(&generics)
1073        }
1074        _ => None,
1075    }

1078// Remove any implicit dereferences from `lvalue`
1079fn simplify_hir_lvalue(lvalue: HirLValue) -> HirLValue {
1080    if let HirLValue::Dereference { lvalue, implicitly_added: true, .. } = lvalue {
1081        simplify_hir_lvalue(*lvalue)
1082    } else {
1083        lvalue
