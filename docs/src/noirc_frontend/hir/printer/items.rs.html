29pub enum Item {
30    Module(Module),
31    DataType(DataType),
32    Trait(Trait),
33    TypeAlias(TypeAliasId),
34    PrimitiveType(PrimitiveType),
35    Global(GlobalId),
36    Function(FuncId),
37    /// An inherent `impl` block declared in a module other than the one defining its type.
38    /// Such impls are emitted in their declaring module rather than next to the type, so the
39    /// reconstructed source preserves the module-private visibility of their methods.
40    Impl(Impl),
41    /// A trait `impl` block declared in a module other than the one defining its type. Emitted in
42    /// its declaring module for the same reason as [`Item::Impl`]: so the impl's method bodies can
43    /// still see the module-private items they referenced in the original source.
44    TraitImpl(TraitImpl),
45}
46
47impl Item {
48    pub fn module_def_id(&self) -> Option<ModuleDefId> {
49        Some(match self {
50            Item::Module(module) => ModuleDefId::ModuleId(module.id),
51            Item::DataType(data_type) => ModuleDefId::TypeId(data_type.id),
52            Item::Trait(trait_) => ModuleDefId::TraitId(trait_.id),
53            Item::TypeAlias(type_alias_id) => ModuleDefId::TypeAliasId(*type_alias_id),
54            Item::PrimitiveType(_) => return None,
55            Item::Global(global_id) => ModuleDefId::GlobalId(*global_id),
56            Item::Function(func_id) => ModuleDefId::FunctionId(*func_id),
57            Item::Impl(_) => return None,
58            Item::TraitImpl(_) => return None,
59        })
60    }
61}

122pub(super) struct ItemBuilder<'context> {
123    crate_id: CrateId,
124    interner: &'context NodeInterner,
125    def_maps: &'context DefMaps,
126    /// All trait implementations in the current crate.
127    trait_impls: HashSet<TraitImplId>,
128    /// Whether inherent `impl` blocks declared in a module other than the one defining their type
129    /// should be emitted in their declaring module (`true`, used by `nargo expand` to reconstruct
130    /// faithful, visibility-preserving source) rather than next to the type (`false`, used by
131    /// `nargo doc`, which groups every impl of a type under that type like rustdoc).
132    relocate_impls: bool,
133    /// Inherent and trait `impl` blocks to emit away from their type because they were declared in
134    /// a different module, keyed by their declaring module and paired with the location used to
135    /// order them among that module's other items. Always empty when `relocate_impls` is `false`.
136    /// See [`Item::Impl`] and [`Item::TraitImpl`].
137    relocated_items: BTreeMap<ModuleId, Vec<(Location, Item)>>,
138}

140impl<'context> ItemBuilder<'context> {
141    pub(super) fn new(
142        crate_id: CrateId,
143        interner: &'context NodeInterner,
144        def_maps: &'context DefMaps,
145        relocate_impls: bool,
146    ) -> Self {
147        let trait_impls = interner.get_trait_implementations_in_crate(crate_id);
148        let mut builder = Self {
149            crate_id,
150            interner,
151            def_maps,
152            trait_impls,
153            relocate_impls,
154            relocated_items: BTreeMap::new(),
155        };
156        builder.relocated_items = builder.collect_relocated_items();
157        builder
158    }

160    /// Builds the map of `impl` blocks that must be emitted away from their type because they were
161    /// declared in a different module. An impl declared in the same module as its type keeps being
162    /// emitted next to the type (see [`Self::build_data_type`] / [`Self::build_data_type_trait_impls`]).
163    /// Empty unless [`Self::relocate_impls`] is set.
164    fn collect_relocated_items(&self) -> BTreeMap<ModuleId, Vec<(Location, Item)>> {
165        let mut relocated: BTreeMap<ModuleId, Vec<(Location, Item)>> = BTreeMap::new();
166        if !self.relocate_impls {
167            return relocated;
168        }
169
170        for impl_id in self.interner.get_impls_in_crate(self.crate_id) {
171            let impl_ = self.interner.get_impl(impl_id);
172            let Type::DataType(data_type, _) = impl_.typ.follow_bindings() else {
173                continue;
174            };
175            let type_module = data_type.borrow().id.parent_module_id(self.def_maps);
176            if impl_.module_id == type_module {
177                continue;
178            }
179            if let Some(impl_) = self.build_impl(impl_id) {
180                relocated
181                    .entry(impl_.module_id)
182                    .or_default()
183                    .push((impl_.location, Item::Impl(impl_)));
184            }
185        }
186
187        for trait_impl_id in &self.trait_impls {
188            if let Some(declaring_module) = self.relocated_trait_impl_module(*trait_impl_id) {
189                let trait_impl = self.build_trait_impl(*trait_impl_id);
190                relocated
191                    .entry(declaring_module)
192                    .or_default()
193                    .push((trait_impl.location, Item::TraitImpl(trait_impl)));
194            }
195        }
196
197        for items in relocated.values_mut() {
198            items.sort_by_key(|(location, _)| *location);
199        }
200        relocated
201    }

203    pub(super) fn build_module(&mut self, module_id: ModuleId) -> Module {
204        let attributes = self.interner.try_module_attributes(module_id);
205        let name = attributes.map(|attributes| attributes.name.clone());
206        let module_data = &self.def_maps[&self.crate_id][module_id.local_id];
207        let is_contract = module_data.is_contract;
208
209        let definitions = module_data.definitions();
210
211        let mut definitions = definitions
212            .types()
213            .iter()
214            .chain(definitions.values())
215            .flat_map(|(_name, scope)| scope.values())
216            .map(|(module_def_id, visibility, _is_prelude)| {
217                let location = self.module_def_id_location(*module_def_id);
218                (*module_def_id, *visibility, location)
219            })
220            .collect::<Vec<_>>();
221
222        // Make sure definitions are sorted according to location so the output is more similar to the original code
223        definitions.sort_by_key(|(_module_def_id, _visibility, location)| *location);
224
225        // Gather all ModuleDefId's for definitions so we can exclude them when we'll list imports now
226        let definitions_module_def_ids =
227            definitions.iter().map(|(module_def_id, ..)| *module_def_id).collect::<HashSet<_>>();
228
229        let scope = module_data.scope();
230        let mut imports = scope
231            .types()
232            .iter()
233            .chain(scope.values())
234            .flat_map(|(name, scope)| scope.values().map(|value| (name.clone(), value)))
235            .filter_map(|(name, (module_def_id, visibility, is_prelude))| {
236                if !definitions_module_def_ids.contains(module_def_id) {
237                    Some(Import {
238                        name,
239                        id: *module_def_id,
240                        visibility: *visibility,
241                        is_prelude: *is_prelude,
242                    })
243                } else {
244                    None
245                }
246            })
247            .collect::<Vec<_>>();
248
249        imports.sort_by_key(|import| import.name.location());
250
251        let mut located_items = definitions
252            .into_iter()
253            .filter_map(|(module_def_id, visibility, location)| {
254                let structure = self.build_module_def_id(module_def_id)?;
255                Some((location, visibility, structure))
256            })
257            .collect::<Vec<_>>();
258
259        // Impls declared in this module but whose type lives elsewhere are emitted here, ordered by
260        // location among the module's other items. An `impl` block has no visibility keyword in
261        // Noir, so the paired visibility is an inert placeholder: relocated impls are shown via
262        // `show_impl` / `show_trait_impl`, which never consult the item-level visibility.
263        if let Some(relocated) = self.relocated_items.remove(&module_id) {
264            for (location, item) in relocated {
265                located_items.push((location, ItemVisibility::Private, item));
266            }
267            located_items.sort_by_key(|(location, _, _)| *location);
268        }
269
270        let items = located_items
271            .into_iter()
272            .map(|(_location, visibility, item)| (visibility, item))
273            .collect();
274
275        Module { id: module_id, name, is_contract, imports, items }
276    }

278    fn module_def_id_location(&self, module_def_id: ModuleDefId) -> Location {
279        // We already have logic to go from a ReferenceId to a location, so we use that here
280        let reference_id = module_def_id_to_reference_id(module_def_id);
281        self.interner.reference_location(reference_id)
282    }

284    fn build_module_def_id(&mut self, module_def_id: ModuleDefId) -> Option<Item> {
285        Some(match module_def_id {
286            ModuleDefId::ModuleId(module_id) => Item::Module(self.build_module(module_id)),
287            ModuleDefId::TypeId(type_id) => self.build_data_type(type_id),
288            ModuleDefId::TypeAliasId(type_alias_id) => Item::TypeAlias(type_alias_id),
289            ModuleDefId::TraitId(trait_id) => self.build_trait(trait_id),
290            ModuleDefId::TraitAssociatedTypeId(_) => return None,
291            ModuleDefId::GlobalId(global_id) => Item::Global(global_id),
292            ModuleDefId::FunctionId(func_id) => Item::Function(func_id),
293        })
294    }

296    fn build_data_type(&self, type_id: TypeId) -> Item {
297        let data_type = self.interner.get_type(type_id);
298
299        let mut impls = if let Some(methods) =
300            self.interner.get_type_methods(&Type::DataType(data_type.clone(), vec![]))
301        {
302            self.build_impls(methods.values())
303        } else {
304            Vec::new()
305        };
306
307        // When relocating, impls declared in a different module than the type are emitted in their
308        // own module (see [`Self::collect_relocated_impls`]); only keep the ones declared alongside
309        // the type here. Otherwise every impl of the type stays grouped under it.
310        if self.relocate_impls {
311            let type_module = type_id.parent_module_id(self.def_maps);
312            impls.retain(|impl_| impl_.module_id == type_module);
313        }
314
315        let data_type = data_type.borrow();
316        let trait_impls = self.build_data_type_trait_impls(&data_type);
317
318        Item::DataType(DataType { id: type_id, impls, trait_impls })
319    }

321    fn build_impls<'a, 'b>(&'a self, methods: impl Iterator<Item = &'b Methods>) -> Vec<Impl> {
322        // Collect the distinct inherent `impl` blocks that own these methods, so each block is
323        // reconstructed individually (preserving its generics and where clause).
324        let mut impl_ids: BTreeSet<ImplId> = BTreeSet::new();
325        for methods in methods {
326            for method in &methods.direct {
327                if let Some(impl_id) = self.interner.function_meta(&method.method).impl_id {
328                    impl_ids.insert(impl_id);
329                }
330            }
331        }
332
333        impl_ids.into_iter().filter_map(|impl_id| self.build_impl(impl_id)).collect()
334    }

336    /// Reconstructs a single inherent `impl` block. Returns `None` if the block has no methods to
337    /// show (e.g. it only holds synthesized enum-variant constructors).
338    fn build_impl(&self, impl_id: ImplId) -> Option<Impl> {
339        let impl_ = self.interner.get_impl(impl_id);
340
341        let mut methods = impl_
342            .methods
343            .iter()
344            // Don't show enum variant functions
345            .filter(|func_id| self.interner.function_meta(func_id).enum_variant_index.is_none())
346            .map(|func_id| {
347                let func_meta = self.interner.function_meta(func_id);
348                let modifiers = self.interner.function_modifiers(func_id);
349                let location = func_meta.name.location;
350                (modifiers.visibility, *func_id, location)
351            })
352            .collect::<Vec<_>>();
353
354        if methods.is_empty() {
355            return None;
356        }
357
358        methods.sort_by_key(|(_, _, location)| *location);
359
360        let methods =
361            methods.into_iter().map(|(visibility, func_id, _)| (visibility, func_id)).collect();
362
363        Some(Impl {
364            generics: impl_.generics.clone(),
365            typ: impl_.typ.clone(),
366            where_clause: impl_.where_clause.clone(),
367            methods,
368            doc_comments: impl_.doc_comments.clone(),
369            module_id: impl_.module_id,
370            location: impl_.location,
371        })
372    }

374    fn build_data_type_trait_impls(&self, data_type: &crate::DataType) -> Vec<TraitImpl> {
375        let mut trait_impls = self
376            .trait_impls
377            .iter()
378            .filter_map(|trait_impl_id| {
379                // Trait impls declared in a different module than their type are emitted in that
380                // module (see [`Self::collect_relocated_items`]), not next to the type.
381                if self.relocate_impls && self.relocated_trait_impl_module(*trait_impl_id).is_some()
382                {
383                    return None;
384                }
385                let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
386                let trait_impl = trait_impl.borrow();
387                if type_mentions_data_type(&trait_impl.typ, data_type) {
388                    Some((*trait_impl_id, trait_impl.location))
389                } else {
390                    None
391                }
392            })
393            .collect::<Vec<_>>();
394
395        self.sort_trait_impls(&mut trait_impls);
396
397        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
398    }

400    fn build_trait_impls_for_trait(&self, trait_id: TraitId) -> Vec<TraitImpl> {
401        let mut trait_impls = Vec::new();
402
403        for trait_impl_id in &self.trait_impls {
404            // Trait impls declared in a different module than their type are emitted in that
405            // module (see [`Self::collect_relocated_items`]), not under the trait.
406            if self.relocate_impls && self.relocated_trait_impl_module(*trait_impl_id).is_some() {
407                continue;
408            }
409            let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
410            let trait_impl = trait_impl.borrow();
411            if trait_impl.trait_id != trait_id {
412                continue;
413            }
414
415            trait_impls.push((*trait_impl_id, trait_impl.location));
416        }
417
418        self.sort_trait_impls(&mut trait_impls);
419
420        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
421    }
422
423    fn sort_trait_impls(&self, trait_impls: &mut [(TraitImplId, Location)]) {
424        trait_impls.sort_by_key(|(trait_impl_id, location)| {
425            let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
426            let trait_impl = trait_impl.borrow();
427            let trait_ = self.interner.get_trait(trait_impl.trait_id);
428            (*location, trait_.name.to_string())
429        });
430    }

432    fn build_trait_impl(&self, trait_impl_id: TraitImplId) -> TraitImpl {
433        let trait_impl = self.interner.get_trait_implementation(trait_impl_id);
434        let trait_impl = trait_impl.borrow();
435        let external_types = self.type_only_mention_types_outside_current_crate(&trait_impl.typ);
436
437        let mut type_var_names = BTreeSet::new();
438        let ordered_generics = self.interner.get_ordered_generics_for_impl(trait_impl_id);
439        for generic in ordered_generics {
440            gather_named_type_vars(generic, &mut type_var_names);
441        }
442        gather_named_type_vars(&trait_impl.typ, &mut type_var_names);
443
444        TraitImpl {
445            generics: type_var_names,
446            id: trait_impl_id,
447            methods: trait_impl.methods.clone(),
448            external_types,
449            location: trait_impl.location,
450        }
451    }

453    /// The module a trait `impl` block was written in, derived from a method it actually defines.
454    ///
455    /// Returns `None` when the impl defines no methods of its own — either a marker impl
456    /// (`impl Trait for T {}`) or one that only inherits trait defaults. Such impls have no body
457    /// that could reference module-private items, so they never need relocating. Inherited default
458    /// methods are skipped because they share the trait's `FuncId`, whose source module is the
459    /// trait's, not the impl's.
460    fn trait_impl_declaring_module(&self, trait_impl_id: TraitImplId) -> Option<ModuleId> {
461        let trait_impl = self.interner.get_trait_implementation(trait_impl_id);
462        let trait_impl = trait_impl.borrow();
463
464        let trait_ = self.interner.get_trait(trait_impl.trait_id);
465        let default_method_ids: HashSet<FuncId> = trait_.method_ids.values().copied().collect();
466
467        let func_id = trait_impl.methods.iter().find(|f| !default_method_ids.contains(f))?;
468        let func_meta = self.interner.function_meta(func_id);
469        Some(ModuleId { krate: func_meta.source_crate, local_id: func_meta.source_module })
470    }

472    /// If a trait impl on an in-crate data type was declared in a module other than the one
473    /// defining that type, returns the declaring module so the impl can be emitted there. This
474    /// mirrors inherent-impl relocation and keeps the impl's body able to see module-private items
475    /// it referenced in the original source.
476    fn relocated_trait_impl_module(&self, trait_impl_id: TraitImplId) -> Option<ModuleId> {
477        let declaring_module = self.trait_impl_declaring_module(trait_impl_id)?;
478        let trait_impl = self.interner.get_trait_implementation(trait_impl_id);
479        let typ = trait_impl.borrow().typ.clone();
480        let Type::DataType(data_type, _) = typ.follow_bindings() else {
481            return None;
482        };
483        let type_id = data_type.borrow().id;
484        if type_id.krate() != self.crate_id {
485            return None;
486        }
487        let type_module = type_id.parent_module_id(self.def_maps);
488        (declaring_module != type_module).then_some(declaring_module)
489    }

491    fn build_trait(&self, trait_id: TraitId) -> Item {
492        let trait_ = self.interner.get_trait(trait_id);
493
494        let mut func_ids = trait_
495            .method_ids
496            .values()
497            .map(|func_id| {
498                let location = self.interner.function_meta(func_id).location;
499                (func_id, location)
500            })
501            .collect::<Vec<_>>();
502
503        // Make sure functions are shown in the same order they were defined
504        func_ids.sort_by_key(|(_func_id, location)| *location);
505
506        let methods = func_ids.into_iter().map(|(func_id, _)| *func_id).collect();
507        let trait_impls = self.build_trait_impls_for_trait(trait_id);
508
509        Item::Trait(Trait { id: trait_id, methods, trait_impls })
510    }

512    fn type_only_mention_types_outside_current_crate(&self, typ: &Type) -> bool {
513        match typ {
514            Type::Array(typ, length) => {
515                self.type_only_mention_types_outside_current_crate(length)
516                    && self.type_only_mention_types_outside_current_crate(typ)
517            }
518            Type::Vector(typ) => self.type_only_mention_types_outside_current_crate(typ),
519            Type::FmtString(length, typ) => {
520                self.type_only_mention_types_outside_current_crate(length)
521                    && self.type_only_mention_types_outside_current_crate(typ)
522            }
523            Type::Tuple(types) => {
524                types.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
525            }
526            Type::DataType(data_type, generics) => {
527                let data_type = data_type.borrow();
528                data_type.id.krate() != self.crate_id
529                    && generics
530                        .iter()
531                        .all(|typ| self.type_only_mention_types_outside_current_crate(typ))
532            }
533            Type::Alias(_type_alias, generics) => {
534                generics.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
535            }
536            Type::TraitAsType(trait_id, _, generics) => {
537                let trait_ = self.interner.get_trait(*trait_id);
538                trait_.id.0.krate != self.crate_id
539                    && generics
540                        .ordered
541                        .iter()
542                        .all(|typ| self.type_only_mention_types_outside_current_crate(typ))
543                    && generics.named.iter().all(|named_type| {
544                        self.type_only_mention_types_outside_current_crate(&named_type.typ)
545                    })
546            }
547            Type::CheckedCast { from, to: _ } => {
548                self.type_only_mention_types_outside_current_crate(from)
549            }
550            Type::Function(args, ret, env, _) => {
551                args.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
552                    && self.type_only_mention_types_outside_current_crate(ret)
553                    && self.type_only_mention_types_outside_current_crate(env)
554            }
555            Type::Reference(typ, _) => self.type_only_mention_types_outside_current_crate(typ),
556            Type::Forall(_, typ) => self.type_only_mention_types_outside_current_crate(typ),
557            Type::InfixExpr(lhs, _, rhs, _) => {
558                self.type_only_mention_types_outside_current_crate(lhs)
559                    && self.type_only_mention_types_outside_current_crate(rhs)
560            }
561            Type::Unit
562            | Type::Bool
563            | Type::Integer(..)
564            | Type::FieldElement
565            | Type::String(_)
566            | Type::Quoted(_)
567            | Type::Constant(..)
568            | Type::TypeVariable(..)
569            | Type::NamedGeneric(..)
570            | Type::Error => true,
571        }
572    }

584        );
585        self.add_primitive_type(

587            items,

589        self.add_primitive_type(

591            items,

593        self.add_primitive_type(Type::Integer(Signedness::Signed, IntegerBitSize::Eight), items);
594        self.add_primitive_type(Type::Integer(Signedness::Signed, IntegerBitSize::Sixteen), items);
595        self.add_primitive_type(

601            items,

604        self.add_primitive_type(Type::String(Box::new(Type::Error)), items);

615
616    fn add_primitive_type(&self, typ: Type, items: &mut Vec<(ItemVisibility, Item)>) {
617        let mut impls = if let Some(methods) = self.interner.get_type_methods(&typ) {
618            self.build_impls(methods.values())
619        } else {
620            Vec::new()
621        };
622        if matches!(typ, Type::Bool | Type::Integer(..) | Type::FieldElement) {
623            // Numeric types seem to share all Field impls
624            impls.retain(|impl_| impl_.typ == typ);
625        }
626
627        let trait_impls = self.build_primitive_type_trait_impls(&typ);
628        let primitive_type = PrimitiveType { typ: typ.clone(), impls, trait_impls };
629        items.push((ItemVisibility::Public, Item::PrimitiveType(primitive_type)));
630    }
631
632    fn build_primitive_type_trait_impls(&self, primitive_type: &Type) -> Vec<TraitImpl> {
633        let mut trait_impls = self
634            .trait_impls
635            .iter()
636            .filter_map(|trait_impl_id| {
637                let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
638                let trait_impl = trait_impl.borrow();
639                if type_mentions_primitive_type(&trait_impl.typ, primitive_type) {
640                    Some((*trait_impl_id, trait_impl.location))
641                } else {
642                    None
643                }
644            })
645            .collect::<Vec<_>>();
646
647        self.sort_trait_impls(&mut trait_impls);
648
649        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
650    }
651}

653fn gather_named_type_vars(typ: &Type, type_vars: &mut BTreeSet<(String, Kind)>) {
654    match typ {
655        Type::Array(typ, length) => {
656            gather_named_type_vars(length, type_vars);
657            gather_named_type_vars(typ, type_vars);
658        }
659        Type::Vector(typ) => {
660            gather_named_type_vars(typ, type_vars);
661        }
662        Type::FmtString(length, typ) => {
663            gather_named_type_vars(length, type_vars);
664            gather_named_type_vars(typ, type_vars);
665        }
666        Type::Tuple(types) => {
667            for typ in types {
668                gather_named_type_vars(typ, type_vars);
669            }
670        }
671        Type::DataType(_, generics) | Type::Alias(_, generics) => {
672            for typ in generics {
673                gather_named_type_vars(typ, type_vars);
674            }
675        }
676        Type::TraitAsType(_, _, trait_generics) => {
677            for typ in &trait_generics.ordered {
678                gather_named_type_vars(typ, type_vars);
679            }
680            for named_type in &trait_generics.named {
681                gather_named_type_vars(&named_type.typ, type_vars);
682            }
683        }
684        Type::NamedGeneric(NamedGeneric { type_var, name, .. }) => {
685            type_vars.insert((name.to_string(), type_var.kind()));
686        }
687        Type::CheckedCast { from, to: _ } => {
688            gather_named_type_vars(from, type_vars);
689        }
690        Type::Function(args, ret, env, _) => {
691            for typ in args {
692                gather_named_type_vars(typ, type_vars);
693            }
694            gather_named_type_vars(ret, type_vars);
695            gather_named_type_vars(env, type_vars);
696        }
697        Type::Reference(typ, _) => {
698            gather_named_type_vars(typ, type_vars);
699        }
700        Type::Forall(_, typ) => {
701            gather_named_type_vars(typ, type_vars);
702        }
703        Type::InfixExpr(lhs, _, rhs, _) => {
704            gather_named_type_vars(lhs, type_vars);
705            gather_named_type_vars(rhs, type_vars);
706        }
707        Type::String(typ) => {
708            gather_named_type_vars(typ, type_vars);
709        }
710        Type::Unit
711        | Type::FieldElement
712        | Type::Integer(..)
713        | Type::Bool
714        | Type::Quoted(_)
715        | Type::Constant(..)
716        | Type::TypeVariable(_)
717        | Type::Error => (),
718    }
719}

721fn type_mentions_data_type(typ: &Type, data_type: &crate::DataType) -> bool {
722    match typ {
723        Type::Array(typ, length) => {
724            type_mentions_data_type(length, data_type) || type_mentions_data_type(typ, data_type)
725        }
726        Type::Vector(typ) => type_mentions_data_type(typ, data_type),
727        Type::FmtString(length, typ) => {
728            type_mentions_data_type(length, data_type) || type_mentions_data_type(typ, data_type)
729        }
730        Type::Tuple(types) => types.iter().any(|typ| type_mentions_data_type(typ, data_type)),
731        Type::DataType(other_data_type, generics) => {
732            let other_data_type = other_data_type.borrow();
733            data_type.id == other_data_type.id
734                || generics.iter().any(|typ| type_mentions_data_type(typ, data_type))
735        }
736        Type::Alias(_type_alias, generics) => {
737            generics.iter().any(|typ| type_mentions_data_type(typ, data_type))
738        }
739        Type::TraitAsType(_, _, generics) => {
740            generics.ordered.iter().any(|typ| type_mentions_data_type(typ, data_type))
741                || generics
742                    .named
743                    .iter()
744                    .any(|named_type| type_mentions_data_type(&named_type.typ, data_type))
745        }
746        Type::CheckedCast { from: _, to } => type_mentions_data_type(to, data_type),
747        Type::Function(args, ret, env, _) => {
748            args.iter().any(|typ| type_mentions_data_type(typ, data_type))
749                || type_mentions_data_type(ret, data_type)
750                || type_mentions_data_type(env, data_type)
751        }
752        Type::Reference(typ, _) => type_mentions_data_type(typ, data_type),
753        Type::Forall(_, typ) => type_mentions_data_type(typ, data_type),
754        Type::InfixExpr(lhs, _, rhs, _) => {
755            type_mentions_data_type(lhs, data_type) || type_mentions_data_type(rhs, data_type)
756        }
757        Type::Unit
758        | Type::Bool
759        | Type::Integer(..)
760        | Type::FieldElement
761        | Type::String(_)
762        | Type::Quoted(_)
763        | Type::Constant(..)
764        | Type::TypeVariable(..)
765        | Type::NamedGeneric(..)
766        | Type::Error => false,
767    }
768}

770fn type_mentions_primitive_type(typ: &Type, target_type: &Type) -> bool {
771    if typ == target_type {
772        return true;
773    }
774
775    match typ {
776        Type::Array(typ, length) => {
777            type_mentions_primitive_type(length, target_type)
778                || type_mentions_primitive_type(typ, target_type)
779        }
780        Type::Vector(typ) => type_mentions_primitive_type(typ, target_type),
781        Type::FmtString(length, typ) => {
782            type_mentions_primitive_type(length, target_type)
783                || type_mentions_primitive_type(typ, target_type)
784        }
785        Type::Tuple(types) => {
786            types.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
787        }
788        Type::DataType(_, generics) => {
789            generics.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
790        }
791        Type::Alias(_type_alias, generics) => {
792            generics.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
793        }
794        Type::TraitAsType(_, _, generics) => {
795            generics.ordered.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
796                || generics
797                    .named
798                    .iter()
799                    .any(|named_type| type_mentions_primitive_type(&named_type.typ, target_type))
800        }
801        Type::CheckedCast { from: _, to } => type_mentions_primitive_type(to, target_type),
802        Type::Function(args, ret, env, _) => {

804                || type_mentions_primitive_type(ret, target_type)

808        Type::Forall(_, typ) => type_mentions_primitive_type(typ, target_type),
809        Type::InfixExpr(lhs, _, rhs, _) => {
810            type_mentions_primitive_type(lhs, target_type)
811                || type_mentions_primitive_type(rhs, target_type)
812        }
813        Type::Unit
814        | Type::Bool
815        | Type::Integer(..)
816        | Type::FieldElement
817        | Type::String(_)
818        | Type::Quoted(_)
819        | Type::Constant(..)
820        | Type::TypeVariable(..)
821        | Type::NamedGeneric(..)
822        | Type::Error => false,
823    }
