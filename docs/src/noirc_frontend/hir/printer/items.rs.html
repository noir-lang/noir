30pub enum Item {
31    Module(Module),
32    DataType(DataType),
33    Trait(Trait),
34    TypeAlias(TypeAliasId),
35    PrimitiveType(PrimitiveType),
36    Global(GlobalId),
37    Function(FuncId),
38}

40impl Item {
41    pub fn module_def_id(&self) -> ModuleDefId {
42        match self {
43            Item::Module(module) => ModuleDefId::ModuleId(module.id),
44            Item::DataType(data_type) => ModuleDefId::TypeId(data_type.id),
45            Item::Trait(trait_) => ModuleDefId::TraitId(trait_.id),
46            Item::TypeAlias(type_alias_id) => ModuleDefId::TypeAliasId(*type_alias_id),
47            Item::PrimitiveType(_) => panic!("No ModuleDefId for PrimitiveType"),
48            Item::Global(global_id) => ModuleDefId::GlobalId(*global_id),
49            Item::Function(func_id) => ModuleDefId::FunctionId(*func_id),
50        }
51    }
52}

113    pub(super) fn new(
114        crate_id: CrateId,
115        interner: &'context NodeInterner,
116        def_maps: &'context DefMaps,
117    ) -> Self {

122    pub(super) fn build_module(&mut self, module_id: ModuleId) -> Module {
123        let attributes = self.interner.try_module_attributes(module_id);
124        let name = attributes.map(|attributes| attributes.name.clone());
125        let module_data = &self.def_maps[&self.crate_id][module_id.local_id];
126        let is_contract = module_data.is_contract;
127
128        let definitions = module_data.definitions();
129
130        let mut definitions = definitions
131            .types()
132            .iter()
133            .chain(definitions.values())
134            .flat_map(|(_name, scope)| scope.values())
135            .map(|(module_def_id, visibility, _is_prelude)| {
136                let location = self.module_def_id_location(*module_def_id);
137                (*module_def_id, *visibility, location)
138            })
139            .collect::<Vec<_>>();
140
141        // Make sure definitions are sorted according to location so the output is more similar to the original code
142        definitions.sort_by_key(|(_module_def_id, _visibility, location)| *location);
143
144        // Gather all ModuleDefId's for definitions so we can exclude them when we'll list imports now
145        let definitions_module_def_ids =
146            definitions.iter().map(|(module_def_id, ..)| *module_def_id).collect::<HashSet<_>>();
147
148        let scope = module_data.scope();
149        let mut imports = scope
150            .types()
151            .iter()
152            .chain(scope.values())
153            .flat_map(|(name, scope)| scope.values().map(|value| (name.clone(), value)))
154            .filter_map(|(name, (module_def_id, visibility, is_prelude))| {
155                if !definitions_module_def_ids.contains(module_def_id) {
156                    Some(Import {
157                        name,
158                        id: *module_def_id,
159                        visibility: *visibility,
160                        is_prelude: *is_prelude,
161                    })
162                } else {
163                    None
164                }
165            })
166            .collect::<Vec<_>>();
167
168        imports.sort_by_key(|import| import.name.location());
169
170        let items = definitions
171            .into_iter()
172            .filter_map(|(module_def_id, visibility, _location)| {
173                let structure = self.build_module_def_id(module_def_id)?;
174                Some((visibility, structure))
175            })
176            .collect();
177
178        Module { id: module_id, name, is_contract, imports, items }
179    }

181    fn module_def_id_location(&self, module_def_id: ModuleDefId) -> Location {
182        // We already have logic to go from a ReferenceId to a location, so we use that here
183        let reference_id = module_def_id_to_reference_id(module_def_id);
184        self.interner.reference_location(reference_id)
185    }

187    fn build_module_def_id(&mut self, module_def_id: ModuleDefId) -> Option<Item> {
188        Some(match module_def_id {
189            ModuleDefId::ModuleId(module_id) => Item::Module(self.build_module(module_id)),
190            ModuleDefId::TypeId(type_id) => self.build_data_type(type_id),
191            ModuleDefId::TypeAliasId(type_alias_id) => Item::TypeAlias(type_alias_id),
192            ModuleDefId::TraitId(trait_id) => self.build_trait(trait_id),
193            ModuleDefId::TraitAssociatedTypeId(_) => return None,
194            ModuleDefId::GlobalId(global_id) => Item::Global(global_id),
195            ModuleDefId::FunctionId(func_id) => Item::Function(func_id),
196        })
197    }

199    fn build_data_type(&self, type_id: TypeId) -> Item {
200        let data_type = self.interner.get_type(type_id);
201
202        let impls = if let Some(methods) =
203            self.interner.get_type_methods(&Type::DataType(data_type.clone(), vec![]))
204        {
205            self.build_impls(methods.values())
206        } else {
207            Vec::new()
208        };
209
210        let data_type = data_type.borrow();
211        let trait_impls = self.build_data_type_trait_impls(&data_type);
212
213        Item::DataType(DataType { id: type_id, impls, trait_impls })
214    }
215
216    fn build_impls<'a, 'b>(&'a self, methods: impl Iterator<Item = &'b Methods>) -> Vec<Impl> {
217        // Gather all impl methods
218        // First split methods by impl methods and trait impl methods
219        let mut impl_methods = Vec::new();
220
221        for methods in methods {
222            impl_methods.extend(methods.direct.clone());
223        }
224
225        // Don't show enum variant functions
226        impl_methods.retain(|method| {
227            let meta = self.interner.function_meta(&method.method);
228            meta.enum_variant_index.is_none()
229        });
230
231        // Split them by the impl type. For example here we'll group
232        // all of `Foo<i32>` methods in one bucket, all of `Foo<Field>` in another, and
233        // all of `Foo<T>` in another one.
234        #[allow(clippy::mutable_key_type)]
235        let mut impl_methods_by_type: BTreeMap<Type, Vec<ImplMethod>> = BTreeMap::new();
236        for method in impl_methods {
237            impl_methods_by_type.entry(method.typ.clone()).or_default().push(method);
238        }
239
240        impl_methods_by_type
241            .into_iter()
242            .map(|(typ, methods)| self.build_impl(typ, methods))
243            .collect()
244    }
245
246    fn build_impl(&self, typ: Type, methods: Vec<ImplMethod>) -> Impl {
247        let mut generics = BTreeSet::new();
248        gather_named_type_vars(&typ, &mut generics);
249
250        let mut methods = methods
251            .into_iter()
252            .map(|method| {
253                let func_id = method.method;
254                let func_meta = self.interner.function_meta(&func_id);
255                let modifiers = self.interner.function_modifiers(&func_id);
256                let location = func_meta.name.location;
257                (modifiers.visibility, func_id, location)
258            })
259            .collect::<Vec<_>>();
260
261        methods.sort_by_key(|(_, _, location)| *location);
262
263        let methods =
264            methods.into_iter().map(|(visibility, func_id, _)| (visibility, func_id)).collect();
265
266        Impl { generics, typ, methods }
267    }
268
269    fn build_data_type_trait_impls(&self, data_type: &crate::DataType) -> Vec<TraitImpl> {
270        let mut trait_impls = self
271            .trait_impls
272            .iter()
273            .filter_map(|trait_impl_id| {
274                let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
275                let trait_impl = trait_impl.borrow();
276                if type_mentions_data_type(&trait_impl.typ, data_type) {
277                    Some((*trait_impl_id, trait_impl.location))
278                } else {
279                    None
280                }
281            })
282            .collect::<Vec<_>>();
283
284        self.sort_trait_impls(&mut trait_impls);
285
286        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
287    }
288
289    fn build_trait_impls_for_trait(&self, trait_id: TraitId) -> Vec<TraitImpl> {
290        let mut trait_impls = Vec::new();
291
292        for trait_impl_id in &self.trait_impls {
293            let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
294            let trait_impl = trait_impl.borrow();
295            if trait_impl.trait_id != trait_id {
296                continue;
297            }
298
299            trait_impls.push((*trait_impl_id, trait_impl.location));
300        }
301
302        self.sort_trait_impls(&mut trait_impls);
303
304        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
305    }
306
307    fn sort_trait_impls(&self, trait_impls: &mut [(TraitImplId, Location)]) {
308        trait_impls.sort_by_key(|(trait_impl_id, location)| {
309            let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
310            let trait_impl = trait_impl.borrow();
311            let trait_ = self.interner.get_trait(trait_impl.trait_id);
312            (*location, trait_.name.to_string())
313        });
314    }
315
316    fn build_trait_impl(&self, trait_impl_id: TraitImplId) -> TraitImpl {
317        let trait_impl = self.interner.get_trait_implementation(trait_impl_id);
318        let trait_impl = trait_impl.borrow();
319        let external_types = self.type_only_mention_types_outside_current_crate(&trait_impl.typ);
320
321        let mut type_var_names = BTreeSet::new();
322        let ordered_generics = self.interner.get_ordered_generics_for_impl(trait_impl_id);
323        for generic in ordered_generics {
324            gather_named_type_vars(generic, &mut type_var_names);
325        }
326        gather_named_type_vars(&trait_impl.typ, &mut type_var_names);
327
328        TraitImpl {
329            generics: type_var_names,
330            id: trait_impl_id,
331            methods: trait_impl.methods.clone(),
332            external_types,
333        }
334    }
335
336    fn build_trait(&self, trait_id: TraitId) -> Item {
337        let trait_ = self.interner.get_trait(trait_id);
338
339        let mut func_ids = trait_
340            .method_ids
341            .values()
342            .map(|func_id| {
343                let location = self.interner.function_meta(func_id).location;
344                (func_id, location)
345            })
346            .collect::<Vec<_>>();
347
348        // Make sure functions are shown in the same order they were defined
349        func_ids.sort_by_key(|(_func_id, location)| *location);
350
351        let methods = func_ids.into_iter().map(|(func_id, _)| *func_id).collect();
352        let trait_impls = self.build_trait_impls_for_trait(trait_id);
353
354        Item::Trait(Trait { id: trait_id, methods, trait_impls })
355    }

357    fn type_only_mention_types_outside_current_crate(&self, typ: &Type) -> bool {
358        match typ {
359            Type::Array(length, typ) => {
360                self.type_only_mention_types_outside_current_crate(length)
361                    && self.type_only_mention_types_outside_current_crate(typ)
362            }
363            Type::Vector(typ) => self.type_only_mention_types_outside_current_crate(typ),
364            Type::FmtString(length, typ) => {
365                self.type_only_mention_types_outside_current_crate(length)
366                    && self.type_only_mention_types_outside_current_crate(typ)
367            }
368            Type::Tuple(types) => {
369                types.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
370            }
371            Type::DataType(data_type, generics) => {
372                let data_type = data_type.borrow();
373                data_type.id.krate() != self.crate_id
374                    && generics
375                        .iter()
376                        .all(|typ| self.type_only_mention_types_outside_current_crate(typ))
377            }
378            Type::Alias(_type_alias, generics) => {
379                generics.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
380            }
381            Type::TraitAsType(trait_id, _, generics) => {
382                let trait_ = self.interner.get_trait(*trait_id);
383                trait_.id.0.krate != self.crate_id
384                    && generics
385                        .ordered
386                        .iter()
387                        .all(|typ| self.type_only_mention_types_outside_current_crate(typ))
388                    && generics.named.iter().all(|named_type| {
389                        self.type_only_mention_types_outside_current_crate(&named_type.typ)
390                    })
391            }
392            Type::CheckedCast { from, to: _ } => {
393                self.type_only_mention_types_outside_current_crate(from)
394            }
395            Type::Function(args, ret, env, _) => {
396                args.iter().all(|typ| self.type_only_mention_types_outside_current_crate(typ))
397                    && self.type_only_mention_types_outside_current_crate(ret)
398                    && self.type_only_mention_types_outside_current_crate(env)
399            }
400            Type::Reference(typ, _) => self.type_only_mention_types_outside_current_crate(typ),
401            Type::Forall(_, typ) => self.type_only_mention_types_outside_current_crate(typ),
402            Type::InfixExpr(lhs, _, rhs, _) => {
403                self.type_only_mention_types_outside_current_crate(lhs)
404                    && self.type_only_mention_types_outside_current_crate(rhs)
405            }
406            Type::Unit
407            | Type::Bool
408            | Type::Integer(..)
409            | Type::FieldElement
410            | Type::String(_)
411            | Type::Quoted(_)
412            | Type::Constant(..)
413            | Type::TypeVariable(..)
414            | Type::NamedGeneric(..)
415            | Type::Error => true,
416        }
417    }

421        self.add_primitive_type(Type::Integer(Signedness::Unsigned, IntegerBitSize::One), items);
422        self.add_primitive_type(Type::Integer(Signedness::Unsigned, IntegerBitSize::Eight), items);
423        self.add_primitive_type(
424            Type::Integer(Signedness::Unsigned, IntegerBitSize::Sixteen),
425            items,
426        );
427        self.add_primitive_type(
428            Type::Integer(Signedness::Unsigned, IntegerBitSize::ThirtyTwo),
429            items,
430        );
431        self.add_primitive_type(
432            Type::Integer(Signedness::Unsigned, IntegerBitSize::SixtyFour),
433            items,
434        );
435        self.add_primitive_type(
436            Type::Integer(Signedness::Unsigned, IntegerBitSize::HundredTwentyEight),
437            items,
438        );
439        self.add_primitive_type(Type::Integer(Signedness::Signed, IntegerBitSize::Eight), items);
440        self.add_primitive_type(Type::Integer(Signedness::Signed, IntegerBitSize::Sixteen), items);
441        self.add_primitive_type(
442            Type::Integer(Signedness::Signed, IntegerBitSize::ThirtyTwo),
443            items,
444        );
445        self.add_primitive_type(
446            Type::Integer(Signedness::Signed, IntegerBitSize::SixtyFour),
447            items,
448        );
449        self.add_primitive_type(Type::FieldElement, items);
450        self.add_primitive_type(Type::String(Box::new(Type::Error)), items);
451        self.add_primitive_type(
452            Type::FmtString(Box::new(Type::Error), Box::new(Type::Error)),
453            items,
454        );
455        self.add_primitive_type(Type::Array(Box::new(Type::Error), Box::new(Type::Error)), items);
456        self.add_primitive_type(Type::Vector(Box::new(Type::Error)), items);
457        for quoted_type in QuotedType::iter() {
458            self.add_primitive_type(Type::Quoted(quoted_type), items);
459        }
460    }

462    fn add_primitive_type(&self, typ: Type, items: &mut Vec<(ItemVisibility, Item)>) {
463        let mut impls = if let Some(methods) = self.interner.get_type_methods(&typ) {
464            self.build_impls(methods.values())
465        } else {
466            Vec::new()
467        };
468        if matches!(typ, Type::Bool | Type::Integer(..) | Type::FieldElement) {
469            // Numeric types seem to share all Field impls
470            impls.retain(|impl_| impl_.typ == typ);
471        }
472
473        let trait_impls = self.build_primitive_type_trait_impls(&typ);
474        let primitive_type = PrimitiveType { typ: typ.clone(), impls, trait_impls };
475        items.push((ItemVisibility::Public, Item::PrimitiveType(primitive_type)));
476    }
477
478    fn build_primitive_type_trait_impls(&self, primitive_type: &Type) -> Vec<TraitImpl> {
479        let mut trait_impls = self
480            .trait_impls
481            .iter()
482            .filter_map(|trait_impl_id| {
483                let trait_impl = self.interner.get_trait_implementation(*trait_impl_id);
484                let trait_impl = trait_impl.borrow();
485                if type_mentions_primitive_type(&trait_impl.typ, primitive_type) {
486                    Some((*trait_impl_id, trait_impl.location))
487                } else {
488                    None
489                }
490            })
491            .collect::<Vec<_>>();
492
493        self.sort_trait_impls(&mut trait_impls);
494
495        trait_impls.into_iter().map(|(trait_impl, _)| self.build_trait_impl(trait_impl)).collect()
496    }

499fn gather_named_type_vars(typ: &Type, type_vars: &mut BTreeSet<(String, Kind)>) {
500    match typ {
501        Type::Array(length, typ) => {
502            gather_named_type_vars(length, type_vars);
503            gather_named_type_vars(typ, type_vars);
504        }
505        Type::Vector(typ) => {
506            gather_named_type_vars(typ, type_vars);
507        }
508        Type::FmtString(length, typ) => {
509            gather_named_type_vars(length, type_vars);
510            gather_named_type_vars(typ, type_vars);
511        }
512        Type::Tuple(types) => {
513            for typ in types {
514                gather_named_type_vars(typ, type_vars);
515            }
516        }
517        Type::DataType(_, generics) | Type::Alias(_, generics) => {
518            for typ in generics {
519                gather_named_type_vars(typ, type_vars);
520            }
521        }
522        Type::TraitAsType(_, _, trait_generics) => {
523            for typ in &trait_generics.ordered {
524                gather_named_type_vars(typ, type_vars);
525            }
526            for named_type in &trait_generics.named {
527                gather_named_type_vars(&named_type.typ, type_vars);
528            }
529        }
530        Type::NamedGeneric(NamedGeneric { type_var, name, .. }) => {
531            type_vars.insert((name.to_string(), type_var.kind()));
532        }
533        Type::CheckedCast { from, to: _ } => {
534            gather_named_type_vars(from, type_vars);
535        }
536        Type::Function(args, ret, env, _) => {
537            for typ in args {
538                gather_named_type_vars(typ, type_vars);
539            }
540            gather_named_type_vars(ret, type_vars);
541            gather_named_type_vars(env, type_vars);
542        }
543        Type::Reference(typ, _) => {
544            gather_named_type_vars(typ, type_vars);
545        }
546        Type::Forall(_, typ) => {
547            gather_named_type_vars(typ, type_vars);
548        }
549        Type::InfixExpr(lhs, _, rhs, _) => {
550            gather_named_type_vars(lhs, type_vars);
551            gather_named_type_vars(rhs, type_vars);
552        }
553        Type::String(typ) => {
554            gather_named_type_vars(typ, type_vars);
555        }
556        Type::Unit
557        | Type::FieldElement
558        | Type::Integer(..)
559        | Type::Bool
560        | Type::Quoted(_)
561        | Type::Constant(..)
562        | Type::TypeVariable(_)
563        | Type::Error => (),
564    }
565}

567fn type_mentions_data_type(typ: &Type, data_type: &crate::DataType) -> bool {
568    match typ {
569        Type::Array(length, typ) => {
570            type_mentions_data_type(length, data_type) || type_mentions_data_type(typ, data_type)
571        }
572        Type::Vector(typ) => type_mentions_data_type(typ, data_type),
573        Type::FmtString(length, typ) => {
574            type_mentions_data_type(length, data_type) || type_mentions_data_type(typ, data_type)
575        }
576        Type::Tuple(types) => types.iter().any(|typ| type_mentions_data_type(typ, data_type)),
577        Type::DataType(other_data_type, generics) => {
578            let other_data_type = other_data_type.borrow();
579            data_type.id == other_data_type.id
580                || generics.iter().any(|typ| type_mentions_data_type(typ, data_type))
581        }
582        Type::Alias(_type_alias, generics) => {
583            generics.iter().any(|typ| type_mentions_data_type(typ, data_type))
584        }
585        Type::TraitAsType(_, _, generics) => {
586            generics.ordered.iter().any(|typ| type_mentions_data_type(typ, data_type))
587                || generics
588                    .named
589                    .iter()
590                    .any(|named_type| type_mentions_data_type(&named_type.typ, data_type))
591        }
592        Type::CheckedCast { from: _, to } => type_mentions_data_type(to, data_type),
593        Type::Function(args, ret, env, _) => {
594            args.iter().any(|typ| type_mentions_data_type(typ, data_type))
595                || type_mentions_data_type(ret, data_type)
596                || type_mentions_data_type(env, data_type)
597        }
598        Type::Reference(typ, _) => type_mentions_data_type(typ, data_type),
599        Type::Forall(_, typ) => type_mentions_data_type(typ, data_type),
600        Type::InfixExpr(lhs, _, rhs, _) => {
601            type_mentions_data_type(lhs, data_type) || type_mentions_data_type(rhs, data_type)
602        }
603        Type::Unit
604        | Type::Bool
605        | Type::Integer(..)
606        | Type::FieldElement
607        | Type::String(_)
608        | Type::Quoted(_)
609        | Type::Constant(..)
610        | Type::TypeVariable(..)
611        | Type::NamedGeneric(..)
612        | Type::Error => false,
613    }
614}
615
616fn type_mentions_primitive_type(typ: &Type, target_type: &Type) -> bool {
617    if typ == target_type {
618        return true;
619    }
620
621    match typ {
622        Type::Array(length, typ) => {
623            type_mentions_primitive_type(length, target_type)
624                || type_mentions_primitive_type(typ, target_type)
625        }
626        Type::Vector(typ) => type_mentions_primitive_type(typ, target_type),
627        Type::FmtString(length, typ) => {
628            type_mentions_primitive_type(length, target_type)
629                || type_mentions_primitive_type(typ, target_type)
630        }
631        Type::Tuple(types) => {
632            types.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
633        }
634        Type::DataType(_, generics) => {
635            generics.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
636        }
637        Type::Alias(_type_alias, generics) => {
638            generics.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
639        }
640        Type::TraitAsType(_, _, generics) => {
641            generics.ordered.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
642                || generics
643                    .named
644                    .iter()
645                    .any(|named_type| type_mentions_primitive_type(&named_type.typ, target_type))
646        }
647        Type::CheckedCast { from: _, to } => type_mentions_primitive_type(to, target_type),
648        Type::Function(args, ret, env, _) => {
649            args.iter().any(|typ| type_mentions_primitive_type(typ, target_type))
650                || type_mentions_primitive_type(ret, target_type)
651                || type_mentions_primitive_type(env, target_type)
652        }
653        Type::Reference(typ, _) => type_mentions_primitive_type(typ, target_type),
654        Type::Forall(_, typ) => type_mentions_primitive_type(typ, target_type),
655        Type::InfixExpr(lhs, _, rhs, _) => {
656            type_mentions_primitive_type(lhs, target_type)
657                || type_mentions_primitive_type(rhs, target_type)
658        }
659        Type::Unit
660        | Type::Bool
661        | Type::Integer(..)
662        | Type::FieldElement
663        | Type::String(_)
664        | Type::Quoted(_)
665        | Type::Constant(..)
666        | Type::TypeVariable(..)
667        | Type::NamedGeneric(..)
668        | Type::Error => false,
669    }
