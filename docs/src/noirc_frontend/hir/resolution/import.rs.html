41type ImportResolutionResult = Result<ResolvedImport, PathResolutionError>;

43#[derive(Debug, Clone, PartialEq, Eq, Error)]
44pub enum PathResolutionError {
45    #[error("Could not resolve '{0}' in path")]
46    Unresolved(Ident),
47    #[error("{0} is private and not visible from the current module")]
48    Private(Ident),
49    #[error("There is no super module")]
50    NoSuper(Location),
51    #[error("turbofish (`::<_>`) not allowed on {item}")]
52    TurbofishNotAllowedOnItem { item: String, location: Location },
53    #[error("{ident} is a {kind}, not a module")]
54    NotAModule { ident: Ident, kind: &'static str },
55    #[error(
56        "trait `{trait_name}` which provides `{ident}` is implemented but not in scope, please import it"
57    )]
58    TraitMethodNotInScope { ident: Ident, trait_name: String },
59    #[error("Could not resolve '{ident}' in path")]
60    UnresolvedWithPossibleTraitsToImport { ident: Ident, traits: Vec<String> },
61    #[error("Multiple applicable items in scope")]
62    MultipleTraitsInScope { ident: Ident, traits: Vec<String> },
63}
64
65impl PathResolutionError {
66    pub fn location(&self) -> Location {
67        match self {
68            PathResolutionError::NoSuper(location)
69            | PathResolutionError::TurbofishNotAllowedOnItem { location, .. } => *location,
70            PathResolutionError::Unresolved(ident)
71            | PathResolutionError::Private(ident)
72            | PathResolutionError::NotAModule { ident, .. }
73            | PathResolutionError::TraitMethodNotInScope { ident, .. }
74            | PathResolutionError::MultipleTraitsInScope { ident, .. }
75            | PathResolutionError::UnresolvedWithPossibleTraitsToImport { ident, .. } => {
76                ident.location()
77            }
78        }
79    }
80}

96impl<'a> From<&'a PathResolutionError> for CustomDiagnostic {
97    fn from(error: &'a PathResolutionError) -> Self {
98        match &error {
99            PathResolutionError::Unresolved(ident) => {
100                CustomDiagnostic::simple_error(error.to_string(), String::new(), ident.location())
101            }
102            PathResolutionError::Private(ident) => CustomDiagnostic::simple_error(
103                error.to_string(),
104                format!("{ident} is private"),
105                ident.location(),
106            ),
107            PathResolutionError::NoSuper(location) => {
108                CustomDiagnostic::simple_error(error.to_string(), String::new(), *location)
109            }
110            PathResolutionError::TurbofishNotAllowedOnItem { item: _, location } => {
111                CustomDiagnostic::simple_error(error.to_string(), String::new(), *location)
112            }
113            PathResolutionError::NotAModule { ident, kind: _ } => {
114                CustomDiagnostic::simple_error(error.to_string(), String::new(), ident.location())
115            }
116            PathResolutionError::TraitMethodNotInScope { ident, .. } => {
117                CustomDiagnostic::simple_error(error.to_string(), String::new(), ident.location())
118            }
119            PathResolutionError::UnresolvedWithPossibleTraitsToImport { ident, traits } => {
120                let mut traits = vecmap(traits, |trait_name| format!("`{trait_name}`"));
121                traits.sort();
122                CustomDiagnostic::simple_error(
123                    error.to_string(),
124                    format!(
125                        "The following traits which provide `{ident}` are implemented but not in scope: {}",
126                        traits.join(", ")
127                    ),
128                    ident.location(),
129                )
130            }
131            PathResolutionError::MultipleTraitsInScope { ident, traits } => {
132                let mut traits = vecmap(traits, |trait_name| format!("`{trait_name}`"));
133                traits.sort();
134                CustomDiagnostic::simple_error(
135                    error.to_string(),
136                    format!(
137                        "Multiple traits which provide `{ident}` are implemented and in scope: {}",
138                        traits.join(", ")
139                    ),
140                    ident.location(),
141                )
142            }
143        }
144    }
145}

147/// Resolves a Path in a `use` statement, assuming it's located in `importing_module`.
148///
149/// If the imported name can't be found, `Err` will be returned. If it can be found, `Ok`
150/// will be returned with a potential list of errors if, for example, one of the segments
151/// is not accessible from the importing module (e.g. because it's private).
152pub fn resolve_import(
153    path: Path,
154    importing_module: ModuleId,
155    def_maps: &DefMaps,
156    usage_tracker: &mut UsageTracker,
157    references_tracker: Option<ReferencesTracker>,
158) -> ImportResolutionResult {
159    let path = path_to_typed_path(path);
160    let (path, module_id, references_tracker) =
161        resolve_path_kind(path, importing_module, def_maps, references_tracker)?;
162    let mut solver =
163        ImportSolver::new(importing_module, def_maps, usage_tracker, references_tracker);
164    solver.resolve_name_in_module(path, module_id)
165}

167fn path_to_typed_path(path: Path) -> TypedPath {
168    let segments = vecmap(path.segments, path_segment_to_typed_path_segment);
169    let kind_location = path.kind_location;
170    TypedPath { segments, kind: path.kind, location: path.location, kind_location }
171}
172
173fn path_segment_to_typed_path_segment(segment: PathSegment) -> TypedPathSegment {
174    assert!(segment.generics.is_none(), "generics should not be present in a use path segment");
175    TypedPathSegment::without_generics(segment.ident, segment.location)
176}

178/// Given a `TypedPath` and a [ModuleId] it's being used in, this function returns a `TypedPath`
179/// and a [ModuleId] where that `TypedPath` should be resolved.
180///
181/// For a [PathKind::Absolute] with a value such as `::foo::bar::baz`, the path will be turned into a
182/// [PathKind::Plain] with the first segment (the crate `foo`) removed, leaving just `bar::baz`
183/// to be resolved within `foo`. For other cases the path kind stays the same, it's just paired
184/// up with the module where it should be looked up. If the module cannot be found, and error is
185/// returned.
186///
187/// The third value in the tuple is a reference tracker that must be passed to this
188/// method, which is used in case the path kind is `dep`: the segment after `dep`
189/// will be linked to the root module of the external dependency.
190pub fn resolve_path_kind<'r>(
191    path: TypedPath,
192    importing_module: ModuleId,
193    def_maps: &DefMaps,
194    references_tracker: Option<ReferencesTracker<'r>>,
195) -> Result<(TypedPath, ModuleId, Option<ReferencesTracker<'r>>), PathResolutionError> {
196    let mut solver =
197        PathResolutionTargetResolver { importing_module, def_maps, references_tracker };
198    let (path, module_id) = solver.resolve(path)?;
199    Ok((path, module_id, solver.references_tracker))
200}
201
202/// Returns `true` if the first segment of a `TypedPath` in the `starting_module`
203/// should always be visible to the `importing_module`.
204///
205/// Assumes that we have called [resolve_path_kind] before.
206pub(crate) fn first_segment_is_always_visible(
207    path: &TypedPath,
208    importing_module: ModuleId,
209    starting_module: ModuleId,
210) -> bool {
211    match path.kind {
212        PathKind::Crate | PathKind::Super => true,
213        PathKind::Plain => importing_module == starting_module,
214        PathKind::Resolved(_) => false,
215        PathKind::Absolute => {
216            unreachable!("ICE: Absolute path kinds should have been turned into Plain.")
217        }
218    }
219}

221struct PathResolutionTargetResolver<'def_maps, 'references_tracker> {
222    importing_module: ModuleId,
223    def_maps: &'def_maps BTreeMap<CrateId, CrateDefMap>,
224    references_tracker: Option<ReferencesTracker<'references_tracker>>,
225}
226
227impl PathResolutionTargetResolver<'_, '_> {
228    /// Resolve a `TypedPath` based on its [PathKind] to the target [ModuleId].
229    fn resolve(&mut self, path: TypedPath) -> Result<(TypedPath, ModuleId), PathResolutionError> {
230        match path.kind {
231            PathKind::Crate => self.resolve_crate_path(path, self.importing_module.krate),
232            PathKind::Plain => self.resolve_plain_path(path, self.importing_module),
233            PathKind::Absolute => self.resolve_absolute_path(path),
234            PathKind::Super => self.resolve_super_path(path),
235            PathKind::Resolved(crate_id) => self.resolve_crate_path(path, crate_id),
236        }
237    }
238
239    /// Resolve a path such as `crate::foo::bar` or `$crate::foo::bar`.
240    ///
241    /// Returns a path with its kind unchanged, paired up with the importing or defining module itself as the target.
242    fn resolve_crate_path(
243        &self,
244        path: TypedPath,
245        krate: CrateId,
246    ) -> Result<(TypedPath, ModuleId), PathResolutionError> {
247        let root_module = self.def_maps[&krate].root();
248        let current_module = ModuleId { krate, local_id: root_module };
249        Ok((path, current_module))
250    }
251
252    /// Resolve a path such as `foo::bar`:
253    /// * check if `foo` module can be found in the current importing module
254    /// * if not, treat the path as if it were `::foo::bar` and look for a `foo` crate instead
255    fn resolve_plain_path(
256        &mut self,
257        path: TypedPath,
258        current_module: ModuleId,
259    ) -> Result<(TypedPath, ModuleId), PathResolutionError> {
260        // There is a possibility that the import path is empty. In that case, early return.
261        // This happens on import statements such as `use crate` or `use std`.
262        if path.segments.is_empty() {
263            return Ok((path, current_module));
264        }
265
266        let first_segment =
267            &path.segments.first().expect("ICE: could not fetch first segment").ident;
268        if get_module(self.def_maps, current_module).find_name(first_segment).is_none() {
269            // Resolve externally when first segment is unresolved
270            return self.resolve_absolute_path(path);
271        }
272
273        Ok((path, current_module))
274    }
275
276    /// Resolve a path such as `::foo:bar::baz`:
277    /// * find the `foo` crate among the dependencies of the current importing module
278    /// * remove the crate `foo` from the path, returning a plain path `bar::baz` along with the dependency module
279    fn resolve_absolute_path(
280        &mut self,
281        mut path: TypedPath,
282    ) -> Result<(TypedPath, ModuleId), PathResolutionError> {
283        // Use extern_prelude to get the dep
284        let current_def_map = &self.def_maps[&self.importing_module.krate];
285
286        // Fetch the root module from the prelude
287        let crate_name = &path.segments.first().unwrap().ident;
288        let dep_module = current_def_map
289            .extern_prelude
290            .get(crate_name.as_str())
291            .ok_or_else(|| PathResolutionError::Unresolved(crate_name.to_owned()))?;
292
293        if let Some(references_tracker) = &mut self.references_tracker {
294            let location = crate_name.location();
295            references_tracker.add_reference(ModuleDefId::ModuleId(*dep_module), location, false);
296        }
297
298        // Now the path can be solved starting from the second segment as a plain path
299        path.kind = PathKind::Plain;
300        path.segments.remove(0);
301
302        Ok((path, *dep_module))
303    }
304
305    /// Resolve a path such as `super::foo::bar`:
306    /// * get the parent of the current importing module
307    /// * return the path still with [PathKind::Super], paired up with the parent module
308    fn resolve_super_path(
309        &self,
310        path: TypedPath,
311    ) -> Result<(TypedPath, ModuleId), PathResolutionError> {
312        let Some(parent_module_id) = get_module(self.def_maps, self.importing_module).parent else {
313            return Err(PathResolutionError::NoSuper(path.kind_location));
314        };
315
316        let current_module =
317            ModuleId { krate: self.importing_module.krate, local_id: parent_module_id };
318        Ok((path, current_module))
319    }
320}

322struct ImportSolver<'def_maps, 'usage_tracker, 'references_tracker> {
323    importing_module: ModuleId,
324    def_maps: &'def_maps BTreeMap<CrateId, CrateDefMap>,
325    usage_tracker: &'usage_tracker mut UsageTracker,
326    references_tracker: Option<ReferencesTracker<'references_tracker>>,
327}
328
329impl<'def_maps, 'usage_tracker, 'references_tracker>
330    ImportSolver<'def_maps, 'usage_tracker, 'references_tracker>
331{
332    fn new(
333        importing_module: ModuleId,
334        def_maps: &'def_maps BTreeMap<CrateId, CrateDefMap>,
335        usage_tracker: &'usage_tracker mut UsageTracker,
336        references_tracker: Option<ReferencesTracker<'references_tracker>>,
337    ) -> Self {
338        Self { importing_module, def_maps, usage_tracker, references_tracker }
339    }
340
341    /// Resolves a [TypedPath] assuming it is inside `starting_module`.
342    ///
343    /// This is very similar to `Elaborator::resolve_name_in_module`.
344    fn resolve_name_in_module(
345        &mut self,
346        path: TypedPath,
347        starting_module: ModuleId,
348    ) -> ImportResolutionResult {
349        // There is a possibility that the import path is empty. In that case, early return.
350        if path.segments.is_empty() {
351            return Ok(ResolvedImport {
352                namespace: PerNs::types(starting_module.into()),
353                errors: Vec::new(),
354            });
355        }
356
357        let first_segment_is_always_visible =
358            first_segment_is_always_visible(&path, self.importing_module, starting_module);
359
360        // The current module and module ID as we resolve path segments
361        let mut current_module_id = starting_module;
362        let mut current_module = get_module(self.def_maps, starting_module);
363
364        let first_segment =
365            &path.segments.first().expect("ICE: could not fetch first segment").ident;
366        let mut current_ns = current_module.find_name(first_segment);
367        if current_ns.is_none() {
368            return Err(PathResolutionError::Unresolved(first_segment.clone()));
369        }
370
371        self.usage_tracker.mark_as_referenced(current_module_id, first_segment);
372
373        let mut errors = Vec::new();
374        for (index, (last_segment, current_segment)) in
375            path.segments.iter().zip(path.segments.iter().skip(1)).enumerate()
376        {
377            let last_ident = &last_segment.ident;
378            let current_ident = &current_segment.ident;
379
380            let (typ, visibility) = match current_ns.types {
381                None => {
382                    return Err(PathResolutionError::Unresolved(last_ident.clone()));
383                }
384                Some((typ, visibility, _)) => (typ, visibility),
385            };
386
387            self.add_reference(typ, last_segment.location, last_segment.ident.is_self_type_name());
388
389            // In the type namespace, only Mod can be used in a path.
390            current_module_id = match typ {
391                ModuleDefId::ModuleId(id) => id,
392                ModuleDefId::TypeId(id) => id.module_id(),
393                ModuleDefId::TypeAliasId(..) => {
394                    return Err(PathResolutionError::NotAModule {
395                        ident: last_segment.ident.clone(),
396                        kind: "type alias",
397                    });
398                }
399                ModuleDefId::TraitAssociatedTypeId(..) => {
400                    return Err(PathResolutionError::NotAModule {
401                        ident: last_segment.ident.clone(),
402                        kind: "associated type",
403                    });
404                }
405                ModuleDefId::TraitId(id) => id.0,
406                ModuleDefId::FunctionId(_) => panic!("functions cannot be in the type namespace"),
407                ModuleDefId::GlobalId(_) => panic!("globals cannot be in the type namespace"),
408            };
409
410            if !((first_segment_is_always_visible && index == 0)
411                || self.item_in_module_is_visible(current_module_id, visibility))
412            {
413                errors.push(PathResolutionError::Private(last_ident.clone()));
414            }
415
416            current_module = &self.def_maps[&current_module_id.krate][current_module_id.local_id];
417
418            // Check if namespace
419            let found_ns = current_module.find_name(current_ident);
420            if found_ns.is_none() {
421                return Err(PathResolutionError::Unresolved(current_ident.clone()));
422            }
423
424            self.usage_tracker.mark_as_referenced(current_module_id, current_ident);
425
426            current_ns = found_ns;
427        }
428
429        let (module_def_id, visibility, _) =
430            current_ns.values.or(current_ns.types).expect("Found empty namespace");
431
432        self.add_reference(module_def_id, path.segments.last().unwrap().ident.location(), false);
433
434        if !self.item_in_module_is_visible(current_module_id, visibility) {
435            errors.push(PathResolutionError::Private(path.last_ident()));
436        }
437
438        Ok(ResolvedImport { namespace: current_ns, errors })
439    }
440
441    fn add_reference(
442        &mut self,
443        reference_id: ModuleDefId,
444        location: Location,
445        is_self_type_name: bool,
446    ) {
447        if let Some(references_tracker) = &mut self.references_tracker {
448            references_tracker.add_reference(reference_id, location, is_self_type_name);
449        }
450    }
451
452    fn item_in_module_is_visible(&self, module: ModuleId, visibility: ItemVisibility) -> bool {
453        item_in_module_is_visible(self.def_maps, self.importing_module, module, visibility)
454    }
455}

457fn get_module(def_maps: &DefMaps, module: ModuleId) -> &ModuleData {
458    let message = "A crate should always be present for a given crate id";
459    &def_maps.get(&module.krate).expect(message)[module.local_id]
