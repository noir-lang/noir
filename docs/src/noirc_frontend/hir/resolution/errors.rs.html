331impl<'a> From<&'a ResolverError> for Diagnostic {
332    /// Only user errors can be transformed into a Diagnostic
333    /// ICEs will make the compiler panic, as they could affect the
334    /// soundness of the generated program
335    fn from(error: &'a ResolverError) -> Diagnostic {
336        match error {
337            ResolverError::DuplicateDefinition { name, first_location, second_location} => {
338                let mut diag = Diagnostic::simple_error(
339                    format!("duplicate definitions of {name} found"),
340                    "second definition found here".to_string(),
341                    *second_location,
342                );
343                diag.add_secondary("first definition found here".to_string(), *first_location);
344                diag
345            }
346            ResolverError::UnusedVariable { ident } => {
347                let mut diagnostic = Diagnostic::simple_warning(
348                    format!("unused variable {ident}"),
349                    "unused variable".to_string(),
350                    ident.location(),
351                );
352                diagnostic.unnecessary = true;
353                diagnostic
354            }
355            ResolverError::UnusedItem { ident, item} => {
356                let item_type = item.item_type();
357
358                let mut diagnostic =
359                    if let UnusedItem::Struct(..) = item {
360                        Diagnostic::simple_warning(
361                            format!("{item_type} `{ident}` is never constructed"),
362                            format!("{item_type} is never constructed"),
363                            ident.location(),
364                        )
365                    } else {
366                        Diagnostic::simple_warning(
367                            format!("unused {item_type} {ident}"),
368                            format!("unused {item_type}"),
369                            ident.location(),
370                        )
371                    };
372                diagnostic.unnecessary = true;
373                diagnostic
374            }
375            ResolverError::UnconditionalRecursion { name, location} => {
376                Diagnostic::simple_warning(
377                    format!("function `{name}` cannot return without recursing"),
378                    "function cannot return without recursing".to_string(),
379                    *location,
380                )
381            }
382            ResolverError::VariableNotDeclared { name, location } =>  {
383                if name == "_" {
384                    Diagnostic::simple_error(
385                        "in expressions, `_` can only be used on the left-hand side of an assignment".to_string(),
386                        "`_` not allowed here".to_string(),
387                        *location,
388                    )
389                } else {
390                    Diagnostic::simple_error(
391                        format!("cannot find `{name}` in this scope"),
392                        "not found in this scope".to_string(),
393                        *location,
394                    )
395                }
396            }
397            ResolverError::PathResolutionError(error) => error.into(),
398            ResolverError::Expected { location, expected, found: got } => Diagnostic::simple_error(
399                format!("expected {expected}, found {got}"),
400                String::new(),
401                *location,
402            ),
403            ResolverError::DuplicateField { field } => Diagnostic::simple_error(
404                format!("duplicate field {field}"),
405                String::new(),
406                field.location(),
407            ),
408            ResolverError::NoSuchField { field, struct_definition } => {
409                Diagnostic::simple_error(
410                    format!("no such field {field} defined in struct {struct_definition}"),
411                    String::new(),
412                    field.location(),
413                )
414            }
415            ResolverError::MissingFields { location, missing_fields, struct_definition } => {
416                let plural = if missing_fields.len() != 1 { "s" } else { "" };
417                let remaining_fields_names = match &missing_fields[..] {
418                    [field1] => field1.clone(),
419                    [field1, field2] => format!("{field1} and {field2}"),
420                    [field1, field2, field3] => format!("{field1}, {field2} and {field3}"),
421                    _ => {
422                        let len = missing_fields.len() - 3;
423                        let len_plural = if len != 1 {"s"} else {""};
424
425                        let truncated_fields = format!(" and {len} other field{len_plural}");
426                        let missing_fields = &missing_fields[0..3];
427                        format!("{}{truncated_fields}", missing_fields.join(", "))
428                    }
429                };
430
431                Diagnostic::simple_error(
432                    format!("missing field{plural} {remaining_fields_names} in struct {struct_definition}"),
433                    String::new(),
434                    *location,
435                )
436            }
437            ResolverError::UnnecessaryMut { first_mut, second_mut } => {
438                let mut error = Diagnostic::simple_error(
439                    "'mut' here is not necessary".to_owned(),
440                    "".to_owned(),
441                    *second_mut,
442                );
443                error.add_secondary(
444                    "Pattern was already made mutable from this 'mut'".to_owned(),
445                    *first_mut,
446                );
447                error
448            }
449            ResolverError::UnnecessaryPub { ident, position } => {
450                let mut diag = Diagnostic::simple_error(
451                    format!("unnecessary pub keyword on {position} for function {ident}"),
452                    format!("unnecessary pub {position}"),
453                    ident.location(),
454                );
455                diag.add_note("The `pub` keyword only has effects on arguments to the entry-point function of a program. Thus, adding it to other function parameters can be deceiving and should be removed".to_owned());
456                diag
457            }
458            ResolverError::NecessaryPub { ident } => {
459                let mut diag = Diagnostic::simple_error(
460                    format!("missing pub keyword on return type of function {ident}"),
461                    "missing pub on return type".to_string(),
462                    ident.location(),
463                );
464                diag.add_note("The `pub` keyword is mandatory for the entry-point function return type because the verifier cannot retrieve private witness and thus the function will not be able to return a 'priv' value".to_owned());
465                diag
466            }
467            ResolverError::NoSuchNumericTypeVariable { path } => Diagnostic::simple_error(
468                format!("Cannot find a global or generic type parameter named `{path}`"),
469                "Only globals or generic type parameters are allowed to be used as an array type's length".to_string(),
470                path.location,
471            ),
472            ResolverError::NonStructUsedInConstructor { typ, location } => Diagnostic::simple_error(
473                "Only struct types can be used in constructor expressions".into(),
474                format!("{typ} has no fields to construct it with"),
475                *location,
476            ),
477            ResolverError::GenericsOnSelfType { location } => Diagnostic::simple_error(
478                "Cannot apply generics to Self type".into(),
479                "Use an explicit type name or apply the generics at the start of the impl instead".into(),
480                *location,
481            ),
482            ResolverError::GenericsOnAssociatedType { location } => Diagnostic::simple_error(
483                "Generic Associated Types (GATs) are currently unsupported in Noir".into(),
484                "Cannot apply generics to an associated type".into(),
485                *location,
486            ),
487            ResolverError::ParserError(error) => error.as_ref().into(),
488            ResolverError::InvalidClosureEnvironment { location, typ } => Diagnostic::simple_error(
489                format!("{typ} is not a valid closure environment type"),
490                "Closure environment must be a tuple or unit type".to_string(), *location
491            ),
492            ResolverError::NestedVectors { location } => Diagnostic::simple_error(
493                "Nested vectors, i.e. vectors within an array or vector, are not supported".into(),
494                "Try to use a constant sized array or BoundedVec instead".into(),
495                *location,
496            ),
497            ResolverError::AbiAttributeOutsideContract { location, usage_location } => {
498                let mut diagnostic = Diagnostic::simple_error(
499                    "#[abi(tag)] attributes can only be used in contracts".to_string(),
500                    "misplaced #[abi(tag)] attribute".to_string(),
501                    *location,
502                );
503                if let Some(usage_location) = usage_location {
504                    diagnostic.add_secondary(
505                        "the type is used outside of a contract".to_string(),
506                        *usage_location,
507                    );
508                }
509                diagnostic
510            }
511            ResolverError::LowLevelFunctionOutsideOfStdlib { location } => Diagnostic::simple_error(
512                "Definition of low-level function outside of standard library".into(),
513                "Usage of the `#[foreign]` or `#[builtin]` function attributes are not allowed outside of the Noir standard library".into(),
514                *location,
515            ),
516            ResolverError::OracleNameClashesWithStdlib { location } => {
517                Diagnostic::simple_error(
518                    error.to_string(),
519                    "Naming an `#[oracle]` function the same as one in the Noir standard library could lead to unexpected behavior".into(),
520                    *location,
521                )
