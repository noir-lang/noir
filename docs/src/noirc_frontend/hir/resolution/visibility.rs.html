7/// Returns true if an item with the given visibility in the target module
8/// is visible from the current module. For example:
9/// ```text
10/// mod foo {
11///     ^^^ <-- target module
12///   pub(crate) fn bar() {}
13///   ^^^^^^^^^^ <- visibility
14/// }
15/// ```
16pub fn item_in_module_is_visible(
17    def_maps: &DefMaps,
18    current_module: ModuleId,
19    target_module: ModuleId,
20    visibility: ItemVisibility,
21) -> bool {
22    // Note that if the target module is in a different crate from the current module then we will either
23    // return true as the target module is public or return false as it is private without looking at the `CrateDefMap` in either case.
24    let same_crate = target_module.krate == current_module.krate;
25
26    match visibility {
27        ItemVisibility::Public => true,
28        ItemVisibility::PublicCrate => same_crate,
29        ItemVisibility::Private => {
30            if !same_crate {
31                return false;
32            }
33
34            let target_crate_def_map = &def_maps[&target_module.krate];
35            module_is_descendant_of_target(
36                target_crate_def_map,
37                current_module.local_id,
38                target_module.local_id,
39            ) || module_is_parent_of_struct_module(
40                target_crate_def_map,
41                current_module.local_id,
42                target_module.local_id,
43            )
44        }
45    }
46}

48/// Returns true if `module` and every one of its ancestor modules (up to, but not including,
49/// the crate root) are visible from `current_module`.
50///
51/// This mimics the behavior of path resolution, which checks the visibility of each module
52/// segment as it walks a path.
53pub fn module_is_visible(
54    module: ModuleId,
55    current_module: ModuleId,
56    interner: &NodeInterner,
57    def_maps: &DefMaps,
58) -> bool {
59    // Each module's visibility is declared in its parent, so we check it against the parent just
60    // as path resolution checks a segment's visibility against the module it lives in. The crate
61    // root has no parent and is always reachable (it is entered via the extern prelude).
62    let mut module = module;
63    while let Some(parent) = module.parent(def_maps) {
64        let visibility = module_def_id_visibility(ModuleDefId::ModuleId(module), interner);
65        if !item_in_module_is_visible(def_maps, current_module, parent, visibility) {
66            return false;
67        }
68        module = parent;
69    }
70    true
71}

73/// Returns true if `current` is a (potentially nested) child module of `target`.
74/// This is also true if `current == target`.
75fn module_is_descendant_of_target(
76    def_map: &CrateDefMap,
77    current: LocalModuleId,
78    target: LocalModuleId,
79) -> bool {
80    if current == target {
81        return true;
82    }
83
84    def_map[current]
85        .parent
86        .is_some_and(|parent| module_is_descendant_of_target(def_map, parent, target))
87}

89/// Returns true if `target` is a struct and its parent is `current`.
90fn module_is_parent_of_struct_module(
91    def_map: &CrateDefMap,
92    current: LocalModuleId,
93    target: LocalModuleId,
94) -> bool {
95    let module_data = &def_map[target];
96    module_data.is_type && module_data.parent == Some(current)
97}

119/// Returns true if the trait that `func_id` belongs to (as a trait declaration or trait impl)
120/// is visible from `current_module`. Returns true if `func_id` is not a trait method, or if
121/// its function meta has not been registered yet (in which case there's nothing to check).
122pub fn trait_visibility_for_method_is_satisfied(
123    func_id: FuncId,
124    current_module: ModuleId,
125    interner: &NodeInterner,
126    def_maps: &DefMaps,
127) -> bool {
128    let Some(func_meta) = interner.try_function_meta(&func_id) else { return true };
129    let visibility = interner.function_modifiers(&func_id).visibility;
130
131    if let Some(trait_id) = func_meta.trait_id {
132        return trait_member_is_visible(trait_id, visibility, current_module, def_maps);
133    }
134    if let Some(trait_impl_id) = func_meta.trait_impl {
135        let trait_id = interner.get_trait_implementation(trait_impl_id).borrow().trait_id;
136        return trait_member_is_visible(trait_id, visibility, current_module, def_maps);
137    }
138    true
139}

141/// Returns whether a struct or trait member with the given visibility is visible from `current_module_id`.
142fn type_member_is_visible(
143    type_module_id: ModuleId,
144    visibility: ItemVisibility,
145    current_module_id: ModuleId,
146    def_maps: &DefMaps,
147) -> bool {
148    match visibility {
149        ItemVisibility::Public => true,
150        ItemVisibility::PublicCrate => {
151            let type_parent_module_id =
152                type_module_id.parent(def_maps).expect("Expected parent module to exist");
153            type_parent_module_id.krate == current_module_id.krate
154        }
155        ItemVisibility::Private => {
156            let type_parent_module_id =
157                type_module_id.parent(def_maps).expect("Expected parent module to exist");
158            if type_parent_module_id.krate != current_module_id.krate {
159                return false;
160            }
161
162            if type_parent_module_id.local_id == current_module_id.local_id {
163                return true;
164            }
165
166            let def_map = &def_maps[&current_module_id.krate];
167            module_is_descendant_of_target(
168                def_map,
169                current_module_id.local_id,
170                type_parent_module_id.local_id,
171            )
172        }
173    }
174}

176/// Returns whether a method call `func_id` on an object of type `object_type` is visible from
177/// `current_module`.
178///
179/// If there's a self type at the current location it must be passed as `self_type`. This is
180/// used for the case of calling, inside a generic trait impl, a private method on the same
181/// type as `self_type` regardless of its generic arguments (in this case the call is allowed).
182pub fn method_call_is_visible(
183    self_type: Option<&Type>,
184    object_type: &Type,
185    func_id: FuncId,
186    current_module: ModuleId,
187    interner: &NodeInterner,
188    def_maps: &DefMaps,
189) -> bool {
190    let modifiers = interner.function_modifiers(&func_id);
191    match modifiers.visibility {
192        ItemVisibility::Public => true,
193        ItemVisibility::PublicCrate | ItemVisibility::Private => {
194            let func_meta = interner.function_meta(&func_id);
195
196            if func_meta.trait_id.is_some() || func_meta.trait_impl.is_some() {
197                return trait_visibility_for_method_is_satisfied(
198                    func_id,
199                    current_module,
200                    interner,
201                    def_maps,
202                );
203            }
204
205            // A private method defined on `Foo<i32>` should be visible when calling
206            // it from an impl on `Foo<i64>`, even though the generics are different.
207            if let Some(self_type) = self_type
208                && is_same_type_regardless_generics(self_type, object_type)
209            {
210                if modifiers.visibility.is_private() {
211                    // Only allow accessing private methods of a type if we are under the same
212                    // module where the type was defined. OTOH if we are in an `impl Foo<i32>`
213                    // block in a different module, extending the type with new methods, then
214                    // we should only access public parts defined in other modules, or private
215                    // ones defined in the same extension.
216                    let def_map = &def_maps[&current_module.krate];
217                    // Cannot call `type_member_is_visible` because it goes up to the parent;
218                    // the `func_meta.source_module` already seems to be the parent.
219                    return module_is_descendant_of_target(
220                        def_map,
221                        current_module.local_id,
222                        func_meta.source_module,
223                    );
224                } else {
225                    // If visibility is PublicCrate, then we are good, because is_same_type_regardless_generics
226                    // already checked that the types are the same, so we are in the same crate.
227                    return true;
228                }
229            }
230
231            if let Some(struct_id) = func_meta.type_id {
232                // For inherent impl methods, check visibility against the impl's
233                // defining module (source_module). This prevents private methods defined
234                // in `impl super::S` inside `mod private` from being callable outside
235                // `mod private` via `s.method()`.
236                if func_meta.trait_impl.is_none() {
237                    let source_module = ModuleId {
238                        krate: func_meta.source_crate,
239                        local_id: func_meta.source_module,
240                    };
241                    let type_module = struct_id.module_id();
242                    if source_module != type_module {
243                        return item_in_module_is_visible(
244                            def_maps,
245                            current_module,
246                            source_module,
247                            modifiers.visibility,
248                        );
249                    }
250                }
251
252                return struct_member_is_visible(
253                    struct_id,
254                    modifiers.visibility,
255                    current_module,
256                    def_maps,
257                );
258            }
259
260            if object_type.is_primitive() {
261                let func_module = interner.function_module(func_id);
262                return item_in_module_is_visible(
263                    def_maps,
264                    current_module,
265                    func_module,
266                    modifiers.visibility,
267                );
268            }
269
270            true
271        }
272    }
273}

275/// Returns whether two types are the same disregarding their generic arguments.
276fn is_same_type_regardless_generics(type1: &Type, type2: &Type) -> bool {
277    if type1 == type2 {
278        return true;
279    }
280
281    match (type1.follow_bindings(), type2.follow_bindings()) {
282        (Type::Array(..), Type::Array(..)) => true,
283        (Type::Vector(..), Type::Vector(..)) => true,
284        (Type::String(..), Type::String(..)) => true,
285        (Type::FmtString(..), Type::FmtString(..)) => true,
286        (Type::Tuple(..), Type::Tuple(..)) => true,
287        (Type::Function(..), Type::Function(..)) => true,
288        (Type::DataType(data_type1, ..), Type::DataType(data_type2, ..)) => {
289            data_type1.borrow().id == data_type2.borrow().id
290        }
291        (Type::Reference(type1, _), _) => is_same_type_regardless_generics(&type1, type2),
292        (_, Type::Reference(type2, _)) => is_same_type_regardless_generics(type1, &type2),
293        _ => false,
294    }
295}

297pub fn module_def_id_visibility(
298    module_def_id: ModuleDefId,
299    interner: &NodeInterner,
300) -> ItemVisibility {
301    match module_def_id {
302        ModuleDefId::ModuleId(module_id) => {
303            let attributes = interner.try_module_attributes(module_id);
304            attributes.map_or(ItemVisibility::Private, |a| a.visibility)
305        }
306        ModuleDefId::FunctionId(func_id) => interner.function_modifiers(&func_id).visibility,
307        ModuleDefId::TypeId(type_id) => {
308            let data_type = interner.get_type(type_id);
309            data_type.borrow().visibility
310        }
311        ModuleDefId::TypeAliasId(type_alias_id) => {
312            let type_alias = interner.get_type_alias(type_alias_id);
313            type_alias.borrow().visibility
314        }
315        ModuleDefId::TraitAssociatedTypeId(_) => ItemVisibility::Public,
316        ModuleDefId::TraitId(trait_id) => {
317            let trait_ = interner.get_trait(trait_id);
318            trait_.visibility
319        }
320        ModuleDefId::GlobalId(global_id) => {
321            let global_info = interner.get_global(global_id);
322            global_info.visibility
323        }
324    }
