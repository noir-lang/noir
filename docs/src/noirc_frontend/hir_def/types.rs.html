157    pub type_var: TypeVariable,
158    /// The name of the generic type.
159    ///
160    /// If this is an associated type, then it has the format `"<{object} as {trait}>::{name}"`

167impl NamedGeneric {
168    /// Create a [NamedGeneric].
169    ///
170    /// If an object and trait name pair is given in `as_trait`, this is assumed
171    /// to be an associated type, and its name will be formatted as `"<{object} as {trait}>::{name}"`
172    /// to disambiguate from other generics in scope.
173    pub fn new(
174        type_var: TypeVariable,
175        implicit: bool,
176        name: &Rc<String>,
177        as_trait: Option<(&str, &str)>,
178    ) -> Self {
179        let name = match as_trait {
180            // TODO(#10858): The compiler rejects `trait Foo { fn foo_bar() -> <Self as Foo>::Bar; }` (unlike Rust),
181            // so in order to be able to parse back expanded code, we have to format it as `Self::Bar`.
182            Some((object, _)) if object == SELF_TYPE_NAME => Rc::new(format!("{object}::{name}")),
183            Some((object, trait_name)) => Rc::new(format!("<{object} as {trait_name}>::{name}")),
184            None => name.clone(),
185        };
186        Self { type_var, name, implicit }
187    }
188}

190/// A Kind is the type of a Type. These are used since only certain kinds of types are allowed in
191/// certain positions.
192///
193/// For example, the type of a struct field or a function parameter is expected to be
194/// a type of kind * (represented here as `Normal`). Types used in positions where a number
195/// is expected (such as in an array length position) are expected to be of kind `Kind::Numeric`.
196#[derive(PartialEq, Eq, Clone, Hash, Debug, PartialOrd, Ord)]
197pub enum Kind {
198    /// Can bind to any type
199    // TODO(https://github.com/noir-lang/noir/issues/6194): evaluate need for and usage of
200    Any,
201
202    /// Can bind to any type, except Type::Constant and Type::InfixExpr
203    Normal,
204
205    /// A generic integer or field type. This is a more specific kind of TypeVariable
206    /// that can only be bound to Type::Field, Type::Integer, or other polymorphic integers.
207    /// This is the type of undecorated integer literals like `46`. Typing them in this way
208    /// allows them to be polymorphic over the actual integer/field type used without requiring
209    /// type annotations on each integer literal.
210    IntegerOrField,
211
212    /// A generic integer type. This is a more specific kind of TypeVariable
213    /// that can only be bound to Type::Integer, or other polymorphic integers.
214    Integer,
215
216    /// Can bind to a Type::Constant or Type::InfixExpr of the given kind
217    Numeric(Box<Type>),
218}

220impl Kind {
221    // Kind::Numeric constructor helper
222    pub fn numeric(typ: Type) -> Kind {
223        Kind::Numeric(Box::new(typ))
224    }
225
226    pub(crate) fn is_error(&self) -> bool {
227        match self.follow_bindings() {
228            Self::Numeric(typ) => *typ == Type::Error,
229            _ => false,
230        }
231    }
232
233    pub(crate) fn u32() -> Self {
234        Self::numeric(Type::u32())
235    }
236
237    pub(crate) fn follow_bindings(&self) -> Self {
238        match self {
239            Self::Any => Self::Any,
240            Self::Normal => Self::Normal,
241            Self::Integer => Self::Integer,
242            Self::IntegerOrField => Self::IntegerOrField,
243            Self::Numeric(typ) => Self::numeric(typ.follow_bindings()),
244        }
245    }
246
247    /// Returns the default type this type variable should be bound to if it is still unbound
248    /// during monomorphization.
249    pub(crate) fn default_type(&self) -> Option<Type> {
250        match self {
251            Kind::IntegerOrField => Some(Type::default_int_or_field_type()),
252            Kind::Integer => Some(Type::default_int_type()),
253            Kind::Numeric(_typ) => {
254                // Even though we have a type here, that type cannot be used as
255                // the default type of a numeric generic.
256                // For example, if we have `let N: u32` and we don't know
257                // what `N` is, we can't assume it's `u32`.
258                None
259            }
260            Kind::Any | Kind::Normal => None,
261        }
262    }
263
264    fn integral_maximum_size(&self) -> Option<FieldElement> {
265        match self.follow_bindings() {
266            Kind::Any | Kind::IntegerOrField | Kind::Integer | Kind::Normal => None,
267            Self::Numeric(typ) => typ.integral_maximum_size(),
268        }
269    }
270
271    fn integral_minimum_size(&self) -> Option<SignedField> {
272        match self.follow_bindings() {
273            Kind::Any | Kind::IntegerOrField | Kind::Integer | Kind::Normal => None,
274            Self::Numeric(typ) => typ.integral_minimum_size(),
275        }
276    }
277
278    /// Ensure the given value fits in self.integral_maximum_size()
279    pub(crate) fn ensure_value_fits(
280        &self,
281        value: SignedField,
282        location: Location,
283    ) -> Result<SignedField, TypeCheckError> {
284        if let Some(maximum_size) = self.integral_maximum_size() {
285            if value > SignedField::positive(maximum_size) {
286                return Err(TypeCheckError::OverflowingConstant {
287                    value,
288                    kind: self.clone(),
289                    maximum_size,
290                    location,
291                });
292            }
293        }
294
295        if let Some(minimum_size) = self.integral_minimum_size() {
296            if value < minimum_size {
297                return Err(TypeCheckError::UnderflowingConstant {
298                    value,
299                    kind: self.clone(),
300                    minimum_size,
301                    location,
302                });
303            }
304        }
305
306        Ok(value)
307    }

309    /// Return the corresponding IntegerTypeSuffix if this is a numeric type kind.
310    /// Note that `Kind::IntegerOrField` and `Kind::Integer` resolve to types and are thus not numeric.
311    pub(crate) fn as_integer_type_suffix(&self) -> Option<IntegerTypeSuffix> {
312        match self {
313            Kind::Numeric(typ) => typ.as_integer_type_suffix(),
314            _ => None,
315        }
316    }

318    pub(crate) fn is_normal_or_any(&self) -> bool {
319        matches!(self, Kind::Normal | Kind::Any)
320    }

332    }

368/// Pretty print type bindings for debugging
369#[allow(unused)]
370pub fn type_bindings_to_string(bindings: &TypeBindings) -> String {
371    if bindings.is_empty() {
372        return "bindings: (none)".to_string();
373    }
374
375    let mut ret = if bindings.len() == 1 {
376        "1 binding:".to_string()
377    } else {
378        format!("{} bindings:", bindings.len())
379    };
380
381    for (var, _, binding) in bindings.values() {
382        ret += &format!("\n    {var:?} := {binding:?}");
383    }
384
385    ret
386}

578    /// Returns the field matching the given field name, as well as its visibility and field index.
579    /// Always returns None if this is not a struct type.
580    pub fn get_field(
581        &self,
582        field_name: &str,
583        generic_args: &[Type],
584    ) -> Option<(Type, ItemVisibility, usize)> {
585        assert_eq!(self.generics.len(), generic_args.len());
586
587        let mut fields = self.fields_raw()?.iter().enumerate();
588        fields.find(|(_, field)| field.name.as_str() == field_name).map(|(i, field)| {
589            let generics = self.generics.iter().zip(generic_args);
590            let substitutions = generics
591                .map(|(old, new)| {
592                    (old.type_var.id(), (old.type_var.clone(), old.type_var.kind(), new.clone()))
593                })
594                .collect();
595
596            (field.typ.substitute(&substitutions), field.visibility, i)
597        })
598    }

650    fn get_fields_substitutions(
651        &self,
652        generic_args: &[Type],
653    ) -> HashMap<TypeVariableId, (TypeVariable, Kind, Type)> {
654        assert_eq!(
655            self.generics.len(),
656            generic_args.len(),
657            "get_fields_substitutions: expected the number of generics to equal the number of generic_args"
658        );

661            .iter()
662            .zip(generic_args)
663            .map(|(old, new)| {
664                (old.type_var.id(), (old.type_var.clone(), old.type_var.kind(), new.clone()))
665            })

723    /// Requires the `Shared<DataType>` handle of self to create the given function type.
724    /// Panics if this is not an enum.
725    ///
726    /// The function type uses the variant "as written" ie. no generic substitutions.
727    /// Although the returned function is technically generic, Type::Function is returned
728    /// instead of Type::Forall.
729    pub fn variant_function_type(&self, variant_index: usize, this: Shared<DataType>) -> Type {
730        let variant = self.variant_at(variant_index);
731        let args = variant.params.clone();
732        assert_eq!(this.borrow().id, self.id);
733        let generics = self.generic_types();
734        let ret = Box::new(Type::DataType(this, generics));
735        Type::Function(args, ret, Box::new(Type::Unit), false)
736    }

739    /// Requires the `Shared<DataType>` handle of self to create the given function type.
740    /// Panics if this is not an enum.
741    pub fn variant_function_type_with_forall(
742        &self,
743        variant_index: usize,
744        this: Shared<DataType>,
745    ) -> Type {
746        let function_type = self.variant_function_type(variant_index, this);
747        let typevars = vecmap(&self.generics, |generic| generic.type_var.clone());
748        Type::Forall(typevars, Box::new(function_type))
749    }

937/// A TypeVariable is a mutable reference that is either
938/// bound to some type, or unbound with a given TypeVariableId.

940pub struct TypeVariable(TypeVariableId, Shared<TypeBinding>);

942impl TypeVariable {
943    pub fn unbound(id: TypeVariableId, type_var_kind: Kind) -> Self {
944        TypeVariable(id, Shared::new(TypeBinding::Unbound(id, type_var_kind)))
945    }

947    pub fn id(&self) -> TypeVariableId {
948        self.0
949    }
950
951    /// Bind this type variable to a value.
952    ///
953    /// Panics if this TypeVariable is already Bound.
954    /// Also Panics if the ID of this TypeVariable occurs within the given
955    /// binding, as that would cause an infinitely recursive type.
956    pub fn bind(&self, typ: Type) {
957        let id = match &*self.1.borrow() {
958            TypeBinding::Bound(binding) => {
959                unreachable!("TypeVariable::bind, cannot bind bound var {} to {}", binding, typ)
960            }
961            TypeBinding::Unbound(id, _) => *id,
962        };
963
964        assert!(!typ.occurs(id), "{self:?} occurs within {typ:?}");
965        *self.1.borrow_mut() = TypeBinding::Bound(typ);
966    }
967
968    pub fn try_bind(
969        &self,
970        binding: Type,
971        kind: &Kind,
972        location: Location,
973    ) -> Result<(), TypeCheckError> {
974        if !binding.kind().unifies(kind) {
975            return Err(TypeCheckError::TypeKindMismatch {
976                expected_kind: kind.clone(),
977                expr_kind: binding.kind(),
978                expr_location: location,
979            });
980        }
981
982        let id = match &*self.1.borrow() {
983            TypeBinding::Bound(binding) => {
984                unreachable!("Expected unbound, found bound to {binding}")
985            }
986            TypeBinding::Unbound(id, _) => *id,
987        };
988
989        if binding.occurs(id) {
990            Err(TypeCheckError::CyclicType { location, typ: binding })
991        } else {
992            *self.1.borrow_mut() = TypeBinding::Bound(binding);
993            Ok(())
994        }
995    }

1000    }

1130impl std::fmt::Display for Type {
1131    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
1132        match self {
1133            Type::FieldElement => {
1134                write!(f, "Field")
1135            }
1136            Type::Array(len, typ) => {
1137                write!(f, "[{typ}; {len}]")
1138            }
1139            Type::Vector(typ) => {
1140                write!(f, "[{typ}]")
1141            }
1142            Type::Integer(sign, num_bits) => match sign {
1143                Signedness::Signed => write!(f, "i{num_bits}"),
1144                Signedness::Unsigned => write!(f, "u{num_bits}"),
1145            },
1146            Type::TypeVariable(var) => {
1147                let binding = &var.1;
1148                match &*binding.borrow() {
1149                    TypeBinding::Unbound(_, type_var_kind) => match type_var_kind {
1150                        Kind::Any | Kind::Normal => write!(f, "{}", var.borrow()),
1151                        Kind::Integer => write!(f, "{}", Type::default_int_type()),
1152                        Kind::IntegerOrField => write!(f, "Field"),
1153                        Kind::Numeric(_typ) => write!(f, "_"),
1154                    },
1155                    TypeBinding::Bound(binding) => {
1156                        write!(f, "{binding}")
1157                    }
1158                }
1159            }
1160            Type::DataType(s, args) => {
1161                let args = vecmap(args, |arg| arg.to_string());
1162                if args.is_empty() {
1163                    write!(f, "{}", s.borrow())
1164                } else {
1165                    write!(f, "{}<{}>", s.borrow(), args.join(", "))
1166                }
1167            }
1168            Type::Alias(alias, args) => {
1169                let args = vecmap(args, |arg| arg.to_string());
1170                if args.is_empty() {
1171                    write!(f, "{}", alias.borrow())
1172                } else {
1173                    write!(f, "{}<{}>", alias.borrow(), args.join(", "))
1174                }
1175            }
1176            Type::TraitAsType(_id, name, generics) => {
1177                write!(f, "impl {name}{generics}")
1178            }
1179            Type::Tuple(elements) => {
1180                let elements = vecmap(elements, ToString::to_string);
1181                if elements.len() == 1 {
1182                    write!(f, "({},)", elements[0])
1183                } else {
1184                    write!(f, "({})", elements.join(", "))
1185                }
1186            }
1187            Type::Bool => write!(f, "bool"),
1188            Type::String(len) => write!(f, "str<{len}>"),
1189            Type::FmtString(len, elements) => {
1190                write!(f, "fmtstr<{len}, {elements}>")
1191            }
1192            Type::Unit => write!(f, "()"),
1193            Type::Error => write!(f, "error"),
1194            Type::NamedGeneric(NamedGeneric { type_var, name, .. }) => match &*type_var.borrow() {
1195                TypeBinding::Bound(type_var) => type_var.fmt(f),
1196                TypeBinding::Unbound(_, _) if name.is_empty() => write!(f, "_"),
1197                TypeBinding::Unbound(_, _) => write!(f, "{name}"),
1198            },
1199            Type::CheckedCast { to, .. } => write!(f, "{to}"),
1200            Type::Constant(x, _kind) => write!(f, "{x}"),
1201            Type::Forall(typevars, typ) => {
1202                let typevars = vecmap(typevars, |var| var.id().to_string());
1203                write!(f, "forall {}. {}", typevars.join(" "), typ)
1204            }
1205            Type::Function(args, ret, env, unconstrained) => {
1206                if *unconstrained {
1207                    write!(f, "unconstrained ")?;
1208                }
1209
1210                let closure_env_text = match **env {
1211                    Type::Unit => "".to_string(),
1212                    _ => format!("[{env}]"),
1213                };
1214
1215                let args = vecmap(args.iter(), ToString::to_string);
1216
1217                write!(f, "fn{closure_env_text}({}) -> {ret}", args.join(", "))
1218            }
1219            Type::Reference(element, mutable) if *mutable => {
1220                write!(f, "&mut {element}")
1221            }
1222            Type::Reference(element, _) => {
1223                write!(f, "&{element}")
1224            }
1225            Type::Quoted(quoted) => write!(f, "{quoted}"),
1226            Type::InfixExpr(lhs, op, rhs, _) => {
1227                let this = self.canonicalize_checked();
1228
1229                // Prevent infinite recursion
1230                if this != *self { write!(f, "{this}") } else { write!(f, "({lhs} {op} {rhs})") }
1231            }
1232        }
1233    }
1234}

1433    ///
1434    /// This should try filter out types which would cause a panic in `abi_gen::abi_type_from_hir_type`,
1435    /// but it has to be more permissive, as we don't have all information yet; some only become apparent
1436    /// after monomorphization.

1438        match self {
1439            Type::FieldElement | Type::Integer(_, _) | Type::Bool | Type::String(_) => true,
1440

1442            Type::TypeVariable(binding) => match &*binding.borrow() {
1443                TypeBinding::Bound(typ) => typ.is_message_compatible(is_monomorphized),

1447            },

1452            }

1456            }
1457            Type::CheckedCast { to, .. } => to.is_message_compatible(is_monomorphized),

1464            | Type::InfixExpr(..)

1557    /// Determines if a type contains a self referring alias by tracking visited TypeAliasId.
1558    ///
1559    /// - `aliases` is a mutable set of TypeAliasId to track visited aliases
1560    /// - it returns `true` if a cyclic alias is detected, `false` otherwise
1561    ///
1562    /// Note: cloning the `aliases` parameter when calling this function recursively in multiple
1563    /// branches, e.g. as done with [`Type::InfixExpr`], can prevent tests like
1564    /// `ensure_repeated_aliases_in_tuples_are_not_detected_as_cyclic_aliases` and
1565    /// `ensure_repeated_aliases_in_arrays_are_not_detected_as_cyclic_aliases` from failing
1566    /// due to the same non-cyclic alias being detected twice in different recursive calls
1567    pub fn has_cyclic_alias(&self) -> bool {
1568        let mut aliases = im::HashSet::<TypeAliasId>::default();
1569        let mut seen_data_types = SeenDataTypes::default();
1570        self.has_cyclic_alias_helper(&mut aliases, &mut seen_data_types)
1571    }
1572
1573    /// `aliases` is a mutable set of TypeAliasId to track visited aliases
1574    fn has_cyclic_alias_helper(
1575        &self,
1576        aliases: &mut im::HashSet<TypeAliasId>,
1577        seen_data_types: &mut SeenDataTypes,
1578    ) -> bool {
1579        match self {
1580            Type::NamedGeneric(NamedGeneric { type_var, .. }) | Type::TypeVariable(type_var) => {
1581                type_var.has_cyclic_alias(aliases, seen_data_types)
1582            }
1583            Type::InfixExpr(lhs, _op, rhs, _) => {
1584                lhs.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1585                    || rhs.has_cyclic_alias_helper(aliases, seen_data_types)
1586            }
1587            Type::Alias(def, generics) => {
1588                let alias_id = def.borrow().id;
1589                if aliases.contains(&alias_id) {
1590                    true
1591                } else {
1592                    aliases.insert(alias_id);
1593                    def.borrow()
1594                        .get_type(generics)
1595                        .has_cyclic_alias_helper(aliases, seen_data_types)
1596                }
1597            }
1598            Type::TraitAsType(_id, _name, generics) => {
1599                generics.ordered.iter().any(|generic| {
1600                    generic.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1601                }) || generics.named.iter().any(|generic| {
1602                    generic.typ.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1603                })
1604            }
1605            Type::String(len) => len.has_cyclic_alias_helper(aliases, seen_data_types),
1606            Type::Array(len, typ) => {
1607                len.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1608                    || typ.has_cyclic_alias_helper(aliases, seen_data_types)
1609            }
1610            Type::Vector(typ) => typ.has_cyclic_alias_helper(aliases, seen_data_types),
1611            Type::DataType(s, args) => {
1612                let data_type = s.borrow();
1613                let key = (data_type.id, args.clone());
1614                if seen_data_types.insert(key) {
1615                    data_type.get_fields(args).unwrap_or_else(Vec::new).iter().any(
1616                        |(_name, field, _visibility)| {
1617                            field.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1618                        },
1619                    ) || data_type.get_variants(args).unwrap_or_else(Vec::new).iter().any(
1620                        |(_name, variant)| {
1621                            variant.iter().any(|variant_field| {
1622                                variant_field
1623                                    .has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1624                            })
1625                        },
1626                    )
1627                } else {
1628                    false
1629                }
1630            }
1631            Type::Tuple(elements) => elements.iter().any(|element| {
1632                element.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1633            }),
1634            Type::FmtString(len, elements) => {
1635                len.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1636                    || (*elements).has_cyclic_alias_helper(aliases, seen_data_types)
1637            }
1638            Type::CheckedCast { to, from } => {
1639                to.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1640                    || from.has_cyclic_alias_helper(aliases, seen_data_types)
1641            }
1642            Type::Constant(_x, kind) => kind.has_cyclic_alias(aliases, seen_data_types),
1643            Type::Forall(typevars, typ) => {
1644                typevars
1645                    .iter()
1646                    .any(|typevar| typevar.has_cyclic_alias(&mut aliases.clone(), seen_data_types))
1647                    || typ.has_cyclic_alias_helper(aliases, seen_data_types)
1648            }
1649            Type::Function(args, ret, env, _unconstrained) => {
1650                args.iter()
1651                    .any(|arg| arg.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types))
1652                    || ret.has_cyclic_alias_helper(&mut aliases.clone(), seen_data_types)
1653                    || env.has_cyclic_alias_helper(aliases, seen_data_types)
1654            }
1655            Type::Reference(element, _mutable) => {
1656                element.has_cyclic_alias_helper(aliases, seen_data_types)
1657            }
1658            Type::FieldElement
1659            | Type::Integer(_, _)
1660            | Type::Bool
1661            | Type::Unit
1662            | Type::Error
1663            | Type::Quoted(_) => false,
1664        }
1665    }

1679    /// Creates an `InfixExpr` that results from the compiler trying to unify something like
1680    /// `4 = a * b` into `a = 4 / b` (where `4 / b` is the "inverted" expression).

1685    pub fn new_infix_expr(
1686        lhs: Box<Type>,
1687        op: BinaryTypeOperator,
1688        rhs: Box<Type>,
1689        inversion: bool,
1690    ) -> Type {
1691        // If this infix expression contains an error then it is eventually an error itself.
1692        if matches!(*lhs, Type::Error) || matches!(*rhs, Type::Error) {
1693            return Type::Error;
1694        }
1695
1696        // If an InfixExpr like this is tried to be created:
1697        //
1698        // a * (b / a)
1699        //
1700        // where `b / a` resulted from the compiler creating an inverted InfixExpr from a previous
1701        // unification (that is, the compiler had `b = a / y` and ended up doing `y = b / a` where
1702        // `y` is `rhs` here) then we can simplify this to just `b` because there wasn't an actual
1703        // division in the original expression, so multiplying it back is just going back to the
1704        // original `y`
1705        if let Type::InfixExpr(rhs_lhs, rhs_op, rhs_rhs, true) = &*rhs {
1706            if op.approx_inverse() == Some(*rhs_op) && lhs == *rhs_rhs {
1707                return *rhs_lhs.clone();
1708            }
1709        }
1710
1711        // Same thing but on the other side.
1712        if let Type::InfixExpr(lhs_lhs, lhs_op, lhs_rhs, true) = &*lhs {
1713            if op.approx_inverse() == Some(*lhs_op) && rhs == *lhs_rhs {
1714                return *lhs_lhs.clone();
1715            }
1716        }
1717
1718        Self::InfixExpr(lhs, op, rhs, inversion)
1719    }

1786            | Type::TraitAsType(..)

2218    /// Try to bind a PolymorphicInt variable to self, succeeding if self is an integer, field,
2219    /// other PolymorphicInt type, or type variable. If successful, the binding is placed in the
2220    /// given TypeBindings map rather than linked immediately.
2221    fn try_bind_to_polymorphic_int(
2222        &self,
2223        var: &TypeVariable,
2224        bindings: &mut TypeBindings,
2225        only_integer: bool,
2226    ) -> Result<(), UnificationError> {
2227        let target_id = match &*var.borrow() {
2228            TypeBinding::Bound(_) => unreachable!(),
2229            TypeBinding::Unbound(id, _) => *id,
2230        };
2231
2232        if !self.kind().unifies(&Kind::IntegerOrField) {
2233            return Err(UnificationError);
2234        }
2235
2236        let this = self.substitute(bindings).follow_bindings();
2237        match &this {
2238            Type::Integer(..) => {
2239                bindings.insert(target_id, (var.clone(), Kind::Integer, this));
2240                Ok(())
2241            }
2242            Type::FieldElement if !only_integer => {
2243                bindings.insert(target_id, (var.clone(), Kind::IntegerOrField, this));
2244                Ok(())
2245            }
2246            Type::TypeVariable(self_var) => {
2247                let borrow = self_var.borrow();
2248                match &*borrow {
2249                    TypeBinding::Bound(typ) => {
2250                        typ.try_bind_to_polymorphic_int(var, bindings, only_integer)
2251                    }
2252                    // Avoid infinitely recursive bindings
2253                    TypeBinding::Unbound(id, _) if *id == target_id => Ok(()),
2254                    TypeBinding::Unbound(new_target_id, Kind::IntegerOrField) => {
2255                        let type_var_kind = Kind::IntegerOrField;
2256                        if only_integer {
2257                            let var_clone = var.clone();
2258                            Kind::Integer.unify(&type_var_kind)?;
2259                            // Integer is more specific than IntegerOrField so we bind the type
2260                            // variable to Integer instead.
2261                            let clone = Type::TypeVariable(var_clone);
2262                            bindings
2263                                .insert(*new_target_id, (self_var.clone(), type_var_kind, clone));
2264                        } else {
2265                            bindings.insert(
2266                                target_id,
2267                                (var.clone(), Kind::IntegerOrField, this.clone()),
2268                            );
2269                        }
2270                        Ok(())
2271                    }
2272                    TypeBinding::Unbound(_new_target_id, Kind::Integer) => {
2273                        Kind::Integer.unify(&Kind::Integer)?;
2274                        bindings.insert(target_id, (var.clone(), Kind::Integer, this.clone()));
2275                        Ok(())
2276                    }

2278                        let var_clone = var.clone();

2280                        let clone_kind =
2281                            if only_integer { Kind::Integer } else { Kind::IntegerOrField };
2282                        clone_kind.unify(type_var_kind)?;

2284                        bindings.insert(*new_target_id, (self_var.clone(), clone_kind, clone));

2293    /// Try to bind the given type variable to self. Although the given type variable
2294    /// is expected to be of Kind::Normal, this binding can still fail
2295    /// if the given type variable occurs within `self` as that would create a recursive type.
2296    ///
2297    /// If successful, the binding is placed in the
2298    /// given TypeBindings map rather than linked immediately.
2299    fn try_bind_to(
2300        &self,
2301        var: &TypeVariable,
2302        bindings: &mut TypeBindings,
2303        kind: Kind,
2304    ) -> Result<(), UnificationError> {
2305        let target_id = match &*var.borrow() {
2306            TypeBinding::Bound(_) => unreachable!(),
2307            TypeBinding::Unbound(id, _) => *id,
2308        };
2309
2310        if !self.kind().unifies(&kind) {
2311            return Err(UnificationError);
2312        }
2313
2314        let this = self.substitute(bindings).follow_bindings();
2315        if let Some((binding, kind)) = this.get_inner_type_variable() {
2316            match &*binding.borrow() {
2317                TypeBinding::Bound(typ) => return typ.try_bind_to(var, bindings, kind),
2318                // Don't recursively bind the same id to itself
2319                TypeBinding::Unbound(id, _) if *id == target_id => return Ok(()),
2320                _ => (),
2321            }
2322        }
2323
2324        // Check if the target id occurs within `this` before binding. Otherwise this could
2325        // cause infinitely recursive types
2326        if this.occurs(target_id) {
2327            Err(UnificationError)
2328        } else {
2329            bindings.insert(target_id, (var.clone(), this.kind(), this.clone()));
2330            Ok(())
2331        }
2332    }

2375    pub(crate) fn evaluate_to_signed_field_helper(
2376        &self,
2377        kind: &Kind,
2378        location: Location,
2379        run_simplifications: bool,
2380    ) -> Result<SignedField, TypeCheckError> {
2381        if let Some((binding, binding_kind)) = self.get_inner_type_variable() {
2382            match &*binding.borrow() {
2383                TypeBinding::Bound(binding) => {
2384                    if kind.unifies(&binding_kind) {
2385                        return binding.evaluate_to_signed_field_helper(
2386                            &binding_kind,
2387                            location,
2388                            run_simplifications,
2389                        );
2390                    }
2391                }
2392                TypeBinding::Unbound(..) => (),
2393            }
2394        }
2395
2396        match self.canonicalize_with_simplifications(run_simplifications) {
2397            Type::Constant(x, constant_kind) => {
2398                if kind.unifies(&constant_kind) {
2399                    kind.ensure_value_fits(x, location)
2400                } else {
2401                    Err(TypeCheckError::TypeKindMismatch {
2402                        expected_kind: constant_kind,
2403                        expr_kind: kind.clone(),
2404                        expr_location: location,
2405                    })
2406                }
2407            }
2408            Type::InfixExpr(lhs, op, rhs, _) => {
2409                let infix_kind = lhs.infix_kind(&rhs);
2410                if kind.unifies(&infix_kind) {
2411                    let lhs_value = lhs.evaluate_to_signed_field_helper(
2412                        &infix_kind,
2413                        location,
2414                        run_simplifications,
2415                    )?;
2416                    let rhs_value = rhs.evaluate_to_signed_field_helper(
2417                        &infix_kind,
2418                        location,
2419                        run_simplifications,
2420                    )?;
2421                    op.function(lhs_value, rhs_value, &infix_kind, location)
2422                } else {
2423                    Err(TypeCheckError::TypeKindMismatch {
2424                        expected_kind: kind.clone(),
2425                        expr_kind: infix_kind,
2426                        expr_location: location,
2427                    })
2428                }
2429            }
2430            Type::CheckedCast { from, to } => {
2431                let to_value = to.evaluate_to_signed_field(kind, location)?;
2432
2433                // if both 'to' and 'from' evaluate to a constant,

2445                            to,
2446                            from,
2447                            to_value,

2449                            location,

2452                } else {

2529    /// are matched against the type list and are eligible for substitution - similar

2531    /// is used and generic substitutions are provided manually by users.

2533    /// Expects the given type vector to be the same length as the Forall type variables.
2534    pub fn instantiate_with_bindings_and_turbofish(

2536        bindings: TypeBindings,
2537        turbofish_types: Vec<Type>,
2538        interner: &NodeInterner,
2539        implicit_generic_count: usize,
2540    ) -> (Type, TypeBindings) {

2542            Type::Forall(typevars, typ) => {
2543                assert_eq!(
2544                    turbofish_types.len() + implicit_generic_count,

2546                    "Turbofish operator used with incorrect generic count which was not caught by name resolution"
2547                );
2548
2549                let implicit_and_turbofish_bindings = (0..implicit_generic_count)
2550                    .map(|_| interner.next_type_variable())
2551                    .chain(turbofish_types);

2554                    .iter()

2564                (instantiated, replacements)

2599    /// This helper function only differs in the additional parameter which, if set,
2600    /// allows substitutions on already-bound type variables. This should be `false`
2601    /// for most uses, but is currently needed during monomorphization when instantiating
2602    /// trait functions to shed any previous bindings from recursive parent calls to the
2603    /// same trait.
2604    fn substitute_helper(
2605        &self,
2606        type_bindings: &TypeBindings,
2607        substitute_bound_typevars: bool,
2608    ) -> Type {
2609        if type_bindings.is_empty() {
2610            return self.clone();
2611        }
2612
2613        let recur_on_binding = |id, replacement: &Type| {
2614            // Prevent recurring forever if there's a `T := T` binding
2615            if replacement.type_variable_id() == Some(id) {
2616                replacement.clone()
2617            } else {
2618                replacement.substitute_helper(type_bindings, substitute_bound_typevars)
2619            }
2620        };
2621
2622        let substitute_binding = |binding: &TypeVariable| {
2623            // Check the id first to allow substituting to
2624            // type variables that have already been bound over.
2625            // This is needed for monomorphizing trait impl methods.
2626            match type_bindings.get(&binding.0) {
2627                Some((_, _kind, replacement)) if substitute_bound_typevars => {
2628                    recur_on_binding(binding.0, replacement)
2629                }
2630                _ => match &*binding.borrow() {
2631                    TypeBinding::Bound(binding) => {
2632                        binding.substitute_helper(type_bindings, substitute_bound_typevars)
2633                    }
2634                    TypeBinding::Unbound(id, _) => match type_bindings.get(id) {
2635                        Some((_, kind, replacement)) => {
2636                            assert!(
2637                                kind.unifies(&replacement.kind()),
2638                                "while substituting (unbound): expected kind of unbound TypeVariable ({:?}) to match the kind of its binding ({:?})",
2639                                kind,
2640                                replacement.kind()
2641                            );
2642
2643                            recur_on_binding(binding.0, replacement)
2644                        }
2645                        None => self.clone(),
2646                    },
2647                },
2648            }
2649        };
2650
2651        match self {
2652            Type::Array(size, element) => {
2653                let size = size.substitute_helper(type_bindings, substitute_bound_typevars);
2654                let element = element.substitute_helper(type_bindings, substitute_bound_typevars);
2655                Type::Array(Box::new(size), Box::new(element))
2656            }
2657            Type::Vector(element) => {
2658                let element = element.substitute_helper(type_bindings, substitute_bound_typevars);
2659                Type::Vector(Box::new(element))
2660            }
2661            Type::String(size) => {
2662                let size = size.substitute_helper(type_bindings, substitute_bound_typevars);
2663                Type::String(Box::new(size))
2664            }
2665            Type::FmtString(size, fields) => {
2666                let size = size.substitute_helper(type_bindings, substitute_bound_typevars);
2667                let fields = fields.substitute_helper(type_bindings, substitute_bound_typevars);
2668                Type::FmtString(Box::new(size), Box::new(fields))
2669            }
2670            Type::CheckedCast { from, to } => {
2671                let from = from.substitute_helper(type_bindings, substitute_bound_typevars);
2672                let to = to.substitute_helper(type_bindings, substitute_bound_typevars);
2673                Type::CheckedCast { from: Box::new(from), to: Box::new(to) }
2674            }
2675            Type::NamedGeneric(NamedGeneric { type_var, .. }) | Type::TypeVariable(type_var) => {
2676                substitute_binding(type_var)
2677            }
2678
2679            // Do not substitute_helper fields, it can lead to infinite recursion
2680            // and we should not match fields when type checking anyway.
2681            Type::DataType(fields, args) => {
2682                let args = vecmap(args, |arg| {
2683                    arg.substitute_helper(type_bindings, substitute_bound_typevars)
2684                });
2685                Type::DataType(fields.clone(), args)
2686            }
2687            Type::Alias(alias, args) => {
2688                let args = vecmap(args, |arg| {
2689                    arg.substitute_helper(type_bindings, substitute_bound_typevars)
2690                });
2691                Type::Alias(alias.clone(), args)
2692            }
2693            Type::Tuple(fields) => {
2694                let fields = vecmap(fields, |field| {
2695                    field.substitute_helper(type_bindings, substitute_bound_typevars)
2696                });
2697                Type::Tuple(fields)
2698            }
2699            Type::Forall(typevars, typ) => {
2700                // Trying to substitute_helper a variable within a nested Forall
2701                // is usually impossible and indicative of an error in the type checker somewhere.
2702                for var in typevars {
2703                    assert!(!type_bindings.contains_key(&var.id()));
2704                }
2705                let typ = Box::new(typ.substitute_helper(type_bindings, substitute_bound_typevars));
2706                Type::Forall(typevars.clone(), typ)
2707            }
2708            Type::Function(args, ret, env, unconstrained) => {
2709                let args = vecmap(args, |arg| {
2710                    arg.substitute_helper(type_bindings, substitute_bound_typevars)
2711                });
2712                let ret = Box::new(ret.substitute_helper(type_bindings, substitute_bound_typevars));
2713                let env = Box::new(env.substitute_helper(type_bindings, substitute_bound_typevars));
2714                Type::Function(args, ret, env, *unconstrained)
2715            }
2716            Type::Reference(element, mutable) => Type::Reference(
2717                Box::new(element.substitute_helper(type_bindings, substitute_bound_typevars)),
2718                *mutable,
2719            ),
2720
2721            Type::TraitAsType(s, name, generics) => {
2722                let ordered = vecmap(&generics.ordered, |arg| {
2723                    arg.substitute_helper(type_bindings, substitute_bound_typevars)
2724                });
2725                let named = vecmap(&generics.named, |arg| {
2726                    let typ = arg.typ.substitute_helper(type_bindings, substitute_bound_typevars);
2727                    NamedType { name: arg.name.clone(), typ }
2728                });
2729                Type::TraitAsType(*s, name.clone(), TraitGenerics { ordered, named })
2730            }
2731            Type::InfixExpr(lhs, op, rhs, inversion) => {
2732                let lhs = lhs.substitute_helper(type_bindings, substitute_bound_typevars);
2733                let rhs = rhs.substitute_helper(type_bindings, substitute_bound_typevars);
2734                Type::InfixExpr(Box::new(lhs), *op, Box::new(rhs), *inversion)
2735            }
2736
2737            Type::FieldElement
2738            | Type::Integer(_, _)
2739            | Type::Bool
2740            | Type::Constant(_, _)
2741            | Type::Error
2742            | Type::Quoted(_)
2743            | Type::Unit => self.clone(),
2744        }
2745    }

2747    /// True if the given TypeVariableId is free anywhere within self
2748    pub fn occurs(&self, target_id: TypeVariableId) -> bool {
2749        match self {
2750            Type::Array(len, elem) => len.occurs(target_id) || elem.occurs(target_id),
2751            Type::Vector(elem) => elem.occurs(target_id),
2752            Type::String(len) => len.occurs(target_id),
2753            Type::FmtString(len, fields) => {
2754                let len_occurs = len.occurs(target_id);
2755                let field_occurs = fields.occurs(target_id);
2756                len_occurs || field_occurs
2757            }
2758            Type::DataType(_, generic_args) | Type::Alias(_, generic_args) => {
2759                generic_args.iter().any(|arg| arg.occurs(target_id))
2760            }
2761            Type::TraitAsType(_, _, args) => {
2762                args.ordered.iter().any(|arg| arg.occurs(target_id))
2763                    || args.named.iter().any(|arg| arg.typ.occurs(target_id))
2764            }
2765            Type::Tuple(fields) => fields.iter().any(|field| field.occurs(target_id)),
2766            Type::CheckedCast { from, to } => from.occurs(target_id) || to.occurs(target_id),
2767            Type::NamedGeneric(NamedGeneric { type_var, .. }) | Type::TypeVariable(type_var) => {
2768                match &*type_var.borrow() {
2769                    TypeBinding::Bound(binding) => {
2770                        type_var.id() == target_id || binding.occurs(target_id)
2771                    }
2772                    TypeBinding::Unbound(id, _) => *id == target_id,
2773                }
2774            }
2775            Type::Forall(typevars, typ) => {
2776                !typevars.iter().any(|var| var.id() == target_id) && typ.occurs(target_id)
2777            }
2778            Type::Function(args, ret, env, _unconstrained) => {
2779                args.iter().any(|arg| arg.occurs(target_id))
2780                    || ret.occurs(target_id)
2781                    || env.occurs(target_id)
2782            }
2783            Type::Reference(element, _) => element.occurs(target_id),
2784            Type::InfixExpr(lhs, _op, rhs, _) => lhs.occurs(target_id) || rhs.occurs(target_id),
2785
2786            Type::FieldElement
2787            | Type::Integer(_, _)
2788            | Type::Bool
2789            | Type::Constant(_, _)
2790            | Type::Error
2791            | Type::Quoted(_)
2792            | Type::Unit => false,
2793        }
2794    }

2796    /// Follow any TypeVariable bindings within this type. Doing so ensures
2797    /// that if the bindings are rebound or unbound from under the type then the
2798    /// returned type will not change (because it will no longer contain the
2799    /// links that may be unbound).
2800    ///
2801    /// Expected to be called on an instantiated type (with no Type::Foralls)
2802    pub fn follow_bindings(&self) -> Type {
2803        fn helper(this: &Type, mut i: u32) -> Type {
2804            if i >= TYPE_RECURSION_LIMIT {
2805                panic!("Type recursion limit reached - types are too large")
2806            }
2807            i += 1;
2808            let recur = |typ| helper(typ, i);
2809
2810            use Type::*;
2811            match this {
2812                Array(size, elem) => Array(Box::new(recur(size)), Box::new(recur(elem))),
2813                Vector(elem) => Vector(Box::new(recur(elem))),
2814                String(size) => String(Box::new(recur(size))),
2815                FmtString(size, args) => {
2816                    let size = Box::new(recur(size));
2817                    let args = Box::new(recur(args));
2818                    FmtString(size, args)
2819                }
2820                DataType(def, args) => {
2821                    let args = vecmap(args, recur);
2822                    DataType(def.clone(), args)
2823                }
2824                Alias(def, args) => {
2825                    // We don't need to vecmap(args, recur) since we're recursively
2826                    // calling recur here already.
2827                    recur(&def.borrow().get_type(args))
2828                }
2829                Tuple(args) => Tuple(vecmap(args, recur)),
2830                CheckedCast { from, to } => {
2831                    let from = Box::new(recur(from));
2832                    let to = Box::new(recur(to));
2833                    CheckedCast { from, to }
2834                }
2835                TypeVariable(var) | NamedGeneric(types::NamedGeneric { type_var: var, .. }) => {
2836                    if let TypeBinding::Bound(typ) = &*var.borrow() {
2837                        return recur(typ);
2838                    }
2839                    this.clone()
2840                }
2841                Function(args, ret, env, unconstrained) => {
2842                    let args = vecmap(args, recur);
2843                    let ret = Box::new(recur(ret));
2844                    let env = Box::new(recur(env));
2845                    Function(args, ret, env, *unconstrained)
2846                }
2847
2848                Reference(element, mutable) => Reference(Box::new(recur(element)), *mutable),
2849
2850                TraitAsType(s, name, args) => {
2851                    let ordered = vecmap(&args.ordered, recur);
2852                    let named = vecmap(&args.named, |arg| NamedType {
2853                        name: arg.name.clone(),
2854                        typ: recur(&arg.typ),
2855                    });
2856                    TraitAsType(*s, name.clone(), TraitGenerics { ordered, named })
2857                }
2858                InfixExpr(lhs, op, rhs, inversion) => {
2859                    let lhs = recur(lhs);
2860                    let rhs = recur(rhs);
2861                    InfixExpr(Box::new(lhs), *op, Box::new(rhs), *inversion)
2862                }
2863
2864                // Expect that this function should only be called on instantiated types
2865                Forall(..) => unreachable!(),
2866                FieldElement | Integer(_, _) | Bool | Constant(_, _) | Unit | Quoted(_) | Error => {
2867                    this.clone()
2868                }
2869            }
2870        }
2871        helper(self, 0)
2872    }

2969                    let mut binding = binding.clone();
2970                    drop(type_binding);
2971                    binding.replace_named_generics_with_type_variables();

2973                } else {
2974                    drop(type_binding);
2975                    *self = Type::TypeVariable(type_var.clone());

2978            Type::Function(args, ret, env, _unconstrained) => {

2985            Type::Reference(elem, _) => elem.replace_named_generics_with_type_variables(),
2986            Type::Forall(_, typ) => typ.replace_named_generics_with_type_variables(),
2987            Type::InfixExpr(lhs, _op, rhs, _) => {
2988                lhs.replace_named_generics_with_type_variables();
2989                rhs.replace_named_generics_with_type_variables();

3576            (Array(lhs_len, lhs_elem), Array(rhs_len, rhs_elem)) => {
3577                lhs_len == rhs_len && lhs_elem == rhs_elem
3578            }
3579            (Vector(lhs_elem), Vector(rhs_elem)) => lhs_elem == rhs_elem,
3580            (Integer(lhs_sign, lhs_bits), Integer(rhs_sign, rhs_bits)) => {
3581                lhs_sign == rhs_sign && lhs_bits == rhs_bits
3582            }
3583            (String(lhs_len), String(rhs_len)) => lhs_len == rhs_len,
3584            (FmtString(lhs_len, lhs_env), FmtString(rhs_len, rhs_env)) => {
3585                lhs_len == rhs_len && lhs_env == rhs_env
3586            }
3587            (Tuple(lhs_types), Tuple(rhs_types)) => lhs_types == rhs_types,
3588            (DataType(lhs_struct, lhs_generics), DataType(rhs_struct, rhs_generics)) => {
3589                lhs_struct == rhs_struct && lhs_generics == rhs_generics
3590            }
3591            (Alias(lhs_alias, lhs_generics), Alias(rhs_alias, rhs_generics)) => {
3592                lhs_alias == rhs_alias && lhs_generics == rhs_generics
3593            }
3594            (TraitAsType(lhs_trait, _, lhs_generics), TraitAsType(rhs_trait, _, rhs_generics)) => {
3595                lhs_trait == rhs_trait && lhs_generics == rhs_generics
3596            }
3597            (
3598                Function(lhs_args, lhs_ret, lhs_env, lhs_unconstrained),
3599                Function(rhs_args, rhs_ret, rhs_env, rhs_unconstrained),
3600            ) => {
3601                let args_and_ret_eq = lhs_args == rhs_args && lhs_ret == rhs_ret;
3602                args_and_ret_eq && lhs_env == rhs_env && lhs_unconstrained == rhs_unconstrained
3603            }
3604            (Reference(lhs_elem, lhs_mut), Reference(rhs_elem, rhs_mut)) => {
3605                lhs_elem == rhs_elem && lhs_mut == rhs_mut
3606            }
3607            (Forall(lhs_vars, lhs_type), Forall(rhs_vars, rhs_type)) => {
3608                lhs_vars == rhs_vars && lhs_type == rhs_type
3609            }
3610            (CheckedCast { to, .. }, other) | (other, CheckedCast { to, .. }) => **to == *other,
3611            (Constant(lhs, lhs_kind), Constant(rhs, rhs_kind)) => {
3612                lhs == rhs && lhs_kind == rhs_kind
3613            }
3614            (Quoted(lhs), Quoted(rhs)) => lhs == rhs,
3615            (InfixExpr(l_lhs, l_op, l_rhs, _), InfixExpr(r_lhs, r_op, r_rhs, _)) => {
3616                l_lhs == r_lhs && l_op == r_op && l_rhs == r_rhs
3617            }
3618            // Two implicitly added unbound named generics are equal
3619            (
3620                NamedGeneric(types::NamedGeneric {
3621                    type_var: lhs_var,
3622                    implicit: true,
3623                    name: lhs_name,
3624                    ..
3625                }),
3626                NamedGeneric(types::NamedGeneric {
3627                    type_var: rhs_var,
3628                    implicit: true,
3629                    name: rhs_name,
3630                    ..
3631                }),
3632            ) => {
3633                lhs_var.borrow().is_unbound()
3634                    && rhs_var.borrow().is_unbound()
3635                    && lhs_name == rhs_name
3636                    || lhs_var.id() == rhs_var.id()
3637            }
3638            // Special case: we consider unbound named generics and type variables to be equal to each
3639            // other if their type variable ids match. This is important for some corner cases in
3640            // monomorphization where we call `replace_named_generics_with_type_variables` but
3641            // still want them to be equal for canonicalization checks in arithmetic generics.
3642            // Without this we'd fail the `serialize` test.
3643            (
3644                NamedGeneric(types::NamedGeneric { type_var: lhs_var, .. }) | TypeVariable(lhs_var),
3645                NamedGeneric(types::NamedGeneric { type_var: rhs_var, .. }) | TypeVariable(rhs_var),
3646            ) => lhs_var.id() == rhs_var.id(),
3647            _ => false,
3648        }
3649    }
3650}
