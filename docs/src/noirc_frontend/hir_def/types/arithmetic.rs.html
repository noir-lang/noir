7impl Type {
8    /// Try to canonicalize the representation of this type.
9    /// Currently the only type with a canonical representation is
10    /// `Type::Infix` where for each consecutive commutative operator
11    /// we sort the non-constant operands by `Type: Ord` and place all constant
12    /// operands at the end, constant folded.
13    ///
14    /// For example:
15    /// - `canonicalize[((1 + N) + M) + 2] = (M + N) + 3`
16    /// - `canonicalize[A + 2 * B + 3 - 2] = A + (B * 2) + 3 - 2`
17    pub fn canonicalize(&self) -> Type {
18        match self.follow_bindings() {
19            Type::CheckedCast { from, to } => Type::CheckedCast {
20                from: Box::new(from.canonicalize_checked()),
21                to: Box::new(to.canonicalize_unchecked()),
22            },
23
24            other => {
25                let non_checked_cast = false;
26                let run_simplifications = true;
27                other.canonicalize_helper(non_checked_cast, run_simplifications)
28            }
29        }
30    }
31
32    pub(crate) fn canonicalize_with_simplifications(&self, run_simplifications: bool) -> Type {
33        self.follow_bindings().canonicalize_helper(false, run_simplifications)
34    }
35
36    /// Only simplify constants and drop/skip any CheckedCast's
37    pub(crate) fn canonicalize_checked(&self) -> Type {
38        self.follow_bindings().canonicalize_checked_helper()
39    }
40
41    /// Only simplify constants and drop/skip any CheckedCast's
42    fn canonicalize_checked_helper(&self) -> Type {
43        let found_checked_cast = true;
44        let skip_simplifications = false;
45        // We expect `self` to have already called `follow_bindings`
46        self.canonicalize_helper(found_checked_cast, skip_simplifications)
47    }
48
49    /// Run all simplifications and drop/skip any CheckedCast's
50    fn canonicalize_unchecked(&self) -> Type {
51        let found_checked_cast = true;
52        let run_simplifications = true;
53        // We expect `self` to have already called `follow_bindings`
54        self.canonicalize_helper(found_checked_cast, run_simplifications)
55    }
56
57    /// If `found_checked_cast`, then drop additional CheckedCast's
58    ///
59    /// If `run_simplifications` is false, then only:
60    /// - Attempt to evaluate each sub-expression to a constant
61    /// - Drop nested CheckedCast's
62    ///
63    /// Otherwise also attempt try_simplify_partial_constants, sort_commutative,
64    /// and other simplifications
65    fn canonicalize_helper(&self, found_checked_cast: bool, run_simplifications: bool) -> Type {
66        match self {
67            Type::InfixExpr(lhs, op, rhs, inversion) => {
68                let kind = lhs.infix_kind(rhs);
69                let dummy_location = Location::dummy();
70
71                let evaluate = |typ: &Type| {
72                    typ.evaluate_to_signed_field_helper(&kind, dummy_location, run_simplifications)
73                };
74                let lhs_evaluated = evaluate(lhs);
75                let rhs_evaluated = evaluate(rhs);
76
77                // evaluate_to_field_element also calls canonicalize so if we just called
78                // `self.evaluate_to_field_element(..)` we'd get infinite recursion.
79                if let Ok(lhs_value) = lhs_evaluated
80                    && let Ok(rhs_value) = rhs_evaluated
81                    && let Ok(result) = op.function(lhs_value, rhs_value, &kind, dummy_location)
82                {
83                    return Type::Constant(result, kind);
84                }
85
86                let lhs = lhs.canonicalize_helper(found_checked_cast, run_simplifications);
87                let rhs = rhs.canonicalize_helper(found_checked_cast, run_simplifications);
88
89                // See if this is `X * 1` or `X / 1` in which case we can simplify it to `X`
90                if matches!(op, BinaryTypeOperator::Multiplication | BinaryTypeOperator::Division)
91                    && let Ok(rhs_value) = rhs_evaluated
92                    && rhs_value.is_one()
93                {
94                    return lhs;
95                }
96
97                // See if this is `X + 0` or `X - 0`, in which case we can simplify it to `X`
98                if matches!(op, BinaryTypeOperator::Addition | BinaryTypeOperator::Subtraction)
99                    && let Ok(rhs_value) = rhs_evaluated
100                    && rhs_value.is_zero()
101                {
102                    return lhs;
103                }
104
105                if !run_simplifications {
106                    return Type::InfixExpr(Box::new(lhs), *op, Box::new(rhs), *inversion);
107                }
108
109                if let Some(result) = Self::try_simplify_non_constants_in_lhs(&lhs, *op, &rhs) {
110                    return result.canonicalize_unchecked();
111                }
112
113                if let Some(result) = Self::try_simplify_non_constants_in_rhs(&lhs, *op, &rhs) {
114                    return result.canonicalize_unchecked();
115                }
116
117                // Try to simplify partially constant expressions in the form `(N op1 C1) op2 C2`
118                // where C1 and C2 are constants that can be combined (e.g. N + 5 - 3 = N + 2)
119                if let Some(result) = Self::try_simplify_partial_constants(&lhs, *op, &rhs) {
120                    return result.canonicalize_unchecked();
121                }
122
123                if op.is_commutative() {
124                    return Self::sort_commutative(&lhs, *op, &rhs);
125                }
126
127                Type::InfixExpr(Box::new(lhs), *op, Box::new(rhs), *inversion)
128            }
129            Type::CheckedCast { from, to } => {
130                let inner_found_checked_cast = true;
131                let to = to.canonicalize_helper(inner_found_checked_cast, run_simplifications);
132
133                if found_checked_cast {
134                    return to;
135                }
136
137                let from = from.canonicalize_checked();
138
139                Type::CheckedCast { from: Box::new(from), to: Box::new(to) }
140            }
141            Type::TypeVariable(type_var) => match &*type_var.borrow() {
142                TypeBinding::Bound(binding) => {
143                    binding.canonicalize_helper(found_checked_cast, run_simplifications)
144                }
145                TypeBinding::Unbound(_, _) => Type::TypeVariable(type_var.clone()),
146            },
147            other => other.clone(),
148        }
149    }

151    fn sort_commutative(lhs: &Type, op: BinaryTypeOperator, rhs: &Type) -> Type {
152        let mut queue = vec![lhs.clone(), rhs.clone()];
153
154        // Maps each term to the number of times that term was used.
155        let mut sorted = BTreeMap::new();
156
157        let zero_value = if op == BinaryTypeOperator::Addition {
158            SignedField::zero()
159        } else {
160            SignedField::one()
161        };
162        let mut constant = zero_value;
163
164        // Push each non-constant term to `sorted` to sort them. Recur on InfixExprs with the same operator.
165        while let Some(item) = queue.pop() {
166            match item.canonicalize_unchecked() {
167                Type::InfixExpr(lhs_inner, new_op, rhs_inner, _) if new_op == op => {
168                    queue.push(*lhs_inner);
169                    queue.push(*rhs_inner);
170                }
171                Type::Constant(new_constant, new_constant_kind) => {
172                    let dummy_location = Location::dummy();
173                    if let Ok(result) =
174                        op.function(constant, new_constant, &new_constant_kind, dummy_location)
175                    {
176                        constant = result;
177                    } else {
178                        let constant = Type::Constant(new_constant, new_constant_kind);
179                        *sorted.entry(constant).or_default() += 1;
180                    }
181                }
182                other => {
183                    *sorted.entry(other).or_default() += 1;
184                }
185            }
186        }
187
188        if let Some(first) = sorted.pop_first() {
189            let (mut typ, first_type_count) = first.clone();
190
191            // - 1 since `typ` already is set to the first instance
192            for _ in 0..first_type_count - 1 {
193                typ = Type::infix_expr(Box::new(typ), op, Box::new(first.0.clone()));
194            }
195
196            for (rhs, rhs_count) in sorted {
197                for _ in 0..rhs_count {
198                    typ = Type::infix_expr(Box::new(typ), op, Box::new(rhs.clone()));
199                }
200            }
201
202            if constant != zero_value {
203                let constant = Type::Constant(constant, lhs.infix_kind(rhs));
204                typ = Type::infix_expr(Box::new(typ), op, Box::new(constant));
205            }
206
207            typ
208        } else {
209            // Every type must have been a constant
210            Type::Constant(constant, lhs.infix_kind(rhs))
211        }
212    }

214    /// Try to simplify non-constant expressions in the form `(N op1 M) op2 M`
215    /// where the two `M` terms are expected to cancel out.
216    /// Precondition: `lhs & rhs are in canonical form`
217    ///
218    /// - Simplifies `(N +/- M) -/+ M` to `N`
219    /// - Simplifies `(N * M) ÷ M` to `N`
220    fn try_simplify_non_constants_in_lhs(
221        lhs: &Type,
222        op: BinaryTypeOperator,
223        rhs: &Type,
224    ) -> Option<Type> {
225        match lhs {
226            Type::CheckedCast { from, to } => {
227                // Apply operation directly to `from` while attempting simplification to `to`.
228                let from = Type::infix_expr(from.clone(), op, Box::new(rhs.clone()));
229                let to = Self::try_simplify_non_constants_in_lhs(to, op, rhs)?;
230                Some(Type::CheckedCast { from: Box::new(from), to: Box::new(to) })
231            }
232            Type::InfixExpr(l_lhs, l_op, l_rhs, _) => {
233                // Note that this is exact, syntactic equality, not unification.
234                // `rhs` is expected to already be in canonical form.
235                if l_op.approx_inverse() != Some(op)
236                    || *l_op == BinaryTypeOperator::Division
237                    || l_rhs.canonicalize_unchecked() != *rhs
238                {
239                    return None;
240                }
241
242                Some(*l_lhs.clone())
243            }
244            _ => None,
245        }
246    }
247
248    /// Try to simplify non-constant expressions in the form `N op1 (M op1 N)`
249    /// where the two `M` terms are expected to cancel out.
250    /// Precondition: `lhs & rhs are in canonical form`
251    ///
252    /// Unlike `try_simplify_non_constants_in_lhs` we can't simplify `N / (M * N)`
253    /// Since that should simplify to `1 / M` instead of `M`.
254    ///
255    /// - Simplifies `N +/- (M -/+ N)` to `M`
256    /// - Simplifies `N * (M ÷ N)` to `M`
257    fn try_simplify_non_constants_in_rhs(
258        lhs: &Type,
259        op: BinaryTypeOperator,
260        rhs: &Type,
261    ) -> Option<Type> {
262        match rhs {
263            Type::CheckedCast { from, to } => {
264                // Apply operation directly to `from` while attempting simplification to `to`.
265                let from = Type::infix_expr(Box::new(lhs.clone()), op, from.clone());
266                let to = Self::try_simplify_non_constants_in_rhs(lhs, op, to)?;
267                Some(Type::CheckedCast { from: Box::new(from), to: Box::new(to) })
268            }
269            Type::InfixExpr(r_lhs, r_op, r_rhs, _) => {
270                // `N / (M * N)` should be simplified to `1 / M`, but we only handle
271                // simplifying to `M` in this function.
272                if op == BinaryTypeOperator::Division && *r_op == BinaryTypeOperator::Multiplication
273                {
274                    return None;
275                }
276
277                // Note that this is exact, syntactic equality, not unification.
278                // `lhs` is expected to already be in canonical form.
279                if r_op.inverse() != Some(op) || *lhs != r_rhs.canonicalize_unchecked() {
280                    return None;
281                }
282
283                Some(*r_lhs.clone())
284            }
285            _ => None,
286        }
287    }

289    /// Given:
290    ///   lhs = `N op C1`
291    ///   rhs = C2
292    /// Returns: `(N, op, C1, C2)` if C1 and C2 are constants.
293    ///   Note that the operator here is within the `lhs` term, the operator
294    ///   separating lhs and rhs is not needed.
295    /// Precondition: `lhs & rhs are in canonical form`
296    fn parse_partial_constant_expr(
297        lhs: &Type,
298        rhs: &Type,
299    ) -> Option<(Box<Type>, BinaryTypeOperator, SignedField, SignedField)> {
300        let kind = lhs.infix_kind(rhs);
301        let dummy_location = Location::dummy();
302        let rhs = rhs.evaluate_to_signed_field(&kind, dummy_location).ok()?;
303
304        let Type::InfixExpr(l_type, l_op, l_rhs, _) = lhs.follow_bindings() else {
305            return None;
306        };
307
308        let l_rhs = l_rhs.evaluate_to_signed_field(&kind, dummy_location).ok()?;
309        Some((l_type, l_op, l_rhs, rhs))
310    }
311
312    /// Try to simplify partially constant expressions in the form `(N op1 C1) op2 C2`
313    /// where C1 and C2 are constants that can be combined (e.g. N + 5 - 3 = N + 2)
314    /// Precondition: `lhs & rhs are in canonical form`
315    ///
316    /// - Simplifies `(N +/- C1) +/- C2` to `N +/- (C1 +/- C2)` if C1 and C2 are constants.
317    /// - Simplifies `(N * C1) ÷ C2` to `N * (C1 ÷ C2)` if C1 and C2 are constants which divide
318    ///   without a remainder.
319    fn try_simplify_partial_constants(
320        lhs: &Type,
321        mut op: BinaryTypeOperator,
322        rhs: &Type,
323    ) -> Option<Type> {
324        use BinaryTypeOperator::*;
325        let (l_type, l_op, l_const, r_const) = Type::parse_partial_constant_expr(lhs, rhs)?;
326
327        match (l_op, op) {
328            (Addition | Subtraction, Addition | Subtraction) => {
329                // If l_op is a subtraction we want to inverse the rhs operator.
330                if l_op == Subtraction {
331                    op = op.inverse()?;
332                }
333                let dummy_location = Location::dummy();
334                let result =
335                    op.function(l_const, r_const, &lhs.infix_kind(rhs), dummy_location).ok()?;
336                let constant = Type::Constant(result, lhs.infix_kind(rhs));
337                Some(Type::infix_expr(l_type, l_op, Box::new(constant)))
338            }
339            (Multiplication, Division) => {
340                // We ensure the result divides evenly to preserve integer division semantics
341                // TODO(https://github.com/noir-lang/noir/issues/11013): do the division simplification
342                // also in case of Field elements
343                let l_128: Option<i128> = l_const.try_to_signed();
344                let r_128: Option<i128> = r_const.try_to_signed();
345                let divides_evenly = if let (Some(l_128), Some(r_128)) = (l_128, r_128) {
346                    l_128.checked_rem(r_128) == Some(0)
347                } else {
348                    false
349                };
350
351                // If op is a division we need to ensure it divides evenly
352                if op == Division && (r_const.is_zero() || !divides_evenly) {
353                    None
354                } else {
355                    let dummy_location = Location::dummy();
356                    let result =
357                        op.function(l_const, r_const, &lhs.infix_kind(rhs), dummy_location).ok()?;
358                    let constant = Box::new(Type::Constant(result, lhs.infix_kind(rhs)));
359                    Some(Type::infix_expr(l_type, l_op, constant))
360                }
361            }
362            _ => None,
363        }
364    }

408    fn solves_n_minus_one_plus_one_through_checked_casts() {
409        // We want to test that the inclusion of a `CheckedCast` won't prevent us from canonicalizing
410        // the expression `(N - 1) + 1` to `N` if there exists a `CheckedCast` on the `N - 1` term.
411
412        let n = Type::NamedGeneric(NamedGeneric {
413            type_var: TypeVariable::unbound(TypeVariableId(0), Kind::u32()),
414            name: std::rc::Rc::new("N".to_owned()),
415            implicit: false,
416            original_type_var_id: None,
417        });
418        let n_minus_one: Type = n.clone() - 1u32.into();
419        let checked_cast_n_minus_one =
420            Type::CheckedCast { from: Box::new(n_minus_one.clone()), to: Box::new(n_minus_one) };
421
422        let n_minus_one_plus_one = checked_cast_n_minus_one.clone() + 1u32.into();
423
424        let canonicalized_typ = n_minus_one_plus_one.canonicalize();
425
426        assert_eq!(n, canonicalized_typ);

431
432        let canonicalized_typ = one_plus_n_minus_one.canonicalize();
433
434        assert_eq!(n, canonicalized_typ);

595            (variable_index in any::<usize>())
596        -> Type {
597            assert!(num_variables != 0);
598            let id = TypeVariableId(variable_index % num_variables);
599            let kind = Kind::numeric(typ.clone());
600            let var = TypeVariable::unbound(id, kind);
601            Type::TypeVariable(var)
602        }

606        (0..num_variables).map(move |id| {

613    fn infix_expr_helper(lhs_op_rhs: (Type, BinaryTypeOperator, Type)) -> Type {
614        let (lhs, op, rhs) = lhs_op_rhs;
615        assert_eq!(lhs.kind(), rhs.kind());
616        let op = if lhs.kind().integral_maximum_size().is_none()
617            && matches!(op, BinaryTypeOperator::Modulo)
618        {
619            // `Field` does not currently support modulo
620            BinaryTypeOperator::Addition
621        } else {
622            op
623        };
624        Type::infix_expr(Box::new(lhs), op, Box::new(rhs))
625    }

627    fn arbitrary_infix_expr_without_variables_helper(
628        typ: Type,
629        arbitrary_value: BoxedStrategy<SignedField>,
630    ) -> impl Strategy<Value = Type> {
631        let leaf = prop_oneof![
632            arbitrary_value
633                .prop_map(move |value| Type::Constant(value, Kind::numeric(typ.clone()))),
634        ];
635
636        leaf.prop_recursive(
637            8,   // 8 levels deep maximum
638            256, // Shoot for maximum size of 256 nodes
639            10,  // We put up to 10 items per collection
640            |inner| (inner.clone(), any::<BinaryTypeOperator>(), inner).prop_map(infix_expr_helper),
641        )
642    }

654    fn arbitrary_infix_expr_with_variables(
655        typ: Type,
656        arbitrary_value: BoxedStrategy<SignedField>,
657        num_variables: usize,

686        fn arbitrary_infix_expr_with_bindings_sized(num_variables: usize)
687            (infix_type_gen in arbitrary_infix_expr_type_gen(num_variables))
688            (values in collection::vec(infix_type_gen.clone().2, num_variables), infix_type_gen in Just(infix_type_gen))

694                }))
695                .collect();
696            (infix_expr, typ, bindings)
697        }

711    fn convert_infix_type_expr_to_expr(infix_expr: &Type) -> Expression {
712        let kind = match infix_expr {
713            Type::InfixExpr(lhs, op, rhs, _inversion) => {
714                let lhs = convert_infix_type_expr_to_expr(lhs);
715                let rhs = convert_infix_type_expr_to_expr(rhs);
716                let binary_op_kind = op.operator_to_binary_op_kind_helper();
717                let location = Location::dummy();
718                let operator = Located::from(location, binary_op_kind);
719                ExpressionKind::Infix(Box::new(InfixExpression { lhs, operator, rhs }))
720            }
721            Type::Constant(value, kind) => {
722                let integer_type_suffix = match kind {
723                    Kind::Numeric(typ) => {
724                        typ.as_integer_type_suffix().expect("ICE: unexpected numeric type {typ:?}")
725                    }
726                    kind => unimplemented!(
727                        "convert_infix_type_expr_to_expr: unexpected non-numeric kind: {kind}"
728                    ),
729                };
730                let literal = Literal::Integer(*value, Some(integer_type_suffix));
731                ExpressionKind::Literal(literal)
732            }
733            Type::TypeVariable(type_var) => unimplemented!(
734                "convert_infix_type_expr_to_expr: unexpected type variable {type_var:?}"
735            ),
736            infix_expr => unimplemented!("unexpected infix expression type: {infix_expr:?}"),
737        };
738        let location = Location::dummy();
739        Expression { kind, location }
740    }

742    // Convert a numeric Value to a `Type::Constant` or panic if `Value::to_signed_field` fails
743    // (expected to happen when it's not numeric)
744    fn numeric_value_to_type(value: Value) -> Type {
745        let kind_type = value.get_type();
746        let kind = Kind::numeric(kind_type.into_owned());
747        let value = value.to_signed_field().expect("ICE: numeric_value_to_type: expected a ");
748        Type::Constant(value, kind)
749    }

827        #[test]
828        fn compare_to_comptime(infix_type in arbitrary_infix_expr()) {
829            let (infix, typ) = infix_type;
830            let kind = Kind::numeric(typ.clone());
831            let expr = convert_infix_type_expr_to_expr(&infix);
832
833            let root = Path::new(".");
834            let file_manager = FileManager::new(root);
835            let parsed_files = HashMap::new();
836            let mut context = Context::from_ref_file_manager(
837                &file_manager,
838                &parsed_files,
839            );
840            let options = ElaboratorOptions {
841                debug_comptime_in_file: None,
842                enabled_unstable_features: &[],
843                disable_required_unstable_features: false,
844            };
845            let mut elaborator = Elaborator::from_context(
846                &mut context,
847                CrateId::Root(0),
848                options,
849            );
850            let (expr_id, expr_type) = elaborator.elaborate_expression(expr);
851
852            assert_eq!(elaborator.errors, vec![]);
853            assert_eq!(typ, expr_type);
854
855            let current_function = None;
856            let mut interpreter = Interpreter::new(
857                &mut elaborator,
858                current_function,
859            );
860
861            let result_value = interpreter.evaluate(expr_id);
862            assert_eq!(elaborator.errors, vec![]);
863
864            let result_type = match result_value {
865                Ok(result_value) => numeric_value_to_type(result_value),
866                Err(_) => Type::Error,
867            };
868
869            let infix = Type::CheckedCast {
870                from: Box::new(infix.clone()),
871                to: Box::new(infix)
872            };
873
874            let infix_canonicalized = infix.canonicalize();
875            prop_assert_eq!(infix_canonicalized.kind(), kind);
876            prop_assert!(infix.kind().unifies(&result_type.kind()));
877
878            let infix_canonicalized = match infix_canonicalized {
879                Type::CheckedCast { from, .. } => {
880                    *from
881                }
882                _ => infix_canonicalized,
883            };
884
885            // expect an error from the interpreter when canonicalize does not produce a constant
886            if matches!(infix_canonicalized, Type::Constant(..)) {
887                prop_assert_eq!(infix_canonicalized, result_type);
888            } else {
889                prop_assert!(matches!(result_type, Type::Error));
890            }
891        }

896        // TODO(https://github.com/noir-lang/noir/issues/11013): This test demonstrates that
897        // try_simplify_partial_constants() does not simplify expressions with FieldElements

907        let n = Type::TypeVariable(var_n);

911            0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x01, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
912            0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00,
913            0x00, 0x00, 0x00, 0x00,
914        ]));

916        let mul_expr = n.clone() * large_field.into();
917        let div_expr = mul_expr.clone() / SignedField::from(2u8).into();
