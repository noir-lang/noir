17enum FunctionCoercionResult {
18    NoCoercion,
19    Coerced(Type),
20    UnconstrainedMismatch(Type),
21}

23/// When unifying types we sometimes need to adjust the algorithm a bit.
24#[derive(Copy, Clone, Debug, Eq, PartialEq)]
25enum UnificationFlags {
26    /// Nothing special to do.
27    None,
28    /// If the right-hand side is `expr op constant`, don't try to move the constant to the left-hand side.
29    DoNotMoveConstantsOnTheRight,
30}

32impl Kind {
33    /// Unifies this kind with the other. Returns true on success
34    pub(crate) fn unifies(&self, other: &Kind) -> bool {
35        match (self, other) {
36            // Kind::Any unifies with everything
37            (Kind::Any, _) | (_, Kind::Any) => true,
38
39            // Kind::Normal unifies with Kind::Integer and Kind::IntegerOrField
40            (Kind::Normal, Kind::Integer | Kind::IntegerOrField)
41            | (Kind::Integer | Kind::IntegerOrField, Kind::Normal) => true,
42
43            // Kind::Integer unifies with Kind::IntegerOrField
44            (Kind::Integer | Kind::IntegerOrField, Kind::Integer | Kind::IntegerOrField) => true,
45
46            // Kind::IntegerOrField unifies with Kind::Numeric(_)
47            (Kind::IntegerOrField, Kind::Numeric(_typ))
48            | (Kind::Numeric(_typ), Kind::IntegerOrField) => true,
49
50            // Kind::Numeric unifies along its Type argument
51            (Kind::Numeric(lhs), Kind::Numeric(rhs)) => {
52                let mut bindings = TypeBindings::default();
53                let unifies = lhs.try_unify(rhs, &mut bindings).is_ok();
54                if unifies {
55                    Type::apply_type_bindings(bindings);
56                }
57                unifies
58            }
59
60            // everything unifies with itself
61            (lhs, rhs) => lhs == rhs,
62        }
63    }
64
65    pub(crate) fn unify(&self, other: &Kind) -> Result<(), UnificationError> {
66        if self.unifies(other) { Ok(()) } else { Err(UnificationError) }
67    }
68}

70impl Type {
71    /// Try to unify this type with another, setting any type variables found
72    /// equal to the other type in the process. When comparing types, unification
73    /// (including try_unify) are almost always preferred over Type::eq as unification
74    /// will correctly handle generic types.
75    pub fn unify(&self, expected: &Type) -> Result<(), UnificationError> {
76        let mut bindings = TypeBindings::default();
77
78        self.try_unify(expected, &mut bindings).map(|()| {
79            // Commit any type bindings on success
80            Self::apply_type_bindings(bindings);
81        })
82    }

84    /// `try_unify` is a bit of a misnomer since although errors are not committed,
85    /// any unified bindings are on success.
86    pub fn try_unify(
87        &self,
88        other: &Type,
89        bindings: &mut TypeBindings,
90    ) -> Result<(), UnificationError> {
91        self.try_unify_with_flags(other, UnificationFlags::None, bindings)
92    }

94    fn try_unify_with_flags(
95        &self,
96        other: &Type,
97        flags: UnificationFlags,
98        bindings: &mut TypeBindings,
99    ) -> Result<(), UnificationError> {
100        use Type::*;
101
102        let lhs = self.follow_bindings_shallow();
103        let rhs = other.follow_bindings_shallow();
104
105        let lhs = match lhs.as_ref() {
106            InfixExpr(..) => Cow::Owned(self.substitute(bindings).canonicalize()),
107            other => Cow::Borrowed(other),
108        };
109
110        let rhs = match rhs.as_ref() {
111            InfixExpr(..) => Cow::Owned(other.substitute(bindings).canonicalize()),
112            other => Cow::Borrowed(other),
113        };
114
115        match (lhs.as_ref(), rhs.as_ref()) {
116            (Error, _) | (_, Error) => Ok(()),
117
118            (Alias(alias, args), other) | (other, Alias(alias, args)) => {
119                let alias = alias.borrow().get_type(args);
120                alias.try_unify_with_flags(other, flags, bindings)
121            }
122
123            (TypeVariable(var), other) | (other, TypeVariable(var)) => match &*var.borrow() {
124                TypeBinding::Bound(typ) => {
125                    if typ.is_numeric_value() {
126                        other.try_unify_to_type_variable(var, flags, bindings, |bindings| {
127                            let only_integer = matches!(typ, Integer(..));
128                            other.try_bind_to_polymorphic_int(var, bindings, only_integer)
129                        })
130                    } else {
131                        other.try_unify_to_type_variable(var, flags, bindings, |bindings| {
132                            other.try_bind_to(var, bindings, typ.kind())
133                        })
134                    }
135                }
136                TypeBinding::Unbound(_id, Kind::IntegerOrField) => other
137                    .try_unify_to_type_variable(var, flags, bindings, |bindings| {
138                        let only_integer = false;
139                        other.try_bind_to_polymorphic_int(var, bindings, only_integer)
140                    }),
141                TypeBinding::Unbound(_id, Kind::Integer) => {
142                    other.try_unify_to_type_variable(var, flags, bindings, |bindings| {
143                        let only_integer = true;
144                        other.try_bind_to_polymorphic_int(var, bindings, only_integer)
145                    })
146                }
147                TypeBinding::Unbound(_id, type_var_kind) => {
148                    other.try_unify_to_type_variable(var, flags, bindings, |bindings| {
149                        other.try_bind_to(var, bindings, type_var_kind.clone())
150                    })
151                }
152            },
153
154            (Array(len_a, elem_a), Array(len_b, elem_b)) => {
155                len_a.try_unify(len_b, bindings)?;
156                elem_a.try_unify(elem_b, bindings)
157            }
158
159            (Vector(elem_a), Vector(elem_b)) => elem_a.try_unify(elem_b, bindings),
160
161            (String(len_a), String(len_b)) => len_a.try_unify(len_b, bindings),
162
163            (FmtString(len_a, elements_a), FmtString(len_b, elements_b)) => {
164                len_a.try_unify(len_b, bindings)?;
165                elements_a.try_unify(elements_b, bindings)
166            }
167
168            (Tuple(elements_a), Tuple(elements_b)) => {
169                if elements_a.len() != elements_b.len() {
170                    Err(UnificationError)
171                } else {
172                    for (a, b) in elements_a.iter().zip(elements_b) {
173                        a.try_unify(b, bindings)?;
174                    }
175                    Ok(())
176                }
177            }
178
179            // No recursive try_unify call for struct fields. Don't want
180            // to mutate shared type variables within struct definitions.
181            // This isn't possible currently but will be once noir gets generic types
182            (DataType(id_a, args_a), DataType(id_b, args_b)) => {
183                if id_a == id_b && args_a.len() == args_b.len() {
184                    for (a, b) in args_a.iter().zip(args_b) {
185                        a.try_unify(b, bindings)?;
186                    }
187                    Ok(())
188                } else {
189                    Err(UnificationError)
190                }
191            }
192
193            (CheckedCast { to, .. }, other) | (other, CheckedCast { to, .. }) => {
194                to.try_unify_with_flags(other, flags, bindings)
195            }
196
197            (NamedGeneric(types::NamedGeneric { type_var, .. }), other)
198            | (other, NamedGeneric(types::NamedGeneric { type_var, .. }))
199                if !type_var.borrow().is_unbound() =>
200            {
201                if let TypeBinding::Bound(link) = &*type_var.borrow() {
202                    link.try_unify_with_flags(other, flags, bindings)
203                } else {
204                    unreachable!("If guard ensures binding is bound")
205                }
206            }
207
208            (
209                NamedGeneric(types::NamedGeneric { type_var: binding_a, .. }),
210                NamedGeneric(types::NamedGeneric { type_var: binding_b, .. }),
211            ) => {
212                // Bound NamedGenerics are caught by the check above
213                assert!(binding_a.borrow().is_unbound());
214                assert!(binding_b.borrow().is_unbound());
215
216                if binding_a.0 == binding_b.0 {
217                    binding_a.kind().unify(&binding_b.kind())
218                } else {
219                    Err(UnificationError)
220                }
221            }
222
223            (
224                Function(params_a, ret_a, env_a, unconstrained_a),
225                Function(params_b, ret_b, env_b, unconstrained_b),
226            ) => {
227                if unconstrained_a == unconstrained_b && params_a.len() == params_b.len() {
228                    for (a, b) in params_a.iter().zip(params_b.iter()) {
229                        a.try_unify(b, bindings)?;
230                    }
231
232                    env_a.try_unify(env_b, bindings)?;
233                    ret_b.try_unify(ret_a, bindings)
234                } else {
235                    Err(UnificationError)
236                }
237            }
238
239            (Reference(elem_a, mutable_a), Reference(elem_b, mutable_b)) => {
240                if mutable_a == mutable_b {
241                    elem_a.try_unify(elem_b, bindings)
242                } else {
243                    Err(UnificationError)
244                }
245            }
246
247            (InfixExpr(lhs_a, op_a, rhs_a, _), InfixExpr(lhs_b, op_b, rhs_b, _)) => {
248                if op_a == op_b {
249                    // We need to preserve the original bindings since if syntactic equality
250                    // fails we fall back to other equality strategies.
251                    let mut new_bindings = bindings.clone();
252                    let lhs_result = lhs_a.try_unify(lhs_b, &mut new_bindings);
253                    let rhs_result = rhs_a.try_unify(rhs_b, &mut new_bindings);
254
255                    if lhs_result.is_ok() && rhs_result.is_ok() {
256                        *bindings = new_bindings;
257                        return Ok(());
258                    }
259                }
260
261                lhs.try_unify_by_isolating_an_unbound_type_variable(&rhs, bindings).or_else(|_| {
262                    lhs.try_unify_by_moving_single_constant_term(&rhs, flags, bindings)
263                })
264            }
265
266            (Constant(value, kind), other) | (other, Constant(value, kind)) => {
267                let dummy_location = Location::dummy();
268                let other = other.substitute(bindings);
269                if let Ok(other_value) = other.evaluate_to_signed_field(kind, dummy_location) {
270                    if *value == other_value && kind.unifies(&other.kind()) {
271                        Ok(())
272                    } else {
273                        Err(UnificationError)
274                    }
275                } else if let InfixExpr(lhs, op, rhs, _) = other {
276                    if let Some(inverse) = op.approx_inverse() {
277                        // Handle cases like `4 = a + b` by trying to solve to `a = 4 - b`
278                        let new_type = Type::inverted_infix_expr(
279                            Box::new(Constant(*value, kind.clone())),
280                            inverse,
281                            rhs.clone(),
282                        );
283
284                        new_type.try_unify(&lhs, bindings)?;
285                        Ok(())
286                    } else {
287                        Err(UnificationError)
288                    }
289                } else {
290                    Err(UnificationError)
291                }
292            }
293
294            (other_a, other_b) => {
295                if other_a == other_b {
296                    Ok(())
297                } else {
298                    Err(UnificationError)
299                }
300            }
301        }
302    }

304    /// Try to unify a type variable to `self`.
305    /// This is a helper function factored out from try_unify.
306    fn try_unify_to_type_variable(
307        &self,
308        type_variable: &TypeVariable,
309        flags: UnificationFlags,
310        bindings: &mut TypeBindings,
311
312        // Bind the type variable to a type. This is factored out since depending on the
313        // Kind, there are different methods to check whether the variable can
314        // bind to the given type or not.
315        bind_variable: impl FnOnce(&mut TypeBindings) -> Result<(), UnificationError>,
316    ) -> Result<(), UnificationError> {
317        match &*type_variable.borrow() {
318            // If it is already bound, unify against what it is bound to
319            TypeBinding::Bound(link) => link.try_unify_with_flags(self, flags, bindings),
320            TypeBinding::Unbound(id, _) => {
321                // We may have already "bound" this type variable in this call to
322                // try_unify, so check those bindings as well.
323                match bindings.get(id) {
324                    Some((_, kind, binding)) => {
325                        if !kind.unifies(&binding.kind()) {
326                            return Err(UnificationError);
327                        }
328                        binding.clone().try_unify_with_flags(self, flags, bindings)
329                    }
330
331                    // Otherwise, bind it
332                    None => bind_variable(bindings),
333                }
334            }
335        }
336    }

338    /// Try to unify the following equations:
339    /// - `A + rhs = other` -> `A = other - rhs`
340    /// - `A - rhs = other` -> `A = other + rhs`
341    /// - `lhs + B = other` -> `B = other - lhs`
342    /// - `lhs - B = other` -> `B = lhs - other`
343    /// - `other = A + rhs` -> `A = other - rhs`
344    /// - `other = A - rhs` -> `A = other + rhs`
345    /// - `other = lhs + B` -> `B = other - lhs`
346    /// - `other = lhs - B` -> `B = lhs - other`
347    pub(super) fn try_unify_by_isolating_an_unbound_type_variable(
348        &self,
349        other: &Type,
350        bindings: &mut TypeBindings,
351    ) -> Result<(), UnificationError> {
352        self.try_unify_by_isolating_an_unbound_type_variable_in_self(other, bindings).or_else(
353            |_| other.try_unify_by_isolating_an_unbound_type_variable_in_self(self, bindings),
354        )
355    }
356
357    /// Try to unify the following equations:
358    /// - `A + rhs = other` -> `A = other - rhs`
359    /// - `A - rhs = other` -> `A = other + rhs`
360    /// - `lhs + B = other` -> `B = other - lhs`
361    /// - `lhs - B = other` -> `B = lhs - other`
362    fn try_unify_by_isolating_an_unbound_type_variable_in_self(
363        &self,
364        other: &Type,
365        bindings: &mut TypeBindings,
366    ) -> Result<(), UnificationError> {
367        if let Type::InfixExpr(lhs_lhs, lhs_op, lhs_rhs, _) = self {
368            let lhs_op_inverse = lhs_op.inverse();
369
370            // Check if it's `A + rhs = other` or `A - rhs = other`
371            if let (Some(op_a_inverse), Type::TypeVariable(lhs_lhs_var)) =
372                (lhs_op_inverse, lhs_lhs.as_ref())
373                && lhs_lhs_var.1.borrow().is_unbound()
374            {
375                // We can say that `A = other - rhs` or `A = other + rhs` respectively
376                let new_rhs =
377                    Type::infix_expr(Box::new(other.clone()), op_a_inverse, lhs_rhs.clone());
378
379                let mut tmp_bindings = bindings.clone();
380                if lhs_lhs.try_unify(&new_rhs, &mut tmp_bindings).is_ok() {
381                    *bindings = tmp_bindings;
382                    return Ok(());
383                }
384            }
385
386            // Check if it's `lhs + B = other`
387            if let (BinaryTypeOperator::Addition, Type::TypeVariable(lhs_rhs_var)) =
388                (lhs_op, lhs_rhs.as_ref())
389                && lhs_rhs_var.1.borrow().is_unbound()
390            {
391                // We can say that `B = other - lhs`
392                let new_rhs = Type::inverted_infix_expr(
393                    Box::new(other.clone()),
394                    BinaryTypeOperator::Subtraction,
395                    lhs_lhs.clone(),
396                );
397
398                let mut tmp_bindings = bindings.clone();
399                if lhs_rhs.try_unify(&new_rhs, &mut tmp_bindings).is_ok() {
400                    *bindings = tmp_bindings;
401                    return Ok(());
402                }
403            }
404
405            // Check if it's `lhs - B = other`
406            if let (BinaryTypeOperator::Subtraction, Type::TypeVariable(lhs_rhs_var)) =
407                (lhs_op, lhs_rhs.as_ref())
408                && lhs_rhs_var.1.borrow().is_unbound()
409            {
410                // We can say that `B = lhs - other`
411                let new_rhs = Type::inverted_infix_expr(
412                    lhs_lhs.clone(),
413                    BinaryTypeOperator::Subtraction,
414                    Box::new(other.clone()),
415                );
416
417                let mut tmp_bindings = bindings.clone();
418                if lhs_rhs.try_unify(&new_rhs, &mut tmp_bindings).is_ok() {
419                    *bindings = tmp_bindings;
420                    return Ok(());
421                }
422            }
423        }
424
425        Err(UnificationError)
426    }

428    /// Try to unify the following equations:
429    /// - `(..a..) + 1 = (..b..)` -> `(..a..) = (..b..) - 1`
430    /// - `(..a..) - 1 = (..b..)` -> `(..a..) = (..b..) + 1`
431    /// - `(..a..) = (..b..) + 1` -> `(..b..) = (..a..) - 1`
432    /// - `(..a..) = (..b..) - 1` -> `(..b..) = (..a..) + 1`
433    fn try_unify_by_moving_single_constant_term(
434        &self,
435        other: &Type,
436        flags: UnificationFlags,
437        bindings: &mut TypeBindings,
438    ) -> Result<(), UnificationError> {
439        let result = self.try_unify_by_moving_single_constant_term_in_self(other, bindings);
440        if result.is_ok() {
441            return Ok(());
442        }
443
444        if flags != UnificationFlags::DoNotMoveConstantsOnTheRight {
445            let result = other.try_unify_by_moving_single_constant_term_in_self(self, bindings);
446            if result.is_ok() {
447                return Ok(());
448            }
449        }
450
451        Err(UnificationError)
452    }

454    /// Try to unify the following equations:
455    /// - `(..a..) + 1 = (..b..)` -> `(..a..) = (..b..) - 1`
456    /// - `(..a..) - 1 = (..b..)` -> `(..a..) = (..b..) + 1`
457    fn try_unify_by_moving_single_constant_term_in_self(
458        &self,
459        other: &Type,
460        bindings: &mut TypeBindings,
461    ) -> Result<(), UnificationError> {
462        if let Type::InfixExpr(lhs_lhs, lhs_op, lhs_rhs, _) = self
463            && let Some(lhs_op_inverse) = lhs_op.approx_inverse()
464        {
465            let kind = lhs_lhs.infix_kind(lhs_rhs);
466            let dummy_location = Location::dummy();
467            let lhs_rhs = lhs_rhs.substitute(bindings);
468            if let Ok(value) = lhs_rhs.evaluate_to_signed_field(&kind, dummy_location) {
469                let lhs_rhs = Box::new(Type::Constant(value, kind));
470                let new_rhs =
471                    Type::inverted_infix_expr(Box::new(other.clone()), lhs_op_inverse, lhs_rhs);
472
473                let mut tmp_bindings = bindings.clone();
474
475                // Since we are going to move a constant from one side to the other, we don't want
476                // to try moving the constant back to where it was because it would lead to infinite recursion.
477                let flags = UnificationFlags::DoNotMoveConstantsOnTheRight;
478                if lhs_lhs.try_unify_with_flags(&new_rhs, flags, &mut tmp_bindings).is_ok() {
479                    *bindings = tmp_bindings;
480                    return Ok(());
481                }
482            }
483        }
484
485        Err(UnificationError)
486    }

488    /// Similar to `unify` but if the check fails this will attempt to coerce the
489    /// argument to the target type. When this happens, the given expression is wrapped in
490    /// a new expression to convert its type. E.g. `array` -> `array.as_slice()`
491    ///
492    /// Currently the only type coercion in Noir is `[T; N]` into `[T]` via `.as_slice()`.
493    pub fn unify_with_coercions(
494        &self,
495        expected: &Type,
496        expression: ExprId,
497        location: Location,
498        interner: &mut NodeInterner,
499        errors: &mut Vec<CompilationError>,
500        make_error: impl FnOnce() -> CompilationError,
501    ) {
502        let mut bindings = TypeBindings::default();
503
504        if let Ok(()) = self.try_unify(expected, &mut bindings) {
505            Type::apply_type_bindings(bindings);
506            return;
507        }
508
509        if self.try_array_to_vector_coercion(expected, expression, interner) {
510            return;
511        }
512
513        if self.try_string_to_ctstring_coercion(expected, expression, interner) {
514            return;
515        }
516
517        if self.try_reference_coercion(expected) {
518            return;
519        }
520
521        // Try to coerce `fn (..) -> T` to `unconstrained fn (..) -> T`
522        match self.try_fn_to_unconstrained_fn_coercion(expected) {
523            FunctionCoercionResult::NoCoercion => errors.push(make_error()),
524            FunctionCoercionResult::Coerced(coerced_self) => {
525                coerced_self.unify_with_coercions(
526                    expected, expression, location, interner, errors, make_error,
527                );
528            }
529            FunctionCoercionResult::UnconstrainedMismatch(coerced_self) => {
530                errors.push(CompilationError::TypeError(TypeCheckError::UnsafeFn { location }));
531
532                coerced_self.unify_with_coercions(
533                    expected, expression, location, interner, errors, make_error,
534                );
535            }
536        }
537    }

539    /// If `self` and `expected` are function types, tries to coerce `self` to `expected`.
540    /// Returns `None` if no coercion can be applied, otherwise returns `self` coerced to `expected`.
541    fn try_fn_to_unconstrained_fn_coercion(&self, expected: &Type) -> FunctionCoercionResult {
542        // If `self` and `expected` are function types, `self` can be coerced to `expected`
543        // if `self` is unconstrained and `expected` is not. The other way around is an error, though.
544        if let (
545            Type::Function(params, ret, env, unconstrained_self),
546            Type::Function(_, _, _, unconstrained_expected),
547        ) = (self.follow_bindings(), expected.follow_bindings())
548        {
549            let coerced_type = Type::Function(params, ret, env, unconstrained_expected);
550
551            match (unconstrained_self, unconstrained_expected) {
552                (true, true) | (false, false) => FunctionCoercionResult::NoCoercion,
553                (false, true) => FunctionCoercionResult::Coerced(coerced_type),
554                (true, false) => FunctionCoercionResult::UnconstrainedMismatch(coerced_type),
555            }
556        } else {
557            FunctionCoercionResult::NoCoercion
558        }
559    }

561    /// Try to apply the array to vector coercion to this given type pair and expression.
562    /// If self can be converted to target this way, do so and return true to indicate success.
563    fn try_array_to_vector_coercion(
564        &self,
565        target: &Type,
566        expression: ExprId,
567        interner: &mut NodeInterner,
568    ) -> bool {
569        let this = self.follow_bindings();
570        let target = target.follow_bindings();
571
572        if let (Type::Array(_size, element1), Type::Vector(element2)) = (&this, &target) {
573            // We can only do the coercion if the `as_vector` method exists.
574            // This is usually true, but some tests don't have access to the standard library.
575            if let Some(as_slice) = interner.lookup_direct_method(&this, "as_slice", true) {
576                // Still have to ensure the element types match.
577                // Don't need to issue an error here if not, it will be done in unify_with_coercions
578                let mut bindings = TypeBindings::default();
579                if element1.try_unify(element2, &mut bindings).is_ok() {
580                    invoke_function_on_expression(expression, this, target, as_slice, interner);
581                    Self::apply_type_bindings(bindings);
582                    return true;
583                }
584            }
585        }
586        false
587    }

589    fn try_string_to_ctstring_coercion(
590        &self,
591        target: &Type,
592        expression: ExprId,
593        interner: &mut NodeInterner,
594    ) -> bool {
595        let this = self.follow_bindings();
596        let target = target.follow_bindings();
597
598        let Type::Quoted(QuotedType::CtString) = &target else {
599            return false;
600        };
601
602        match &this {
603            Type::String(..) | Type::FmtString(..) => {
604                // as_ctstring is defined as a trait method
605                for (func_id, trait_id) in interner.lookup_trait_methods(&this, "as_ctstring", true)
606                {
607                    // Look up the one that's in the standard library.
608                    let trait_ = interner.get_trait(trait_id);
609                    if trait_.crate_id.is_stdlib() && trait_.name.as_str() == "AsCtString" {
610                        invoke_function_on_expression(expression, this, target, func_id, interner);
611                        return true;
612                    }
613                }
614            }
615            _ => (),
616        }
617
618        false
619    }
620
621    /// Attempt to coerce reference types, returning true if possible.
622    pub(crate) fn try_reference_coercion(&self, target: &Type) -> bool {
623        let this = self.follow_bindings();
624        let target = target.follow_bindings();
625
626        match (&this, &target) {
627            // Coerce `&mut T` to `&T`, and `&T` to `&T`
628            (Type::Reference(this_elem, true), Type::Reference(target_elem, false))
629            | (Type::Reference(this_elem, false), Type::Reference(target_elem, false)) => {
630                // Still have to ensure the element types match.
631                // Don't need to issue an error here if not, it will be done in unify_with_coercions
632                let mut bindings = TypeBindings::default();
633                if this_elem.try_unify(target_elem, &mut bindings).is_ok() {
634                    Self::apply_type_bindings(bindings);
635                    return true;
636                }
637                false
638            }
639            _ => false,
640        }
641    }
642}
643
644/// Wraps a given `expression` in `expression.method()`
645fn invoke_function_on_expression(
646    expression: ExprId,
647    expression_type: Type,
648    target_type: Type,
649    method: crate::node_interner::FuncId,
650    interner: &mut NodeInterner,
651) {
652    let method_id = interner.function_definition_id(method);
653    let location = interner.expr_location(&expression);
654    let as_slice = HirExpression::Ident(HirIdent::non_trait_method(method_id, location), None);
655    let func_type = Type::Function(
656        vec![expression_type.clone()],
657        Box::new(target_type.clone()),
658        Box::new(Type::Unit),
659        false,
660    );
661    let func = interner.push_expr_full(as_slice, location, func_type);
662
663    // Copy the expression and give it a new ExprId. The old one
664    // will be mutated in place into a Call expression.
665    let argument = interner.expression(&expression);
666    let argument = interner.push_expr_full(argument, location, expression_type);
667
668    let arguments = vec![argument];
669    let is_macro_call = false;
670    let call = HirExpression::Call(HirCallExpression { func, arguments, location, is_macro_call });
671    interner.replace_expr(&expression, call);
672    interner.push_expr_type(expression, target_type);
673}

848    #[test]
849    fn does_not_recurse_forever_when_moving_single_constant_terms() {
850        let mut types = Types::new();
851        let mut bindings = TypeBindings::default();
852
853        // (A / B) - 1 = C * D
854        let (a, _) = types.type_variable();
855        let (b, _) = types.type_variable();
856        let (c, _) = types.type_variable();
857        let (d, _) = types.type_variable();
858        let one = constant(1);
859
860        let left = subtract(&divide(&a, &b), &one);
861        let right = multiply(&c, &d);
862
863        // This shouldn't unify. The idea is the compiler will try to do this:
864        //
865        // 1. (A / B) - 1 = C * D
866        // 2. (A / B) = (C * D) + 1
867        //
868        // It can't solve either A or B, so it will try to move the `1` back to
869        // the right side... except that we prevent that recursion.
870        assert!(left.try_unify(&right, &mut bindings).is_err());
871    }
