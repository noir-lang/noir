19/// A HirExpression is the result of an Expression in the AST undergoing
20/// name resolution. It is almost identical to the Expression AST node, but
21/// references other HIR nodes indirectly via IDs rather than directly via
22/// boxing. Variables in HirExpressions are tagged with their DefinitionId
23/// from the definition that refers to them so there is no ambiguity with names.
24#[derive(Debug, Clone)]
25pub enum HirExpression {
26    // The optional vec here is the optional list of generics
27    // provided by the turbofish operator, if it was used
28    Ident(HirIdent, Option<Vec<Type>>),
29    Literal(HirLiteral),
30    Block(HirBlockExpression),
31    Prefix(HirPrefixExpression),
32    Infix(HirInfixExpression),
33    Index(HirIndexExpression),
34    Constructor(HirConstructorExpression),
35    EnumConstructor(HirEnumConstructorExpression),
36    MemberAccess(HirMemberAccess),
37    Call(HirCallExpression),
38    Constrain(HirConstrainExpression),
39    Cast(HirCastExpression),
40    If(HirIfExpression),
41    Match(HirMatch),
42    Tuple(Vec<ExprId>),
43    Lambda(HirLambda),
44    Quote(Tokens),
45    Unquote(Tokens),
46    Unsafe(HirBlockExpression),
47    Error,
48}

264impl HirMethodReference {
265    /// Return the [FuncId] of a method if it's known.
266    ///
267    /// Returns `None` for trait methods don't have a know function definition.
268    pub fn func_id(&self, interner: &NodeInterner) -> Option<FuncId> {
269        match self {
270            HirMethodReference::FuncId(func_id) => Some(*func_id),
271            HirMethodReference::TraitItemId(HirTraitMethodReference { definition, .. }) => {
272                match &interner.try_definition(*definition)?.kind {
273                    DefinitionKind::Function(func_id) => Some(*func_id),
274                    _ => None,
275                }
276            }
277        }
278    }
279
280    /// Looks up definition of a function and its implementation kind (a normal function or a trait method),
281    /// and interns an identifier we can use to call the function.
282    pub fn into_function_id_and_name(
283        self,
284        object_type: Type,
285        generics: Option<Vec<Type>>,
286        location: Location,
287        interner: &mut NodeInterner,
288    ) -> (PushedExpr<HasLocation>, HirIdent) {
289        let (id, impl_kind) = match self {
290            HirMethodReference::FuncId(func_id) => {
291                (interner.function_definition_id(func_id), ImplKind::NotATraitMethod)
292            }
293            HirMethodReference::TraitItemId(HirTraitMethodReference {
294                definition,
295                trait_id,
296                trait_generics,
297                assumed,
298            }) => {
299                let constraint = TraitConstraint {
300                    typ: Self::find_self_type(definition, object_type, interner),
301                    trait_bound: ResolvedTraitBound { trait_id, trait_generics, location },
302                };
303
304                (definition, ImplKind::TraitItem(TraitItem { definition, constraint, assumed }))
305            }
306        };
307        let func_var = HirIdent { location, id, impl_kind };
308        let func = interner
309            .push_expr(HirExpression::Ident(func_var.clone(), generics))
310            .push_location(interner, location);
311        (func, func_var)
312    }
313
314    /// Find what the trait's Self type should be to construct the trait constraint, given the type
315    /// of its first argument. Assuming this is a method call, it should be sufficient to expect
316    /// Self is somewhere within the first argument. Usually the first parameter is either `Self`
317    /// or `&mut Self`.
318    ///
319    /// This defaults to `object_type` if there are any unexpected type errors (expected to be
320    /// issued elsewhere).
321    fn find_self_type(
322        definition: DefinitionId,
323        object_type: Type,
324        interner: &NodeInterner,
325    ) -> Type {
326        // Get the function type, e.g: `fn(&mut Self, i32) -> Field`
327        let function_type = interner.definition_type(definition);
328
329        // Instantiate, e.g: `fn(&mut ?Self, A, B) -> C`
330        let (instantiated, _bindings) = function_type.instantiate(interner);
331        let instantiated = instantiated.follow_bindings_shallow();
332
333        let first_parameter = match instantiated.as_ref() {
334            Type::Function(parameters, _, _, _) if !parameters.is_empty() => &parameters[0],
335            _ => return object_type,
336        };
337
338        // Match the first parameter, expected to be `?Self` or `&mut ?Self` to our object type. If
339        // successful, we expect `bindings` to have a single binding of `?Self` to either our
340        // object type or its element type, if it is a reference. Note that unifying here also
341        // handles the case where `object_type` is unbound but is discovered (here) that it must at
342        // least be a reference.
343        let mut bindings = TypeBindings::default();
344        if object_type.try_unify(first_parameter, &mut bindings).is_ok() && bindings.len() == 1 {
345            // We're expecting `Self` or `&mut Self`, where `Self` becomes a type variable after
346            // instantiation, so the value of `Self` for the constraint should be what it is bound
347            // to in `bindings`.
348            bindings.into_values().next().unwrap().2
349        } else {
350            object_type
351        }
352    }
353}

476#[derive(Debug, Clone, Hash, PartialEq, Eq, PartialOrd, Ord)]
477pub enum Constructor {
478    True,
479    False,
480    Unit,
481    Int(SignedField),
482    Tuple(Vec<Type>),
483    Variant(Type, usize),
484    Range(SignedField, SignedField),
485}
486
487impl Constructor {
488    pub fn variant_index(&self) -> usize {
489        match self {
490            Constructor::False
491            | Constructor::Int(_)
492            | Constructor::Unit
493            | Constructor::Tuple(_)
494            | Constructor::Range(_, _) => 0,
495            Constructor::True => 1,
496            Constructor::Variant(_, index) => *index,
497        }
498    }
499
500    /// True if this constructor constructs an enum value.
501    /// Enums contain a tag value and often have values to
502    /// unpack for each different variant index.
503    pub fn is_enum(&self) -> bool {
504        match self {
505            Constructor::Variant(typ, _) => match typ.follow_bindings_shallow().as_ref() {
506                Type::DataType(def, _) => def.borrow().is_enum(),
507                _ => false,
508            },
509            _ => false,
510        }
511    }
512
513    /// True if this constructor constructs a tuple or struct value.
514    /// Tuples or structs will still have values to unpack but do not
515    /// store a tag value internally.
516    pub fn is_tuple_or_struct(&self) -> bool {
517        match self {
518            Constructor::Tuple(_) => true,
519            Constructor::Variant(typ, _) => match typ.follow_bindings_shallow().as_ref() {
520                Type::DataType(def, _) => def.borrow().is_struct(),
521                _ => false,
522            },
523            _ => false,
524        }
525    }
526
527    /// Return all the constructors of this type from one constructor. Intended to be used
528    /// for error reporting in cases where there are at least 2 constructors.
529    pub(crate) fn all_constructors(&self) -> Vec<(Constructor, /*arg count:*/ usize)> {
530        match self {
531            Constructor::True | Constructor::False => {
532                vec![(Constructor::True, 0), (Constructor::False, 0)]
533            }
534            Constructor::Unit => vec![(Constructor::Unit, 0)],
535            Constructor::Tuple(args) => vec![(self.clone(), args.len())],
536            Constructor::Variant(typ, _) => {
537                let typ = typ.follow_bindings();
538                let Type::DataType(def, generics) = &typ else {
539                    unreachable!(
540                        "Constructor::Variant should have a DataType type, but found {typ:?}"
541                    );
542                };
543
544                let def_ref = def.borrow();
545                if let Some(variants) = def_ref.get_variants(generics) {
546                    vecmap(variants.into_iter().enumerate(), |(i, (_, fields))| {
547                        (Constructor::Variant(typ.clone(), i), fields.len())
548                    })
549                } else
550                /* def is a struct */
551                {
552                    let field_count = def_ref.fields_raw().map(|fields| fields.len()).unwrap_or(0);
553                    vec![(Constructor::Variant(typ.clone(), 0), field_count)]
554                }
555            }
556
557            // Nothing great to return for these
558            Constructor::Int(_) | Constructor::Range(..) => Vec::new(),
559        }
560    }
561}

563impl std::fmt::Display for Constructor {
564    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
565        match self {
566            Constructor::True => write!(f, "true"),
567            Constructor::False => write!(f, "false"),
568            Constructor::Unit => write!(f, "()"),
569            Constructor::Int(x) => write!(f, "{x}"),
570            // We already print the arguments of a constructor after this in the format of `(x, y)`.
571            // In that case it is already in the format of a tuple so there's nothing more we need
572            // to do here. This is implicitly assuming we never display a constructor without also
573            // displaying its arguments though.
574            Constructor::Tuple(_) => Ok(()),
575            Constructor::Variant(typ, variant_index) => {
576                if let Type::DataType(def, _) = typ {
577                    let def = def.borrow();
578                    if let Some(variant) = def.get_variant_as_written(*variant_index) {
579                        write!(f, "{}", variant.name)?;
580                    } else if def.is_struct() {
581                        write!(f, "{}", def.name)?;
582                    }
583                }
584                Ok(())
585            }
586            Constructor::Range(start, end) => write!(f, "{start} .. {end}"),
587        }
588    }
