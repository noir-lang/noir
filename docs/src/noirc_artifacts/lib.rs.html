4//! This module defines the structure of Nargo's different compilation artifacts.
5//!
6//! These artifacts are intended to remain independent of any applications being built on top of Noir.
7//! Should any projects require/desire a different artifact format, it's expected that they will write a transformer
8//! to generate them using these artifacts as a starting point.
9
10use serde::{Deserializer, Serializer, de::Visitor};

18/// Serialize `hash` as `String`, so that it doesn't get truncated in Javascript.
19fn serialize_hash<S>(hash: &u64, s: S) -> Result<S::Ok, S::Error>
20where
21    S: Serializer,
22{
23    s.serialize_str(&hash.to_string())
24}
25
26/// Deserialize `hash` from `String` in JSON.
27fn deserialize_hash<'de, D>(deserializer: D) -> Result<u64, D::Error>
28where
29    D: Deserializer<'de>,
30{
31    use serde::de::Error;
32
33    // Backwards compatible with `hash` serialized as a number.
34    struct StringOrU64;
35
36    impl Visitor<'_> for StringOrU64 {
37        type Value = u64;
38
39        fn expecting(&self, formatter: &mut std::fmt::Formatter) -> std::fmt::Result {
40            formatter.write_str("String or u64")
41        }
42
43        fn visit_str<E>(self, value: &str) -> Result<u64, E>
44        where
45            E: Error,
46        {
47            value.parse().map_err(E::custom)
48        }
49
50        fn visit_u64<E>(self, value: u64) -> Result<Self::Value, E> {
51            Ok(value)
52        }
53    }
54    deserializer.deserialize_any(StringOrU64)
55}
