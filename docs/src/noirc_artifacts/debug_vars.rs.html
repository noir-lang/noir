1use acvm::AcirField;
2use noirc_printable_type::{PrintableType, PrintableValue, decode_printable_value};
3use std::collections::HashMap;
4
5use crate::debug::{DebugFnId, DebugFunction, DebugInfo, DebugTypeId, DebugVarId, DebugVariable};
6
7#[derive(Debug, Default, Clone)]
8pub struct DebugVars<F> {
9    variables: HashMap<DebugVarId, DebugVariable>,
10    functions: HashMap<DebugFnId, DebugFunction>,
11    types: HashMap<DebugTypeId, PrintableType>,
12    frames: Vec<(DebugFnId, HashMap<DebugVarId, PrintableValue<F>>)>,
13}
14
15pub struct StackFrame<'a, F> {
16    pub function_name: &'a str,
17    pub function_params: Vec<&'a str>,
18    pub variables: Vec<(&'a str, &'a PrintableValue<F>, &'a PrintableType)>,
19}
20
21impl<F: AcirField> DebugVars<F> {
22    pub fn insert_debug_info(&mut self, info: &DebugInfo) {
23        self.variables.extend(info.variables.clone());
24        self.types.extend(info.types.clone());
25        self.functions.extend(info.functions.clone());
26    }
27
28    pub fn get_variables(&self) -> Vec<StackFrame<F>> {
29        self.frames.iter().map(|(fn_id, frame)| self.build_stack_frame(fn_id, frame)).collect()
30    }
31
32    pub fn current_stack_frame(&self) -> Option<StackFrame<F>> {
33        self.frames.last().map(|(fn_id, frame)| self.build_stack_frame(fn_id, frame))
34    }
35
36    fn lookup_var(&self, var_id: DebugVarId) -> Option<(&str, &PrintableType)> {
37        self.variables.get(&var_id).and_then(|debug_var| {
38            let printable_type = self.types.get(&debug_var.debug_type_id)?;
39            Some((debug_var.name.as_str(), printable_type))
40        })
41    }
42
43    fn build_stack_frame<'a>(
44        &'a self,
45        fn_id: &DebugFnId,
46        frame: &'a HashMap<DebugVarId, PrintableValue<F>>,
47    ) -> StackFrame<'a, F> {
48        let debug_fn = &self.functions.get(fn_id).expect("failed to find function metadata");
49
50        let params: Vec<&str> =
51            debug_fn.arg_names.iter().map(|arg_name| arg_name.as_str()).collect();
52        let vars: Vec<(&str, &PrintableValue<F>, &PrintableType)> = frame
53            .iter()
54            .filter_map(|(var_id, var_value)| {
55                self.lookup_var(*var_id).map(|(name, typ)| (name, var_value, typ))
56            })
57            .collect();
58
59        StackFrame {
60            function_name: debug_fn.name.as_str(),
61            function_params: params,
62            variables: vars,
63        }
64    }
65
66    pub fn assign_var(&mut self, var_id: DebugVarId, values: &[F]) {
67        let type_id = &self.variables.get(&var_id).unwrap().debug_type_id;
68        let printable_type = self.types.get(type_id).unwrap();
69
70        self.frames
71            .last_mut()
72            .expect("unexpected empty stack frames")
73            .1
74            .insert(var_id, decode_printable_value(&mut values.iter().copied(), printable_type));
75    }
76
77    pub fn assign_field(&mut self, var_id: DebugVarId, indexes: Vec<u32>, values: &[F]) {
78        let current_frame = &mut self.frames.last_mut().expect("unexpected empty stack frames").1;
79        let mut cursor: &mut PrintableValue<F> = current_frame
80            .get_mut(&var_id)
81            .unwrap_or_else(|| panic!("value unavailable for var_id {var_id:?}"));
82        let cursor_type_id = &self
83            .variables
84            .get(&var_id)
85            .unwrap_or_else(|| panic!("variable {var_id:?} not found"))
86            .debug_type_id;
87        let mut cursor_type = self
88            .types
89            .get(cursor_type_id)
90            .unwrap_or_else(|| panic!("type unavailable for type id {cursor_type_id:?}"));
91        for index in indexes.iter() {
92            (cursor, cursor_type) = match (cursor, cursor_type) {
93                (
94                    PrintableValue::Vec { array_elements, is_vector },
95                    PrintableType::Array { length, typ },
96                ) => {
97                    assert!(!*is_vector, "vector has array type");
98                    if *index >= *length {
99                        panic!("unexpected field index past array length")
100                    }
101                    if *length != array_elements.len() as u32 {
102                        panic!("type/array length mismatch")
103                    }
104                    (array_elements.get_mut(*index as usize).unwrap(), &*Box::leak(typ.clone()))
105                }
106                (
107                    PrintableValue::Vec { array_elements, is_vector },
108                    PrintableType::Vector { typ },
109                ) => {
110                    assert!(*is_vector, "vector doesn't have vector type");
111                    (array_elements.get_mut(*index as usize).unwrap(), &*Box::leak(typ.clone()))
112                }
113                (
114                    PrintableValue::Struct(field_map),
115                    PrintableType::Struct { name: _name, fields },
116                ) => {
117                    if *index as usize >= fields.len() {
118                        panic!("unexpected field index past struct field length")
119                    }
120                    let (key, typ) = fields.get(*index as usize).unwrap();
121                    (field_map.get_mut(key).unwrap(), typ)
122                }
123                (
124                    PrintableValue::Vec { array_elements, is_vector },
125                    PrintableType::Tuple { types },
126                ) => {
127                    assert!(!*is_vector, "vector has tuple type");
128                    if *index >= types.len() as u32 {
129                        panic!(
130                            "unexpected field index ({index}) past tuple length ({})",
131                            types.len()
132                        );
133                    }
134                    if types.len() != array_elements.len() {
135                        panic!("type/array length mismatch")
136                    }
137                    let typ = types.get(*index as usize).unwrap();
138                    (array_elements.get_mut(*index as usize).unwrap(), typ)
139                }
140                _ => {
141                    panic!("unexpected assign field of {cursor_type:?} type");
142                }
143            };
144        }
145        *cursor = decode_printable_value(&mut values.iter().copied(), cursor_type);
146    }
147
148    pub fn assign_deref(&mut self, _var_id: DebugVarId, _values: &[F]) {
149        unimplemented![]
150    }
151
152    pub fn get_type(&self, var_id: DebugVarId) -> Option<&PrintableType> {
153        self.variables.get(&var_id).and_then(|debug_var| self.types.get(&debug_var.debug_type_id))
154    }
155
156    pub fn drop_var(&mut self, var_id: DebugVarId) {
157        self.frames.last_mut().expect("unexpected empty stack frames").1.remove(&var_id);
158    }
159
160    pub fn push_fn(&mut self, fn_id: DebugFnId) {
161        self.frames.push((fn_id, HashMap::default()));
162    }
163
164    pub fn pop_fn(&mut self) {
165        self.frames.pop();
166    }
