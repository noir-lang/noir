35/// A Debug Artifact stores, for a given program, the debug info for every function
36/// along with a map of file Id to the source code so locations in debug info can be mapped to source code they point to.
37#[derive(Debug, Serialize, Deserialize)]
38pub struct DebugArtifact {
39    pub debug_symbols: Vec<DebugInfo>,
40    pub file_map: BTreeMap<FileId, DebugFile>,
41}
42
43impl DebugArtifact {
44    pub fn new(debug_symbols: Vec<DebugInfo>, file_manager: &FileManager) -> Self {
45        let mut file_map = BTreeMap::new();
46
47        let mut files_with_debug_symbols: BTreeSet<FileId> = debug_symbols
48            .iter()
49            .flat_map(|function_symbols| {
50                function_symbols.acir_locations.values().flat_map(|call_stack_id| {
51                    function_symbols
52                        .location_tree
53                        .get_call_stack(*call_stack_id)
54                        .into_iter()
55                        .map(|location| location.file)
56                })
57            })
58            .collect();
59
60        let files_with_brillig_debug_symbols: BTreeSet<FileId> = debug_symbols
61            .iter()
62            .flat_map(|function_symbols| {
63                function_symbols.brillig_locations.values().flat_map(|brillig_location_map| {
64                    brillig_location_map.values().flat_map(|call_stack_id| {
65                        function_symbols
66                            .location_tree
67                            .get_call_stack(*call_stack_id)
68                            .into_iter()
69                            .map(|location| location.file)
70                    })
71                })
72            })
73            .collect();
74
75        files_with_debug_symbols.extend(files_with_brillig_debug_symbols);
76
77        for file_id in files_with_debug_symbols {
78            let file_path = file_manager.path(file_id).expect("file should exist");
79            let file_source = file_manager.fetch_file(file_id).expect("file should exist");
80
81            file_map.insert(
82                file_id,
83                DebugFile { source: file_source.to_string(), path: file_path.to_path_buf() },
84            );
85        }
86
87        Self { debug_symbols, file_map }
88    }

121    /// Given a location, returns a Span relative to its line's
122    /// position in the file. This is useful when processing a file's
123    /// contents on a per-line-basis.
124    pub fn location_in_line(&self, location: Location) -> Result<Range<usize>, Error> {
125        let location_start = location.span.start() as usize;
126        let location_end = location.span.end() as usize;
127        let line_index = self.line_index(location.file, location_start)?;
128        let line_span = self.line_range(location.file, line_index)?;
129
130        let line_length =
131            if line_span.end > line_span.start { line_span.end - (line_span.start + 1) } else { 0 };
132        let start_in_line = location_start - line_span.start;
133
134        // The location might continue beyond the line,
135        // so we need a bounds check
136        let end_in_line = location_end - line_span.start;
137        let end_in_line = std::cmp::min(end_in_line, line_length);
138
139        Ok(Range { start: start_in_line, end: end_in_line })
140    }
141
142    /// Given a location, returns a Span relative to its last line's
143    /// position in the file. This is useful when processing a file's
144    /// contents on a per-line-basis.
145    pub fn location_in_end_line(&self, location: Location) -> Result<Range<usize>, Error> {
146        let end_line_index = self.location_end_line_index(location)?;
147        let line_span = self.line_range(location.file, end_line_index)?;
148        let location_end = location.span.end() as usize;
149        let end_in_line = location_end - line_span.start;
150        Ok(Range { start: 0, end: end_in_line })
151    }
152
153    /// Given a location, returns the last line index
154    /// of its file
155    pub fn last_line_index(&self, location: Location) -> Result<usize, Error> {
156        let source = self.source(location.file)?;
157        self.line_index(location.file, source.len())
158    }

200impl<'a> Files<'a> for DebugArtifact {
201    type FileId = FileId;
202    type Name = PathString;
203    type Source = &'a str;
204
205    fn name(&self, file_id: Self::FileId) -> Result<Self::Name, Error> {
206        let name = self.file_map.get(&file_id).ok_or(Error::FileMissing);
207        let name: Self::Name = name.map(|file| file.path.clone().into())?;
208
209        // See if we can make the file path a bit shorter/easier to read if it starts with the current directory
210        if let Ok(current_dir) = std::env::current_dir()
211            && let Ok(name_without_prefix) = name.clone().into_path_buf().strip_prefix(current_dir)
212        {
213            return Ok(PathString::from_path(name_without_prefix.to_path_buf()));
214        }
215
216        Ok(name)
217    }
218
219    fn source(&'a self, file_id: Self::FileId) -> Result<Self::Source, Error> {
220        self.file_map.get(&file_id).ok_or(Error::FileMissing).map(|file| file.source.as_ref())
221    }
222
223    fn line_index(&self, file_id: Self::FileId, byte_index: usize) -> Result<usize, Error> {
224        let file = self.file_map.get(&file_id).ok_or(Error::FileMissing)?;
225        SimpleFile::new(PathString::from(file.path.clone()), file.source.clone())
226            .line_index((), byte_index)
227    }
228
229    fn line_range(&self, file_id: Self::FileId, line_index: usize) -> Result<Range<usize>, Error> {
230        let file = self.file_map.get(&file_id).ok_or(Error::FileMissing)?;
231        SimpleFile::new(PathString::from(file.path.clone()), file.source.clone())
232            .line_range((), line_index)
233    }
234}

244impl ProgramDebugInfo {
245    pub fn serialize_compressed_base64_json<S>(
246        debug_info: &ProgramDebugInfo,
247        s: S,
248    ) -> Result<S::Ok, S::Error>
249    where
250        S: Serializer,
251    {
252        let json_str = serde_json::to_string(debug_info).map_err(S::Error::custom)?;
253
254        let mut encoder = DeflateEncoder::new(Vec::new(), Compression::default());
255        encoder.write_all(json_str.as_bytes()).map_err(S::Error::custom)?;
256        let compressed_data = encoder.finish().map_err(S::Error::custom)?;
257
258        let encoded_b64 = base64::prelude::BASE64_STANDARD.encode(compressed_data);
259        s.serialize_str(&encoded_b64)
260    }
261
262    pub fn deserialize_compressed_base64_json<'de, D>(
263        deserializer: D,
264    ) -> Result<ProgramDebugInfo, D::Error>
265    where
266        D: Deserializer<'de>,
267    {
268        let encoded_b64: String = Deserialize::deserialize(deserializer)?;
269
270        let compressed_data =
271            base64::prelude::BASE64_STANDARD.decode(encoded_b64).map_err(D::Error::custom)?;
272
273        let mut decoder = DeflateDecoder::new(&compressed_data[..]);
274        let mut decompressed_data = Vec::new();
275        decoder.read_to_end(&mut decompressed_data).map_err(D::Error::custom)?;
276
277        let json_str = String::from_utf8(decompressed_data).map_err(D::Error::custom)?;
278        serde_json::from_str(&json_str).map_err(D::Error::custom)
279    }
280}

323
324    /// Updates the locations map when the [`Circuit`][acvm::acir::circuit::Circuit] is modified.
325    ///
326    /// The [`OpcodeLocation`]s are generated with the ACIR, but passing the ACIR through a transformation step
327    /// renders the old `OpcodeLocation`s invalid. The AcirTransformationMap is able to map the old `OpcodeLocation` to the new ones.
328    /// Note: One old `OpcodeLocation` might have transformed into more than one new `OpcodeLocation`.
329    #[tracing::instrument(level = "trace", skip(self, update_map))]
330    pub fn update_acir(&mut self, update_map: AcirTransformationMap) {
331        let old_locations = mem::take(&mut self.acir_locations);
332
333        for (old_opcode_location, source_locations) in old_locations {
334            update_map.new_acir_locations(old_opcode_location).for_each(|new_opcode_location| {
335                self.acir_locations.insert(new_opcode_location, source_locations);
336            });
337        }
338    }
339
340    pub fn acir_opcode_location(&self, loc: &AcirOpcodeLocation) -> Option<Vec<Location>> {
341        self.acir_locations
342            .get(loc)
343            .map(|call_stack_id| self.location_tree.get_call_stack(*call_stack_id))
344    }
345
346    pub fn opcode_location(&self, loc: &OpcodeLocation) -> Option<Vec<Location>> {
347        match loc {
348            OpcodeLocation::Brillig { .. } => None, //TODO: need brillig function id in order to look into brillig_locations
349            OpcodeLocation::Acir(loc) => self.acir_opcode_location(&AcirOpcodeLocation::new(*loc)),
350        }
351    }
352}

360#[derive(Debug, Clone, Serialize, Deserialize, Default, Hash)]
361pub struct LocationTree {
362    pub locations: Vec<LocationNodeDebugInfo>,
363}
364
365impl LocationTree {
366    /// Construct a CallStack from a CallStackId
367    pub fn get_call_stack(&self, mut call_stack: CallStackId) -> CallStack {
368        let mut result = Vec::new();
369        while let Some(parent) = self.locations[call_stack.index()].parent {
370            result.push(self.locations[call_stack.index()].value);
371            call_stack = parent;
372        }
373        result.reverse();
374        result
375    }
376}

383                .locations
384                .iter()
385                .map(|node| LocationNodeDebugInfo { value: node.value, parent: node.parent })
386                .collect(),
387        }
388    }
