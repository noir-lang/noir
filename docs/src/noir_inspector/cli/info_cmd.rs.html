38/// Resolves the input file path for profiling.
39/// Priority:
40/// 1. Explicit --input-file argument
41/// 2. Prover.toml in program directory (parent of target/)
42/// 3. Prover.json in program directory
43fn resolve_input_file(
44    artifact_path: &Path,
45    explicit_input: Option<&PathBuf>,
46) -> eyre::Result<PathBuf> {
47    if let Some(input_path) = explicit_input {
48        if !input_path.exists() {
49            return Err(eyre::eyre!("Input file not found: {}", input_path.display()));
50        }
51        return Ok(input_path.clone());
52    }
53
54    // Artifact is typically at: <program_dir>/target/<name>.json
55    // Input files are at: <program_dir>/Prover.toml
56    let artifact_dir =
57        artifact_path.parent().ok_or_else(|| eyre::eyre!("Cannot determine artifact directory"))?;
58
59    let program_dir =
60        artifact_dir.parent().ok_or_else(|| eyre::eyre!("Cannot determine program directory"))?;
61
62    // Try Prover.toml first
63    let toml_path = program_dir.join("Prover.toml");
64    if toml_path.exists() {
65        return Ok(toml_path);
66    }
67
68    // Try Prover.json
69    let json_path = program_dir.join("Prover.json");
70    if json_path.exists() {
71        return Ok(json_path);
72    }
73
74    Err(eyre::eyre!(
75        "No input file found. Expected {} or {}",
76        toml_path.display(),
77        json_path.display()
78    ))
79}

81/// validates that a program artifact is pure Brillig
82/// 1. It has at least one Brillig function 2. All ACIR functions are wrappers with exactly 1 opcode each (the Brillig call)
83///    typically the result of compiling with `nargo export --force-brillig`
84fn validate_brillig_artifact(
85    program: &noirc_artifacts::program::ProgramArtifact,
86) -> eyre::Result<()> {
87    // check that it has brillig functions
88    if program.bytecode.unconstrained_functions.is_empty() {
89        return Err(eyre::eyre!(
90            "Cannot profile execution: program has no brillig functions.\n\
91             Compile with `nargo export --force-brillig` to generate pure brillig programs."
92        ));
93    }
94
95    let mut functions = program.bytecode.functions.iter();
96
97    match functions.len() {
98        1 => { /* ok */ }
99        0 => return Err(eyre::eyre!("Program contains no functions.")),
100        _ => {
101            return Err(eyre::eyre!(
102                "Cannot profile execution: program contains multiple functions.\n\
103                 Compile with `nargo export --force-brillig` to generate pure Brillig programs."
104            ));
105        }
106    }
107
108    let circuit = functions.next().unwrap();
109
110    // Pure brillig functions are represented as ACIR functions with exactly 1 opcode (the brillig call).
111    if circuit.opcodes.len() != 1 {
112        let function_name = &circuit.function_name;
113        return Err(eyre::eyre!(
114            "Cannot profile execution: function '{function_name}' contains ACIR constraints.\n\
115                 Compile with `nargo export --force-brillig` to generate pure Brillig programs.",
116        ));
117    }
118
119    Ok(())
120}

122/// Profile a single program's execution
123fn profile_program_execution(
124    program: noirc_artifacts::program::ProgramArtifact,
125    package_name: String,
126    input_file: &Path,
127) -> eyre::Result<ProgramInfo> {
128    // Validate that program is pure brillig before execution
129    validate_brillig_artifact(&program)?;
130
131    // Read inputs from file
132    let (inputs_map, _) = read_inputs_from_file(input_file, &program.abi)?;
133    let initial_witness = program.abi.encode(&inputs_map, None)?;
134
135    // Execute with profiling
136    let (_, profiling_samples) = execute_program_with_profiling(
137        &program.bytecode,
138        initial_witness,
139        &Bn254BlackBoxSolver,
140        &mut DefaultForeignCallBuilder::default().build(),
141    )
142    .map_err(|e| eyre::eyre!("Execution failed: {}", e))?;
143
144    // Build profiling report
145    Ok(ProgramInfo {
146        package_name,
147        functions: vec![FunctionInfo { name: "main".to_string(), opcodes: 0 }],
148        unconstrained_functions_opcodes: profiling_samples.len(),
149        unconstrained_functions: vec![FunctionInfo {
150            name: "main".to_string(),
151            opcodes: profiling_samples.len(),
152        }],
153    })
154}

156pub(crate) fn run(args: InfoCommand) -> eyre::Result<()> {
157    let artifact = Artifact::read_from_file(&args.artifact)?;
158
159    let programs = if args.profile_execution {
160        let input_file = resolve_input_file(&args.artifact, args.input_file.as_ref())?;
161
162        match artifact {
163            Artifact::Program(program) => {
164                let package_name = args
165                    .artifact
166                    .with_extension("")
167                    .file_name()
168                    .map(|s| s.to_string_lossy().to_string())
169                    .unwrap_or_else(|| "artifact".to_string());
170
171                vec![profile_program_execution(program, package_name, &input_file)?]
172            }
173            Artifact::Contract(_) => {
174                unreachable!("profile-execution conflicts with contract-fn argument");
175            }
176        }
177    } else {
178        match artifact {
179            Artifact::Program(program) => {
180                let package_name = args
181                    .artifact
182                    .with_extension("")
183                    .file_name()
184                    .map(|s| s.to_string_lossy().to_string())
185                    .unwrap_or_else(|| "artifact".to_string());
186
187                vec![count_opcodes_and_gates_in_program(program, package_name)]
188            }
189            Artifact::Contract(contract) => contract
190                .functions
191                .into_iter()
192                .filter(|f| args.contract_fn.as_ref().map(|n| *n == f.name).unwrap_or(true))
193                .map(|f| {
194                    let package_name = format!("{}::{}", contract.name, f.name);
195                    let program = f.into_compiled_program(
196                        contract.noir_version.clone(),
197                        contract.file_map.clone(),
198                    );
199                    count_opcodes_and_gates_in_program(program.into(), package_name)
200                })
201                .collect::<Vec<_>>(),
202        }
203    };
204
205    let info_report = InfoReport { programs };
206    show_info_report(info_report, args.json);
207
208    Ok(())
209}
