16/// Verifies an ECDSA signature over the Secp256r1 elliptic curve.
17///
18/// This function implements ECDSA signature verification on the Secp256r1 curve
19///
20/// # Parameters:
21///
22/// * `hashed_msg` - The 32-byte hash of the message that was signed
23/// * `public_key_x_bytes` - The x-coordinate of the public key (32 bytes, big-endian)
24/// * `public_key_y_bytes` - The y-coordinate of the public key (32 bytes, big-endian)
25/// * `signature` - The 64-byte signature in (r, s) format, where r and s are 32 bytes each
26///
27/// Returns `true` if the signature is valid, `false` otherwise.
28///
29/// The function returns an error if the following is not true:
30/// - The signature components `r` and `s` must be non-zero
31/// - The public key point must lie on the Secp256r1 curve
32///
33/// The function do not validate a signature if:
34/// - The signature is not "low S" normalized per BIP 0062 to prevent malleability
35///
36/// The function will panic if `hashed_msg >= p256::NistP256::ORDER`.
37/// According to ECDSA specification, the message hash leftmost bits should be truncated
38/// up to the curve order length, and then reduced modulo the curve order.
39pub(super) fn verify_signature(
40    hashed_msg: &[u8; 32],
41    public_key_x_bytes: &[u8; 32],
42    public_key_y_bytes: &[u8; 32],
43    signature: &[u8; 64],
44) -> Result<bool, BlackBoxResolutionError> {
45    // Convert the inputs into k256 data structures
46    let Ok(signature) = Signature::try_from(signature.as_slice()) else {
47        // Signature `r` and `s` are forbidden from being zero.
48        return Err(BlackBoxResolutionError::Failed(
49            BlackBoxFunc::EcdsaSecp256r1,
50            "Signature provided for ECDSA verification is zero".to_string(),
51        ));
52    };
53
54    let point = Sec1Point::<p256::NistP256>::from_affine_coordinates(
55        public_key_x_bytes.into(),
56        public_key_y_bytes.into(),
57        false,
58    );
59
60    let pubkey = PublicKey::from_sec1_point(&point);
61    if pubkey.is_none().into() {
62        // Public key must sit on the Secp256r1 curve.
63        return Err(BlackBoxResolutionError::Failed(
64            BlackBoxFunc::EcdsaSecp256k1,
65            "Invalid public key provided for ECDSA verification".to_string(),
66        ));
67    }
68    let pubkey = pubkey.unwrap();
69
70    // Convert the hashed message to a scalar.
71    // Per ECDSA specification (SEC 1, section 4.1.4), if `hashed_msg >= p256::NistP256::ORDER`,
72    // the message hash should be reduced modulo the curve order.
73    let z = <Scalar as Reduce<p256::U256>>::reduce(&p256::U256::from_be_slice(hashed_msg));
74
75    // Finished converting bytes into data structures
76
77    let r = signature.r();
78    let s = signature.s();
79
80    // Ensure signature is "low S" normalized ala BIP 0062
81    if s.is_high().into() {
82        log::warn!(
83            "Signature provided for ECDSA verification is not properly normalized (high S value)"
84        );
85        return Ok(false);
86    }
87
88    let s_inv = s.invert().unwrap();
89    let u1 = z * s_inv;
90    let u2 = *r * s_inv;
91
92    #[allow(non_snake_case)]
93    let R: AffinePoint = ((ProjectivePoint::GENERATOR * u1)
94        + (ProjectivePoint::from(*pubkey.as_affine()) * u2))
95        .to_affine();
96
97    match R.to_sec1_point(false).coordinates() {
98        Coordinates::Uncompressed { x, y: _ } => {
99            // The conversion from R.x to a scalar can fail if R.x >= curve_order (a possible but rare case).
100            // In this case, the signature is invalid per ECDSA specification, so we return false.
101            // The prover will handle this gracefully - it should generate a proof that fails verification.
102            Ok(Scalar::from_repr(*x).into_option().map_or_else(
103                || {
104                    log::warn!("Failed to convert R.x coordinate to scalar for ECDSA verification");
105                    false
106                },
107                |scalar| scalar == *r,
108            ))
109        }
110        Coordinates::Identity => Ok(false),
111        _ => Err(BlackBoxResolutionError::Failed(
112            BlackBoxFunc::EcdsaSecp256r1,
113            "Unexpected coordinate encoding".to_string(),
114        )),
115    }
116}
