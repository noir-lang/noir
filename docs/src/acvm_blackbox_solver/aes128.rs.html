1use crate::BlackBoxResolutionError;
2use libaes::Cipher; // cSpell:disable-line
3
4pub fn aes128_encrypt(
5    inputs: &[u8],
6    iv: [u8; 16],
7    key: [u8; 16],
8) -> Result<Vec<u8>, BlackBoxResolutionError> {
9    if inputs.len() % 16 != 0 {
10        return Err(BlackBoxResolutionError::Failed(
11            acir::BlackBoxFunc::AES128Encrypt,
12            "input length must be a multiple of 16".to_string(),
13        ));
14    }
15
16    let mut cipher = Cipher::new_128(&key);
17    // Disable auto padding - input must be block-aligned (multiple of 16)
18    // and output will be the same size as input.
19    // See: https://github.com/keepsimple1/libaes/blob/e45afaa1e9f248375e797a52eaf40eeb0ba8515a/tests/aes.rs#L325
20    cipher.set_auto_padding(false);
21    let encrypted = cipher.cbc_encrypt(&iv, inputs);
22
23    // We validate input.len() % 16 == 0 above, and libaes with auto_padding disabled only returns empty for non-block-aligned input.
24    if encrypted.is_empty() && !inputs.is_empty() {
25        unreachable!("AES encryption returned empty output for non-empty block-aligned input");
26    }
27
28    Ok(encrypted)
29}
