7pub fn bit_xor<F: AcirField>(lhs: F, rhs: F, num_bits: u32) -> F {
8    bitwise_op(lhs, rhs, num_bits, |lhs_byte, rhs_byte| lhs_byte ^ rhs_byte)

11/// Performs a bitwise operation on two field elements by treating them as byte arrays.
12///
13/// Both field elements are converted to little-endian byte arrays and masked to keep only
14/// the lowest `num_bits` bits. The provided operation `op` is then applied byte-by-byte,
15/// and the result is converted back to a field element.
16/// This function works for any `num_bits` value and does not assume it to be a multiple of 8.
17fn bitwise_op<F: AcirField>(lhs: F, rhs: F, num_bits: u32, op: fn(u8, u8) -> u8) -> F {
18    // We could explicitly expect `num_bits` to be a multiple of 8 as most backends assume bytes:
19    // assert!(num_bits % 8 == 0, "num_bits is not a multiple of 8, it is {num_bits}");
20
21    let lhs_bytes = mask_to_le_bytes(lhs, num_bits);
22    let rhs_bytes = mask_to_le_bytes(rhs, num_bits);
23
24    let and_byte_arr: Vec<_> =
25        lhs_bytes.into_iter().zip(rhs_bytes).map(|(left, right)| op(left, right)).collect();
26
27    F::from_le_bytes_reduce(&and_byte_arr)
28}
29
30// mask_to methods will not remove any bytes from the field
31// they are simply zeroed out
32// Whereas truncate_to will remove those bits and make the byte array smaller
33fn mask_to_le_bytes<F: AcirField>(field: F, num_bits: u32) -> Vec<u8> {
34    let mut bytes = field.to_le_bytes();
35    mask_vector_le(&mut bytes, num_bits as usize);
36    bytes
37}
38
39/// Mask a byte array in-place to only keep the lowest `num_bits`
40fn mask_vector_le(bytes: &mut [u8], num_bits: usize) {
41    let total_bits = bytes.len() * 8;
42    if num_bits >= total_bits {
43        // nothing to mask, all bits are used
44        return;
45    }
46
47    // Find which byte contains the last bit we want to keep
48    let array_mask_index = num_bits / 8;
49    // Find how many bits to keep in that byte (0-7)
50    let mask_power = num_bits % 8;
51
52    // If `mask_power` is non-zero, this keeps only the lower `mask_power` bits of the byte.
53    // If `mask_power` is zero (when `num_bits` is a multiple of 8), this zeros out the byte,
54    // which is correct since that byte is the first one beyond what we want to keep.
55    bytes[array_mask_index] &= 2u8.pow(mask_power as u32) - 1;
56
57    // Zero out all remaining bytes
58    for byte in &mut bytes[(array_mask_index + 1)..] {
59        *byte = 0;
60    }
61}

63#[cfg(test)]
64mod tests {
65    use acir::FieldElement;
66    use proptest::prelude::*;
67
68    use crate::{bit_and, bit_xor};
69
70    proptest! {
71        #[test]
72        fn matches_bitwise_and_on_u128s(x in 0..=u128::MAX, y in 0..=u128::MAX, bit_size in 128u32..) {
73            let x_as_field = FieldElement::from(x);
74            let y_as_field = FieldElement::from(y);
75
76            let x_and_y = x & y;
77            let x_and_y_as_field = bit_and(x_as_field, y_as_field, bit_size);
78
79            prop_assert_eq!(x_and_y_as_field, FieldElement::from(x_and_y), "AND on fields should match that on integers");
80        }
81
82        #[test]
83        fn matches_bitwise_xor_on_u128s(x in 0..=u128::MAX, y in 0..=u128::MAX, bit_size in 128u32..) {
84            let x_as_field = FieldElement::from(x);
85            let y_as_field = FieldElement::from(y);
86
87            let x_xor_y = x ^ y;
88            let x_xor_y_as_field = bit_xor(x_as_field, y_as_field, bit_size);
89
90            prop_assert_eq!(x_xor_y_as_field, FieldElement::from(x_xor_y), "XOR on fields should match that on integers");
91        }
92    }
