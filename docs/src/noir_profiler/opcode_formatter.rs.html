5fn format_blackbox_function<F: AcirField>(call: &BlackBoxFuncCall<F>) -> String {
6    match call {
7        BlackBoxFuncCall::AES128Encrypt { .. } => "aes128_encrypt".to_string(),
8        BlackBoxFuncCall::AND { .. } => "and".to_string(),
9        BlackBoxFuncCall::XOR { .. } => "xor".to_string(),
10        BlackBoxFuncCall::RANGE { .. } => "range".to_string(),
11        BlackBoxFuncCall::Blake2s { .. } => "blake2s".to_string(),
12        BlackBoxFuncCall::Blake3 { .. } => "blake3".to_string(),
13        BlackBoxFuncCall::EcdsaSecp256k1 { .. } => "ecdsa_secp256k1".to_string(),
14        BlackBoxFuncCall::EcdsaSecp256r1 { .. } => "ecdsa_secp256r1".to_string(),
15        BlackBoxFuncCall::MultiScalarMul { .. } => "multi_scalar_mul".to_string(),
16        BlackBoxFuncCall::EmbeddedCurveAdd { .. } => "embedded_curve_add".to_string(),
17        BlackBoxFuncCall::Keccakf1600 { .. } => "keccakf1600".to_string(),
18        BlackBoxFuncCall::RecursiveAggregation { .. } => "recursive_aggregation".to_string(),
19        BlackBoxFuncCall::Poseidon2Permutation { .. } => "poseidon2_permutation".to_string(),
20        BlackBoxFuncCall::Sha256Compression { .. } => "sha256_compression".to_string(),
21    }
22}

40fn format_acir_opcode_kind<F: AcirField>(opcode: &AcirOpcode<F>) -> String {
41    match opcode {
42        AcirOpcode::AssertZero(_) => "arithmetic".to_string(),
43        AcirOpcode::BlackBoxFuncCall(call) => {
44            format!("blackbox::{}", format_blackbox_function(call))
45        }
46        AcirOpcode::MemoryOp { .. } => "memory::op".to_string(),
47        AcirOpcode::MemoryInit { .. } => "memory::init".to_string(),
48        AcirOpcode::BrilligCall { id, .. } => format!("brillig_call({id})"),
49        AcirOpcode::Call { .. } => "acir_call".to_string(),
50    }
51}

53fn format_binary_field_op(op: &BinaryFieldOp) -> String {

59        BinaryFieldOp::IntegerDiv => "div".to_string(),
60        BinaryFieldOp::Equals => "eq".to_string(),
61        BinaryFieldOp::LessThan => "lt".to_string(),
62        BinaryFieldOp::LessThanEquals => "lte".to_string(),
63    }
64}

83fn format_brillig_opcode_kind<F>(opcode: &BrilligOpcode<F>) -> String {
84    match opcode {
85        BrilligOpcode::BinaryFieldOp { op, .. } => format!("field::{}", format_binary_field_op(op)),
86        BrilligOpcode::BinaryIntOp { op, bit_size, .. } => {
87            format!("{bit_size}::{}", format_binary_int(op))
88        }
89        BrilligOpcode::Not { .. } => "not".to_string(),
90        BrilligOpcode::BlackBox(func) => format!("blackbox::{}", format_blackbox_op(func)),
91        BrilligOpcode::Call { .. } => "call".to_string(),
92        BrilligOpcode::CalldataCopy { .. } => "calldata_copy".to_string(),
93        BrilligOpcode::Cast { .. } => "cast".to_string(),
94        BrilligOpcode::ConditionalMov { .. } => "cmov".to_string(),
95        BrilligOpcode::Const { .. } => "const".to_string(),
96        BrilligOpcode::IndirectConst { .. } => "iconst".to_string(),
97        BrilligOpcode::ForeignCall { function, .. } => format!("foreign_call({function})"),
98        BrilligOpcode::Jump { .. } => "jump".to_string(),
99        BrilligOpcode::JumpIf { .. } => "jump_if".to_string(),
100        BrilligOpcode::Load { .. } => "load".to_string(),
101        BrilligOpcode::Mov { .. } => "mov".to_string(),
102        BrilligOpcode::Return => "return".to_string(),
103        BrilligOpcode::Stop { .. } => "stop".to_string(),
104        BrilligOpcode::Store { .. } => "store".to_string(),
105        BrilligOpcode::Trap { .. } => "trap".to_string(),
106    }
107}

109pub(crate) fn format_acir_opcode<F: AcirField>(opcode: &AcirOpcode<F>) -> String {
110    format!("acir::{}", format_acir_opcode_kind(opcode))
