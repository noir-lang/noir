98impl FlamegraphGenerator for InfernoFlamegraphGenerator {
99    fn generate_flamegraph<'files, S: Sample>(
100        &self,
101        samples: Vec<S>,
102        debug_symbols: &DebugInfo,
103        files: &'files impl Files<'files, FileId = fm::FileId>,
104        artifact_name: &str,
105        function_name: &str,
106        output_path: &Path,
107    ) -> eyre::Result<()> {
108        let folded_lines = generate_folded_sorted_lines(samples, debug_symbols, files);
109        let flamegraph_file = std::fs::File::create(output_path)?;
110        let flamegraph_writer = BufWriter::new(flamegraph_file);
111
112        let mut options = Options::default();
113        options.hash = true;
114        options.deterministic = true;
115        options.title = format!("Artifact: {artifact_name}, Function: {function_name}");
116        options.frame_height = 24;
117        options.color_diffusion = true;
118        options.min_width = 0.0;
119        options.count_name = self.count_name.clone();
120        options.text_truncate_direction = TextTruncateDirection::Right;
121
122        from_lines(
123            &mut options,
124            folded_lines.iter().map(|as_string| as_string.as_str()),
125            flamegraph_writer,
126        )?;
127
128        Ok(())
129    }
130}

132fn generate_folded_sorted_lines<'files, S: Sample>(
133    samples: Vec<S>,
134    debug_symbols: &DebugInfo,
135    files: &'files impl Files<'files, FileId = fm::FileId>,
136) -> Vec<String> {
137    // Create a nested hashmap with the stack items, folding the gates for all the callsites that are equal
138    let mut folded_stack_items = BTreeMap::new();
139
140    let mut resolution_cache: HashMap<OpcodeLocation, Vec<String>> = HashMap::default();
141    for sample in samples {
142        let mut location_names = Vec::with_capacity(sample.call_stack().len());
143        for opcode_location in sample.call_stack() {
144            let callsite_labels = resolution_cache
145                .entry(*opcode_location)
146                .or_insert_with(|| {
147                    find_callsite_labels(
148                        debug_symbols,
149                        opcode_location,
150                        sample.brillig_function_id(),
151                        files,
152                    )
153                })
154                .clone();
155
156            location_names.extend(callsite_labels);
157        }
158
159        // We move `sample` by calling `sample.opcode()` so we want to fetch the sample count here.
160        let count = sample.count();
161
162        if let Some(opcode) = sample.opcode() {
163            location_names.push(opcode);
164        }
165
166        add_locations_to_folded_stack_items(&mut folded_stack_items, location_names, count);
167    }
168
169    to_folded_sorted_lines(&folded_stack_items, Default::default())
170}

172fn find_callsite_labels<'files>(
173    debug_symbols: &DebugInfo,
174    opcode_location: &OpcodeLocation,
175    brillig_function_id: Option<BrilligFunctionId>,
176    files: &'files impl Files<'files, FileId = fm::FileId>,
177) -> Vec<String> {
178    let mut procedure_id = None;
179    let source_locations = match opcode_location {
180        OpcodeLocation::Acir(idx) => {
181            debug_symbols.acir_opcode_location(&AcirOpcodeLocation::new(*idx)).unwrap_or_default()
182        }
183        OpcodeLocation::Brillig { .. } => {
184            if let (Some(brillig_function_id), Some(brillig_location)) =
185                (brillig_function_id, opcode_location.to_brillig_location())
186            {
187                let procedure_locs = debug_symbols.brillig_procedure_locs.get(&brillig_function_id);
188                if let Some(procedure_locs) = procedure_locs {
189                    for (procedure, range) in procedure_locs.iter() {
190                        if brillig_location.0 >= range.0 && brillig_location.0 <= range.1 {
191                            procedure_id = Some(*procedure);
192                            break;
193                        }
194                    }
195                }
196                let brillig_locations = debug_symbols.brillig_locations.get(&brillig_function_id);
197
198                if let Some(brillig_locations) = brillig_locations {
199                    brillig_locations
200                        .get(&brillig_location)
201                        .map(|call_stack| debug_symbols.location_tree.get_call_stack(*call_stack))
202                        .unwrap_or_default()
203                } else {
204                    vec![]
205                }
206            } else {
207                vec![]
208            }
209        }
210    };
211
212    let mut callsite_labels: Vec<_> = source_locations
213        .into_iter()
214        .map(|location| location_to_callsite_label(location, files))
215        .collect();
216
217    if let Some(procedure_id) = procedure_id {
218        callsite_labels.push(format!("procedure::{}", ProcedureId::from_debug_id(procedure_id)));
219    }
220
221    callsite_labels
222}

224fn location_to_callsite_label<'files>(
225    location: Location,
226    files: &'files impl Files<'files, FileId = fm::FileId>,
227) -> String {
228    let filename =
229        Path::new(&files.name(location.file).expect("should have a file path").to_string())
230            .file_name()
231            .map(|os_str| os_str.to_string_lossy().to_string())
232            .unwrap_or("invalid_path".to_string());
233    let source = files.source(location.file).expect("should have a file source");
234
235    let code_vector = source
236        .as_ref()
237        .chars()
238        .skip(location.span.start() as usize)
239        .take(location.span.end() as usize - location.span.start() as usize)
240        .collect::<String>();
241
242    // ";" is used for frame separation, and is not allowed by inferno
243    // Check code vector for ";" and replace it with 'GREEK QUESTION MARK' (U+037E)
244    let code_vector = code_vector.replace(';', "\u{037E}");
245
246    let (line, column) = line_and_column_from_span(source.as_ref(), &location.span);
247
248    format!("{filename}:{line}:{column}::{code_vector}")
249}

251fn add_locations_to_folded_stack_items(
252    stack_items: &mut BTreeMap<String, FoldedStackItem>,
253    locations: Vec<String>,
254    count: usize,
255) {
256    let mut child_map = stack_items;
257    for (index, location) in locations.iter().enumerate() {
258        let current_item = child_map.entry(location.clone()).or_default();
259
260        child_map = &mut current_item.nested_items;
261
262        if index == locations.len() - 1 {
263            current_item.total_samples += count;
264        }
265    }
266}

268/// Creates a vector of lines in the format that inferno expects from a nested hashmap of stack items
269/// The lines have to be sorted in the following way, exploring the graph in a depth-first manner:
270/// main 100
271/// main::foo 0
272/// main::foo::bar 200
273/// main::baz 27
274/// main::baz::qux 800
275fn to_folded_sorted_lines(
276    folded_stack_items: &BTreeMap<String, FoldedStackItem>,
277    parent_stacks: im::Vector<String>,
278) -> Vec<String> {
279    let mut result_vector = Vec::with_capacity(folded_stack_items.len());
280
281    for (location, folded_stack_item) in folded_stack_items.iter() {
282        if folded_stack_item.total_samples > 0 {
283            let frame_list: Vec<String> =
284                parent_stacks.iter().cloned().chain(std::iter::once(location.clone())).collect();
285            let line: String =
286                format!("{} {}", frame_list.join(";"), folded_stack_item.total_samples);
287
288            result_vector.push(line);
289        };
290
291        let mut new_parent_stacks = parent_stacks.clone();
292        new_parent_stacks.push_back(location.clone());
293        let child_lines =
294            to_folded_sorted_lines(&folded_stack_item.nested_items, new_parent_stacks);
295
296        result_vector.extend(child_lines);
297    }
298
299    result_vector
300}

321    fn find_spans_for(source: &str, needle: &str) -> Vec<Span> {
322        let mut spans = Vec::new();
323        let mut start = 0;
324        while let Some(start_idx) = source[start..].find(needle) {
325            let start_idx = start + start_idx;
326            let end_idx = start_idx + needle.len();
327            spans.push(Span::inclusive(start_idx as u32, end_idx as u32 - 1));
328            start = end_idx;
329        }
330        spans
331    }
