14/// Generates a flamegraph mapping backend opcodes to their associated locations in the source code.
15#[derive(Debug, Clone, Args)]
16pub(crate) struct GatesFlamegraphCommand {
17    /// The path to the artifact JSON file
18    #[clap(long, short)]
19    artifact_path: PathBuf,
20
21    /// Path to the Noir backend binary
22    #[clap(long, short)]
23    backend_path: PathBuf,
24
25    /// Command to get a gates report from the backend. Defaults to "gates"
26    #[clap(long, short = 'g', default_value = "gates")]
27    backend_gates_command: String,
28
29    /// Optional arguments for the backend gates command
30    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
31    backend_extra_args: Vec<String>,
32
33    /// The output folder for the flamegraph svg files
34    #[clap(long, short)]
35    output: PathBuf,
36
37    /// The output name for the flamegraph svg files
38    #[clap(long, short = 'f')]
39    output_filename: Option<String>,
40}
41
42pub(crate) fn run(args: GatesFlamegraphCommand) -> eyre::Result<()> {
43    run_with_provider(
44        &args.artifact_path,
45        &BackendGatesProvider {
46            backend_path: args.backend_path,
47            gates_command: args.backend_gates_command,
48            extra_args: args.backend_extra_args,
49        },
50        &InfernoFlamegraphGenerator { count_name: "gates".to_string() },
51        &args.output,
52        args.output_filename,
53    )
54}
55
56fn run_with_provider<Provider: GatesProvider, Generator: FlamegraphGenerator>(
57    artifact_path: &Path,
58    gates_provider: &Provider,
59    flamegraph_generator: &Generator,
60    output_path: &Path,
61    output_filename: Option<String>,
62) -> eyre::Result<()> {
63    let mut program =
64        read_program_from_file(artifact_path).context("Error reading program from file")?;
65
66    let backend_gates_response =
67        gates_provider.get_gates(artifact_path).context("Error querying backend for gates")?;
68
69    let bytecode = std::mem::take(&mut program.bytecode);
70    let debug_artifact: DebugArtifact = program.into();
71
72    let num_functions = bytecode.functions.len();
73    for (func_idx, (func_gates, circuit)) in
74        backend_gates_response.functions.into_iter().zip(bytecode.functions).enumerate()
75    {
76        // We can have repeated names if there are functions with the same name in different
77        // modules or functions that use generics. Thus, add the unique function index as a suffix.
78        let function_name = if num_functions > 1 {
79            format!("{}_{}", circuit.function_name.as_str(), func_idx)
80        } else {
81            circuit.function_name.to_owned()
82        };
83
84        println!(
85            "Opcode count: {}, Total gates by opcodes: {}, Circuit size: {}",
86            func_gates.acir_opcodes,
87            func_gates.gates_per_opcode.iter().sum::<usize>(),
88            func_gates.circuit_size
89        );
90
91        let samples = func_gates
92            .gates_per_opcode
93            .into_iter()
94            .zip(circuit.opcodes)
95            .enumerate()
96            .map(|(index, (gates, opcode))| CompilationSample {
97                opcode: Some(format_acir_opcode(&opcode)),
98                call_stack: vec![OpcodeLocation::Acir(index)],
99                count: gates,
100                brillig_function_id: None,
101            })
102            .collect();
103
104        let output_filename = if let Some(output_filename) = &output_filename {
105            format!("{output_filename}_{function_name}_gates.svg")
106        } else {
107            format!("{function_name}_gates.svg")
108        };
109        flamegraph_generator.generate_flamegraph(
110            samples,
111            &debug_artifact.debug_symbols[func_idx],
112            &debug_artifact,
113            artifact_path.to_str().unwrap(),
114            &function_name,
115            &Path::new(&output_path).join(Path::new(&output_filename)),
116        )?;
117    }
118
119    Ok(())
120}
