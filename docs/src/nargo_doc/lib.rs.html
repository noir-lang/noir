37/// Returns the root module in a crate.
38pub fn crate_module(
39    crate_id: CrateId,
40    crate_graph: &CrateGraph,
41    def_maps: &DefMaps,
42    interner: &NodeInterner,
43    file_manager: &FileManager,
44    item_id_to_converted_item: &mut HashMap<ItemId, ConvertedItem>,
45) -> (Module, Vec<BrokenLink>) {
46    let module = noirc_frontend::hir::printer::crate_to_module(crate_id, def_maps, interner);
47    let mut builder = DocItemBuilder::new(
48        interner,
49        crate_id,
50        crate_graph,
51        def_maps,
52        file_manager,
53        item_id_to_converted_item,
54    );
55    // Use convert_item here instead of convert_module so that the root module is tracked
56    // in `item_id_to_converted_item`.
57    let item = builder.convert_item(expand_items::Item::Module(module), ItemVisibility::Public);
58    let Item::Module(mut module) = item else {
59        panic!("Expected root item to be a module");
60    };
61    builder.process_module_reexports(&mut module);
62    (module, builder.broken_links)
63}

65struct DocItemBuilder<'a> {
66    interner: &'a NodeInterner,
67    crate_id: CrateId,
68    crate_graph: &'a CrateGraph,
69    def_maps: &'a DefMaps,
70    file_manager: &'a FileManager,
71    current_module_id: LocalModuleId,
72    current_type: Option<CurrentType>,
73    /// The minimum visibility of the current module. For example,
74    /// if the visibilities of parents modules are [pub, pub(crate), pub] then
75    /// this will be `pub(crate)`.
76    visibility: ItemVisibility,
77    /// Maps an ItemId to the item it converted to.
78    /// This is needed because if an item is publicly exported, but the item
79    /// isn't publicly visible (because its parent module is private) then we'll
80    /// include the item directly under the module that publicly exports it.
81    /// We do this by looking up the item in this map.
82    item_id_to_converted_item: &'a mut HashMap<ItemId, ConvertedItem>,
83    module_imports: HashMap<ModuleId, Vec<expand_items::Import>>,
84    /// Trait constraints in scope.
85    /// These are set when a trait or trait impl is visited.
86    trait_constraints: Vec<TraitConstraint>,
87    broken_links: Vec<BrokenLink>,
88    link_finder: LinkFinder,
89}

147pub struct ConvertedItem {
148    item: Item,
149    crate_id: CrateId,
150    // This is the maximum visibility the item has considering it's nested
151    // in modules. For example, in `pub(crate) mod a { pub struct B {} }`,
152    // struct B will have `pub(crate)` visibility here as it can't be publicly
153    // reached.
154    visibility: ItemVisibility,
155}

157impl DocItemBuilder<'_> {
158    fn convert_item(&mut self, item: expand_items::Item, visibility: ItemVisibility) -> Item {
159        let module_def_id = if let expand_items::Item::PrimitiveType(..) = item {
160            None
161        } else {
162            Some(item.module_def_id())
163        };
164        let converted_item = match item {
165            expand_items::Item::Module(module) => {
166                let old_visibility = self.visibility;
167                self.visibility = old_visibility.min(visibility);
168
169                let old_module_id = self.current_module_id;
170                self.current_module_id = module.id.local_id;
171
172                let module = self.convert_module(module);
173
174                self.visibility = old_visibility;
175                self.current_module_id = old_module_id;
176
177                Item::Module(module)
178            }
179            expand_items::Item::DataType(item_data_type) => {
180                let type_id = item_data_type.id;
181                self.current_type = Some(CurrentType::Type(type_id));
182                let shared_data_type = self.interner.get_type(type_id);
183                let data_type = shared_data_type.borrow();
184                let comments = self.doc_comments(ReferenceId::Type(type_id));
185                let mut has_private_fields = false;
186                let fields = if data_type.is_enum() {
187                    // Enums are shown as structs for now
188                    Vec::new()
189                } else {
190                    data_type
191                        .get_fields_as_written()
192                        .unwrap()
193                        .iter()
194                        .enumerate()
195                        .filter(|(_, field)| {
196                            if field.visibility == ItemVisibility::Public {
197                                true
198                            } else {
199                                has_private_fields = true;
200                                false
201                            }
202                        })
203                        .map(|(index, field)| {
204                            let comments =
205                                self.doc_comments(ReferenceId::StructMember(type_id, index));
206                            let r#type = self.convert_type(&field.typ);
207                            StructField { name: field.name.to_string(), r#type, comments }
208                        })
209                        .collect()
210                };
211                let generics = vecmap(&data_type.generics, |generic| self.convert_generic(generic));
212                let impls = vecmap(item_data_type.impls, |impl_| self.convert_impl(impl_));
213                let trait_impls =
214                    vecmap(item_data_type.trait_impls, |impl_| self.convert_trait_impl(impl_));
215                let id = get_type_id(type_id, self.interner);
216                self.current_type = None;
217                Item::Struct(Struct {
218                    id,
219                    name: data_type.name.to_string(),
220                    generics,
221                    fields,
222                    has_private_fields,
223                    impls,
224                    trait_impls,
225                    comments,
226                })
227            }
228            expand_items::Item::Trait(item_trait) => {
229                let trait_id = item_trait.id;
230                self.current_type = Some(CurrentType::Trait(trait_id));
231                let trait_ = self.interner.get_trait(trait_id);
232                let name = trait_.name.to_string();
233                let comments = self.doc_comments(ReferenceId::Trait(trait_id));
234                let generics = vecmap(&trait_.generics, |generic| self.convert_generic(generic));
235                let where_clause = vecmap(&trait_.where_clause, |constraint| {
236                    self.convert_trait_constraint(constraint)
237                });
238                self.trait_constraints = where_clause.clone();
239                let mut required_methods = Vec::new();
240                let mut provided_methods = Vec::new();
241                for func_id in &item_trait.methods {
242                    let has_body = self.interner.function(func_id).try_as_expr().is_some();
243                    let function = self.convert_function(*func_id);
244                    if has_body {
245                        provided_methods.push(function);
246                    } else {
247                        required_methods.push(function);
248                    }
249                }
250                self.trait_constraints.clear();
251                let trait_impls = vecmap(item_trait.trait_impls, |trait_impl| {
252                    self.convert_trait_impl(trait_impl)
253                });
254                let parents = vecmap(&trait_.trait_bounds, |bound| self.convert_trait_bound(bound));
255
256                let mut associated_types = Vec::new();
257                let mut associated_constants = Vec::new();
258
259                for associated_type in &trait_.associated_types {
260                    let name = associated_type.name.to_string();
261
262                    if let Kind::Numeric(numeric_type) = associated_type.kind() {
263                        let r#type = self.convert_type(&numeric_type);
264                        associated_constants.push(AssociatedConstant { name, r#type });
265                    } else {
266                        let bounds = if let Some(trait_bounds) =
267                            trait_.associated_type_bounds.get(associated_type.name.as_str())
268                        {
269                            vecmap(trait_bounds, |trait_bound| {
270                                self.convert_trait_bound(trait_bound)
271                            })
272                        } else {
273                            Vec::new()
274                        };
275                        associated_types.push(AssociatedType { name, bounds });
276                    }
277                }
278
279                let id = get_trait_id(trait_id, self.interner);
280
281                self.current_type = None;
282
283                Item::Trait(Trait {
284                    id,
285                    name,
286                    generics,
287                    bounds: parents,
288                    where_clause,
289                    associated_types,
290                    associated_constants,
291                    comments,
292                    required_methods,
293                    provided_methods,
294                    trait_impls,
295                })
296            }
297            expand_items::Item::TypeAlias(type_alias_id) => {
298                let type_alias = self.interner.get_type_alias(type_alias_id);
299                let type_alias = type_alias.borrow();
300                let name = type_alias.name.to_string();
301                let r#type = self.convert_type(&type_alias.typ);
302                let comments = self.doc_comments(ReferenceId::Alias(type_alias_id));
303                let generics =
304                    vecmap(&type_alias.generics, |generic| self.convert_generic(generic));
305                let id = get_type_alias_id(type_alias_id, self.interner);
306                Item::TypeAlias(TypeAlias { id, name, comments, r#type, generics })
307            }
308            expand_items::Item::PrimitiveType(primitive_type) => {
309                let kind = match &primitive_type.typ {
310                    noirc_frontend::Type::String(..) => PrimitiveTypeKind::Str,
311                    noirc_frontend::Type::FmtString(..) => PrimitiveTypeKind::Fmtstr,
312                    noirc_frontend::Type::Array(..) => PrimitiveTypeKind::Array,
313                    noirc_frontend::Type::Vector(..) => PrimitiveTypeKind::Vector,
314                    _ => {
315                        let Type::Primitive(kind) = self.convert_type(&primitive_type.typ) else {
316                            panic!("Expected primitive type");
317                        };
318                        kind
319                    }
320                };
321                self.current_type = Some(CurrentType::PrimitiveType(kind));
322                let impls = vecmap(primitive_type.impls, |impl_| self.convert_impl(impl_));
323                let trait_impls =
324                    vecmap(primitive_type.trait_impls, |impl_| self.convert_trait_impl(impl_));
325                let comments =
326                    self.interner.primitive_docs.get(&kind.to_string()).cloned().map(|comments| {
327                        vecmap(comments, |comment| comment.contents).join("\n").trim().to_string()
328                    });
329                let comments = comments.map(|comments| {
330                    let links = self.find_links_in_comments(&comments);
331                    (comments, links)
332                });
333                self.current_type = None;
334                Item::PrimitiveType(PrimitiveType { kind, impls, trait_impls, comments })
335            }
336            expand_items::Item::Global(global_id) => {
337                let global_info = self.interner.get_global(global_id);
338                let definition_id = global_info.definition_id;
339                let definition = self.interner.definition(definition_id);
340                let comptime = matches!(
341                    self.interner.get_global_let_statement(global_id),
342                    Some(HirLetStatement { comptime: true, .. })
343                );
344                let mutable = definition.mutable;
345                let name = global_info.ident.to_string();
346                let typ = self.interner.definition_type(definition_id);
347                let r#type = self.convert_type(&typ);
348                let comments = self.doc_comments(ReferenceId::Global(global_id));
349                let id = get_global_id(global_id, self.interner);
350                Item::Global(Global { id, name, comments, comptime, mutable, r#type })
351            }
352            expand_items::Item::Function(func_id) => Item::Function(self.convert_function(func_id)),
353        };
354        if let Some(module_def_id) = module_def_id {
355            let id = get_module_def_id(module_def_id, self.interner);
356            self.item_id_to_converted_item.insert(
357                id,
358                ConvertedItem {
359                    item: converted_item.clone(),
360                    crate_id: self.crate_id,
361                    visibility: self.visibility,
362                },
363            );
364        }
365        converted_item
366    }

368    fn convert_module(&mut self, module: expand_items::Module) -> Module {
369        let name = module.name.unwrap_or_default();
370        let comments = self.doc_comments(ReferenceId::Module(module.id));
371        let items = module
372            .items
373            .into_iter()
374            .map(|(visibility, item)| (visibility, self.convert_item(item, visibility)))
375            .collect();
376        self.module_imports.insert(module.id, module.imports);
377        let is_contract = module.is_contract;
378        let id = get_module_id(module.id, self.interner);
379        Module { id, module_id: module.id, name, comments, items, is_contract }
380    }

382    fn convert_impl(&mut self, impl_: expand_items::Impl) -> Impl {
383        let generics = vecmap(impl_.generics, |(name, kind)| {
384            let numeric = self.kind_to_numeric(kind);
385            Generic { name, numeric }
386        });
387        let r#type = self.convert_type(&impl_.typ);
388        let methods = impl_
389            .methods
390            .into_iter()
391            .filter(|(visibility, _)| visibility == &ItemVisibility::Public)
392            .map(|(_, func_id)| self.convert_function(func_id))
393            .collect();
394        Impl { generics, r#type, methods }
395    }

397    fn convert_trait_impl(&mut self, item_trait_impl: expand_items::TraitImpl) -> TraitImpl {
398        let trait_impl_id = item_trait_impl.id;
399
400        let trait_impl = self.interner.get_trait_implementation(trait_impl_id);
401        let trait_impl = trait_impl.borrow();
402
403        let generics = vecmap(item_trait_impl.generics, |(name, kind)| {
404            let numeric = self.kind_to_numeric(kind);
405            Generic { name, numeric }
406        });
407        let where_clause = vecmap(&trait_impl.where_clause, |constraint| {
408            self.convert_trait_constraint(constraint)
409        });
410        self.trait_constraints = where_clause.clone();
411        let methods = vecmap(item_trait_impl.methods, |func_id| self.convert_function(func_id));
412        self.trait_constraints.clear();
413
414        let trait_ = self.interner.get_trait(trait_impl.trait_id);
415        let trait_name = trait_.name.to_string();
416        let trait_id = get_trait_id(trait_.id, self.interner);
417        let trait_generics = vecmap(&trait_impl.trait_generics, |typ| self.convert_type(typ));
418        let r#type = self.convert_type(&trait_impl.typ);
419        TraitImpl { r#type, generics, methods, trait_id, trait_name, trait_generics, where_clause }
420    }

422    fn convert_trait_constraint(
423        &self,
424        constraint: &noirc_frontend::hir_def::traits::TraitConstraint,
425    ) -> TraitConstraint {
426        let r#type = self.convert_type(&constraint.typ);
427        let bound = self.convert_trait_bound(&constraint.trait_bound);
428        TraitConstraint { r#type, bound }
429    }

431    fn convert_trait_bound(&self, trait_bound: &ResolvedTraitBound) -> TraitBound {
432        let trait_ = self.interner.get_trait(trait_bound.trait_id);
433        let trait_name = trait_.name.to_string();
434        let trait_id = get_trait_id(trait_.id, self.interner);
435        let (ordered_generics, named_generics) =

437        TraitBound { trait_id, trait_name, ordered_generics, named_generics }
438    }

440    fn convert_type(&self, typ: &noirc_frontend::Type) -> Type {
441        match typ {
442            noirc_frontend::Type::Unit => Type::Unit,
443            noirc_frontend::Type::FieldElement => Type::Primitive(PrimitiveTypeKind::Field),
444            noirc_frontend::Type::Bool => Type::Primitive(PrimitiveTypeKind::Bool),
445            noirc_frontend::Type::Integer(signedness, bit_size) => match signedness {
446                Signedness::Unsigned => match bit_size {
447                    IntegerBitSize::One => Type::Primitive(PrimitiveTypeKind::U1),
448                    IntegerBitSize::Eight => Type::Primitive(PrimitiveTypeKind::U8),
449                    IntegerBitSize::Sixteen => Type::Primitive(PrimitiveTypeKind::U16),
450                    IntegerBitSize::ThirtyTwo => Type::Primitive(PrimitiveTypeKind::U32),
451                    IntegerBitSize::SixtyFour => Type::Primitive(PrimitiveTypeKind::U64),
452                    IntegerBitSize::HundredTwentyEight => Type::Primitive(PrimitiveTypeKind::U128),
453                },
454                Signedness::Signed => match bit_size {
455                    IntegerBitSize::One => panic!("There is no signed 1-bit integer"),
456                    IntegerBitSize::Eight => Type::Primitive(PrimitiveTypeKind::I8),
457                    IntegerBitSize::Sixteen => Type::Primitive(PrimitiveTypeKind::I16),
458                    IntegerBitSize::ThirtyTwo => Type::Primitive(PrimitiveTypeKind::I32),
459                    IntegerBitSize::SixtyFour => Type::Primitive(PrimitiveTypeKind::I64),
460                    IntegerBitSize::HundredTwentyEight => {
461                        panic!("There is no signed 128-bit integer")
462                    }
463                },
464            },
465            noirc_frontend::Type::Quoted(quoted) => match quoted {
466                noirc_frontend::QuotedType::Expr => Type::Primitive(PrimitiveTypeKind::Expr),
467                noirc_frontend::QuotedType::Quoted => Type::Primitive(PrimitiveTypeKind::Quoted),
468                noirc_frontend::QuotedType::Type => Type::Primitive(PrimitiveTypeKind::Type),
469                noirc_frontend::QuotedType::TypedExpr => {
470                    Type::Primitive(PrimitiveTypeKind::TypedExpr)
471                }
472                noirc_frontend::QuotedType::TypeDefinition => {
473                    Type::Primitive(PrimitiveTypeKind::TypeDefinition)
474                }
475                noirc_frontend::QuotedType::TraitConstraint => {
476                    Type::Primitive(PrimitiveTypeKind::TraitConstraint)
477                }
478                noirc_frontend::QuotedType::TraitDefinition => {
479                    Type::Primitive(PrimitiveTypeKind::TraitDefinition)
480                }
481                noirc_frontend::QuotedType::TraitImpl => {
482                    Type::Primitive(PrimitiveTypeKind::TraitImpl)
483                }
484                noirc_frontend::QuotedType::UnresolvedType => {
485                    Type::Primitive(PrimitiveTypeKind::UnresolvedType)
486                }
487                noirc_frontend::QuotedType::FunctionDefinition => {
488                    Type::Primitive(PrimitiveTypeKind::FunctionDefinition)
489                }
490                noirc_frontend::QuotedType::Module => Type::Primitive(PrimitiveTypeKind::Module),
491                noirc_frontend::QuotedType::CtString => {
492                    Type::Primitive(PrimitiveTypeKind::CtString)
493                }
494            },
495            noirc_frontend::Type::Array(length, element) => Type::Array {
496                length: Box::new(self.convert_type(length)),
497                element: Box::new(self.convert_type(element)),
498            },
499            noirc_frontend::Type::Vector(element) => {
500                Type::Vector { element: Box::new(self.convert_type(element)) }
501            }
502            noirc_frontend::Type::String(length) => {
503                Type::String { length: Box::new(self.convert_type(length)) }
504            }
505            noirc_frontend::Type::FmtString(length, element) => Type::FmtString {
506                length: Box::new(self.convert_type(length)),
507                element: Box::new(self.convert_type(element)),
508            },
509            noirc_frontend::Type::Tuple(types) => {
510                Type::Tuple(vecmap(types, |typ| self.convert_type(typ)))
511            }
512            noirc_frontend::Type::DataType(data_type, generics) => {
513                let data_type = data_type.borrow();
514                // Enums are shown as structs for now
515                let id = get_type_id(data_type.id, self.interner);
516                let name = data_type.name.to_string();
517                let generics = vecmap(generics, |typ| self.convert_type(typ));
518                Type::Struct { id, name, generics }
519            }
520            noirc_frontend::Type::Alias(type_alias, generics) => {
521                let type_alias = type_alias.borrow();
522                let id = get_type_alias_id(type_alias.id, self.interner);
523                let name = type_alias.name.to_string();
524                let generics = vecmap(generics, |typ| self.convert_type(typ));
525                Type::TypeAlias { id, name, generics }
526            }
527            noirc_frontend::Type::TypeVariable(type_var) => {
528                if let TypeBinding::Bound(typ) = &*type_var.borrow() {
529                    self.convert_type(typ)
530                } else {
531                    Type::Generic("_".to_string())
532                }
533            }
534            noirc_frontend::Type::TraitAsType(trait_id, _, trait_generics) => {
535                let trait_ = self.interner.get_trait(*trait_id);
536                let trait_id = get_trait_id(trait_.id, self.interner);
537                let trait_name = trait_.name.to_string();
538                let (ordered_generics, named_generics) =
539                    self.convert_trait_generics(trait_generics);
540                Type::TraitAsType { trait_id, trait_name, ordered_generics, named_generics }
541            }
542            noirc_frontend::Type::NamedGeneric(NamedGeneric { name, type_var, .. }) => {
543                if let TypeBinding::Bound(typ) = &*type_var.borrow() {
544                    self.convert_type(typ)
545                } else {
546                    Type::Generic(name.to_string())
547                }
548            }
549            noirc_frontend::Type::CheckedCast { from: _, to } => self.convert_type(to),
550            noirc_frontend::Type::Function(args, return_type, env, unconstrained) => {
551                Type::Function {
552                    params: vecmap(args, |typ| self.convert_type(typ)),
553                    return_type: Box::new(self.convert_type(return_type)),
554                    env: Box::new(self.convert_type(env)),
555                    unconstrained: *unconstrained,
556                }
557            }
558            noirc_frontend::Type::Reference(typ, mutable) => {
559                Type::Reference { r#type: Box::new(self.convert_type(typ)), mutable: *mutable }
560            }
561            noirc_frontend::Type::Constant(signed_field, _kind) => {
562                Type::Constant(signed_field.to_string())
563            }
564            noirc_frontend::Type::InfixExpr(lhs, operator, rhs, _) => Type::InfixExpr {
565                lhs: Box::new(self.convert_type(lhs)),
566                operator: operator.to_string(),
567                rhs: Box::new(self.convert_type(rhs)),
568            },
569            noirc_frontend::Type::Forall(..) => {
570                panic!("Should not need to print Type::Forall")
571            }
572            noirc_frontend::Type::Error => {
573                panic!("Should not need to print Type::Error")
574            }
575        }
576    }

578    fn convert_trait_generics(
579        &self,
580        trait_generics: &noirc_frontend::hir::type_check::generics::TraitGenerics,
581    ) -> (Vec<Type>, std::collections::BTreeMap<String, Type>) {
582        let ordered_generics = vecmap(&trait_generics.ordered, |typ| self.convert_type(typ));
583        let named_generics = trait_generics
584            .named
585            .iter()
586            .filter(|named_type| {
587                if let noirc_frontend::Type::NamedGeneric(NamedGeneric { implicit: true, .. }) =
588                    &named_type.typ
589                {
590                    return false;
591                }
592                true
593            })
594            .map(|named_type| (named_type.name.to_string(), self.convert_type(&named_type.typ)))
595            .collect();
596        (ordered_generics, named_generics)
597    }

599    fn convert_function(&mut self, func_id: FuncId) -> Function {
600        let modifiers = self.interner.function_modifiers(&func_id);
601        let func_meta = self.interner.function_meta(&func_id);
602        let unconstrained = modifiers.is_unconstrained;
603        let comptime = modifiers.is_comptime;
604        let name = modifiers.name.to_string();
605        let comments = self.doc_comments(ReferenceId::Function(func_id));
606        let generics = vecmap(&func_meta.direct_generics, |generic| self.convert_generic(generic));
607        let params = vecmap(func_meta.parameters.iter(), |(pattern, typ, _visibility)| {
608            let is_self = self.pattern_is_self(pattern);
609
610            // `&mut self` is represented as a mutable reference type, not as a mutable pattern
611            let mut mut_ref = false;
612            let name = if is_self && matches!(typ, noirc_frontend::Type::Reference(..)) {
613                mut_ref = true;
614                "self".to_string()
615            } else {
616                self.pattern_to_string(pattern)
617            };
618
619            let r#type = self.convert_type(typ);
620            FunctionParam { name, r#type, mut_ref }
621        });
622        let return_type = self.convert_type(func_meta.return_type());
623        let trait_constraints = func_meta.trait_constraints.clone();
624        let where_clause =
625            vecmap(trait_constraints, |constraint| self.convert_trait_constraint(&constraint));
626
627        // Only keep trait constraints if they aren't already present because they exist in the
628        // parent trait/impl.
629        let where_clause = where_clause
630            .iter()
631            .filter(|trait_constraint| !self.trait_constraints.contains(trait_constraint))
632            .cloned()
633            .collect::<Vec<_>>();
634
635        let attributes = self.interner.function_attributes(&func_id);
636        let deprecated = attributes.get_deprecated_note();
637
638        let id = get_function_id(func_id, self.interner);
639
640        Function {
641            id,
642            name,
643            comments,
644            unconstrained,
645            comptime,
646            generics,
647            params,
648            return_type,
649            where_clause,
650            deprecated,
651        }
652    }

667    /// Goes over a module's imports. If an import is a re-export of a private item,
668    /// the item is added to the module's items.
669    fn process_module_reexports(&mut self, module: &mut Module) {
670        // Process this module's sub-modules first because when we actually process the
671        // imports we might add a publicly exported module into this module's items,
672        // visiting it twice.
673        for (_visibility, item) in &mut module.items {
674            if let Item::Module(sub_module) = item {
675                self.process_module_reexports(sub_module);
676            }
677        }
678
679        let imports = self.module_imports.remove(&module.module_id).unwrap();
680
681        for import in imports {
682            if import.visibility == ItemVisibility::Private {
683                continue;
684            }
685
686            let item_id = get_module_def_id(import.id, self.interner);
687            if let Some(converted_item) = self.item_id_to_converted_item.get(&item_id) {
688                // Check if this is a re-export of a private item. The private item won't show up in
689                // its module docs (because it's private) so it's included directly under
690                // the module that re-exports it (this is how rustdoc works too).
691                //
692                // We also check if this a re-export of an item from another crate. In that
693                // case we'll also include that item directly under this module as the other
694                // crate might not be part of the workspace (just a dependency).
695                if converted_item.visibility < import.visibility
696                    || converted_item.crate_id != self.crate_id
697                {
698                    let mut item = converted_item.item.clone();
699                    item.set_name(import.name.to_string());
700
701                    module.items.push((import.visibility, item));
702                    continue;
703                }
704            }
705
706            // This is an internal or external re-export
707            let id = get_module_def_id(import.id, self.interner);
708            module.items.push((
709                import.visibility,
710                Item::Reexport(Reexport {
711                    id,
712                    item_name: self.get_module_def_id_name(import.id),
713                    name: import.name.to_string(),
714                }),
715            ));
716        }
717    }

719    fn doc_comments(&mut self, id: ReferenceId) -> Option<(String, Links)> {
720        self.link_finder.reset();
721
722        let comments = self.interner.doc_comments(id)?;
723        let mut links = Vec::new();
724        let mut line = 0;
725
726        // Go comment by comment so that broken link locations are more accurate.
727        for comment in comments {
728            let mut comment_links = self.find_links_in_comments(&comment.contents);
729            for link in &mut comment_links {
730                link.line += line;
731
732                if link.target.is_none() {
733                    let location = self.link_location(comment, link.line, link.start, link.end);
734                    let broken_link = BrokenLink { text: link.path.clone(), location };
735                    self.broken_links.push(broken_link);
736                }
737            }
738            links.extend(comment_links);
739            line += comment.contents.lines().count().max(1);
740        }
741
742        let comments =
743            vecmap(comments, |comment| comment.contents.clone()).join("\n").trim().to_string();
744        Some((comments, links))
745    }

747    /// The idea of this method is to find occurrences of markdown links and references in comments.
748    /// For each of these we try to resolve them to a ModuleDefId of sort, which
749    /// is actually represented as a Link to a type, method, module, etc.
750    ///
751    /// The doc generator ([html::to_html]) will then replace occurrences of these links
752    /// with resolved HTML links.
753    fn find_links_in_comments(&mut self, comments: &str) -> Links {
754        let current_module_id = ModuleId { krate: self.crate_id, local_id: self.current_module_id };
755        let links = self.link_finder.find_links(
756            comments,
757            current_module_id,
758            self.current_type,
759            self.interner,
760            self.def_maps,
761            self.crate_graph,
762        );
763        vecmap(links, |link| {
764            let target = link.target.map(|target| match target {
765                links::LinkTarget::TopLevelItem(module_def_id) => {
766                    LinkTarget::TopLevelItem(get_module_def_id(module_def_id, self.interner))
767                }
768                links::LinkTarget::Method(module_def_id, func_id) => {
769                    let name = self.interner.function_name(&func_id).to_string();
770                    LinkTarget::Method(get_module_def_id(module_def_id, self.interner), name)
771                }
772                links::LinkTarget::StructMember(type_id, index) => {
773                    let data_type = self.interner.get_type(type_id);
774                    let name = data_type.borrow().field_at(index).name.to_string();
775                    LinkTarget::StructMember(get_type_id(type_id, self.interner), name)
776                }
777                links::LinkTarget::PrimitiveType(primitive_type_kind) => {
778                    LinkTarget::PrimitiveType(primitive_type_kind)
779                }
780                links::LinkTarget::PrimitiveTypeFunction(primitive_type_kind, func_id) => {
781                    let name = self.interner.function_name(&func_id).to_string();
782                    LinkTarget::PrimitiveTypeFunction(primitive_type_kind, name)
783                }
784            });
785            Link {
786                name: link.name,
787                path: link.path,
788                target,
789                line: link.line,
790                start: link.start,
791                end: link.end,
792            }
793        })
794    }

796    /// Returns the actual [`Location`] of a link inside `comment`, one that is at the given
797    /// `line`, `start` and `end`.
798    fn link_location(
799        &self,
800        comment: &DocComment,
801        line: usize,
802        start: usize,
803        end: usize,
804    ) -> Location {
805        let location = comment.location();
806        let file = location.file;
807        let source = self.file_manager.fetch_file(file).unwrap();
808        let text: &str = &source[location.span.start() as usize..location.span.end() as usize];
809        let offset = link_offset(text, line) + start;
810        let span_start = location.span.start() + offset as u32;
811        let span_end = span_start + (end - start) as u32;
812        let span = Span::from(span_start..span_end);
813        Location::new(span, file)
814    }

831                definition.name == "self"
832            }

838    fn get_module_def_id_name(&self, id: ModuleDefId) -> String {
839        match id {
840            ModuleDefId::ModuleId(module_id) => {
841                let attributes = self.interner.try_module_attributes(module_id);
842                attributes.map(|attributes| &attributes.name).cloned().unwrap_or_else(String::new)
843            }
844            ModuleDefId::FunctionId(func_id) => self.interner.function_name(&func_id).to_string(),
845            ModuleDefId::TypeId(type_id) => {
846                let data_type = self.interner.get_type(type_id);
847                let data_type = data_type.borrow();
848                data_type.name.to_string()
849            }
850            ModuleDefId::TypeAliasId(type_alias_id) => {
851                let type_alias = self.interner.get_type_alias(type_alias_id);
852                let type_alias = type_alias.borrow();
853                type_alias.name.to_string()
854            }
855            ModuleDefId::TraitId(trait_id) => {
856                let trait_ = self.interner.get_trait(trait_id);
857                trait_.name.to_string()
858            }
859            ModuleDefId::TraitAssociatedTypeId(id) => {
860                let associated_type = self.interner.get_trait_associated_type(id);
861                associated_type.name.to_string()
862            }
863            ModuleDefId::GlobalId(global_id) => {
864                let global_info = self.interner.get_global(global_id);
865                global_info.ident.to_string()
866            }
867        }
868    }

917/// Returns the offset in `text` at which the actual comment text start at the given `line`,
918/// assuming `text` is the entire doc comment text.
919/// The `start` and `end` position of links are relative to the line in which a link appears,
920/// and because broken links need to be reported as [`Location`]s we need to adjust their
921/// span accordingly.
922fn link_offset(text: &str, line: usize) -> usize {
923    // Easy: for line comments we just need to skip the comment prefix
924    if text.starts_with("/// ") || text.starts_with("//! ") {
925        return 4;
926    }
927    if text.starts_with("///") || text.starts_with("//!") {
928        return 3;
929    }
930
931    // If the text contains "\r\n" we assume that's the newline style used.
932    let rn = text.contains("\r\n");
933    let newline_width = if rn { 2 } else { 1 };
934
935    // A bit more tricky: block comments.
936    let mut offset = 0;
937    let lines = text.lines().collect::<Vec<_>>();
938
939    // The line number in the doc comment we are in. This isn't exactly the `index` we get
940    // from `iter().enumerate()` because if the first line is just "/**" or "/*!" (with optional
941    // trailing spaces) then that line is not counted as the first line of the comment (the next
942    // one will).
943    let mut current_line_number: usize = 0;
944
945    for (line_index, line_text) in lines.iter().enumerate() {
946        // Special check for the first line
947        if line_index == 0 {
948            // We first skip past "/**" or "/*!" (one of those must come).
949            let new_line_text =
950                line_text.strip_prefix("/**").or_else(|| line_text.strip_prefix("/*!")).unwrap();
951            offset += 3;
952
953            // Next we skip any spaces after that
954            let line_text_length = new_line_text.len();
955            let new_line_text = new_line_text.trim_start();
956            if new_line_text.is_empty() {
957                // If the line is empty, the entire line is skipped. Note that we proceed
958                // with the next line without incrementing `current_line_number`.
959                offset += new_line_text.len() + newline_width;
960                continue;
961            } else {
962                // Otherwise we just skip the spaces.
963                offset += line_text_length - new_line_text.len();
964            }
965        }
966
967        // Did we reach the line we were looking for?
968        if current_line_number == line {
969            let line_text_length = line_text.len();
970            let line_text = line_text.trim_start();
971            // Adjust offset to account for leading spaces
972            offset += line_text_length - line_text.len();
973
974            // Does every line in the comment start with "*" (except for the new first line)
975            let all_stars = block_comment_has_all_leading_stars(text);
976
977            // If every line starts with "*" we need to skip past it, and any spaces after it.
978            if all_stars {
979                if let Some(line_text) = line_text.strip_prefix('*') {
980                    offset += 1;
981                    let line_text_length = line_text.len();
982                    let line_text = line_text.trim_start();
983                    // Adjust offset to account for leading spaces after the "*"
984                    offset += line_text_length - line_text.len();
985                }
986            }
987
988            break;
989        }
990
991        offset += line_text.len() + newline_width;
992        current_line_number += 1;
993    }
994    offset
995}
