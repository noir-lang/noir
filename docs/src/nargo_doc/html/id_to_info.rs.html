10pub(super) struct ItemInfo {
11    pub(super) path: Vec<String>,
12    pub(super) uri: String,
13    pub(super) class: &'static str,
14    /// Overall visibility of this item, including parent modules.
15    /// For example, if this is a public item inside a private module,
16    /// its visibility will be private.
17    pub(super) visibility: ItemVisibility,
18}
19
20/// Computes an ItemInfo for every item in the workspace, indexed by its Id.
21pub(super) fn compute_id_to_info(workspace: &Workspace) -> HashMap<ItemId, ItemInfo> {
22    let mut id_to_info = HashMap::new();
23    let mut path = Vec::new();
24
25    for krate in workspace.all_crates() {
26        let module = &krate.root_module;
27        let uri = krate.uri();
28        let class = module.class();
29        let visibility = ItemVisibility::Public;
30        id_to_info.insert(module.id.clone(), ItemInfo { path: Vec::new(), uri, class, visibility });
31
32        path.push(krate.name.to_string());
33
34        for (visibility, item) in &krate.root_module.items {
35            compute_id_to_info_in_item(item, *visibility, &mut id_to_info, &mut path);
36        }
37
38        path.pop();
39    }
40
41    id_to_info
42}
43
44fn compute_id_to_info_in_item(
45    item: &Item,
46    visibility: ItemVisibility,
47    id_to_info: &mut HashMap<ItemId, ItemInfo>,
48    path: &mut Vec<String>,
49) {
50    match item {
51        Item::Module(module) => {
52            let uri = format!("{}/{}", path.join("/"), module.uri());
53            let class = module.class();
54            id_to_info
55                .insert(module.id.clone(), ItemInfo { path: path.clone(), uri, class, visibility });
56
57            path.push(module.name.clone());
58            for (item_visibility, item) in &module.items {
59                let visibility = visibility.min(*item_visibility);
60                compute_id_to_info_in_item(item, visibility, id_to_info, path);
61            }
62            path.pop();
63        }
64        Item::Struct(struct_) => {
65            let uri = format!("{}/{}", path.join("/"), struct_.uri());
66            let class = struct_.class();
67            id_to_info.insert(
68                struct_.id.clone(),
69                ItemInfo { path: path.clone(), uri, class, visibility },
70            );
71        }
72        Item::Trait(trait_) => {
73            let uri = format!("{}/{}", path.join("/"), trait_.uri());
74            let class = trait_.class();
75            id_to_info
76                .insert(trait_.id.clone(), ItemInfo { path: path.clone(), uri, class, visibility });
77        }
78        Item::TypeAlias(type_alias) => {
79            let uri = format!("{}/{}", path.join("/"), type_alias.uri());
80            let class = type_alias.class();
81            id_to_info.insert(
82                type_alias.id.clone(),
83                ItemInfo { path: path.clone(), uri, class, visibility },
84            );
85        }
86        Item::Function(function) => {
87            let uri = format!("{}/{}", path.join("/"), function.uri());
88            let class = function.class();
89            id_to_info.insert(
90                function.id.clone(),
91                ItemInfo { path: path.clone(), uri, class, visibility },
92            );
93        }
94        Item::Global(global) => {
95            let uri = format!("{}/{}", path.join("/"), global.uri());
96            let class = global.class();
97            id_to_info
98                .insert(global.id.clone(), ItemInfo { path: path.clone(), uri, class, visibility });
99        }
100        Item::PrimitiveType(_) | Item::Reexport(_) => {}
101    }
102}
