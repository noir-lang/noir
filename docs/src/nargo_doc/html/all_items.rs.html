1use noirc_frontend::ast::ItemVisibility;
2
3use crate::items::{Item, Workspace};

5/// All items in a workspace.
6/// Each field is a list of (path, name) tuples.
7pub(super) struct AllItems {
8    pub(super) structs: Vec<(Vec<String>, String)>,
9    pub(super) traits: Vec<(Vec<String>, String)>,
10    pub(super) type_aliases: Vec<(Vec<String>, String)>,
11    pub(super) primitive_types: Vec<(Vec<String>, String)>,
12    pub(super) functions: Vec<(Vec<String>, String)>,
13    pub(super) globals: Vec<(Vec<String>, String)>,
14}
15
16pub(super) fn compute_all_items(workspace: &Workspace) -> AllItems {
17    let mut all_items = AllItems {
18        structs: Vec::new(),
19        traits: Vec::new(),
20        type_aliases: Vec::new(),
21        primitive_types: Vec::new(),
22        globals: Vec::new(),
23        functions: Vec::new(),
24    };
25
26    let mut current_path = Vec::new();
27    for krate in &workspace.crates {
28        current_path.push(krate.name.clone());
29        for (visibility, item) in &krate.root_module.items {
30            if visibility == &ItemVisibility::Public {
31                gather_all_items_in_item(item, &mut current_path, &mut all_items);
32            }
33        }
34        current_path.pop();
35    }
36
37    all_items.structs.sort();
38    all_items.traits.sort();
39    all_items.type_aliases.sort();
40    all_items.primitive_types.sort();
41    all_items.functions.sort();
42    all_items.globals.sort();
43
44    all_items
45}
46
47fn gather_all_items_in_item(item: &Item, current_path: &mut Vec<String>, all_items: &mut AllItems) {
48    match item {
49        Item::Module(module) => {
50            current_path.push(module.name.clone());
51            for (visibility, item) in &module.items {
52                if visibility == &ItemVisibility::Public {
53                    gather_all_items_in_item(item, current_path, all_items);
54                }
55            }
56            current_path.pop();
57        }
58        Item::Struct(struct_) => {
59            all_items.structs.push((current_path.clone(), struct_.name.clone()));
60        }
61        Item::Trait(trait_) => {
62            all_items.traits.push((current_path.clone(), trait_.name.clone()));
63        }
64        Item::TypeAlias(type_alias) => {
65            all_items.type_aliases.push((current_path.clone(), type_alias.name.clone()));
66        }
67        Item::PrimitiveType(primitive_type) => {
68            all_items.primitive_types.push((current_path.clone(), primitive_type.kind.to_string()));
69        }
70        Item::Global(global) => {
71            all_items.globals.push((current_path.clone(), global.name.clone()));
72        }
73        Item::Function(function) => {
74            all_items.functions.push((current_path.clone(), function.name.clone()));
75        }
76        Item::Reexport(_) => {}
77    }
78}
