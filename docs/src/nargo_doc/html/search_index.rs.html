9/// A single searchable item in the generated documentation.
10#[derive(Debug, Clone, PartialEq, Eq, Serialize)]
11pub(super) struct SearchEntry {
12    /// The item's name, e.g. `BoundedVec`.
13    pub(super) name: String,
14    /// The fully qualified module path of the item, e.g. `std::collections::bounded_vec`.
15    pub(super) path: String,
16    /// The kind of the item, matching its CSS class, e.g. `struct`, `trait`, `fn`, `module`.
17    pub(super) kind: String,
18    /// The URL of the item's page, relative to the documentation root.
19    pub(super) url: String,
20    /// A short, single-line plain-text summary taken from the item's doc comments.
21    #[serde(rename = "desc")]
22    pub(super) description: String,
23}
24
25/// Gathers every searchable item across the workspace crates.
26///
27/// Items that are publicly re-exported from another crate are inlined into the re-exporting
28/// module at conversion time (see `process_module_reexports`), so they are reachable here just
29/// like items defined locally.
30pub(super) fn compute_search_index(workspace: &Workspace) -> Vec<SearchEntry> {
31    let mut entries = Vec::new();
32    let mut current_path = Vec::new();
33
34    for krate in &workspace.crates {
35        if !krate.root_module.has_public_items() {
36            continue;
37        }
38
39        current_path.push(krate.name.clone());
40        for (visibility, item) in &krate.root_module.items {
41            if visibility == &ItemVisibility::Public {
42                gather_search_entries_in_item(item, &mut current_path, &mut entries);
43            }
44        }
45        current_path.pop();
46    }
47
48    // Sort for a stable order. Members such as trait-impl methods are gathered in a
49    // non-deterministic order, which would otherwise make the generated file differ between runs.
50    // The order only affects how equally-ranked results tie-break at search time.
51    entries.sort_by(|a, b| {
52        a.name
53            .cmp(&b.name)
54            .then_with(|| a.path.cmp(&b.path))
55            .then_with(|| a.kind.cmp(&b.kind))
56            .then_with(|| a.url.cmp(&b.url))
57    });
58
59    entries
60}

62fn gather_search_entries_in_item(
63    item: &Item,
64    current_path: &mut Vec<String>,
65    entries: &mut Vec<SearchEntry>,
66) {
67    match item {
68        Item::Module(module) => {
69            push_entry(entries, current_path, &module.name, module.class(), &module.uri(), module);
70            current_path.push(module.name.clone());
71            for (visibility, item) in &module.items {
72                if visibility == &ItemVisibility::Public {
73                    gather_search_entries_in_item(item, current_path, entries);
74                }
75            }
76            current_path.pop();
77        }
78        Item::Struct(struct_) => {
79            push_entry(
80                entries,
81                current_path,
82                &struct_.name,
83                struct_.class(),
84                &struct_.uri(),
85                struct_,
86            );
87
88            let context = TypeContext::new(current_path, &struct_.name, &struct_.uri());
89            for field in &struct_.fields {
90                push_member_entry(
91                    entries,
92                    &context,
93                    &field.name,
94                    "field",
95                    &format!("structfield.{}", field.name),
96                    &field.comments,
97                );
98            }
99            push_methods(entries, &context, &struct_.impls);
100            push_trait_impl_methods(entries, &context, &struct_.trait_impls);
101        }
102        Item::Trait(trait_) => {
103            push_entry(entries, current_path, &trait_.name, trait_.class(), &trait_.uri(), trait_);
104
105            let context = TypeContext::new(current_path, &trait_.name, &trait_.uri());
106            for method in trait_.required_methods.iter().chain(&trait_.provided_methods) {
107                push_member_entry(
108                    entries,
109                    &context,
110                    &method.name,
111                    "method",
112                    &method.name,
113                    &method.comments,
114                );
115            }
116        }
117        Item::TypeAlias(type_alias) => {
118            push_entry(
119                entries,
120                current_path,
121                &type_alias.name,
122                type_alias.class(),
123                &type_alias.uri(),
124                type_alias,
125            );
126        }
127        Item::PrimitiveType(primitive_type) => {
128            let name = primitive_type.kind.to_string();
129            push_entry(
130                entries,
131                current_path,
132                &name,
133                primitive_type.class(),
134                &primitive_type.uri(),
135                primitive_type,
136            );
137
138            let context = TypeContext::new(current_path, &name, &primitive_type.uri());
139            push_methods(entries, &context, &primitive_type.impls);
140            push_trait_impl_methods(entries, &context, &primitive_type.trait_impls);
141        }
142        Item::Global(global) => {
143            push_entry(entries, current_path, &global.name, global.class(), &global.uri(), global);
144        }
145        Item::Function(function) => {
146            push_entry(
147                entries,
148                current_path,
149                &function.name,
150                function.class(),
151                &function.uri(),
152                function,
153            );
154        }
155        Item::Reexport(_) => {}
156    }
157}

159fn push_entry(
160    entries: &mut Vec<SearchEntry>,
161    current_path: &[String],
162    name: &str,
163    kind: &str,
164    uri: &str,
165    item: &dyn ItemProperties,
166) {
167    let path = current_path.join("::");
168    let url = format!("{}/{uri}", current_path.join("/"));
169    entries.push(SearchEntry {
170        name: name.to_string(),
171        path,
172        kind: kind.to_string(),
173        url,
174        description: description(item.comments()),
175    });
176}

178/// Identifies the type whose members are being indexed: its qualified path (e.g.
179/// `std::collections::bounded_vec::BoundedVec`) and the URL of its page.
180struct TypeContext {
181    path: String,
182    url: String,
183}
184
185impl TypeContext {
186    fn new(current_path: &[String], type_name: &str, type_uri: &str) -> Self {
187        Self {
188            path: format!("{}::{type_name}", current_path.join("::")),
189            url: format!("{}/{type_uri}", current_path.join("/")),
190        }
191    }
192}
193
194/// Indexes the methods of a type's inherent `impl` blocks. Each method's page is its type's
195/// page, anchored at the method name.
196fn push_methods(entries: &mut Vec<SearchEntry>, context: &TypeContext, impls: &[Impl]) {
197    for impl_ in impls {
198        for method in &impl_.methods {
199            push_member_entry(
200                entries,
201                context,
202                &method.name,
203                "method",
204                &method.name,
205                &method.comments,
206            );
207        }
208    }
209}
210
211/// Indexes the methods of a type's trait `impl` blocks. Trait-impl methods don't get their own
212/// anchor, so they link to the surrounding `impl` block on the type's page.
213fn push_trait_impl_methods(
214    entries: &mut Vec<SearchEntry>,
215    context: &TypeContext,
216    trait_impls: &[TraitImpl],
217) {
218    for trait_impl in trait_impls {
219        let anchor = super::trait_impl_anchor(trait_impl);
220        for method in &trait_impl.methods {
221            push_member_entry(entries, context, &method.name, "method", &anchor, &method.comments);
222        }
223    }
224}

226/// Adds an entry for a member (a field or method) of a type. Its qualified path includes the
227/// type name, and its URL is the type's page anchored at `anchor`.
228fn push_member_entry(
229    entries: &mut Vec<SearchEntry>,
230    context: &TypeContext,
231    name: &str,
232    kind: &str,
233    anchor: &str,
234    comments: &Option<Comments>,
235) {
236    entries.push(SearchEntry {
237        name: name.to_string(),
238        path: context.path.clone(),
239        kind: kind.to_string(),
240        url: format!("{}#{anchor}", context.url),
241        description: description(comments.as_ref()),
242    });
243}

245/// Returns a short, single-line plain-text description from an item's doc comments.
246fn description(comments: Option<&Comments>) -> String {
247    let Some((markdown, _links)) = comments else {
248        return String::new();
249    };
250
251    markdown_summary(markdown).lines().next().unwrap_or("").trim().to_string()
252}

254/// Renders the search index as the contents of `search-index.js`.
255///
256/// The index is assigned to a global variable (rather than fetched) so that search works when
257/// the documentation is opened directly from the filesystem, where `fetch` is blocked by the
258/// browser's same-origin policy.
259pub(super) fn render_search_index_js(entries: &[SearchEntry]) -> String {
260    let mut output = String::from("window.searchIndex = [\n");
261    for entry in entries {
262        output.push_str("  ");
263        output.push_str(
264            &serde_json::to_string(entry).expect("a search entry is always serializable"),
265        );
266        output.push_str(",\n");
267    }
268    output.push_str("];\n");
269    output
270}

367    #[test]
368    fn collects_items_with_qualified_path_and_url() {
369        let comments: Comments = ("The Foo struct.\n\nMore details.".to_string(), Vec::new());
370        let submodule =
371            Item::Module(module("submod", vec![(ItemVisibility::Public, function_item("bar"))]));
372        let root = vec![
373            (ItemVisibility::Public, struct_item("Foo", Some(comments))),
374            (ItemVisibility::Public, submodule),
375        ];
376
377        let entries = compute_search_index(&workspace(vec![krate("mylib", root)]));
378
379        let foo = entries.iter().find(|e| e.name == "Foo").expect("Foo should be indexed");
380        assert_eq!(foo.path, "mylib");
381        assert_eq!(foo.kind, "struct");
382        assert_eq!(foo.url, "mylib/struct.Foo.html");
383        assert_eq!(foo.description, "The Foo struct.");
384
385        let submod = entries.iter().find(|e| e.name == "submod").expect("submod should be indexed");
386        assert_eq!(submod.path, "mylib");
387        assert_eq!(submod.kind, "module");
388        assert_eq!(submod.url, "mylib/submod/index.html");
389
390        let bar = entries.iter().find(|e| e.name == "bar").expect("bar should be indexed");
391        assert_eq!(bar.path, "mylib::submod");
392        assert_eq!(bar.kind, "fn");
393        assert_eq!(bar.url, "mylib/submod/fn.bar.html");
394    }
