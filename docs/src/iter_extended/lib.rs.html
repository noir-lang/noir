6/// Equivalent to .into_iter().map(f).collect::<Vec<_>>()
7pub fn vecmap<T, U, F>(iterable: T, f: F) -> Vec<U>
8where
9    T: IntoIterator,
10    F: FnMut(T::Item) -> U,
11{

44/// Given an iterator over a Result, filter out the Ok values from the Err values
45/// and return both in separate `Vec`s. Unlike other collect-like functions over Results,
46/// this function will always consume the entire iterator.
47pub fn partition_results<It, T, E, F>(iterable: It, mut f: F) -> (Vec<T>, Vec<E>)
48where
49    It: IntoIterator,
50    F: FnMut(It::Item) -> Result<T, E>,
51{
52    let mut oks = vec![];
53    let mut errors = vec![];
54
55    for elem in iterable {
56        match f(elem) {
57            Ok(ok) => oks.push(ok),
58            Err(error) => errors.push(error),
59        }
60    }
61
62    (oks, errors)
63}
