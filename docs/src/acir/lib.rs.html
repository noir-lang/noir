66    /// Technical DTO for deserializing in Barretenberg while ignoring
67    /// the Brillig opcodes, so that we can add more without affecting it.
68    ///
69    /// This could be achieved in other ways, for example by having a
70    /// version of `Program` that deserializes into opaque bytes,
71    /// which would require a 2 step (de)serialization process.
72    ///
73    /// This one is simpler. The cost is that msgpack will deserialize
74    /// into a JSON-like structure, but since we won't be interpreting it,
75    /// it's okay if new tags appear.
76    #[derive(Clone, PartialEq, Eq, Serialize, Deserialize, Default, Hash, MsgpackTagged)]
77    struct ProgramWithoutBrillig<F: AcirField> {
78        #[tag(0)]
79        pub functions: Vec<Circuit<F>>,
80        /// We want to ignore this field. By setting its type as `unit`
81        /// it will not be deserialized, but it will correctly maintain
82        /// the position of the others (although in this case it doesn't)
83        /// matter since it's the last field.
84        #[tag(1)]
85        pub unconstrained_functions: (),
86    }

187    /// Regenerate C++ code for serializing our domain model based on serde,
188    /// intended to be used by Barretenberg.
189    ///
190    /// If `should_overwrite()` returns `false` then just check if the old file hash is the
191    /// same as the new one, to guard against unintended changes in the serialization format.
192    fn serde_cpp_codegen(
193        namespace: &str,
194        path: &Path,
195        registry: &Registry,
196        tag_registry: &TagRegistry,
197        code: CustomCode,
198    ) {
199        let old_hash = if path.is_file() {
200            let old_source = std::fs::read(path).expect("failed to read existing code");
201            let old_source = String::from_utf8(old_source).expect("old source not UTF-8");
202            Some(rustc_hash::FxBuildHasher.hash_one(&old_source))
203        } else {
204            None
205        };
206        let msgpack_code = MsgPackCodeGenerator::generate(
207            namespace,
208            registry,
209            tag_registry,
210            code,
211            MsgPackCodeConfig::from_env(),
212        );
213
214        // Create C++ class definitions.
215        let mut source = Vec::new();
216        // We use `serde_generate` to take advantage of its integration with `serde_reflection` but only use our
217        // custom msgpack code generation.
218        let config = serde_generate::CodeGeneratorConfig::new(namespace.to_string())
219            .with_encodings(vec![])
220            .with_custom_code(msgpack_code);
221        let generator = serde_generate::cpp::CodeGenerator::new(&config);
222        generator.output(&mut source, registry).expect("failed to generate C++ code");
223
224        // Further massaging of the generated code
225        let mut source = String::from_utf8(source).expect("not a UTF-8 string");
226        replace_throw(&mut source);
227        MsgPackCodeGenerator::add_preamble(&mut source);
228        MsgPackCodeGenerator::add_helpers(&mut source, namespace);
229        MsgPackCodeGenerator::replace_array_with_shared_ptr(&mut source);
230        MsgPackCodeGenerator::add_autogen_header(&mut source);
231
232        if !should_overwrite()
233            && let Some(old_hash) = old_hash
234        {
235            let new_hash = rustc_hash::FxBuildHasher.hash_one(&source);
236            assert_eq!(new_hash, old_hash, "Serialization format has changed",);
237        }
238
239        write_to_file(source.as_bytes(), path);
240    }

242    /// Get a boolean flag env var.
243    fn env_flag(name: &str, default: bool) -> bool {
244        let Ok(s) = std::env::var(name) else {
245            return default;
246        };
247        match s.as_str() {
248            "1" | "true" | "yes" => true,
249            "0" | "false" | "no" => false,
250            _ => default,
251        }

260    fn write_to_file(bytes: &[u8], path: &Path) -> String {
261        let display = path.display();
262
263        let parent_dir = path.parent().unwrap();
264        if !parent_dir.is_dir() {
265            std::fs::create_dir_all(parent_dir).unwrap();
266        }
267
268        let mut file = match File::create(path) {
269            Err(why) => panic!("couldn't create {display}: {why}"),
270            Ok(file) => file,
271        };
272
273        match file.write_all(bytes) {
274            Err(why) => panic!("couldn't write to {display}: {why}"),
275            Ok(_) => display.to_string(),
276        }
277    }

307    /// Assert that every product / sum container in the serde-reflection
308    /// `registry` is also present in the `MsgpackTagged` `tag_registry`,
309    /// and that every product is in *canonical* (tag-ascending) source
310    /// order. Panics with a focused message on the first miss.
311    ///
312    /// **Why the coverage check.** The int-keyed dispatch branch in the
313    /// generated C++ needs each field's u8 tag (for structs) and each
314    /// variant's u8 tag (for enums). That metadata only exists on the
315    /// `MsgpackTagged` side. A type in `registry` but not in
316    /// `tag_registry` means somebody added a wire type without deriving
317    /// `MsgpackTagged` — fail loudly at codegen time, not silently at
318    /// runtime.
319    ///
320    /// **Why the order check.** Under `Format::MsgpackCompact` a struct
321    /// is emitted as a fixarray in *source* order; under
322    /// `Format::MsgpackTagged` with the `Array` strategy it's a fixarray
323    /// in *tag-ascending* order. Both shapes look identical to the C++
324    /// decoder (just `msgpack::type::ARRAY`), so byte-different wires
325    /// for the same logical value would silently land on the wrong
326    /// fields if the two orders ever diverge. Forcing
327    /// `tag_order_matches_source` keeps them lockstep — both formats
328    /// produce byte-identical arrays, and the codegen reads positionally
329    /// in tag-ascending order.
330    fn assert_tag_registry_covers(registry: &Registry, tag_registry: &TagRegistry) {
331        use serde_reflection::ContainerFormat;
332        for (name, container) in registry {
333            // Only named-struct and enum containers need an entry in the
334            // `MsgpackTagged` registry: their int-keyed dispatch branch
335            // depends on per-field / per-variant tag metadata. Unit
336            // structs, newtypes, and (currently `unimplemented!`) tuple
337            // structs pass through to their inner type (or are no-ops)
338            // and don't register themselves in `TagRegistry`.
339            let needs_tags =
340                matches!(container, ContainerFormat::Struct(_) | ContainerFormat::Enum(_),);
341            if !needs_tags {
342                continue;
343            }
344            let Some(entry) = tag_registry.get(name) else {
345                panic!(
346                    "MsgpackTagged tag registry is missing {name:?} — the type is on \
347                     the wire (serde-reflection traced it as a struct/enum) but \
348                     doesn't derive `MsgpackTagged`. Add `#[derive(MsgpackTagged)]` \
349                     to the type, and a `register_into` call at the codegen test \
350                     site if the type isn't reachable from a type that's already \
351                     registered.",
352                );
353            };
354            if let Some(p) = entry.tagged().as_product()
355                && !p.tag_order_matches_source
356            {
357                panic!(
358                    "MsgpackTagged product {name:?} declares its fields in an order \
359                     that doesn't match tag-ascending. The C++ codegen can't tell \
360                     `MsgpackCompact` (source-order array) from \
361                     `MsgpackTagged::Array` (tag-ascending array) on the wire — \
362                     reorder the Rust fields so `#[tag(N)]` values are increasing \
363                     in source order, or drop the type from the C++ wire types.",
364                );
365            }
366        }
367    }

369    /// Generate custom code for the msgpack machinery in Barretenberg.
370    /// See https://github.com/AztecProtocol/aztec-packages/blob/master/barretenberg/cpp/src/barretenberg/serialize/msgpack.hpp
371    struct MsgPackCodeGenerator<'a> {
372        config: MsgPackCodeConfig,
373        namespace: Vec<String>,
374        code: CustomCode,
375        /// Carries the integer-tag metadata produced by `MsgpackTagged`
376        /// derives. The serde-reflection `Registry` only knows field/variant
377        /// *names*; consulting this registry lets us pair each name with its
378        /// stable u8 tag so the generated C++ can dispatch on int-keyed
379        /// `MsgpackTagged` wires the same way the Rust decoder does.
380        tag_registry: &'a TagRegistry,
381    }

383    impl<'a> MsgPackCodeGenerator<'a> {
384        /// Prepend a banner marking the file as auto-generated and pointing
385        /// readers at the generator. Without it, the committed `.cpp` files

393// To regenerate, run:

395//     NOIR_CODEGEN_OVERWRITE=1 cargo test -p acir cpp_codegen
396//

411        /// Add helper functions to cut down repetition in the generated code.
412        pub(crate) fn add_helpers(source: &mut String, namespace: &str) {
413            // Based on https://github.com/AztecProtocol/msgpack-c/blob/54e9865b84bbdc73cfbf8d1d437dbf769b64e386/include/msgpack/v1/adaptor/detail/cpp11_define_map.hpp#L75
414            // Using a `struct Helpers` with `static` methods, because top level functions turn up as duplicates in `wasm-ld`.
415            // cSpell:disable
416            let helpers = r#"
417    struct Helpers {
418        static std::map<std::string, msgpack::object const*> make_kvmap(
419            msgpack::object const& o,
420            std::string const& name
421        ) {
422            if (o.type != msgpack::type::MAP) {
423                std::cerr << o << std::endl;
424                throw_or_abort("expected MAP for " + name);
425            }
426            std::map<std::string, msgpack::object const*> kvmap;
427            for (uint32_t i = 0; i < o.via.map.size; ++i) {
428                if (o.via.map.ptr[i].key.type != msgpack::type::STR) {
429                    std::cerr << o << std::endl;
430                    throw_or_abort("expected STR for keys of " + name);
431                }
432                kvmap.emplace(
433                    std::string(
434                        o.via.map.ptr[i].key.via.str.ptr,
435                        o.via.map.ptr[i].key.via.str.size),
436                    &o.via.map.ptr[i].val);
437            }
438            return kvmap;
439        }

441        template<typename T>
442        static void conv_fld_from_kvmap(
443            std::map<std::string, msgpack::object const*> const& kvmap,
444            std::string const& struct_name,
445            std::string const& field_name,
446            T& field,
447            bool is_optional
448        ) {
449            auto it = kvmap.find(field_name);
450            if (it != kvmap.end()) {
451                if (!is_optional && it->second->type == msgpack::type::NIL) {
452                    throw_or_abort("nil value for required field: " + struct_name + "::" + field_name);
453                }
454                try {
455                    it->second->convert(field);
456                } catch (const msgpack::type_error&) {
457                    std::cerr << *it->second << std::endl;
458                    throw_or_abort("error converting into field " + struct_name + "::" + field_name);
459                }
460            } else if (!is_optional) {
461                throw_or_abort("missing field: " + struct_name + "::" + field_name);
462            }
463        }

465        template<typename T>
466        static void conv_fld_from_array(
467            msgpack::object_array const& array,
468            std::string const& struct_name,
469            std::string const& field_name,
470            T& field,
471            uint32_t index
472        ) {
473            if (index >= array.size) {
474                throw_or_abort("index out of bounds: " + struct_name + "::" + field_name + " at " + std::to_string(index));
475            }
476            auto element = array.ptr[index];
477            if (element.type == msgpack::type::NIL) {
478                throw_or_abort("nil value for required field: " + struct_name + "::" + field_name);
479            }
480            try {
481                element.convert(field);
482            } catch (const msgpack::type_error&) {
483                std::cerr << element << std::endl;
484                throw_or_abort("error converting into field " + struct_name + "::" + field_name);
485            }
486        }

488        /// Convert `val` into `field`, or throw a focused error mentioning
489        /// the struct + field name. Used by the int-keyed dispatch path
490        /// where each `switch` case populates one field directly.
491        template<typename T>
492        static void convert_or_throw(
493            msgpack::object const& val,
494            std::string const& struct_name,
495            std::string const& field_name,
496            T& field
497        ) {
498            try {
499                val.convert(field);
500            } catch (const msgpack::type_error&) {
501                std::cerr << val << std::endl;
502                throw_or_abort("error converting into field " + struct_name + "::" + field_name);
503            }
504        }

506        /// Whether `o` is a non-empty MAP whose first key is an integer.
507        /// This is the signature of `Format::MsgpackTagged`: int keys for
508        /// struct field tags and enum variant tags. Legacy `Format::Msgpack`
509        /// keys are always strings, so a positive-integer first key is a
510        /// reliable shape discriminator between the two.
511        static bool is_int_keyed_map(msgpack::object const& o) {
512            return o.type == msgpack::type::MAP
513                && o.via.map.size > 0
514                && o.via.map.ptr[0].key.type == msgpack::type::POSITIVE_INTEGER;
515        }
516
517        /// Iterate an int-keyed MAP and invoke `dispatch(tag, val)` for each
518        /// `(u8, msgpack::object)` entry. The per-tag `switch` inside the
519        /// caller's lambda decides which field (or variant) to populate;
520        /// unknown tags fall through to `default` and are silently skipped,
521        /// matching the `MsgpackTagged` decoder's forward-compat policy
522        /// (`allow_unknown_tags` / retired tags drained).
523        template<typename Dispatch>
524        static void int_map_dispatch(
525            msgpack::object const& o,
526            std::string const& name,
527            Dispatch&& dispatch
528        ) {
529            for (uint32_t i = 0; i < o.via.map.size; ++i) {
530                uint8_t tag;
531                try {
532                    o.via.map.ptr[i].key.convert(tag);
533                } catch (const msgpack::type_error&) {
534                    std::cerr << o.via.map.ptr[i].key << std::endl;
535                    throw_or_abort("expected u8 tag in int-keyed map for " + name);
536                }
537                dispatch(tag, o.via.map.ptr[i].val);
538            }
539        }

541        /// Cap a `MAP` or `ARRAY` entry count against `active + reserved`.
542        /// Under-length wires are caught downstream (`conv_fld_from_array`
543        /// errors out of bounds; `conv_fld_from_kvmap` errors on missing
544        /// required keys), so we only need the upper bound here.
545        ///
546        /// * Up to `reserved` extra trailing entries are tolerated as
547        ///   retired fields (`#[tagged(reserved(...))]` on the Rust side).
548        /// * Anything beyond that is forward-compat drift that the
549        ///   producer only emits when newer fields were added. The
550        ///   Rust-side cue is `#[tagged(allow_unknown_tags)]`; the
551        ///   message points at it so a reviewer can see the opt-in.
552        static void check_size(
553            uint32_t actual,
554            std::string const& name,
555            uint32_t active,
556            uint32_t reserved
557        ) {
558            uint32_t max_size = active + reserved;
559            if (actual > max_size) {
560                throw_or_abort(
561                    name + " has " + std::to_string(actual) +
562                    " entries but at most " + std::to_string(max_size) +
563                    " are expected (" + std::to_string(active) +
564                    " active + " + std::to_string(reserved) +
565                    " reserved); opt into `#[tagged(allow_unknown_tags)]` on the Rust type to accept extras");
566            }
567        }
568    };
569    "#;
570            // cSpell:enable
571            let pos = source.find(&format!("namespace {namespace}")).expect("namespace");
572            source.insert_str(pos, &format!("namespace {namespace} {{{helpers}}}\n\n"));
573        }

575        /// Reduce the opcode size in C++ by doing what Adam came up with in https://github.com/zefchain/serde-reflection/issues/75
576        fn replace_array_with_shared_ptr(source: &mut String) {
577            // Capture `std::array<$TYPE, $LEN>`
578            let re = Regex::new(r#"std::array<\s*([^,<>]+?)\s*,\s*([0-9]+)\s*>"#)
579                .expect("failed to create regex");
580
581            let fixed =
582                re.replace_all(source, "std::shared_ptr<std::array<${1}, ${2}>>").into_owned();
583
584            *source = fixed;
585        }

587        /// Add custom code for msgpack serialization and deserialization.
588        ///
589        /// Walks the serde-reflection `registry` and asserts that every
590        /// product/sum container is also present in the `MsgpackTagged`
591        /// `tag_registry` — every wire type must derive `MsgpackTagged`
592        /// for the int-keyed dispatch to have tag metadata available.
593        /// Products additionally have to be in *canonical* (tag-ascending)
594        /// source order — otherwise `MsgpackCompact` (source-order array)
595        /// and `MsgpackTagged::Array` (tag-ascending array) would produce
596        /// indistinguishable but byte-different wires on the same input,
597        /// and the C++ side can't tell which to expect.
598        pub(crate) fn generate(
599            namespace: &str,
600            registry: &Registry,
601            tag_registry: &'a TagRegistry,
602            code: CustomCode,
603            config: MsgPackCodeConfig,
604        ) -> CustomCode {
605            assert_tag_registry_covers(registry, tag_registry);
606            let mut g = Self { namespace: vec![namespace.to_string()], code, config, tag_registry };
607            for (name, container) in registry {
608                g.generate_container(name, container);
609            }
610            g.code
611        }

613        /// Append custom code of an item in the current namespace.
614        fn add_code(&mut self, name: &str, code: &str) {
615            let mut ns = self.namespace.clone();
616            ns.push(name.to_string());
617            let c = self.code.entry(ns).or_default();
618            if !c.is_empty() && code.contains('\n') {
619                c.push('\n');
620            }
621            c.push_str(code);
622            c.push('\n');
623        }

625        fn generate_container(&mut self, name: &str, container: &ContainerFormat) {
626            use serde_reflection::ContainerFormat::*;
627            match container {
628                UnitStruct => {
629                    self.generate_unit_struct(name);
630                }
631                NewTypeStruct(_format) => {
632                    self.generate_newtype(name);
633                }
634                TupleStruct(formats) => {
635                    self.generate_tuple(name, formats);
636                }
637                Struct(fields) => {
638                    // Top-level struct — look up its Product in the
639                    // tag registry. `assert_tag_registry_covers` has
640                    // already validated coverage.
641                    let product: Product = self
642                        .tag_registry
643                        .get(name)
644                        .expect("assert_tag_registry_covers should have caught this")
645                        .tagged()
646                        .as_product()
647                        .expect("struct name should map to a Product");
648                    self.generate_struct(name, fields, product);
649                }
650                Enum(variants) => {
651                    self.generate_enum(name, variants);
652                }
653            }
654        }

656        /// Unit structs don't have fields to put into the data.
657        fn generate_unit_struct(&mut self, name: &str) {
658            // Ostensibly we could use `MSGPACK_FIELDS();`, but because of how enum unpacking
659            // expects each variant to have `msgpack_unpack`, we generate two empty methods.
660            // self.msgpack_fields(name, std::iter::empty());
661            self.msgpack_pack(name, "");
662            self.msgpack_unpack(name, "");
663        }
664
665        /// Regular structs pack into a map. `product` is the
666        /// `MsgpackTagged` metadata for this shape: either the top-level
667        /// type's entry (for ordinary structs) or the variant's payload
668        /// (when recursing into a struct-variant). It carries the
669        /// per-field u8 tag the int-keyed dispatch branch needs.
670        fn generate_struct(&mut self, name: &str, fields: &[Named<Format>], product: Product) {
671            // We could use the `MSGPACK_FIELDS` macro with the following:
672            // self.msgpack_fields(name, fields.iter().map(|f| f.name.clone()));
673            // Unfortunately it doesn't seem to deal with missing optional fields,
674            // which would mean we can't delete fields even if they were optional:
675            // https://github.com/AztecProtocol/msgpack-c/blob/54e9865b84bbdc73cfbf8d1d437dbf769b64e386/include/msgpack/v1/adaptor/detail/cpp11_define_map.hpp#L33-L45
676
677            // Or we can generate code for individual fields, which relies on
678            // the `add_helpers` to add some utility functions. This way the
679            // code is more verbose, but also easier to control, e.g. we can
680            // raise errors telling specifically which field was wrong,
681            // or we could reject the data if there was a new field we could
682            // not recognize, or we could even handle aliases.
683
684            // We treat unit fields as special, using them to ignore fields during deserialization:
685            // * in 'map' format we skip over them, never try to deserialize them from the map
686            // * in 'tuple' format we jump over their index, ignoring whatever is in that position
687            fn is_unit(field: &Named<Format>) -> bool {
688                matches!(field.value, Format::Unit)
689            }
690
691            let non_unit_field_count = fields.iter().filter(|f| !is_unit(f)).count();
692
693            self.msgpack_pack(name, &{
694                if self.config.pack_compact {
695                    // Pack as ARRAY
696                    let mut body = format!(
697                        "
698    packer.pack_array({});",
699                        fields.len()
700                    );
701                    for field in fields {
702                        let field_name = &field.name;
703                        body.push_str(&format!(
704                            r#"
705    packer.pack({field_name});"#
706                        ));
707                    }
708                    body
709                } else {
710                    // Pack as MAP
711                    let mut body = format!(
712                        "
713    packer.pack_map({non_unit_field_count});",
714                    );
715                    for field in fields {
716                        if is_unit(field) {
717                            continue;
718                        }
719                        let field_name = &field.name;
720                        body.push_str(&format!(
721                            r#"
722    packer.pack(std::make_pair("{field_name}", {field_name}));"#
723                        ));
724                    }
725                    body
726                }
727            });
728
729            self.msgpack_unpack(name, &{
730                // Dispatch on wire shape:
731                //   * MAP with INT first key → `Format::MsgpackTagged`
732                //     (int-keyed: dispatch by u8 tag).
733                //   * MAP with STR keys → legacy `Format::Msgpack`
734                //     (string-keyed: look up by field name).
735                //   * ARRAY → legacy `Format::MsgpackCompact` and also
736                //     `Format::MsgpackTagged::Array` (both emit fields in
737                //     tag-ascending order under the codegen invariant
738                //     `tag_order_matches_source`, so positional indexing
739                //     decodes both correctly).
740                // cSpell:disable
741                let mut body = format!(
742                    r#"
743    std::string name = "{name}";
744    if (o.type == msgpack::type::MAP) {{
745        if (Helpers::is_int_keyed_map(o)) {{
746            Helpers::int_map_dispatch(o, name, [&](uint8_t tag, msgpack::object const& val) {{
747                switch (tag) {{"#
748                );
749                // cSpell:enable
750                for field in fields {
751                    let field_name = &field.name;
752                    let tag = product.tag_for(field_name).unwrap_or_else(|| {
753                        panic!(
754                            "field {field_name:?} of {name:?} is not in the MsgpackTagged \
755                             Product — serde-reflection and #[derive(MsgpackTagged)] disagree \
756                             on which fields are on the wire",
757                        )
758                    });
759                    if is_unit(field) {
760                        // Field is `()` in Rust / `std::monostate` in C++.
761                        // The wire still carries a value at this tag; we
762                        // consume it without binding to any C++ member,
763                        // mirroring `deserialize_unit`'s skip-any semantics.
764                        // cSpell:disable
765                        body.push_str(&format!(
766                            r#"
767                    case {tag}:
768                        // Field is `std::monostate` — wire entry intentionally discarded.
769                        break;"#
770                        ));
771                        // cSpell:enable
772                        continue;
773                    }
774                    // cSpell:disable
775                    body.push_str(&format!(
776                        r#"
777                    case {tag}:
778                        Helpers::convert_or_throw(val, name, "{field_name}", {field_name});
779                        break;"#
780                    ));
781                    // cSpell:enable
782                }
783                // Reserved tags: retired in the Rust type via
784                // `#[tagged(reserved(...))]`. The Rust decoder skips
785                // them silently regardless of `allow_unknown_tags`; do
786                // the same here, with an explicit case so the strict
787                // default below can distinguish "retired" from "never
788                // declared".
789                if !product.reserved.is_empty() {
790                    for &reserved_tag in product.reserved {
791                        body.push_str(&format!(
792                            "
793                    case {reserved_tag}:"
794                        ));
795                    }
796                    body.push_str(
797                        r#"
798                        // Reserved tag (retired field) — skip silently.
799                        break;"#,
800                    );
801                }
802                // Default branch: strict by default (reject unknown tags
803                // so a C++ reviewer can see the type-level intent). Opt
804                // into silent forward-compat with `#[tagged(allow_unknown_tags)]`
805                // on the Rust struct.
806                if product.allow_unknown_tags {
807                    body.push_str(
808                        r#"
809                    default:
810                        // `#[tagged(allow_unknown_tags)]` on the Rust side:
811                        // silently skip any tag we don't recognize.
812                        break;"#,
813                    );
814                } else {
815                    // cSpell:disable
816                    body.push_str(&format!(
817                        r#"
818                    default:
819                        std::cerr << val << std::endl;
820                        throw_or_abort("unknown tag for {name}: " + std::to_string(tag));"#
821                    ));
822                    // cSpell:enable
823                }
824                // cSpell:disable
825                body.push_str(
826                    r#"
827                }
828            });
829        } else {"#,
830                );
831                // cSpell:enable
832                // String-keyed map branch (`Format::Msgpack`): the
833                // lookup is by field name, so extra wire keys would
834                // otherwise pass unnoticed. Same `active + reserved`
835                // ceiling and same `allow_unknown_tags` opt-out as the
836                // ARRAY branch.
837                if !product.allow_unknown_tags {
838                    body.push_str(&format!(
839                        r#"
840            Helpers::check_size(o.via.map.size, name, {active}, {reserved});"#,
841                        active = fields.len(),
842                        reserved = product.reserved.len(),
843                    ));
844                }
845                // cSpell:disable
846                body.push_str(
847                    r#"
848            auto kvmap = Helpers::make_kvmap(o, name);"#,
849                );
850                // cSpell:enable
851                for field in fields {
852                    if is_unit(field) {
853                        continue;
854                    }
855                    let field_name = &field.name;
856                    let is_optional = matches!(field.value, Format::Option(_));
857                    // cSpell:disable
858                    body.push_str(&format!(
859                        r#"
860            Helpers::conv_fld_from_kvmap(kvmap, name, "{field_name}", {field_name}, {is_optional});"#
861                    ));
862                    // cSpell:enable
863                }
864                body.push_str(
865                    "
866        }
867    } else if (o.type == msgpack::type::ARRAY) {
868        auto array = o.via.array;",
869                );
870                // Cap the array length: an older reader of a newer wire
871                // (forward-compat) should reject extra trailing items
872                // unless the type opts into `#[tagged(allow_unknown_tags)]`;
873                // a newer reader of an older wire that retired trailing
874                // fields (backward-compat) gets `reserved.len()` extra
875                // trailing positions tolerated either way.
876                //
877                // Under-length wires are caught downstream by
878                // `Helpers::conv_fld_from_array` (it errors when its
879                // index is past `array.size`), so we only need the
880                // upper bound here.
881                if !product.allow_unknown_tags {
882                    body.push_str(&format!(
883                        r#"
884        Helpers::check_size(array.size, name, {active}, {reserved});"#,
885                        active = fields.len(),
886                        reserved = product.reserved.len(),
887                    ));
888                }
889                for (index, field) in fields.iter().enumerate() {
890                    if is_unit(field) {
891                        continue;
892                    }
893                    let field_name = &field.name;
894                    // cSpell:disable
895                    body.push_str(&format!(
896                        r#"
897        Helpers::conv_fld_from_array(array, name, "{field_name}", {field_name}, {index});"#
898                    ));
899                    // cSpell:enable
900                }
901
902                body.push_str(
903                    r#"
904    } else {
905        throw_or_abort("expected MAP or ARRAY for " + name);
906    }"#,
907                );
908                body
909            });
910        }

937        /// Enums serialize as a single element map keyed by the variant type name.
938        fn generate_enum(&mut self, name: &str, variants: &BTreeMap<u32, Named<VariantFormat>>) {
939            // Look up the `Sum` upfront — both the per-variant recursion
940            // (each variant's payload Product comes from here) and the
941            // int-keyed `msgpack_unpack` body below need it. Coverage is
942            // already enforced by `assert_tag_registry_covers`.
943            let sum: Sum = self
944                .tag_registry
945                .get(name)
946                .expect("assert_tag_registry_covers should have caught this")
947                .tagged()
948                .as_sum()
949                .expect("enum name should map to a Sum");
950
951            // Resolve a variant's payload Product by its serde name. The
952            // `MsgpackTagged` macro and serde-derive agree on names, so
953            // a miss here is an internal bug.
954            let payload_for = |variant_name: &str| -> Product {
955                sum.variants.iter().find(|v| v.name == variant_name).map_or_else(
956                    || {
957                        panic!(
958                            "variant {variant_name:?} of enum {name:?} is not in the \
959                             MsgpackTagged Sum — serde-reflection and \
960                             #[derive(MsgpackTagged)] disagree on variants",
961                        )
962                    },
963                    |v| v.payload,
964                )
965            };
966
967            // Recurse into the variants
968            self.namespace.push(name.to_string());
969            for variant in variants.values() {
970                let payload = payload_for(&variant.name);
971                self.generate_variant(&variant.name, &variant.value, payload);
972            }
973            self.namespace.pop();
974
975            // Pack the enum itself
976            self.msgpack_pack(name, &{
977                let cases = variants
978                    .iter()
979                    .map(|(i, v)| {
980                        format!(
981                            r#"
982        case {i}:
983            tag = "{}";
984            is_unit = {};
985            break;"#,
986                            v.name,
987                            matches!(v.value, VariantFormat::Unit)
988                        )
989                    })
990                    .collect::<Vec<_>>()
991                    .join("");
992
993                format!(
994                    r#"
995    std::string tag;
996    bool is_unit;
997    switch (value.index()) {{
998        {cases}
999        default:
1000            throw_or_abort("unknown enum '{name}' variant index: " + std::to_string(value.index()));
1001    }}
1002    if (is_unit) {{
1003        packer.pack(tag);
1004    }} else {{
1005        std::visit([&packer, tag](const auto& arg) {{
1006            packer.pack_map(1);
1007            packer.pack(tag);
1008            packer.pack(arg);
1009        }}, value);
1010    }}"#
1011                )
1012            });
1013
1014            // Unpack the enum:
1015            //   * MAP with INT key → `Format::MsgpackTagged` variant
1016            //     (int-keyed dispatch on the u8 tag).
1017            //   * MAP with STR key → legacy `Format::Msgpack` variant
1018            //     (string-keyed dispatch on variant name).
1019            //   * STR top-level → legacy `Format::MsgpackCompact` unit
1020            //     variant (bare variant-name string).
1021            // The 1-entry-size invariant of the MAP cases is enforced
1022            // upfront.
1023            self.msgpack_unpack(name, &{
1024                // cSpell:disable
1025                let mut body = format!(
1026                    r#"
1027
1028    if (o.type != msgpack::type::object_type::MAP && o.type != msgpack::type::object_type::STR) {{
1029        std::cerr << o << std::endl;
1030        throw_or_abort("expected MAP or STR for enum '{name}'; got type " + std::to_string(o.type));
1031    }}
1032    if (o.type == msgpack::type::object_type::MAP && o.via.map.size != 1) {{
1033        throw_or_abort("expected 1 entry for enum '{name}'; got " + std::to_string(o.via.map.size));
1034    }}
1035    if (Helpers::is_int_keyed_map(o)) {{
1036        // `Format::MsgpackTagged` — int-keyed variant.
1037        uint8_t tag;
1038        try {{
1039            o.via.map.ptr[0].key.convert(tag);
1040        }} catch (const msgpack::type_error&) {{
1041            std::cerr << o << std::endl;
1042            throw_or_abort("expected u8 variant tag for enum '{name}'");
1043        }}
1044        switch (tag) {{"#
1045                );
1046                // cSpell:enable
1047
1048                for v in variants.values() {
1049                    let variant = &v.name;
1050                    let tag = sum.variants.iter().find(|reg_v| reg_v.name == variant).map_or_else(
1051                        || {
1052                            panic!(
1053                                "variant {variant:?} of enum {name:?} is not in the MsgpackTagged \
1054                                 Sum — serde-reflection and #[derive(MsgpackTagged)] disagree",
1055                            )
1056                        },
1057                        |reg_v| reg_v.tag,
1058                    );
1059                    if matches!(v.value, VariantFormat::Unit) {
1060                        // Unit variants carry a `nil` payload that the
1061                        // variant's empty `msgpack_unpack` would accept,
1062                        // but constructing the value and assigning is
1063                        // simpler and avoids reading `o.via.map.ptr[0].val`
1064                        // (which Barretenberg's `-Werror=unused-variable`
1065                        // flags when all variants of an enum are unit).
1066                        body.push_str(&format!(
1067                            r#"
1068            case {tag}: {{
1069                {variant} v;
1070                value = v;
1071                break;
1072            }}"#
1073                        ));
1074                    } else {
1075                        // cSpell:disable
1076                        body.push_str(&format!(
1077                            r#"
1078            case {tag}: {{
1079                {variant} v;
1080                try {{
1081                    o.via.map.ptr[0].val.convert(v);
1082                }} catch (const msgpack::type_error&) {{
1083                    std::cerr << o << std::endl;
1084                    throw_or_abort("error converting into enum variant '{name}::{variant}'");
1085                }}
1086                value = v;
1087                break;
1088            }}"#
1089                        ));
1090                        // cSpell:enable
1091                    }
1092                }
1093
1094                // Reserved variant tags: retired variants on the Rust side
1095                // (`#[tagged(reserved(...))]` on the enum). If the enum
1096                // marks a unit variant with `#[tagged(on_reserved)]`,
1097                // legacy wires carrying a retired tag route to that
1098                // fallback; otherwise we throw with a "retired" message
1099                // that's distinguishable from the "unknown" case below.
1100                let lookup_unit_variant = |fallback_tag: u8| -> &str {
1101                    sum.variants.iter().find(|v| v.tag == fallback_tag).map_or_else(
1102                        || {
1103                            panic!(
1104                                "MsgpackTagged Sum for enum {name:?} declares a fallback \
1105                                 tag {fallback_tag} that doesn't match any registered variant",
1106                            )
1107                        },
1108                        |v| v.name,
1109                    )
1110                };
1111                if !sum.reserved.is_empty() {
1112                    for &reserved_tag in sum.reserved {
1113                        body.push_str(&format!(
1114                            "
1115            case {reserved_tag}:"
1116                        ));
1117                    }
1118                    if let Some(fallback_tag) = sum.on_reserved_tag {
1119                        let fallback_name = lookup_unit_variant(fallback_tag);
1120                        // cSpell:disable
1121                        body.push_str(&format!(
1122                            r#" {{
1123                // `#[tagged(on_reserved)]` fallback: retired tag routes to
1124                // the designated unit variant (payload discarded).
1125                {fallback_name} v;
1126                value = v;
1127                break;
1128            }}"#
1129                        ));
1130                        // cSpell:enable
1131                    } else {
1132                        // cSpell:disable
1133                        body.push_str(&format!(
1134                            r#"
1135                std::cerr << o << std::endl;
1136                throw_or_abort("retired variant tag for enum '{name}' (declare `#[tagged(on_reserved)]` on a unit variant to route legacy data here): " + std::to_string(tag));"#
1137                        ));
1138                        // cSpell:enable
1139                    }
1140                }
1141
1142                // Default branch for unknown variant tags (not active,
1143                // not reserved). Routes to `#[tagged(on_unknown)]` if
1144                // set — the forward-compat opt-in for newer producers
1145                // introducing variants this code doesn't know about.
1146                if let Some(fallback_tag) = sum.on_unknown_tag {
1147                    let fallback_name = lookup_unit_variant(fallback_tag);
1148                    // cSpell:disable
1149                    body.push_str(&format!(
1150                        r#"
1151            default: {{
1152                // `#[tagged(on_unknown)]` fallback: any tag we don't recognize
1153                // (and isn't reserved) routes here.
1154                {fallback_name} v;
1155                value = v;
1156                break;
1157            }}"#
1158                    ));
1159                    // cSpell:enable
1160                } else {
1161                    // cSpell:disable
1162                    body.push_str(&format!(
1163                        r#"
1164            default:
1165                std::cerr << o << std::endl;
1166                throw_or_abort("unknown '{name}' enum variant tag: " + std::to_string(tag));"#
1167                    ));
1168                    // cSpell:enable
1169                }
1170                // cSpell:disable
1171                body.push_str(
1172                    r#"
1173        }
1174    } else {"#,
1175                );
1176                // cSpell:enable
1177                // Reuse the existing format-string body for the legacy
1178                // string-keyed dispatch path. The `format!(name = ...)`
1179                // substitution from the previous body is preserved in the
1180                // already-appended text; the section below is plain C++
1181                // with no further interpolation.
1182                body.push_str(&format!(
1183                    r#"
1184        // `Format::Msgpack` (MAP, string-keyed) or `Format::MsgpackCompact`
1185        // unit variant (bare STR) — both dispatch on the variant name.
1186        std::string tag;
1187        try {{
1188            if (o.type == msgpack::type::object_type::MAP) {{
1189                o.via.map.ptr[0].key.convert(tag);
1190            }} else {{
1191                o.convert(tag);
1192            }}
1193        }} catch(const msgpack::type_error&) {{
1194            std::cerr << o << std::endl;
1195            throw_or_abort("error converting tag to string for enum '{name}'");
1196        }}"#
1197                ));
1198                // cSpell:enable
1199
1200                for (i, v) in variants {
1201                    let variant = &v.name;
1202                    body.push_str(&format!(
1203                        r#"
1204        {}if (tag == "{variant}") {{
1205            {variant} v;"#,
1206                        if *i == 0 { "" } else { "else " }
1207                    ));
1208
1209                    if !matches!(v.value, VariantFormat::Unit) {
1210                        // cSpell:disable
1211                        body.push_str(&format!(
1212                            r#"
1213            try {{
1214                o.via.map.ptr[0].val.convert(v);
1215            }} catch (const msgpack::type_error&) {{
1216                std::cerr << o << std::endl;
1217                throw_or_abort("error converting into enum variant '{name}::{variant}'");
1218            }}
1219            "#
1220                        ));
1221                        // cSpell:enable
1222                    }
1223                    // Closing brace of if statement
1224                    body.push_str(
1225                        r#"
1226            value = v;
1227        }"#,
1228                    );
1229                }
1230                // cSpell:disable
1231                body.push_str(&format!(
1232                    r#"
1233        else {{
1234            std::cerr << o << std::endl;
1235            throw_or_abort("unknown '{name}' enum variant: " + tag);
1236        }}
1237    }}"#
1238                ));
1239                // cSpell:enable
1240
1241                body
1242            });
1243        }

1245        /// Generate msgpack code for nested enum variants. `payload` is
1246        /// the variant's `MsgpackTagged` `Product`, looked up from the
1247        /// parent enum's `Sum`. Only the `Struct` variant case consults
1248        /// it (it carries the int-keyed dispatch metadata for that
1249        /// variant's named fields); the other cases are tag-irrelevant
1250        /// (unit / passthrough newtype / unimplemented tuple).
1251        fn generate_variant(&mut self, name: &str, variant: &VariantFormat, payload: Product) {
1252            match variant {
1253                VariantFormat::Variable(_) => {
1254                    unreachable!("internal construct")
1255                }
1256                VariantFormat::Unit => self.generate_unit_struct(name),
1257                VariantFormat::NewType(_format) => self.generate_newtype(name),
1258                VariantFormat::Tuple(formats) => self.generate_tuple(name, formats),
1259                VariantFormat::Struct(fields) => self.generate_struct(name, fields, payload),
1260            }
1261        }

1263        /// Use the `MSGPACK_FIELDS` macro with a list of fields.
1264        /// This one takes care of serializing and deserializing as well.
1265        ///
1266        /// Uses [define_map](https://github.com/AztecProtocol/msgpack-c/blob/54e9865b84bbdc73cfbf8d1d437dbf769b64e386/include/msgpack/v1/adaptor/detail/cpp11_define_map.hpp#L75-L88) under the hood.
1267        #[allow(dead_code)]
1268        fn msgpack_fields(&mut self, name: &str, fields: impl Iterator<Item = String>) {
1269            let fields = fields.collect::<Vec<_>>().join(", ");
1270            let code = format!("MSGPACK_FIELDS({fields});");
1271            self.add_code(name, &code);
1272        }

1274        /// Add a `msgpack_pack` implementation.
1275        fn msgpack_pack(&mut self, name: &str, body: &str) {
1276            if self.config.no_pack {
1277                return;
1278            }
1279            let code = Self::make_fn("void msgpack_pack(auto& packer) const", body);
1280            self.add_code(name, &code);

1326        fn make_fn(header: &str, body: &str) -> String {
1327            let body = body.trim_end();
1328            if body.is_empty() {
1329                format!("{header} {{}}")
1330            } else if !body.contains('\n') {
1331                format!("{header} {{ {body} }}")
1332            } else if body.starts_with('\n') {
1333                format!("{header} {{{body}\n}}")
1334            } else {
1335                format!("{header} {{\n{body}\n}}")
1336            }
1337        }
